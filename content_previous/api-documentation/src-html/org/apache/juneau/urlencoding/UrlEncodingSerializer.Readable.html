029/**
030 * Serializes POJO models to URL-encoded notation with UON-encoded values (a notation for URL-encoded query paramter values).
031 *
032 *
033 * <h6 class='topic'>Media types</h6>
034 * <p>
035 *    Handles <code>Accept</code> types: <code>application/x-www-form-urlencoded</code>
036 * <p>
037 *    Produces <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
038 *
039 *
040 * <h6 class='topic'>Description</h6>
041 * <p>
042 *    This serializer provides several serialization options.  Typically, one of the predefined DEFAULT serializers will be sufficient.
043 *    However, custom serializers can be constructed to fine-tune behavior.
044 *
045 *
046 * <h6 class='topic'>Configurable properties</h6>
047 * <p>
048 *    This class has the following properties associated with it:
049 * <ul>
050 *    <li>{@link UonSerializerContext}
051 *    <li>{@link BeanContext}
052 * </ul>
053 * <p>
054 *    The following shows a sample object defined in Javascript:
055 * </p>
056 * <p class='bcode'>
057 *    {
058 *       id: 1,
059 *       name: <js>'John Smith'</js>,
060 *       uri: <js>'http://sample/addressBook/person/1'</js>,
061 *       addressBookUri: <js>'http://sample/addressBook'</js>,
062 *       birthDate: <js>'1946-08-12T00:00:00Z'</js>,
063 *       otherIds: <jk>null</jk>,
064 *       addresses: [
065 *          {
066 *             uri: <js>'http://sample/addressBook/address/1'</js>,
067 *             personUri: <js>'http://sample/addressBook/person/1'</js>,
068 *             id: 1,
069 *             street: <js>'100 Main Street'</js>,
070 *             city: <js>'Anywhereville'</js>,
071 *             state: <js>'NY'</js>,
072 *             zip: 12345,
073 *             isCurrent: <jk>true</jk>,
074 *          }
075 *       ]
076 *    }
077 * </p>
078 * <p>
079 *    Using the "strict" syntax defined in this document, the equivalent
080 *       URL-encoded notation would be as follows:
081 * </p>
082 * <p class='bcode'>
083 *    <xa>id</xa>=$n(<xs>1</xs>)
084 *    &amp;<xa>name</xa>=<xs>John+Smith</xs>,
085 *    &amp;<xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
086 *    &amp;<xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
087 *    &amp;<xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
088 *    &amp;<xa>otherIds</xa>=<xs>%00</xs>,
089 *    &amp;<xa>addresses</xa>=$a(
090 *       $o(
091 *          <xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
092 *          <xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
093 *          <xa>id</xa>=$n(<xs>1</xs>),
094 *          <xa>street</xa>=<xs>100+Main+Street</xs>,
095 *          <xa>city</xa>=<xs>Anywhereville</xs>,
096 *          <xa>state</xa>=<xs>NY</xs>,
097 *          <xa>zip</xa>=$n(<xs>12345</xs>),
098 *          <xa>isCurrent</xa>=$b(<xs>true</xs>)
099 *       )
100 *    )
101 * </p>
102 * <p>
103 *    A secondary "lax" syntax is available when the data type of the
104 *       values are already known on the receiving end of the transmission:
105 * </p>
106 * <p class='bcode'>
107 *    <xa>id</xa>=<xs>1</xs>,
108 *    &amp;<xa>name</xa>=<xs>John+Smith</xs>,
109 *    &amp;<xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
110 *    &amp;<xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
111 *    &amp;<xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
112 *    &amp;<xa>otherIds</xa>=<xs>%00</xs>,
113 *    &amp;<xa>addresses</xa>=(
114 *       (
115 *          <xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
116 *          <xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
117 *          <xa>id</xa>=<xs>1</xs>,
118 *          <xa>street</xa>=<xs>100+Main+Street</xs>,
119 *          <xa>city</xa>=<xs>Anywhereville</xs>,
120 *          <xa>state</xa>=<xs>NY</xs>,
121 *          <xa>zip</xa>=<xs>12345</xs>,
122 *          <xa>isCurrent</xa>=<xs>true</xs>
123 *       )
124 *    )
125 * </p>
126 *
127 *
128 * <h6 class='topic'>Examples</h6>
129 * <p class='bcode'>
130 *    <jc>// Serialize a Map</jc>
131 *    Map m = <jk>new</jk> ObjectMap(<js>"{a:'b',c:1,d:false,e:['f',1,false],g:{h:'i'}}"</js>);
132 *
133 *    <jc>// Serialize to value equivalent to JSON.</jc>
134 *    <jc>// Produces "a=b&amp;c=$n(1)&amp;d=$b(false)&amp;e=$a(f,$n(1),$b(false))&amp;g=$o(h=i)"</jc>
135 *    String s = UrlEncodingSerializer.<jsf>DEFAULT</jsf>.serialize(s);
136 *
137 *    <jc>// Serialize to simplified value (for when data type is already known by receiver).</jc>
138 *    <jc>// Produces "a=b&amp;c=1&amp;d=false&amp;e=(f,1,false)&amp;g=(h=i))"</jc>
139 *    String s = UrlEncodingSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
140 *
141 *    <jc>// Serialize a bean</jc>
142 *    <jk>public class</jk> Person {
143 *       <jk>public</jk> Person(String s);
144 *       <jk>public</jk> String getName();
145 *       <jk>public int</jk> getAge();
146 *       <jk>public</jk> Address getAddress();
147 *       <jk>public boolean</jk> deceased;
148 *    }
149 *
150 *    <jk>public class</jk> Address {
151 *       <jk>public</jk> String getStreet();
152 *       <jk>public</jk> String getCity();
153 *       <jk>public</jk> String getState();
154 *       <jk>public int</jk> getZip();
155 *    }
156 *
157 *    Person p = <jk>new</jk> Person(<js>"John Doe"</js>, 23, <js>"123 Main St"</js>, <js>"Anywhere"</js>, <js>"NY"</js>, 12345, <jk>false</jk>);
158 *
159 *    <jc>// Produces "name=John+Doe&amp;age=23&amp;address=$o(street=123+Main+St,city=Anywhere,state=NY,zip=$n(12345))&amp;deceased=$b(false)"</jc>
160 *    String s = UrlEncodingSerializer.<jsf>DEFAULT</jsf>.serialize(s);
161 *
162 *    <jc>// Produces "name=John+Doe&amp;age=23&amp;address=(street=123+Main+St,city=Anywhere,state=NY,zip=12345)&amp;deceased=false)"</jc>
163 *    String s = UrlEncodingSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
164 * </p>
165 *
166 * @author James Bognar (james.bognar@salesforce.com)
167 */

223   /**
224    * Workhorse method. Determines the type of object, and then calls the
225    * appropriate type-specific serialization method.
226    */
227   @SuppressWarnings({ "rawtypes", "unchecked" })
228   private SerializerWriter serializeAnything(UrlEncodingSerializerSession session, UonWriter out, Object o) throws Exception {
229      BeanContext bc = session.getBeanContext();
230
231      boolean addClassAttr;      // Add "_class" attribute to element?
232      ClassMeta<?> aType;        // The actual type
233      ClassMeta<?> gType;        // The generic type
234
235      aType = session.push("root", o, object());
236      session.indent--;
237      if (aType == null)
238         aType = object();
239
240      gType = aType.getTransformedClassMeta();
241      addClassAttr = (session.isAddClassAttrs());
242
243      // Transform if necessary
244      PojoTransform transform = aType.getPojoTransform();            // The transform
245      if (transform != null) {
246         o = transform.transform(o);
247
248         // If the transform's getTransformedClass() method returns Object, we need to figure out
249         // the actual type now.
250         if (gType.isObject())
251            gType = bc.getClassMetaForObject(o);
252      }
253
254      if (gType.isMap()) {
255         if (o instanceof BeanMap)
256            serializeBeanMap(session, out, (BeanMap)o, addClassAttr);
257         else
258            serializeMap(session, out, (Map)o, gType);
259      } else if (gType.hasToObjectMapMethod()) {
260         serializeMap(session, out, gType.toObjectMap(o), gType);
261      } else if (gType.isBean()) {
262         serializeBeanMap(session, out, bc.forBean(o), addClassAttr);
263      } else if (gType.isCollection()) {
264         serializeMap(session, out, getCollectionMap((Collection)o), bc.getMapClassMeta(Map.class, Integer.class, gType.getElementType()));
265      } else {
266         // All other types can't be serialized as key/value pairs, so we create a
267         // mock key/value pair with a "_value" key.
268         out.append("_value=");
269         super.serializeAnything(session, out, o, null, null, null, false, true);
270      }
271
272      session.pop();
273      return out;
274   }
275
276   private Map<Integer,Object> getCollectionMap(Collection<?> c) {
277      Map<Integer,Object> m = new TreeMap<Integer,Object>();
278      int i = 0;
279      for (Object o : c)
280         m.put(i++, o);
281      return m;
282   }

284   @SuppressWarnings({ "rawtypes", "unchecked" })
285   private SerializerWriter serializeMap(UrlEncodingSerializerSession session, UonWriter out, Map m, ClassMeta<?> type) throws Exception {
286
287      m = session.sort(m);
288
289      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
290
291      int depth = session.getIndent();
292      boolean addAmp = false;
293
294      Iterator mapEntries = m.entrySet().iterator();
295
296      while (mapEntries.hasNext()) {
297         Map.Entry e = (Map.Entry) mapEntries.next();
298         Object value = e.getValue();
299         Object key = session.generalize(e.getKey(), keyType);
300
301
302         if (session.shouldUseExpandedParams(value)) {
303            Iterator i = value instanceof Collection ? ((Collection)value).iterator() : ArrayUtils.iterator(value);
304            while (i.hasNext()) {
305               if (addAmp)
306                  out.cr(depth).append('&');
307               out.appendObject(key, false, true, true).append('=');
308               super.serializeAnything(session, out, i.next(), null, (key == null ? null : key.toString()), null, false, true);
309               addAmp = true;
310            }
311         } else {
312            if (addAmp)
313               out.cr(depth).append('&');
314            out.appendObject(key, false, true, true).append('=');
315            super.serializeAnything(session, out, value, valueType, (key == null ? null : key.toString()), null, false, true);
316            addAmp = true;
317         }
318      }
319
320      return out;
321   }

323   @SuppressWarnings({ "rawtypes" })
324   private SerializerWriter serializeBeanMap(UrlEncodingSerializerSession session, UonWriter out, BeanMap m, boolean addClassAttr) throws Exception {
325      int depth = session.getIndent();
326
327      Iterator mapEntries = m.entrySet().iterator();
328
329      // Print out "_class" attribute on this bean if required.
330      if (addClassAttr) {
331         String attr = "_class";
332         out.appendObject(attr, false, false, true).append('=').append(m.getClassMeta().getInnerClass().getName());
333         if (mapEntries.hasNext())
334            out.cr(depth).append('&');
335      }
336
337      boolean addAmp = false;
338
339      while (mapEntries.hasNext()) {
340         BeanMapEntry p = (BeanMapEntry)mapEntries.next();
341         BeanPropertyMeta pMeta = p.getMeta();
342
343         String key = p.getKey();
344         Object value = null;
345         try {
346            value = p.getValue();
347         } catch (StackOverflowError e) {
348            throw e;
349         } catch (Throwable t) {
350            session.addBeanGetterWarning(pMeta, t);
351         }
352
353         if (session.canIgnoreValue(pMeta.getClassMeta(), key, value))
354            continue;
355
356         if (value != null && session.shouldUseExpandedParams(pMeta)) {
357            ClassMeta cm = pMeta.getClassMeta();
358            // Transformed object array bean properties may be transformed resulting in ArrayLists,
359            // so we need to check type if we think it's an array.
360            Iterator i = (cm.isCollection() || value instanceof Collection) ? ((Collection)value).iterator() : ArrayUtils.iterator(value);
361            while (i.hasNext()) {
362               if (addAmp)
363                  out.cr(depth).append('&');
364
365               out.appendObject(key, false, true, true).append('=');
366
367               super.serializeAnything(session, out, i.next(), pMeta.getClassMeta().getElementType(), key, pMeta, false, true);
368
369               addAmp = true;
370            }
371         } else {
372            if (addAmp)
373               out.cr(depth).append('&');
374
375            out.appendObject(key, false, true, true).append('=');
376
377            super.serializeAnything(session, out, value, pMeta.getClassMeta(), key, pMeta, false, true);
378
379            addAmp = true;
380         }
381
382      }
383      return out;
384   }

390   /**
391    * Converts the specified object to a string using this serializers {@link BeanContext#convertToType(Object, Class)} method
392    *    and runs {@link URLEncoder#encode(String,String)} against the results.
393    * Useful for constructing URL parts.
394    *
395    * @param o The object to serialize.
396    * @return The serialized object.
397    */
398   public String serializeUrlPart(Object o) {
399      try {
400         // Shortcut for simple types.
401         ClassMeta<?> cm = getBeanContext().getClassMetaForObject(o);
402         if (cm != null)
403            if (cm.isCharSequence() || cm.isNumber() || cm.isBoolean())
404               return o.toString();
405
406         StringWriter w = new StringWriter();
407         UonSerializerSession s = createSession(w, null, null);
408         super.doSerialize(s, o);
409         return w.toString();
410      } catch (Exception e) {
411         throw new RuntimeException(e);
412      }
413   }

420   @Override /* Serializer */
421   public UrlEncodingSerializerSession createSession(Object output, ObjectMap properties, Method javaMethod) {
422      return new UrlEncodingSerializerSession(getContext(UrlEncodingSerializerContext.class), getBeanContext(), output, properties, javaMethod);

425   @Override /* Serializer */
426   protected void doSerialize(SerializerSession session, Object o) throws Exception {
427      UrlEncodingSerializerSession s = (UrlEncodingSerializerSession)session;
428      serializeAnything(s, s.getWriter(), o);
429   }
