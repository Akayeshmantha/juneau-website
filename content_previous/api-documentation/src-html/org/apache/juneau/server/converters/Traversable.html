013package org.apache.juneau.server.converters;

015import javax.servlet.http.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.serializer.*;
019import org.apache.juneau.server.*;
020import org.apache.juneau.utils.*;
021
022/**
023 * Converter for enablement of {@link PojoRest} support on response objects returned by a <code>@RestMethod</code> method.
024 * <p>
025 *    When enabled, objects in a POJO tree returned by the REST method can be addressed through additional URL path information.
026 *
027 * <p class='bcode'>
028 *    <jc>// Resource method on resource "http://localhost:8080/sample/addressBook"</jc>
029 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, converters=Traversable.<jk>class</jk>)
030 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res) {
031 *       <jk>return new</jk> AddressBook();
032 *    }
033 *
034 *    <jc>// Sample usage</jc>
035 *    <jk>public static void</jk> main(String[] args) {
036 *       <jc>// Get the zip code of the 2nd address of the first person in the address book.</jc>
037 *       RestCall r = <jk>new</jk> RestClient().doGet(<js>"http://localhost:8080/sample/addressBook/0/addresses/1/zip"</js>);
038 *       <jk>int</jk> zip = r.getResponse(Integer.<jk>class</jk>);
039 *    }
040 * </p>
041 * <p>
042 *    See {@link PojoRest} for additional information on addressing elements in a POJO tree using URL notation.
043 */
044public final class Traversable implements RestConverter {
045
046   @Override /* RestConverter */
047   @SuppressWarnings({"rawtypes", "unchecked"})
048   public Object convert(RestRequest req, Object o, ClassMeta cm) throws RestException {
049      if (o == null)
050         return null;
051
052      if (req.getPathRemainder() != null) {
053         try {
054            if (cm.getPojoTransform() != null)
055               o = cm.getPojoTransform().transform(o);
056            PojoRest p = new PojoRest(o, req.getReaderParser());
057            o = p.get(req.getPathRemainder());
058         } catch (SerializeException e) {
059            throw new RestException(HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e);
060         } catch (PojoRestException e) {
061            throw new RestException(e.getStatus(), e);
062         }
063      }
064
065      return o;
066   }
067}
