035/**
036 * Represents a connection to a remote REST resource.
037 * <p>
038 *    Instances of this class are created by the various {@code doX()} methods on the {@link RestClient} class.
039 * <p>
040 *    This class uses only Java standard APIs.  Requests can be built up using a fluent interface with method chaining, like so...
041 *
042 * <p class='bcode'>
043 *    RestClient client = <jk>new</jk> RestClient();
044 *    RestCall c = client.doPost(<jsf>URL</jsf>).setInput(o).setHeader(x,y);
045 *    MyBean b = c.getResponse(MyBean.<jk>class</jk>);
046 * </p>
047 * <p>
048 *    The actual connection and request/response transaction occurs when calling one of the <code>getResponseXXX()</code> methods.
049 *
050 * <h6 class='topic'>Additional Information</h6>
051 * <ul>
052 *    <li><a class='doclink' href='package-summary.html#RestClient'>org.apache.juneau.client &gt; REST client API</a> for more information and code examples.
053 * </ul>
054 *
055 * @author James Bognar (james.bognar@salesforce.com)
056 */

079   /**
080    * Constructs a REST call with the specified method name.
081    *
082    * @param client The client that created this request.
083    * @param request The wrapped Apache HTTP client request object.
084    * @throws RestCallException If an exception or non-200 response code occurred during the connection attempt.

086   protected RestCall(RestClient client, HttpRequestBase request) throws RestCallException {
087      this.client = client;
088      this.request = request;
089      for (RestCallInterceptor i : this.client.interceptors)
090         addInterceptor(i);

093   /**
094    * Sets the input for this REST call.
095    *
096    * @param input The input to be sent to the REST resource (only valid for PUT and POST) requests. <br>
097    *    Can be of the following types:
098    *    <ul class='spaced-list'>
099    *       <li>{@link Reader} - Raw contents of {@code Reader} will be serialized to remote resource.
100    *       <li>{@link InputStream} - Raw contents of {@code InputStream} will be serialized to remote resource.
101    *       <li>{@link Object} - POJO to be converted to text using the {@link Serializer} registered with the {@link RestClient}.
102    *       <li>{@link HttpEntity} - Bypass Juneau serialization and pass HttpEntity directly to HttpClient.
103    *    </ul>
104    * @return This object (for method chaining).
105    * @throws RestCallException If a retry was attempted, but the entity was not repeatable.
106    */
107   public RestCall setInput(final Object input) throws RestCallException {
108      if (! (request instanceof HttpEntityEnclosingRequestBase))
109         throw new RestCallException(0, "Method does not support content entity.", request.getMethod(), request.getURI(), null);
110      HttpEntity entity = (input instanceof HttpEntity ? (HttpEntity)input : new RestRequestEntity(input, client.serializer));
111      ((HttpEntityEnclosingRequestBase)request).setEntity(entity);
112      if (retries > 1 && ! entity.isRepeatable())
113         throw new RestCallException("Rest call set to retryable, but entity is not repeatable.");
114      return this;
115   }

131   /**
132    * Make this call retryable if an error response (>=400) is received.
133    *
134    * @param retries The number of retries to attempt.
135    * @param interval The time in milliseconds between attempts.
136    * @param retryOn Optional object used for determining whether a retry should be attempted.
137    *    If <jk>null</jk>, uses {@link RetryOn#DEFAULT}.
138    * @return This object (for method chaining).
139    * @throws RestCallException If current entity is not repeatable.
140    */
141   public RestCall setRetryable(int retries, long interval, RetryOn retryOn) throws RestCallException {
142      if (request instanceof HttpEntityEnclosingRequestBase) {
143      HttpEntity e = ((HttpEntityEnclosingRequestBase)request).getEntity();
144      if (e != null && ! e.isRepeatable())
145         throw new RestCallException("Attempt to make call retryable, but entity is not repeatable.");
146      }
147      this.retries = retries;
148      this.retryInterval = interval;
149      this.retryOn = (retryOn == null ? RetryOn.DEFAULT : retryOn);
150      return this;
151

453   /**
454    * Method used to execute an HTTP response where you're only interested in the HTTP response code.
455    * <p>
456    * The response entity is discarded unless one of the pipe methods have been specified to pipe the
457    *     output to an output stream or writer.
458    *
459    * <dl>
460    *    <dt>Example:</dt>
461    *    <dd>
462    * <p class='bcode'>
463    *    <jk>try</jk> {
464    *       RestClient client = <jk>new</jk> RestClient();
465    *       <jk>int</jk> rc = client.doGet(url).execute();
466    *       <jc>// Succeeded!</jc>
467    *    } <jk>catch</jk> (RestCallException e) {
468    *       <jc>// Failed!</jc>
469    *    }
470    * </p>
471    *    </dd>
472    * </dl>
473    *
474    * @return This object (for method chaining).
475    * @throws RestCallException If an exception or non-200 response code occurred during the connection attempt.
476    */
477   public int run() throws RestCallException {
478      connect();
479      try {
480         StatusLine status = response.getStatusLine();
481         int sc = status.getStatusCode();
482         if (sc >= 400 && ! ignoreErrors)
483            throw new RestCallException(sc, status.getReasonPhrase(), request.getMethod(), request.getURI(), getResponseAsString()).setHttpResponse(response);
484         if (outputStreams.size() > 0 || writers.size() > 0)
485            getReader();
486         return sc;
487      } catch (RestCallException e) {
488         isFailed = true;
489         throw e;
490      } catch (IOException e) {
491         isFailed = true;
492         throw new RestCallException(e).setHttpResponse(response);
493      } finally {
494         close();
495      }
496   }

498   /**
499    * Connects to the REST resource.
500    * <p>
501    *    If this is a <code>PUT</code> or <code>POST</code>, also sends the input to the remote resource.<br>
502    * <p>
503    *    Typically, you would only call this method if you're not interested in retrieving the body of the HTTP response.
504    *    Otherwise, you're better off just calling one of the {@link #getReader()}/{@link #getResponse(Class)}/{@link #pipeTo(Writer)}
505    *    methods directly which automatically call this method already.
506    *
507    * @return This object (for method chaining).
508    * @throws RestCallException If an exception or <code>400+</code> HTTP status code occurred during the connection attempt.
509    */
510   public RestCall connect() throws RestCallException {
511
512      if (isConnected)
513         return this;
514      isConnected = true;
515
516      try {
517         int sc = 0;
518         while (retries > 0) {
519            retries--;
520            Exception ex = null;
521            try {
522         response = client.execute(request);
523               sc = (response == null || response.getStatusLine() == null) ? -1 : response.getStatusLine().getStatusCode();
524            } catch (Exception e) {
525               ex = e;
526               sc = -1;
527               if (response != null)
528                  EntityUtils.consumeQuietly(response.getEntity());
529            }
530            if (! retryOn.onCode(sc))
531               retries = 0;
532            if (retries > 0) {
533               for (RestCallInterceptor rci : interceptors)
534                  rci.onRetry(this, sc, request, response, ex);
535               request.reset();
536               long w = retryInterval;
537               synchronized(this) {
538                  wait(w);
539               }
540            } else if (ex != null) {
541               throw ex;
542            }
543         }
544         for (RestCallInterceptor rci : interceptors)
545            rci.onConnect(this, sc, request, response);
546         if (response == null)
547            throw new RestCallException("HttpClient returned a null response");
548         StatusLine sl = response.getStatusLine();
549         String method = request.getMethod();
550         sc = sl.getStatusCode(); // Read it again in case it was changed by one of the interceptors.
551         if (sc >= 400 && ! ignoreErrors)
552            throw new RestCallException(sc, sl.getReasonPhrase(), method, request.getURI(), getResponseAsString()).setHttpResponse(response);
553         if ((sc == 307 || sc == 302) && allowRedirectsOnPosts && method.equalsIgnoreCase("POST")) {
554            if (redirectOnPostsTries-- < 1)
555               throw new RestCallException(sc, "Maximum number of redirects occurred.  Location header: " + response.getFirstHeader("Location"), method, request.getURI(), getResponseAsString());
556            Header h = response.getFirstHeader("Location");
557            if (h != null) {
558               reset();
559               request.setURI(URI.create(h.getValue()));
560               retries++;  // Redirects should affect retries.
561               connect();
562            }
563         }
564
565      } catch (RestCallException e) {
566         isFailed = true;
567         try {
568         close();
569         } catch (RestCallException e2) { /* Ignore */ }
570         throw e;
571      } catch (Exception e) {
572         isFailed = true;
573         close();
574         throw new RestCallException(e).setHttpResponse(response);
575      }
576
577      return this;
578   }
579
580   private void reset() {
581      if (response != null)
582         EntityUtils.consumeQuietly(response.getEntity());
583      request.reset();
584      isConnected = false;
585      isClosed = false;
586      isFailed = false;
587      if (capturedResponseWriter != null)
588         capturedResponseWriter.getBuffer().setLength(0);
589   }

591   /**
592    * Connects to the remote resource (if <code>connect()</code> hasn't already been called) and returns the HTTP response message body as a reader.
593    * <p>
594    *    If an {@link Encoder} has been registered with the {@link RestClient}, then the underlying input stream
595    *       will be wrapped in the encoded stream (e.g. a <code>GZIPInputStream</code>).
596    * <p>
597    *    If present, automatically handles the <code>charset</code> value in the <code>Content-Type</code> response header.
598    * <p>
599    *    <b>IMPORTANT:</b>  It is your responsibility to close this reader once you have finished with it.
600    *
601    * @return The HTTP response message body reader.  <jk>null</jk> if response was successful but didn't contain a body (e.g. HTTP 204).
602    * @throws IOException If an exception occurred while streaming was already occurring.
603    */
604   public Reader getReader() throws IOException {
605      InputStream is = getInputStream();
606      if (is == null)
607         return null;
608
609      // Figure out what the charset of the response is.
610      String cs = null;
611      Header contentType = response.getLastHeader("Content-Type");
612      String ct = contentType == null ? null : contentType.getValue();
613
614      // First look for "charset=" in Content-Type header of response.
615      if (ct != null && ct.contains("charset="))
616         cs = ct.substring(ct.indexOf("charset=")+8).trim();
617
618      if (cs == null)
619         cs = "UTF-8";
620
621      Reader isr = new InputStreamReader(is, cs);
622
623      if (writers.size() > 0) {
624         StringWriter sw = new StringWriter();
625         writers.add(sw, true);
626         IOPipe.create(isr, writers).byLines(byLines).run();
627         return new StringReader(sw.toString());
628      }
629
630      return new InputStreamReader(is, cs);
631   }

633   /**
634    * Returns the response text as a string if {@link #captureResponse()} was called on this object.
635    * <p>
636    * Note that while similar to {@link #getResponseAsString()}, this method can be called multiple times
637    *    to retrieve the response text multiple times.
638    * <p>
639    * Note that this method returns <jk>null</jk> if you have not called one of the methods that cause
640    *    the response to be processed.  (e.g. {@link #run()}, {@link #getResponse()}, {@link #getResponseAsString()}.
641    *
642    * @return The captured response, or <jk>null</jk> if {@link #captureResponse()} has not been called.
643    * @throws IllegalStateException If trying to call this method before the response is consumed.
644    */
645   public String getCapturedResponse() {
646      if (! isClosed)
647         throw new IllegalStateException("This method cannot be called until the response has been consumed.");
648      if (capturedResponse == null && capturedResponseWriter != null && capturedResponseWriter.getBuffer().length() > 0)
649         capturedResponse = capturedResponseWriter.toString();
650      return capturedResponse;
651   }

677   /**
678    * Returns the value of the <code>Content-Length</code> header.
679    *
680    * @return The value of the <code>Content-Length</code> header, or <code>-1</code> if header is not present.
681    * @throws IOException
682    */
683   public int getContentLength() throws IOException {
684      connect();
685      Header h = response.getLastHeader("Content-Length");
686      if (h == null)
687         return -1;
688      long l = Long.parseLong(h.getValue());
689      if (l > Integer.MAX_VALUE)
690         return Integer.MAX_VALUE;
691      return (int)l;
692   }

694   /**
695    * Connects to the remote resource (if <code>connect()</code> hasn't already been called) and returns the HTTP response message body as an input stream.
696    * <p>
697    *    If an {@link Encoder} has been registered with the {@link RestClient}, then the underlying input stream
698    *       will be wrapped in the encoded stream (e.g. a <code>GZIPInputStream</code>).
699    * <p>
700    *    <b>IMPORTANT:</b>  It is your responsibility to close this reader once you have finished with it.
701    *
702    * @return The HTTP response message body input stream. <jk>null</jk> if response was successful but didn't contain a body (e.g. HTTP 204).
703    * @throws IOException If an exception occurred while streaming was already occurring.
704    * @throws IllegalStateException If an attempt is made to read the response more than once.
705    */
706   public InputStream getInputStream() throws IOException {
707      if (isClosed)
708         throw new IllegalStateException("Method cannot be called.  Response has already been consumed.");
709      connect();
710      if (response == null)
711         throw new RestCallException("Response was null");
712      if (response.getEntity() == null)  // HTTP 204 results in no content.
713         return null;
714      InputStream is = response.getEntity().getContent();
715
716      if (outputStreams.size() > 0) {
717         ByteArrayInOutStream baios = new ByteArrayInOutStream();
718         outputStreams.add(baios, true);
719         IOPipe.create(is, baios).run();
720         return baios.getInputStream();
721      }
722      return is;
723   }

851   <T> T getResponse(ClassMeta<T> type) throws IOException, ParseException {
852      try {
853      Parser p = getParser();
854      T o = null;
855         if (! p.isReaderParser()) {
856         InputStream is = getInputStream();
857         o = ((InputStreamParser)p).parse(is, type);
858      } else {
859         Reader r = getReader();
860         o = ((ReaderParser)p).parse(r, type);
861         }
862      return o;
863      } catch (ParseException e) {
864         isFailed = true;
865         throw e;
866      } catch (IOException e) {
867         isFailed = true;
868         throw e;
869      } finally {
870         close();
871      }
872   }

920   /**
921    * Cleans up this HTTP call.
922    *
923    * @return This object (for method chaining).
924    * @throws RestCallException Can be thrown by one of the {@link RestCallInterceptor#onClose(RestCall)} calls.
925    */
926   public RestCall close() throws RestCallException {
927      if (response != null)
928         EntityUtils.consumeQuietly(response.getEntity());
929      isClosed = true;
930      if (! isFailed)
931         for (RestCallInterceptor r : interceptors)
932            r.onClose(this);
933      return this;
934   }
