038   /**
039    * Encodes invalid XML text characters.
040    * <p>
041    * Encodes <js>'&'</js>, <js>'&lt;'</js>, and <js>'&gt;'</js> as XML entities.<br>
042    * Encodes any other invalid XML text characters to <code>_x####_</code> sequences.
043    *
044    * @param o The object being encoded.
045    * @return The encoded string.
046    */
047   public static final String encodeText(Object o) {
048
049      if (o == null)
050         return "_x0000_";
051
052      String s = o.toString();
053
054      try {
055         if (needsTextEncoding(s))
056            return encodeTextInner(new StringBuilderWriter(s.length()*2), s).toString();
057      } catch (IOException e) {
058         throw new RuntimeException(e); // Never happens
059      }
060
061      return s;
062   }

160   private static final Writer encodeTextInner(Writer w, String s) throws IOException {
161      final int len = s.length();
162      for (int i = 0; i < len; i++) {
163         char c = s.charAt(i);
164         if (c == '&')
165            w.append("&amp;");
166         else if (c == '<')
167            w.append("&lt;");
168         else if (c == '>')
169            w.append("&gt;");
170         else if (c == '_' && isEscapeSequence(s,i))
171            appendPaddedHexChar(w, c);
172         else if ((i == 0 || i == len-1) && Character.isWhitespace(c))
173            appendPaddedHexChar(w, c);
174         else if (isValidXmlCharacter(c))
175            w.append(c);
176         else if (c == 0x09 || c == 0x0A || c == 0x0D)
177            w.append("&#x000").append(Integer.toHexString(c)).append(";");
178         else
179            appendPaddedHexChar(w, c);
180      }
181      return w;
182   }
183
184   private static final Writer encodeTextInvalidCharsInner(Writer w, String s) throws IOException {
185      final int len = s.length();
186      for (int i = 0; i < len; i++) {
187         char c = s.charAt(i);
188         if ((i == 0 || i == len-1) && Character.isWhitespace(c))
189            appendPaddedHexChar(w, c);
190         else if (c == '_' && isEscapeSequence(s,i))
191            appendPaddedHexChar(w, c);
192         else if (isValidXmlCharacter(c))
193            w.append(c);
194         else
195            appendPaddedHexChar(w, c);
196      }
197      return w;
198   }

200   private static final Writer encodeTextXmlCharsInner(Writer w, String s) throws IOException {
201      final int len = s.length();
202      for (int i = 0; i < len; i++) {
203         char c = s.charAt(i);
204         if (c == '&')
205            w.append("&amp;");
206         else if (c == '<')
207            w.append("&lt;");
208         else if (c == '>')
209            w.append("&gt;");
210         else
211            w.append(c);
212      }
213      return w;
214   }

216   private static final boolean needsTextEncoding(String s) {
217      // See if we need to convert the string.
218      // Conversion is somewhat expensive, so make sure we need to do so before hand.
219      final int len = s.length();
220      for (int i = 0; i < len; i++) {
221         char c = s.charAt(i);
222         if ((i == 0 || i == len-1) && Character.isWhitespace(c))
223            return true;
224         if (c == '&' || c == '<' || c == '>' || c == '\n' || ! isValidXmlCharacter(c) || (c == '_' && isEscapeSequence(s,i)))
225            return true;
226      }
227      return false;
228   }

235   /**
236    * Translates any _x####_ sequences (introduced by the various encode methods) back into their original characters.
237    *
238    * @param s The string being decoded.
239    * @return The decoded string.
240    */
241   public static final String decode(String s) {
242      if (s == null) return null;
243      if (s.length() == 0)
244         return s;
245      if (s.indexOf('_') == -1)
246         return s;
247
248      StringBuffer sb = new StringBuffer(s.length());
249      for (int i = 0; i < s.length(); i++) {
250         char c = s.charAt(i);
251         if (c == '_' && isEscapeSequence(s,i)) {
252
253            int x = Integer.parseInt(s.substring(i+2, i+6), 16);
254
255            // If we find _x0000_, then that means a null.
256            if (x == 0)
257               return null;
258
259            sb.append((char)x);
260            i+=6;
261         } else {
262            sb.append(c);
263         }
264      }
265      return sb.toString();
266   }

273   /**
274    * Serializes and encodes the specified object as valid XML attribute name.
275    *
276    * @param w The writer to send the output to.
277    * @param o The object being serialized.
278    * @return This object (for method chaining).
279    * @throws IOException If a problem occurred.
280    */
281   public static final Writer encodeAttr(Writer w, Object o) throws IOException {
282
283      if (o == null)
284         return w.append("_x0000_");
285
286      String s = o.toString();
287
288      if (needsAttributeEncoding(s))
289         return encodeAttrInner(w, s);
290
291      w.append(s);
292      return w;
293   }
294
295   private static final Writer encodeAttrInner(Writer w, String s) throws IOException {
296      for (int i = 0; i < s.length(); i++) {
297         char c = s.charAt(i);
298         if (c == '&')
299            w.append("&amp;");
300         else if (c == '<')
301            w.append("&lt;");
302         else if (c == '>')
303            w.append("&gt;");
304         else if (c == '\'')
305            w.append("&apos;");
306         else if (c == '"')
307            w.append("&quot;");
308         else if (c == '_' && isEscapeSequence(s,i))
309            appendPaddedHexChar(w, c);
310         else if (isValidXmlCharacter(c))
311            w.append(c);
312         else
313            appendPaddedHexChar(w, c);
314      }
315      return w;
316   }
317
318   private static boolean needsAttributeEncoding(String s) {
319      // See if we need to convert the string.
320      // Conversion is somewhat expensive, so make sure we need to do so before hand.
321      for (int i = 0; i < s.length(); i++) {
322         char c = s.charAt(i);
323         if (c == '&' || c == '<' || c == '>' || c == '\n' || c == '\'' || c == '"' || ! isValidXmlCharacter(c))
324            return true;
325      }
326      return false;
327   }

356   /**
357    * Encodes any invalid XML element name characters to <code>_x####_</code> sequences.
358    *
359    * @param o The object being encoded.
360    * @return The encoded element name string.
361    */
362   public static final String encodeElementName(Object o) {
363      if (o == null)
364         return "_x0000_";
365
366      String s = o.toString();
367
368      try {
369         if (needsElementNameEncoding(s))
370            return encodeElementNameInner(new StringBuilderWriter(s.length() * 2), s).toString();
371      } catch (IOException e) {
372         throw new RuntimeException(e); // Never happens
373      }
374
375      return s;
376   }
377
378   private static final Writer encodeElementNameInner(Writer w, String s) throws IOException {
379      for (int i = 0; i < s.length(); i++) {
380         char c = s.charAt(i);
381         if ((c >= 'A' && c <= 'Z')
382               || (c == '_' && ! isEscapeSequence(s,i))
383               || (c >= 'a' && c <= 'z')
384               || (i != 0 && (
385                     c == '-'
386                     || c == '.'
387                     || (c >= '0' && c <= '9')
388                     || c == '\u00b7'
389                     || (c >= '\u0300' && c <= '\u036f')
390                     || (c >= '\u203f' && c <= '\u2040')
391                  ))
392               || (c >= '\u00c0' && c <= '\u00d6')
393               || (c >= '\u00d8' && c <= '\u00f6')
394               || (c >= '\u00f8' && c <= '\u02ff')
395               || (c >= '\u0370' && c <= '\u037d')
396               || (c >= '\u037f' && c <= '\u1fff')
397               || (c >= '\u200c' && c <= '\u200d')
398               || (c >= '\u2070' && c <= '\u218f')
399               || (c >= '\u2c00' && c <= '\u2fef')
400               || (c >= '\u3001' && c <= '\ud7ff')
401               || (c >= '\uf900' && c <= '\ufdcf')
402               || (c >= '\ufdf0' && c <= '\ufffd')) {
403            w.append(c);
404         }  else {
405            appendPaddedHexChar(w, c);
406         }
407      }
408      return w;
409   }
410
411   private static final boolean needsElementNameEncoding(String s) {
412      // Note that this doesn't need to be perfect, just fast.
413      for (int i = 0; i < s.length(); i++) {
414         char c = s.charAt(i);
415         if (! (c >= '0' && c <= '9' || c >= 'a' && c <= 'z' || c >= 'A' && c <= 'Z'))
416            return true;
417         if (i == 0 && (c >= '0' && c <= '9'))
418            return true;
419      }
420      return false;
421   }

428   /**
429    * Utility method for reading XML mixed content from an XML element and returning it as text.
430    *
431    * @param r The reader to read from.
432    * @return The contents read as a string.
433    * @throws XMLStreamException
434    * @throws IOException
435    */
436   public static String readXmlContents(XMLStreamReader r) throws XMLStreamException, IOException {
437      StringWriter sw = new StringWriter();
438      XmlWriter w = new XmlWriter(sw, false, false, '"', null, null, false, null);
439      try {
440         int depth = 0;
441         do {
442            int event = r.next();
443            if (event == START_ELEMENT) {
444               depth++;
445               QName n = r.getName();
446               w.oTag(n.getPrefix(), n.getLocalPart());
447               for (int i = 0; i < r.getNamespaceCount(); i++)
448                  w.attr(r.getNamespacePrefix(i), "xmlns", r.getNamespaceURI(i));
449               for (int i = 0; i < r.getAttributeCount(); i++)
450                  w.attr(r.getAttributePrefix(i), r.getAttributeLocalName(i), r.getAttributeValue(i));
451               w.append('>');
452            } else if (r.hasText()) {
453               w.encodeTextXmlChars(r.getText());
454            } else if (event == ATTRIBUTE) {
455               // attributes handled above.
456            } else if (event == END_ELEMENT) {
457               QName n = r.getName();
458               if (depth > 0)
459                  w.eTag(n.getPrefix(), n.getLocalPart());
460               depth--;
461            }
462            if (depth < 0)
463               return sw.toString();
464         } while (true);
465      } finally {
466         w.close();
467      }
468   }

470   // Returns true if the specified character can safely be used in XML text or an attribute.
471   private static final boolean isValidXmlCharacter(char c) {
472      return (c >= 0x20 && c <= 0xD7FF) /*|| c == 0xA || c == 0xD*/ || (c >= 0xE000 && c <= 0xFFFD);
473   }
474
475   // Returns true if the string at the specified position is of the form "_x####_"
476   // where '#' are hexadecimal characters.
477   private static final boolean isEscapeSequence(String s, int i) {
478      return s.length() > i+6
479         && s.charAt(i) == '_'
480         && s.charAt(i+1) == 'x'
481         && isHexCharacter(s.charAt(i+2))
482         && isHexCharacter(s.charAt(i+3))
483         && isHexCharacter(s.charAt(i+4))
484         && isHexCharacter(s.charAt(i+5))
485         && s.charAt(i+6) == '_';
486   }
487
488   // Returns true if the character is a hexadecimal character
489   private static final boolean isHexCharacter(char c) {
490      return (c >= '0' && c <= '9') || (c >= 'A' && c <= 'F');
491   }
492
493   // Converts an integer to a hexadecimal string padded to 4 places.
494   private static final Writer appendPaddedHexChar(Writer out, int num) throws IOException {
495      out.append("_x");
496      char[] n = new char[4];
497      int a = num%16;
498      n[3] = (char)(a > 9 ? 'A'+a-10 : '0'+a);
499      int base = 16;
500      for (int i = 1; i < 4; i++) {
501         a = (num/base)%16;
502         base <<= 4;
503         n[3-i] = (char)(a > 9 ? 'A'+a-10 : '0'+a);
504      }
505      for (int i = 0; i < 4; i++)
506         out.append(n[i]);
507      return out.append('_');
508   }

510   /**
511    * Find the namespace given a list of <ja>@Xml</ja> and <ja>@XmlSchema</ja> annotations.
512    * The annotations should be a child-to-parent ordering of annotations found on
513    *    a class or method.
514    *
515    * @param xmls The list of <ja>@Xml</ja> annotations.
516    * @param schemas The list of <ja>@XmlSchema</ja> annotations.
517    * @return The namespace, or <jk>null</jk> if it couldn't be found.
518    */
519   public static Namespace findNamespace(List<Xml> xmls, List<XmlSchema> schemas) {
520
521      for (Xml xml : xmls) {
522         Namespace ns = findNamespace(xml.prefix(), xml.namespace(), xmls, schemas);
523         if (ns != null)
524            return ns;
525      }
526
527      for (XmlSchema schema : schemas) {
528         Namespace ns = findNamespace(schema.prefix(), schema.namespace(), null, schemas);
529         if (ns != null)
530            return ns;
531      }
532
533      return null;
534   }
535
536   private static Namespace findNamespace(String prefix, String ns, List<Xml> xmls, List<XmlSchema> schemas) {
537
538      // If both prefix and namespace specified, use that Namespace mapping.
539      if (! (prefix.isEmpty() || ns.isEmpty()))
540         return NamespaceFactory.get(prefix, ns);
541
542      // If only prefix specified, need to search for namespaceURI.
543      if (! prefix.isEmpty()) {
544         if (xmls != null)
545            for (Xml xml2 : xmls)
546               if (xml2.prefix().equals(prefix) && ! xml2.namespace().isEmpty())
547                  return NamespaceFactory.get(prefix, xml2.namespace());
548         for (XmlSchema schema : schemas) {
549            if (schema.prefix().equals(prefix) && ! schema.namespace().isEmpty())
550               return NamespaceFactory.get(prefix, schema.namespace());
551            for (XmlNs xmlNs : schema.xmlNs())
552               if (xmlNs.prefix().equals(prefix))
553                  return NamespaceFactory.get(prefix, xmlNs.namespaceURI());
554         }
555         throw new BeanRuntimeException("Found @Xml.prefix annotation with no matching URI.  prefix='"+prefix+"'");
556      }
557
558      // If only namespaceURI specified, need to search for prefix.
559      if (! ns.isEmpty()) {
560         if (xmls != null)
561            for (Xml xml2 : xmls)
562               if (xml2.namespace().equals(ns) && ! xml2.prefix().isEmpty())
563                  return NamespaceFactory.get(xml2.prefix(), ns);
564         for (XmlSchema schema : schemas) {
565            if (schema.namespace().equals(ns) && ! schema.prefix().isEmpty())
566               return NamespaceFactory.get(schema.prefix(), ns);
567            for (XmlNs xmlNs : schema.xmlNs())
568               if (xmlNs.namespaceURI().equals(ns))
569                  return NamespaceFactory.get(xmlNs.prefix(), ns);
570         }
571      }
572
573      return null;
574   }
