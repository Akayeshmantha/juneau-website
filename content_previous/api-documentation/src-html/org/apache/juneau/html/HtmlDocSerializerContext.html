013package org.apache.juneau.html;
014
015import java.util.*;
016
017import org.apache.juneau.*;
018
019/**
020 * Properties associated with the {@link HtmlDocSerializer} class.
021 * <p>
022 *    These are typically specified via <ja>@RestResource.properties()</ja> and <ja>@RestMethod.properties()</ja> annotations,
023 *       although they can also be set programmatically via the <code>RestResponse.setProperty()</code> method.
024 *
025 * <h6 class='topic'>Example</h6>
026 * <p class='bcode'>
027 *    <ja>@RestResource</ja>(
028 *       messages=<js>"nls/AddressBookResource"</js>,
029 *       properties={
030 *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_title</jsf>, value=<js>"$L{title}"</js>),
031 *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_description</jsf>, value=<js>"$L{description}"</js>),
032 *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_links</jsf>, value=<js>"{options:'?method=OPTIONS',doc:'doc'}"</js>)
033 *       }
034 *    )
035 *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
036 * </p>
037 * <p>
038 *    The <code>$L{...}</code> variable represent localized strings pulled from the resource bundle identified by the <code>messages</code> annotation.
039 *    These variables are replaced at runtime based on the HTTP request locale.
040 *    Several built-in runtime variable types are defined, and the API can be extended to include user-defined variables.
041 * </p>
042 *
043 * @author James Bognar (james.bognar@salesforce.com)
044 */
045public final class HtmlDocSerializerContext extends HtmlSerializerContext {
046
047   /**
048    * Adds a title at the top of a page.
049    *
050    * <dl>
051    *    <dt>Example:</dt>
052    *    <dd>
053    * <p>
054    *    The <code>AddressBookResource</code> sample class uses this property...
055    * </p>
056    * <p class='bcode'>
057    *    <ja>@RestResource</ja>(
058    *       messages=<js>"nls/AddressBookResource"</js>,
059    *       properties={
060    *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_title</jsf>, value=<js>"$L{title}"</js>)
061    *       }
062    *    )
063    *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
064    * </p>
065    * <p>
066    *    ...with this property in <code>AddressBookResource.properties</code>...
067    * </p>
068    * <p class='bcode'>
069    *    title = <js>AddressBook sample resource</js>
070    * </p>
071    * <p>
072    *    ...to produce this title on the HTML page...
073    * </p>
074    *       <img class='bordered' src='doc-files/HTML_TITLE.png'>
075    *    </dd>
076    * </dl>
077    */
078   public static final String HTMLDOC_title = "HtmlSerializer.title";
079
080   /**
081    * Adds a description right below the title of a page.
082    *
083    * <dl>
084    *    <dt>Example:</dt>
085    *    <dd>
086    * <p>
087    *    The <code>AddressBookResource</code> sample class uses this property...
088    * </p>
089    * <p class='bcode'>
090    *    <ja>@RestResource</ja>(
091    *       messages=<js>"nls/AddressBookResource"</js>,
092    *       properties={
093    *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_description</jsf>, value=<js>"description"</js>, type=<jsf>NLS</jsf>)
094    *       }
095    *    )
096    *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
097    * </p>
098    * <p>
099    *    ...with this property in <code>AddressBookResource.properties</code>...
100    * </p>
101    * <p class='bcode'>
102    *    description = <js>Simple address book POJO sample resource</js>
103    * </p>
104    * <p>
105    *    ...to produce this description on the HTML page...
106    * </p>
107    *       <img class='bordered' src='doc-files/HTML_DESCRIPTION.png'>
108    *    </dd>
109    * </dl>
110    */
111   public static final String HTMLDOC_description = "HtmlSerializer.description";
112
113   /**
114    * Adds a list of hyperlinks immediately under the title and description but above the content of the page.
115    * <p>
116    *    This can be used to provide convenient hyperlinks when viewing the REST interface from a browser.
117    * <p>
118    *    The value is a JSON object string where the keys are anchor text and the values are URLs.
119    * <p>
120    *    Relative URLs are considered relative to the servlet path.
121    *    For example, if the servlet path is <js>"http://localhost/myContext/myServlet"</js>, and the
122    *       URL is <js>"foo"</js>, the link becomes <js>"http://localhost/myContext/myServlet/foo"</js>.
123    *    Absolute (<js>"/myOtherContext/foo"</js>) and fully-qualified (<js>"http://localhost2/foo"</js>) URLs
124    *       can also be used.
125    *
126    * <dl>
127    *    <dt>Example:</dt>
128    *    <dd>
129    * <p>
130    *    The <code>AddressBookResource</code> sample class uses this property...
131    * </p>
132    * <p class='bcode'>
133    *    <ja>@RestResource</ja>(
134    *       messages=<js>"nls/AddressBookResource"</js>,
135    *       properties={
136    *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_links</jsf>, value=<js>"{options:'?method=OPTIONS',doc:'doc'}"</js>)
137    *       }
138    *    )
139    *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
140    * </p>
141    * <p>
142    *    ...to produce this list of links on the HTML page...
143    * </p>
144    *       <img class='bordered' src='doc-files/HTML_LINKS.png'>
145    *    </dd>
146    * </dl>
147    */
148   public static final String HTMLDOC_links = "HtmlDocSerializer.links";
149
150   /**
151    * Similar to {@link #HTMLDOC_links} except appends on to the existing list of links.
152    */
153   public static final String HTMLDOC_links_add = "HtmlDocSerializer.links.add";
154
155   /**
156    * Adds a link to the specified stylesheet URL (<l>String</l>, default=<jk>null</jk>).
157    * <p>
158    *    If not specified, defaults to the built-in stylesheet located at <js>"/servletPath/style.css"</js>.
159    *    Note that this stylesheet is controlled by the <code><ja>@RestResource</ja>.style()</code> annotation.
160    */
161   public static final String HTMLDOC_cssUrl = "HtmlDocSerializer.cssUrl";
162
163   /**
164    * Imports the specified CSS page URLs into the page (<l>String[]</l>, default=<code>[]</code>).
165    */
166   public static final String HTMLDOC_cssImports = "HtmlDocSerializer.cssImports";
167
168   /**
169    * Append to the {@link #HTMLDOC_cssImports} property.
170    */
171   public static final String HTMLDOC_cssImports_add = "HtmlDocSerializer.cssImports.add";
172
173   /**
174    * Adds <js>"* {white-space:nowrap}"</js> to the style header to prevent word wrapping.
175    */
176   public static final String HTMLDOC_nowrap = "HtmlDocSerializer.nowrap";
177
178   final String[] cssImports;
179   final Map<String,String> links;
180   final String title, description, cssUrl;
181   final boolean nowrap;
182
183   /**
184    * Constructor.
185    * <p>
186    * Typically only called from {@link ContextFactory#getContext(Class)}.
187    *
188    * @param cf The factory that created this context.
189    */
190   public HtmlDocSerializerContext(ContextFactory cf) {
191      super(cf);
192      cssImports = cf.getProperty(HTMLDOC_cssImports, String[].class, new String[0]);
193      title = cf.getProperty(HTMLDOC_title, String.class, null);
194      description = cf.getProperty(HTMLDOC_description, String.class, null);
195      cssUrl = cf.getProperty(HTMLDOC_cssUrl, String.class, null);
196      nowrap = cf.getProperty(HTMLDOC_nowrap, boolean.class, false);
197      links = cf.getMap(HTMLDOC_links, String.class, String.class, Collections.<String,String>emptyMap());
198   }
199}
