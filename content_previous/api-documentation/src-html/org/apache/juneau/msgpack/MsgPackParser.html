025/**
026 * Parses a MessagePack stream into a POJO model.
027 *
028 *
029 * <h6 class='topic'>Media types</h6>
030 * <p>
031 *    Handles <code>Content-Type</code> types: <code>octal/msgpack</code>
032 *
033 *
034 * <h6 class='topic'>Configurable properties</h6>
035 * <p>
036 *    This class has the following properties associated with it:
037 * <ul>
038 *    <li>{@link MsgPackParserContext}
039 * </ul>
040 *
041 *
042 * @author James Bognar (james.bognar@salesforce.com)
043 */
044@SuppressWarnings({ "rawtypes", "unchecked" })
045@Consumes({"octal/msgpack"})
046public final class MsgPackParser extends InputStreamParser {
047
048   /** Default parser, all default settings.*/
049   public static final MsgPackParser DEFAULT = new MsgPackParser().lock();
050
051   /**
052    * Workhorse method.
053    */
054   private <T> T parseAnything(MsgPackParserSession session, ClassMeta<T> nt, MsgPackInputStream is, Object outer) throws Exception {
055
056      BeanContext bc = session.getBeanContext();
057      if (nt == null)
058         nt = (ClassMeta<T>)object();
059      PojoTransform<T,Object> transform = (PojoTransform<T,Object>)nt.getPojoTransform();
060      ClassMeta<?> ft = nt.getTransformedClassMeta();
061      session.setCurrentClass(ft);
062
063      Object o = null;
064      DataType dt = is.readDataType();
065      int length = (int)is.readLength();
066
067      if (dt != DataType.NULL) {
068         if (dt == BOOLEAN)
069            o = is.readBoolean();
070         else if (dt == INT)
071            o = is.readInt();
072         else if (dt == LONG)
073            o = is.readLong();
074         else if (dt == FLOAT)
075            o = is.readFloat();
076         else if (dt == DOUBLE)
077            o = is.readDouble();
078         else if (dt == STRING)
079            o = session.trim(is.readString());
080         else if (dt == BIN)
081            o = is.readBinary();
082         else if (dt == ARRAY && ft.isObject()) {
083            ObjectList ol = new ObjectList(bc);
084            for (int i = 0; i < length; i++)
085               ol.add(parseAnything(session, object(), is, outer));
086            o = ol;
087         } else if (dt == MAP && ft.isObject()) {
088            ObjectMap om = new ObjectMap(bc);
089            for (int i = 0; i < length; i++)
090               om.put(parseAnything(session, string(), is, outer), parseAnything(session, object(), is, om));
091            o = om.cast();
092         }
093
094         if (ft.isObject()) {
095            // Do nothing.
096         } else if (ft.isBoolean() || ft.isCharSequence() || ft.isChar() || ft.isNumber()) {
097            o = bc.convertToType(o, ft);
098         } else if (ft.isMap()) {
099            if (dt == MAP) {
100               Map m = (ft.canCreateNewInstance(outer) ? (Map)ft.newInstance(outer) : new ObjectMap(bc));
101               for (int i = 0; i < length; i++) {
102                  Object key = parseAnything(session, ft.getKeyType(), is, outer);
103                  ClassMeta<?> vt = ft.getValueType();
104                  Object value = parseAnything(session, vt, is, m);
105                  setName(vt, value, key);
106                  m.put(key, value);
107               }
108               o = m;
109            } else {
110               throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, ft);
111            }
112         } else if (ft.canCreateNewInstanceFromObjectMap(outer)) {
113            ObjectMap m = new ObjectMap(bc);
114            for (int i = 0; i < length; i++)
115               m.put(parseAnything(session, string(), is, outer), parseAnything(session, object(), is, m));
116            o = ft.newInstanceFromObjectMap(outer, m);
117         } else if (ft.canCreateNewBean(outer)) {
118            if (dt == MAP) {
119               BeanMap m = bc.newBeanMap(outer, ft.getInnerClass());
120               for (int i = 0; i < length; i++) {
121                  String pName = parseAnything(session, string(), is, m.getBean(false));
122                  BeanPropertyMeta<?> bpm = m.getPropertyMeta(pName);
123                  if (bpm == null) {
124                     if (pName.equals("_class"))
125                        parseAnything(session, bc.string(), is, null);
126                     else
127                        onUnknownProperty(session, pName, m, 0, is.getPosition());
128                  } else {
129                     ClassMeta<?> cm = bpm.getClassMeta();
130                     Object value = parseAnything(session, cm, is, m.getBean(false));
131                     setName(cm, value, pName);
132                     bpm.set(m, value);
133                  }
134               }
135               o = m.getBean();
136            } else {
137               throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, ft);
138            }
139         } else if (ft.canCreateNewInstanceFromString(outer) && dt == STRING) {
140            o = ft.newInstanceFromString(outer, o == null ? "" : o.toString());
141         } else if (ft.canCreateNewInstanceFromNumber(outer) && dt.isOneOf(INT, LONG, FLOAT, DOUBLE)) {
142            o = ft.newInstanceFromNumber(outer, (Number)o);
143         } else if (ft.isCollection()) {
144            if (dt == MAP) {
145               ObjectMap m = new ObjectMap(bc);
146               for (int i = 0; i < length; i++)
147                  m.put(parseAnything(session, string(), is, outer), parseAnything(session, object(), is, m));
148               o = m.cast();
149            } else if (dt == ARRAY) {
150               Collection l = (ft.canCreateNewInstance(outer) ? (Collection)ft.newInstance() : new ObjectList(bc));
151               for (int i = 0; i < length; i++)
152                  l.add(parseAnything(session, ft.getElementType(), is, l));
153               o = l;
154            } else {
155               throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, ft);
156            }
157         } else if (ft.isArray()) {
158            if (dt == MAP) {
159               ObjectMap m = new ObjectMap(bc);
160               for (int i = 0; i < length; i++)
161                  m.put(parseAnything(session, string(), is, outer), parseAnything(session, object(), is, m));
162               o = m.cast();
163            } else if (dt == ARRAY) {
164               Collection l = (ft.isCollection() && ft.canCreateNewInstance(outer) ? (Collection)ft.newInstance() : new ObjectList(bc));
165               for (int i = 0; i < length; i++)
166                  l.add(parseAnything(session, ft.getElementType(), is, l));
167               o = bc.toArray(ft, l);
168            } else {
169               throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, ft);
170            }
171         } else if (dt == MAP) {
172            ObjectMap m = new ObjectMap(bc);
173            for (int i = 0; i < length; i++)
174               m.put(parseAnything(session, string(), is, outer), parseAnything(session, object(), is, m));
175            if (m.containsKey("_class"))
176               o = m.cast();
177            else
178               throw new ParseException(session, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", ft.getInnerClass().getName(), ft.getNotABeanReason());
179         } else {
180            throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, ft);
181         }
182      }
183
184      if (transform != null && o != null)
185         o = transform.normalize(o, nt);
186
187      if (outer != null)
188         setParent(nt, o, outer);
189
190      return (T)o;
191   }

194   // Overridden methods

197   @Override /* Parser */
198   public MsgPackParserSession createSession(Object input, ObjectMap op, Method javaMethod, Object outer) {
199      return new MsgPackParserSession(getContext(MsgPackParserContext.class), getBeanContext(), input, op, javaMethod, outer);
200   }
201
202   @Override /* Parser */
203   protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
204      MsgPackParserSession s = (MsgPackParserSession)session;
205      type = s.getBeanContext().normalizeClassMeta(type);
206      MsgPackInputStream is = s.getInputStream();
207      T o = parseAnything(s, type, is, s.getOuter());
208      return o;
209   }
210
