013package org.apache.juneau;
014
015/**
016 * Superclass of all classes that have a locked state.
017 * <p>
018 * Used to mark bean contexts, serializers, and parsers as read-only so that
019 *    settings can no longer be modified.
020 * <p>
021 * Also keeps track of when the object has been cloned and allows for lazy cloning through
022 *    the {@link #onUnclone()} method.  The idea behind this is that certain expensive fields don't
023 *    need to be cloned unless the object is actually being modified.
024 * <p>
025 * Calling {@link #lock()} on the object causes it to be put into a read-only state.
026 * Once called, subsequent calls to {@link #checkLock()} will cause {@link LockedException LockedExceptions}
027 *    to be thrown.
028 * <p>
029 * As a rule, cloned objects are unlocked by default.
030 *
031 * @author James Bognar (james.bognar@salesforce.com)
032 */
033public abstract class Lockable implements Cloneable {
034
035   private boolean isLocked = false;
036   private boolean isCloned = false;
037
038   /**
039    * Locks this object so that settings on it cannot be modified.
040    *
041    * @return This object (for method chaining).
042    */
043   public Lockable lock() {
044      isLocked = true;
045      return this;
046   }
047
048   /**
049    * @return <code><jk>true</jk></code> if this object has been locked.
050    */
051   public boolean isLocked() {
052      return isLocked;
053   }
054
055   /**
056    * Causes a {@link LockedException} to be thrown if this object has been locked.
057    * <p>
058    *    Also calls {@link #onUnclone()} if this is the first time this method has been called since cloning.
059    *
060    * @throws LockedException If {@link #lock()} has been called on this object.
061    */
062   public void checkLock() throws LockedException {
063      if (isLocked)
064         throw new LockedException();
065      if (isCloned)
066         onUnclone();
067      isCloned = false;
068   }
069
070   /**
071    * Subclass can override this method to handle lazy-cloning on the first time {@link #checkLock()} is called after
072    * the object has been cloned.
073    */
074   public void onUnclone() {}
075
076   /**
077    * Creates an unlocked clone of this object.
078    *
079    * @throws CloneNotSupportedException If class cannot be cloned.
080    */
081   @Override /* Object */
082   public Lockable clone() throws CloneNotSupportedException {
083      Lockable c = (Lockable)super.clone();
084      c.isLocked = false;
085      c.isCloned = true;
086      return c;
087   }
088}
