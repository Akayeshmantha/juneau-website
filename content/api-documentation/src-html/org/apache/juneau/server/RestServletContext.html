016
017import org.apache.juneau.*;

023/**
024 * Configurable properties on the {@link RestServlet} class.
025 * <p>
026 * Properties can be set on the {@link RestServlet} class using the {@link RestResource#properties} or {@link RestMethod#properties} annotations.
027 * <p>
028 * These properties can also be passed in as servlet init parameters or system properties.
029 * <p>
030 * Some of these properties are only applicable on the servlet class, and others can be specified on the servlet class or method.<br>
031 * These distinctions are noted below.
032 * <p>
033 * See {@link ContextFactory} for more information about context properties.
034 *
035 * @author James Bognar (james.bognar@salesforce.com)
036 */
037public final class RestServletContext extends Context {
038
039   /**
040    * Allow header URL parameters ({@link Boolean}, default=<jk>true</jk>).
041    * <p>
042    * When enabled, headers such as <js>"Accept"</js> and <js>"Content-Type"</js> to be passed in as URL query parameters.
043    * For example:  <js>"?Accept=text/json&Content-Type=text/json"</js>
044    * <p>
045    * Parameter names are case-insensitive.
046    * <p>
047    * Useful for debugging REST interface using only a browser.
048    * <p>
049    * Applicable to servlet class only.
050    */
051   public static final String REST_allowHeaderParams = "RestServlet.allowHeaderParams";
052
053   /**
054    * Allow <js>"method"</js> URL parameter for specific HTTP methods (String, default=<js>""</js>, example=<js>"HEAD,OPTIONS"</js>).
055    * <p>
056    * When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular GET request.
057    * For example:  <js>"?method=OPTIONS"</js>
058    * <p>
059    * Parameter name is case-insensitive.  Use "*" to represent all methods.  For backwards compatibility, "true" also means "*".
060    * <p>
061    * Note that per the <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec9.html">HTTP specification</a>, special care should
062    *    be taken when allowing non-safe (POST, PUT, DELETE) methods to be invoked through GET requests.
063    * <p>
064    * Applicable to servlet class only.
065    */
066   public static final String REST_allowMethodParam = "RestServlet.allowMethodParam";
067
068   /**
069    * Allow <js>"content"</js> URL parameter ({@link Boolean}, default=<jk>true</jk>).
070    * <p>
071    * When enabled, the HTTP body content on PUT and POST requests can be passed in as text using the <js>"content"</js> URL parameter.
072    * For example:  <js>"?content={name:'John%20Smith',age:45}"</js>
073    * <p>
074    * Parameter name is case-insensitive.
075    * <p>
076    * Useful for debugging PUT and POST methods using only a browser.
077    * <p>
078    * Applicable to servlet class only.
079    */
080   public static final String REST_allowContentParam = "RestServlet.allowContentParam";
081
082   /**
083    * Render stack traces in HTTP response bodies when errors occur ({@link Boolean}, default=<jk>false</jk>).
084    * <p>
085    * When enabled, Java stack traces will be rendered in the output response.
086    * Useful for debugging, although allowing stack traces to be rendered may cause security concerns.
087    * <p>
088    * Applicable to servlet class only.
089    */
090   public static final String REST_renderResponseStackTraces = "RestServlet.renderResponseStackTraces";
091
092   /**
093    * Use stack trace hashes ({@link Boolean}, default=<jk>true</jk>).
094    * <p>
095    * When enabled, the number of times an exception has occurred will be determined based on stack trace hashsums,
096    * made available through the {@link RestException#getOccurrence()} method.
097    * <p>
098    * Applicable to servlet class only.
099    */
100   public static final String REST_useStackTraceHashes = "RestServlet.useStackTraceHashes";
101
102   /**
103    * The default character encoding for the request and response if not specified on the request ({@link String}>, default=<js>"utf-8"</js>).
104    * <p>
105    * Applicable to servlet class and methods.
106    */
107   public static final String REST_defaultCharset = "RestServlet.defaultCharset";
108
109   /**
110    * The expected format of request parameters ({@link String}, default=<js>"UON"</js>).
111    * <p>
112    * Possible values:
113    * <ul class='spaced-list'>
114    *    <li><js>"UON"</js> - URL-Encoded Object Notation.<br>
115    *       This notation allows for request parameters to contain arbitrarily complex POJOs.
116    *    <li><js>"PLAIN"</js> - Plain text.<br>
117    *       This treats request parameters as plain text.<br>
118    *       Only POJOs directly convertable from <l>Strings</l> can be represented in parameters when using this mode.
119    * </ul>
120    * <p>
121    * Note that the parameter value <js>"(foo)"</js> is interpreted as <js>"(foo)"</js> when using plain mode, but
122    *    <js>"foo"</js> when using UON mode.
123    * <p>
124    * The format can also be specified per-parameter using the {@link Param#format() @Param.format()} and {@link QParam#format() @QParam.format()}
125    *    annotations.
126    * <p>
127    * Applicable to servlet class and methods.
128    */
129   public static final String REST_paramFormat = "RestServlet.paramFormat";
130
131   //--------------------------------------------------------------------------------
132   // Automatically added properties.
133   //--------------------------------------------------------------------------------
134
135   /**
136    * The request servlet path.
137    * <p>
138    * Automatically added to properties return by {@link RestServlet#createRequestProperties(org.apache.juneau.ObjectMap, RestRequest)}
139    *    and are therefore available through {@link SerializerSession#getProperties()} and {@link ParserSession#getProperties()}.
140    * <p>
141    * Equivalent to the value returned by {@link RestRequest#getServletPath()}
142    */
143   public static final String REST_servletPath = "RestServlet.servletPath";
144
145   /**
146    * The request servlet URI.
147    * <p>
148    * Equivalent to the value returned by {@link RestRequest#getServletURI()}
149    */
150   public static final String REST_servletURI = "RestServlet.servletURI";
151
152   /**
153    * The request servlet URI.
154    * <p>
155    * Equivalent to the value returned by {@link RestRequest#getRelativeServletURI()}
156    */
157   public static final String REST_relativeServletURI = "RestServlet.relativeServletURI";
158
159   /**
160    * The request URI path info.
161    * <p>
162    * Automatically added to properties return by {@link RestServlet#createRequestProperties(org.apache.juneau.ObjectMap, RestRequest)}
163    *    and are therefore available through {@link SerializerSession#getProperties()} and {@link ParserSession#getProperties()}.
164    * <p>
165    * Equivalent to the value returned by {@link RestRequest#getPathInfo()}
166    */
167   public static final String REST_pathInfo = "RestServlet.pathInfo";
168
169   /**
170    * The request URI.
171    * <p>
172    * Automatically added to properties return by {@link RestServlet#createRequestProperties(org.apache.juneau.ObjectMap, RestRequest)}
173    *    and are therefore available through {@link SerializerSession#getProperties()} and {@link ParserSession#getProperties()}.
174    * <p>
175    * Equivalent to the value returned by {@link RestRequest#getRequestURI()}
176    */
177   public static final String REST_requestURI = "RestServlet.requestURI";
178
179   /**
180    * The request method.
181    * <p>
182    * Automatically added to properties return by {@link RestServlet#createRequestProperties(org.apache.juneau.ObjectMap, RestRequest)}
183    *    and are therefore available through {@link SerializerSession#getProperties()} and {@link ParserSession#getProperties()}.
184    * <p>
185    * Equivalent to the value returned by {@link RestRequest#getMethod()}
186    */
187   public static final String REST_method = "RestServlet.method";
188
189
190   final boolean allowHeaderParams, allowContentParam, renderResponseStackTraces, useStackTraceHashes;
191   final String defaultCharset, paramFormat;
192   final Set<String> allowMethodParams;
193
194   /**
195    * Constructor.
196    * <p>
197    * Typically only called from {@link ContextFactory#getContext(Class)}.
198    *
199    * @param cf The factory that created this context.
200    */
201   public RestServletContext(ContextFactory cf) {
202      super(cf);
203      allowHeaderParams = cf.getProperty(REST_allowHeaderParams, boolean.class, true);
204      allowContentParam = cf.getProperty(REST_allowContentParam, boolean.class, true);
205      renderResponseStackTraces = cf.getProperty(REST_renderResponseStackTraces, boolean.class, false);
206      useStackTraceHashes = cf.getProperty(REST_useStackTraceHashes, boolean.class, true);
207      defaultCharset = cf.getProperty(REST_defaultCharset, String.class, "utf-8");
208      paramFormat = cf.getProperty(REST_paramFormat, String.class, "");
209
210      Set<String> s = new LinkedHashSet<String>();
211      for (String m : StringUtils.split(cf.getProperty(REST_allowMethodParam, String.class, ""), ','))
212         if (m.equals("true"))  // For backwards compatibility when this was a boolean field.
213            s.add("*");
214         else
215            s.add(m.toUpperCase());
216      allowMethodParams = Collections.unmodifiableSet(s);
217   }
