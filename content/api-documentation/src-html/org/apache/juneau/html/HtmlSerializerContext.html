013package org.apache.juneau.html;
014
015import org.apache.juneau.*;
016import org.apache.juneau.xml.*;
017
018/**
019 * Configurable properties on the {@link HtmlSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * The following convenience methods are also provided for setting context properties:
025 * <ul>
026 *    <li>{@link HtmlSerializer#setProperty(String,Object)}
027 *    <li>{@link HtmlSerializer#setProperties(ObjectMap)}
028 *    <li>{@link HtmlSerializer#addNotBeanClasses(Class[])}
029 *    <li>{@link HtmlSerializer#addTransforms(Class[])}
030 *    <li>{@link HtmlSerializer#addImplClass(Class,Class)}
031 * </ul>
032 * <p>
033 * See {@link ContextFactory} for more information about context properties.
034 *
035 * @author James Bognar (james.bognar@salesforce.com)
036 */
037public class HtmlSerializerContext extends XmlSerializerContext {
038
039   /**
040    * Anchor text source ({@link String}, default={@link #TO_STRING}).
041    * <p>
042    * When creating anchor tags (e.g. <code><xt>&lt;a</xt> <xa>href</xa>=<xs>'...'</xs><xt>&gt;</xt>text<xt>&lt;/a&gt;</xt></code>)
043    *    in HTML, this setting defines what to set the inner text to.
044    * <p>
045    * Possible values:
046    * <ul class='spaced-list'>
047    *    <li>{@link #TO_STRING} / <js>"toString"</js> - Set to whatever is returned by {@link #toString()} on the object.
048    *    <li>{@link #URI} / <js>"uri"</js> - Set to the URI value.
049    *    <li>{@link #LAST_TOKEN} / <js>"lastToken"</js> - Set to the last token of the URI value.
050    *    <li>{@link #PROPERTY_NAME} / <js>"propertyName"</js> - Set to the bean property name.
051    *    <li>{@link #URI_ANCHOR} / <js>"uriAnchor"</js> - Set to the anchor of the URL.  (e.g. <js>"http://localhost:9080/foobar#anchorTextHere"</js>)
052    * </ul>
053    */
054   public static final String HTML_uriAnchorText = "HtmlSerializer.uriAnchorText";
055
056   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
057   public static final String PROPERTY_NAME = "PROPERTY_NAME";
058   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
059   public static final String TO_STRING = "TO_STRING";
060   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
061   public static final String URI = "URI";
062   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
063   public static final String LAST_TOKEN = "LAST_TOKEN";
064   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
065   public static final String URI_ANCHOR = "URI_ANCHOR";
066
067
068   /**
069    * Look for URLs in {@link String Strings} ({@link Boolean}, default=<jk>true</jk>).
070    * <p>
071    * If a string looks like a URL (e.g. starts with <js>"http://"</js> or <js>"https://"</js>, then treat it like a URL
072    *    and make it into a hyperlink based on the rules specified by {@link #HTML_uriAnchorText}.
073    */
074   public static final String HTML_detectLinksInStrings = "HtmlSerializer.detectLinksInStrings";
075
076   /**
077    * Look for link labels in the <js>"label"</js> parameter of the URL ({@link Boolean}, default=<jk>true</jk>).
078    * <p>
079    * If the URL has a label parameter (e.g. <js>"?label=foobar"</js>), then use that as the anchor text of the link.
080    * <p>
081    * The parameter name can be changed via the {@link #HTML_labelParameter} property.
082    */
083   public static final String HTML_lookForLabelParameters = "HtmlSerializer.lookForLabelParameters";
084
085   /**
086    * The parameter name to use when using {@link #HTML_lookForLabelParameters} ({@link String}, default=<js>"label"</js>).
087    */
088   public static final String HTML_labelParameter = "HtmlSerializer.labelParameter";
089
090   final String uriAnchorText;
091   final boolean lookForLabelParameters, detectLinksInStrings;
092   final String labelParameter;
093
094   /**
095    * Constructor.
096    * <p>
097    * Typically only called from {@link ContextFactory#getContext(Class)}.
098    *
099    * @param cf The factory that created this context.
100    */
101   public HtmlSerializerContext(ContextFactory cf) {
102      super(cf);
103      uriAnchorText = cf.getProperty(HTML_uriAnchorText, String.class, TO_STRING);
104      lookForLabelParameters = cf.getProperty(HTML_lookForLabelParameters, Boolean.class, true);
105      detectLinksInStrings = cf.getProperty(HTML_detectLinksInStrings, Boolean.class, true);
106      labelParameter = cf.getProperty(HTML_labelParameter, String.class, "label");
107   }
108}
