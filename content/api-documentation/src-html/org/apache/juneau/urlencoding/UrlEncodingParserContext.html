013package org.apache.juneau.urlencoding;
014
015import org.apache.juneau.*;
016
017/**
018 * Configurable properties on the {@link UrlEncodingParser} class.
019 * <p>
020 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
021 * returned {@link CoreApi#getContextFactory()}.
022 * <p>
023 * The following convenience methods are also provided for setting context properties:
024 * <ul>
025 *    <li>{@link UrlEncodingParser#setProperty(String,Object)}
026 *    <li>{@link UrlEncodingParser#setProperties(ObjectMap)}
027 *    <li>{@link UrlEncodingParser#addNotBeanClasses(Class[])}
028 *    <li>{@link UrlEncodingParser#addTransforms(Class[])}
029 *    <li>{@link UrlEncodingParser#addImplClass(Class,Class)}
030 * </ul>
031 * <p>
032 * See {@link ContextFactory} for more information about context properties.
033 *
034 * @author James Bognar (james.bognar@salesforce.com)
035 */
036public class UrlEncodingParserContext extends UonParserContext {
037
038   /**
039    * Serialize bean property collections/arrays as separate key/value pairs ({@link Boolean}, default=<jk>false</jk>).
040    * <p>
041    *    If <jk>false</jk>, serializing the array <code>[1,2,3]</code> results in <code>?key=$a(1,2,3)</code>.
042    *    If <jk>true</jk>, serializing the same array results in <code>?key=1&key=2&key=3</code>.
043    * <p>
044    *    Example:
045    * <p class='bcode'>
046    *    <jk>public class</jk> A {
047    *       <jk>public</jk> String[] f1 = {<js>"a"</js>,<js>"b"</js>};
048    *       <jk>public</jk> List&lt;String&gt; f2 = <jk>new</jk> LinkedList&lt;String&gt;(Arrays.<jsm>asList</jsm>(<jk>new</jk> String[]{<js>"c"</js>,<js>"d"</js>}));
049    *    }
050    *
051    *    UrlEncodingSerializer s1 = <jk>new</jk> UrlEncodingParser();
052    *    UrlEncodingSerializer s2 = <jk>new</jk> UrlEncodingParser().setProperty(UrlEncodingContext.<jsf>URLENC_expandedParams</jsf>, <jk>true</jk>);
053    *
054    *    String s1 = p1.serialize(<jk>new</jk> A()); <jc>// Produces "f1=(a,b)&f2=(c,d)"</jc>
055    *    String s2 = p2.serialize(<jk>new</jk> A()); <jc>// Produces "f1=a&f1=b&f2=c&f2=d"</jc>
056    * </p>
057    * <p>
058    *    <b>Important note:</b>  If parsing multi-part parameters, it's highly recommended to use Collections or Lists
059    *    as bean property types instead of arrays since arrays have to be recreated from scratch every time a value
060    *    is added to it.
061    * <p>
062    *    This option only applies to beans.
063    */
064   public static final String URLENC_expandedParams = "UrlEncoding.expandedParams";
065
066
067   final boolean
068      expandedParams;
069
070   /**
071    * Constructor.
072    * <p>
073    * Typically only called from {@link ContextFactory#getContext(Class)}.
074    *
075    * @param cf The factory that created this context.
076    */
077   public UrlEncodingParserContext(ContextFactory cf) {
078      super(cf);
079      this.expandedParams = cf.getProperty(URLENC_expandedParams, boolean.class, false);
080   }
081}
