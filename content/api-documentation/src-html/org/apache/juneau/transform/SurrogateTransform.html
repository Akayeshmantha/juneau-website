024/**
025 * Specialized {@link PojoTransform} for surrogate classes.
026 * <p>
027 * Surrogate classes are used in place of other classes during serialization.
028 * For example, you may want to use a surrogate class to change the names or order of bean
029 * properties on a bean.
030 * <p>
031 * The following is an example of a surrogate class change changes a property name:
032 * <p class='bcode'>
033 *    <jk>public class</jk> SurrogateClass {
034 *       <jk>public</jk> String surrogateField;  <jc>// New bean property</jc>
035 *
036 *       <jk>public</jk> SurrogateClass(NormalClass normalClass) {
037 *          <jk>this</jk>.surrogateField = normalClass.normalField;
038 *       }
039 *    }
040 * </p>
041 * <p>
042 * Optionally, a public static method can be used to untransform a class during parsing:
043 * <p class='bcode'>
044 *    <jk>public class</jk> SurrogateClass {
045 *       ...
046 *       <jk>public static</jk> NormalClass <jsm>toNormalClass</jsm>(SurrogateClass surrogateClass) {
047 *          <jk>return new</jk> NormalClass(surrogateClass.transformedField);
048 *       }
049 *    }
050 * </p>
051 * <p>
052 * Surrogate classes must conform to the following:
053 * <ul class='spaced-list'>
054 *    <li>It must have a one or more public constructors that take in a single parameter whose type is the normal types.
055 *       (It is possible to define a class as a surrogate for multiple class types by using multiple constructors with
056 *       different parameter types).
057 *    <li>It optionally can have a public static method that takes in a single parameter whose type is the transformed type
058 *       and returns an instance of the normal type.  This is called the untransform method.  The method can be called anything.
059 *    <li>If an untransform method is present, the class must also contain a no-arg constructor (so that the transformed class
060 *       can be instantiated by the parser before being converted into the normal class by the untransform method).
061 * </ul>
062 * <p>
063 * Surrogate classes are associated with serializers and parsers using the {@link CoreApi#addTransforms(Class...)} method.
064 * <p class='bcode'>
065 *    <ja>@Test</ja>
066 *    <jk>public void</jk> test() <jk>throws</jk> Exception {
067 *       JsonSerializer s = <jk>new</jk> JsonSerializer.Simple().addTransforms(Surrogate.<jk>class</jk>);
068 *       JsonParser p = <jk>new</jk> JsonParser().addTransforms(Surrogate.<jk>class</jk>);
069 *       String r;
070 *       Normal n = Normal.<jsm>create</jsm>();
071 *
072 *       r = s.serialize(n);
073 *       assertEquals(<js>"{f2:'f1'}"</js>, r);
074 *
075 *       n = p.parse(r, Normal.<jk>class</jk>);
076 *       assertEquals(<js>"f1"</js>, n.f1);
077 *    }
078 *
079 *    <jc>// The normal class</jc>
080 *    <jk>public class</jk> Normal {
081 *       <jk>public</jk> String f1;
082 *
083 *       <jk>public static</jk> Normal <jsm>create</jsm>() {
084 *          Normal n = <jk>new</jk> Normal();
085 *          n.f1 = <js>"f1"</js>;
086 *          <jk>return</jk> n;
087 *       }
088 *    }
089 *
090 *    <jc>// The surrogate class</jc>
091 *    <jk>public static class</jk> Surrogate {
092 *       <jk>public</jk> String f2;
093 *
094 *       <jc>// Surrogate constructor</jc>
095 *       <jk>public</jk> Surrogate(Normal n) {
096 *          f2 = n.f1;
097 *       }
098 *
099 *       <jc>// Constructor used during parsing (only needed if untransform method specified)</jc>
100 *       <jk>public</jk> Surrogate() {}
101 *
102 *       <jc>// Untransform method (optional)</jc>
103 *       <jk>public static</jk> Normal <jsm>toNormal</jsm>(Surrogate f) {
104 *          Normal n = <jk>new</jk> Normal();
105 *          n.f1 = f.f2;
106 *          <jk>return</jk> n;
107 *       }
108 *    }
109 * </p>
110 * <p>
111 * It should be noted that a surrogate class is functionally equivalent to the following {@link PojoTransform} implementation:
112 * <p class='bcode'>
113 *    <jk>public static class</jk> SurrogateTransform <jk>extends</jk> PojoTransform&lt;Normal,Surrogate&gt; {
114 *       <jk>public</jk> Surrogate transform(Normal n) <jk>throws</jk> SerializeException {
115 *          <jk>return new</jk> Surrogate(n);
116 *       }
117 *       <jk>public</jk> Normal normalize(Surrogate s, ClassMeta<?> hint) <jk>throws</jk> ParseException {
118 *          <jk>return</jk> Surrogate.<jsm>toNormal</jsm>(s);
119 *       }
120 *    }
121 * </p>
122 *
123 * @author James Bognar (james.bognar@salesforce.com)
124 * @param <T> The class type that this transform applies to.
125 * @param <F> The transformed class type.
126 */
127public class SurrogateTransform<T,F> extends PojoTransform<T,F> {
128
129   private Constructor<F> constructor;   // public F(T t);
130   private Method untransformMethod;        // public static T valueOf(F f);
131
132   /**
133    * Constructor.
134    *
135    * @param forClass The normal class.
136    * @param constructor The constructor on the surrogate class that takes the normal class as a parameter.
137    * @param untransformMethod The static method that converts surrogate objects into normal objects.
138    */
139   protected SurrogateTransform(Class<T> forClass, Constructor<F> constructor, Method untransformMethod) {
140      super(forClass, constructor.getDeclaringClass());
141      this.constructor = constructor;
142      this.untransformMethod = untransformMethod;
143   }
144
145   /**
146    * Given the specified surrogate class, return the list of POJO transforms.
147    * A transform is returned for each public 1-arg constructor found.
148    * Returns an empty list if no public 1-arg constructors are found.
149    *
150    * @param c The surrogate class.
151    * @return The list of POJO transforms that apply to this class.
152    */
153   @SuppressWarnings({"unchecked", "rawtypes"})
154   public static List<SurrogateTransform<?,?>> findTransforms(Class<?> c) {
155      List<SurrogateTransform<?,?>> l = new LinkedList<SurrogateTransform<?,?>>();
156      for (Constructor<?> cc : c.getConstructors()) {
157         if (cc.getAnnotation(BeanIgnore.class) == null) {
158            Class<?>[] pt = cc.getParameterTypes();
159
160            // Only constructors with one parameter.
161            // Ignore instance class constructors.
162            if (pt.length == 1 && pt[0] != c.getDeclaringClass()) {
163               int mod = cc.getModifiers();
164               if (Modifier.isPublic(mod)) {  // Only public constructors.
165
166                  // Find the untransform method if there is one.
167                  Method untransformMethod = null;
168                  for (Method m : c.getMethods()) {
169                     if (pt[0].equals(m.getReturnType())) {
170                        Class<?>[] mpt = m.getParameterTypes();
171                        if (mpt.length == 1 && mpt[0].equals(c)) { // Only methods with one parameter and where the return type matches this class.
172                           int mod2 = m.getModifiers();
173                           if (Modifier.isPublic(mod2) && Modifier.isStatic(mod2))  // Only public static methods.
174                              untransformMethod = m;
175                        }
176                     }
177                  }
178
179                  l.add(new SurrogateTransform(pt[0], cc, untransformMethod));
180               }
181            }
182         }
183      }
184      return l;
185   }
186
187   @Override /* PojoTransform */
188   public F transform(T o) throws SerializeException {
189      try {
190         return constructor.newInstance(o);
191      } catch (Exception e) {
192         throw new SerializeException(e);
193      }
194   }
195
196   @Override /* PojoTransform */
197   @SuppressWarnings("unchecked")
198   public T normalize(F f, ClassMeta<?> hint) throws ParseException {
199      if (untransformMethod == null)
200         throw new ParseException("static valueOf({0}) method not implement on surrogate class ''{1}''", f.getClass().getName(), getNormalClass().getName());
201      try {
202         return (T)untransformMethod.invoke(null, f);
203      } catch (Exception e) {
204         throw new ParseException(e);
205      }
206   }
207}
