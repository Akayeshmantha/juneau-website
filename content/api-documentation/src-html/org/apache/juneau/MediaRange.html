013package org.apache.juneau;
014
015import java.util.*;
016import java.util.Map.*;
017
018/**
019 * Describes a single type used in content negotiation between an HTTP client and server, as described in
020 * Section 14.1 and 14.7 of RFC2616 (the HTTP/1.1 specification).
021 */
022public final class MediaRange implements Comparable<MediaRange>  {
023
024   private final String type;                      // The media type (e.g. "text" for Accept, "utf-8" for Accept-Charset)
025   private final String subType;                   // The media sub-type (e.g. "json" for Accept, not used for Accept-Charset)
026   private final Float qValue;
027   private final Map<String,Set<String>> parameters, extensions;
028
029   /**
030    * Returns the media type enclosed by this media range.
031    * <p>
032    * Examples:
033    * <ul>
034    *    <li><js>"text/html"</js>
035    *    <li><js>"text/*"</js>
036    *    <li><js>"*\/*"</js>
037    * </ul>
038    *
039    * @return The media type of this media range, lowercased, never <jk>null</jk>.
040    */
041   public String getMediaType() {
042      return type + "/" + subType;
043   }
044
045   /**
046    * Return just the type portion of this media range.
047    *
048    * @return The type portion of this media range.
049    */
050   public String getType() {
051      return type;
052   }
053
054   /**
055    * Returns the <js>'q'</js> (quality) value for this type, as described in Section 3.9 of RFC2616.
056    * <p>
057    * The quality value is a float between <code>0.0</code> (unacceptable) and <code>1.0</code> (most acceptable).
058    * <p>
059    * If 'q' value doesn't make sense for the context (e.g. this range was extracted from a <js>"content-*"</js> header, as opposed to <js>"accept-*"</js>
060    * header, its value will always be <js>"1"</js>.
061    *
062    * @return The 'q' value for this type, never <jk>null</jk>.
063    */
064   public Float getQValue() {
065      return qValue;
066   }
067
068   /**
069    * Returns the optional set of parameters associated to the type as returned by {@link #getMediaType()}.
070    * <p>
071    * The parameters are those values as described in standardized MIME syntax.
072    * An example of such a parameter in string form might be <js>"level=1"</js>.
073    * <p>
074    * Values are lowercase and never <jk>null</jk>.
075    *
076    * @return The optional list of parameters, never <jk>null</jk>.
077    */
078   public Map<String,Set<String>> getParameters() {
079      return parameters;
080   }
081
082   /**
083    * Returns the optional set of custom extensions defined for this type.
084    * <p>
085    * Values are lowercase and never <jk>null</jk>.
086    *
087    * @return The optional list of extensions, never <jk>null</jk>.
088    */
089   public Map<String,Set<String>> getExtensions() {
090      return extensions;
091   }
092
093   /**
094    * Provides a string representation of this media range, suitable for use as an <code>Accept</code> header value.
095    * <p>
096    * The literal text generated will be all lowercase.
097    *
098    * @return A media range suitable for use as an Accept header value, never <code>null</code>.
099    */
100   @Override /* Object */
101   public String toString() {
102      StringBuffer sb = new StringBuffer().append(type).append('/').append(subType);
103
104      if (! parameters.isEmpty())
105         for (Entry<String,Set<String>> e : parameters.entrySet()) {
106            String k = e.getKey();
107            for (String v : e.getValue())
108               sb.append(';').append(k).append('=').append(v);
109         }
110
111      // '1' is equivalent to specifying no qValue. If there's no extensions, then we won't include a qValue.
112      if (qValue.floatValue() == 1.0) {
113         if (! extensions.isEmpty()) {
114            sb.append(";q=").append(qValue);
115            for (Entry<String,Set<String>> e : extensions.entrySet()) {
116               String k = e.getKey();
117               for (String v : e.getValue())
118                  sb.append(';').append(k).append('=').append(v);
119            }
120         }
121      } else {
122         sb.append(";q=").append(qValue);
123         for (Entry<String,Set<String>> e : extensions.entrySet()) {
124            String k = e.getKey();
125            for (String v : e.getValue())
126               sb.append(';').append(k).append('=').append(v);
127         }
128      }
129      return sb.toString();
130   }
131
132   /**
133    * Returns <jk>true</jk> if the specified object is also a <code>MediaType</code>, and has the same qValue, type, parameters, and extensions.
134    *
135    * @return <jk>true</jk> if object is equivalent.
136    */
137   @Override /* Object */
138   public boolean equals(Object o) {
139
140      if (o == null || !(o instanceof MediaRange))
141         return false;
142
143      if (this == o)
144         return true;
145
146      MediaRange o2 = (MediaRange) o;
147      return qValue.equals(o2.qValue)
148         && type.equals(o2.type)
149         && subType.equals(o2.subType)
150         && parameters.equals(o2.parameters)
151         && extensions.equals(o2.extensions);
152   }
153
154   /**
155    * Returns a hash based on this instance's <code>media-type</code>.
156    *
157    * @return A hash based on this instance's <code>media-type</code>.
158    */
159   @Override /* Object */
160   public int hashCode() {
161      return type.hashCode() + subType.hashCode();
162   }
163
164   /**
165    * Creates a <code>MediaRange</code> object with the referenced values.
166    *
167    * @param type The MIME type of this media range (e.g. <js>"application"</js> in <js>"application/json"</js>)
168    * @param subType The MIME subtype of this media range (e.g. <js>"json"</js> in <js>"application/json"</js>).
169    * @param parameters The optional parameters for this range.
170    * @param qValue The quality value of this range.  Must be between <code>0</code> and <code>1.0</code>.
171    * @param extensions The optional extensions to this quality value.
172    */
173   private MediaRange(String type, String subType, Map<String,Set<String>> parameters, Float qValue, Map<String,Set<String>> extensions) {
174      this.type = type;
175      this.subType = subType;
176      this.parameters = (parameters == null ? new TreeMap<String,Set<String>>() : parameters);
177      this.extensions = (extensions == null ? new TreeMap<String,Set<String>>() : extensions);
178      this.qValue = qValue;
179   }
180
181   /**
182    * Parses an <code>Accept</code> header value into an array of media ranges.
183    * <p>
184    * The returned media ranges are sorted such that the most acceptable media is available at ordinal position <js>'0'</js>, and the least acceptable at position n-1.
185    * <p>
186    * The syntax expected to be found in the referenced <code>value</code> complies with the syntax described in RFC2616, Section 14.1, as described below:
187    * <p class='bcode'>
188    *    Accept         = "Accept" ":"
189    *                      #( media-range [ accept-params ] )
190    *
191    *    media-range    = ( "*\/*"
192    *                      | ( type "/" "*" )
193    *                      | ( type "/" subtype )
194    *                      ) *( ";" parameter )
195    *    accept-params  = ";" "q" "=" qvalue *( accept-extension )
196    *    accept-extension = ";" token [ "=" ( token | quoted-string ) ]
197    * </p>
198    * This method can also be used on other headers such as <code>Accept-Charset</code> and <code>Accept-Encoding</code>...
199    * <p class='bcode'>
200    *    Accept-Charset = "Accept-Charset" ":"
201    *    1#( ( charset | "*" )[ ";" "q" "=" qvalue ] )
202    * </p>
203    *
204    * @param value The value to parse.  If <jk>null</jk> or empty, returns a single <code>MediaRange</code> is returned that represents all types.
205    * @return The media ranges described by the string.
206    *    The ranges are sorted such that the most acceptable media is available at ordinal position <js>'0'</js>, and the least acceptable at position n-1.
207    */
208   public static MediaRange[] parse(String value) {
209
210      Set<MediaRange> ranges = new TreeSet<MediaRange>();
211
212      if (value == null || value.length() == 0)
213         return new MediaRange[]{new MediaRange("*", "*", null, 1f, null)};
214
215      value = value.toLowerCase(Locale.ENGLISH);
216
217      for (String r : value.trim().split("\\s*,\\s*")) {
218         r = r.trim();
219
220         if (r.isEmpty())
221            continue;
222
223         String[] tokens = r.split("\\s*;\\s*");
224
225         tokens[0] = tokens[0].replace(' ', '+');
226
227         // There is at least a type.
228         String[] t = tokens[0].split("/");
229         String type = t[0], subType = (t.length == 1 ? "*" : t[1]);
230
231         // Only the type of the range is specified
232         if (tokens.length == 1) {
233            ranges.add(new MediaRange(type, subType, null, 1f, null));
234            continue;
235         }
236
237         Float qValue = 1f;
238         Map<String,Set<String>> params = new TreeMap<String,Set<String>>();
239         Map<String,Set<String>> exts = new TreeMap<String,Set<String>>();
240
241         boolean isInExtensions = false;
242         for (int i = 1; i < tokens.length; i++) {
243            String[] parm = tokens[i].split("\\s*=\\s*");
244            if (parm.length == 2) {
245               String k = parm[0], v = parm[1];
246               if (isInExtensions) {
247                  if (! exts.containsKey(parm[0]))
248                     exts.put(parm[0], new TreeSet<String>());
249                  exts.get(parm[0]).add(parm[1]);
250               } else if (k.equals("q")) {
251                  qValue = new Float(v);
252                  isInExtensions = true;
253               } else /*(! isInExtensions)*/ {
254                  if (! params.containsKey(parm[0]))
255                     params.put(parm[0], new TreeSet<String>());
256                  params.get(parm[0]).add(parm[1]);
257               }
258            }
259         }
260
261         ranges.add(new MediaRange(type, subType, params, qValue, exts));
262      }
263
264      return ranges.toArray(new MediaRange[ranges.size()]);
265   }
266
267   /**
268    * Compares two MediaRanges for equality.
269    * <p>
270    * The values are first compared according to <code>qValue</code> values.
271    * Should those values be equal, the <code>type</code> is then lexicographically compared (case-insensitive) in ascending order,
272    *    with the <js>"*"</js> type demoted last in that order.
273    * <code>MediaRanges</code> with the same type but different sub-types are compared - a more specific subtype is
274    *    promoted over the 'wildcard' subtype.
275    * <code>MediaRanges</code> with the same types but with extensions are promoted over those same types with no extensions.
276    *
277    * @param o The range to compare to.  Never <jk>null</jk>.
278    */
279   @Override /* Comparable */
280   public int compareTo(MediaRange o) {
281
282      // Compare q-values.
283      int qCompare = Float.compare(o.qValue, qValue);
284      if (qCompare != 0)
285         return qCompare;
286
287      // Compare media-types.
288      // Note that '*' comes alphabetically before letters, so just do a reverse-alphabetical comparison.
289      int i = o.type.compareTo(type);
290      if (i == 0)
291         i = o.subType.compareTo(subType);
292      return i;
293   }
294
295   /**
296    * Returns <jk>true</jk> if the specified <code>MediaRange</code> matches this range.
297    * <p>
298    * This implies the types and subtypes are the same as or encompasses the other (e.g. <js>'application/xml'</js> and <js>'application/*'</js>).
299    *
300    * @param o The other media rage.
301    * @return <jk>true</jk> if the media ranges are the same or one encompasses the other.
302    */
303   public boolean matches(MediaRange o) {
304      if (this == o)
305         return true;
306
307      if (qValue == 0 || o.qValue == 0)
308         return false;
309
310      if (type.equals(o.type) || (type.equals("*")) || (o.type.equals("*")))
311         if (subType.equals(o.subType) || subType.equals("*") || o.subType.equals("*"))
312            return true;
313
314      return false;
315   }
