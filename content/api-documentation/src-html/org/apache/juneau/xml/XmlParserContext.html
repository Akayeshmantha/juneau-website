013package org.apache.juneau.xml;
014
015import javax.xml.stream.*;
016import javax.xml.stream.util.*;
017
018import org.apache.juneau.*;
019import org.apache.juneau.parser.*;
020
021/**
022 * Configurable properties on the {@link XmlParser} class.
023 * <p>
024 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
025 * returned {@link CoreApi#getContextFactory()}.
026 * <p>
027 * The following convenience methods are also provided for setting context properties:
028 * <ul>
029 *    <li>{@link XmlParser#setProperty(String,Object)}
030 *    <li>{@link XmlParser#setProperties(ObjectMap)}
031 *    <li>{@link XmlParser#addNotBeanClasses(Class[])}
032 *    <li>{@link XmlParser#addTransforms(Class[])}
033 *    <li>{@link XmlParser#addImplClass(Class,Class)}
034 * </ul>
035 * <p>
036 * See {@link ContextFactory} for more information about context properties.
037 *
038 * @author James Bognar (james.bognar@salesforce.com)
039 */
040public final class XmlParserContext extends ParserContext {
041
042   /**
043    * XMLSchema-instance namespace URI ({@link String}, default=<js>"http://www.w3.org/2001/XMLSchema-instance"</js>).
044    * <p>
045    * The XMLSchema namespace.
046    */
047   public static final String XML_xsiNs = "XmlParser.xsiNs";
048
049   /**
050    * Trim whitespace from text elements ({@link Boolean}, default=<jk>true</jk>).
051    * <p>
052    * If <jk>true</jk>, whitespace in text elements will be automatically trimmed.
053    */
054   public static final String XML_trimWhitespace = "XmlParser.trimWhitespace";
055
056   /**
057    * Set validating mode ({@link Boolean}, default=<jk>false</jk>).
058    * <p>
059    * If <jk>true</jk>, XML document will be validated.
060    * See {@link XMLInputFactory#IS_VALIDATING} for more info.
061    */
062   public static final String XML_validating = "XmlParser.validating";
063
064   /**
065    * Set coalescing mode ({@link Boolean}, default=<jk>false</jk>).
066    * <p>
067    * If <jk>true</jk>, XML text elements will be coalesced.
068    * See {@link XMLInputFactory#IS_COALESCING} for more info.
069    */
070   public static final String XML_coalescing = "XmlParser.coalescing";
071
072   /**
073    * Replace entity references ({@link Boolean}, default=<jk>true</jk>).
074    * <p>
075    * If <jk>true</jk>, entity references will be replace during parsing.
076    * See {@link XMLInputFactory#IS_REPLACING_ENTITY_REFERENCES} for more info.
077    */
078   public static final String XML_replaceEntityReferences = "XmlParser.replaceEntityReferences";
079
080   /**
081    * XML reporter ({@link XMLReporter}, default=<jk>null</jk>).
082    * <p>
083    * Associates an {@link XMLReporter} with this parser.
084    * <p>
085    * Note:  Reporters are not copied to new parsers during a clone.
086    */
087   public static final String XML_reporter = "XmlParser.reporter";
088
089   /**
090    * XML resolver ({@link XMLResolver}, default=<jk>null</jk>).
091    * <p>
092    * Associates an {@link XMLResolver} with this parser.
093    */
094   public static final String XML_resolver = "XmlParser.resolver";
095
096   /**
097    * XML event allocator. ({@link XMLEventAllocator}, default=<jk>false</jk>).
098    * <p>
099    * Associates an {@link XMLEventAllocator} with this parser.
100    */
101   public static final String XML_eventAllocator = "XmlParser.eventAllocator";
102
103   /**
104    * Preserve root element during generalized parsing ({@link Boolean}, default=<jk>false</jk>).
105    * <p>
106    * If <jk>true</jk>, when parsing into a generic {@link ObjectMap}, the map will
107    *    contain a single entry whose key is the root element name.
108    *
109    * Example:
110    * <table class='styled'>
111    *    <tr>
112    *       <td>XML</td>
113    *       <td>ObjectMap.toString(), preserveRootElement==false</td>
114    *       <td>ObjectMap.toString(), preserveRootElement==true</td>
115    *    </tr>
116    *    <tr>
117    *       <td><code><xt>&lt;root&gt;&lt;a&gt;</xt>foobar<xt>&lt;/a&gt;&lt;/root&gt;</xt><code></td>
118    *       <td><code>{ a:<js>'foobar'</js> }</code></td>
119    *       <td><code>{ root: { a:<js>'foobar'</js> }}</code></td>
120    *    </tr>
121    * </table>
122    *
123    */
124   public static final String XML_preserveRootElement = "XmlParser.preserveRootElement";
125
126   final String xsiNs;
127   final boolean
128      trimWhitespace,
129      validating,
130      coalescing,
131      replaceEntityReferences,
132      preserveRootElement;
133   final XMLReporter reporter;
134   final XMLResolver resolver;
135   final XMLEventAllocator eventAllocator;
136
137   /**
138    * Constructor.
139    * <p>
140    * Typically only called from {@link ContextFactory#getContext(Class)}.
141    *
142    * @param cf The factory that created this context.
143    */
144   public XmlParserContext(ContextFactory cf) {
145      super(cf);
146      xsiNs = cf.getProperty(XML_xsiNs, String.class, "http://www.w3.org/2001/XMLSchema-instance");
147      trimWhitespace = cf.getProperty(XML_trimWhitespace, boolean.class, true);
148      validating = cf.getProperty(XML_validating, boolean.class, false);
149      coalescing = cf.getProperty(XML_coalescing, boolean.class, false);
150      replaceEntityReferences = cf.getProperty(XML_replaceEntityReferences, boolean.class, true);
151      preserveRootElement = cf.getProperty(XML_preserveRootElement, boolean.class, false);
152      reporter = cf.getProperty(XML_reporter, XMLReporter.class, null);
153      resolver = cf.getProperty(XML_resolver, XMLResolver.class, null);
154      eventAllocator = cf.getProperty(XML_eventAllocator, XMLEventAllocator.class, null);
155   }
156}
