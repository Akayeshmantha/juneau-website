013package org.apache.juneau.xml;
014
015import java.util.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.xml.annotation.*;
019
020/**
021 * Metadata on beans specific to the XML serializers and parsers pulled from the {@link Xml @Xml} annotation on the class.
022 *
023 * @author James Bognar (james.bognar@salesforce.com)
024 * @param <T> The bean class type.
025 */
026public class XmlBeanMeta<T> {
027
028   // XML related fields
029   private final Map<String,BeanPropertyMeta<T>> xmlAttrs;                        // Map of bean properties that are represented as XML attributes.
030   private final BeanPropertyMeta<T> xmlContent;                                  // Bean property that is represented as XML content within the bean element.
031   private final XmlContentHandler<T> xmlContentHandler;                          // Class used to convert bean to XML content.
032   private final Map<String,BeanPropertyMeta<T>> childElementProperties;          // Properties defined with @Xml.childName annotation.
033   private final BeanMeta<T> beanMeta;
034
035   /**
036    * Constructor.
037    *
038    * @param beanMeta The metadata on the bean that this metadata applies to.
039    * @param pNames Only look at these property names.  If <jk>null</jk>, apply to all bean properties.
040    */
041   public XmlBeanMeta(BeanMeta<T> beanMeta, String[] pNames) {
042      this.beanMeta = beanMeta;
043      Class<T> c = beanMeta.getClassMeta().getInnerClass();
044
045      Map<String,BeanPropertyMeta<T>> tXmlAttrs = new LinkedHashMap<String,BeanPropertyMeta<T>>();
046      BeanPropertyMeta<T> tXmlContent = null;
047      XmlContentHandler<T> tXmlContentHandler = null;
048      Map<String,BeanPropertyMeta<T>> tChildElementProperties = new LinkedHashMap<String,BeanPropertyMeta<T>>();
049
050      for (BeanPropertyMeta<T> p : beanMeta.getPropertyMetas(pNames)) {
051         XmlFormat xf = p.getXmlMeta().getXmlFormat();
052         if (xf == XmlFormat.ATTR)
053            tXmlAttrs.put(p.getName(), p);
054         else if (xf == XmlFormat.CONTENT) {
055            if (tXmlContent != null)
056               throw new BeanRuntimeException(c, "Multiple instances of CONTENT properties defined on class.  Only one property can be designated as such.");
057            tXmlContent = p;
058            tXmlContentHandler = p.getXmlMeta().getXmlContentHandler();
059         }
060         // Look for any properties that are collections with @Xml.childName specified.
061         String n = p.getXmlMeta().getChildName();
062         if (n != null) {
063            if (tChildElementProperties.containsKey(n))
064               throw new BeanRuntimeException(c, "Multiple properties found with the name ''{0}''.", n);
065            tChildElementProperties.put(n, p);
066         }
067      }
068
069      xmlAttrs = Collections.unmodifiableMap(tXmlAttrs);
070      xmlContent = tXmlContent;
071      xmlContentHandler = tXmlContentHandler;
072      childElementProperties = (tChildElementProperties.isEmpty() ? null : Collections.unmodifiableMap(tChildElementProperties));
073   }
074
075   /**
076    * Returns the list of properties annotated with an {@link Xml#format()} of {@link XmlFormat#ATTR}.
077    * In other words, the list of properties that should be rendered as XML attributes instead of child elements.
078    *
079    * @return Metadata on the XML attribute properties of the bean.
080    */
081   protected Map<String,BeanPropertyMeta<T>> getXmlAttrProperties() {
082      return xmlAttrs;
083   }
084
085   /**
086    * Returns the bean property annotated with an {@link Xml#format()} value of {@link XmlFormat#CONTENT}
087    *
088    * @return The bean property, or <jk>null</jk> if annotation is not specified.
089    */
090   protected BeanPropertyMeta<T> getXmlContentProperty() {
091      return xmlContent;
092   }
093
094   /**
095    * Return the XML content handler for this bean.
096    *
097    * @return The XML content handler for this bean, or <jk>null</jk> if no content handler is defined.
098    */
099   protected XmlContentHandler<T> getXmlContentHandler() {
100      return xmlContentHandler;
101   }
102
103   /**
104    * Returns the child element properties for this bean.
105    * See {@link Xml#childName()}
106    *
107    * @return The child element properties for this bean, or <jk>null</jk> if no child element properties are defined.
108    */
109   protected Map<String,BeanPropertyMeta<T>> getChildElementProperties() {
110      return childElementProperties;
111   }
112
113   /**
114    * Returns bean property meta with the specified name.
115    * This is identical to calling {@link BeanMeta#getPropertyMeta(String)} except it first retrieves
116    *    the bean property meta based on the child name (e.g. a property whose name is "people", but whose child name is "person").
117    *
118    * @param fieldName The bean property name.
119    * @return The property metadata.
120    */
121   protected BeanPropertyMeta<T> getPropertyMeta(String fieldName) {
122      if (childElementProperties != null) {
123         BeanPropertyMeta<T> bpm = childElementProperties.get(fieldName);
124         if (bpm != null)
125            return bpm;
126      }
127      return beanMeta.getPropertyMeta(fieldName);
128   }
129}
