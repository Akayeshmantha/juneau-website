025/**
026 * Transforms {@link Calendar Calendars} to {@link String Strings}.
027 *
028 *
029 * <h6 class='topic'>Behavior-specific subclasses</h6>
030 * <p>
031 *    The following direct subclasses are provided for convenience:
032 * <ul class='spaced-list'>
033 *    <li>{@link ToString} - Transforms to {@link String Strings} using the {@code Date.toString()} method.
034 *    <li>{@link ISO8601DT} - Transforms to ISO8601 date-time strings.
035 *    <li>{@link ISO8601DTZ} - Same as {@link ISO8601DT}, except always serializes in GMT.
036 *    <li>{@link RFC2822DT} - Transforms to RFC2822 date-time strings.
037 *    <li>{@link RFC2822DTZ} - Same as {@link RFC2822DT}, except always serializes in GMT.
038 *    <li>{@link RFC2822D} - Transforms to RFC2822 date strings.
039 *    <li>{@link Simple} - Transforms to simple <js>"yyyy/MM/dd HH:mm:ss"</js> strings.
040 *    <li>{@link Medium} - Transforms to {@link DateFormat#MEDIUM} strings.
041 * </ul>
042 *
043 *
044 * @author James Bognar (james.bognar@salesforce.com)
045 */
046public class CalendarTransform extends PojoTransform<Calendar,String> {
047
048   private static final TimeZone GMT = TimeZone.getTimeZone("GMT");
049
050   /**
051    * Transforms {@link Calendar Calendars} to {@link String Strings} using the {@code Date.toString()} method.
052    *
053    * <dl>
054    *    <dt>Example output:</dt>
055    *    <dd>
056    *       <ul>
057    *          <li><js>"Wed Jul 04 15:30:45 EST 2001"</js>
058    *       </ul>
059    *    </dd>
060    * </dl>
061    */
062   public static class ToString extends CalendarTransform {
063      /** Constructor */
064      public ToString() {
065         super("EEE MMM dd HH:mm:ss zzz yyyy");
066      }
067   }
068
069   /**
070    * Transforms {@link Calendar Calendars} to ISO8601 date-time strings.
071    *
072    * <dl>
073    *    <dt>Example output:</dt>
074    *    <dd>
075    *       <ul>
076    *          <li><js>"2001-07-04T15:30:45-05:00"</js>
077    *          <li><js>"2001-07-04T15:30:45Z"</js>
078    *       </ul>
079    *    </dd>
080    *    <dt>Example input:</dt>
081    *    <dd>
082    *       <ul>
083    *          <li><js>"2001-07-04T15:30:45-05:00"</js>
084    *          <li><js>"2001-07-04T15:30:45Z"</js>
085    *          <li><js>"2001-07-04T15:30:45.1Z"</js>
086    *          <li><js>"2001-07-04T15:30Z"</js>
087    *          <li><js>"2001-07-04"</js>
088    *          <li><js>"2001-07"</js>
089    *          <li><js>"2001"</js>
090    *        </ul>
091    *    </dd>
092    * </dl>
093    */
094   public static class ISO8601DT extends CalendarTransform {
095
096      /** Constructor */
097      public ISO8601DT() {}
098
099      @Override /* PojoTransform */
100      public Calendar normalize(String o, ClassMeta<?> hint) throws ParseException {
101         try {
102            if (StringUtils.isEmpty(o))
103               return null;
104            return convert(DatatypeConverter.parseDateTime(o), hint);
105         } catch (Exception e) {
106            throw new ParseException(e);
107         }
108      }
109
110      @Override /* PojoTransform */
111      public String transform(Calendar o) {
112         return DatatypeConverter.printDateTime(o);
113      }
114   }
115
116   /**
117    * Same as {@link ISO8601DT}, except always serializes in GMT.
118    * <p>
119    * Example output: <js>"2001-07-04T15:30:45Z"</js>
120    */
121   public static class ISO8601DTZ extends CalendarTransform {
122
123      /** Constructor */
124      public ISO8601DTZ() {}
125
126      @Override /* PojoTransform */
127      public Calendar normalize(String o, ClassMeta<?> hint) throws ParseException {
128         try {
129            if (StringUtils.isEmpty(o))
130               return null;
131            return convert(DatatypeConverter.parseDateTime(o), hint);
132         } catch (Exception e) {
133            throw new ParseException(e);
134         }
135      }
136
137      @Override /* PojoTransform */
138      public String transform(Calendar o) {
139         if (o.getTimeZone().getRawOffset() != 0) {
140            Calendar c = Calendar.getInstance(GMT);
141            c.setTime(o.getTime());
142            o = c;
143         }
144         return DatatypeConverter.printDateTime(o);
145      }
146   }
147
148   /**
149    * Transforms {@link Calendar Calendars} to RFC2822 date-time strings.
150    */
151   public static class RFC2822DT extends CalendarTransform {
152      /** Constructor */
153      public RFC2822DT() {
154         super("EEE, dd MMM yyyy HH:mm:ss Z");
155      }
156   }
157
158   /**
159    * Same as {@link RFC2822DT}, except always serializes in GMT.
160    * <p>
161    * Example output: <js>"Wed, 31 Jan 2001 12:34:56 +0000"</js>
162    */
163   public static class RFC2822DTZ extends CalendarTransform {
164      /** Constructor */
165      public RFC2822DTZ() {
166         super("EEE, dd MMM yyyy HH:mm:ss 'GMT'", GMT);
167      }
168   }
169
170   /**
171    * Transforms {@link Calendar Calendars} to RFC2822 date strings.
172    */
173   public static class RFC2822D extends CalendarTransform {
174      /** Constructor */
175      public RFC2822D() {
176         super("dd MMM yyyy");
177      }
178   }
179
180   /**
181    * Transforms {@link Calendar Calendars} to simple <js>"yyyy/MM/dd HH:mm:ss"</js> strings.
182    */
183   public static class Simple extends CalendarTransform {
184      /** Constructor */
185      public Simple() {
186         super("yyyy/MM/dd HH:mm:ss");
187      }
188   }
189
190   /**
191    * Transforms {@link Calendar Calendars} to {@link DateFormat#MEDIUM} strings.
192    */
193   public static class Medium extends CalendarTransform {
194      /** Constructor */
195      public Medium() {
196         super(DateFormat.getDateInstance(DateFormat.MEDIUM));
197      }
198   }
199
200   /** The formatter to convert dates to Strings. */
201   private DateFormat format;
202
203   private TimeZone timeZone;
204
205   /**
206    * Default constructor.
207    * <p>
208    *    This constructor is used when <code>transform()</code> and <code>normalize()</code> are overridden by subclasses.
209    */
210   public CalendarTransform() {}
211
212   /**
213    * Construct a transform using the specified date format string that will be
214    *    used to construct a {@link SimpleDateFormat} that will be used to convert
215    *    dates to strings.
216    *
217    * @param simpleDateFormat The {@link SimpleDateFormat} pattern.
218    */
219   public CalendarTransform(String simpleDateFormat) {
220      this(new SimpleDateFormat(simpleDateFormat));
221   }
222
223   /**
224    * Construct a transform using the specified date format string that will be
225    *    used to construct a {@link SimpleDateFormat} that will be used to convert
226    *    dates to strings.
227    *
228    * @param simpleDateFormat The {@link SimpleDateFormat} pattern.
229    * @param timeZone The time zone to associate with the date pattern.
230    */
231   public CalendarTransform(String simpleDateFormat, TimeZone timeZone) {
232      this(new SimpleDateFormat(simpleDateFormat));
233      format.setTimeZone(timeZone);
234      this.timeZone = timeZone;
235   }
236
237   /**
238    * Construct a transform using the specified {@link DateFormat} that will be used to convert
239    *    dates to strings.
240    *
241    * @param format The format to use to convert dates to strings.
242    */
243   public CalendarTransform(DateFormat format) {
244      super();
245      this.format = format;
246   }
247
248   /**
249    * Converts the specified {@link Calendar} to a {@link String}.
250    */
251   @Override /* PojoTransform */
252   public String transform(Calendar o) {
253      DateFormat df = format;
254      TimeZone tz1 = o.getTimeZone();
255      TimeZone tz2 = format.getTimeZone();
256      if (timeZone == null && ! tz1.equals(tz2)) {
257         df = (DateFormat)format.clone();
258         df.setTimeZone(tz1);
259      }
260      return df.format(o.getTime());
261   }
262
263   /**
264    * Converts the specified {@link String} to a {@link Calendar}.
265    */
266   @Override /* PojoTransform */
267   public Calendar normalize(String o, ClassMeta<?> hint) throws ParseException {
268      try {
269         if (StringUtils.isEmpty(o))
270            return null;
271         return convert(format.parse(o), hint);
272      } catch (Exception e) {
273         throw new ParseException(e);
274      }
275   }
276
277   private static Calendar convert(Calendar in, ClassMeta<?> hint) throws Exception {
278      if (hint.isInstance(in) || ! hint.canCreateNewInstance())
279         return in;
280      Calendar c = (Calendar)hint.newInstance();
281      c.setTime(in.getTime());
282      c.setTimeZone(in.getTimeZone());
283      return c;
284   }
285
286   private static Calendar convert(Date in, ClassMeta<?> hint) throws Exception {
287      if (hint == null || ! hint.canCreateNewInstance())
288         hint = BeanContext.DEFAULT.getClassMeta(GregorianCalendar.class);
289      Calendar c = (Calendar)hint.newInstance();
290      c.setTime(in);
291      return c;
292   }
