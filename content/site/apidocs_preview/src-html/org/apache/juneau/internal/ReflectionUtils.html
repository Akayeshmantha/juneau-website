017import java.io.*;
018import java.lang.annotation.*;

022/**
023 * Reflection utilities.
024 */
025public final class ReflectionUtils {
026
027   /**
028    * Returns <jk>true</jk> if the {@link #getAnnotation(Class, Method, int)} returns a value.
029    *
030    * @param a The annotation to check for.
031    * @param m The method containing the parameter to check.
032    * @param index The parameter index.
033    * @return <jk>true</jk> if the {@link #getAnnotation(Class, Method, int)} returns a value.
034    */
035   public static boolean hasAnnotation(Class<? extends Annotation> a, Method m, int index) {
036      return getAnnotation(a, m, index) != null;
037   }
038
039   /**
040    * Returns <jk>true</jk> if the {@link #getAnnotation(Class, Method)} returns a value.
041    *
042    * @param a The annotation to check for.
043    * @param m The method to check.
044    * @return <jk>true</jk> if the {@link #getAnnotation(Class, Method)} returns a value.
045    */
046   public static boolean hasAnnotation(Class<? extends Annotation> a, Method m) {
047      return getAnnotation(a, m) != null;
048   }
049
050   /**
051    * Returns <jk>true</jk> if the {@link #getAnnotation(Class, Class)} returns a value.
052    *
053    * @param a The annotation to check for.
054    * @param c The class to check.
055    * @return <jk>true</jk> if the {@link #getAnnotation(Class, Class)} returns a value.
056    */
057   public static boolean hasAnnotation(Class<? extends Annotation> a, Class<?> c) {
058      return getAnnotation(a, c) != null;
059   }
060
061   /**
062    * Returns the specified annotation if it exists on the specified parameter or parameter type class.
063    *
064    * @param a The annotation to check for.
065    * @param m The method containing the parameter to check.
066    * @param index The parameter index.
067    * @return <jk>true</jk> if the {@link #getAnnotation(Class, Method, int)} returns a value.
068    */
069   @SuppressWarnings("unchecked")
070   public static <T extends Annotation> T getAnnotation(Class<T> a, Method m, int index) {
071      for (Annotation a2 :  m.getParameterAnnotations()[index])
072         if (a.isInstance(a2))
073            return (T)a2;
074      Type t = m.getGenericParameterTypes()[index];
075      if (t instanceof Class)
076         return getAnnotation(a, (Class<?>)t);
077      return null;
078   }
079
080   /**
081    * Returns the specified annotation if it exists on the specified method or return type class.
082    *
083    * @param a The annotation to check for.
084    * @param m The method to check.
085    * @return <jk>true</jk> if the {@link #getAnnotation(Class, Method, int)} returns a value.
086    */
087   @SuppressWarnings("unchecked")
088   public static <T extends Annotation> T getAnnotation(Class<T> a, Method m) {
089      for (Annotation a2 :  m.getAnnotations())
090         if (a.isInstance(a2))
091            return (T)a2;
092      Type t = m.getGenericReturnType();
093      if (t instanceof Class)
094         return getAnnotation(a, (Class<?>)t);
095      return null;
096   }
097
098   /**
099    * Similar to {@link Class#getAnnotation(Class)} except also searches annotations on interfaces.
100    *
101    * @param <T> The annotation class type.
102    * @param a The annotation class.
103    * @param c The annotated class.
104    * @return The annotation, or <jk>null</jk> if not found.
105    */
106   public static <T extends Annotation> T getAnnotation(Class<T> a, Class<?> c) {
107      if (c == null)
108         return null;
109
110      T t = getDeclaredAnnotation(a, c);
111      if (t != null)
112         return t;
113
114      t = getAnnotation(a, c.getSuperclass());
115      if (t != null)
116         return t;
117
118      for (Class<?> c2 : c.getInterfaces()) {
119         t = getAnnotation(a, c2);
120         if (t != null)
121            return t;
122      }
123      return null;
124   }
125
126   /**
127    * Returns the specified annotation only if it's been declared on the specified class.
128    *
129    * <p>
130    * More efficient than calling {@link Class#getAnnotation(Class)} since it doesn't recursively look for the class
131    * up the parent chain.
132    *
133    * @param <T> The annotation class type.
134    * @param a The annotation class.
135    * @param c The annotated class.
136    * @return The annotation, or <jk>null</jk> if not found.
137    */
138   @SuppressWarnings("unchecked")
139   public static <T extends Annotation> T getDeclaredAnnotation(Class<T> a, Class<?> c) {
140      for (Annotation a2 : c.getDeclaredAnnotations())
141         if (a2.annotationType() == a)
142            return (T)a2;
143      return null;
144   }
145
146   /**
147    * Returns all instances of the specified annotation on the specified class.
148    *
149    * <p>
150    * Searches all superclasses and superinterfaces.
151    * Results are ordered child-to-parent.
152    *
153    * @param <T> The annotation class type.
154    * @param a The annotation class type.
155    * @param c The class being searched.
156    * @return The found matches, or an empty array if annotation was not found.
157    */
158   public static <T extends Annotation> List<T> findAnnotations(Class<T> a, Class<?> c) {
159      List<T> l = new LinkedList<>();
160      appendAnnotations(a, c, l);
161      return l;
162   }
163
164   /**
165    * Same as {@link #findAnnotations(Class, Class)} but returns the list in parent-to-child order.
166    *
167    * @param a The annotation class type.
168    * @param c The class being searched.
169    * @return The found matches, or an empty array if annotation was not found.
170    */
171   public static <T extends Annotation> List<T> findAnnotationsParentFirst(Class<T> a, Class<?> c) {
172      List<T> l = findAnnotations(a, c);
173      Collections.reverse(l);
174      return l;
175   }
176
177   /**
178    * Same as {@link #findAnnotations(Class, Class)} except returns the annotations as a map with the keys being the
179    * class on which the annotation was found.
180    *
181    * <p>
182    * Results are ordered child-to-parent.
183    *
184    * @param <T> The annotation class type.
185    * @param a The annotation class type.
186    * @param c The class being searched.
187    * @return The found matches, or an empty map if annotation was not found.
188    */
189   public static <T extends Annotation> LinkedHashMap<Class<?>,T> findAnnotationsMap(Class<T> a, Class<?> c) {
190      LinkedHashMap<Class<?>,T> m = new LinkedHashMap<>();
191      findAnnotationsMap(a, c, m);
192      return m;
193   }
194
195   /**
196    * Same as {@link #findAnnotationsMap(Class, Class)} except returns results in parent-to-child order.
197    *
198    * @param <T> The annotation class type.
199    * @param a The annotation class type.
200    * @param c The class being searched.
201    * @return The found matches, or an empty map if annotation was not found.
202    */
203   public static <T extends Annotation> LinkedHashMap<Class<?>,T> findAnnotationsMapParentFirst(Class<T> a, Class<?> c) {
204      return CollectionUtils.reverse(findAnnotationsMap(a, c));
205   }
206
207   private static <T extends Annotation> void findAnnotationsMap(Class<T> a, Class<?> c, Map<Class<?>,T> m) {
208      if (c == null)
209         return;
210
211      T t = getDeclaredAnnotation(a, c);
212      if (t != null)
213         m.put(c, t);
214
215      findAnnotationsMap(a, c.getSuperclass(), m);
216
217      for (Class<?> c2 : c.getInterfaces())
218         findAnnotationsMap(a, c2, m);
219   }
220
221   /**
222    * Finds and appends the specified annotation on the specified class and superclasses/interfaces to the specified
223    * list.
224    *
225    * @param a The annotation.
226    * @param c The class.
227    * @param l The list of annotations.
228    */
229   public static <T extends Annotation> void appendAnnotations(Class<T> a, Class<?> c, List<T> l) {
230      if (c == null)
231         return;
232
233      addIfNotNull(l, getDeclaredAnnotation(a, c));
234
235      if (c.getPackage() != null)
236         addIfNotNull(l, c.getPackage().getAnnotation(a));
237
238      appendAnnotations(a, c.getSuperclass(), l);
239
240      for (Class<?> c2 : c.getInterfaces())
241         appendAnnotations(a, c2, l);
242   }

244   /**
245    * Similar to {@link Class#getResourceAsStream(String)} except looks up the parent hierarchy for the existence of
246    * the specified resource.
247    *
248    * @param c The class to return the resource on.
249    * @param name The resource name.
250    * @return An input stream on the specified resource, or <jk>null</jk> if the resource could not be found.
251    */
252   public static InputStream getResource(Class<?> c, String name) {
253      if (name == null)
254         return null;
255      while (c != null) {
256         InputStream is = c.getResourceAsStream(name);
257         if (is != null)
258            return is;
259         c = c.getSuperclass();
260      }
261      return null;
262   }
