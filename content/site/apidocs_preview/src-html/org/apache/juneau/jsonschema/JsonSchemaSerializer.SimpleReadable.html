022/**
023 * Serializes POJO metadata to HTTP responses as JSON.
024 *
025 * <h5 class='topic'>Media types</h5>
026 *
027 * Handles <code>Accept</code> types:  <code><b>application/json+schema, text/json+schema</b></code>
028 * <p>
029 * Produces <code>Content-Type</code> types:  <code><b>application/json</b></code>
030 *
031 * <h5 class='topic'>Description</h5>
032 *
033 * Produces the JSON-schema for the JSON produced by the {@link JsonSerializer} class with the same properties.
034 */
035public class JsonSchemaSerializer extends JsonSerializer {

043   /**
044    * Configuration property:  Add descriptions to types.
045    *
046    * <h5 class='section'>Property:</h5>
047    * <ul>
048    *    <li><b>Name:</b>  <js>"JsonSchemaSerializer.addDescriptionsTo.s"</js>
049    *    <li><b>Data type:</b>  <code>String</code>
050    *    <li><b>Default:</b>  Empty string.
051    *    <li><b>Session property:</b>  <jk>false</jk>
052    *    <li><b>Methods:</b>
053    *       <ul>
054    *          <li class='jm'>{@link JsonSchemaSerializerBuilder#addDescriptionsTo(String)}
055    *       </ul>
056    * </ul>
057    *
058    * <h5 class='section'>Description:</h5>
059    * <p>
060    * Identifies which categories of types that descriptions should be automatically added to generated schemas.
061    * <p>
062    * The description is the result of calling {@link ClassMeta#getReadableName()}.
063    * <p>
064    * The format is a comma-delimited list of any of the following values:
065    *
066    * <ul class='doctree'>
067    *    <li class='jf'>{@link TypeCategory#BEAN BEAN}
068    *    <li class='jf'>{@link TypeCategory#COLLECTION COLLECTION}
069    *    <li class='jf'>{@link TypeCategory#ARRAY ARRAY}
070    *    <li class='jf'>{@link TypeCategory#MAP MAP}
071    *    <li class='jf'>{@link TypeCategory#STRING STRING}
072    *    <li class='jf'>{@link TypeCategory#NUMBER NUMBER}
073    *    <li class='jf'>{@link TypeCategory#BOOLEAN BOOLEAN}
074    *    <li class='jf'>{@link TypeCategory#ANY ANY}
075    *    <li class='jf'>{@link TypeCategory#OTHER OTHER}
076    * </ul>
077    */

324   private final boolean useBeanDefs, allowNestedExamples, allowNestedDescriptions;
325   private final BeanDefMapper beanDefMapper;
326   private final Set<TypeCategory> addExamplesTo, addDescriptionsTo;
327   private final Map<String,ObjectMap> defaultSchemas;
328
329   /**
330    * Constructor.
331    *
332    * @param ps Initialize with the specified config property store.
333    */
334   public JsonSchemaSerializer(PropertyStore ps) {
335      super(
336         ps.builder()
337            .set(SERIALIZER_detectRecursions, true)
338            .set(SERIALIZER_ignoreRecursions, true)
339            .build(),
340         "application/json", "application/json+schema,text/json+schema"
341      );
342
343      useBeanDefs = getBooleanProperty(JSONSCHEMA_useBeanDefs, false);
344      allowNestedExamples = getBooleanProperty(JSONSCHEMA_allowNestedExamples, false);
345      allowNestedDescriptions = getBooleanProperty(JSONSCHEMA_allowNestedDescriptions, false);
346      beanDefMapper = getInstanceProperty(JSONSCHEMA_beanDefMapper, BeanDefMapper.class, BasicBeanDefMapper.class);
347      addExamplesTo = TypeCategory.parse(getStringProperty(JSONSCHEMA_addExamplesTo, null));
348      addDescriptionsTo = TypeCategory.parse(getStringProperty(JSONSCHEMA_addDescriptionsTo, null));
349      defaultSchemas = getMapProperty(JSONSCHEMA_defaultSchemas, ObjectMap.class);
350   }
351
352   @Override /* Context */
353   public JsonSchemaSerializerBuilder builder() {
354      return new JsonSchemaSerializerBuilder(getPropertyStore());
355   }

357   /**

359    *

362    *
363    * @return A new {@link JsonSerializerBuilder} object.
364    */

366      return new JsonSchemaSerializerBuilder();
367   }
