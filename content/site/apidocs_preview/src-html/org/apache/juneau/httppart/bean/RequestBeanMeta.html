013package org.apache.juneau.httppart.bean;
014
015import static org.apache.juneau.internal.ClassUtils.*;
016
017import java.lang.annotation.*;
018import java.lang.reflect.*;
019import java.util.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.http.annotation.*;
023import org.apache.juneau.httppart.*;
024import org.apache.juneau.internal.*;
025
026/**
027 * Represents the metadata gathered from a parameter or class annotated with {@link Request}.
028 */
029public class RequestBeanMeta {
030
031   /**
032    * Create metadata from specified parameter.
033    *
034    * @param m The method containing the parameter or parameter type annotated with {@link Request}.
035    * @param i The parameter index.
036    * @param ps
037    *    Configuration information used to instantiate part serializers and part parsers.
038    *    <br>Can be <jk>null</jk>.
039    * @return Metadata about the parameter, or <jk>null</jk> if parameter or parameter type not annotated with {@link Request}.
040    */
041   public static RequestBeanMeta create(Method m, int i, PropertyStore ps) {
042      if (! hasAnnotation(Request.class, m, i))
043         return null;
044      return new RequestBeanMeta.Builder(ps).apply(m, i).build();
045   }
046
047   /**
048    * Create metadata from specified class.
049    *
050    * @param c The class annotated with {@link Request}.
051    * @param ps
052    *    Configuration information used to instantiate part serializers and part parsers.
053    *    <br>Can be <jk>null</jk>.
054    * @return Metadata about the class, or <jk>null</jk> if class not annotated with {@link Request}.
055    */
056   public static RequestBeanMeta create(Class<?> c, PropertyStore ps) {
057      if (! hasAnnotation(Request.class, c))
058         return null;
059      return new RequestBeanMeta.Builder(ps).apply(c).build();
060   }
061
062   //-----------------------------------------------------------------------------------------------------------------
063   // Instance
064   //-----------------------------------------------------------------------------------------------------------------
065
066   private final ClassMeta<?> cm;
067   private final Map<String,RequestBeanPropertyMeta> properties, propertiesByGetter;
068   private final HttpPartSerializer serializer;
069   private final HttpPartParser parser;
070
071   RequestBeanMeta(Builder b) {
072      this.cm = b.cm;
073      this.serializer = ClassUtils.newInstance(HttpPartSerializer.class, b.serializer, true, b.ps);
074      this.parser = ClassUtils.newInstance(HttpPartParser.class, b.parser, true, b.ps);
075      Map<String,RequestBeanPropertyMeta> properties = new LinkedHashMap<>(), propertiesByGetter = new LinkedHashMap<>();
076      for (Map.Entry<String,RequestBeanPropertyMeta.Builder> e : b.properties.entrySet()) {
077         RequestBeanPropertyMeta pm = e.getValue().build(serializer, parser);
078         properties.put(e.getKey(), pm);
079         propertiesByGetter.put(pm.getGetter(), pm);
080
081      }
082      this.properties = Collections.unmodifiableMap(properties);
083      this.propertiesByGetter = Collections.unmodifiableMap(propertiesByGetter);
084   }
085
086   static class Builder {
087      ClassMeta<?> cm;
088      PropertyStore ps;
089      Class<? extends HttpPartSerializer> serializer;
090      Class<? extends HttpPartParser> parser;
091      Map<String,RequestBeanPropertyMeta.Builder> properties = new LinkedHashMap<>();
092
093      Builder(PropertyStore ps) {
094         this.ps = ps;
095      }
096
097      Builder apply(Method m, int i) {
098         return apply(m.getParameterTypes()[i]).apply(getAnnotation(Request.class, m, i));
099      }
100
101      Builder apply(Class<?> c) {
102         apply(getAnnotation(Request.class, c));
103         this.cm = BeanContext.DEFAULT.getClassMeta(c);
104         BeanMeta<?> bm = BeanContext.DEFAULT.getBeanMeta(c);
105         for (BeanPropertyMeta bp : bm.getPropertyMetas()) {
106            String n = bp.getName();
107            Annotation a = bp.getAnnotation(Path.class);
108            String g = bp.getGetter().getName();
109            if (a != null) {
110               HttpPartSchemaBuilder s = HttpPartSchema.create().apply(a);
111               getProperty(n, HttpPartType.PATH).apply(s).getter(g);
112            }
113            a = bp.getAnnotation(Header.class);
114            if (a != null) {
115               HttpPartSchemaBuilder s = HttpPartSchema.create().apply(a);
116               getProperty(n, HttpPartType.HEADER).apply(s).getter(g);
117            }
118            a = bp.getAnnotation(Query.class);
119            if (a != null) {
120               HttpPartSchemaBuilder s = HttpPartSchema.create().apply(a);
121               getProperty(n, HttpPartType.QUERY).apply(s).getter(g);
122            }
123            a = bp.getAnnotation(FormData.class);
124            if (a != null) {
125               HttpPartSchemaBuilder s = HttpPartSchema.create().apply(a);
126               getProperty(n, HttpPartType.FORMDATA).apply(s).getter(g);
127            }
128            a = bp.getAnnotation(Body.class);
129            if (a != null) {
130               HttpPartSchemaBuilder s = HttpPartSchema.create().apply(a);
131               getProperty(n, HttpPartType.BODY).apply(s).getter(g);
132            }
133         }
134         return this;
135      }
136
137      Builder apply(Request a) {
138         if (a != null) {
139            if (a.partSerializer() != HttpPartSerializer.Null.class)
140               serializer = a.partSerializer();
141            if (a.partParser() != HttpPartParser.Null.class)
142               parser = a.partParser();
143         }
144         return this;
145      }
146
147      RequestBeanMeta build() {
148         return new RequestBeanMeta(this);
149      }
150
151      private RequestBeanPropertyMeta.Builder getProperty(String name, HttpPartType partType) {
152         RequestBeanPropertyMeta.Builder b = properties.get(name);
153         if (b == null) {
154            b = RequestBeanPropertyMeta.create().name(name).partType(partType);
155            properties.put(name, b);
156         }
157         return b;
158      }
159   }
160
161   /**
162    * Returns metadata about the bean property with the specified property name.
163    *
164    * @param name The bean property name.
165    * @return Metadata about the bean property, or <jk>null</jk> if none found.
166    */
167   public RequestBeanPropertyMeta getProperty(String name) {
168      return properties.get(name);
169   }
170
171   /**
172    * Returns metadata about the bean property with the specified method getter name.
173    *
174    * @param name The bean method getter name.
175    * @return Metadata about the bean property, or <jk>null</jk> if none found.
176    */
177   public RequestBeanPropertyMeta getPropertyByGetter(String name) {
178      return propertiesByGetter.get(name);
179   }
180
181   /**
182    * Returns metadata about the class.
183    *
184    * @return Metadata about the class.
185    */
186   public ClassMeta<?> getClassMeta() {
187      return cm;
188   }
189}
