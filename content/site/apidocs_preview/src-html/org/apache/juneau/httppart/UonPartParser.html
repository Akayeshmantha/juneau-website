013package org.apache.juneau.httppart;
014
015import java.lang.reflect.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.parser.*;
019import org.apache.juneau.uon.*;
020
021/**
022 * Parses HTTP headers, query/form-data parameters, and path variables into POJOs.
023 *
024 * <p>
025 * This parser expects UON notation for all parts by default.
026 */
027public class UonPartParser extends UonParser implements HttpPartParser {
028
029   //-------------------------------------------------------------------------------------------------------------------
030   // Predefined instances
031   //-------------------------------------------------------------------------------------------------------------------
032
033   /** Reusable instance of {@link UonPartParser}. */
034   public static final UonPartParser DEFAULT = new UonPartParser(PropertyStore.DEFAULT);
035
036
037   //-------------------------------------------------------------------------------------------------------------------
038   // Instance
039   //-------------------------------------------------------------------------------------------------------------------
040
041   /**
042    * Constructor.
043    *
044    * @param ps The property store containing all the settings for this object.
045    */
046   public UonPartParser(PropertyStore ps) {
047      super(
048         ps.builder()
049            .build(),
050         "application/x-www-form-urlencoded"
051      );
052   }
053
054   @Override /* Context */
055   public UonPartParserBuilder builder() {
056      return new UonPartParserBuilder(getPropertyStore());
057   }
058
059   /**
060    * Instantiates a new clean-slate {@link UonPartParserBuilder} object.
061    *
062    * <p>
063    * This is equivalent to simply calling <code><jk>new</jk> UonPartParserBuilder()</code>.
064    *
065    * <p>
066    * Note that this method creates a builder initialized to all default settings, whereas {@link #builder()} copies
067    * the settings of the object called on.
068    *
069    * @return A new {@link UonPartParserBuilder} object.
070    */
071   public static UonPartParserBuilder create() {
072      return new UonPartParserBuilder();
073   }
074
075   //-------------------------------------------------------------------------------------------------------------------
076   // Entry point methods
077   //-------------------------------------------------------------------------------------------------------------------
078
079   @Override
080   public UonPartParserSession createSession(ParserSessionArgs args) {
081      return new UonPartParserSession(this, args);
082   }
083
084   @Override
085   public UonPartParserSession createSession() {
086      return new UonPartParserSession(this, ParserSessionArgs.DEFAULT);
087   }
088//
089// @Override /* HttpPartParser */
090// public <T> T parse(HttpPartType partType, HttpPartSchema schema, String in, ClassMeta<T> toType) throws ParseException, SchemaValidationException {
091//    return createSession().parse(partType, schema, in, toType);
092// }
093
094   @Override /* HttpPartParser */
095   public <T> T parse(HttpPartType partType, HttpPartSchema schema, String in, Class<T> toType) throws ParseException, SchemaValidationException {
096      return createSession().parse(partType, schema, in, toType);
097   }
098
099   @Override /* HttpPartParser */
100   public <T> T parse(HttpPartType partType, HttpPartSchema schema, String in, Type toType, Type...toTypeArgs) throws ParseException, SchemaValidationException {
101      return createSession().parse(partType, schema, in, toType, toTypeArgs);
102   }
103//
104// @Override /* HttpPartParser */
105// public <T> T parse(HttpPartSchema schema, String in, ClassMeta<T> toType) throws ParseException, SchemaValidationException {
106//    return createSession().parse(null, schema, in, toType);
107// }
108
109   @Override /* HttpPartParser */
110   public <T> T parse(HttpPartSchema schema, String in, Class<T> toType) throws ParseException, SchemaValidationException {
111      return createSession().parse(null, schema, in, toType);
112   }
113
114   @Override /* HttpPartParser */
115   public <T> T parse(HttpPartSchema schema, String in, Type toType, Type...toTypeArgs) throws ParseException, SchemaValidationException {
116      return createSession().parse(null, schema, in, toType, toTypeArgs);
117   }
118}
