013package org.apache.juneau.http.annotation;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019
020import org.apache.juneau.httppart.*;
021
022/**
023 * Request bean annotation.
024 *
025 * <p>
026 * Identifies an interface to use to interact with HTTP parts of an HTTP request through a bean.
027 *
028 * <p>
029 * Can be used in the following locations:
030 * <ul>
031 *    <li>Java method arguments and argument-types of client-side <ja>@Remoteable</ja>-annotated REST interface proxies.
032 *    <li>Java method arguments and argument-types of server-side <ja>@RestMethod</ja>-annotated REST Java methods.
033 * </ul>
034 *
035 * <h5 class='topic'>Server-side REST</h5>
036 *
037 * Annotation that can be applied to a parameter of a <ja>@RestMethod</ja>-annotated method to identify it as an interface for retrieving HTTP parts through a bean interface.
038 *
039 * <h5 class='section'>Example:</h5>
040 * <p class='bcode w800'>
041 *    <ja>@RestMethod</ja>(path=<js>"/mypath/{p1}/{p2}/*"</js>)
042 *    <jk>public void</jk> myMethod(<ja>@RequestBean</ja> MyRequestBean rb) {...}
043 *
044 *    <jk>public interface</jk> MyRequestBean {
045 *
046 *       <ja>@Path</ja> <jc>// Path variable name inferred from getter.</jc>
047 *       String getP1();
048 *
049 *       <ja>@Path</ja>(<js>"p2"</js>)
050 *       String getX();
051 *
052 *       <ja>@Path</ja>(<js>"/*"</js>)
053 *       String getRemainder();
054 *
055 *       <ja>@Query</ja>
056 *       String getQ1();
057 *
058 *       <ja>@Query</ja>
059 *       <ja>@BeanProperty</ja>(name=<js>"q2"</js>)
060 *       String getQuery2();
061 *
062 *    <jc>// Schema-based query parameter:  Pipe-delimited lists of comma-delimited lists of integers.</jc>
063 *       <ja>@Query</ja>(
064 *          collectionFormat=<js>"pipes"</js>
065 *          items=<ja>@Items</ja>(
066 *             items=<ja>@SubItems</ja>(
067 *                collectionFormat=<js>"csv"</js>
068 *                type=<js>"integer"</js>
069 *             )
070 *          )
071 *       )
072 *       <jk>int</jk>[][] getQ3();
073 *
074 *       <ja>@Header</ja>(<js>"*"</js>)
075 *       Map&lt;String,Object&gt; getHeaders();
076 * </p>
077 * <p class='bcode w800'>
078 *    <jc>// Same as above but annotation defined on interface.</jc>
079 *    <ja>@RestMethod</ja>(path=<js>"/mypath/{p1}/{p2}/*"</js>)
080 *    <jk>public void</jk> myMethod(MyRequestBean rb) {...}
081 *
082 * <ja>@RequestBean</ja>
083 *    <jk>public interface</jk> MyRequestBean {...}
084 *
085 * <p>
086 * The return types of the getters must be the supported parameter types for the HTTP-part annotation used.
087 * <br>Schema-based serialization and parsing is allowed just as if used as individual parameter types.
088 *
089 * <p>
090 * TODO - Swagger-info in <ja>@RequestBean</ja>s is not currently reflected in generated Swagger documentation.
091 *
092 * <h5 class='section'>See Also:</h5>
093 * <ul>
094 *    <li class='link'><a class="doclink" href="../../../../../overview-summary.html#juneau-rest-server.HttpPartAnnotations.RequestBean">Overview &gt; juneau-rest-server &gt; @RequestBean</a>
095 * </ul>
096 *
097 * <h5 class='topic'>Client-side REST</h5>
098 *
099 * Annotation applied to Java method arguments of interface proxies to denote a bean with remoteable annotations.
100
101 * <h5 class='section'>Example:</h5>
102 * <p class='bcode w800'>
103 *    <ja>@Remoteable</ja>(path=<js>"/myproxy"</js>)
104 *    <jk>public interface</jk> MyProxy {
105 *
106 *       <ja>@RemoteMethod</ja>(path=<js>"/mymethod/{p1}/{p2}"</js>)
107 *       String myProxyMethod(<ja>@RequestBean</ja> MyRequestBean bean);
108 *    }
109 *
110 *    <jk>public class</jk> MyRequestBean {
111 *
112 *       <ja>@Path</ja> <jc>// Path variable name inferred from getter.</jc>
113 *       <jk>public</jk> String getP1() {...}
114 *
115 *       <ja>@Path</ja>(<js>"p2"</js>)
116 *       <jk>public</jk> String getX() {...}
117 *
118 *       <ja>@Path</ja>(<js>"/*"</js>)
119 *       <jk>public</jk> String getRemainder() {...}
120 *
121 *       <ja>@Query</ja>
122 *       <jk>public</jk> String getQ1() {...}
123 *
124 *       <ja>@Query</ja>
125 *       <ja>@BeanProperty</ja>(name=<js>"q2"</js>)
126 *       <jk>public</jk> String getQuery2() {...}
127 *
128 *    <jc>// Schema-based query parameter:  Pipe-delimited lists of comma-delimited lists of integers.</jc>
129 *       <ja>@Query</ja>(
130 *          collectionFormat=<js>"pipes"</js>
131 *          items=<ja>@Items</ja>(
132 *             items=<ja>@SubItems</ja>(
133 *                collectionFormat=<js>"csv"</js>
134 *                type=<js>"integer"</js>
135 *             )
136 *          )
137 *       )
138 *       <jk>public</jk> <jk>int</jk>[][] getQ3() {...}
139 *
140 *       <ja>@Header</ja>(<js>"*"</js>)
141 *       <jk>public</jk> Map&lt;String,Object&gt; getHeaders() {...}
142 *    }
143 * </p>
144 *
145 * <h5 class='section'>See Also:</h5>
146 * <ul class='doctree'>
147 *    <li class='link'><a class='doclink' href='../../../../../overview-summary.html#juneau-rest-client.3rdPartyProxies'>Overview &gt; juneau-rest-client &gt; Interface Proxies Against 3rd-party REST Interfaces</a>
148 * </ul>
149 */
150@Documented
151@Target(PARAMETER)
152@Retention(RUNTIME)
153@Inherited
154public @interface RequestBean {
155
156   /**
157    * Specifies the {@link HttpPartSerializer} class used for serializing values to strings.
158    *
159    * <p>
160    * Overrides for this part the part serializer defined on the REST client which by default is {@link OpenApiPartSerializer}.
161    */
162   Class<? extends HttpPartSerializer> serializer() default HttpPartSerializer.Null.class;
163
164   /**
165    * Specifies the {@link HttpPartParser} class used for parsing strings to values.
166    *
167    * <p>
168    * Overrides for this part the part parser defined on the REST resource which by default is {@link OpenApiPartParser}.
169    */
170   Class<? extends HttpPartParser> parser() default HttpPartParser.Null.class;
171}
