013package org.apache.juneau.http.annotation;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019
020import org.apache.juneau.*;
021import org.apache.juneau.annotation.*;
022import org.apache.juneau.httppart.*;
023import org.apache.juneau.json.*;
024import org.apache.juneau.jsonschema.*;
025
026/**
027 * TODO
028 */
029@Documented
030@Target({PARAMETER,TYPE,METHOD})
031@Retention(RUNTIME)
032@Inherited
033public @interface ResponseBody {
034
035   /**
036    * Specifies the {@link HttpPartSerializer} class used for serializing values to strings.
037    *
038    * <p>
039    * Overrides for this part the part serializer defined on the REST resource which by default is {@link OpenApiPartSerializer}.
040    */
041   Class<? extends HttpPartSerializer> partSerializer() default HttpPartSerializer.Null.class;
042
043   /**
044    * Specifies whether a part serializer should be used for serializing this value.
045    *
046    * <p>
047    * If <jk>false</jk>, then it indicates that normal Juneau serializers (e.g. {@link JsonSerializer}) should be used for this part.
048    */
049   public boolean usePartSerializer() default false;
050
051   /**
052    * <mk>schema</mk> field of the Swagger <a class="doclink" href="https://swagger.io/specification/v2/#responseObject">Response</a> object.
053    *
054    * <h5 class='section'>Used for:</h5>
055    * <ul class='spaced-list'>
056    *    <li>
057    *       Server-side schema-based serializing and serializing validation.
058    *    <li>
059    *       Server-side generated Swagger documentation.
060    * </ul>
061    */
062   Schema schema() default @Schema;
063
064   /**
065    * A serialized example of the body of a response.
066    *
067    * <p>
068    * This is the <a class='doclink' href='../../../../../overview-summary.html#juneau-marshall.JsonDetails.SimplifiedJson'>Simplified JSON</a> of an example of the body.
069    *
070    * <p>
071    * This value is converted to a POJO and then serialized to all the registered serializers on the REST method to produce examples for all
072    * supported language types.
073    * <br>These values are then used to automatically populate the {@link #examples} field.
074    *
075    * <h5 class='section'>Example:</h5>
076    * <p class='bcode w800'>
077    *    <jc>// A JSON representation of a PetCreate object.</jc>
078    *    <ja>@ResponseBody</ja>(
079    *       example=<js>"{name:'Doggie',price:9.99,species:'Dog',tags:['friendly','cute']}"</js>
080    *    )
081    * </p>
082    *
083    * <p>
084    * There are several other options for defining this example:
085    * <ul class='spaced-list'>
086    *    <li>
087    *       Defining an <js>"x-example"</js> field in the inherited Swagger JSON response object (classpath file or <code><ja>@ResourceSwagger</ja>(value)</code>/<code><ja>@MethodSwagger</ja>(value)</code>).
088    *    <li>
089    *       Defining an <js>"x-example"</js> field in the Swagger Schema Object for the response object (including referenced <js>"$ref"</js> schemas).
090    *    <li>
091    *       Allowing Juneau to auto-generate a code example.
092    * </ul>
093    *
094    * <p>
095    * The latter is important because Juneau also supports auto-generation of JSON-Schema from POJO classes using {@link JsonSchemaSerializer} which has several of it's own
096    * options for auto-detecting and calculation POJO examples.
097    *
098    * <p>
099    * In particular, examples can be defined via static methods, fields, and annotations on the classes themselves.
100    *
101    * <p class='bcode w800'>
102    *    <jc>// Annotation on class.</jc>
103    *    <ja>@Example</ja>(<js>"{name:'Doggie',price:9.99,species:'Dog',tags:['friendly','cute']}"</js>)
104    *    <jk>public class</jk> PetCreate {
105    *       ...
106    *    }
107    * </p>
108    * <p class='bcode w800'>
109    *    <jc>// Annotation on static method.</jc>
110    *    <jk>public class</jk> PetCreate {
111    *
112    *       <ja>@Example</ja>
113    *       <jk>public static</jk> PetCreate <jsm>sample</jsm>() {
114    *          <jk>return new</jk> PetCreate(<js>"Doggie"</js>, 9.99f, <js>"Dog"</js>, <jk>new</jk> String[] {<js>"friendly"</js>,<js>"cute"</js>});
115    *       }
116    *    }
117    * </p>
118    * <p class='bcode w800'>
119    *    <jc>// Static method with specific name 'example'.</jc>
120    *    <jk>public class</jk> PetCreate {
121    *
122    *       <jk>public static</jk> PetCreate <jsm>example</jsm>() {
123    *          <jk>return new</jk> PetCreate(<js>"Doggie"</js>, 9.99f, <js>"Dog"</js>, <jk>new</jk> String[] {<js>"friendly"</js>,<js>"cute"</js>});
124    *       }
125    *    }
126    * </p>
127    * <p class='bcode w800'>
128    *    <jc>// Static field.</jc>
129    *    <jk>public class</jk> PetCreate {
130    *
131    *       <ja>@Example</ja>
132    *       <jk>public static</jk> PetCreate <jsf>EXAMPLE</jsf> = <jk>new</jk> PetCreate(<js>"Doggie"</js>, 9.99f, <js>"Dog"</js>, <jk>new</jk> String[] {<js>"friendly"</js>,<js>"cute"</js>});
133    *    }
134    * </p>
135    *
136    * <p>
137    * Examples can also be specified via generic properties as well using the {@link BeanContext#BEAN_examples} property at either the class or method level.
138    * <p class='bcode w800'>
139    *    <jc>// Examples defined at class level.</jc>
140    *    <ja>@RestResource</ja>(
141    *       properties={
142    *          <ja>@Property</ja>(
143    *             name=<jsf>BEAN_examples</jsf>,
144    *             value=<js>"{'org.apache.juneau.examples.rest.petstore.PetCreate': {name:'Doggie',price:9.99,species:'Dog',tags:['friendly','cute']}}"</js>
145    *          )
146    *       }
147    *    )
148    * </p>
149    *
150    * <h5 class='section'>Used for:</h5>
151    * <ul class='spaced-list'>
152    *    <li>
153    *       Server-side generated Swagger documentation.
154    * </ul>
155    *
156    * <h5 class='section'>See also:</h5>
157    * <ul>
158    *    <li class='ja'>{@link Example}
159    *    <li class='jc'>{@link BeanContext}
160    *    <ul>
161    *       <li class='jf'>{@link BeanContext#BEAN_examples BEAN_examples}
162    *    </ul>
163    *    <li class='jc'>{@link JsonSchemaSerializer}
164    *    <ul>
165    *       <li class='jf'>{@link JsonSchemaSerializer#JSONSCHEMA_addExamplesTo JSONSCHEMA_addExamplesTo}
166    *       <li class='jf'>{@link JsonSchemaSerializer#JSONSCHEMA_allowNestedExamples JSONSCHEMA_allowNestedExamples}
167    *    </ul>
168    * </ul>
169    *
170    * <h5 class='section'>Notes:</h5>
171    * <ul class='spaced-list'>
172    *    <li>
173    *       The format is any <a class='doclink' href='../../../../../overview-summary.html#juneau-marshall.JsonDetails.SimplifiedJson'>Simplified JSON</a> if the object can be converted to a POJO using {@link JsonParser#DEFAULT} or a simple String if the object
174    *       has a schema associated with it can be converted from a String.
175    *       <br>Multiple lines are concatenated with newlines.
176    *    <li>
177    *       The format of this object can also be a simple String if the body has a schema associated with it, meaning it's meant to be treated as an HTTP part.
178    *    <li>
179    *       Supports <a class="doclink" href="../../../../../overview-summary.html#DefaultRestSvlVariables">initialization-time and request-time variables</a>
180    *       (e.g. <js>"$L{my.localized.variable}"</js>).
181    * </ul>
182    */
183   String[] example() default {};
184
185   /**
186    * Serialized examples of the body of a response.
187    *
188    * <p>
189    * This is a <a class='doclink' href='../../../../../overview-summary.html#juneau-marshall.JsonDetails.SimplifiedJson'>Simplified JSON</a> object whose keys are media types and values are string representations of that value.
190    *
191    * <p>
192    * In general you won't need to populate this value directly since it will automatically be calculated based on the value provided in the {@link #example()} field.
193    * <br>However, this field allows you to override the behavior and show examples for only specified media types or different examples for different media types.
194    *
195    * <p class='bcode w800'>
196    *    <jc>// A JSON representation of a PetCreate object.</jc>
197    *    <ja>@ResponseBody</ja>(
198    *       examples={
199    *          <js>"'application/json':'{name:\\'Doggie\\',species:\\'Dog\\'}',"</js>,
200    *          <js>"'text/uon':'(name:Doggie,species=Dog)'"</js>
201    *       }
202    *    )
203    * </p>
204    *
205    * <h5 class='section'>Used for:</h5>
206    * <ul class='spaced-list'>
207    *    <li>
208    *       Server-side generated Swagger documentation.
209    * </ul>
210    *
211    * <h5 class='section'>Notes:</h5>
212    * <ul class='spaced-list'>
213    *    <li>
214    *       The format is a <a class='doclink' href='../../../../../overview-summary.html#juneau-marshall.JsonDetails.SimplifiedJson'>Simplified JSON</a> object with string keys (media type) and string values (example for that media type) .
215    *    <li>
216    *       The leading/trailing <code>{ }</code> characters are optional.
217    *    <li>
218    *       Multiple lines are concatenated with newlines so that you can format the value to be readable:
219    *    <li>
220    *       Supports <a class="doclink" href="../../../../../overview-summary.html#DefaultRestSvlVariables">initialization-time and request-time variables</a>
221    *       (e.g. <js>"$L{my.localized.variable}"</js>).
222    *    <li>
223    *       Resolution of variables is delayed until request time and occurs before parsing.
224    *       <br>This allows you to, for example, pull in a JSON construct from a properties file based on the locale of the HTTP request.
225    * </ul>
226    */
227   String[] examples() default {};
228
229   /**
230    * Free-form value for the Swagger <a class="doclink" href="https://swagger.io/specification/v2/#responseObject">Response</a> object.
231    *
232    * <p>
233    * This is a <a class='doclink' href='../../../../../overview-summary.html#juneau-marshall.JsonDetails.SimplifiedJson'>Simplified JSON</a> object that makes up the swagger information for this field.
234    *
235    * <h5 class='section'>Notes:</h5>
236    * <ul class='spaced-list'>
237    *    <li>
238    *       Note that the only swagger field you can't specify using this value is <js>"code"</js> whose value needs to be known during servlet initialization.
239    *    <li>
240    *       The format is a <a class='doclink' href='../../../../../overview-summary.html#juneau-marshall.JsonDetails.SimplifiedJson'>Simplified JSON</a> object.
241    *    <li>
242    *       The leading/trailing <code>{ }</code> characters are optional.
243    *       <br>The following two example are considered equivalent:
244    *       <p class='bcode w800'>
245    *    <ja>@ResponseBody</ja>(api=<js>"{schema:{type:'string'}}}"</js>)
246    *       </p>
247    *       <p class='bcode w800'>
248    *    <ja>@ResponseBody</ja>(api=<js>"schema: {type:'string'}"</js>)
249    *       </p>
250    *    <li>
251    *       Multiple lines are concatenated with newlines so that you can format the value to be readable.
252    *    <li>
253    *       Supports <a class="doclink" href="../../../../../overview-summary.html#DefaultRestSvlVariables">initialization-time and request-time variables</a>
254    *       (e.g. <js>"$L{my.localized.variable}"</js>).
255    *    <li>
256    *       Values defined in this field supersede values pulled from the Swagger JSON file and are superseded by individual values defined on this annotation.
257    * </ul>
258    */
259   String[] api() default {};
260}
