262   /**
263    * Same as {@link #getObject(Class, String, Object)}, but value is referenced through section name and key instead of full key.
264    *
265    * @param c The class to convert the value to.
266    * @param sectionName The section name.  Must not be <jk>null</jk>.
267    * @param sectionKey The section key.  Must not be <jk>null</jk>.
268    * @param def The default value if section or key does not exist.
269    * @throws ParseException If parser could not parse the value or if a parser is not registered with this config file.
270    * @return The value, or the default value if the section or value doesn't exist.
271    */
272   @SuppressWarnings("unchecked")
273   public <T> T getObject(Class<T> c, String sectionName, String sectionKey, T def) throws ParseException {
274      String s = get(sectionName, sectionKey);
275      if (s == null)
276         return def;
277      if (c == String.class)
278         return (T)s;
279      if (c == Integer.class || c == int.class)
280         return (T)(StringUtils.isEmpty(s) ? def : Integer.valueOf(parseIntWithSuffix(s)));
281      if (c == Boolean.class || c == boolean.class)
282         return (T)(StringUtils.isEmpty(s) ? def : Boolean.valueOf(Boolean.parseBoolean(s)));
283      if (c == String[].class) {
284         String[] r = StringUtils.isEmpty(s) ? new String[0] : StringUtils.split(s, ',');
285         return (T)(r.length == 0 ? def : r);
286      }
287      if (c.isArray()) {
288         Class<?> ce = c.getComponentType();
289         if (StringUtils.isEmpty(s))
290            return def;
291         String[] r = StringUtils.split(s, ',');
292         Object o = Array.newInstance(ce, r.length);
293         for (int i = 0; i < r.length; i++)
294            Array.set(o, i, getParser().parse(r[i], ce));
295         return (T)o;
296      }
297      if (StringUtils.isEmpty(s))
298         return def;
299      return getParser().parse(s, c);
300   }

412   /**
413    * Adds or replaces an entry with the specified key with the specified value.
414    * <p>
415    * The format of the entry depends on the data type of the value.
416    * <ul class='spaced-list'>
417    *    <li>Simple types (<code>String</code>, <code>Number</code>, <code>Boolean</code>, primitives)
418    *       are serialized as plain strings.
419    *    <li>Arrays and collections of simple types are serialized as comma-delimited lists of plain strings.
420    *    <li>Other types (e.g. beans) are serialized using the serializer registered with this config file.
421    *    <li>Arrays and collections of other types are serialized as comma-delimited lists of serialized strings of each entry.
422    * </ul>
423    *
424    * @param key The key.  See {@link #getString(String)} for a description of the key.
425    * @param value The new value.
426    * @param encoded If <jk>true</jk>, value is encoded by the registered encoder when the config file is persisted to disk.
427    * @return The previous value, or <jk>null</jk> if the section or key did not previously exist.
428    * @throws SerializeException If serializer could not serialize the value or if a serializer is not registered with this config file.
429    * @throws UnsupportedOperationException If config file is read only.
430    */
431   public final String put(String key, Object value, boolean encoded) throws SerializeException {
432      assertFieldNotNull(key, "key");
433      if (value == null)
434         value = "";
435      Class<?> c = value.getClass();
436      if (isSimpleType(c))
437         return put(getSectionName(key), getSectionKey(key), value.toString(), encoded);
438      if (c.isAssignableFrom(Collection.class)) {
439         Collection<?> c2 = (Collection<?>)value;
440         String[] r = new String[c2.size()];
441         int i = 0;
442         for (Object o2 : c2) {
443            boolean isSimpleType = o2 == null ? true : isSimpleType(o2.getClass());
444            r[i++] = (isSimpleType ? Array.get(value, i).toString() : getSerializer().toString(Array.get(value, i)));
445         }
446         return put(getSectionName(key), getSectionKey(key), StringUtils.join(r, ','), encoded);
447      } else if (c.isArray()) {
448         boolean isSimpleType = isSimpleType(c.getComponentType());
449         String[] r = new String[Array.getLength(value)];
450         for (int i = 0; i < r.length; i++) {
451            r[i] = (isSimpleType ? Array.get(value, i).toString() : getSerializer().toString(Array.get(value, i)));
452         }
453         return put(getSectionName(key), getSectionKey(key), StringUtils.join(r, ','), encoded);
454      }
455      return put(getSectionName(key), getSectionKey(key), getSerializer().toString(value), encoded);
456   }

458   private static boolean isSimpleType(Class<?> c) {
459      return (c == String.class || c.isPrimitive() || c.isAssignableFrom(Number.class) || c == Boolean.class);
460   }

462   /**
463    * Returns the specified section as a map of key/value pairs.
464    *
465    * @param sectionName The section name to retrieve.
466    * @return A map of the section, or <jk>null</jk> if the section was not found.
467    */
468   public final ObjectMap getSectionMap(String sectionName) {
469      readLock();
470      try {
471         Set<String> keys = getSectionKeys(sectionName);
472         if (keys == null)
473            return null;
474         ObjectMap m = new ObjectMap();
475         for (String key : keys)
476            m.put(key, get(sectionName, key));
477         return m;
478      } finally {
479         readUnlock();
480      }
481   }

483   /**
484    * Copies the entries in a section to the specified bean by calling the public setters on that bean.
485    *
486    * @param sectionName The section name to write from.
487    * @param bean The bean to set the properties on.
488    * @param ignoreUnknownProperties If <jk>true</jk>, don't throw an {@link IllegalArgumentException} if this section
489    *    contains a key that doesn't correspond to a setter method.
490    * @param permittedPropertyTypes If specified, only look for setters whose property types
491    *    are those listed.  If not specified, use all setters.
492    * @return An object map of the changes made to the bean.
493    * @throws ParseException If parser was not set on this config file or invalid properties were found in the section.
494    * @throws IllegalArgumentException
495    * @throws IllegalAccessException
496    * @throws InvocationTargetException
497    */
498   public final ObjectMap writeProperties(String sectionName, Object bean, boolean ignoreUnknownProperties, Class<?>...permittedPropertyTypes) throws ParseException, IllegalArgumentException, IllegalAccessException, InvocationTargetException {
499      assertFieldNotNull(bean, "bean");
500      ObjectMap om = new ObjectMap();
501      readLock();
502      try {
503         Set<String> keys = getSectionKeys(sectionName);
504         if (keys == null)
505            throw new IllegalArgumentException("Section not found");
506         keys = new LinkedHashSet<String>(keys);
507         for (Method m : bean.getClass().getMethods()) {
508            int mod = m.getModifiers();
509            if (isPublic(mod) && (!isStatic(mod)) && m.getName().startsWith("set") && m.getParameterTypes().length == 1) {
510               Class<?> pt = m.getParameterTypes()[0];
511               if (permittedPropertyTypes == null || permittedPropertyTypes.length == 0 || ArrayUtils.contains(pt, permittedPropertyTypes)) {
512                  String propName = Introspector.decapitalize(m.getName().substring(3));
513                  Object value = getObject(pt, sectionName, propName, null);
514                  if (value != null) {
515                     m.invoke(bean, value);
516                     om.put(propName, value);
517                     keys.remove(propName);
518                  }
519               }
520            }
521         }
522         if (! (ignoreUnknownProperties || keys.isEmpty()))
523            throw new ParseException("Invalid properties found in config file section ["+sectionName+"]: " + JsonSerializer.DEFAULT_LAX.toString(keys));
524         return om;
525      } finally {
526         readUnlock();
527      }
528   }

542   /**
543    * Converts this config file section to the specified bean instance.
544    *
545    * @param sectionName The section name to write from.
546    * @param c The bean class to create.
547    * @param ignoreUnknownProperties If <jk>false</jk>, throws a {@link ParseException} if
548    *    the section contains an entry that isn't a bean property name.
549    * @return A new bean instance.
550    * @throws ParseException
551    */
552   public final <T> T getSectionAsBean(String sectionName, Class<T> c, boolean ignoreUnknownProperties) throws ParseException {
553      assertFieldNotNull(c, "c");
554      readLock();
555      try {
556         BeanMap<T> bm = getParser().getBeanContext().createSession().newBeanMap(c);
557         for (String k : getSectionKeys(sectionName)) {
558            BeanPropertyMeta bpm = bm.getPropertyMeta(k);
559            if (bpm == null) {
560               if (! ignoreUnknownProperties)
561                  throw new ParseException("Unknown property {0} encountered", k);
562            } else {
563               bm.put(k, getObject(bpm.getClassMeta().getInnerClass(), sectionName + '/' + k));
564            }
565         }
566         return bm.getBean();
567      } finally {
568         readUnlock();
569      }
570   }

747   private static int parseIntWithSuffix(String s) {
748      assertFieldNotNull(s, "s");
749      int m = 1;
750      if (s.endsWith("M")) {
751         m = 1024*1024;
752         s = s.substring(0, s.length()-1).trim();
753      } else if (s.endsWith("K")) {
754         m = 1024;
755         s = s.substring(0, s.length()-1).trim();
756      }
757      return Integer.parseInt(s) * m;
758   }
