075   public ConfigFileImpl(File file, boolean readOnly, Encoder encoder, WriterSerializer serializer, ReaderParser parser, Charset charset) throws IOException {
076      this.file = file;
077      this.encoder = encoder == null ? XorEncoder.INSTANCE : encoder;
078      this.serializer = serializer == null ? JsonSerializer.DEFAULT : serializer;
079      this.parser = parser == null ? JsonParser.DEFAULT : parser;
080      this.charset = charset == null ? Charset.defaultCharset() : charset;
081      load();
082      this.readOnly = readOnly;
083      if (readOnly) {
084         this.sections = Collections.unmodifiableMap(this.sections);
085         for (Section s : sections.values())
086            s.setReadOnly();
087      }
088   }

125   @Override /* ConfigFile */
126   public ConfigFileImpl load() throws IOException {
127      Reader r = null;
128      if (file != null && file.exists())
129         r = new InputStreamReader(new FileInputStream(file), charset);
130      else
131         r = new StringReader("");
132      try {
133         load(r);
134      } finally {
135         r.close();
136      }
137      return this;
138   }

140   @Override /* ConfigFile */
141   public ConfigFileImpl load(Reader r) throws IOException {
142      assertFieldNotNull(r, "r");
143      writeLock();
144      try {
145         this.sections = Collections.synchronizedMap(new LinkedHashMap<String,Section>());
146         BufferedReader in = new BufferedReader(r);
147         try {
148            writeLock();
149            hasBeenModified = false;
150            try {
151               sections.clear();
152               String line = null;
153               Section section = getSection(null, true);
154               ArrayList<String> lines = new ArrayList<String>();
155               boolean canAppend = false;
156               while ((line = in.readLine()) != null) {
157                  if (isSection(line)) {
158                     section.addLines(null, lines.toArray(new String[lines.size()]));
159                     lines.clear();
160                     canAppend = false;
161                     String sn = StringUtils.replaceUnicodeSequences(line.substring(line.indexOf('[')+1, line.indexOf(']')).trim());
162                     section = getSection(sn, true).addHeaderComments(section.removeTrailingComments());
163                  } else {
164                     char c = line.isEmpty() ? 0 : line.charAt(0);
165                     if ((c == ' ' || c == '\t') && canAppend && ! (isComment(line) || isAssignment(line)))
166                        lines.add(lines.remove(lines.size()-1) + '\n' + line.substring(1));
167                     else {
168                        lines.add(line);
169                        if (isAssignment(line))
170                           canAppend = true;
171                        else
172                           canAppend = canAppend && ! (StringUtils.isEmpty(line) || isComment(line));
173                     }
174                  }
175               }
176               section.addLines(null, lines.toArray(new String[lines.size()]));
177               in.close();
178               if (hasBeenModified)  // Set when values need to be encoded.
179                  save();
180               if (file != null)
181                  modifiedTimestamp = file.lastModified();
182            } finally {
183               writeUnlock();
184            }
185         } finally {
186            in.close();
187         }
188      } finally {
189         writeUnlock();
190      }
191      for (ConfigFileListener l : listeners)
192         l.onLoad(this);
193      return this;
194   }

559   @Override /* ConfigFile */
560   public ConfigFileImpl save() throws IOException {
561      writeLock();
562      try {
563         if (file == null)
564            throw new UnsupportedOperationException("No backing file specified for config file.");
565         Writer out = new OutputStreamWriter(new FileOutputStream(file), charset);
566         try {
567            serializeTo(out);
568            hasBeenModified = false;
569            modifiedTimestamp = file.lastModified();
570         } finally {
571            out.close();
572         }
573         for (ConfigFileListener l : listeners)
574            l.onSave(this);
575         return this;
576      } finally {
577         writeUnlock();
578      }
579   }

581   @Override /* ConfigFile */
582   public ConfigFileImpl serializeTo(Writer out, ConfigFileFormat format) throws IOException {
583      readLock();
584      try {
585         PrintWriter pw = (out instanceof PrintWriter ? (PrintWriter)out : new PrintWriter(out));
586         for (Section s : sections.values())
587            s.writeTo(pw, format);
588         pw.flush();
589         pw.close();
590         out.close();
591      } finally {
592         readUnlock();
593      }
594      return this;
595   }

633   @Override /* ConfigFile */
634   public ConfigFile merge(ConfigFile cf) {
635      assertFieldNotNull(cf, "cf");
636      Set<String> changes = createChanges();
637      writeLock();
638      try {
639         for (String sectionName : this.keySet())
640            if (! cf.containsKey(sectionName))
641               remove(sectionName, changes);
642
643         for (Map.Entry<String,Section> e : cf.entrySet())
644            put(e.getKey(), e.getValue(), changes);
645
646      } finally {
647         writeUnlock();
648      }
649      signalChanges(changes);
650      return this;
651   }

709   /*
710    * Finds the keys that are different between the two sections and adds it to
711    * the specified set.
712    */
713   private void findChanges(Set<String> s, Section a, Section b) {
714      if (s == null)
715         return;
716      String sname = (a == null ? b.name : a.name);
717      if (a == null) {
718         for (String k : b.keySet())
719            s.add(getFullKey(sname, k));
720      } else if (b == null) {
721         for (String k : a.keySet())
722            s.add(getFullKey(sname, k));
723      } else {
724         for (String k : a.keySet())
725            addChange(s, sname, k, a.get(k), b.get(k));
726         for (String k : b.keySet())
727            addChange(s, sname, k, a.get(k), b.get(k));
728      }
729   }
730
731   private void addChange(Set<String> changes, String section, String key, String oldVal, String newVal) {
732      if (! StringUtils.isEquals(oldVal, newVal))
733         changes.add(getFullKey(section, key));
734   }

736   private Set<String> createChanges() {
737      return (listeners.size() > 0 ? new LinkedHashSet<String>() : null);
738   }

740   private void signalChanges(Set<String> changes) {
741      if (changes != null && ! changes.isEmpty())
742         for (ConfigFileListener l : listeners)
743            l.onChange(this, changes);
744   }
