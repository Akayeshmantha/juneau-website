087   public ConfigFileImpl(File file, boolean readOnly, Encoder encoder, WriterSerializer serializer, ReaderParser parser,
088         Charset charset) throws IOException {
089      this.file = file;
090      this.encoder = encoder == null ? XorEncoder.INSTANCE : encoder;
091      this.serializer = serializer == null ? JsonSerializer.DEFAULT : serializer;
092      this.parser = parser == null ? JsonParser.DEFAULT : parser;
093      this.charset = charset == null ? Charset.defaultCharset() : charset;
094      load();
095      this.readOnly = readOnly;
096      if (readOnly) {
097         this.sections = Collections.unmodifiableMap(this.sections);
098         for (Section s : sections.values())
099            s.setReadOnly();
100      }
101      this.pBeanSession = this.parser.createSession();
102   }

153   @Override /* ConfigFile */
154   public ConfigFileImpl load(Reader r) throws IOException {
155      assertFieldNotNull(r, "r");
156      writeLock();
157      try {
158         this.sections = Collections.synchronizedMap(new LinkedHashMap<String,Section>());
159         try (BufferedReader in = new BufferedReader(r)) {
160            writeLock();
161            hasBeenModified = false;
162            try {
163               sections.clear();
164               String line = null;
165               Section section = getSection(null, true);
166               ArrayList<String> lines = new ArrayList<>();
167               boolean canAppend = false;
168               while ((line = in.readLine()) != null) {
169                  if (isSection(line)) {
170                     section.addLines(null, lines.toArray(new String[lines.size()]));
171                     lines.clear();
172                     canAppend = false;
173                     String sn = replaceUnicodeSequences(line.substring(line.indexOf('[')+1, line.indexOf(']')).trim());
174                     section = getSection(sn, true).addHeaderComments(section.removeTrailingComments());
175                  } else {
176                     char c = line.isEmpty() ? 0 : line.charAt(0);
177                     if ((c == ' ' || c == '\t') && canAppend && ! (isComment(line) || isAssignment(line)))
178                        lines.add(lines.remove(lines.size()-1) + '\n' + line.substring(1));
179                     else {
180                        lines.add(line);
181                        if (isAssignment(line))
182                           canAppend = true;
183                        else
184                           canAppend = canAppend && ! (StringUtils.isEmpty(line) || isComment(line));
185                     }
186                  }
187               }
188               section.addLines(null, lines.toArray(new String[lines.size()]));
189               in.close();
190               if (hasBeenModified)  // Set when values need to be encoded.
191                  save();
192               if (file != null)
193                  modifiedTimestamp = file.lastModified();
194            } finally {
195               writeUnlock();
196            }
197         }
198      } finally {
199         writeUnlock();
200      }
201      for (ConfigFileListener l : listeners)
202         l.onLoad(this);
203      return this;
204   }

206   @Override /* ConfigFile */
207   protected String serialize(Object value, Serializer serializer, boolean newline) throws SerializeException {
208      if (value == null)
209         return "";
210      if (serializer == null)
211         serializer = this.serializer;
212      Class<?> c = value.getClass();
213      if (isSimpleType(c))
214         return value.toString();
215
216      String r = null;
217      if (newline)
218         r = "\n" + (String)serializer.serialize(value);
219      else
220         r = (String)serializer.serialize(value);
221
222      if (r.startsWith("'"))
223         return r.substring(1, r.length()-1);
224      return r;
225   }

227   @Override /* ConfigFile */
228   @SuppressWarnings({ "unchecked" })
229   protected <T> T parse(String s, Parser parser, Type type, Type...args) throws ParseException {
230
231      if (StringUtils.isEmpty(s))
232         return null;
233
234      if (isSimpleType(type))
235         return (T)pBeanSession.convertToType(s, (Class<?>)type);
236
237      char s1 = firstNonWhitespaceChar(s);
238      if (isArray(type) && s1 != '[')
239         s = '[' + s + ']';
240      else if (s1 != '[' && s1 != '{' && ! "null".equals(s))
241         s = '\'' + s + '\'';
242
243      if (parser == null)
244         parser = this.parser;
245
246      return parser.parse(s, type, args);
247   }

249   private static boolean isSimpleType(Type t) {
250      if (! (t instanceof Class))
251         return false;
252      Class<?> c = (Class<?>)t;
253      return (c == String.class || c.isPrimitive() || c.isAssignableFrom(Number.class) || c == Boolean.class || c.isEnum());
254   }
255
256   private static boolean isArray(Type t) {
257      if (! (t instanceof Class))
258         return false;
259      Class<?> c = (Class<?>)t;
260      return (c.isArray());
261   }

721         writeUnlock();
722      }
723      signalChanges(changes);

780   /*
781    * Finds the keys that are different between the two sections and adds it to
782    * the specified set.
783    */
784   static final void findChanges(Set<String> s, Section a, Section b) {
785      if (s == null)
786         return;
787      String sname = (a == null ? b.name : a.name);
788      if (a == null) {
789         for (String k : b.keySet())
790            s.add(getFullKey(sname, k));
791      } else if (b == null) {
792         for (String k : a.keySet())
793            s.add(getFullKey(sname, k));
794      } else {
795         for (String k : a.keySet())
796            addChange(s, sname, k, a.get(k), b.get(k));
797         for (String k : b.keySet())
798            addChange(s, sname, k, a.get(k), b.get(k));
799      }
800   }

802   static final void addChange(Set<String> changes, String section, String key, String oldVal, String newVal) {
803      if (! isEquals(oldVal, newVal))
804         changes.add(getFullKey(section, key));
805   }

807   final Set<String> createChanges() {
808      return (listeners.size() > 0 ? new LinkedHashSet<String>() : null);
809   }

811   final void signalChanges(Set<String> changes) {
812      if (changes != null && ! changes.isEmpty())
813         for (ConfigFileListener l : listeners)
814            l.onChange(this, changes);
815   }
