094   @Override /* Map */
095   public void clear() {
096      Set<String> changes = createChanges();
097      writeLock();
098      try {
099         if (changes != null)
100            for (String k : keySet())
101               changes.add(getFullKey(name, k));
102         entries.clear();
103         lines.clear();
104         headerComments.clear();
105      } finally {
106         writeUnlock();
107      }
108      signalChanges(changes);
109   }

230   /**
231    * Sets the specified value in this section.

241      signalChanges(changes);
242      return s;
243   }
244
245   String put(String key, String value, boolean encoded, Set<String> changes) {
246      writeLock();
247      try {
248         addLine(key, encoded);
249         String prev = entries.put(key, value);
250         addChange(changes, key, prev, value);
251         return prev;
252      } finally {
253         writeUnlock();
254      }
255   }

257   @Override /* Map */
258   public void putAll(Map<? extends String,? extends String> map) {
259      Set<String> changes = createChanges();
260      for (Map.Entry<? extends String,? extends String> e : map.entrySet())
261         put(e.getKey(), e.getValue(), false, changes);
262      signalChanges(changes);
263   }

265   @Override /* Map */
266   public String remove(Object key) {
267      Set<String> changes = createChanges();
268      String old = remove(key, changes);
269      signalChanges(changes);
270      return old;
271   }
272
273   String remove(Object key, Set<String> changes) {
274      writeLock();
275      try {
276         String prev = entries.remove(key);
277         addChange(changes, key.toString(), prev, null);
278         removeLine(key.toString());
279         return prev;
280      } finally {
281         writeUnlock();
282      }
283   }
284
285   void removeLine(String key) {
286      for (Iterator<String> i = lines.iterator(); i.hasNext();) {
287         String k = i.next();
288         if (k.startsWith("*") || k.startsWith(">")) {
289            if (k.substring(1).equals(key)) {
290               i.remove();
291               break;
292            }
293         }
294      }
295   }

312   /**
313    * Returns <jk>true</jk> if the specified entry is encoded.
314    * 
315    * @param key The key.
316    * @return <jk>true</jk> if the specified entry is encoded.
317    */
318   public boolean isEncoded(String key) {
319      readLock();
320      try {
321         for (String s : lines)
322            if (s.length() > 1)
323               if (s.substring(1).equals(key))
324                  return s.charAt(0) == '*';
325         return false;
326      } finally {
327         readUnlock();
328      }
329   }

331   /**
332    * Adds header comments to this section.
333    * 
334    * @see ConfigFile#addHeaderComments(String, String...) for a description.
335    * @param comments The comment lines to add to this section.
336    * @return This object (for method chaining).
337    */
338   public Section addHeaderComments(List<String> comments) {
339      writeLock();
340      try {
341         for (String c : comments) {
342            if (c == null)
343               c = "";
344            if (! c.startsWith("#"))
345               c = "#" + c;
346            this.headerComments.add(c);
347         }
348         return this;
349      } finally {
350         writeUnlock();
351      }
352   }
353
354   /**
355    * Removes all header comments from this section.
356    */
357   public void clearHeaderComments() {
358      writeLock();
359      try {
360         this.headerComments.clear();
361      } finally {
362         writeUnlock();
363      }
364   }

366   /**
367    * Serialize this section.
368    * 
369    * @param out What to serialize to.
370    * @param format The format (e.g. INI, BATCH, SHELL).
371    */
372   public void writeTo(PrintWriter out, ConfigFileFormat format) {
373      readLock();
374      try {
375         if (format == INI) {
376            for (String s : headerComments)
377               out.append(s).println();
378            if (! name.equals("default"))
379               out.append('[').append(name).append(']').println();
380            for (String l : lines) {
381               char c = (l.length() > 0 ? l.charAt(0) : 0);
382               if (c == '>' || c == '*'){
383                  boolean encode = c == '*';
384                  String key = l.substring(1);
385                  String val = entries.get(key);
386                  if (val.indexOf('\n') != -1)
387                     val = val.replaceAll("(\\r?\\n)", "$1\t");
388                  if (val.indexOf('=') != -1)
389                     val = val.replace("=", "\\u003D");
390                  if (val.indexOf('#') != -1)
391                     val = val.replace("#", "\\u0023");
392                  out.append(key);
393                  if (encode)
394                     out.append('*');
395                  out.append(" = ");
396                  if (encode)
397                     out.append('{').append(configFile.getEncoder().encode(key, val)).append('}');
398                  else
399                     out.append(val);
400                  out.println();
401               } else {
402                  out.append(l).println();
403               }
404            }
405
406         } else if (format == BATCH) {
407            String section = name.replaceAll("\\.\\/", "_");
408            for (String l : headerComments) {
409               l = trimComment(l);
410               if (! l.isEmpty())
411                  out.append("rem ").append(l);
412               out.println();
413            }
414            for (String l : lines) {
415               char c = (l.length() > 0 ? l.charAt(0) : 0);
416               if (c == '>' || c == '*') {
417                  String key = l.substring(1);
418                  String val = entries.get(key);
419                  out.append("set ");
420                  if (! name.equals("default"))
421                     out.append(section).append('_');
422                  out.append(key.replaceAll("\\.\\/", "_")).append(" = ").append(val).println();
423               } else {
424                  l = trimComment(l);
425                  if (! l.isEmpty())
426                     out.append("rem ").append(l);
427                  out.println();
428               }
429            }
430
431         } else if (format == SHELL) {
432            String section = name.replaceAll("\\.\\/", "_");
433            for (String l : headerComments) {
434               l = trimComment(l);
435               if (! l.isEmpty())
436                  out.append("# ").append(l);
437               out.println();
438            }
439            for (String l : lines) {
440               char c = (l.length() > 0 ? l.charAt(0) : 0);
441               if (c == '>' || c == '*'){
442                  String key = l.substring(1);
443                  String val = entries.get(key).replaceAll("\\\\", "\\\\\\\\");
444                  out.append("export ");
445                  if (! name.equals("default"))
446                     out.append(section).append('_');
447                  out.append(key.replaceAll("\\.\\/", "_")).append('=').append('"').append(val).append('"').println();
448               } else {
449                  l = trimComment(l);
450                  if (! l.isEmpty())
451                     out.append("# ").append(l);
452                  out.println();
453               }
454            }
455         }
456      } finally {
457         readUnlock();
458      }
459   }

466   /*
467    * Add lines to this section.
468    */
469   Section addLines(Set<String> changes, String...l) {
470      writeLock();
471      try {
472         if (l == null)
473            l = new String[0];
474         for (int i = 0; i < l.length; i++) {
475            String line = l[i];
476            if (line == null)
477               line = "";
478            if (isComment(line))
479               this.lines.add(line);
480            else if (isAssignment(line)) {
481               // Key/value pairs are stored as either ">key" or "*key";
482               String key = replaceUnicodeSequences(line.substring(0, line.indexOf('=')).trim());
483               String val = replaceUnicodeSequences(line.substring(line.indexOf('=')+1).trim());
484               boolean encoded = key.length() > 1 && key.endsWith("*");
485               if (encoded) {
486                  key = key.substring(0, key.lastIndexOf('*'));
487                  String v = val.toString().trim();
488                  if (v.startsWith("{") && v.endsWith("}"))
489                     val = configFile.getEncoder().decode(key, v.substring(1, v.length()-1));
490                  else
491                     configFile.setHasBeenModified();
492               }
493               if (containsKey(key)) {
494                  entries.remove(key);
495                  lines.remove('*' + key);
496                  lines.remove('>' + key);
497               }
498               lines.add((encoded ? '*' : '>') + key);
499               addChange(changes, key, entries.put(key, val), val);
500            } else {
501               this.lines.add(line);
502            }
503         }
504         return this;
505      } finally {
506         writeUnlock();
507      }
508   }

510   /*
511    * Remove all "#*" lines at the end of this section so they can
512    * be associated with the next section.
513    */
514   List<String> removeTrailingComments() {
515      LinkedList<String> l = new LinkedList<>();
516      while ((! lines.isEmpty()) && lines.getLast().startsWith("#"))
517         l.addFirst(lines.removeLast());
518      return l;
519   }

526   private void addLine(String key, boolean encoded) {
527      for (Iterator<String> i = lines.iterator(); i.hasNext();) {
528         String k = i.next();
529         if ((k.startsWith("*") || k.startsWith(">")) && k.substring(1).equals(key)) {
530            if (k.startsWith("*") && encoded || k.startsWith(">") && ! encoded)
531               return;
532            i.remove();
533         }
534      }
535      lines.add((encoded ? "*" : ">") + key);
536   }

538   void readLock() {
539      lock.readLock().lock();
540   }
541
542   void readUnlock() {
543      lock.readLock().unlock();
544   }
545
546   void writeLock() {
547      if (readOnly)
548         throw new UnsupportedOperationException("Cannot modify read-only ConfigFile.");
549      lock.writeLock().lock();
550   }
551
552   void writeUnlock() {
553      lock.writeLock().unlock();
554   }
555
556   private static String trimComment(String s) {
557      return s.replaceAll("^\\s*\\#\\s*", "").trim();
558   }
559
560   Set<String> createChanges() {
561      return (configFile != null && configFile.getListeners().size() > 0 ? new LinkedHashSet<String>() : null);
562   }
563
564   void signalChanges(Set<String> changes) {
565      if (changes != null && ! changes.isEmpty())
566         for (ConfigFileListener l : configFile.getListeners())
567            l.onChange(configFile, changes);
568   }
569
570   void addChange(Set<String> changes, String key, String oldVal, String newVal) {
571      if (changes != null)
572         if (! isEquals(oldVal, newVal))
573            changes.add(getFullKey(name, key));
574   }
