013package org.apache.juneau.transform;
014
015import static org.apache.juneau.internal.ClassUtils.*;
016
017import java.util.*;
018
019import org.apache.juneau.*;
020import org.apache.juneau.parser.*;
021import org.apache.juneau.serializer.*;
022
023/**
024 * Used to swap out non-serializable objects with serializable replacements during serialization, and vis-versa during parsing.
025 *
026 * <h5 class='section'>Description:</h5>
027 *
028 * <code>PojoSwaps</code> are used to extend the functionality of the serializers and parsers to be able to handle
029 * POJOs that aren't automatically handled by the serializers or parsers.  For example, JSON does not have a standard
030 * representation for rendering dates.
031 * By defining a special {@code Date} swap and associating it with a serializer and parser, you can convert a
032 * {@code Date} object to a {@code String} during serialization, and convert that {@code String} object back into a
033 * {@code Date} object during parsing.
034 *
035 * <p>
036 * Swaps MUST declare a public no-arg constructor so that the bean context can instantiate them.
037 *
038 * <p>
039 * <code>PojoSwaps</code> are associated with instances of {@link BeanContext BeanContexts} by passing the swap
040 * class to the {@link CoreObjectBuilder#pojoSwaps(Class...)} method.
041 * <br>When associated with a bean context, fields of the specified type will automatically be converted when the
042 * {@link BeanMap#get(Object)} or {@link BeanMap#put(String, Object)} methods are called.
043 *
044 * <p>
045 * <code>PojoSwaps</code> have two parameters:
046 * <ol>
047 *    <li>{@code <T>} - The normal representation of an object.
048 *    <li>{@code <S>} - The swapped representation of an object.
049 * </ol>
050 * <br>{@link Serializer Serializers} use swaps to convert objects of type T into objects of type S, and on calls to
051 * {@link BeanMap#get(Object)}.
052 * <br>{@link Parser Parsers} use swaps to convert objects of type S into objects of type T, and on calls to
053 * {@link BeanMap#put(String,Object)}.
054 *
055 * <h6 class='topic'>Subtypes</h6>
056 *
057 * The following abstract subclasses are provided for common swap types:
058 * <ol>
059 *    <li>{@link StringSwap} - Objects swapped with strings.
060 *    <li>{@link MapSwap} - Objects swapped with {@link ObjectMap ObjectMaps}.
061 * </ol>
062 *
063 * <h6 class='topic'>Localization</h6>
064 *
065 * Swaps have access to the session locale and timezone through the {@link BeanSession#getLocale()} and
066 * {@link BeanSession#getTimeZone()} methods.
067 * This allows you to specify localized swap values when needed.
068 * If using the REST server API, the locale and timezone are set based on the <code>Accept-Language</code> and
069 * <code>Time-Zone</code> headers on the request.
070 *
071 * <h6 class='topic'>Swap Class Type {@code <S>}</h6>
072 *
073 * The swapped object representation of an object must be an object type that the serializers can natively convert to
074 * JSON (or language-specific equivalent).
075 * The list of valid transformed types are as follows...
076 * <ul class='spaced-list'>
077 *    <li>
078 *       {@link String}
079 *    <li>
080 *       {@link Number}
081 *    <li>
082 *       {@link Boolean}
083 *    <li>
084 *       {@link Collection} containing anything on this list.
085 *    <li>
086 *       {@link Map} containing anything on this list.
087 *    <li>
088 *       A java bean with properties of anything on this list.
089 *    <li>
090 *       An array of anything on this list.
091 * </ul>
092 *
093 * <h6 class='topic'>Normal Class Type {@code <T>}</h6>
094 *
095 * The normal object representation of an object.
096 *
097 * <h6 class='topic'>One-way vs. Two-way Serialization</h6>
098 *
099 * Note that while there is a unified interface for handling swaps during both serialization and parsing,
100 * in many cases only one of the {@link #swap(BeanSession, Object)} or {@link #unswap(BeanSession, Object, ClassMeta)}
101 * methods will be defined because the swap is one-way.
102 * For example, a swap may be defined to convert an {@code Iterator} to a {@code ObjectList}, but
103 * it's not possible to unswap an {@code Iterator}.
104 * In that case, the {@code swap(Object}} method would be implemented, but the {@code unswap(ObjectMap)} object would
105 * not, and the swap would be associated on the serializer, but not the parser.
106 * Also, you may choose to serialize objects like {@code Dates} to readable {@code Strings}, in which case it's not
107 * possible to re-parse it back into a {@code Date}, since there is no way for the {@code Parser} to know it's a
108 * {@code Date} from just the JSON or XML text.
109 *
110 * <h5 class='section'>Additional information:</h5>
111 *
112 * See <a class='doclink' href='package-summary.html#TOC'>org.apache.juneau.transform</a> for more information.
113 *
114 * @param <T> The normal form of the class.
115 * @param <S> The swapped form of the class.
116 */
117public abstract class PojoSwap<T,S> {
118
119   /**
120    * Represents a non-existent pojo swap.
121    */
122   @SuppressWarnings({ "unchecked", "rawtypes" })
123   public final static PojoSwap NULL = new PojoSwap(null, null) {};
124
125   private final Class<T> normalClass;
126   private final Class<?> swapClass;
127   private ClassMeta<?> swapClassMeta;
128
129   /**
130    * Constructor.
131    */
132   @SuppressWarnings("unchecked")
133   protected PojoSwap() {
134      normalClass = (Class<T>)resolveParameterType(PojoSwap.class, 0, this.getClass());
135      swapClass = resolveParameterType(PojoSwap.class, 1, this.getClass());
136   }
137
138   /**
139    * Constructor for when the normal and transformed classes are already known.
140    *
141    * @param normalClass The normal class (cannot be serialized).
142    * @param swapClass The transformed class (serializable).
143    */
144   protected PojoSwap(Class<T> normalClass, Class<?> swapClass) {
145      this.normalClass = normalClass;
146      this.swapClass = swapClass;
147   }
148
149   /**
150    * If this transform is to be used to serialize non-serializable POJOs, it must implement this method.
151    *
152    * <p>
153    * The object must be converted into one of the following serializable types:
154    * <ul class='spaced-list'>
155    *    <li>
156    *       {@link String}
157    *    <li>
158    *       {@link Number}
159    *    <li>
160    *       {@link Boolean}
161    *    <li>
162    *       {@link Collection} containing anything on this list.
163    *    <li>
164    *       {@link Map} containing anything on this list.
165    *    <li>
166    *       A java bean with properties of anything on this list.
167    *    <li>
168    *       An array of anything on this list.
169    * </ul>
170    *
171    * @param session
172    *    The bean session to use to get the class meta.
173    *    This is always going to be the same bean context that created this swap.
174    * @param o The object to be transformed.
175    * @return The transformed object.
176    * @throws Exception If a problem occurred trying to convert the output.
177    */
178   public S swap(BeanSession session, T o) throws Exception {
179      throw new SerializeException("Swap method not implemented on PojoSwap ''{0}''", this.getClass().getName());
180   }
181
182   /**
183    * If this transform is to be used to reconstitute POJOs that aren't true Java beans, it must implement this method.
184    *
185    * @param session
186    *    The bean session to use to get the class meta.
187    *    This is always going to be the same bean context that created this swap.
188    * @param f The transformed object.
189    * @param hint
190    *    If possible, the parser will try to tell you the object type being created.
191    *    For example, on a serialized date, this may tell you that the object being created must be of type
192    *    {@code GregorianCalendar}.
193    *    <br>This may be <jk>null</jk> if the parser cannot make this determination.
194    * @return The narrowed object.
195    * @throws Exception If this method is not implemented.
196    */
197   public T unswap(BeanSession session, S f, ClassMeta<?> hint) throws Exception {
198      throw new ParseException("Unswap method not implemented on PojoSwap ''{0}''", this.getClass().getName());
199   }
200
201   /**
202    * Returns the T class, the normalized form of the class.
203    *
204    * @return The normal form of this class.
205    */
206   public Class<T> getNormalClass() {
207      return normalClass;
208   }
209
210   /**
211    * Returns the G class, the generalized form of the class.
212    *
213    * <p>
214    * Subclasses must override this method if the generalized class is {@code Object}, meaning it can produce multiple
215    * generalized forms.
216    *
217    * @return The transformed form of this class.
218    */
219   public Class<?> getSwapClass() {
220      return swapClass;
221   }
222
223   /**
224    * Returns the {@link ClassMeta} of the transformed class type.
225    *
226    * <p>
227    * This value is cached for quick lookup.
228    *
229    * @param beanContext
230    *    The bean context to use to get the class meta.
231    *    This is always going to be the same bean context that created this swap.
232    * @return The {@link ClassMeta} of the transformed class type.
233    */
234   public ClassMeta<?> getSwapClassMeta(BeanContext beanContext) {
235      if (swapClassMeta == null)
236         swapClassMeta = beanContext.getClassMeta(swapClass);
237      return swapClassMeta;
238   }
239
240   /**
241    * Checks if the specified object is an instance of the normal class defined on this swap.
242    *
243    * @param o The object to check.
244    * @return
245    *    <jk>true</jk> if the specified object is a subclass of the normal class defined on this transform.
246    *    <jk>null</jk> always return <jk>false</jk>.
247    */
248   public boolean isNormalObject(Object o) {
249      if (o == null)
250         return false;
251      return isParentClass(normalClass, o.getClass());
252   }
253
254   /**
255    * Checks if the specified object is an instance of the swap class defined on this swap.
256    *
257    * @param o The object to check.
258    * @return
259    *    <jk>true</jk> if the specified object is a subclass of the transformed class defined on this transform.
260    *    <jk>null</jk> always return <jk>false</jk>.
261    */
262   public boolean isSwappedObject(Object o) {
263      if (o == null)
264         return false;
265      return isParentClass(swapClass, o.getClass());
266   }
267
268
269   //--------------------------------------------------------------------------------
270   // Overridden methods
271   //--------------------------------------------------------------------------------
272
273   @Override /* Object */
274   public String toString() {
275      return getClass().getSimpleName() + '<' + getNormalClass().getSimpleName() + "," + getSwapClass().getSimpleName() + '>';
276   }
277}
