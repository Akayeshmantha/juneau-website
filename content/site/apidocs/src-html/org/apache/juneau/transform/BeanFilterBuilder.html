013package org.apache.juneau.transform;
014
015import static org.apache.juneau.internal.ClassUtils.*;
016
017import java.beans.*;
018import java.util.*;
019
020import org.apache.juneau.*;
021
022/**
023 * Builder class for {@link BeanFilter} objects.
024 *
025 * <p>
026 * Bean filter builders must have a public no-arg constructor.
027 * Builder settings should be set in the constructor using the provided setters on this class.
028 *
029 * <h5 class='section'>Example:</h5>
030 * <p class='bcode'>
031 *    <jc>// Create our serializer with a bean filter.</jc>
032 *    WriterSerializer s = <jk>new</jk> JsonSerializerBuilder().beanFilters(AddressFilter.<jk>class</jk>).build();
033 *
034 *    Address a = <jk>new</jk> Address();
035 *    String json = s.serialize(a); <jc>// Serializes only street, city, state.</jc>
036 *
037 *    <jc>// Filter class defined via setters</jc>
038 *    <jk>public class</jk> AddressFilter <jk>extends</jk> BeanFilterBuilder {
039 *       <jk>public</jk> AddressFilter() {
040 *          super(Address.<jk>class</jk>);
041 *          setProperties(<js>"street"</js>,<js>"city"</js>,<js>"state"</js>);
042 *       }
043 *    }
044 * </p>
045 *
046 * <h5 class='section'>Additional information:</h5>
047 * See <a class='doclink' href='package-summary.html#TOC'>org.apache.juneau.transform</a> for more information.
048 */
049@SuppressWarnings("hiding")
050public abstract class BeanFilterBuilder {
051
052   Class<?> beanClass;
053   String typeName;
054   String[] properties, excludeProperties;
055   Class<?> interfaceClass, stopClass;
056   boolean sortProperties;
057   PropertyNamer propertyNamer;
058   List<Class<?>> beanDictionary;
059
060   /**
061    * Constructor.
062    *
063    * @param beanClass The bean class that this filter applies to.
064    */
065   public BeanFilterBuilder(Class<?> beanClass) {
066      this.beanClass = beanClass;
067   }
068
069   /**
070    * Specifies the type name for this bean.
071    *
072    * @param typeName The dictionary name associated with this bean.
073    * @return This object (for method chaining).
074    */
075   public BeanFilterBuilder typeName(String typeName) {
076      this.typeName = typeName;
077      return this;
078   }
079
080   /**
081    * Specifies the set and order of names of properties associated with the bean class.
082    *
083    * <p>
084    * The order specified is the same order that the entries will be returned by the {@link BeanMap#entrySet()} and
085    * related methods.
086    * Entries in the list can also contain comma-delimited lists that will be split.
087    *
088    * @param properties The properties associated with the bean class.
089    * @return This object (for method chaining).
090    */
091   public BeanFilterBuilder properties(String...properties) {
092      this.properties = properties;
093      return this;
094   }
095
096   /**
097    * Specifies the list of properties to ignore on a bean.
098    *
099    * @param excludeProperties The list of properties to ignore on a bean.
100    * @return This object (for method chaining).
101    */
102   public BeanFilterBuilder excludeProperties(String...excludeProperties) {
103      this.excludeProperties = excludeProperties;
104      return this;
105   }
106
107   /**
108    * Identifies a class to be used as the interface class for this and all subclasses.
109    *
110    * <p>
111    * When specified, only the list of properties defined on the interface class will be used during serialization.
112    * Additional properties on subclasses will be ignored.
113    *
114    * <p class='bcode'>
115    *    <jc>// Parent class</jc>
116    *    <jk>public abstract class</jk> A {
117    *       <jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
118    *    }
119    *
120    *    <jc>// Sub class</jc>
121    *    <jk>public class</jk> A1 <jk>extends</jk> A {
122    *       <jk>public</jk> String <jf>f1</jf> = <js>"f1"</js>;
123    *    }
124    *
125    *    <jc>// Filter class</jc>
126    *    <jk>public class</jk> AFilter <jk>extends</jk> BeanFilterBuilder {
127    *       <jk>public</jk> AFilter() {
128    *          super(A.<jk>class</jk>);
129    *          setInterfaceClass(A.<jk>class</jk>);
130    *       }
131    *    }
132    *
133    *    JsonSerializer s = new JsonSerializerBuilder().beanFilters(AFilter.<jk>class</jk>).build();
134    *    A1 a1 = <jk>new</jk> A1();
135    *    String r = s.serialize(a1);
136    *    <jsm>assertEquals</jsm>(<js>"{f0:'f0'}"</js>, r);  <jc>// Note f1 is not serialized</jc>
137    * </p>
138    *
139    * <p>
140    * Note that this filter can be used on the parent class so that it filters to all child classes, or can be set
141    * individually on the child classes.
142    *
143    * @param interfaceClass The interface class to use for this bean class.
144    * @return This object (for method chaining).
145    */
146   public BeanFilterBuilder interfaceClass(Class<?> interfaceClass) {
147      this.interfaceClass = interfaceClass;
148      return this;
149   }
150
151   /**
152    * Identifies a stop class for this class and all subclasses.
153    *
154    * <p>
155    * Identical in purpose to the stop class specified by {@link Introspector#getBeanInfo(Class, Class)}.
156    * Any properties in the stop class or in its base classes will be ignored during analysis.
157    *
158    * <p>
159    * For example, in the following class hierarchy, instances of <code>C3</code> will include property <code>p3</code>,
160    * but not <code>p1</code> or <code>p2</code>.
161    *
162    * <p class='bcode'>
163    *    <jk>public class</jk> C1 {
164    *       <jk>public int</jk> getP1();
165    *    }
166    *
167    *    <jk>public class</jk> C2 <jk>extends</jk> C1 {
168    *       <jk>public int</jk> getP2();
169    *    }
170    *
171    *    <ja>@Bean</ja>(stopClass=C2.<jk>class</jk>)
172    *    <jk>public class</jk> C3 <jk>extends</jk> C2 {
173    *       <jk>public int</jk> getP3();
174    *    }
175    * </p>
176    *
177    * @param stopClass
178    * @return This object (for method chaining).
179    */
180   public BeanFilterBuilder stopClass(Class<?> stopClass) {
181      this.stopClass = stopClass;
182      return this;
183   }
184
185   /**
186    * Sort properties in alphabetical order.
187    *
188    * @param sortProperties
189    * @return This object (for method chaining).
190    */
191   public BeanFilterBuilder sortProperties(boolean sortProperties) {
192      this.sortProperties = sortProperties;
193      return this;
194   }
195
196   /**
197    * The property namer to use to name bean properties.
198    *
199    * @param propertyNamer The property namer instance.
200    * @return This object (for method chaining).
201    */
202   public BeanFilterBuilder propertyNamer(PropertyNamer propertyNamer) {
203      this.propertyNamer = propertyNamer;
204      return this;
205   }
206
207   /**
208    * The property namer to use to name bean properties.
209    *
210    * @param c The property namer class.  Must have a public no-arg constructor.
211    * @return This object (for method chaining).
212    * @throws Exception Thrown from constructor method.
213    */
214   public BeanFilterBuilder propertyNamer(Class<? extends PropertyNamer> c) throws Exception {
215      this.propertyNamer = newInstance(PropertyNamer.class, c);
216      return this;
217   }
218
219   /**
220    * Sets the contents of this bean's bean dictionary.
221    *
222    * @param c The classes to set on this bean's bean dictionary.
223    * @return This object (for method chaining).
224    */
225   public BeanFilterBuilder setBeanDictionary(Class<?>...c) {
226      beanDictionary = new ArrayList<Class<?>>(Arrays.asList(c));
227      return this;
228   }
229
230   /**
231    * Adds classes to this bean's bean dictionary.
232    *
233    * @param c The classes to add to this bean's bean dictionary.
234    * @return This object (for method chaining).
235    */
236   public BeanFilterBuilder beanDictionary(Class<?>...c) {
237      if (beanDictionary == null)
238         beanDictionary = new ArrayList<Class<?>>(Arrays.asList(c));
239      else for (Class<?> cc : c)
240         beanDictionary.add(cc);
241      return this;
242   }
243
244   /**
245    * Creates a {@link BeanFilter} with settings in this builder class.
246    *
247    * @return A new {@link BeanFilter} instance.
248    */
249   public BeanFilter build() {
250      return new BeanFilter(this);
251   }
252}
