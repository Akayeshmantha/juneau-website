013package org.apache.juneau.urlencoding;
014
015import java.lang.reflect.*;
016import java.util.*;
017
018import org.apache.juneau.*;
019import org.apache.juneau.http.*;
020import org.apache.juneau.json.*;
021import org.apache.juneau.uon.*;
022
023/**
024 * Session object that lives for the duration of a single use of {@link UrlEncodingSerializer}.
025 *
026 * <p>
027 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
028 */
029public class UrlEncodingSerializerSession extends UonSerializerSession {
030
031   private final boolean expandedParams;
032
033   /**
034    * Create a new session using properties specified in the context.
035    *
036    * @param ctx
037    *    The context creating this session object.
038    *    The context contains all the configuration settings for this object.
039    * @param encode Overrides the {@link UonSerializerContext#UON_encodeChars} setting.
040    * @param output The output object.  See {@link JsonSerializerSession#getWriter()} for valid class types.
041    * @param op
042    *    The override properties.
043    *    These override any context properties defined in the context.
044    * @param javaMethod The java method that called this serializer, usually the method in a REST servlet.
045    * @param locale
046    *    The session locale.
047    *    If <jk>null</jk>, then the locale defined on the context is used.
048    * @param timeZone
049    *    The session timezone.
050    *    If <jk>null</jk>, then the timezone defined on the context is used.
051    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
052    * @param uriContext
053    *    The URI context.
054    *    Identifies the current request URI used for resolution of URIs to absolute or root-relative form.
055    */
056   public UrlEncodingSerializerSession(UrlEncodingSerializerContext ctx, Boolean encode, ObjectMap op, Object output,
057         Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType, UriContext uriContext) {
058      super(ctx, encode, op, output, javaMethod, locale, timeZone, mediaType, uriContext);
059      if (op == null || op.isEmpty()) {
060         expandedParams = ctx.expandedParams;
061      } else {
062         expandedParams = op.getBoolean(UrlEncodingContext.URLENC_expandedParams, false);
063      }
064   }
065
066   /**
067    * Returns <jk>true</jk> if the specified bean property should be expanded as multiple key-value pairs.
068    *
069    * @param pMeta The metadata on the bean property.
070    * @return <jk>true</jk> if the specified bean property should be expanded as multiple key-value pairs.
071    */
072   public final boolean shouldUseExpandedParams(BeanPropertyMeta pMeta) {
073      ClassMeta<?> cm = pMeta.getClassMeta();
074      if (cm.isCollectionOrArray()) {
075         if (expandedParams)
076            return true;
077         if (pMeta.getBeanMeta().getClassMeta().getExtendedMeta(UrlEncodingClassMeta.class).isExpandedParams())
078            return true;
079      }
080      return false;
081   }
082
083   /**
084    * Returns <jk>true</jk> if the specified value should be represented as an expanded parameter list.
085    *
086    * @param value The value to check.
087    * @return <jk>true</jk> if the specified value should be represented as an expanded parameter list.
088    */
089   public final boolean shouldUseExpandedParams(Object value) {
090      if (value == null || ! expandedParams)
091         return false;
092      ClassMeta<?> cm = getClassMetaForObject(value).getSerializedClassMeta();
093      if (cm.isCollectionOrArray()) {
094         if (expandedParams)
095            return true;
096      }
097      return false;
098   }
099}
