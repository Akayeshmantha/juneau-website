013package org.apache.juneau.urlencoding;
014
015import org.apache.juneau.*;
016import org.apache.juneau.serializer.*;
017
018/**
019 * Configurable properties on the {@link UonSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * See {@link ContextFactory} for more information about context properties.
025 *
026 * <h5 class='section'>Inherited configurable properties:</h5>
027 * <ul class='javahierarchy'>
028 *    <li class='c'><a class="doclink" href="../BeanContext.html#ConfigProperties">BeanContext</a> - Properties associated with handling beans on serializers and parsers.
029 *    <ul>
030 *       <li class='c'><a class="doclink" href="../serializer/SerializerContext.html#ConfigProperties">SerializerContext</a> - Configurable properties common to all serializers.
031 *    </ul>
032 * </ul>
033 */
034public class UonSerializerContext extends SerializerContext {
035
036   /**
037    * <b>Configuration property:</b>  Encode non-valid URI characters.
038    * <p>
039    * <ul>
040    *    <li><b>Name:</b> <js>"UonSerializer.encodeChars"</js>
041    *    <li><b>Data type:</b> <code>Boolean</code>
042    *    <li><b>Default:</b> <jk>false</jk> for {@link UonSerializer}, <jk>true</jk> for {@link UrlEncodingSerializer}
043    *    <li><b>Session-overridable:</b> <jk>true</jk>
044    * </ul>
045    * <p>
046    * Encode non-valid URI characters with <js>"%xx"</js> constructs.
047    * <p>
048    * If <jk>true</jk>, non-valid URI characters will be converted to <js>"%xx"</js> sequences.
049    * Set to <jk>false</jk> if parameter value is being passed to some other code that will already
050    *    perform URL-encoding of non-valid URI characters.
051    */
052   public static final String UON_encodeChars = "UonSerializer.encodeChars";
053
054   /**
055    * <b>Configuration property:</b>  Add <js>"_type"</js> properties when needed.
056    * <p>
057    * <ul>
058    *    <li><b>Name:</b> <js>"UonSerializer.addBeanTypeProperties"</js>
059    *    <li><b>Data type:</b> <code>Boolean</code>
060    *    <li><b>Default:</b> <jk>false</jk>
061    *    <li><b>Session-overridable:</b> <jk>true</jk>
062    * </ul>
063    * <p>
064    * If <jk>true</jk>, then <js>"_type"</js> properties will be added to beans if their type cannot be inferred through reflection.
065    * This is used to recreate the correct objects during parsing if the object types cannot be inferred.
066    * For example, when serializing a {@code Map<String,Object>} field, where the bean class cannot be determined from the value type.
067    * <p>
068    * When present, this value overrides the {@link SerializerContext#SERIALIZER_addBeanTypeProperties} setting and is
069    * provided to customize the behavior of specific serializers in a {@link SerializerGroup}.
070    */
071   public static final String UON_addBeanTypeProperties = "UonSerializer.addBeanTypeProperties";
072
073
074   final boolean
075      encodeChars,
076      addBeanTypeProperties;
077
078   /**
079    * Constructor.
080    * <p>
081    * Typically only called from {@link ContextFactory#getContext(Class)}.
082    *
083    * @param cf The factory that created this context.
084    */
085   public UonSerializerContext(ContextFactory cf) {
086      super(cf);
087      encodeChars = cf.getProperty(UON_encodeChars, boolean.class, false);
088      addBeanTypeProperties = cf.getProperty(UON_addBeanTypeProperties, boolean.class, cf.getProperty(SERIALIZER_addBeanTypeProperties, boolean.class, true));
089   }
090
091   @Override /* Context */
092   public ObjectMap asMap() {
093      return super.asMap()
094         .append("UonSerializerContext", new ObjectMap()
095            .append("encodeChars", encodeChars)
096            .append("addBeanTypeProperties", addBeanTypeProperties)
097         );
098   }
099}
