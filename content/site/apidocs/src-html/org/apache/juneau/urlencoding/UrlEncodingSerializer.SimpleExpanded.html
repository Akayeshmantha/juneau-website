028/**
029 * Serializes POJO models to URL-encoded notation with UON-encoded values (a notation for URL-encoded query paramter values).
030 *
031 * <h5 class='section'>Media types:</h5>
032 * <p>
033 *    Handles <code>Accept</code> types: <code>application/x-www-form-urlencoded</code>
034 * <p>
035 *    Produces <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
036 *
037 * <h5 class='section'>Description:</h5>
038 * <p>
039 *    This serializer provides several serialization options.  Typically, one of the predefined DEFAULT serializers will be sufficient.
040 *    However, custom serializers can be constructed to fine-tune behavior.
041 *
042 * <h5 class='section'>Configurable properties:</h5>
043 * <p>
044 *    This class has the following properties associated with it:
045 * <ul>
046 *    <li>{@link UonSerializerContext}
047 *    <li>{@link BeanContext}
048 * </ul>
049 * <p>
050 *    The following shows a sample object defined in Javascript:
051 * </p>
052 * <p class='bcode'>
053 *    {
054 *       id: 1,
055 *       name: <js>'John Smith'</js>,
056 *       uri: <js>'http://sample/addressBook/person/1'</js>,
057 *       addressBookUri: <js>'http://sample/addressBook'</js>,
058 *       birthDate: <js>'1946-08-12T00:00:00Z'</js>,
059 *       otherIds: <jk>null</jk>,
060 *       addresses: [
061 *          {
062 *             uri: <js>'http://sample/addressBook/address/1'</js>,
063 *             personUri: <js>'http://sample/addressBook/person/1'</js>,
064 *             id: 1,
065 *             street: <js>'100 Main Street'</js>,
066 *             city: <js>'Anywhereville'</js>,
067 *             state: <js>'NY'</js>,
068 *             zip: 12345,
069 *             isCurrent: <jk>true</jk>,
070 *          }
071 *       ]
072 *    }
073 * </p>
074 * <p>
075 *    Using the "strict" syntax defined in this document, the equivalent
076 *       URL-encoded notation would be as follows:
077 * </p>
078 * <p class='bcode'>
079 *    <xa>id</xa>=$n(<xs>1</xs>)
080 *    &amp;<xa>name</xa>=<xs>John+Smith</xs>,
081 *    &amp;<xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
082 *    &amp;<xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
083 *    &amp;<xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
084 *    &amp;<xa>otherIds</xa>=<xs>%00</xs>,
085 *    &amp;<xa>addresses</xa>=$a(
086 *       $o(
087 *          <xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
088 *          <xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
089 *          <xa>id</xa>=$n(<xs>1</xs>),
090 *          <xa>street</xa>=<xs>100+Main+Street</xs>,
091 *          <xa>city</xa>=<xs>Anywhereville</xs>,
092 *          <xa>state</xa>=<xs>NY</xs>,
093 *          <xa>zip</xa>=$n(<xs>12345</xs>),
094 *          <xa>isCurrent</xa>=$b(<xs>true</xs>)
095 *       )
096 *    )
097 * </p>
098 * <p>
099 *    A secondary "lax" syntax is available when the data type of the
100 *       values are already known on the receiving end of the transmission:
101 * </p>
102 * <p class='bcode'>
103 *    <xa>id</xa>=<xs>1</xs>,
104 *    &amp;<xa>name</xa>=<xs>John+Smith</xs>,
105 *    &amp;<xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
106 *    &amp;<xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
107 *    &amp;<xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
108 *    &amp;<xa>otherIds</xa>=<xs>%00</xs>,
109 *    &amp;<xa>addresses</xa>=(
110 *       (
111 *          <xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
112 *          <xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
113 *          <xa>id</xa>=<xs>1</xs>,
114 *          <xa>street</xa>=<xs>100+Main+Street</xs>,
115 *          <xa>city</xa>=<xs>Anywhereville</xs>,
116 *          <xa>state</xa>=<xs>NY</xs>,
117 *          <xa>zip</xa>=<xs>12345</xs>,
118 *          <xa>isCurrent</xa>=<xs>true</xs>
119 *       )
120 *    )
121 * </p>
122 *
123 * <h5 class='section'>Example:</h5>
124 * <p class='bcode'>
125 *    <jc>// Serialize a Map</jc>
126 *    Map m = <jk>new</jk> ObjectMap(<js>"{a:'b',c:1,d:false,e:['f',1,false],g:{h:'i'}}"</js>);
127 *
128 *    <jc>// Serialize to value equivalent to JSON.</jc>
129 *    <jc>// Produces "a=b&amp;c=$n(1)&amp;d=$b(false)&amp;e=$a(f,$n(1),$b(false))&amp;g=$o(h=i)"</jc>
130 *    String s = UrlEncodingSerializer.<jsf>DEFAULT</jsf>.serialize(s);
131 *
132 *    <jc>// Serialize to simplified value (for when data type is already known by receiver).</jc>
133 *    <jc>// Produces "a=b&amp;c=1&amp;d=false&amp;e=(f,1,false)&amp;g=(h=i))"</jc>
134 *    String s = UrlEncodingSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
135 *
136 *    <jc>// Serialize a bean</jc>
137 *    <jk>public class</jk> Person {
138 *       <jk>public</jk> Person(String s);
139 *       <jk>public</jk> String getName();
140 *       <jk>public int</jk> getAge();
141 *       <jk>public</jk> Address getAddress();
142 *       <jk>public boolean</jk> deceased;
143 *    }
144 *
145 *    <jk>public class</jk> Address {
146 *       <jk>public</jk> String getStreet();
147 *       <jk>public</jk> String getCity();
148 *       <jk>public</jk> String getState();
149 *       <jk>public int</jk> getZip();
150 *    }
151 *
152 *    Person p = <jk>new</jk> Person(<js>"John Doe"</js>, 23, <js>"123 Main St"</js>, <js>"Anywhere"</js>, <js>"NY"</js>, 12345, <jk>false</jk>);
153 *
154 *    <jc>// Produces "name=John+Doe&amp;age=23&amp;address=$o(street=123+Main+St,city=Anywhere,state=NY,zip=$n(12345))&amp;deceased=$b(false)"</jc>
155 *    String s = UrlEncodingSerializer.<jsf>DEFAULT</jsf>.serialize(s);
156 *
157 *    <jc>// Produces "name=John+Doe&amp;age=23&amp;address=(street=123+Main+St,city=Anywhere,state=NY,zip=12345)&amp;deceased=false)"</jc>
158 *    String s = UrlEncodingSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
159 * </p>
160 */

216   /**
217    * Workhorse method. Determines the type of object, and then calls the
218    * appropriate type-specific serialization method.
219    */
220   @SuppressWarnings({ "rawtypes", "unchecked" })
221   private SerializerWriter serializeAnything(UrlEncodingSerializerSession session, UonWriter out, Object o) throws Exception {
222
223      boolean addTypeProperty;      // Add "_type" attribute to element?
224      ClassMeta<?> aType;        // The actual type
225      ClassMeta<?> sType;        // The serialized type
226
227      aType = session.push("root", o, object());
228      session.indent--;
229      if (aType == null)
230         aType = object();
231
232      sType = aType.getSerializedClassMeta();
233      addTypeProperty = (session.isAddBeanTypeProperties());
234
235      // Swap if necessary
236      PojoSwap swap = aType.getPojoSwap();
237      if (swap != null) {
238         o = swap.swap(session, o);
239
240         // If the getSwapClass() method returns Object, we need to figure out
241         // the actual type now.
242         if (sType.isObject())
243            sType = session.getClassMetaForObject(o);
244      }
245
246      if (sType.isMap()) {
247         if (o instanceof BeanMap)
248            serializeBeanMap(session, out, (BeanMap)o, addTypeProperty);
249         else
250            serializeMap(session, out, (Map)o, sType);
251      } else if (sType.isBean()) {
252         serializeBeanMap(session, out, session.toBeanMap(o), addTypeProperty);
253      } else if (sType.isCollection()) {
254         serializeMap(session, out, getCollectionMap((Collection)o), session.getClassMeta(Map.class, Integer.class, sType.getElementType()));
255      } else {
256         // All other types can't be serialized as key/value pairs, so we create a
257         // mock key/value pair with a "_value" key.
258         out.append("_value=");
259         super.serializeAnything(session, out, o, null, null, null, false, true);
260      }
261
262      session.pop();
263      return out;
264   }
265
266   private Map<Integer,Object> getCollectionMap(Collection<?> c) {
267      Map<Integer,Object> m = new TreeMap<Integer,Object>();
268      int i = 0;
269      for (Object o : c)
270         m.put(i++, o);
271      return m;
272   }

274   @SuppressWarnings({ "rawtypes", "unchecked" })
275   private SerializerWriter serializeMap(UrlEncodingSerializerSession session, UonWriter out, Map m, ClassMeta<?> type) throws Exception {
276
277      m = session.sort(m);
278
279      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
280
281      int depth = session.getIndent();
282      boolean addAmp = false;
283
284      Iterator mapEntries = m.entrySet().iterator();
285
286      while (mapEntries.hasNext()) {
287         Map.Entry e = (Map.Entry) mapEntries.next();
288         Object value = e.getValue();
289         Object key = session.generalize(e.getKey(), keyType);
290
291
292         if (session.shouldUseExpandedParams(value)) {
293            Iterator i = value instanceof Collection ? ((Collection)value).iterator() : ArrayUtils.iterator(value);
294            while (i.hasNext()) {
295               if (addAmp)
296                  out.cr(depth).append('&');
297               out.appendObject(key, false, true, true).append('=');
298               super.serializeAnything(session, out, i.next(), null, (key == null ? null : key.toString()), null, false, true);
299               addAmp = true;
300            }
301         } else {
302            if (addAmp)
303               out.cr(depth).append('&');
304            out.appendObject(key, false, true, true).append('=');
305            super.serializeAnything(session, out, value, valueType, (key == null ? null : key.toString()), null, false, true);
306            addAmp = true;
307         }
308      }
309
310      return out;
311   }
312
313   @SuppressWarnings({ "rawtypes" })
314   private SerializerWriter serializeBeanMap(UrlEncodingSerializerSession session, UonWriter out, BeanMap<?> m, boolean addTypeProperty) throws Exception {
315      int depth = session.getIndent();
316
317      boolean addAmp = false;
318
319      for (BeanPropertyValue p : m.getValues(session.isTrimNulls(), addTypeProperty ? session.createBeanTypeNameProperty(m) : null)) {
320         BeanPropertyMeta pMeta = p.getMeta();
321         ClassMeta<?> cMeta = p.getClassMeta();
322
323         String key = p.getName();
324         Object value = p.getValue();
325         Throwable t = p.getThrown();
326         if (t != null)
327            session.addBeanGetterWarning(pMeta, t);
328
329         if (session.canIgnoreValue(cMeta, key, value))
330            continue;
331
332         if (value != null && session.shouldUseExpandedParams(pMeta)) {
333            // Transformed object array bean properties may be transformed resulting in ArrayLists,
334            // so we need to check type if we think it's an array.
335            Iterator i = (cMeta.isCollection() || value instanceof Collection) ? ((Collection)value).iterator() : ArrayUtils.iterator(value);
336            while (i.hasNext()) {
337               if (addAmp)
338                  out.cr(depth).append('&');
339
340               out.appendObject(key, false, true, true).append('=');
341
342               super.serializeAnything(session, out, i.next(), cMeta.getElementType(), key, pMeta, false, true);
343
344               addAmp = true;
345            }
346         } else {
347            if (addAmp)
348               out.cr(depth).append('&');
349
350            out.appendObject(key, false, true, true).append('=');
351
352            super.serializeAnything(session, out, value, cMeta, key, pMeta, false, true);
353
354            addAmp = true;
355         }
356
357      }
358      return out;
359   }

366   /**
367    * Converts the specified object to a string using this serializers {@link BeanSession#convertToType(Object, Class)} method
368    *    and runs {@link URLEncoder#encode(String,String)} against the results.
369    * Useful for constructing URL parts.
370    *
371    * @param o The object to serialize.
372    * @return The serialized object.
373    */
374   public String serializeUrlPart(Object o) {
375      try {
376         // Shortcut for simple types.
377         ClassMeta<?> cm = getBeanContext().getClassMetaForObject(o);
378         if (cm != null)
379            if (cm.isCharSequence() || cm.isNumber() || cm.isBoolean())
380               return o.toString();
381
382         StringWriter w = new StringWriter();
383         UonSerializerSession s = createSession(w, null, null, null, null, MediaType.UON);
384         super.doSerialize(s, o);
385         return w.toString();
386      } catch (Exception e) {
387         throw new RuntimeException(e);
388      }
389   }

401   @Override /* Serializer */
402   protected void doSerialize(SerializerSession session, Object o) throws Exception {
403      UrlEncodingSerializerSession s = (UrlEncodingSerializerSession)session;
404      serializeAnything(s, s.getWriter(), o);
405   }

412   /**
413    * <b>Configuration property:</b>  Serialize bean property collections/arrays as separate key/value pairs.
414    * <p>
415    * <ul>
416    *    <li><b>Name:</b> <js>"UrlEncoding.expandedParams"</js>
417    *    <li><b>Data type:</b> <code>Boolean</code>
418    *    <li><b>Default:</b> <jk>false</jk>
419    *    <li><b>Session-overridable:</b> <jk>true</jk>
420    * </ul>
421    * <p>
422    * If <jk>false</jk>, serializing the array <code>[1,2,3]</code> results in <code>?key=$a(1,2,3)</code>.
423    * If <jk>true</jk>, serializing the same array results in <code>?key=1&amp;key=2&amp;key=3</code>.
424    *
425    * <h5 class='section'>Example:</h5>
426    * <p class='bcode'>
427    *    <jk>public class</jk> A {
428    *       <jk>public</jk> String[] f1 = {<js>"a"</js>,<js>"b"</js>};
429    *       <jk>public</jk> List&lt;String&gt; f2 = <jk>new</jk> LinkedList&lt;String&gt;(Arrays.<jsm>asList</jsm>(<jk>new</jk> String[]{<js>"c"</js>,<js>"d"</js>}));
430    *    }
431    *
432    *    UrlEncodingSerializer s1 = <jk>new</jk> UrlEncodingParser();
433    *    UrlEncodingSerializer s2 = <jk>new</jk> UrlEncodingParser().setExpandedParams(<jk>true</jk>);
434    *
435    *    String s1 = p1.serialize(<jk>new</jk> A()); <jc>// Produces "f1=(a,b)&amp;f2=(c,d)"</jc>
436    *    String s2 = p2.serialize(<jk>new</jk> A()); <jc>// Produces "f1=a&amp;f1=b&amp;f2=c&amp;f2=d"</jc>
437    * </p>
438    * <p>
439    * This option only applies to beans.
440    * <p>
441    * <h5 class='section'>Notes:</h5>
442    * <ul>
443    *    <li>If parsing multi-part parameters, it's highly recommended to use Collections or Lists
444    *       as bean property types instead of arrays since arrays have to be recreated from scratch every time a value
445    *       is added to it.
446    * </ul>
447    * <p>
448    * <h5 class='section'>Notes:</h5>
449    * <ul>
450    *    <li>This is equivalent to calling <code>setProperty(<jsf>URLENC_expandedParams</jsf>, value)</code>.
451    *    <li>This introduces a slight performance penalty.
452    * </ul>
453    *
454    * @param value The new value for this property.
455    * @return This object (for method chaining).
456    * @throws LockedException If {@link #lock()} was called on this class.
457    * @see UrlEncodingSerializerContext#URLENC_expandedParams
458    */
459   public UrlEncodingSerializer setExpandedParams(boolean value) throws LockedException {
460      return setProperty(URLENC_expandedParams, value);
461   }
