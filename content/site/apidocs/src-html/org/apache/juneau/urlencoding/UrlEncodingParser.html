025/**
026 * Parses URL-encoded text into POJO models.
027 *
028 * <h5 class='section'>Media types:</h5>
029 *
030 * Handles <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
031 *
032 * <h5 class='section'>Description:</h5>
033 *
034 * Parses URL-Encoded text (e.g. <js>"foo=bar&amp;baz=bing"</js>) into POJOs.
035 *
036 * <p>
037 * Expects parameter values to be in UON notation.
038 *
039 * <p>
040 * This parser uses a state machine, which makes it very fast and efficient.
041 *
042 * <h5 class='section'>Configurable properties:</h5>
043 *
044 * This class has the following properties associated with it:
045 * <ul>
046 *    <li>{@link UonParserContext}
047 *    <li>{@link BeanContext}
048 * </ul>
049 */
050@SuppressWarnings({ "unchecked", "hiding" })
051public class UrlEncodingParser extends UonParser implements PartParser {
052
053   /** Reusable instance of {@link UrlEncodingParser}. */
054   public static final UrlEncodingParser DEFAULT = new UrlEncodingParser(PropertyStore.create());
055
056
057   private final UrlEncodingParserContext ctx;
058
059   /**
060    * Constructor.
061    *
062    * @param propertyStore The property store containing all the settings for this object.
063    */
064   public UrlEncodingParser(PropertyStore propertyStore) {
065      super(propertyStore.copy().append(UON_decodeChars, true), "application/x-www-form-urlencoded");
066      this.ctx = createContext(UrlEncodingParserContext.class);
067   }
068
069   @Override /* CoreObject */
070   public UrlEncodingParserBuilder builder() {
071      return new UrlEncodingParserBuilder(propertyStore);
072   }
073
074   /**
075    * Parse a URL query string into a simple map of key/value pairs.
076    *
077    * @param qs The query string to parse.
078    * @param map The map to parse into.  If <jk>null</jk>, then a new {@link TreeMap} will be used.
079    * @return A sorted {@link TreeMap} of query string entries.
080    * @throws Exception
081    */
082   public Map<String,String[]> parseIntoSimpleMap(String qs, Map<String,String[]> map) throws Exception {
083
084      Map<String,String[]> m = map == null ? new TreeMap<String,String[]>() : map;
085
086      if (isEmpty(qs))
087         return m;
088
089      // We're reading from a string, so we don't need to make sure close() is called on the pipe.
090      ParserPipe p = new ParserPipe(qs, false, false, null, null);
091      UonReader r = new UonReader(p, true);
092
093      final int S1=1; // Looking for attrName start.
094      final int S2=2; // Found attrName start, looking for = or & or end.
095      final int S3=3; // Found =, looking for valStart.
096      final int S4=4; // Found valStart, looking for & or end.
097
098      try {
099         int c = r.peekSkipWs();
100         if (c == '?')
101            r.read();
102
103         int state = S1;
104         String currAttr = null;
105         while (c != -1) {
106            c = r.read();
107            if (state == S1) {
108               if (c != -1) {
109                  r.unread();
110                  r.mark();
111                  state = S2;
112               }
113            } else if (state == S2) {
114               if (c == -1) {
115                  add(m, r.getMarked(), null);
116               } else if (c == '\u0001') {
117                  m.put(r.getMarked(0,-1), null);
118                  state = S1;
119               } else if (c == '\u0002') {
120                  currAttr = r.getMarked(0,-1);
121                  state = S3;
122               }
123            } else if (state == S3) {
124               if (c == -1 || c == '\u0001') {
125                  add(m, currAttr, "");
126               } else {
127                  if (c == '\u0002')
128                     r.replace('=');
129                  r.unread();
130                  r.mark();
131                  state = S4;
132               }
133            } else if (state == S4) {
134               if (c == -1) {
135                  add(m, currAttr, r.getMarked());
136               } else if (c == '\u0001') {
137                  add(m, currAttr, r.getMarked(0,-1));
138                  state = S1;
139               } else if (c == '\u0002') {
140                  r.replace('=');
141               }
142            }
143         }
144      } finally {
145         r.close();
146      }
147
148      return m;
149   }
150
151   private static void add(Map<String,String[]> m, String key, String val) {
152      boolean b = m.containsKey(key);
153      if (val == null) {
154         if (! b)
155            m.put(key, null);
156      } else if (b && m.get(key) != null) {
157         m.put(key, append(m.get(key), val));
158      } else {
159         m.put(key, new String[]{val});
160      }
161   }
162
163   @Override /* PartParser */
164   public <T> T parse(PartType partType, String in, ClassMeta<T> type) throws ParseException {
165      if (in == null)
166         return null;
167      if (type.isString() && in.length() > 0) {
168         // Shortcut - If we're returning a string and the value doesn't start with "'" or is "null", then
169         // just return the string since it's a plain value.
170         // This allows us to bypass the creation of a UonParserSession object.
171         char x = firstNonWhitespaceChar(in);
172         if (x != '\'' && x != 'n' && in.indexOf('~') == -1)
173            return (T)in;
174         if (x == 'n' && "null".equals(in))
175            return null;
176      }
177      UonParserSession session = createParameterSession();
178      ParserPipe pipe = session.createPipe(in);
179      try {
180         UonReader r = session.getUonReader(pipe, false);
181         return session.parseAnything(type, r, null, true, null);
182      } catch (ParseException e) {
183         throw e;
184      } catch (Exception e) {
185         throw new ParseException(session.getLastLocation(), e);
186      } finally {
187         pipe.close();
188         session.close();
189      }
190   }
