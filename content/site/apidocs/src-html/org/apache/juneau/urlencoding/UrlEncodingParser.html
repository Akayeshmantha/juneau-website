024/**
025 * Parses URL-encoded text into POJO models.
026 *
027 * <h5 class='section'>Media types:</h5>
028 *
029 * Handles <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
030 *
031 * <h5 class='section'>Description:</h5>
032 *
033 * Parses URL-Encoded text (e.g. <js>"foo=bar&amp;baz=bing"</js>) into POJOs.
034 *
035 * <p>
036 * Expects parameter values to be in UON notation.
037 *
038 * <p>
039 * This parser uses a state machine, which makes it very fast and efficient.
040 */
041@SuppressWarnings({ "unchecked" })
042public class UrlEncodingParser extends UonParser implements PartParser {
043
044   //-------------------------------------------------------------------------------------------------------------------
045   // Configurable properties
046   //-------------------------------------------------------------------------------------------------------------------
047
048   private static final String PREFIX = "UrlEncodingParser.";
049
050   /**
051    * Parser bean property collections/arrays as separate key/value pairs ({@link Boolean}, default=<jk>false</jk>).
052    *
053    * <p>
054    * This is the parser-side equivalent of the {@link #URLENC_expandedParams} setting.
055    *
056    * <p>
057    * This option only applies to beans.
058    *
059    * <h5 class='section'>Notes:</h5>
060    * <ul>
061    *    <li>If parsing multi-part parameters, it's highly recommended to use <code>Collections</code> or <code>Lists</code>
062    *       as bean property types instead of arrays since arrays have to be recreated from scratch every time a value
063    *       is added to it.
064    * </ul>
065    */
066   public static final String URLENC_expandedParams = PREFIX + "expandedParams";
067

115   /**
116    * Parse a URL query string into a simple map of key/value pairs.
117    *
118    * @param qs The query string to parse.
119    * @param map The map to parse into.  If <jk>null</jk>, then a new {@link TreeMap} will be used.
120    * @return A sorted {@link TreeMap} of query string entries.
121    * @throws Exception
122    */
123   public Map<String,String[]> parseIntoSimpleMap(String qs, Map<String,String[]> map) throws Exception {
124
125      Map<String,String[]> m = map == null ? new TreeMap<String,String[]>() : map;
126
127      if (isEmpty(qs))
128         return m;
129
130      try (ParserPipe p = new ParserPipe(qs, false, false, null, null)) {
131         
132         final int S1=1; // Looking for attrName start.
133         final int S2=2; // Found attrName start, looking for = or & or end.
134         final int S3=3; // Found =, looking for valStart.
135         final int S4=4; // Found valStart, looking for & or end.
136
137         try (UonReader r = new UonReader(p, true)) {
138            int c = r.peekSkipWs();
139            if (c == '?')
140               r.read();
141
142            int state = S1;
143            String currAttr = null;
144            while (c != -1) {
145               c = r.read();
146               if (state == S1) {
147                  if (c != -1) {
148                     r.unread();
149                     r.mark();
150                     state = S2;
151                  }
152               } else if (state == S2) {
153                  if (c == -1) {
154                     add(m, r.getMarked(), null);
155                  } else if (c == '\u0001') {
156                     m.put(r.getMarked(0,-1), null);
157                     state = S1;
158                  } else if (c == '\u0002') {
159                     currAttr = r.getMarked(0,-1);
160                     state = S3;
161                  }
162               } else if (state == S3) {
163                  if (c == -1 || c == '\u0001') {
164                     add(m, currAttr, "");
165                  } else {
166                     if (c == '\u0002')
167                        r.replace('=');
168                     r.unread();
169                     r.mark();
170                     state = S4;
171                  }
172               } else if (state == S4) {
173                  if (c == -1) {
174                     add(m, currAttr, r.getMarked());
175                  } else if (c == '\u0001') {
176                     add(m, currAttr, r.getMarked(0,-1));
177                     state = S1;
178                  } else if (c == '\u0002') {
179                     r.replace('=');
180                  }
181               }
182            }
183         }
184
185         return m;
186      }
187   }
188
189   private static void add(Map<String,String[]> m, String key, String val) {
190      boolean b = m.containsKey(key);
191      if (val == null) {
192         if (! b)
193            m.put(key, null);
194      } else if (b && m.get(key) != null) {
195         m.put(key, append(m.get(key), val));
196      } else {
197         m.put(key, new String[]{val});
198      }
199   }
200
201   @Override /* PartParser */
202   public <T> T parse(PartType partType, String in, ClassMeta<T> type) throws ParseException {
203      if (in == null)
204         return null;
205      if (type.isString() && in.length() > 0) {
206         // Shortcut - If we're returning a string and the value doesn't start with "'" or is "null", then
207         // just return the string since it's a plain value.
208         // This allows us to bypass the creation of a UonParserSession object.
209         char x = firstNonWhitespaceChar(in);
210         if (x != '\'' && x != 'n' && in.indexOf('~') == -1)
211            return (T)in;
212         if (x == 'n' && "null".equals(in))
213            return null;
214      }
215      UonParserSession session = createParameterSession();
216      try (ParserPipe pipe = session.createPipe(in)) {
217         try (UonReader r = session.getUonReader(pipe, false)) {
218            return session.parseAnything(type, r, null, true, null);
219         }
220      } catch (ParseException e) {
221         throw e;
222      } catch (Exception e) {
223         throw new ParseException(e);
224      }
225   }
