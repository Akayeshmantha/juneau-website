026/**
027 * Parses URL-encoded text into POJO models.
028 *
029 * <h5 class='section'>Media types:</h5>
030 *
031 * Handles <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
032 *
033 * <h5 class='section'>Description:</h5>
034 *
035 * Parses URL-Encoded text (e.g. <js>"foo=bar&amp;baz=bing"</js>) into POJOs.
036 *
037 * <p>
038 * Expects parameter values to be in UON notation.
039 *
040 * <p>
041 * This parser uses a state machine, which makes it very fast and efficient.
042 *
043 * <h5 class='section'>Configurable properties:</h5>
044 *
045 * This class has the following properties associated with it:
046 * <ul>
047 *    <li>{@link UonParserContext}
048 *    <li>{@link BeanContext}
049 * </ul>
050 */
051@SuppressWarnings({ "unchecked", "hiding" })
052@Consumes("application/x-www-form-urlencoded")
053public class UrlEncodingParser extends UonParser implements PartParser {
054
055   /** Reusable instance of {@link UrlEncodingParser}. */
056   public static final UrlEncodingParser DEFAULT = new UrlEncodingParser(PropertyStore.create());
057
058
059   private final UrlEncodingParserContext ctx;
060
061   /**
062    * Constructor.
063    *
064    * @param propertyStore The property store containing all the settings for this object.
065    */
066   public UrlEncodingParser(PropertyStore propertyStore) {
067      super(propertyStore.copy().append(UON_decodeChars, true));
068      this.ctx = createContext(UrlEncodingParserContext.class);
069   }
070
071   @Override /* CoreObject */
072   public UrlEncodingParserBuilder builder() {
073      return new UrlEncodingParserBuilder(propertyStore);
074   }
075
076   /**
077    * Parse a URL query string into a simple map of key/value pairs.
078    *
079    * @param qs The query string to parse.
080    * @param map The map to parse into.  If <jk>null</jk>, then a new {@link TreeMap} will be used.
081    * @return A sorted {@link TreeMap} of query string entries.
082    * @throws Exception
083    */
084   public Map<String,String[]> parseIntoSimpleMap(String qs, Map<String,String[]> map) throws Exception {
085
086      Map<String,String[]> m = map == null ? new TreeMap<String,String[]>() : map;
087
088      if (isEmpty(qs))
089         return m;
090
091      // We're reading from a string, so we don't need to make sure close() is called on the pipe.
092      ParserPipe p = new ParserPipe(qs, false, false, null, null);
093      UonReader r = new UonReader(p, true);
094
095      final int S1=1; // Looking for attrName start.
096      final int S2=2; // Found attrName start, looking for = or & or end.
097      final int S3=3; // Found =, looking for valStart.
098      final int S4=4; // Found valStart, looking for & or end.
099
100      try {
101         int c = r.peekSkipWs();
102         if (c == '?')
103            r.read();
104
105         int state = S1;
106         String currAttr = null;
107         while (c != -1) {
108            c = r.read();
109            if (state == S1) {
110               if (c != -1) {
111                  r.unread();
112                  r.mark();
113                  state = S2;
114               }
115            } else if (state == S2) {
116               if (c == -1) {
117                  add(m, r.getMarked(), null);
118               } else if (c == '\u0001') {
119                  m.put(r.getMarked(0,-1), null);
120                  state = S1;
121               } else if (c == '\u0002') {
122                  currAttr = r.getMarked(0,-1);
123                  state = S3;
124               }
125            } else if (state == S3) {
126               if (c == -1 || c == '\u0001') {
127                  add(m, currAttr, "");
128               } else {
129                  if (c == '\u0002')
130                     r.replace('=');
131                  r.unread();
132                  r.mark();
133                  state = S4;
134               }
135            } else if (state == S4) {
136               if (c == -1) {
137                  add(m, currAttr, r.getMarked());
138               } else if (c == '\u0001') {
139                  add(m, currAttr, r.getMarked(0,-1));
140                  state = S1;
141               } else if (c == '\u0002') {
142                  r.replace('=');
143               }
144            }
145         }
146      } finally {
147         r.close();
148      }
149
150      return m;
151   }
152
153   private static void add(Map<String,String[]> m, String key, String val) {
154      boolean b = m.containsKey(key);
155      if (val == null) {
156         if (! b)
157            m.put(key, null);
158      } else if (b && m.get(key) != null) {
159         m.put(key, append(m.get(key), val));
160      } else {
161         m.put(key, new String[]{val});
162      }
163   }
164
165   @Override /* PartParser */
166   public <T> T parse(PartType partType, String in, ClassMeta<T> type) throws ParseException {
167      if (in == null)
168         return null;
169      if (type.isString() && in.length() > 0) {
170         // Shortcut - If we're returning a string and the value doesn't start with "'" or is "null", then
171         // just return the string since it's a plain value.
172         // This allows us to bypass the creation of a UonParserSession object.
173         char x = firstNonWhitespaceChar(in);
174         if (x != '\'' && x != 'n' && in.indexOf('~') == -1)
175            return (T)in;
176         if (x == 'n' && "null".equals(in))
177            return null;
178      }
179      UonParserSession session = createParameterSession();
180      ParserPipe pipe = session.createPipe(in);
181      try {
182         UonReader r = session.getUonReader(pipe, false);
183         return session.parseAnything(type, r, null, true, null);
184      } catch (ParseException e) {
185         throw e;
186      } catch (Exception e) {
187         throw new ParseException(session.getLastLocation(), e);
188      } finally {
189         pipe.close();
190         session.close();
191      }
192   }

201      return new UrlEncodingParserSession(ctx, args);
202   }
