027/**
028 * Parses URL-encoded text into POJO models.
029 *
030 * <h5 class='section'>Media types:</h5>
031 * <p>
032 * Handles <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
033 *
034 * <h5 class='section'>Description:</h5>
035 * <p>
036 * Parses URL-Encoded text (e.g. <js>"foo=bar&amp;baz=bing"</js>) into POJOs.
037 * <p>
038 * Expects parameter values to be in UON notation.
039 * <p>
040 * This parser uses a state machine, which makes it very fast and efficient.
041 *
042 * <h5 class='section'>Configurable properties:</h5>
043 * <p>
044 * This class has the following properties associated with it:
045 * <ul>
046 *    <li>{@link UonParserContext}
047 *    <li>{@link BeanContext}
048 * </ul>
049 */
050@SuppressWarnings({ "rawtypes", "unchecked", "hiding" })
051@Consumes("application/x-www-form-urlencoded")
052public class UrlEncodingParser extends UonParser {
053

079   private <T> T parseAnything(UrlEncodingParserSession session, ClassMeta<T> eType, ParserReader r, Object outer) throws Exception {
080
081      if (eType == null)
082         eType = (ClassMeta<T>)object();
083      PojoSwap<T,Object> transform = (PojoSwap<T,Object>)eType.getPojoSwap();
084      ClassMeta<?> sType = eType.getSerializedClassMeta();
085
086      int c = r.peekSkipWs();
087      if (c == '?')
088         r.read();
089
090      Object o;
091
092      if (sType.isObject()) {
093         ObjectMap m = new ObjectMap(session);
094         parseIntoMap(session, r, m, session.getClassMeta(Map.class, String.class, Object.class), outer);
095         if (m.containsKey("_value"))
096            o = m.get("_value");
097         else
098            o = session.cast(m, null, eType);
099      } else if (sType.isMap()) {
100         Map m = (sType.canCreateNewInstance() ? (Map)sType.newInstance() : new ObjectMap(session));
101         o = parseIntoMap(session, r, m, sType, m);
102      } else if (sType.canCreateNewBean(outer)) {
103         BeanMap m = session.newBeanMap(outer, sType.getInnerClass());
104         m = parseIntoBeanMap(session, r, m);
105         o = m == null ? null : m.getBean();
106      } else if (sType.isCollection() || sType.isArray() || sType.isArgs()) {
107         // ?1=foo&2=bar...
108         Collection c2 = ((sType.isArray() || sType.isArgs()) || ! sType.canCreateNewInstance(outer)) ? new ObjectList(session) : (Collection)sType.newInstance();
109         Map<Integer,Object> m = new TreeMap<Integer,Object>();
110         parseIntoMap(session, r, m, sType, c2);
111         c2.addAll(m.values());
112         if (sType.isArray())
113            o = ArrayUtils.toArray(c2, sType.getElementType().getInnerClass());
114         else if (sType.isArgs())
115            o = c2.toArray(new Object[c2.size()]);
116         else
117            o = c2;
118      } else {
119         // It could be a non-bean with _type attribute.
120         ObjectMap m = new ObjectMap(session);
121         parseIntoMap(session, r, m, session.getClassMeta(Map.class, String.class, Object.class), outer);
122         if (m.containsKey(session.getBeanTypePropertyName(eType)))
123            o = session.cast(m, null, eType);
124         else if (m.containsKey("_value")) {
125            o = session.convertToType(m.get("_value"), sType);
126         } else {
127            if (sType.getNotABeanReason() != null)
128               throw new ParseException(session, "Class ''{0}'' could not be instantiated as application/x-www-form-urlencoded.  Reason: ''{1}''", sType, sType.getNotABeanReason());
129            throw new ParseException(session, "Malformed application/x-www-form-urlencoded input for class ''{0}''.", sType);
130         }
131      }
132
133      if (transform != null && o != null)
134         o = transform.unswap(session, o, eType);
135
136      if (outer != null)
137         setParent(eType, o, outer);
138
139      return (T)o;
140   }

142   private <K,V> Map<K,V> parseIntoMap(UonParserSession session, ParserReader r, Map<K,V> m, ClassMeta<?> type, Object outer) throws Exception {
143
144      ClassMeta<K> keyType = (ClassMeta<K>)(type.isArgs() || type.isCollectionOrArray() ? session.getClassMeta(Integer.class) : type.getKeyType());
145
146      int c = r.peekSkipWs();
147      if (c == -1)
148         return m;
149
150      final int S1=1; // Looking for attrName start.
151      final int S2=2; // Found attrName end, looking for =.
152      final int S3=3; // Found =, looking for valStart.
153      final int S4=4; // Looking for & or end.
154      boolean isInEscape = false;
155
156      int state = S1;
157      int argIndex = 0;
158      K currAttr = null;
159      while (c != -1) {
160         c = r.read();
161         if (! isInEscape) {
162            if (state == S1) {
163               if (c == -1)
164                  return m;
165               r.unread();
166               Object attr = parseAttr(session, r, true);
167               currAttr = attr == null ? null : convertAttrToType(session, m, session.trim(attr.toString()), keyType);
168               state = S2;
169               c = 0; // Avoid isInEscape if c was '\'
170            } else if (state == S2) {
171               if (c == '\u0002')
172                  state = S3;
173               else if (c == -1 || c == '\u0001') {
174                  m.put(currAttr, null);
175                  if (c == -1)
176                     return m;
177                  state = S1;
178               }
179            } else if (state == S3) {
180               if (c == -1 || c == '\u0001') {
181                  ClassMeta<V> valueType = (ClassMeta<V>)(type.isArgs() ? type.getArg(argIndex++) : type.isCollectionOrArray() ? type.getElementType() : type.getValueType());
182                  V value = convertAttrToType(session, m, "", valueType);
183                  m.put(currAttr, value);
184                  if (c == -1)
185                     return m;
186                  state = S1;
187               } else  {
188                  // For performance, we bypass parseAnything for string values.
189                  ClassMeta<V> valueType = (ClassMeta<V>)(type.isArgs() ? type.getArg(argIndex++) : type.isCollectionOrArray() ? type.getElementType() : type.getValueType());
190                  V value = (V)(valueType.isString() ? super.parseString(session, r.unread(), true) : super.parseAnything(session, valueType, r.unread(), outer, true, null));
191
192                  // If we already encountered this parameter, turn it into a list.
193                  if (m.containsKey(currAttr) && valueType.isObject()) {
194                     Object v2 = m.get(currAttr);
195                     if (! (v2 instanceof ObjectList)) {
196                        v2 = new ObjectList(v2).setBeanSession(session);
197                        m.put(currAttr, (V)v2);
198                     }
199                     ((ObjectList)v2).add(value);
200                  } else {
201                     m.put(currAttr, value);
202                  }
203                  state = S4;
204                  c = 0; // Avoid isInEscape if c was '\'
205               }
206            } else if (state == S4) {
207               if (c == '\u0001')
208                  state = S1;
209               else if (c == -1) {
210                  return m;
211               }
212            }
213         }
214         isInEscape = (c == '\\' && ! isInEscape);
215      }
216      if (state == S1)
217         throw new ParseException(session, "Could not find attribute name on object.");
218      if (state == S2)
219         throw new ParseException(session, "Could not find '=' following attribute name on object.");
220      if (state == S3)
221         throw new ParseException(session, "Dangling '=' found in object entry");
222      if (state == S4)
223         throw new ParseException(session, "Could not find end of object.");
224
225      return null; // Unreachable.
226   }

228   private <T> BeanMap<T> parseIntoBeanMap(UrlEncodingParserSession session, ParserReader r, BeanMap<T> m) throws Exception {
229
230      int c = r.peekSkipWs();
231      if (c == -1)
232         return m;
233
234      final int S1=1; // Looking for attrName start.
235      final int S2=2; // Found attrName end, looking for =.
236      final int S3=3; // Found =, looking for valStart.
237      final int S4=4; // Looking for , or }
238      boolean isInEscape = false;
239
240      int state = S1;
241      String currAttr = "";
242      int currAttrLine = -1, currAttrCol = -1;
243      while (c != -1) {
244         c = r.read();
245         if (! isInEscape) {
246            if (state == S1) {
247               if (c == -1) {
248                  return m;
249               }
250               r.unread();
251               currAttrLine= r.getLine();
252               currAttrCol = r.getColumn();
253               currAttr = parseAttrName(session, r, true);
254               if (currAttr == null)  // Value was '%00'
255                  return null;
256               state = S2;
257            } else if (state == S2) {
258               if (c == '\u0002')
259                  state = S3;
260               else if (c == -1 || c == '\u0001') {
261                  m.put(currAttr, null);
262                  if (c == -1)
263                     return m;
264                  state = S1;
265               }
266            } else if (state == S3) {
267               if (c == -1 || c == '\u0001') {
268                  if (! currAttr.equals(session.getBeanTypePropertyName(m.getClassMeta()))) {
269                     BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
270                     if (pMeta == null) {
271                        onUnknownProperty(session, currAttr, m, currAttrLine, currAttrCol);
272                     } else {
273                        session.setCurrentProperty(pMeta);
274                        // In cases of "&foo=", create an empty instance of the value if createable.
275                        // Otherwise, leave it null.
276                        ClassMeta<?> cm = pMeta.getClassMeta();
277                        if (cm.canCreateNewInstance())
278                           pMeta.set(m, cm.newInstance());
279                        session.setCurrentProperty(null);
280                     }
281                  }
282                  if (c == -1)
283                     return m;
284                  state = S1;
285               } else {
286                  if (! currAttr.equals(session.getBeanTypePropertyName(m.getClassMeta()))) {
287                     BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
288                     if (pMeta == null) {
289                        onUnknownProperty(session, currAttr, m, currAttrLine, currAttrCol);
290                        parseAnything(session, object(), r.unread(), m.getBean(false), true, null); // Read content anyway to ignore it
291                     } else {
292                        session.setCurrentProperty(pMeta);
293                        if (session.shouldUseExpandedParams(pMeta)) {
294                           ClassMeta et = pMeta.getClassMeta().getElementType();
295                           Object value = parseAnything(session, et, r.unread(), m.getBean(false), true, pMeta);
296                           setName(et, value, currAttr);
297                           pMeta.add(m, value);
298                        } else {
299                           ClassMeta<?> cm = pMeta.getClassMeta();
300                           Object value = parseAnything(session, cm, r.unread(), m.getBean(false), true, pMeta);
301                           setName(cm, value, currAttr);
302                           pMeta.set(m, value);
303                        }
304                        session.setCurrentProperty(null);
305                     }
306                  }
307                  state = S4;
308               }
309            } else if (state == S4) {
310               if (c == '\u0001')
311                  state = S1;
312               else if (c == -1) {
313                  return m;
314               }
315            }
316         }
317         isInEscape = (c == '\\' && ! isInEscape);
318      }
319      if (state == S1)
320         throw new ParseException(session, "Could not find attribute name on object.");
321      if (state == S2)
322         throw new ParseException(session, "Could not find '=' following attribute name on object.");
323      if (state == S3)
324         throw new ParseException(session, "Could not find value following '=' on object.");
325      if (state == S4)
326         throw new ParseException(session, "Could not find end of object.");
327
328      return null; // Unreachable.
329   }

331   /**
332    * Parse a URL query string into a simple map of key/value pairs.
333    *
334    * @param qs The query string to parse.
335    * @return A sorted {@link TreeMap} of query string entries.
336    * @throws Exception
337    */
338   public Map<String,String[]> parseIntoSimpleMap(String qs) throws Exception {
339
340      Map<String,String[]> m = new TreeMap<String,String[]>();
341
342      if (StringUtils.isEmpty(qs))
343         return m;
344
345      UonReader r = new UonReader(qs, true);
346
347      final int S1=1; // Looking for attrName start.
348      final int S2=2; // Found attrName start, looking for = or & or end.
349      final int S3=3; // Found =, looking for valStart.
350      final int S4=4; // Found valStart, looking for & or end.
351
352      try {
353         int c = r.peekSkipWs();
354         if (c == '?')
355            r.read();
356
357         int state = S1;
358         String currAttr = null;
359         while (c != -1) {
360            c = r.read();
361            if (state == S1) {
362               if (c != -1) {
363                  r.unread();
364                  r.mark();
365                  state = S2;
366               }
367            } else if (state == S2) {
368               if (c == -1) {
369                  add(m, r.getMarked(), null);
370               } else if (c == '\u0001') {
371                  m.put(r.getMarked(0,-1), null);
372                  state = S1;
373               } else if (c == '\u0002') {
374                  currAttr = r.getMarked(0,-1);
375                  state = S3;
376               }
377            } else if (state == S3) {
378               if (c == -1 || c == '\u0001') {
379                  add(m, currAttr, "");
380               } else {
381                  if (c == '\u0002')
382                     r.replace('=');
383                  r.unread();
384                  r.mark();
385                  state = S4;
386               }
387            } else if (state == S4) {
388               if (c == -1) {
389                  add(m, currAttr, r.getMarked());
390               } else if (c == '\u0001') {
391                  add(m, currAttr, r.getMarked(0,-1));
392                  state = S1;
393               } else if (c == '\u0002') {
394                  r.replace('=');
395               }
396            }
397         }
398      } finally {
399         r.close();
400      }
401
402      return m;
403   }

405   private static void add(Map<String,String[]> m, String key, String val) {
406      boolean b = m.containsKey(key);
407      if (val == null) {
408         if (! b)
409            m.put(key, null);
410      } else if (b && m.get(key) != null) {
411         m.put(key, ArrayUtils.append(m.get(key), val));
412      } else {
413         m.put(key, new String[]{val});
414      }
415   }

417   /**
418    * Parses a single query parameter or header value into the specified class type.
419    *
420    * @param in The input query string value.
421    * @param type The object type to create.
422    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
423    * @param args The type arguments of the class if it's a collection or map.
424    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
425    *    <br>Ignored if the main type is not a map or collection.
426    * @return A new instance of the specified type.
427    * @throws ParseException
428    */
429   public <T> T parsePart(String in, Type type, Type...args) throws ParseException {
430      if (in == null)
431         return null;
432      return (T)parsePart(in, getBeanContext().getClassMeta(type, args));
433   }

435   /**
436    * Parses a single query parameter or header value into the specified class type.
437    *
438    * @param in The input query string value.
439    * @param type The class type of the object to create.
440    * @return A new instance of the specified type.
441    * @throws ParseException
442    */
443   public <T> T parsePart(String in, Class<T> type) throws ParseException {
444      if (in == null)
445         return null;
446      return parsePart(in, getBeanContext().getClassMeta(type));
447   }

449   /**
450    * Same as {@link #parsePart(String, Type, Type...)} except the type has already
451    * been converted to a {@link ClassMeta} object.
452    *
453    * @param in The input query string value.
454    * @param type The class type of the object to create.
455    * @return A new instance of the specified type.
456    * @throws ParseException
457    */
458   public <T> T parsePart(String in, ClassMeta<T> type) throws ParseException {
459      if (in == null)
460         return null;
461      if (type.isString() && in.length() > 0) {
462         // Shortcut - If we're returning a string and the value doesn't start with "'" or is "null", then
463         // just return the string since it's a plain value.
464         // This allows us to bypass the creation of a UonParserSession object.
465         char x = StringUtils.firstNonWhitespaceChar(in);
466         if (x != '\'' && x != 'n' && in.indexOf('~') == -1)
467            return (T)in;
468         if (x == 'n' && "null".equals(in))
469            return null;
470      }
471      UonParserSession session = createParameterSession(in);
472      try {
473         UonReader r = session.getReader();
474         return super.parseAnything(session, type, r, null, true, null);
475      } catch (ParseException e) {
476         throw e;
477      } catch (Exception e) {
478         throw new ParseException(session, e);
479      } finally {
480         session.close();
481      }
482   }

489   @Override /* Parser */
490   public UrlEncodingParserSession createSession(Object input, ObjectMap op, Method javaMethod, Object outer, Locale locale, TimeZone timeZone, MediaType mediaType) {
491      return new UrlEncodingParserSession(ctx, op, input, javaMethod, outer, locale, timeZone, mediaType);
492   }

494   @Override /* Parser */
495   protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
496      UrlEncodingParserSession s = (UrlEncodingParserSession)session;
497      UonReader r = s.getReader();
498      T o = parseAnything(s, type, r, s.getOuter());
499      return o;
500   }
501
502   @Override /* ReaderParser */
503   protected <K,V> Map<K,V> doParseIntoMap(ParserSession session, Map<K,V> m, Type keyType, Type valueType) throws Exception {
504      UrlEncodingParserSession s = (UrlEncodingParserSession)session;
505      UonReader r = s.getReader();
506      if (r.peekSkipWs() == '?')
507         r.read();
508      m = parseIntoMap(s, r, m, session.getClassMeta(Map.class, keyType, valueType), null);
509      return m;
510   }
