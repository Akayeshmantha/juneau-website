028/**
029 * Serializes POJO models to URL-encoded notation with UON-encoded values (a notation for URL-encoded query paramter values).
030 *
031 * <h5 class='section'>Media types:</h5>
032 *
033 * Handles <code>Accept</code> types: <code>application/x-www-form-urlencoded</code>
034 * <p>
035 * Produces <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
036 *
037 * <h5 class='section'>Description:</h5>
038 *
039 * This serializer provides several serialization options.  Typically, one of the predefined DEFAULT serializers will be sufficient.
040 * However, custom serializers can be constructed to fine-tune behavior.
041 *
042 * <h5 class='section'>Configurable properties:</h5>
043 *
044 * This class has the following properties associated with it:
045 * <ul>
046 *    <li>{@link UonSerializerContext}
047 *    <li>{@link BeanContext}
048 * </ul>
049 *
050 * <p>
051 * The following shows a sample object defined in Javascript:
052 * <p class='bcode'>
053 *    {
054 *       id: 1,
055 *       name: <js>'John Smith'</js>,
056 *       uri: <js>'http://sample/addressBook/person/1'</js>,
057 *       addressBookUri: <js>'http://sample/addressBook'</js>,
058 *       birthDate: <js>'1946-08-12T00:00:00Z'</js>,
059 *       otherIds: <jk>null</jk>,
060 *       addresses: [
061 *          {
062 *             uri: <js>'http://sample/addressBook/address/1'</js>,
063 *             personUri: <js>'http://sample/addressBook/person/1'</js>,
064 *             id: 1,
065 *             street: <js>'100 Main Street'</js>,
066 *             city: <js>'Anywhereville'</js>,
067 *             state: <js>'NY'</js>,
068 *             zip: 12345,
069 *             isCurrent: <jk>true</jk>,
070 *          }
071 *       ]
072 *    }
073 * </p>
074 *
075 * <p>
076 * Using the "strict" syntax defined in this document, the equivalent URL-encoded notation would be as follows:
077 * <p class='bcode'>
078 *    <ua>id</ua>=<un>1</un>
079 *    &amp;<ua>name</ua>=<us>'John+Smith'</us>,
080 *    &amp;<ua>uri</ua>=<us>http://sample/addressBook/person/1</us>,
081 *    &amp;<ua>addressBookUri</ua>=<us>http://sample/addressBook</us>,
082 *    &amp;<ua>birthDate</ua>=<us>1946-08-12T00:00:00Z</us>,
083 *    &amp;<ua>otherIds</ua>=<uk>null</uk>,
084 *    &amp;<ua>addresses</ua>=@(
085 *       (
086 *          <ua>uri</ua>=<us>http://sample/addressBook/address/1</us>,
087 *          <ua>personUri</ua>=<us>http://sample/addressBook/person/1</us>,
088 *          <ua>id</ua>=<un>1</un>,
089 *          <ua>street</ua>=<us>'100+Main+Street'</us>,
090 *          <ua>city</ua>=<us>Anywhereville</us>,
091 *          <ua>state</ua>=<us>NY</us>,
092 *          <ua>zip</ua>=<un>12345</un>,
093 *          <ua>isCurrent</ua>=<uk>true</uk>
094 *       )
095 *    )
096 * </p>
097 *
098 * <h5 class='section'>Example:</h5>
099 * <p class='bcode'>
100 *    <jc>// Serialize a Map</jc>
101 *    Map m = <jk>new</jk> ObjectMap(<js>"{a:'b',c:1,d:false,e:['f',1,false],g:{h:'i'}}"</js>);
102 *
103 *    <jc>// Serialize to value equivalent to JSON.</jc>
104 *    <jc>// Produces "a=b&amp;c=1&amp;d=false&amp;e=@(f,1,false)&amp;g=(h=i)"</jc>
105 *    String s = UrlEncodingSerializer.<jsf>DEFAULT</jsf>.serialize(s);
106 *
107 *    <jc>// Serialize a bean</jc>
108 *    <jk>public class</jk> Person {
109 *       <jk>public</jk> Person(String s);
110 *       <jk>public</jk> String getName();
111 *       <jk>public int</jk> getAge();
112 *       <jk>public</jk> Address getAddress();
113 *       <jk>public boolean</jk> deceased;
114 *    }
115 *
116 *    <jk>public class</jk> Address {
117 *       <jk>public</jk> String getStreet();
118 *       <jk>public</jk> String getCity();
119 *       <jk>public</jk> String getState();
120 *       <jk>public int</jk> getZip();
121 *    }
122 *
123 *    Person p = <jk>new</jk> Person(<js>"John Doe"</js>, 23, <js>"123 Main St"</js>, <js>"Anywhere"</js>, <js>"NY"</js>, 12345, <jk>false</jk>);
124 *
125 *    <jc>// Produces "name=John+Doe&amp;age=23&amp;address=(street='123+Main+St',city=Anywhere,state=NY,zip=12345)&amp;deceased=false"</jc>
126 *    String s = UrlEncodingSerializer.<jsf>DEFAULT</jsf>.serialize(s);
127 * </p>
128 */
129@Produces("application/x-www-form-urlencoded")
130@SuppressWarnings("hiding")
131public class UrlEncodingSerializer extends UonSerializer implements PartSerializer {
132
133   /** Reusable instance of {@link UrlEncodingSerializer}, all default settings. */
134   public static final UrlEncodingSerializer DEFAULT = new UrlEncodingSerializer(PropertyStore.create());
135
136   /** Reusable instance of {@link UrlEncodingSerializer.PlainText}. */
137   public static final UrlEncodingSerializer DEFAULT_PLAINTEXT = new PlainText(PropertyStore.create());
138
139   /** Reusable instance of {@link UrlEncodingSerializer.Expanded}. */
140   public static final UrlEncodingSerializer DEFAULT_EXPANDED = new Expanded(PropertyStore.create());
141
142   /** Reusable instance of {@link UrlEncodingSerializer.Readable}. */
143   public static final UrlEncodingSerializer DEFAULT_READABLE = new Readable(PropertyStore.create());
144
145   /**
146    * Equivalent to <code><jk>new</jk> UrlEncodingSerializerBuilder().expandedParams(<jk>true</jk>).build();</code>.
147    */
148   @Produces(value="application/x-www-form-urlencoded",contentType="application/x-www-form-urlencoded")
149   public static class Expanded extends UrlEncodingSerializer {
150
151      /**
152       * Constructor.
153       *
154       * @param propertyStore The property store containing all the settings for this object.
155       */
156      public Expanded(PropertyStore propertyStore) {
157         super(propertyStore.copy().append(URLENC_expandedParams, true));
158      }
159   }
160
161   /**
162    * Equivalent to <code><jk>new</jk> UrlEncodingSerializerBuilder().useWhitespace(<jk>true</jk>).build();</code>.
163    */
164   public static class Readable extends UrlEncodingSerializer {
165
166      /**
167       * Constructor.
168       *
169       * @param propertyStore The property store containing all the settings for this object.
170       */
171      public Readable(PropertyStore propertyStore) {
172         super(propertyStore.copy().append(SERIALIZER_useWhitespace, true));
173      }
174   }
175
176   /**
177    * Equivalent to <code><jk>new</jk> UrlEncodingSerializerBuilder().plainTextParts().build();</code>.
178    */
179   public static class PlainText extends UrlEncodingSerializer {
180
181      /**
182       * Constructor.
183       *
184       * @param propertyStore The property store containing all the settings for this object.
185       */
186      public PlainText(PropertyStore propertyStore) {
187         super(propertyStore.copy().append(UON_paramFormat, "PLAINTEXT"));
188      }
189   }
190
191   private final UrlEncodingSerializerContext ctx;
192
193   /**
194    * Constructor.
195    *
196    * @param propertyStore The property store containing all the settings for this object.
197    */
198   public UrlEncodingSerializer(PropertyStore propertyStore) {
199      super(propertyStore.copy().append(UON_encodeChars, true));
200      this.ctx = createContext(UrlEncodingSerializerContext.class);
201   }
202
203   @Override /* CoreObject */
204   public UrlEncodingSerializerBuilder builder() {
205      return new UrlEncodingSerializerBuilder(propertyStore);
206   }
207
208
209   //--------------------------------------------------------------------------------
210   // Methods for constructing individual parameter values.
211   //--------------------------------------------------------------------------------
212
213   /**
214    * Converts the specified object to a string using this serializers {@link BeanSession#convertToType(Object, Class)}
215    * method and runs {@link URLEncoder#encode(String,String)} against the results.
216    *
217    * <p>
218    * Useful for constructing URL parts.
219    *
220    * @param o The object to serialize.
221    * @param urlEncode
222    *    URL-encode the string if necessary.
223    *    If <jk>null</jk>, then uses the value of the {@link UonSerializerContext#UON_encodeChars} setting.
224    * @param plainTextParams
225    *    Whether we're using plain-text params.
226    *    If <jk>null</jk>, then uses the value from the {@link UrlEncodingSerializerContext#URLENC_paramFormat} setting.
227    * @return The serialized object.
228    */
229   private String serializePart(Object o, Boolean urlEncode, Boolean plainTextParams) {
230      try {
231         // Shortcut for simple types.
232         ClassMeta<?> cm = getBeanContext().getClassMetaForObject(o);
233         if (cm != null) {
234            if (cm.isNumber() || cm.isBoolean())
235               return o.toString();
236            if (cm.isCharSequence()) {
237               String s = o.toString();
238               boolean ptt = (plainTextParams != null ? plainTextParams : ctx.plainTextParams());
239               if (ptt || s.isEmpty() || ! UonUtils.needsQuotes(s))
240                  return (urlEncode ? urlEncode(s) : s);
241            }
242         }
243
244         StringWriter w = new StringWriter();
245         UonSerializerSession s = new UonSerializerSession(ctx, urlEncode, SerializerSessionArgs.DEFAULT);
246         s.serialize(w, o);
247         return w.toString();
248      } catch (Exception e) {
249         throw new RuntimeException(e);
250      }
251   }
252
253
254   //--------------------------------------------------------------------------------
255   // Entry point methods
256   //--------------------------------------------------------------------------------
257
258   @Override /* Serializer */
259   public WriterSerializerSession createSession(SerializerSessionArgs args) {
260      return new UrlEncodingSerializerSession(ctx, null, args);
261   }
262
263   @Override /* PartSerializer */
264   public String serialize(PartType type, Object value) {
265      switch(type) {
266         case HEADER: return serializePart(value, false, true);
267         case FORM_DATA: return serializePart(value, false, null);
268         case PATH: return serializePart(value, false, null);
269         case QUERY: return serializePart(value, false, null);
270         default: return toString(value);
271      }
272   }
