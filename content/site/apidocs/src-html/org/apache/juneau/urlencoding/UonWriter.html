013package org.apache.juneau.urlencoding;
014
015import java.io.*;
016
017import org.apache.juneau.internal.*;
018import org.apache.juneau.serializer.*;

020/**
021 * Specialized writer for serializing UON-encoded text.
022 * <p>
023 * <h5 class='section'>Notes:</h5>
024 * <ul>
025 *    <li>This class is not intended for external use.
026 * </ul>
027 */
028public final class UonWriter extends SerializerWriter {
029
030   private final UonSerializerSession session;
031   private final boolean simpleMode, encodeChars;
032
033   // Characters that do not need to be URL-encoded in strings.
034   private static final AsciiSet unencodedChars = new AsciiSet("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789;/?:@-_.!*'$(),~=");
035
036   // Characters that do not need to be URL-encoded in attribute names.
037   // Identical to unencodedChars, but excludes '='.
038   private static final AsciiSet unencodedCharsAttrName = new AsciiSet("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789;/?:@-_.!*'$(),~");
039
040   // Characters that need to be preceeded with an escape character.
041   private static final AsciiSet escapedChars = new AsciiSet(",()~=");
042
043   // AsciiSet that maps no characters.
044   private static final AsciiSet emptyCharSet = new AsciiSet("");
045
046   private static char[] hexArray = "0123456789ABCDEF".toCharArray();
047
048   /**
049    * Constructor.
050    *
051    * @param session The session that created this writer.
052    * @param out The writer being wrapped.
053    * @param useIndentation If <jk>true</jk>, tabs will be used in output.
054    * @param simpleMode If <jk>true</jk>, type flags will not be generated in output.
055    * @param encodeChars If <jk>true</jk>, special characters should be encoded.
056    * @param trimStrings If <jk>true</jk>, strings should be trimmed before they're serialized.
057    * @param relativeUriBase The base (e.g. <js>https://localhost:9443/contextPath"</js>) for relative URIs (e.g. <js>"my/path"</js>).
058    * @param absolutePathUriBase The base (e.g. <js>https://localhost:9443"</js>) for relative URIs with absolute paths (e.g. <js>"/contextPath/my/path"</js>).
059    */
060   protected UonWriter(UonSerializerSession session, Writer out, boolean useIndentation, boolean simpleMode, boolean encodeChars, boolean trimStrings, String relativeUriBase, String absolutePathUriBase) {
061      super(out, useIndentation, false, trimStrings, '\'', relativeUriBase, absolutePathUriBase);
062      this.session = session;
063      this.simpleMode = simpleMode;
064      this.encodeChars = encodeChars;
065   }
066
067   /**
068    * Serializes the specified simple object as a UON string value.
069    *
070    * @param o The object being serialized.
071    * @param quoteEmptyStrings Special case where we're serializing an array containing an empty string.
072    * @param isTopAttrName If this is a top-level attribute name we're serializing.
073    * @param isTop If this is a top-level value we're serializing.
074    * @return This object (for method chaining).
075    * @throws IOException Should never happen.
076    */
077   protected UonWriter appendObject(Object o, boolean quoteEmptyStrings, boolean isTopAttrName, boolean isTop) throws IOException {
078
079      char typeFlag = 0;
080
081      if (o == null)
082         o = "\u0000";
083      else if (o.equals("\u0000"))
084         typeFlag = 's';
085
086      String s = session.toString(o);
087//    if (trimStrings)
088//       s = s.trim();
089      if (s.isEmpty()) {
090         if (quoteEmptyStrings)
091            typeFlag = 's';
092      } else if (s.charAt(0) == '(' || s.charAt(0) == '$') {
093         typeFlag = 's';
094      } else if (useIndentation && (s.indexOf('\n') != -1 || (s.charAt(0) <= ' ' && s.charAt(0) != 0))) {
095         // Strings containing newline characters must always be quoted so that they're not confused with whitespace.
096         // Also, strings starting with whitespace must be quoted so that the contents are not ignored when whitespace is ignored.
097         typeFlag = 's';
098      } else if (! simpleMode) {
099         if (o instanceof Boolean)
100            typeFlag = 'b';
101         else if (o instanceof Number)
102            typeFlag = 'n';
103      }
104
105      if (typeFlag != 0)
106         startFlag(typeFlag);
107
108      AsciiSet unenc = (isTopAttrName ? unencodedCharsAttrName : unencodedChars);
109      AsciiSet esc = (isTop && typeFlag == 0 ? emptyCharSet : escapedChars);
110
111      for (int i = 0; i < s.length(); i++) {
112         char c = s.charAt(i);
113         if (esc.contains(c))
114            append('~');
115         if ((!encodeChars) || unenc.contains(c))
116            append(c);
117         else {
118            if (c == ' ')
119               append('+');
120            else {
121               int p = s.codePointAt(i);
122               if (p < 0x0080)
123                  appendHex(p);
124               else if (p < 0x0800) {
125                  int p1=p>>>6;
126                  appendHex(p1+192).appendHex((p&63)+128);
127               } else if (p < 0x10000) {
128                  int p1=p>>>6, p2=p1>>>6;
129                  appendHex(p2+224).appendHex((p1&63)+128).appendHex((p&63)+128);
130               } else {
131                  i++;  // Two-byte codepoint...skip past surrogate pair lower byte.
132                  int p1=p>>>6, p2=p1>>>6, p3=p2>>>6;
133                  appendHex(p3+240).appendHex((p2&63)+128).appendHex((p1&63)+128).appendHex((p&63)+128);
134               }
135            }
136         }
137      }
138
139      if (typeFlag != 0)
140         append(')');
141
142      return this;
143   }
144
145   /**
146    * Prints <code>$f(</code> in normal mode, and <code>(</code> in simple mode.
147    *
148    * @param f The flag character.
149    * @return This object (for method chaining).
150    * @throws IOException
151    */
152   protected UonWriter startFlag(char f) throws IOException {
153      if (f != 's' && ! simpleMode)
154         append('$').append(f);
155      append('(');
156      return this;
157   }
158
159   /**
160    * Prints out a two-byte %xx sequence for the given byte value.
161    */
162   private UonWriter appendHex(int b) throws IOException {
163      if (b > 255)
164         throw new IOException("Invalid value passed to appendHex.  Must be in the range 0-255.  Value=" + b);
165      append('%').append(hexArray[b>>>4]).append(hexArray[b&0x0F]);
166      return this;
167   }
168
169   /**
170    * Appends a URI to the output.
171    *
172    * @param uri The URI to append to the output.
173    * @param isTop If this is a top-level value we're serializing.
174    * @return This object (for method chaining).
175    * @throws IOException
176    */
177   public SerializerWriter appendUri(Object uri, boolean isTop) throws IOException {
178      String s = uri.toString();
179      if (s.indexOf("://") == -1) {
180         if (StringUtils.startsWith(s, '/')) {
181            if (absolutePathUriBase != null)
182               append(absolutePathUriBase);
183         } else {
184            if (relativeUriBase != null) {
185               append(relativeUriBase);
186               if (! relativeUriBase.equals("/"))
187                  append("/");
188
189            }
190         }
191      }
192      return appendObject(s, false, false, isTop);
193   }
