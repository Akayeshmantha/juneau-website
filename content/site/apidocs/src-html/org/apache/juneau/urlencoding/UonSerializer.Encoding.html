026/**
027 * Serializes POJO models to UON (a notation for URL-encoded query parameter values).
028 *
029 * <h6 class='topic'>Media types</h6>
030 * <p>
031 *    Handles <code>Accept</code> types: <code>text/uon</code>
032 * <p>
033 *    Produces <code>Content-Type</code> types: <code>text/uon</code>
034 *
035 * <h6 class='topic'>Description</h6>
036 * <p>
037 *    This serializer provides several serialization options.  Typically, one of the predefined DEFAULT serializers will be sufficient.
038 *    However, custom serializers can be constructed to fine-tune behavior.
039 *
040 * <h6 class='topic'>Configurable properties</h6>
041 * <p>
042 *    This class has the following properties associated with it:
043 * <ul>
044 *    <li>{@link UonSerializerContext}
045 *    <li>{@link BeanContext}
046 * </ul>
047 * <p>
048 *    The following shows a sample object defined in Javascript:
049 * </p>
050 * <p class='bcode'>
051 *    {
052 *       id: 1,
053 *       name: <js>'John Smith'</js>,
054 *       uri: <js>'http://sample/addressBook/person/1'</js>,
055 *       addressBookUri: <js>'http://sample/addressBook'</js>,
056 *       birthDate: <js>'1946-08-12T00:00:00Z'</js>,
057 *       otherIds: <jk>null</jk>,
058 *       addresses: [
059 *          {
060 *             uri: <js>'http://sample/addressBook/address/1'</js>,
061 *             personUri: <js>'http://sample/addressBook/person/1'</js>,
062 *             id: 1,
063 *             street: <js>'100 Main Street'</js>,
064 *             city: <js>'Anywhereville'</js>,
065 *             state: <js>'NY'</js>,
066 *             zip: 12345,
067 *             isCurrent: <jk>true</jk>,
068 *          }
069 *       ]
070 *    }
071 * </p>
072 * <p>
073 *    Using the "strict" syntax defined in this document, the equivalent
074 *       UON notation would be as follows:
075 * </p>
076 * <p class='bcode'>
077 *    $o(
078 *       <xa>id</xa>=$n(<xs>1</xs>),
079 *       <xa>name</xa>=<xs>John+Smith</xs>,
080 *       <xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
081 *       <xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
082 *       <xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
083 *       <xa>otherIds</xa>=<xs>%00</xs>,
084 *       <xa>addresses</xa>=$a(
085 *          $o(
086 *             <xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
087 *             <xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
088 *             <xa>id</xa>=$n(<xs>1</xs>),
089 *             <xa>street</xa>=<xs>100+Main+Street</xs>,
090 *             <xa>city</xa>=<xs>Anywhereville</xs>,
091 *             <xa>state</xa>=<xs>NY</xs>,
092 *             <xa>zip</xa>=$n(<xs>12345</xs>),
093 *             <xa>isCurrent</xa>=$b(<xs>true</xs>)
094 *          )
095 *       )
096 *    )
097 * </p>
098 * <p>
099 *    A secondary "lax" syntax is available when the data type of the
100 *       values are already known on the receiving end of the transmission:
101 * </p>
102 * <p class='bcode'>
103 *    (
104 *       <xa>id</xa>=<xs>1</xs>,
105 *       <xa>name</xa>=<xs>John+Smith</xs>,
106 *       <xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
107 *       <xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
108 *       <xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
109 *       <xa>otherIds</xa>=<xs>%00</xs>,
110 *       <xa>addresses</xa>=(
111 *          (
112 *             <xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
113 *             <xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
114 *             <xa>id</xa>=<xs>1</xs>,
115 *             <xa>street</xa>=<xs>100+Main+Street</xs>,
116 *             <xa>city</xa>=<xs>Anywhereville</xs>,
117 *             <xa>state</xa>=<xs>NY</xs>,
118 *             <xa>zip</xa>=<xs>12345</xs>,
119 *             <xa>isCurrent</xa>=<xs>true</xs>
120 *          )
121 *       )
122 *    )
123 * </p>
124 *
125 * <h6 class='topic'>Example:</h6>
126 * <p class='bcode'>
127 *    <jc>// Serialize a Map</jc>
128 *    Map m = <jk>new</jk> ObjectMap(<js>"{a:'b',c:1,d:false,e:['f',1,false],g:{h:'i'}}"</js>);
129 *
130 *    <jc>// Serialize to value equivalent to JSON.</jc>
131 *    <jc>// Produces "$o(a=b,c=$n(1),d=$b(false),e=$a(f,$n(1),$b(false)),g=$o(h=i))"</jc>
132 *    String s = UonSerializer.<jsf>DEFAULT</jsf>.serialize(s);
133 *
134 *    <jc>// Serialize to simplified value (for when data type is already known by receiver).</jc>
135 *    <jc>// Produces "(a=b,c=1,d=false,e=(f,1,false),g=(h=i))"</jc>
136 *    String s = UonSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
137 *
138 *    <jc>// Serialize a bean</jc>
139 *    <jk>public class</jk> Person {
140 *       <jk>public</jk> Person(String s);
141 *       <jk>public</jk> String getName();
142 *       <jk>public int</jk> getAge();
143 *       <jk>public</jk> Address getAddress();
144 *       <jk>public boolean</jk> deceased;
145 *    }
146 *
147 *    <jk>public class</jk> Address {
148 *       <jk>public</jk> String getStreet();
149 *       <jk>public</jk> String getCity();
150 *       <jk>public</jk> String getState();
151 *       <jk>public int</jk> getZip();
152 *    }
153 *
154 *    Person p = <jk>new</jk> Person(<js>"John Doe"</js>, 23, <js>"123 Main St"</js>, <js>"Anywhere"</js>, <js>"NY"</js>, 12345, <jk>false</jk>);
155 *
156 *    <jc>// Produces "$o(name=John Doe,age=23,address=$o(street=123 Main St,city=Anywhere,state=NY,zip=$n(12345)),deceased=$b(false))"</jc>
157 *    String s = UonSerializer.<jsf>DEFAULT</jsf>.serialize(s);
158 *
159 *    <jc>// Produces "(name=John Doe,age=23,address=(street=123 Main St,city=Anywhere,state=NY,zip=12345),deceased=false)"</jc>
160 *    String s = UonSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
161 * </p>
162 */
163@Produces("text/uon")
164public class UonSerializer extends WriterSerializer {

226   /**
227    * Workhorse method. Determines the type of object, and then calls the
228    * appropriate type-specific serialization method.
229    * @param session The context that exist for the duration of a serialize.
230    * @param out The writer to serialize to.
231    * @param o The object being serialized.
232    * @param eType The expected type of the object if this is a bean property.
233    * @param attrName The bean property name if this is a bean property.  <jk>null</jk> if this isn't a bean property being serialized.
234    * @param pMeta The bean property metadata.
235    * @param quoteEmptyStrings <jk>true</jk> if this is the first entry in an array.
236    * @param isTop If we haven't recursively called this method.
237    *
238    * @return The same writer passed in.
239    * @throws Exception
240    */
241   @SuppressWarnings({ "rawtypes", "unchecked" })
242   protected SerializerWriter serializeAnything(UonSerializerSession session, UonWriter out, Object o, ClassMeta<?> eType,
243         String attrName, BeanPropertyMeta pMeta, boolean quoteEmptyStrings, boolean isTop) throws Exception {
244
245      if (o == null) {
246         out.appendObject(null, false, false, isTop);
247         return out;
248      }
249
250      if (eType == null)
251         eType = object();
252
253      boolean addTypeProperty;      // Add "_type" attribute to element?
254      ClassMeta<?> aType;        // The actual type
255      ClassMeta<?> sType;        // The serialized type
256
257      aType = session.push(attrName, o, eType);
258      boolean isRecursion = aType == null;
259
260      // Handle recursion
261      if (aType == null) {
262         o = null;
263         aType = object();
264      }
265
266      sType = aType.getSerializedClassMeta();
267      addTypeProperty = (session.isAddBeanTypeProperties() && ! eType.equals(aType));
268
269      // Swap if necessary
270      PojoSwap swap = aType.getPojoSwap();
271      if (swap != null) {
272         o = swap.swap(session, o);
273
274         // If the getSwapClass() method returns Object, we need to figure out
275         // the actual type now.
276         if (sType.isObject())
277            sType = session.getClassMetaForObject(o);
278      }
279
280      // '\0' characters are considered null.
281      if (o == null || (sType.isChar() && ((Character)o).charValue() == 0))
282         out.appendObject(null, false, false, isTop);
283      else if (sType.isBean())
284         serializeBeanMap(session, out, session.toBeanMap(o), addTypeProperty);
285      else if (sType.isUri() || (pMeta != null && pMeta.isUri()))
286         out.appendUri(o, isTop);
287      else if (sType.isMap()) {
288         if (o instanceof BeanMap)
289            serializeBeanMap(session, out, (BeanMap)o, addTypeProperty);
290         else
291            serializeMap(session, out, (Map)o, eType);
292      }
293      else if (sType.isCollection()) {
294         serializeCollection(session, out, (Collection) o, eType);
295      }
296      else if (sType.isArray()) {
297         serializeCollection(session, out, toList(sType.getInnerClass(), o), eType);
298      }
299      else {
300         out.appendObject(o, quoteEmptyStrings, false, isTop);
301      }
302
303      if (! isRecursion)
304         session.pop();
305      return out;
306   }

308   @SuppressWarnings({ "rawtypes", "unchecked" })
309   private SerializerWriter serializeMap(UonSerializerSession session, UonWriter out, Map m, ClassMeta<?> type) throws Exception {
310
311      m = session.sort(m);
312
313      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
314
315      int depth = session.getIndent();
316      out.startFlag('o');
317
318      Iterator mapEntries = m.entrySet().iterator();
319
320      while (mapEntries.hasNext()) {
321         Map.Entry e = (Map.Entry) mapEntries.next();
322         Object value = e.getValue();
323         Object key = session.generalize(e.getKey(), keyType);
324         out.cr(depth).appendObject(key, session.isUseWhitespace(), false, false).append('=');
325         serializeAnything(session, out, value, valueType, (key == null ? null : session.toString(key)), null, session.isUseWhitespace(), false);
326         if (mapEntries.hasNext())
327            out.append(',');
328      }
329
330      if (m.size() > 0)
331         out.cr(depth-1);
332      out.append(')');
333
334      return out;
335   }
336
337   private SerializerWriter serializeBeanMap(UonSerializerSession session, UonWriter out, BeanMap<?> m, boolean addTypeProperty) throws Exception {
338      int depth = session.getIndent();
339
340      out.startFlag('o');
341
342      boolean addComma = false;
343
344      for (BeanPropertyValue p : m.getValues(session.isTrimNulls(), addTypeProperty ? session.createBeanTypeNameProperty(m) : null)) {
345         BeanPropertyMeta pMeta = p.getMeta();
346         ClassMeta<?> cMeta = p.getClassMeta();
347
348         String key = p.getName();
349         Object value = p.getValue();
350         Throwable t = p.getThrown();
351         if (t != null)
352            session.addBeanGetterWarning(pMeta, t);
353
354         if (session.canIgnoreValue(cMeta, key, value))
355            continue;
356
357         if (addComma)
358            out.append(',');
359
360         out.cr(depth).appendObject(key, false, false, false).append('=');
361
362         serializeAnything(session, out, value, cMeta, key, pMeta, false, false);
363
364         addComma = true;
365      }
366
367      if (m.size() > 0)
368         out.cr(depth-1);
369      out.append(')');
370
371      return out;
372   }
373
374   @SuppressWarnings({ "rawtypes", "unchecked" })
375   private SerializerWriter serializeCollection(UonSerializerSession session, UonWriter out, Collection c, ClassMeta<?> type) throws Exception {
376
377      ClassMeta<?> elementType = type.getElementType();
378
379      c = session.sort(c);
380
381      out.startFlag('a');
382
383      int depth = session.getIndent();
384      boolean quoteEmptyString = (c.size() == 1 || session.isUseWhitespace());
385
386      for (Iterator i = c.iterator(); i.hasNext();) {
387         out.cr(depth);
388         serializeAnything(session, out, i.next(), elementType, "<iterator>", null, quoteEmptyString, false);
389         if (i.hasNext())
390            out.append(',');
391      }
392
393      if (c.size() > 0)
394         out.cr(depth-1);
395      out.append(')');
396
397      return out;
398   }

404   @Override /* Serializer */
405   public UonSerializerSession createSession(Object output, ObjectMap op, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType) {
406      return new UonSerializerSession(getContext(UonSerializerContext.class), op, output, javaMethod, locale, timeZone, mediaType);
407   }

409   @Override /* Serializer */
410   protected void doSerialize(SerializerSession session, Object o) throws Exception {
411      UonSerializerSession s = (UonSerializerSession)session;
412      serializeAnything(s, s.getWriter(), o, null, "root", null, false, true);
413   }
