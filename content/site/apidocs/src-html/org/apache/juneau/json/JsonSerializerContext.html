013package org.apache.juneau.json;
014
015import org.apache.juneau.*;
016import org.apache.juneau.serializer.*;
017
018/**
019 * Configurable properties on the {@link JsonSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * See {@link ContextFactory} for more information about context properties.
025 *
026 * <h5 class='section'>Inherited configurable properties:</h5>
027 * <ul class='javahierarchy'>
028 *    <li class='c'><a class="doclink" href="../BeanContext.html#ConfigProperties">BeanContext</a> - Properties associated with handling beans on serializers and parsers.
029 *    <ul>
030 *       <li class='c'><a class="doclink" href="../serializer/SerializerContext.html#ConfigProperties">SerializerContext</a> - Configurable properties common to all serializers.
031 *    </ul>
032 * </ul>
033 */
034public final class JsonSerializerContext extends SerializerContext {
035
036   /**
037    * <b>Configuration property:</b>  Simple JSON mode.
038    * <p>
039    * <ul>
040    *    <li><b>Name:</b> <js>"JsonSerializer.simpleMode"</js>
041    *    <li><b>Data type:</b> <code>Boolean</code>
042    *    <li><b>Default:</b> <jk>false</jk>
043    *    <li><b>Session-overridable:</b> <jk>true</jk>
044    * </ul>
045    * <p>
046    * If <jk>true</jk>, JSON attribute names will only be quoted when necessary.
047    * Otherwise, they are always quoted.
048    */
049   public static final String JSON_simpleMode = "JsonSerializer.simpleMode";
050
051   /**
052    * <b>Configuration property:</b>  Prefix solidus <js>'/'</js> characters with escapes.
053    * <p>
054    * <ul>
055    *    <li><b>Name:</b> <js>"JsonSerializer.escapeSolidus"</js>
056    *    <li><b>Data type:</b> <code>Boolean</code>
057    *    <li><b>Default:</b> <jk>false</jk>
058    *    <li><b>Session-overridable:</b> <jk>true</jk>
059    * </ul>
060    * <p>
061    * If <jk>true</jk>, solidus (e.g. slash) characters should be escaped.
062    * The JSON specification allows for either format.
063    * However, if you're embedding JSON in an HTML script tag, this setting prevents
064    *    confusion when trying to serialize <xt>&lt;\/script&gt;</xt>.
065    */
066   public static final String JSON_escapeSolidus = "JsonSerializer.escapeSolidus";
067
068
069   final boolean
070      simpleMode,
071      escapeSolidus;
072
073   /**
074    * Constructor.
075    * <p>
076    * Typically only called from {@link ContextFactory#getContext(Class)}.
077    *
078    * @param cf The factory that created this context.
079    */
080   public JsonSerializerContext(ContextFactory cf) {
081      super(cf);
082      simpleMode = cf.getProperty(JSON_simpleMode, boolean.class, false);
083      escapeSolidus = cf.getProperty(JSON_escapeSolidus, boolean.class, false);
084   }
085
086   @Override /* Context */
087   public ObjectMap asMap() {
088      return super.asMap()
089         .append("JsonSerializerContext", new ObjectMap()
090            .append("simpleMode", simpleMode)
091            .append("escapeSolidus", escapeSolidus)
092         );
093   }
094}
