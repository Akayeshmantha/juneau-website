013package org.apache.juneau.json;
014
015import static org.apache.juneau.internal.ClassUtils.*;
016import static org.apache.juneau.serializer.SerializerContext.*;
017
018import java.lang.reflect.*;
019import java.util.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.annotation.*;
023import org.apache.juneau.serializer.*;
024import org.apache.juneau.transform.*;
025
026/**
027 * Serializes POJO metadata to HTTP responses as JSON.
028 *
029 * <h5 class='section'>Media types:</h5>
030 * <p>
031 * Handles <code>Accept</code> types: <code>application/json+schema, text/json+schema</code>
032 * <p>
033 * Produces <code>Content-Type</code> types: <code>application/json</code>
034 *
035 * <h5 class='section'>Description:</h5>
036 * <p>
037 * Produces the JSON-schema for the JSON produced by the {@link JsonSerializer} class with the same properties.
038 */
039@Produces(value="application/json+schema,text/json+schema",contentType="application/json")
040public final class JsonSchemaSerializer extends JsonSerializer {
041
042   private final JsonSerializerContext ctx;
043
044   /**
045    * Constructor.
046    * @param propertyStore Initialize with the specified config property store.
047    */
048   public JsonSchemaSerializer(PropertyStore propertyStore) {
049      this(propertyStore, null);
050   }
051
052   /**
053    * Constructor.
054    * @param propertyStore Initialize with the specified config property store.
055    * @param overrideProperties
056    */
057   public JsonSchemaSerializer(PropertyStore propertyStore, Map<String,Object> overrideProperties) {
058      super(propertyStore);
059      this.ctx = this.propertyStore.create(overrideProperties).getContext(JsonSerializerContext.class);
060   }
061
062   @Override /* CoreObject */
063   protected ObjectMap getOverrideProperties() {
064      return super.getOverrideProperties().append(SERIALIZER_detectRecursions, true).append(SERIALIZER_ignoreRecursions, true);
065   }
066
067
068   //--------------------------------------------------------------------------------
069   // Entry point methods
070   //--------------------------------------------------------------------------------
071
072   @Override /* Serializer */
073   public JsonSerializerSession createSession(Object output, ObjectMap op, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType) {
074      return new JsonSerializerSession(ctx, op, output, javaMethod, locale, timeZone, mediaType);
075   }
076
077   @Override /* JsonSerializer */
078   protected void doSerialize(SerializerSession session, Object o) throws Exception {
079      JsonSerializerSession s = (JsonSerializerSession)session;
080      ObjectMap schema = getSchema(s, session.getClassMetaForObject(o), "root", null);
081      serializeAnything(s, s.getWriter(), schema, null, "root", null);
082   }
083
084   /*
085    * Creates a schema representation of the specified class type.
086    *
087    * @param eType The class type to get the schema of.
088    * @param ctx Serialize context used to prevent infinite loops.
089    * @param attrName The name of the current attribute.
090    * @return A schema representation of the specified class.
091    * @throws SerializeException If a problem occurred trying to convert the output.
092    */
093   @SuppressWarnings({ "unchecked", "rawtypes" })
094   private ObjectMap getSchema(JsonSerializerSession session, ClassMeta<?> eType, String attrName, String[] pNames) throws Exception {
095      ObjectMap out = new ObjectMap();
096
097      if (eType == null)
098         eType = object();
099
100      ClassMeta<?> aType;        // The actual type (will be null if recursion occurs)
101      ClassMeta<?> sType;        // The serialized type
102
103      aType = session.push(attrName, eType, null);
104
105      sType = eType.getSerializedClassMeta();
106      String type = null;
107
108      if (sType.isEnum() || sType.isCharSequence() || sType.isChar())
109         type = "string";
110      else if (sType.isNumber())
111         type = "number";
112      else if (sType.isBoolean())
113         type = "boolean";
114      else if (sType.isMapOrBean())
115         type = "object";
116      else if (sType.isCollectionOrArray())
117         type = "array";
118      else
119         type = "any";
120
121      out.put("type", type);
122      out.put("description", eType.toString());
123      PojoSwap f = eType.getPojoSwap();
124      if (f != null)
125         out.put("transform", f);
126
127      if (aType != null) {
128         if (sType.isEnum())
129            out.put("enum", getEnumStrings((Class<Enum<?>>)sType.getInnerClass()));
130         else if (sType.isCollectionOrArray()) {
131            ClassMeta componentType = sType.getElementType();
132            if (sType.isCollection() && isParentClass(Set.class, sType.getInnerClass()))
133               out.put("uniqueItems", true);
134            out.put("items", getSchema(session, componentType, "items", pNames));
135         } else if (sType.isBean()) {
136            ObjectMap properties = new ObjectMap();
137            BeanMeta bm = session.getBeanMeta(sType.getInnerClass());
138            if (pNames != null)
139               bm = new BeanMetaFiltered(bm, pNames);
140            for (Iterator<BeanPropertyMeta> i = bm.getPropertyMetas().iterator(); i.hasNext();) {
141               BeanPropertyMeta p = i.next();
142               properties.put(p.getName(), getSchema(session, p.getClassMeta(), p.getName(), p.getProperties()));
143            }
144            out.put("properties", properties);
145         }
146      }
147      session.pop();
148      return out;
149   }
150
151   @SuppressWarnings({ "unchecked", "rawtypes" })
152   private static List<String> getEnumStrings(Class<? extends Enum> c) {
153      List<String> l = new LinkedList<String>();
154      for (Object e : EnumSet.allOf(c))
155         l.add(e.toString());
156      return l;
157   }
158}
