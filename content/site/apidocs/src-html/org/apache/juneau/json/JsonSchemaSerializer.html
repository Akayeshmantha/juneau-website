013package org.apache.juneau.json;
014
015import static org.apache.juneau.internal.ClassUtils.*;
016import static org.apache.juneau.serializer.SerializerContext.*;
017
018import java.util.*;
019
020import org.apache.juneau.*;
021import org.apache.juneau.annotation.*;
022import org.apache.juneau.serializer.*;
023import org.apache.juneau.transform.*;
024
025/**
026 * Serializes POJO metadata to HTTP responses as JSON.
027 *
028 * <h6 class='topic'>Media types</h6>
029 * <p>
030 *    Handles <code>Accept</code> types: <code>application/json+schema, text/json+schema</code>
031 * <p>
032 *    Produces <code>Content-Type</code> types: <code>application/json</code>
033 *
034 * <h6 class='topic'>Description</h6>
035 * <p>
036 *    Produces the JSON-schema for the JSON produced by the {@link JsonSerializer} class with the same properties.
037 */
038@Produces(value="application/json+schema,text/json+schema",contentType="application/json")
039public final class JsonSchemaSerializer extends JsonSerializer {
040
041   /**
042    * Constructor.
043    */
044   public JsonSchemaSerializer() {
045      setProperty(SERIALIZER_detectRecursions, true);
046      setProperty(SERIALIZER_ignoreRecursions, true);
047   }
048
049   /**
050    * Constructor.
051    *
052    * @param config Initialize with the specified config property store.
053    */
054   public JsonSchemaSerializer(ContextFactory config) {
055      getContextFactory().copyFrom(config);
056      setProperty(SERIALIZER_detectRecursions, true);
057      setProperty(SERIALIZER_ignoreRecursions, true);
058   }
059
060   //--------------------------------------------------------------------------------
061   // Overridden methods
062   //--------------------------------------------------------------------------------
063
064   @Override /* JsonSerializer */
065   protected void doSerialize(SerializerSession session, Object o) throws Exception {
066      JsonSerializerSession s = (JsonSerializerSession)session;
067      ObjectMap schema = getSchema(s, session.getClassMetaForObject(o), "root", null);
068      serializeAnything(s, s.getWriter(), schema, null, "root", null);
069   }
070
071   /*
072    * Creates a schema representation of the specified class type.
073    *
074    * @param eType The class type to get the schema of.
075    * @param ctx Serialize context used to prevent infinite loops.
076    * @param attrName The name of the current attribute.
077    * @return A schema representation of the specified class.
078    * @throws SerializeException If a problem occurred trying to convert the output.
079    */
080   @SuppressWarnings({ "unchecked", "rawtypes" })
081   private ObjectMap getSchema(JsonSerializerSession session, ClassMeta<?> eType, String attrName, String[] pNames) throws Exception {
082      ObjectMap out = new ObjectMap();
083
084      if (eType == null)
085         eType = object();
086
087      ClassMeta<?> aType;        // The actual type (will be null if recursion occurs)
088      ClassMeta<?> sType;        // The serialized type
089
090      aType = session.push(attrName, eType, null);
091
092      sType = eType.getSerializedClassMeta();
093      String type = null;
094
095      if (sType.isEnum() || sType.isCharSequence() || sType.isChar())
096         type = "string";
097      else if (sType.isNumber())
098         type = "number";
099      else if (sType.isBoolean())
100         type = "boolean";
101      else if (sType.isMapOrBean())
102         type = "object";
103      else if (sType.isCollectionOrArray())
104         type = "array";
105      else
106         type = "any";
107
108      out.put("type", type);
109      out.put("description", eType.toString());
110      PojoSwap f = eType.getPojoSwap();
111      if (f != null)
112         out.put("transform", f);
113
114      if (aType != null) {
115         if (sType.isEnum())
116            out.put("enum", getEnumStrings((Class<Enum<?>>)sType.getInnerClass()));
117         else if (sType.isCollectionOrArray()) {
118            ClassMeta componentType = sType.getElementType();
119            if (sType.isCollection() && isParentClass(Set.class, sType.getInnerClass()))
120               out.put("uniqueItems", true);
121            out.put("items", getSchema(session, componentType, "items", pNames));
122         } else if (sType.isBean()) {
123            ObjectMap properties = new ObjectMap();
124            BeanMeta bm = session.getBeanMeta(sType.getInnerClass());
125            if (pNames != null)
126               bm = new BeanMetaFiltered(bm, pNames);
127            for (Iterator<BeanPropertyMeta> i = bm.getPropertyMetas().iterator(); i.hasNext();) {
128               BeanPropertyMeta p = i.next();
129               properties.put(p.getName(), getSchema(session, p.getClassMeta(), p.getName(), p.getProperties()));
130            }
131            out.put("properties", properties);
132         }
133      }
134      session.pop();
135      return out;
136   }
137
138   @SuppressWarnings({ "unchecked", "rawtypes" })
139   private List<String> getEnumStrings(Class<? extends Enum> c) {
140      List<String> l = new LinkedList<String>();
141      for (Object e : EnumSet.allOf(c))
142         l.add(e.toString());
143      return l;
144   }
145
146
147   @Override /* Lockable */
148   public JsonSchemaSerializer lock() {
149      super.lock();
150      return this;
151   }
152}
