013package org.apache.juneau.annotation;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019import java.util.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.transform.*;
023
024/**
025 * Used tailor how bean properties get interpreted by the framework.
026 * <p>
027 * Can be used to do the following:
028 * <ul class='spaced-list'>
029 *    <li>Override the name of a property.
030 *    <li>Identify a getter or setter with a non-standard naming convention.
031 *    <li>Identify a specific subclass for a property with a general class type.
032 *    <li>Identify class types of elements in properties of type <code>Collection</code> or <code>Map</code>.
033 *    <li>Hide properties during serialization.
034 *    <li>Associate transforms with bean property values, such as a POJO swap to convert a <code>Calendar</code> field to a string.
035 *    <li>Override the list of properties during serialization on child elements of a property of type <code>Collection</code> or <code>Map</code>.
036 *    <li>Identify a property as the URL for a bean.
037 *    <li>Identify a property as the ID for a bean.
038 * </ul>
039 * <p>
040 * This annotation is applied to public fields and public getter/setter methods of beans.
041 */
042@Documented
043@Target({FIELD,METHOD})
044@Retention(RUNTIME)
045@Inherited
046public @interface BeanProperty {
047
048   /**
049    * Identifies the name of the property.
050    * <p>
051    * Normally, this is automatically inferred from the field name or getter method name
052    *    of the property.  However, this property can be used to assign a different
053    *    property name from the automatically inferred value.
054    * <p>
055    * If the {@link BeanContext#BEAN_beanFieldVisibility} setting on the bean context excludes this field (e.g. the visibility
056    *    is set to PUBLIC, but the field is PROTECTED), this annotation can be used to force the field to be identified as a property.
057    * <p>
058    * <h6 class='topic'>Dynamic beans</h6>
059    * The bean property named <js>"*"</js> is the designated "dynamic property" which allows for "extra" bean properties not otherwise defined.
060    * This is similar in concept to the Jackson <ja>@JsonGetterAll</ja> and <ja>@JsonSetterAll</ja> annotations.
061    * The primary purpose is for backwards compatibility in parsing newer streams with addition information into older beans.
062    * <p>
063    * The following examples show how to define dynamic bean properties.
064    * <p class='bcode'>
065    *    <jc>// Option #1 - A simple public Map field.
066    *    // The field name can be anything.</jc>
067    *    <jk>public class</jk> BeanWithDynaField {
068    *
069    *       <ja>@BeanProperty</ja>(name=<js>"*"</js>)
070    *       <jk>public</jk> Map&lt;String,Object&gt; extraStuff = <jk>new</jk> LinkedHashMap&lt;String,Object&gt;();
071    *    }
072    *
073    *    <jc>// Option #2 - Getters and setters.
074    *    // Method names can be anything.
075    *    // Getter must return a Map with String keys.
076    *    // Setter must take in two arguments.</jc>
077    *    <jk>public class</jk> BeanWithDynaMethods {
078    *
079    *       <ja>@BeanProperty</ja>(name=<js>"*"</js>)
080    *       <jk>public</jk> Map&lt;String,Object&gt; getMyExtraStuff() {
081    *          ...
082    *       }
083    *
084    *       <ja>@BeanProperty</ja>(name=<js>"*"</js>)
085    *       <jk>public void</jk> setAnExtraField(String name, Object value) {
086    *          ...
087    *       }
088    *    }
089    *
090    *    <jc>// Option #3 - Getter only.
091    *    // Properties will be added through the getter.</jc>
092    *    <jk>public class</jk> BeanWithDynaGetterOnly {
093    *
094    *       <ja>@BeanProperty</ja>(name=<js>"*"</js>)
095    *       <jk>public</jk> Map&lt;String,Object&gt; getMyExtraStuff() {
096    *          ...
097    *       }
098    *    }
099    * </p>
100    * <p>
101    * Similar rules apply for value types and swaps.  The property values optionally can be any serializable type
102    * or use swaps.
103    * <p class='bcode'>
104    *    <jc>// A serializable type other than Object.</jc>
105    *    <jk>public class</jk> BeanWithDynaFieldWithListValues {
106    *
107    *       <ja>@BeanProperty</ja>(name=<js>"*"</js>)
108    *       <jk>public</jk> Map&lt;String,List&lt;String&gt;&gt; getMyExtraStuff() {
109    *          ...
110    *       }
111    *    }
112    *
113    *    <jc>// A swapped value.</jc>
114    *    <jk>public class</jk> BeanWithDynaFieldWithSwappedValues {
115    *
116    *       <ja>@BeanProperty</ja>(name=<js>"*"</js>, swap=CalendarSwap.<jsf>ISO8601DTZ</jsf>.<jk>class</jk>)
117    *       <jk>public</jk> Map&lt;String,Calendar&gt; getMyExtraStuff() {
118    *          ...
119    *       }
120    *    }
121    * </p>
122    * <ul class='doctree'>
123    *    <li class='info'>
124    *       Note that if you're not interested in these additional properties, you can also use the {@link BeanContext#BEAN_ignoreUnknownBeanProperties} setting
125    *       to ignore values that don't fit into existing properties.
126    * </ul>
127    */
128   String name() default "";
129
130   /**
131    * A synonym for {@link #name()}.
132    * <p>
133    * The following annotations are equivalent:
134    * <p>
135    * <p class='bcode'>
136    *    <ja>@BeanProperty</ja>(name=<js>"foo"</js>)
137    *
138    *    <ja>@BeanProperty</ja>(<js>"foo"</js>)
139    * </p>
140    */
141   String value() default "";
142
143   /**
144    * Identifies a specialized class type for the property.
145    * <p>
146    * Normally this can be inferred through reflection of the field type or getter return type.
147    * However, you'll want to specify this value if you're parsing beans where the bean property class
148    *    is an interface or abstract class to identify the bean type to instantiate.  Otherwise, you may
149    *    cause an {@link InstantiationException} when trying to set these fields.
150    * <p>
151    * This property must denote a concrete bean class with a no-arg constructor.
152    *
153    * <h5 class='section'>Example:</h5>
154    * <p class='bcode'>
155    *    <jk>public class</jk> MyBean {
156    *
157    *       <jc>// Identify concrete map type.</jc>
158    *       <ja>@BeanProperty</ja>(type=HashMap.<jk>class</jk>)
159    *       <jk>public</jk> Map <jf>p1</jf>;
160    *    }
161    * </p>
162    */
163   Class<?> type() default Object.class;
164
165   /**
166    * For bean properties of maps and collections, this annotation can be used to identify
167    * the class types of the contents of the bean property object when the generic parameter
168    * types are interfaces or abstract classes.
169    *
170    * <h5 class='section'>Example:</h5>
171    * <p class='bcode'>
172    *    <jk>public class</jk> MyBean {
173    *
174    *       <jc>// Identify concrete map type with String keys and Integer values.</jc>
175    *       <ja>@BeanProperty</ja>(type=HashMap.<jk>class</jk>, params={String.<jk>class</jk>,Integer.<jk>class</jk>})
176    *       <jk>public</jk> Map <jf>p1</jf>;
177    *    }
178    * </p>
179    */
180   Class<?>[] params() default {};
181
182   /**
183    * Associates a {@link PojoSwap} or {@link SurrogateSwap} with this bean property that will swap the value object
184    *    with another object during serialization and parsing.
185    * <p>
186    * This annotation supersedes any swaps associated with the bean property type
187    *    class itself.
188    * <p>
189    * Typically used for rendering {@link Date Dates} and {@link Calendar Calendars}
190    *    as a particular string format.
191    *
192    * <h5 class='section'>Example:</h5>
193    * <p class='bcode'>
194    *    <jk>public class</jk> MyClass {
195    *
196    *       <jc>// During serialization, convert to ISO8601 date-time string.</jc>
197    *       <ja>@BeanProperty</ja>(pojoSwap=CalendarSwap.ISO8601DT.<jk>class</jk>)
198    *       <jk>public</jk> Calendar getTime();
199    *    }
200    * </p>
201    */
202   Class<?> swap() default Null.class;
203
204   /**
205    * Used to limit which child properties are rendered by the serializers.
206    * <p>
207    * Can be used on any of the following bean property types:
208    * <ul class='spaced-list'>
209    *    <li>Beans - Only render the specified properties of the bean.
210    *    <li>Maps - Only render the specified entries in the map.
211    *    <li>Bean/Map arrays - Same, but applied to each element in the array.
212    *    <li>Bean/Map collections - Same, but applied to each element in the collection.
213    * </ul>
214    *
215    * <h5 class='section'>Example:</h5>
216    * <p class='bcode'>
217    *    <jk>public class</jk> MyClass {
218    *
219    *       <jc>// Only render 'f1' when serializing this bean property.</jc>
220    *       <ja>@BeanProperty</ja>(properties=<js>"f1"</js>)
221    *       <jk>public</jk> MyChildClass x1 = <jk>new</jk> MyChildClass();
222    *    }
223    *
224    *    <jk>public class</jk> MyChildClass {
225    *       <jk>public int</jk> f1 = 1;
226    *       <jk>public int</jk> f2 = 2;
227    *    }
228    *
229    *    <jc>// Renders "{x1:{f1:1}}"</jc>
230    *    String json = JsonSerializer.<jsf>DEFAULT</jsf>.serialize(<jk>new</jk> MyClass());
231    * </p>
232    */
233   String properties() default "";
234
235   /**
236    * The list of classes that make up the bean dictionary for this bean property.
237    * <p>
238    * The dictionary is a name/class mapping used to find class types during parsing when they cannot be inferred through reflection.
239    * The names are defined through the {@link Bean#typeName()} annotation defined on the bean class.
240    * <p>
241    * This list can consist of the following class types:
242    * <ul>
243    *    <li>Any bean class that specifies a value for {@link Bean#typeName() @Bean.name()};
244    *    <li>Any subclass of {@link BeanDictionaryList} that defines an entire set of mappings.
245    *       Note that the subclass MUST implement a no-arg constructor so that it can be instantiated.
246    *    <li>Any subclass of {@link BeanDictionaryMap} that defines an entire set of mappings.
247    *       Note that the subclass MUST implement a no-arg constructor so that it can be instantiated.
248    * </ul>
249    */
250   Class<?>[] beanDictionary() default {};
251}
