013package org.apache.juneau.server.jaxrs;
014
015import static javax.servlet.http.HttpServletResponse.*;
016
017import java.io.*;
018import java.lang.annotation.*;
019import java.lang.reflect.*;
020
021import javax.ws.rs.*;
022import javax.ws.rs.core.*;
023import javax.ws.rs.ext.*;
024
025import org.apache.juneau.*;
026import org.apache.juneau.internal.*;
027import org.apache.juneau.parser.*;
028import org.apache.juneau.serializer.*;
029import org.apache.juneau.server.annotation.*;
030
031/**
032 * Base class for defining JAX-RS providers based on Juneau serializers and parsers.
033 *
034 * @author James Bognar (james.bognar@salesforce.com)
035 */
036public class BaseProvider implements MessageBodyReader<Object>, MessageBodyWriter<Object> {
037
038   private SerializerGroup serializers = new SerializerGroup();
039   private ParserGroup parsers = new ParserGroup();
040   private ObjectMap properties = new ObjectMap();
041
042   /**
043    * Constructor.
044    */
045   protected BaseProvider() {
046      try {
047         properties = new ObjectMap();
048         JuneauProvider jp = getClass().getAnnotation(JuneauProvider.class);
049         serializers.append(jp.serializers());
050         parsers.append(jp.parsers());
051         for (Property p : jp.properties())
052            properties.put(p.name(), p.value());
053         serializers.addBeanFilters(jp.beanFilters());
054         parsers.addBeanFilters(jp.beanFilters());
055         serializers.addPojoSwaps(jp.pojoSwaps());
056         parsers.addPojoSwaps(jp.pojoSwaps());
057      } catch (Exception e) {
058         throw new RuntimeException(e);
059      }
060   }
061
062   /**
063    * Returns properties defined on the specified method through the {@link RestMethod#properties()}
064    *    annotation specified on the method and the {@link JuneauProvider#properties()} annotation
065    *    specified on the provider class.
066    *
067    * @param a All annotations defined on the method.
068    * @return A map of all properties define on the method.
069    */
070   protected ObjectMap getMethodProperties(Annotation[] a) {
071      ObjectMap m = new ObjectMap().setInner(properties);
072      for (Annotation aa : a) {
073         if (aa instanceof RestMethod) {
074            for (Property p : ((RestMethod)aa).properties())
075               m.put(p.name(), p.value());
076         }
077      }
078      return m;
079   }
080
081   @Override /* MessageBodyWriter */
082   public long getSize(Object o, Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
083      return -1;
084   }
085
086   @Override /* MessageBodyWriter */
087   public boolean isWriteable(Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
088      return serializers.findMatch(mediaType.toString()) != null;
089   }
090
091   @Override /* MessageBodyWriter */
092   public void writeTo(Object o, Class<?> type, Type gType, Annotation[] a, MediaType mediaType,
093         MultivaluedMap<String,Object> headers, OutputStream out) throws IOException, WebApplicationException {
094      try {
095         String mt = serializers.findMatch(mediaType.toString());
096         if (mt == null)
097            throw new WebApplicationException(SC_NOT_ACCEPTABLE);
098         Serializer s = serializers.getSerializer(mt);
099         ObjectMap mp = getMethodProperties(a);
100         mp.append("mediaType", mediaType.toString());
101         if (s.isWriterSerializer()) {
102            WriterSerializer s2 = (WriterSerializer)s;
103            OutputStreamWriter w = new OutputStreamWriter(out, IOUtils.UTF8);
104            SerializerSession session = s.createSession(w, mp, null);
105            s2.serialize(session, o);
106            w.flush();
107            w.close();
108         } else {
109            OutputStreamSerializer s2 = (OutputStreamSerializer)s;
110            SerializerSession session = s.createSession(s2, mp, null);
111            s2.serialize(session, o);
112            out.flush();
113            out.close();
114         }
115      } catch (SerializeException e) {
116         throw new IOException(e);
117      }
118   }
119
120   @Override /* MessageBodyReader */
121   public boolean isReadable(Class<?> type, Type gType, Annotation[] a, MediaType mediaType) {
122      return parsers.findMatch(mediaType.toString()) != null;
123   }
124
125   @Override /* MessageBodyReader */
126   public Object readFrom(Class<Object> type, Type gType, Annotation[] a, MediaType mediaType,
127         MultivaluedMap<String,String> headers, InputStream in) throws IOException, WebApplicationException {
128      try {
129         String mt = parsers.findMatch(mediaType.toString());
130         if (mt == null)
131            throw new WebApplicationException(SC_UNSUPPORTED_MEDIA_TYPE);
132         Parser p = parsers.getParser(mt);
133         BeanContext bc = p.getBeanContext();
134         ClassMeta<?> cm = bc.getClassMeta(gType);
135         ObjectMap mp = getMethodProperties(a);
136         mp.put("mediaType", mediaType.toString());
137         if (p.isReaderParser()) {
138            ReaderParser p2 = (ReaderParser)p;
139            InputStreamReader r = new InputStreamReader(in, IOUtils.UTF8);
140            ParserSession session = p2.createSession(r, mp, null, null);
141            return p2.parse(session, cm);
142         }
143         InputStreamParser p2 = (InputStreamParser)p;
144         ParserSession session = p2.createSession(in, mp, null, null);
145         return p2.parse(session, cm);
146      } catch (ParseException e) {
147         throw new IOException(e);
148      }
149   }
150}
