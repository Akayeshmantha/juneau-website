154   @Override /* Servlet */
155   public synchronized void init(ServletConfig servletConfig) throws ServletException {
156      try {
157         log(FINE, "Servlet {0} init called.", getClass().getName());
158         this.servletConfig = servletConfig;
159
160         if (isInitialized)
161            return;
162
163         super.init(servletConfig);
164
165         // Find resource resource bundle location.
166         for (Map.Entry<Class<?>,RestResource> e : restResourceAnnotationsChildFirst.entrySet()) {
167            Class<?> c = e.getKey();
168            RestResource r = e.getValue();
169            if (! r.messages().isEmpty()) {
170               if (msgs == null)
171                  msgs = new MessageBundle(c, r.messages());
172               else
173                  msgs.addSearchPath(c, r.messages());
174            }
175         }
176         for (RestResource r : restResourceAnnotationsParentFirst.values()) {
177            if (! r.title().isEmpty())
178               title = r.title();
179            if (! r.description().isEmpty())
180               description = r.description();
181            if (! r.clientVersionHeader().isEmpty())
182               clientVersionHeader = r.clientVersionHeader();
183            if (! r.termsOfService().isEmpty())
184               termsOfService = r.termsOfService();
185            if (! r.contact().isEmpty())
186               contact = r.contact();
187            if (! r.license().isEmpty())
188               license = r.license();
189            if (! r.version().isEmpty())
190               version = r.version();
191            if (! r.tags().isEmpty())
192               tags = r.tags();
193            if (! r.externalDocs().isEmpty())
194               externalDocs = r.externalDocs();
195         }
196
197         if (msgs == null)
198            msgs = new MessageBundle(this.getClass(), "");
199         if (clientVersionHeader.isEmpty())
200            clientVersionHeader = "X-Client-Version";
201
202         styleSheet = createStyleSheet();
203         favIcon = createFavIcon();
204         staticFilesMap = Collections.unmodifiableMap(createStaticFilesMap());
205         staticFilesPrefixes = staticFilesMap.keySet().toArray(new String[0]);
206
207         properties = createProperties();
208         beanFilters = createBeanFilters();
209         pojoSwaps = createPojoSwaps();
210         context = ContextFactory.create().setProperties(properties).getContext(RestServletContext.class);
211         beanContext = createBeanContext(properties, beanFilters, pojoSwaps);
212         urlEncodingSerializer = createUrlEncodingSerializer(properties, beanFilters, pojoSwaps).lock();
213         urlEncodingParser = createUrlEncodingParser(properties, beanFilters, pojoSwaps).lock();
214         serializers = createSerializers(properties, beanFilters, pojoSwaps).lock();
215         parsers = createParsers(properties, beanFilters, pojoSwaps).lock();
216         converters = createConverters(properties);
217         encoders = createEncoders(properties);
218         guards = createGuards(properties);
219         mimetypesFileTypeMap = createMimetypesFileTypeMap(properties);
220         defaultRequestHeaders = new TreeMap<String,String>(String.CASE_INSENSITIVE_ORDER);
221         defaultRequestHeaders.putAll(createDefaultRequestHeaders(properties));
222         defaultResponseHeaders = createDefaultResponseHeaders(properties);
223         responseHandlers = createResponseHandlers(properties);
224
225         // Discover the @RestMethod methods available on the resource.
226         List<String> methodsFound = new LinkedList<String>();   // Temporary to help debug transient duplicate method issue.
227         for (java.lang.reflect.Method method : this.getClass().getMethods()) {
228            if (method.isAnnotationPresent(RestMethod.class)) {
229               RestMethod a = method.getAnnotation(RestMethod.class);
230               methodsFound.add(method.getName() + "," + a.name() + "," + a.path());
231               try {
232                  if (! Modifier.isPublic(method.getModifiers()))
233                     throw new RestServletException("@RestMethod method {0}.{1} must be defined as public.", this.getClass().getName(), method.getName());
234
235                  MethodMeta sm = new MethodMeta(method);
236                  javaRestMethods.put(method.getName(), sm);
237                  ResourceMethod rm = restMethods.get(sm.httpMethod);
238                  if (rm == null)
239                     restMethods.put(sm.httpMethod, sm);
240                  else if (rm instanceof MultiMethod)
241                     ((MultiMethod)rm).addSimpleMethod(sm);
242                  else
243                     restMethods.put(sm.httpMethod, new MultiMethod((MethodMeta)rm, sm));
244               } catch (RestServletException e) {
245                  throw new RestServletException("Problem occurred trying to serialize methods on class {0}, methods={1}", this.getClass().getName(), JsonSerializer.DEFAULT_LAX.serialize(methodsFound)).initCause(e);
246               }
247            }
248         }
249
250         for (ResourceMethod m : restMethods.values())
251            m.complete();
252
253         // Discover the child resources.
254         childResources.putAll(createChildrenMap());
255
256         for (RestServlet child : childResources.values())
257            child.init(servletConfig);
258
259         varResolver.addVars(
260            LocalizationVar.class,
261            RequestVar.class,
262            SerializedRequestAttrVar.class,
263            ServletInitParamVar.class,
264            UrlEncodeVar.class
265         );
266
267      } catch (RestException e) {
268         // Thrown RestExceptions are simply caught and rethrown on subsequent calls to service().
269         initException = e;
270         log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
271         title = String.valueOf(initException.getLocalizedMessage());
272      } catch (ServletException e) {
273         initException = e;
274         log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
275         title = String.valueOf(initException.getLocalizedMessage());
276         throw e;
277      } catch (Exception e) {
278         initException = e;
279         log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
280         title = String.valueOf(initException.getLocalizedMessage());
281         throw new ServletException(e);
282      } catch (Throwable e) {
283         initException = new Exception(e);
284         log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
285         title = String.valueOf(initException.getLocalizedMessage());
286         throw new ServletException(e);
287      } finally {
288         isInitialized = true;
289      }
290   }

296   /**
297    * Creates the child resources of this resource.
298    * <p>
299    *    Default implementation calls {@link #createChildren()} and uses the {@link RestResource#path() @RestResource.path()} annotation
300    *       on each child to identify the subpath for the resource which become the keys in this map.
301    *    It then calls the {@link #setParent(RestServlet)} method on the child resource.
302    * </p>
303    * <p>
304    *    Subclasses can override this method to programatically create child resources
305    *       without using the {@link RestResource#children() @RestResource.children()} annotation.
306    *    When overridding this method, you are responsible for calling {@link #setParent(RestServlet)} on the
307    *       child resources.
308    * </p>
309    *
310    * @return The new mutable list of child resource instances.
311    * @throws Exception If an error occurred during servlet instantiation.
312    */
313   protected Map<String,RestServlet> createChildrenMap() throws Exception {
314      Map<String,RestServlet> m = new LinkedHashMap<String,RestServlet>();
315      for (RestServlet r : createChildren()) {
316         r.setParent(this);
317         String p = r.findPath();
318         if (p == null)
319            throw new RestServletException("Child resource ''{0}'' does not define a ''@RestResource.path'' attribute.", r.getClass().getName());
320         m.put(p, r);
321      }
322      return m;
323   }

325   /**
326    * Creates instances of child resources for this servlet.
327    * <p>
328    *    Default implementation uses the {@link RestResource#children() @RestResource.children()} annotation to identify and
329    *       instantiate children.
330    * </p>
331    * <p>
332    *    Subclasses can override this method to programatically create child resources
333    *       without using the {@link RestResource#children() @RestResource.children()} annotation.
334    * </p>
335    *
336    * @return The new mutable list of child resource instances.
337    * @throws Exception If an error occurred during servlet instantiation.
338    */
339   protected List<RestServlet> createChildren() throws Exception {
340      List<RestServlet> l = new LinkedList<RestServlet>();
341      for (Class<?> c : getChildClasses()) {
342         if (isParentClass(RestServlet.class, c))
343            l.add((RestServlet)c.newInstance());
344         else
345            l.add(resolveChild(c));
346      }
347      return l;
348   }

351    * Programmatic equivalent to the {@link RestResource#children() @RestResource.children()} annotation.
352    * <p>
353    *    Subclasses can override this method to provide customized list of child resources.
354    *       (e.g. different children based on values specified in the config file).

364      List<Class<?>> l = new ArrayList<Class<?>>();
365      List<RestResource> rr = ReflectionUtils.findAnnotations(RestResource.class, getClass());

368      return l.toArray(new Class<?>[l.size()]);

371   /**
372    * Creates the class-level properties associated with this servlet.
373    * <p>
374    *    Subclasses can override this method to provide their own class-level properties for this servlet, typically
375    *       by calling <code><jk>super</jk>.createProperties()</code> and appending to the map.
376    *  However, in most cases, the existing set of properties can be added to by overridding {@link #getProperties()}
377    *       and appending to the map returned by <code><jk>super</jk>.getProperties()</code>
378    * </p>
379    * <p>
380    *    By default, the map returned by this method contains the following:
381    * </p>
382    * <ul class='spaced-list'>
383    *    <li>Servlet-init parameters.
384    *    <li>{@link RestResource#properties()} annotations in parent-to-child order.
385    *    <li>{@link SerializerContext#SERIALIZER_relativeUriBase} from {@link ServletConfig#getServletContext()}.
386    * </ul>
387    *
388    * @return The resource properties as an {@link ObjectMap}.
389    */
390   protected ObjectMap createProperties() {
391      ObjectMap m = new ObjectMap();
392
393      ServletContext ctx = servletConfig.getServletContext();
394
395      // Workaround for bug in Jetty that causes context path to always end in "null".
396      String ctxPath = ctx.getContextPath();
397      if (ctxPath.endsWith("null"))
398         ctxPath = ctxPath.substring(0, ctxPath.length()-4);
399      m.put(SERIALIZER_relativeUriBase, ctxPath);
400
401      // Get the initialization parameters.
402      for (Enumeration ep = servletConfig.getInitParameterNames(); ep.hasMoreElements();) {
403         String p = (String)ep.nextElement();
404         String initParam = servletConfig.getInitParameter(p);
405         m.put(p, initParam);
406      }
407
408      // Properties are loaded in parent-to-child order to allow overrides.
409      for (RestResource r : restResourceAnnotationsParentFirst.values())
410         for (Property p : r.properties())
411            m.append(getVarResolver().resolve(p.name()), getVarResolver().resolve(p.value()));
412
413      return m;
414   }

532
533      g.setProperties(properties);
534      g.addBeanFilters(beanFilters).addPojoSwaps(pojoSwaps);

1357    * Resolve a static resource file.
1358    * <p>
1359    *    Subclasses can override this method to provide their own way to resolve files.
1360    * </p>
1361    *
1362    * @param pathInfo The unencoded path info.
1363    * @return The resource, or <jk>null</jk> if the resource could not be resolved.
1364    * @throws IOException
1365    */
1366   protected StreamResource resolveStaticFile(String pathInfo) throws IOException {
1367      if (! staticFilesCache.containsKey(pathInfo)) {
1368         String p = RestUtils.decode(RestUtils.trimSlashes(pathInfo));
1369         if (p.indexOf("..") != -1)
1370            throw new RestException(SC_NOT_FOUND, "Invalid path");
1371         for (Map.Entry<String,String> e : staticFilesMap.entrySet()) {
1372            String key = RestUtils.trimSlashes(e.getKey());
1373            if (p.startsWith(key)) {
1374               String remainder = (p.equals(key) ? "" : p.substring(key.length()));
1375               if (remainder.isEmpty() || remainder.startsWith("/")) {
1376                  String p2 = RestUtils.trimSlashes(e.getValue()) + remainder;
1377                  InputStream is = getResource(p2, null);
1378                  if (is != null) {
1379                     try {
1380                        int i = p2.lastIndexOf('/');
1381                        String name = (i == -1 ? p2 : p2.substring(i+1));
1382                        String mediaType = getMimetypesFileTypeMap().getContentType(name);
1383                        staticFilesCache.put(pathInfo, new StreamResource(is, mediaType).setHeader("Cache-Control", "max-age=86400, public"));
1384                        return staticFilesCache.get(pathInfo);
1385                     } finally {
1386                        is.close();
1387                     }
1388                  }
1389               }
1390            }
1391         }
1392      }
