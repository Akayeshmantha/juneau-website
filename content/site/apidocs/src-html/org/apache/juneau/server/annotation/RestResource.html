033/**
034 * Optionally used to associate metadata on an instance of {@link RestServlet}.
035 * <p>
036 *    Refer to {@link org.apache.juneau.server} doc for information on using this class.
037 *
038 * @author James Bognar (james.bognar@salesforce.com)
039 */
040@Documented
041@Target(TYPE)
042@Retention(RUNTIME)
043@Inherited
044public @interface RestResource {
045
046   /**
047    * Identifies the location of the resource bundle for this class.
048    * <p>
049    * This annotation is used to provide localized messages for the following methods:
050    * <ul>
051    *    <li>{@link RestServlet#getMessage(java.util.Locale, String, Object...)}
052    *    <li>{@link RestServlet#getMethodDescriptions(RestRequest)}
053    *    <li>{@link RestServlet#getLabel(RestRequest)}
054    *    <li>{@link RestServlet#getDescription(RestRequest)}
055    * </ul>
056    * <p>
057    * Refer to the {@link MessageBundle} class for a description of the message key formats
058    *    used in the properties file.
059    * <p>
060    * The value can be a relative path like <js>"nls/Messages"</js>, indicating to look for the
061    *    resource bundle <js>"com.ibm.sample.nls.Messages"</js> if the resource class
062    *    is in <js>"com.ibm.sample"</js>, or it can be an absolute path, like <js>"com.ibm.sample.nls.Messages"</js>
063    */
064   String messages() default "";
065
066   /**
067    * Class-level guards.
068    * <p>
069    * Associates one or more {@link RestGuard RestGuards} with all REST methods defined
070    *    in this class.
071    * These guards get called immediately before execution of any REST method in this class.
072    * <p>
073    * Typically, guards will be used for permissions checking on the user making the request,
074    *    but it can also be used for other purposes like pre-call validation of a request.
075    */
076   Class<? extends RestGuard>[] guards() default {};
077
078   /**
079    * Class-level converters.
080    * <p>
081    * Associates one or more {@link RestConverter converters} with a resource class.
082    * These converters get called immediately after execution of the REST method in the same
083    *    order specified in the annotation.
084    * <p>
085    * Can be used for performing post-processing on the response object before serialization.
086    * <p>
087    * Default converter implementations are provided in the {@link org.apache.juneau.server.converters} package.
088    */
089   Class<? extends RestConverter>[] converters() default {};
090
091   /**
092    * Class-level bean filters.
093    * <p>
094    * Shortcut to add bean filters to the bean contexts of the objects returned by the following methods:
095    * <ul>
096    *    <li>{@link RestServlet#getBeanContext()}
097    *    <li>{@link RestServlet#getSerializers()}
098    *    <li>{@link RestServlet#getParsers()}
099    * </ul>
100    * <p>
101    * If the specified class is an instance of {@link BeanFilterBuilder}, then a filter built from that builder is added.
102    * Any other classes are wrapped in a {@link InterfaceBeanFilterBuilder} to indicate that subclasses should
103    *    be treated as the specified class type.
104    */
105   Class<?>[] beanFilters() default {};
106
107   /**
108    * Class-level POJO swaps.
109    * <p>
110    * Shortcut to add POJO swaps to the bean contexts of the objects returned by the following methods:
111    * <ul>
112    *    <li>{@link RestServlet#getBeanContext()}
113    *    <li>{@link RestServlet#getSerializers()}
114    *    <li>{@link RestServlet#getParsers()}
115    * </ul>
116    * <p>
117    * If the specified class is an instance of {@link PojoSwap}, then that swap is added.
118    * Any other classes are wrapped in a {@link SurrogateSwap}.
119    */
120   Class<?>[] pojoSwaps() default {};
121
122   /**
123    * Class-level properties.
124    * <p>
125    *    Shortcut for specifying class-level properties on this servlet to the objects returned by the following methods:
126    * <ul>
127    *    <li>{@link RestServlet#getBeanContext()}
128    *    <li>{@link RestServlet#getSerializers()}
129    *    <li>{@link RestServlet#getParsers()}
130    * </ul>
131    * <p>
132    *    Any of the following property names can be specified:
133    * <ul>
134    *    <li>{@link RestServletContext}
135    *    <li>{@link BeanContext}
136    *    <li>{@link SerializerContext}
137    *    <li>{@link ParserContext}
138    *    <li>{@link JsonSerializerContext}
139    *    <li>{@link RdfSerializerContext}
140    *    <li>{@link RdfParserContext}
141    *    <li>{@link RdfCommonContext}
142    *    <li>{@link XmlSerializerContext}
143    *    <li>{@link XmlParserContext}
144    * </ul>
145    * <p>
146    * Property values will be converted to the appropriate type.
147    * <p>
148    * In some cases, properties can be overridden at runtime through the {@link RestResponse#setProperty(String, Object)} method
149    *    or through a {@link Properties @Properties} annotated method parameter.
150    */
151   Property[] properties() default {};
152
153   /**
154    * Specifies a list of {@link Serializer} classes to add to the list of serializers available for this servlet.
155    * <p>
156    * This annotation can only be used on {@link Serializer} classes that have no-arg constructors.
157    */
158   Class<? extends Serializer>[] serializers() default {};
159
160   /**
161    * Specifies a list of {@link Parser} classes to add to the list of parsers available for this servlet.
162    * <p>
163    * This annotation can only be used on {@link Parser} classes that have no-arg constructors.
164    */
165   Class<? extends Parser>[] parsers() default {};
166
167   /**
168    * Specifies a list of {@link ResponseHandler} classes that know how to convert POJOs returned
169    *    by REST methods or set via {@link RestResponse#setOutput(Object)} into appropriate
170    *    HTTP responses.
171    * See {@link ResponseHandler} for details.
172    */
173   Class<? extends ResponseHandler>[] responseHandlers() default {};
174
175   /**
176    * Specifies a list of {@link Encoder} to associate with this servlet.
177    * <p>
178    * These can be used to enable various kinds of compression (e.g. <js>"gzip"</js>) on requests and responses.
179    * <p>
180    * This annotation can only be used on {@link Encoder} classes that have no-arg constructors.
181    *
182    * <dl>
183    *    <dt>Example:</dt>
184    *    <dd>
185    *       <p class='bcode'>
186    *    <jc>// Servlet with automated support for GZIP compression</jc>
187    *    <ja>@RestResource</ja>(encoders={GzipEncoder.<jk>class</jk>})
188    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
189    *       ...
190    *    }
191    *       </p>
192    *    </dd>
193    * </dl>
194    */
195   Class<? extends Encoder>[] encoders() default {};
196
197   /**
198    * Specifies default values for request headers.
199    * <p>
200    * Strings are of the format <js>"Header-Name: header-value"</js>.
201    * <p>
202    * Affects values returned by {@link RestRequest#getHeader(String)} when the header is not present on the request.
203    * <p>
204    * The most useful reason for this annotation is to provide a default <code>Accept</code> header when one is not specified
205    *    so that a particular default {@link Serializer} is picked.
206    * <p>
207    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
208    *
209    * <dl>
210    *    <dt>Example:</dt>
211    *    <dd>
212    *       <p class='bcode'>
213    *    <jc>// Assume "text/json" Accept value when Accept not specified</jc>
214    *    <ja>@RestResource</ja>(defaultRequestHeaders={<js>"Accept: text/json"</js>})
215    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
216    *       ...
217    *    }
218    *       </p>
219    *    </dd>
220    * </dl>
221    */
222   String[] defaultRequestHeaders() default {};
223
224   /**
225    * Specifies default values for response headers.
226    * <p>
227    * Strings are of the format <js>"Header-Name: header-value"</js>.
228    * <p>
229    * This is equivalent to calling {@link RestResponse#setHeader(String, String)} programmatically in each of the Java methods.
230    * <p>
231    * The header value will not be set if the header value has already been specified (hence the 'default' in the name).
232    * <p>
233    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
234    *
235    * <dl>
236    *    <dt>Example:</dt>
237    *    <dd>
238    *       <p class='bcode'>
239    *    <jc>// Add a version header attribute to all responses</jc>
240    *    <ja>@RestResource</ja>(defaultResponseHeaders={<js>"X-Version: 1.0"</js>})
241    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
242    *       ...
243    *    }
244    *       </p>
245    *    </dd>
246    * </dl>
247    */
248   String[] defaultResponseHeaders() default {};
249
250   /**
251    * Defines children of this resource.
252    * <p>
253    * A REST child resource is simply another servlet that is initialized as part of the parent
254    *    resource and has a servlet path directly under the parent servlet path.
255    * The main advantage to defining servlets as REST children is that you do not need
256    *    to define them in the <code>web.xml</code> file of the web application.
257    * This can cut down on the number of entries that show up in the <code>web.xml</code> file
258    *    if you are defining large numbers of servlets.
259    * <p>
260    * Child resources must specify a value for {@link #path()} that identifies the subpath of the
261    *    child resource relative to the parent path.
262    * <p>
263    * It should be noted that servlets can be nested arbitrarily deep using this technique (i.e. children can also have children).
264    *
265    * <dl>
266    *    <dt>Servlet initialization:</dt>
267    *    <dd>
268    *       <p>
269    *          A child resource will be initialized immediately after the parent servlet is initialized.  The child resource
270    *          receives the same servlet config as the parent resource.  This allows configuration information such as
271    *          servlet initialization parameters to filter to child resources.
272    *       </p>
273    *    </dd>
274    *    <dt>Runtime behavior:</dt>
275    *    <dd>
276    *       <p>
277    *          As a rule, methods defined on the <code>HttpServletRequest</code> object will behave as if
278    *          the child servlet were deployed as a top-level resource under the child's servlet path.
279    *          For example, the <code>getServletPath()</code> and <code>getPathInfo()</code> methods on the
280    *          <code>HttpServletRequest</code> object will behave as if the child resource were deployed
281    *          using the child's servlet path.
282    *          Therefore, the runtime behavior should be equivalent to deploying the child servlet in
283    *          the <code>web.xml</code> file of the web application.
284    *       </p>
285    *    </dd>
286    * </dl>
287    */
288   Class<?>[] children() default {};
289
290   /**
291    * Identifies the URL subpath relative to the parent resource.
292    * <p>
293    * Typically, this annotation is only applicable to resources defined as children through the {@link #children()}
294    *    annotation.  However, it may be used in other ways (e.g. defining paths for top-level resources in microservices).
295    * <p>
296    * This annotation is ignored on top-level servlets (i.e. servlets defined in <code>web.xml</code> files).
297    * Therefore, implementers can optionally specify a path value for documentation purposes.
298    */
299   String path() default "";
300
301   /**
302    * Optional servlet label.
303    * <p>
304    *    The default value pulls the label from the <code>label</code> entry in the servlet resource bundle.
305    *    (e.g. <js>"label = foo"</js> or <js>"MyServlet.label = foo"</js>).
306    * <p>
307    *    This field can contain variables (e.g. "$L{my.localized.variable}").
308    */
309   String label() default "";
310
311   /**
312    * Optional servlet description.
313    * <p>
314    *    The default value pulls the description from the <code>description</code> entry in the servlet resource bundle.
315    *    (e.g. <js>"description = foo"</js> or <js>"MyServlet.description = foo"</js>).
316    * <p>
317    *    This field can contain variables (e.g. "$L{my.localized.variable}").
318    */
319   String description() default "";
320
321   /**
322    * Optional location of configuration file for this servlet.
323    * <p>
324    *    The configuration file .
325    * <p>
326    *    This field can contain variables (e.g. "$L{my.localized.variable}").
327    */
328   String config() default "";
329
330   /**
331    * The stylesheet to use for HTML views.
332    * <p>
333    *    The name is a path to a stylesheet located in either the classpath or working directory.
334    *    The resulting stylesheet becomes available through the servlet via the URL <js>"[servletpath]/style.css"</js>.
335    * <p>
336    *    The default set of styles located in the <code>org.apache.juneau.server.styles</code> package are:
337    * <ul class='spaced-list'>
338    *    <li><js>"styles/juneau.css"</js> - Theme based on Jazz look-and-feel.
339    *    <li><js>"styles/devops.css"</js> - Theme based on IBM DevOps look-and-feel.
340    * </ul>
341    * <p>
342    *    The classpath search starts with the child servlet class and proceeds up the class hierarchy
343    *    chain.  Since the {@link RestServlet} class is in the <code>org.apache.juneau.server</code> package
344    *    and the predefined styles are in the <code>org.apache.juneau.server.styles</code> package, the paths to
345    *    the predefined styles are prefixed with <js>"styles/"</js>.
346    * <p>
347    *    If the stylesheet cannot be found on the classpath, an attempt to look in the working directory
348    *    for it will be made.  This allows for stylesheets to be placed on the file system in the working
349    *    directory.
350    * <p>
351    *    If the file cannot be located, the request to <js>"[servletpath]/style.css"</js> will return {@link HttpServletResponse#SC_NOT_FOUND}.
352    *
353    * <dl>
354    *    <dt>Example:</dt>
355    *    <dd>
356    *       <p class='bcode'>
357    *    <jk>package</jk> com.ibm.mypackage;
358    *
359    *    <ja>@RestResource</ja>(
360    *       stylesheet=<js>"mystyles/mycss.css"</js>
361    *    )
362    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
363    *    }
364    *       </p>
365    *       <p>
366    *          In this example, the servlet will attempt to find the <code>mycss.css</code> file in the following ordered locations:
367    *       </p>
368    *       <ol>
369    *          <li><code>com.ibm.mypackage.mystyles</code> package.
370    *          <li><code>org.apache.juneau.server.mystyles</code> package (since <code>RestServletDefault</code> is in <code>org.apache.juneau.server</code>).
371    *          <li><code>[working-dir]/mystyles</code> directory.
372    *       </ol>
373    *    </dd>
374    * </dl>
375    */
376   String stylesheet() default "";
377
378   /**
379    * The favicon to use for HTML views.
380    * <p>
381    *    The name is a path to an icon file located in either the classpath or working directory in a similar way
382    *    to how the {@link #stylesheet()} stylesheet is resolved.
383    *    The resulting favicon becomes available in the servlet via the URL <js>"[servletpath]/favicon.ico"</js>.
384    * <p>
385    *    If the file cannot be located, the request to <js>"[servletpath]/favicon.ico"</js> will return {@link HttpServletResponse#SC_NOT_FOUND}.
386    *
387    * <dl>
388    *    <dt>Example:</dt>
389    *    <dd>
390    *       <p class='bcode'>
391    *    <jk>package</jk> com.ibm.mypackage;
392    *
393    *    <ja>@RestResource</ja>(
394    *       favicon=<js>"mydocs/myicon.ico"</js>
395    *    )
396    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
397    *    }
398    *       </p>
399    *       <p>
400    *          In this example, the servlet will attempt to find the <code>myicon.ico</code> file in the following ordered locations:
401    *       </p>
402    *       <ol>
403    *          <li><code>com.ibm.mypackage.mydocs</code> package.
404    *          <li><code>org.apache.juneau.server.mydocs</code> package (since <code>RestServletDefault</code> is in <code>org.apache.juneau.server</code>).
405    *          <li><code>[working-dir]/mydocs</code> directory.
406    *       </ol>
407    *    </dd>
408    * </dl>
409    */
410   String favicon() default "";
411
412   /**
413    * Defines paths and locations of statically served files.
414    * <p>
415    *    This is a JSON map of paths to packages/directories located on either the classpath or working directory.
416    * <p>
417    *    Mappings are cumulative from parent to child.  Child resources can override mappings made on parent resources.
418    * <p>
419    *    If the file cannot be located, the request will return {@link HttpServletResponse#SC_NOT_FOUND}.
420    * <p>
421    *    The media type on the response is determined by the {@link RestServlet#getMimetypesFileTypeMap()} method.
422    *
423    * <dl>
424    *    <dt>Example:</dt>
425    *    <dd>
426    *       <p class='bcode'>
427    *    <jk>package</jk> com.ibm.mypackage;
428    *
429    *    <ja>@RestResource</ja>(
430    *       path=<js>"/myresource"</js>,
431    *       staticFiles=<js>"{htdocs:'docs'}"</js>
432    *    )
433    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
434    *    }
435    *       </p>
436    *       <p>
437    *          In this example, given a GET request to <code>/myresource/htdocs/foobar.html</code>, the servlet will attempt to find the <code>foobar.html</code> file
438    *          in the following ordered locations:
439    *       </p>
440    *       <ol>
441    *          <li><code>com.ibm.mypackage.docs</code> package.
442    *          <li><code>org.apache.juneau.server.docs</code> package (since <code>RestServletDefault</code> is in <code>org.apache.juneau.server</code>).
443    *          <li><code>[working-dir]/docs</code> directory.
444    *       </ol>
445    *    </dd>
446    * </dl>
447    */
448   String staticFiles() default "";
449
450   /**
451    * Specifies the HTTP header name used to identify the client version.
452    * <p>
453    *    The client version is used to support backwards compatibility for breaking REST interface
454    *    changes.  Used in conjunction with {@link RestMethod#clientVersion()} annotation.
455    * <p>
456    *    If not specified, uses <js>"X-Client-Version"</js>.
457    */
458   String clientVersionHeader() default "";
