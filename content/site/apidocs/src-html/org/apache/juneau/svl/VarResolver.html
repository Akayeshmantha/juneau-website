014
015import static java.text.MessageFormat.*;
016import static org.apache.juneau.svl.VarResolverContext.*;
017
018import java.io.*;
019import java.util.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.ini.*;
023import org.apache.juneau.svl.vars.*;

025/**
026 * Utility class for resolving variables of the form <js>"$X{key}"</js> in strings.
027 * <p>
028 * Variables are of the form <code>$X{key}</code>, where <code>X</code> can consist of zero or more ASCII characters.<br>
029 *    The variable key can contain anything, even nested variables that get recursively resolved.
030 * <p>
031 * Variables are defined through the {@link #addVars(Class[])} method.
032 * <p>
033 * The {@link Var} interface defines how variables are converted to values.
034 * <p>
035 * <h5 class='section'>Example:</h5>
036 * <p class='bcode'>
037 *    <jk>public class</jk> SystemPropertiesVar <jk>extends</jk> SimpleVar {
038 *
039 *       <jc>// Must have a no-arg constructor!</jc>
040 *       <jk>public</jk> SystemPropertiesVar() {
041 *          <jk>super</jk>(<js>"S"</js>);
042 *       }
043 *
044 *       <ja>@Override</ja>
045 *       <jk>public</jk> String resolve(VarResolverSession session, String varVal) {
046 *          <jk>return</jk> System.<jsm>getProperty</jsm>(varVal);
047 *       }
048 *    }
049 *
050 *    <jc>// Create a variable resolver that resolves system properties (e.g. "$S{java.home}")</jc>
051 *    VarResolver r = <jk>new</jk> VarResolver().addVars(SystemPropertiesVar.<js>class</js>);
052 *
053 *    <jc>// Use it!</jc>
054 *    System.<jsf>out</jsf>.println(r.resolve(<js>"java.home is set to $S{java.home}"</js>));
055 * </p>
056 *
057 * <h6 class='topic'>Context objects</h6>
058 * <p>
059 * Var resolvers can have zero or more context objects associated with them.
060 * <p>
061 * Context objects are arbitrary objects associated with this var resolver, such as
062 *    a {@link ConfigFile} object.
063 * They are set through the {@link #setContextObject(String, Object)} method.
064 * They can be any class type.
065 * <p>
066 * Context objects can be retrieved by {@link Var} classes through the {@link VarResolverSession#getSessionObject(Class, String)} method.
067 *
068 * <h6 class='topic'>Session objects</h6>
069 * <p>
070 * Session objects are considered more ephemeral than context objects.
071 * While a context object is unlikely to ever change, a session object may change on every
072 *    use of the var resolver.
073 * For example, the server API defines various <code>Var</code> objects that use the <code>RestRequest</code>
074 *    object as a session object for the duration of a single HTTP request.
075 * <p>
076 * Session objects are used by calling the {@link #createSession()} or {@link #createSession(Map)} methods to create an instance
077 *    of a {@link VarResolverSession} object that contains {@link VarResolverSession#resolve(String)} and {@link VarResolverSession#resolveTo(String,Writer)} methods
078 *    that are identical to {@link VarResolver#resolve(String)} and {@link VarResolver#resolveTo(String, Writer)} except that the <code>Var</code> objects
079 *    have access to the session objects through the {@link VarResolverSession#getSessionObject(Class, String)} method.
080 * <p>
081 * Session objects are specified through either the {@link #createSession(Map)} method or the {@link VarResolverSession#setSessionObject(String, Object)} methods.
082 *
083 * <h6 class='topic'>Cloning</h6>
084 * <p>
085 * Var resolvers can be cloned by using the {@link #clone()} method.
086 * Cloning a resolver will copy it's {@link Var} class names and context objects.
087 * <p>
088 * <h5 class='section'>Example:</h5>
089 * <p class='bcode'>
090 *    <jc>// Create a resolver that copies the default resolver and adds $C and $ARG vars.</jc>
091 *    VarResolver myVarResolver = VarResolver.<jsf>DEFAULT</jsf>.clone().addVars(ConfigVar.<jk>class</jk>, ArgsVar.<jk>class</jk>);
092 * </p>
093 *
094 * @see org.apache.juneau.svl
095 */
096public class VarResolver extends CoreApi {
097
098   /**
099    * Default string variable resolver with support for system properties and environment variables:
100    * <p>
101    * <ul>
102    *    <li><code>$S{key}</code>,<code>$S{key,default}</code> - System properties.
103    *    <li><code>$E{key}</code>,<code>$E{key,default}</code> - Environment variables.
104    *    <li><code>$IF{booleanValue,thenValue[,elseValue]}</code> - If-else patterns.
105    *    <li><code>$SWITCH{test,matchPattern,thenValue[,matchPattern,thenValue][,elseValue]}</code> - Switch patterns.
106    * </ul>
107    *
108    * @see SystemPropertiesVar
109    * @see EnvVariablesVar
110    */
111   public static final VarResolver DEFAULT = new VarResolver().addVars(SystemPropertiesVar.class, EnvVariablesVar.class, SwitchVar.class, IfVar.class).lock();
112
113   /**
114    * Construct an empty var resolver with no vars.
115    */
116   public VarResolver() {}
117
118   /**
119    * Register new variables with this resolver.
120    *
121    * @param vars The variable resolver classes.
122    * These classes must subclass from {@link Var} and have no-arg constructors.
123    * @return This object (for method chaining).
124    */
125   public VarResolver addVars(Class<?>...vars) {
126      checkLock();
127      ContextFactory cf = getContextFactory();
128      for (Class<?> v : vars) {
129         try {
130            v.newInstance();
131         } catch (InstantiationException e) {
132            throw new UnsupportedOperationException(format("Cannot instantiate variable class {0}.  Must have a public no-arg constructor.", v.getName()));
133         } catch (RuntimeException e) {
134            throw e;
135         } catch (Exception e) {
136            throw new RuntimeException(e);
137         }
138         cf.addToProperty(SVL_vars, v);
139      }
140      return this;
141   }
142
143   /**
144    * Associates a context object with this resolver.
145    *
146    * @param name The name of the context object.
147    * @param object The context object.
148    * @return This object (for method chaining).
149    */
150   public VarResolver setContextObject(String name, Object object) {
151      getContextFactory().putToProperty(SVL_context, name, object);
152      return this;
153   }
154
155   /**
156    * Creates a new resolver session with no session objects.
157    * <p>
158    * Session objects can be associated with the specified session using the {@link VarResolverSession#setSessionObject(String, Object)} method.
159    *
160    * @return A new resolver session.
161    */
162   public VarResolverSession createSession() {
163      return new VarResolverSession(getContext(VarResolverContext.class), null);
164   }
165
166   /**
167    * Same as {@link #createSession()} except allows you to specify session objects as a map.
168    *
169    * @param sessionObjects The session objects to associate with the sessino.
170    * @return A new resolver session.
171    */
172   public VarResolverSession createSession(Map<String,Object> sessionObjects) {
173      return new VarResolverSession(getContext(VarResolverContext.class), sessionObjects);
174   }
175
176   /**
177    * Resolve variables in the specified string.
178    * <p>
179    * This is a shortcut for calling <code>createSession(<jk>null</jk>).resolve(s);</code>.
180    * This method can only be used if the string doesn't contain variables that rely on the existence of session variables.
181    *
182    * @param s The input string.
183    * @return The string with variables resolved, or the same string if it doesn't contain any variables to resolve.
184    */
185   public String resolve(String s) {
186      return createSession(null).resolve(s);
187   }
188
189   /**
190    * Resolve variables in the specified string and sends the results to the specified writer.
191    * <p>
192    * This is a shortcut for calling <code>createSession(<jk>null</jk>).resolveTo(s, w);</code>.
193    * This method can only be used if the string doesn't contain variables that rely on the existence of session variables.
194    *
195    * @param s The input string.
196    * @param w The writer to send the result to.
197    * @throws IOException
198    */
199   public void resolveTo(String s, Writer w) throws IOException {
200      createSession(null).resolveTo(s, w);
201   }
202
203
204   //--------------------------------------------------------------------------------
205   // Overridden methods
206   //--------------------------------------------------------------------------------
207
208   @Override /* Lockable */
209   public VarResolver lock() {
210      super.lock();
211      return this;
212   }
213
214   @Override /* Lockable */
215   public VarResolver clone() {
216      try {
217         return (VarResolver)super.clone();
218      } catch (CloneNotSupportedException e) {
219         throw new RuntimeException(e);
220      }
221   }
222}
