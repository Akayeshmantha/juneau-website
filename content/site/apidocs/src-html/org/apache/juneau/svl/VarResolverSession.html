023/**
024 * A var resolver session that combines a {@link VarResolver} with one or more session objects.
025 * <p>
026 * Instances of this class are considered light-weight and fast to construct, use, and discard.
027 * <p>
028 * This class contains the workhorse code for var resolution.
029 * <p>
030 * Instances of this class are created through the {@link VarResolver#createSession()} and {@link VarResolver#createSession(Map)}
031 *    methods.
032 * <p>
033 * Instances of this class are NOT guaranteed to be thread safe.
034 *
035 * @see org.apache.juneau.svl
036 */
037public class VarResolverSession extends Session {
038
039   private final VarResolverContext context;
040
041   /**
042    * Constructor.
043    *
044    * @param context The {@link VarResolver} context object that contains the {@link Var Vars} and
045    *    context objects associated with that resolver.
046    * @param sessionObjects
047    */
048   public VarResolverSession(VarResolverContext context, Map<String,Object> sessionObjects) {
049      super(context, null);
050      this.context = context;
051      addToCache(sessionObjects);
052   }
053
054   /**
055    * Adds a session object to this session.
056    *
057    * @param name The name of the session object.
058    * @param o The session object.
059    * @return This method (for method chaining).
060    */
061   public VarResolverSession setSessionObject(String name, Object o) {
062      addToCache(name, o);
063      return this;
064   }
065
066   /**
067    * Resolve all variables in the specified string.
068    *
069    * @param s The string to resolve variables in.
070    * @return The new string with all variables resolved, or the same string if no variables were found.
071    * Null input results in a blank string.
072    */
073   public String resolve(String s) {
074
075      if (s == null)
076         return "";
077      if (s.indexOf('$') == -1 && s.indexOf('\\') == -1)
078         return s;
079
080      // Special case where value consists of a single variable with no embedded variables (e.g. "$X{...}").
081      // This is a common case, so we want an optimized solution that doesn't involve string builders.
082      if (isSimpleVar(s)) {
083         String var = s.substring(1, s.indexOf('{'));
084         String val = s.substring(s.indexOf('{')+1, s.length()-1);
085         Var v = getVar(var);
086         if (v != null) {
087            if (v.streamed) {
088               StringWriter sw = new StringWriter();
089               v.resolveTo(this, sw, val);
090               return sw.toString();
091            }
092            s = v.doResolve(this, val);
093            if (s == null)
094               s = "";
095            return resolve(s);
096         }
097         return s;
098      }
099
100      try {
101         return resolveTo(s, new StringWriter()).toString();
102      } catch (IOException e) {
103         throw new RuntimeException(e); // Never happens.
104      }
105   }
106
107   /**
108    * Checks to see if string is of the simple form "$X{...}" with no embedded variables.
109    * This is a common case, and we can avoid using StringWriters.
110    */
111   private boolean isSimpleVar(String s) {
112      int S1 = 1;    // Not in variable, looking for $
113      int S2 = 2;    // Found $, Looking for {
114      int S3 = 3;    // Found {, Looking for }
115      int S4 = 4;    // Found }
116
117      int length = s.length();
118      int state = S1;
119      for (int i = 0; i < length; i++) {
120         char c = s.charAt(i);
121         if (state == S1) {
122            if (c == '$') {
123               state = S2;
124            } else {
125               return false;
126            }
127         } else if (state == S2) {
128            if (c == '{') {
129               state = S3;
130            } else if (c < 'A' || c > 'z' || (c > 'Z' && c < 'a')) {   // False trigger "$X "
131               return false;
132            }
133         } else if (state == S3) {
134            if (c == '}')
135               state = S4;
136            else if (c == '{' || c == '$')
137               return false;
138         } else if (state == S4) {
139            return false;
140         }
141      }
142      return state == S4;
143   }
144
145   /**
146    * Resolves variables in the specified string and sends the output to the specified writer.
147    * More efficient than first parsing to a string and then serializing to the writer since this
148    * method doesn't need to construct a large string.
149    *
150    * @param s The string to resolve variables in.
151    * @param out The writer to write to.
152    * @return The same writer.
153    * @throws IOException
154    */
155   public Writer resolveTo(String s, Writer out) throws IOException {
156
157      int S1 = 1;    // Not in variable, looking for $
158      int S2 = 2;    // Found $, Looking for {
159      int S3 = 3;    // Found {, Looking for }
160
161      int state = S1;
162      boolean isInEscape = false;
163      boolean hasInternalVar = false;
164      boolean hasInnerEscapes = false;
165      String varType = null;
166      String varVal = null;
167      int x = 0, x2 = 0;
168      int depth = 0;
169      int length = s.length();
170      for (int i = 0; i < length; i++) {
171         char c = s.charAt(i);
172         if (state == S1) {
173            if (isInEscape) {
174               if (c == '\\' || c == '$') {
175                  out.append(c);
176               } else {
177                  out.append('\\').append(c);
178               }
179               isInEscape = false;
180            } else if (c == '\\') {
181               isInEscape = true;
182            } else if (c == '$') {
183               x = i;
184               x2 = i;
185               state = S2;
186            } else {
187               out.append(c);
188            }
189         } else if (state == S2) {
190            if (isInEscape) {
191               isInEscape = false;
192            } else if (c == '\\') {
193               hasInnerEscapes = true;
194               isInEscape = true;
195            } else if (c == '{') {
196               varType = s.substring(x+1, i);
197               x = i;
198               state = S3;
199            } else if (c < 'A' || c > 'z' || (c > 'Z' && c < 'a')) {  // False trigger "$X "
200               if (hasInnerEscapes)
201                  out.append(unEscapeChars(s.substring(x, i+1), new char[]{'\\','{'}));
202               else
203                  out.append(s, x, i+1);
204               x = i + 1;
205               state = S1;
206               hasInnerEscapes = false;
207            }
208         } else if (state == S3) {
209            if (isInEscape) {
210               isInEscape = false;
211            } else if (c == '\\') {
212               isInEscape = true;
213               hasInnerEscapes = true;
214            } else if (c == '{') {
215               depth++;
216               hasInternalVar = true;
217            } else if (c == '}') {
218               if (depth > 0) {
219                  depth--;
220               } else {
221                  varVal = s.substring(x+1, i);
222                  varVal = (hasInternalVar ? resolve(varVal) : varVal);
223                  Var r = getVar(varType);
224                  if (r == null) {
225                     if (hasInnerEscapes)
226                        out.append(unEscapeChars(s.substring(x2, i+1), new char[]{'\\','$','{','}'}));
227                     else
228                        out.append(s, x2, i+1);
229                     x = i+1;
230                  } else {
231                     if (r.streamed)
232                        r.resolveTo(this, out, varVal);
233                     else {
234                        String replacement = r.doResolve(this, varVal);
235                        if (replacement == null)
236                           replacement = "";
237                        // If the replacement also contains variables, replace them now.
238                        if (replacement.indexOf('$') != -1)
239                           replacement = resolve(replacement);
240                        out.append(replacement);
241                     }
242                     x = i+1;
243                  }
244                  state = 1;
245                  hasInnerEscapes = false;
246               }
247            }
248         }
249      }
250      if (isInEscape)
251         out.append('\\');
252      else if (state == S2)
253         out.append('$').append(unEscapeChars(s.substring(x+1), new char[]{'{', '\\'}));
254      else if (state == S3)
255         out.append('$').append(varType).append('{').append(unEscapeChars(s.substring(x+1), new char[]{'\\','$','{','}'}));
256      return out;
257   }

260   /**
261    * Returns the session object with the specified name.
262    * Casts it to the specified class type for you.
263    *
264    * @param c The class type to cast to.
265    * @param name The name of the session object.
266    * @return The session object.  Never <jk>null</jk>.
267    * @throws RuntimeException If session object with specified name does not exist.
268    */
269   public <T> T getSessionObject(Class<T> c, String name) {
270      T t = null;
271      try {
272         t = getFromCache(c, name);
273         if (t == null) {
274            addToCache(name, this.context.getContextObject(name));
275            t = getFromCache(c, name);
276         }
277      } catch (Exception e) {
278         throw new RuntimeException(format("Session object ''{0}'' or context object ''SvlContext.{0}'' could not be converted to type ''{1}''.", name, c.getName()));
279      }
280      if (t == null)
281         throw new RuntimeException(format("Session object ''{0}'' or context object ''SvlContext.{0}'' not found.", name));
282      return t;
283   }
284
285   /**
286    * Returns the {@link Var} with the specified name.
287    *
288    * @param name The var name (e.g. <js>"S"</js>).
289    * @return The {@link Var} instance, or <jk>null</jk> if no <code>Var</code> is associated with the specified name.
290    */
291   protected Var getVar(String name) {
292      return this.context.getVars().get(name);
293   }
