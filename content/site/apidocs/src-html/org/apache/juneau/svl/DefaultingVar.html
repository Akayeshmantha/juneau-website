013package org.apache.juneau.svl;
014
015import static org.apache.juneau.internal.StringUtils.*;
016
017/**
018 * Interface for the resolution of vars with a default value if the <code>resolve()</code> method returns <jk>null</jk>.
019 * <p>
020 * For example, to resolve the system property <js>"myProperty"</js> but resolve to <js>"not found"</js> if the property doesn't exist:
021 *    <js>"$S{myProperty,not found}"</js>
022 * <p>
023 * Subclasses must implement the {@link #resolve(VarResolverSession, String)} method.
024 *
025 * @see org.apache.juneau.svl
026 */
027public abstract class DefaultingVar extends SimpleVar {
028
029   /**
030    * Constructor.
031    *
032    * @param name The name of this variable.
033    */
034   public DefaultingVar(String name) {
035      super(name);
036   }
037
038   @Override /* Var*/
039   public String doResolve(VarResolverSession session, String s) {
040      int i = s.indexOf(',');
041      if (i == -1)
042         return resolve(session, s.trim());
043      String[] s2 = split(s, ',');
044      String v = resolve(session, s2[0]);
045      if (v == null)
046         v = s2[1];
047      return v;
048   }
049}
