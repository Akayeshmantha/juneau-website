054/**
055 * Utility class for interfacing with remote REST interfaces.
056 *
057 *
058 * <h6 class='topic'>Features</h6>
059 * <ul class='spaced-list'>
060 *    <li>Convert POJOs directly to HTTP request message bodies using {@link Serializer} class.
061 *    <li>Convert HTTP response message bodies directly to POJOs using {@link Parser} class.
062 *    <li>Fluent interface.
063 *    <li>Thread safe.
064 *    <li>API for interacting with remoteable services.
065 * </ul>
066 *
067 *
068 * <h6 class='topic'>Additional Information</h6>
069 * <ul>
070 *    <li><a class='doclink' href='package-summary.html#RestClient'>org.apache.juneau.client &gt; REST client API</a> for more information and code examples.
071 * </ul>
072 *
073 * @author James Bognar (james.bognar@salesforce.com)
074 */

205   /**
206    * Creates an instance of an {@link HttpClient} to be used to handle all HTTP communications with the target server.
207    * <p>
208    * This HTTP client is used when the HTTP client is not specified through one of the constructors or the
209    *    {@link #setHttpClient(CloseableHttpClient)} method.
210    * <p>
211    * Subclasses can override this method to provide specially-configured HTTP clients to handle
212    *    stuff such as SSL/TLS certificate handling, authentication, etc.
213    * <p>
214    * The default implementation returns an instance of {@link HttpClient} using the client builder
215    *    returned by {@link #createHttpClientBuilder()}.
216    *
217    * @return The HTTP client to use.
218    * @throws Exception
219    */
220   protected CloseableHttpClient createHttpClient() throws Exception {
221      // Don't call createConnectionManager() if RestClient.setConnectionManager() was called.
222      if (httpClientConnectionManager == null)
223         httpClientBuilder.setConnectionManager(createConnectionManager());
224      return httpClientBuilder.build();
225   }

227   /**
228    * Creates an instance of an {@link HttpClientBuilder} to be used to create
229    *    the {@link HttpClient}.
230    * <p>
231    *    Subclasses can override this method to provide their own client builder.
232    * </p>
233    * <p>
234    *    The predefined method returns an {@link HttpClientBuilder} with the following settings:
235    * </p>
236    * <ul>
237    *    <li>Lax redirect strategy.
238    *    <li>The connection manager returned by {@link #createConnectionManager()}.
239    * </ul>
240    *
241    * @return The HTTP client builder to use to create the HTTP client.
242    */
243   protected HttpClientBuilder createHttpClientBuilder() {
244      HttpClientBuilder b = HttpClientBuilder.create();
245      b.setRedirectStrategy(new AllowAllRedirects());
246      return b;
247   }

249   /**
250    * Creates the {@link HttpClientConnectionManager} returned by {@link #createConnectionManager()}.
251    * <p>
252    *    Subclasses can override this method to provide their own connection manager.
253    * </p>
254    * <p>
255    *    The default implementation returns an instance of a {@link PoolingHttpClientConnectionManager}.
256    * </p>
257    *
258    * @return The HTTP client builder to use to create the HTTP client.
259    */
260   protected HttpClientConnectionManager createConnectionManager() {
261      if (sslOpts != null) {
262         HostnameVerifier hv = null;
263         switch (sslOpts.getHostVerify()) {
264            case LAX: hv = new NoopHostnameVerifier(); break;
265            case DEFAULT: hv = new DefaultHostnameVerifier(); break;
266         }
267
268         for (String p : StringUtils.split(sslOpts.getProtocols(), ',')) {
269            try {
270               TrustManager tm = new SimpleX509TrustManager(sslOpts.getCertValidate() == SSLOpts.CertValidate.LAX);
271
272               SSLContext ctx = SSLContext.getInstance(p);
273               ctx.init(null, new TrustManager[] { tm }, null);
274
275               // Create a socket to ensure this algorithm is acceptable.
276               // This will correctly disallow certain configurations (such as SSL_TLS under FIPS)
277               ctx.getSocketFactory().createSocket().close();
278               SSLConnectionSocketFactory sf = new SSLConnectionSocketFactory(ctx, hv);
279               setSSLSocketFactory(sf);
280
281               Registry<ConnectionSocketFactory> r = RegistryBuilder.<ConnectionSocketFactory> create().register("https", sf).build();
282
283               return (pooled ? new PoolingHttpClientConnectionManager(r) : new BasicHttpClientConnectionManager(r));
284            } catch (Throwable t) {}
285         }
286      }
287
288         // Using pooling connection so that this client is threadsafe.
289      return (pooled ? new PoolingHttpClientConnectionManager() : new BasicHttpClientConnectionManager());
290   }

292   /**
293    * Set up this client to use BASIC auth.
294    *
295    * @param host The auth scope hostname.
296    * @param port The auth scope port.
297    * @param user The username.
298    * @param pw The password.
299    * @return This object (for method chaining).
300    */
301   public RestClient setBasicAuth(String host, int port, String user, String pw) {
302      AuthScope scope = new AuthScope(host, port);
303      Credentials up = new UsernamePasswordCredentials(user, pw);
304      CredentialsProvider p = new BasicCredentialsProvider();
305      p.setCredentials(scope, up);
306      setDefaultCredentialsProvider(p);
307      return this;
308   }

670   /**
671    * Performs a REST call where the entire call is specified in a simple string.
672    * <p>
673    * This method is useful for performing callbacks when the target of a callback is passed in
674    * on an initial request, for example to signal when a long-running process has completed.
675    * <p>
676    * The call string can be any of the following formats:
677    * <ul class='spaced-list'>
678    *    <li><js>"[method] [url]"</js> - e.g. <js>"GET http://localhost/callback"</js>
679    *    <li><js>"[method] [url] [payload]"</js> - e.g. <js>"POST http://localhost/callback some text payload"</js>
680    *    <li><js>"[method] [headers] [url] [payload]"</js> - e.g. <js>"POST {'Content-Type':'text/json'} http://localhost/callback {'some':'json'}"</js>
681    * </ul>
682    * <p>
683    * The payload will always be sent using a simple {@link StringEntity}.
684    *
685    * @param callString The call string.
686    * @return A {@link RestCall} object that can be further tailored before executing the request
687    *    and getting the response as a parsed object.
688    * @throws RestCallException
689    */
690   public RestCall doCallback(String callString) throws RestCallException {
691      String s = callString;
692      try {
693         RestCall rc = null;
694         String method = null, uri = null, content = null;
695         ObjectMap h = null;
696         int i = s.indexOf(' ');
697         if (i != -1) {
698            method = s.substring(0, i).trim();
699            s = s.substring(i).trim();
700            if (s.length() > 0) {
701               if (s.charAt(0) == '{') {
702                  i = s.indexOf('}');
703                  if (i != -1) {
704                     String json = s.substring(0, i+1);
705                     h = JsonParser.DEFAULT.parse(json, ObjectMap.class);
706                     s = s.substring(i+1).trim();
707                  }
708               }
709               if (s.length() > 0) {
710                  i = s.indexOf(' ');
711                  if (i == -1)
712                     uri = s;
713                  else {
714                     uri = s.substring(0, i).trim();
715                     s = s.substring(i).trim();
716                     if (s.length() > 0)
717                        content = s;
718                  }
719               }
720            }
721         }
722         if (method != null && uri != null) {
723            rc = doCall(method, uri, content != null);
724            if (content != null)
725               rc.setInput(new StringEntity(content));
726            if (h != null)
727               for (Map.Entry<String,Object> e : h.entrySet())
728                  rc.setHeader(e.getKey(), e.getValue());
729            return rc;
730         }
731      } catch (Exception e) {
732         throw new RestCallException(e);
733      }
734      throw new RestCallException("Invalid format for call string.");
735   }

763    * Perform a generic REST call.
764    *
765    * @param method The method name (e.g. <js>"GET"</js>, <js>"OPTIONS"</js>).
766    * @param url The URL of the remote REST resource.  Can be any of the following:  {@link String}, {@link URI}, {@link URL}.
767    * @param hasContent Boolean flag indicating if the specified request has content associated with it.

769    *    and getting the response as a parsed object.
770    * @throws RestCallException If any authentication errors occurred.
771    */
772   public RestCall doCall(String method, Object url, boolean hasContent) throws RestCallException {
773      HttpRequestBase req = null;
774      RestCall restCall = null;
775      final String methodUC = method.toUpperCase(Locale.ENGLISH);
776      if (hasContent) {
777         req = new HttpEntityEnclosingRequestBase() {
778            @Override /* HttpRequest */
779            public String getMethod() {
780               return methodUC;
781            }
782         };
783         restCall = new RestCall(this, req);
784         if (contentType != null)
785            restCall.setHeader("Content-Type", contentType);
786      } else {
787         req = new HttpRequestBase() {
788            @Override /* HttpRequest */
789            public String getMethod() {
790               return methodUC;
791            }
792         };
793         restCall = new RestCall(this, req);
794      }
795      try {
796         req.setURI(toURI(url));
797      } catch (URISyntaxException e) {
798         throw new RestCallException(e);
799      }
800      if (accept != null)
801         restCall.setHeader("Accept", accept);
802      for (Map.Entry<String,? extends Object> e : headers.entrySet())
803         restCall.setHeader(e.getKey(), e.getValue());
804      return restCall;
805   }

807   /**
808    * Create a new proxy interface for the specified remoteable service interface.
809    *
810    * @param interfaceClass The interface to create a proxy for.
811    * @return The new proxy interface.
812    * @throws RuntimeException If the Remotable service URI has not been specified on this
813    *    client by calling {@link #setRemoteableServletUri(String)}.
814    */
815   @SuppressWarnings("unchecked")
816   public <T> T getRemoteableProxy(final Class<T> interfaceClass) {
817      if (remoteableServletUri == null)
818         throw new RuntimeException("Remoteable service URI has not been specified.");
819      return (T)Proxy.newProxyInstance(
820         interfaceClass.getClassLoader(),
821         new Class[] { interfaceClass },
822         new InvocationHandler() {
823            @Override /* InvocationHandler */
824            public Object invoke(Object proxy, Method method, Object[] args) {
825               try {
826                  String uri = remoteableServiceUriMap.get(method);
827                  if (uri == null) {
828                     // Constructing this string each time can be time consuming, so cache it.
829                     uri = remoteableServletUri + '/' + interfaceClass.getName() + '/' + ClassUtils.getMethodSignature(method);
830                     remoteableServiceUriMap.put(method, uri);
831                  }
832                  return doPost(uri, args).getResponse(method.getReturnType());
833               } catch (Exception e) {
834                  throw new RuntimeException(e);
835               }
836            }
837      });
838   }

840   private Pattern absUrlPattern = Pattern.compile("^\\w+\\:\\/\\/.*");
841
842   private URI toURI(Object url) throws URISyntaxException {
843      assertFieldNotNull(url, "url");
844      if (url instanceof URI)
845         return (URI)url;
846      if (url instanceof URL)
847         ((URL)url).toURI();
848      String s = url.toString();
849      if (rootUrl != null && ! absUrlPattern.matcher(s).matches()) {
850         if (s.isEmpty())
851            s = rootUrl;
852         else {
853            StringBuilder sb = new StringBuilder(rootUrl);
854            if (! s.startsWith("/"))
855               sb.append('/');
856            sb.append(s);
857            s = sb.toString();
858         }
859      }
860      return new URI(s);
861   }
