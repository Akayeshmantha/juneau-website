013package org.apache.juneau.dto.swagger;
014
015import org.apache.juneau.annotation.*;
016
017/**
018 * Allows adding meta data to a single tag that is used by the <a class="doclink" href="http://swagger.io/specification/#operationObject">Operation Object</a>.
019 * <p>
020 * It is not mandatory to have a Tag Object per tag used there.
021 *
022 * <h5 class='section'>Example:</h5>
023 * <p class='bcode'>
024 *    {
025 *       <js>"name"</js>: <js>"pet"</js>,
026 *       <js>"description"</js>: <js>"Pets operations"</js>
027 *    }
028 * </p>
029 *
030 * <h6 class='topic'>Additional Information</h6>
031 * <ul class='doctree'>
032 *    <li class='link'><a class='doclink' href='../../../../../overview-summary.html#DTOs'>Juneau Data Transfer Objects (org.apache.juneau.dto)</a>
033 *    <ul>
034 *       <li class='sublink'><a class='doclink' href='../../../../../overview-summary.html#DTOs.Swagger'>Swagger</a>
035 *    </ul>
036 *    <li class='jp'><a class='doclink' href='package-summary.html#TOC'>org.apache.juneau.dto.swagger</a>
037 * </ul>
038 */
039@Bean(properties="name,description,externalDocs")
040@SuppressWarnings("hiding")
041public class Tag extends SwaggerElement {
042
043   private String name;
044   private String description;
045   private ExternalDocumentation externalDocs;
046
047   /**
048    * Bean property getter:  <property>name</property>.
049    * <p>
050    * Required. The name of the tag.
051    *
052    * @return The value of the <property>name</property> property on this bean, or <jk>null</jk> if it is not set.
053    */
054   public String getName() {
055      return name;
056   }
057
058   /**
059    * Bean property setter:  <property>name</property>.
060    * <p>
061    * Required. The name of the tag.
062    *
063    * @param name The new value for the <property>name</property> property on this bean.
064    * @return This object (for method chaining).
065    */
066   public Tag setName(String name) {
067      this.name = name;
068      return this;
069   }
070
071   /**
072    * Synonym for {@link #setName(String)}.
073    *
074    * @param name The new value for the <property>name</property> property on this bean.
075    * @return This object (for method chaining).
076    */
077   public Tag name(String name) {
078      return setName(name);
079   }
080
081   /**
082    * Bean property getter:  <property>description</property>.
083    * <p>
084    * A short description for the tag.
085    * <a class="doclink" href="https://help.github.com/articles/github-flavored-markdown">GFM syntax</a> can be used for rich text representation.
086    *
087    * @return The value of the <property>description</property> property on this bean, or <jk>null</jk> if it is not set.
088    */
089   public String getDescription() {
090      return description;
091   }
092
093   /**
094    * Bean property setter:  <property>description</property>.
095    * <p>
096    * A short description for the tag.
097    * <a class="doclink" href="https://help.github.com/articles/github-flavored-markdown">GFM syntax</a> can be used for rich text representation.
098    *
099    * @param description The new value for the <property>description</property> property on this bean.
100    * @return This object (for method chaining).
101    */
102   public Tag setDescription(String description) {
103      this.description = description;
104      return this;
105   }
106
107   /**
108    * Synonym for {@link #setDescription(String)}.
109    *
110    * @param description The new value for the <property>description</property> property on this bean.
111    * @return This object (for method chaining).
112    */
113   public Tag description(String description) {
114      return setDescription(description);
115   }
116
117   /**
118    * Bean property getter:  <property>externalDocs</property>.
119    * <p>
120    * Additional external documentation for this tag.
121    *
122    * @return The value of the <property>externalDocs</property> property on this bean, or <jk>null</jk> if it is not set.
123    */
124   public ExternalDocumentation getExternalDocs() {
125      return externalDocs;
126   }
127
128   /**
129    * Bean property setter:  <property>externalDocs</property>.
130    * <p>
131    * Additional external documentation for this tag.
132    *
133    * @param externalDocs The new value for the <property>externalDocs</property> property on this bean.
134    * @return This object (for method chaining).
135    */
136   public Tag setExternalDocs(ExternalDocumentation externalDocs) {
137      this.externalDocs = externalDocs;
138      return this;
139   }
140
141   /**
142    * Synonym for {@link #setExternalDocs(ExternalDocumentation)}.
143    *
144    * @param externalDocs The new value for the <property>externalDocs</property> property on this bean.
145    * @return This object (for method chaining).
146    */
147   public Tag externalDocs(ExternalDocumentation externalDocs) {
148      return setExternalDocs(externalDocs);
149   }
150}
