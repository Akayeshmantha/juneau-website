021 * Describes a single API operation on a path.
022 *
023 * <h5 class='section'>Example:</h5>
024 * <p class='bcode'>
025 *    {
026 *       <js>"tags"</js>: [
027 *          <js>"pet"</js>
028 *       ],
029 *       <js>"summary"</js>: <js>"Updates a pet in the store with form data"</js>,
030 *       <js>"description"</js>: "",
031 *       <js>"operationId"</js>: <js>"updatePetWithForm"</js>,
032 *       <js>"consumes"</js>: [
033 *          <js>"application/x-www-form-urlencoded"</js>
034 *       ],
035 *       <js>"produces"</js>: [
036 *          <js>"application/json"</js>,
037 *          <js>"application/xml"</js>
038 *       ],
039 *       <js>"parameters"</js>: [
040 *          {
041 *             <js>"name"</js>: <js>"petId"</js>,
042 *             <js>"in"</js>: <js>"path"</js>,
043 *             <js>"description"</js>: <js>"ID of pet that needs to be updated"</js>,
044 *             <js>"required"</js>: <jk>true</jk>,
045 *             <js>"type"</js>: <js>"string"</js>
046 *          },
047 *          {
048 *             <js>"name"</js>: <js>"name"</js>,
049 *             <js>"in"</js>: <js>"formData"</js>,
050 *             <js>"description"</js>: <js>"Updated name of the pet"</js>,
051 *             <js>"required"</js>: <jk>false</jk>,
052 *             <js>"type"</js>: <js>"string"</js>
053 *          },
054 *          {
055 *             <js>"name"</js>: <js>"status"</js>,
056 *             <js>"in"</js>: <js>"formData"</js>,
057 *             <js>"description"</js>: <js>"Updated status of the pet"</js>,
058 *             <js>"required"</js>: <jk>false</jk>,
059 *             <js>"type"</js>: <js>"string"</js>
060 *          }
061 *       ],
062 *       <js>"responses"</js>: {
063 *          <js>"200"</js>: {
064 *             <js>"description"</js>: <js>"Pet updated."</js>
065 *          },
066 *          <js>"405"</js>: {
067 *             <js>"description"</js>: <js>"Invalid input"</js>
068 *          }
069 *       },
070 *       <js>"security"</js>: [
071 *          {
072 *             <js>"petstore_auth"</js>: [
073 *                <js>"write:pets"</js>,
074 *                <js>"read:pets"</js>
075 *             ]
076 *          }
077 *       ]
078 *    }
079 * </p>
