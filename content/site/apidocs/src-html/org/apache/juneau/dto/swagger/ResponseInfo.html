013package org.apache.juneau.dto.swagger;
014
015import java.util.*;
016
017import org.apache.juneau.annotation.*;
018
019/**
020 * Describes a single response from an API Operation.
021 *
022 * <h5 class='section'>Example:</h5>
023 * <p class='bcode'>
024 *    {
025 *       <js>"description"</js>: <js>"A complex object array response"</js>,
026 *       <js>"schema"</js>: {
027 *          <js>"type"</js>: <js>"array"</js>,
028 *          <js>"items"</js>: {
029 *             <js>"$ref"</js>: <js>"#/definitions/VeryComplexType"</js>
030 *          }
031 *       }
032 *    }
033 * </p>
034 *
035 * <h6 class='topic'>Additional Information</h6>
036 * <ul class='doctree'>
037 *    <li class='link'><a class='doclink' href='../../../../../overview-summary.html#DTOs'>Juneau Data Transfer Objects (org.apache.juneau.dto)</a>
038 *    <ul>
039 *       <li class='sublink'><a class='doclink' href='../../../../../overview-summary.html#DTOs.Swagger'>Swagger</a>
040 *    </ul>
041 *    <li class='jp'><a class='doclink' href='package-summary.html#TOC'>org.apache.juneau.dto.swagger</a>
042 * </ul>
043 */
044@Bean(properties="description,schema,headers,examples")
045@SuppressWarnings("hiding")
046public class ResponseInfo extends SwaggerElement {
047
048   private String description;
049   private SchemaInfo schema;
050   private Map<String,HeaderInfo> headers;
051   private Map<String,Object> examples;
052
053   /**
054    * Bean property getter:  <property>description</property>.
055    * <p>
056    * Required. A short description of the response.
057    * <a class="doclink" href="https://help.github.com/articles/github-flavored-markdown">GFM syntax</a> can be used for rich text representation.
058    *
059    * @return The value of the <property>description</property> property on this bean, or <jk>null</jk> if it is not set.
060    */
061   public String getDescription() {
062      return description;
063   }
064
065   /**
066    * Bean property setter:  <property>description</property>.
067    * <p>
068    * Required. A short description of the response.
069    * <a class="doclink" href="https://help.github.com/articles/github-flavored-markdown">GFM syntax</a> can be used for rich text representation.
070    *
071    * @param description The new value for the <property>description</property> property on this bean.
072    * @return This object (for method chaining).
073    */
074   public ResponseInfo setDescription(String description) {
075      this.description = description;
076      return this;
077   }
078
079   /**
080    * Synonym for {@link #setDescription(String)}.
081    *
082    * @param description The new value for the <property>description</property> property on this bean.
083    * @return This object (for method chaining).
084    */
085   public ResponseInfo description(String description) {
086      return setDescription(description);
087   }
088
089   /**
090    * Bean property getter:  <property>schema</property>.
091    * <p>
092    * A definition of the response structure.
093    * It can be a primitive, an array or an object.
094    * If this field does not exist, it means no content is returned as part of the response.
095    * As an extension to the <a class="doclink" href="http://swagger.io/specification/#schemaObject">Schema Object</a>, its root type value may also be <js>"file"</js>.
096    * This SHOULD be accompanied by a relevant produces mime-type.
097    *
098    * @return The value of the <property>schema</property> property on this bean, or <jk>null</jk> if it is not set.
099    */
100   public SchemaInfo getSchema() {
101      return schema;
102   }
103
104   /**
105    * Bean property setter:  <property>schema</property>.
106    * <p>
107    * A definition of the response structure.
108    * It can be a primitive, an array or an object.
109    * If this field does not exist, it means no content is returned as part of the response.
110    * As an extension to the <a class="doclink" href="http://swagger.io/specification/#schemaObject">Schema Object</a>, its root type value may also be <js>"file"</js>.
111    * This SHOULD be accompanied by a relevant produces mime-type.
112    *
113    * @param schema The new value for the <property>schema</property> property on this bean.
114    * @return This object (for method chaining).
115    */
116   public ResponseInfo setSchema(SchemaInfo schema) {
117      this.schema = schema;
118      return this;
119   }
120
121   /**
122    * Synonym for {@link #setSchema(SchemaInfo)}.
123    *
124    * @param schema The new value for the <property>schema</property> property on this bean.
125    * @return This object (for method chaining).
126    */
127   public ResponseInfo schema(SchemaInfo schema) {
128      return setSchema(schema);
129   }
130
131   /**
132    * Bean property getter:  <property>headers</property>.
133    * <p>
134    * A list of headers that are sent with the response.
135    *
136    * @return The value of the <property>headers</property> property on this bean, or <jk>null</jk> if it is not set.
137    */
138   public Map<String,HeaderInfo> getHeaders() {
139      return headers;
140   }
141
142   /**
143    * Bean property setter:  <property>headers</property>.
144    * <p>
145    * A list of headers that are sent with the response.
146    *
147    * @param headers The new value for the <property>headers</property> property on this bean.
148    * @return This object (for method chaining).
149    */
150   public ResponseInfo setHeaders(Map<String,HeaderInfo> headers) {
151      this.headers = headers;
152      return this;
153   }
154
155   /**
156    * Bean property adder:  <property>headers</property>.
157    * <p>
158    * A list of headers that are sent with the response.
159    *
160    * @param name The header name.
161    * @param header The header descriptions
162    * @return This object (for method chaining).
163    */
164   public ResponseInfo addHeader(String name, HeaderInfo header) {
165      if (headers == null)
166         headers = new TreeMap<String,HeaderInfo>();
167      headers.put(name, header);
168      return this;
169   }
170
171   /**
172    * Synonym for {@link #addHeader(String,HeaderInfo)}.
173    *
174    * @param name The header name.
175    * @param header The header descriptions
176    * @return This object (for method chaining).
177    */
178   public ResponseInfo header(String name, HeaderInfo header) {
179      return addHeader(name, header);
180   }
181
182   /**
183    * Bean property getter:  <property>examples</property>.
184    * <p>
185    * An example of the response message.
186    * Keys must be MIME-type strings.
187    *
188    * @return The value of the <property>examples</property> property on this bean, or <jk>null</jk> if it is not set.
189    */
190   public Map<String,Object> getExamples() {
191      return examples;
192   }
193
194   /**
195    * Bean property setter:  <property>examples</property>.
196    * <p>
197    * An example of the response message.
198    * Keys must be MIME-type strings.
199    *
200    * @param examples The new value for the <property>examples</property> property on this bean.
201    * @return This object (for method chaining).
202    */
203   public ResponseInfo setExamples(Map<String,Object> examples) {
204      this.examples = examples;
205      return this;
206   }
207
208   /**
209    * Bean property adder:  <property>examples</property>.
210    * <p>
211    * An example of the response message.
212    *
213    * @param mimeType The mimeType of the example.
214    * @param example The example output.
215    * @return This object (for method chaining).
216    */
217   public ResponseInfo addExample(String mimeType, Object example) {
218      if (examples == null)
219         examples = new TreeMap<String,Object>();
220      examples.put(mimeType, example);
221      return this;
222   }
223
224   /**
225    * Synonym for {@link #addExample(String,Object)}.
226    *
227    * @param mimeType The mimeType of the example.
228    * @param example The example output.
229    * @return This object (for method chaining).
230    */
231   public ResponseInfo example(String mimeType, Object example) {
232      return addExample(mimeType, example);
233   }
234
235   /**
236    * Synonym for {@link #setExamples(Map)}.
237    *
238    * @param examples The new value for the <property>examples</property> property on this bean.
239    * @return This object (for method chaining).
240    */
241   public ResponseInfo examples(Map<String,Object> examples) {
242      return setExamples(examples);
243   }
244}
