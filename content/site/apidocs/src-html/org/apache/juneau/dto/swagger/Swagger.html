379   /**
380    * Bean property adder:  <property>paths</property>.
381    * <p>
382    * Required. The available paths and operations for the API.
383    *
384    * @param path The path template.
385    * @param methodName The HTTP method name.
386    * @param operation The operation that describes the path.
387    * @return This object (for method chaining).
388    */
389   public Swagger addPath(String path, String methodName, Operation operation) {
390      if (paths == null)
391         paths = new TreeMap<String,Map<String,Operation>>();
392      Map<String,Operation> p = paths.get(path);
393      if (p == null) {
394         p = new TreeMap<String,Operation>(new MethodSorter());
395         paths.put(path, p);
396      }
397      p.put(methodName, operation);
398      return this;
399   }
