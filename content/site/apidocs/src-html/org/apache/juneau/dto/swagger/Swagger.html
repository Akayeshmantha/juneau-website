452   /**
453    * Bean property adder:  <property>paths</property>.
454    * <p>
455    * Required. The available paths and operations for the API.
456    *
457    * @param path The path template.
458    * @param methodName The HTTP method name.
459    * @param operation The operation that describes the path.
460    * @return This object (for method chaining).
461    */
462   public Swagger addPath(String path, String methodName, Operation operation) {
463      if (paths == null)
464         paths = new TreeMap<String,Map<String,Operation>>();
465      Map<String,Operation> p = paths.get(path);
466      if (p == null) {
467         p = new TreeMap<String,Operation>(new MethodSorter());
468         paths.put(path, p);
469      }
470      p.put(methodName, operation);
471      return this;
472   }
