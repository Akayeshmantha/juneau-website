013package org.apache.juneau.dto.html5;
014
015import org.apache.juneau.annotation.*;
016
017/**
018 * DTO for an HTML <a class="doclink" href="https://www.w3.org/TR/html5/forms.html#the-select-element">&lt;select&gt;</a> element.
019 * <p>
020 */
021@Bean(typeName="select")
022public class Select extends HtmlElementContainer {
023
024   /**
025    * <a class="doclink" href="https://www.w3.org/TR/html5/forms.html#attr-fe-autofocus">autofocus</a> attribute.
026    * Automatically focus the form control when the page is loaded.
027    * @param autofocus - The new value for this attribute.
028    *    Typically a {@link Boolean} or {@link String}.
029    * @return This object (for method chaining).
030    */
031   public final Select autofocus(Object autofocus) {
032      attr("autofocus", autofocus);
033      return this;
034   }
035
036   /**
037    * <a class="doclink" href="https://www.w3.org/TR/html5/forms.html#attr-fe-disabled">disabled</a> attribute.
038    * Whether the form control is disabled.
039    * @param disabled - The new value for this attribute.
040    *    Typically a {@link Boolean} or {@link String}.
041    * @return This object (for method chaining).
042    */
043   public final Select disabled(Object disabled) {
044      attr("disabled", disabled);
045      return this;
046   }
047
048   /**
049    * <a class="doclink" href="https://www.w3.org/TR/html5/forms.html#attr-fae-form">form</a> attribute.
050    * Associates the control with a form element.
051    * @param form - The new value for this attribute.
052    * @return This object (for method chaining).
053    */
054   public final Select form(String form) {
055      attr("form", form);
056      return this;
057   }
058
059   /**
060    * <a class="doclink" href="https://www.w3.org/TR/html5/forms.html#attr-select-multiple">multiple</a> attribute.
061    * Whether to allow multiple values.
062    * @param multiple - The new value for this attribute.
063    *    Typically a {@link Boolean} or {@link String}.
064    * @return This object (for method chaining).
065    */
066   public final Select multiple(Object multiple) {
067      attr("multiple", multiple);
068      return this;
069   }
070
071   /**
072    * <a class="doclink" href="https://www.w3.org/TR/html5/forms.html#attr-fe-name">name</a> attribute.
073    * Name of form control to use for form submission and in the form.elements API.
074    * @param name - The new value for this attribute.
075    * @return This object (for method chaining).
076    */
077   public final Select name(String name) {
078      attr("name", name);
079      return this;
080   }
081
082   /**
083    * <a class="doclink" href="https://www.w3.org/TR/html5/forms.html#attr-select-required">required</a> attribute.
084    * Whether the control is required for form submission.
085    * @param required - The new value for this attribute.
086    *    Typically a {@link Boolean} or {@link String}.
087    * @return This object (for method chaining).
088    */
089   public final Select required(Object required) {
090      attr("required", required);
091      return this;
092   }
093
094   /**
095    * <a class="doclink" href="https://www.w3.org/TR/html5/forms.html#attr-select-size">size</a> attribute.
096    * Size of the control.
097    * @param size - The new value for this attribute.
098    *    Typically a {@link Number} or {@link String}.
099    * @return This object (for method chaining).
100    */
101   public final Select size(Object size) {
102      attr("size", size);
103      return this;
104   }
105
106   //--------------------------------------------------------------------------------
107   // Overridden methods
108   //--------------------------------------------------------------------------------
109
110   @Override /* HtmlElement */
111   public final Select _class(String _class) {
112      super._class(_class);
113      return this;
114   }
115
116   @Override /* HtmlElement */
117   public final Select id(String id) {
118      super.id(id);
119      return this;
120   }
121
122   @Override /* HtmlElementContainer */
123   public final Select children(Object...children) {
124      super.children(children);
125      return this;
126   }
127
128   @Override /* HtmlElementContainer */
129   public final Select child(Object child) {
130      super.child(child);
131      return this;
132   }
133}
