053   /**
054    * <b>Configuration property:</b>  Encode non-valid URI characters.
055    * <p>
056    * <ul>
057    *    <li><b>Name:</b> <js>"UonSerializer.encodeChars"</js>
058    *    <li><b>Data type:</b> <code>Boolean</code>
059    *    <li><b>Default:</b> <jk>false</jk> for {@link UonSerializer}, <jk>true</jk> for {@link UrlEncodingSerializer}
060    *    <li><b>Session-overridable:</b> <jk>true</jk>
061    * </ul>
062    * <p>
063    * Encode non-valid URI characters with <js>"%xx"</js> constructs.
064    * <p>
065    * If <jk>true</jk>, non-valid URI characters will be converted to <js>"%xx"</js> sequences.
066    * Set to <jk>false</jk> if parameter value is being passed to some other code that will already
067    *    perform URL-encoding of non-valid URI characters.
068    * <p>
069    * <h5 class='section'>Notes:</h5>
070    * <ul>
071    *    <li>This is equivalent to calling <code>property(<jsf>UON_encodeChars</jsf>, value)</code>.
072    *    <li>This introduces a slight performance penalty.
073    * </ul>
074    *
075    * @param value The new value for this property.
076    * @return This object (for method chaining).
077    * @see UonSerializerContext#UON_encodeChars
078    */
079   public UonSerializerBuilder encodeChars(boolean value) {
080      return property(UON_encodeChars, value);
081   }
082
083   /**
084    * Shortcut for calling <code>setEncodeChars(true)</code>.
085    *
086    * @return This object (for method chaining).
087    */
088   public UonSerializerBuilder encoding() {
089      return encodeChars(true);
090   }
