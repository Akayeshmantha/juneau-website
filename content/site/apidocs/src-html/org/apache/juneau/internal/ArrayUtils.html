025   /**
026    * Appends one or more elements to an array.
027    *
028    * @param <T> The element type.
029    * @param array The array to append to.
030    * @param newElements The new elements to append to the array.
031    * @return A new array with the specified elements appended.
032    */
033   @SuppressWarnings("unchecked")
034   public static <T> T[] append(T[] array, T...newElements) {
035      if (array == null)
036         return newElements;
037      if (newElements.length == 0)
038         return array;
039      T[] a = (T[])Array.newInstance(array.getClass().getComponentType(), array.length + newElements.length);
040      for (int i = 0; i < array.length; i++)
041         a[i] = array[i];
042      for (int i = 0; i < newElements.length; i++)
043         a[i+array.length] = newElements[i];
044      return a;
045   }

047   /**
048    * Appends one or more elements to an array.
049    *
050    * @param <T> The element type.
051    * @param array The array to append to.
052    * @param newElements The new elements to append to the array.
053    * @return A new array with the specified elements appended.
054    */
055   @SuppressWarnings("unchecked")
056   public static <T> T[] append(T[] array, Collection<T> newElements) {
057      assertFieldNotNull(array, "array");
058      if (newElements.size() == 0)
059         return array;
060      T[] a = (T[])Array.newInstance(array.getClass().getComponentType(), array.length + newElements.size());
061      for (int i = 0; i < array.length; i++)
062         a[i] = array[i];
063      int l = array.length;
064      for (T t : newElements)
065         a[l++] = t;
066      return a;
067   }

069   /**
070    * Combine an arbitrary number of arrays into a single array.
071    *
072    * @param arrays Collection of arrays to combine.
073    * @return A new combined array, or <jk>null</jk> if all arrays are <jk>null</jk>.
074    */
075   @SuppressWarnings("unchecked")
076   public static <T> T[] combine(T[]...arrays) {
077      assertFieldNotNull(arrays, "arrays");
078      int l = 0;
079      T[] a1 = null;
080      for (T[] a : arrays) {
081         if (a1 == null && a != null)
082            a1 = a;
083         l += (a == null ? 0 : a.length);
084      }
085      if (a1 == null)
086         return null;
087      T[] a = (T[])Array.newInstance(a1.getClass().getComponentType(), l);
088      int i = 0;
089      for (T[] aa : arrays)
090         if (aa != null)
091            for (T t : aa)
092               a[i++] = t;
093      return a;
094   }

096   /**
097    * Creates a new array with reversed entries.
098    *
099    * @param <T> The class type of the array.
100    * @param array The array to reverse.
101    * @return A new array with reversed entries, or <jk>null</jk> if the array was <jk>null</jk>.
102    */
103   @SuppressWarnings("unchecked")
104   public static <T> T[] reverse(T[] array) {
105      assertFieldNotNull(array, "array");
106      Class<T> c = (Class<T>)array.getClass().getComponentType();
107      T[] a2 = (T[])Array.newInstance(c, array.length);
108      for (int i = 0; i < array.length; i++)
109         a2[a2.length-i-1] = array[i];
110      return a2;
111   }

113   /**
114    * Converts the specified array to a <code>Set</code>.
115    * <p>
116    * The order of the entries in the set are the same as the array.
117    *
118    * @param <T> The entry type of the array.
119    * @param array The array being wrapped in a <code>Set</code> interface.
120    * @return The new set.
121    */
122   public static <T> Set<T> asSet(final T[] array) {
123      assertFieldNotNull(array, "array");
124      return new AbstractSet<T>() {
125
126         @Override /* Set */
127         public Iterator<T> iterator() {
128            return new Iterator<T>() {
129               int i = 0;
130
131               @Override /* Iterator */
132               public boolean hasNext() {
133                  return i < array.length;
134               }
135
136               @Override /* Iterator */
137               public T next() {
138                  if (i >= array.length)
139                     throw new NoSuchElementException();
140                  T t = array[i];
141                  i++;
142                  return t;
143               }
144
145               @Override /* Iterator */
146               public void remove() {
147                  throw new UnsupportedOperationException();
148               }
149            };
150         }
151
152         @Override /* Set */
153         public int size() {
154            return array.length;
155         }
156      };
157   }

159   /**
160    * Returns an iterator against an array.
161    * This works with any array type (e.g. <code>String[]</code>, <code>Object[]</code>, <code><jk>int</jk>[]</code>, etc...).
162    *
163    * @param array The array to create an iterator over.
164    * @return An iterator over the specified array.
165    */
166   public static Iterator<Object> iterator(final Object array) {
167      return new Iterator<Object>() {
168         int i = 0;
169         int length = array == null ? 0 : Array.getLength(array);
170
171         @Override /* Iterator */
172         public boolean hasNext() {
173            return i < length;
174         }
175
176         @Override /* Iterator */
177         public Object next() {
178            if (i >= length)
179               throw new NoSuchElementException();
180            return Array.get(array, i++);
181         }
182
183         @Override /* Iterator */
184         public void remove() {
185            throw new UnsupportedOperationException();
186         }
187      };
188   }

190   /**
191    * Converts the specified collection to an array.
192    * Works on both object and primitive arrays.
193    *
194    * @param c The collection to convert to an array.
195    * @param componentType The component type of the collection.
196    * @return A new array.
197    */
198   public static <T> Object toArray(Collection<?> c, Class<T> componentType) {
199      Object a = Array.newInstance(componentType, c.size());
200      Iterator<?> it = c.iterator();
201      int i = 0;
202      while (it.hasNext())
203         Array.set(a, i++, it.next());
204      return a;
205   }

216   /**
217    * Converts the specified array to an <code>ArrayList</code>
218    * 
219    * @param array The array to convert.
220    * @param componentType The type of objects in the array.
221    *    It must match the actual component type in the array.
222    * @return A new {@link ArrayList}
223    */
224   @SuppressWarnings("unchecked")
225   public static <T> List<T> toList(Object array, Class<T> componentType) {
226      List<T> l = new ArrayList<T>(Array.getLength(array));
227      for (int i = 0; i < Array.getLength(array); i++)
228         l.add((T)Array.get(array, i));
229      return l;
230   }

232   /**
233    * Shortcut for calling <code>myList.toArray(new T[myList.size()]);</code>
234    *
235    * @param c The collection being converted to an array.
236    * @param componentType The component type of the array.
237    * @return The collection converted to an array.
238    */
239   @SuppressWarnings("unchecked")
240   public static <T> T[] toObjectArray(Collection<?> c, Class<T> componentType) {
241      Object a = Array.newInstance(componentType, c.size());
242      Iterator<?> it = c.iterator();
243      int i = 0;
244      while (it.hasNext())
245         Array.set(a, i++, it.next());
246      return (T[])a;
247   }

249   /**
250    * Copies the specified array into the specified list.
251    * Works on both object and primitive arrays.
252    *
253    * @param array The array to copy into a list.
254    * @param list The list to copy the values into.
255    * @return The same list passed in.
256    */
257   @SuppressWarnings({"unchecked","rawtypes"})
258   public static List copyToList(Object array, List list) {
259      if (array != null) {
260         int length = Array.getLength(array);
261         for (int i = 0; i < length; i++)
262            list.add(Array.get(array, i));
263      }
264      return list;
265   }

268    * Returns <jk>true</jk> if the specified array contains the specified element

271    * @param element The element to check for.
272    * @param array The array to check.

277      return indexOf(element, array) != -1;

280   /**
281    * Returns the index position of the element in the specified array
282    *    using the {@link Object#equals(Object)} method.
283    *
284    * @param element The element to check for.
285    * @param array The array to check.
286    * @return The index position of the element in the specified array, or
287    *    <code>-1</code> if the array doesn't contain the element, or the array or element is <jk>null</jk>.
288    */
289   public static <T> int indexOf(T element, T[] array) {
290      if (element == null)
291         return -1;
292      if (array == null)
293         return -1;
294      for (int i = 0; i < array.length; i++)
295         if (element.equals(array[i]))
296            return i;
297      return -1;
298   }

333   /**
334    * Converts a primitive wrapper array (e.g. <code>Integer[]</code>) to a primitive array (e.g. <code><jk>int</jk>[]</code>).
335    *
336    * @param o The array to convert.  Must be a primitive wrapper array.
337    * @return A new array.
338    * @throws IllegalArgumentException If object is not a wrapper object array.
339    */
340   public static Object toPrimitiveArray(Object o) {
341      Class<?> c = o.getClass();
342      if (! c.isArray())
343         throw new IllegalArgumentException("Cannot pass non-array objects to toPrimitiveArray()");
344      int l = Array.getLength(o);
345      Class<?> tc = ClassUtils.getPrimitiveForWrapper(c.getComponentType());
346      if (tc == null)
347         throw new IllegalArgumentException("Array type is not a primitive wrapper array.");
348      Object a = Array.newInstance(tc, l);
349      for (int i = 0; i < l; i++)
350         Array.set(a, i, Array.get(o, i));
351      return a;
352   }
