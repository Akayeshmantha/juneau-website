030   /**
031    * Reads the contents of a file into a string.
032    *
033    * @param path The path of the file to read using default character encoding.
034    * @return The contents of the reader as a string, or <jk>null</jk> if file does not exist.
035    * @throws IOException If a problem occurred trying to read from the reader.
036    */
037   public static String readFile(String path) throws IOException {
038      return read(new File(path));
039   }

041   /**
042    * Reads the contents of a file into a string.
043    *
044    * @param in The file to read using default character encoding.
045    * @return The contents of the reader as a string, or <jk>null</jk> if file does not exist.
046    * @throws IOException If a problem occurred trying to read from the reader.
047    */
048   public static String read(File in) throws IOException {
049      if (in == null || ! in.exists())
050         return null;
051      Reader r = new InputStreamReader(new FileInputStream(in), Charset.defaultCharset());
052      return read(r, 0, 1024);
053   }

055   /**
056    * Writes the contents of the specified <code>Reader</code> to the specified file.
057    *
058    * @param out The file to write the output to.
059    * @param in The reader to pipe from.
060    * @return The number of characters written to the file.
061    * @throws IOException
062    */
063   public static int write(File out, Reader in) throws IOException {
064      assertFieldNotNull(out, "out");
065      assertFieldNotNull(in, "in");
066      Writer w = new OutputStreamWriter(new FileOutputStream(out), Charset.defaultCharset());
067      try {
068         return IOPipe.create(in, w).closeOut().run();
069      } finally {
070         w.close();
071      }
072   }

131   /**
132    * Read the specified input stream into a byte array and closes the stream.
133    *
134    * @param in The input stream.
135    * @param bufferSize The expected size of the buffer.
136    * @return The contents of the stream as a byte array.
137    * @throws IOException Thrown by underlying stream.
138    */
139   public static byte[] readBytes(InputStream in, int bufferSize) throws IOException {
140      if (in == null)
141         return null;
142      ByteArrayOutputStream buff = new ByteArrayOutputStream(bufferSize);
143      int nRead;
144      byte[] b = new byte[Math.min(bufferSize, 8192)];
145
146      try {
147         while ((nRead = in.read(b, 0, b.length)) != -1)
148              buff.write(b, 0, nRead);
149
150            buff.flush();
151
152            return buff.toByteArray();
153      } finally {
154         in.close();
155      }
156   }
157
158   /**
159    * Reads a raw stream of bytes from the specified file.
160    *
161    * @param f The file to read.
162    * @return A byte array containing the contents of the file.
163    * @throws IOException
164    */
165   public static byte[] readBytes(File f) throws IOException {
166      if (f == null || ! (f.exists() && f.canRead()))
167         return null;
168
169      FileInputStream fis = new FileInputStream(f);
170      try {
171         return readBytes(fis, (int)f.length());
172      } finally {
173         fis.close();
174      }
175   }

177   /**
178    * Reads the specified input into a {@link String} until the end of the input is reached.
179    * <p>
180    * The {@code Reader} is automatically closed.
181    * <p>
182    * If the {@code Reader} is not an instance of a {@code BufferedReader}, then it gets wrapped in a {@code BufferedReader}.
183    *
184    * @param in The input reader.
185    * @param length Specify a positive number if the length of the input is known.
186    * @param bufferSize Specify the buffer size to use.
187    * @return The contents of the reader as a string.  <jk>null</jk> if reader was null.
188    * @throws IOException If a problem occurred trying to read from the reader.
189    */
190   public static String read(Reader in, int length, int bufferSize) throws IOException {
191      if (in == null)
192         return null;
193      length = (length <= 0 ? bufferSize : length);
194      StringBuilder sb = new StringBuilder(length); // Assume they're ASCII characters.
195      try {
196         char[] buf = new char[Math.min(bufferSize, length)];
197         int i = 0;
198         while ((i = in.read(buf)) != -1)
199            sb.append(buf, 0, i);
200         return sb.toString();
201      } finally {
202         in.close();
203      }
204   }

206   /**
207    * Pipes the contents of the specified reader into the writer.
208    * The reader is closed, the writer is not.
209    *
210    * @param in The reader to pipe from.
211    * @param out The writer to pipe to.
212    * @throws IOException
213    */
214   public static void pipe(Reader in, Writer out) throws IOException {
215      assertFieldNotNull(out, "out");
216      assertFieldNotNull(in, "in");
217      IOPipe.create(in, out).run();
218   }

224    * @return The reader wrapped in a {@link BufferedReader}, or the original {@link Reader} if it's already

228      if (r instanceof BufferedReader || r instanceof StringReader)
229         return r;

233   /**
234    * Counts the number of bytes in the input stream and then closes the stream.
235    *
236    * @param is The input stream to read from.
237    * @return The number of bytes read.
238    * @throws IOException
239    */
240   public static long count(InputStream is) throws IOException {
241      assertFieldNotNull(is, "is");
242      long c = 0;
243      long i;
244      try {
245         while ((i = is.skip(1024)) != 0)
246            c += i;
247      } finally {
248         is.close();
249      }
250      return c;
251   }

273   /**
274    * Given the specified <js>"Content-Length"</js> header value, return an appropriate buffer size.
275    * The maximum buffer size is 1MB.
276    *
277    * @param contentLength The value of the <js>"Content-Length"</js> header.
278    * @return The appropriate buffer size.
279    */
280   public static int getBufferSize(String contentLength) {
281      try {
282         if (! StringUtils.isEmpty(contentLength)) {
283            long l = Long.decode(contentLength);
284            if (l > 1048576)
285               return 1048576;
286            if (l <= 0)
287               return 8192;
288            return (int)l;
289         }
290      } catch (Exception e) {
291         return 8192;
292      }
293      return 8192;
294   }

348   /**
349    * Quietly close all specified input streams, output streams, readers, and writers.
350    *
351    * @param o The list of all objects to quietly close.
352    */
353   public static void closeQuietly(Object...o) {
354      for (Object o2 : o) {
355         if (o2 instanceof InputStream)
356            closeQuietly((InputStream)o2);
357         if (o2 instanceof OutputStream)
358            closeQuietly((OutputStream)o2);
359         if (o2 instanceof Reader)
360            closeQuietly((Reader)o2);
361         if (o2 instanceof Writer)
362            closeQuietly((Writer)o2);
363      }
364   }
