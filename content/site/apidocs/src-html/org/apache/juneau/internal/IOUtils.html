030   /**

032    *

041   /**
042    * Reads the contents of a file into a string.
043    *
044    * @param in The file to read using default character encoding.
045    * @return The contents of the reader as a string, or <jk>null</jk> if file does not exist.
046    * @throws IOException If a problem occurred trying to read from the reader.
047    */
048   public static String read(File in) throws IOException {
049      if (in == null || ! in.exists())
050         return null;
051      Reader r = new InputStreamReader(new FileInputStream(in), Charset.defaultCharset());
052      return read(r, 0, 1024);
053   }

055   /**
056    * Reads the specified object to a <code>String</code>.
057    *
058    * <p>
059    * Can be any of the following object types:
060    * <ul>
061    *    <li>{@link CharSequence}
062    *    <li>{@link File}
063    *    <li>{@link Reader}
064    *    <li>{@link InputStream}
065    *    <li><code><jk>byte</jk>[]</code>
066    * </ul>
067    *
068    * @param o The object to read.
069    * @return The object serialized to a string, or <jk>null</jk> if it wasn't a supported type.
070    * @throws IOException
071    */
072   public static String read(Object o) throws IOException {
073      if (o instanceof CharSequence)
074         return o.toString();
075      if (o instanceof File)
076         return read((File)o);
077      if (o instanceof Reader)
078         return read((Reader)o);
079      if (o instanceof InputStream)
080         return read((InputStream)o);
081      if (o instanceof byte[])
082         return read(new ByteArrayInputStream((byte[])o));
083      return null;
084   }

162   /**
163    * Read the specified input stream into a byte array and closes the stream.
164    *
165    * @param in The input stream.
166    * @param bufferSize The expected size of the buffer.
167    * @return The contents of the stream as a byte array.
168    * @throws IOException Thrown by underlying stream.
169    */
170   public static byte[] readBytes(InputStream in, int bufferSize) throws IOException {
171      if (in == null)
172         return null;
173      ByteArrayOutputStream buff = new ByteArrayOutputStream(bufferSize);
174      int nRead;
175      byte[] b = new byte[Math.min(bufferSize, 8192)];
176
177      try {
178         while ((nRead = in.read(b, 0, b.length)) != -1)
179            buff.write(b, 0, nRead);
180         buff.flush();
181
182         return buff.toByteArray();
183      } finally {
184         in.close();
185      }
186   }

188   /**
189    * Reads a raw stream of bytes from the specified file.
190    *
191    * @param f The file to read.
192    * @return A byte array containing the contents of the file.
193    * @throws IOException
194    */
195   public static byte[] readBytes(File f) throws IOException {
196      if (f == null || ! (f.exists() && f.canRead()))
197         return null;
198
199      FileInputStream fis = new FileInputStream(f);
200      try {
201         return readBytes(fis, (int)f.length());
202      } finally {
203         fis.close();
204      }
205   }

207   /**
208    * Reads the specified input into a {@link String} until the end of the input is reached.
209    *
210    * <p>
211    * The {@code Reader} is automatically closed.
212    *
213    * <p>
214    * If the {@code Reader} is not an instance of a {@code BufferedReader}, then it gets wrapped in a
215    * {@code BufferedReader}.
216    *
217    * @param in The input reader.
218    * @param length Specify a positive number if the length of the input is known.
219    * @param bufferSize Specify the buffer size to use.
220    * @return The contents of the reader as a string.  <jk>null</jk> if reader was null.
221    * @throws IOException If a problem occurred trying to read from the reader.
222    */
223   public static String read(Reader in, int length, int bufferSize) throws IOException {
224      if (in == null)
225         return null;
226      length = (length <= 0 ? bufferSize : length);
227      StringBuilder sb = new StringBuilder(length); // Assume they're ASCII characters.
228      try {
229         char[] buf = new char[Math.min(bufferSize, length)];
230         int i = 0;
231         while ((i = in.read(buf)) != -1)
232            sb.append(buf, 0, i);
233         return sb.toString();
234      } finally {
235         in.close();
236      }
237   }

269   /**
270    * Counts the number of bytes in the input stream and then closes the stream.
271    *
272    * @param is The input stream to read from.
273    * @return The number of bytes read.
274    * @throws IOException
275    */
276   public static long count(InputStream is) throws IOException {
277      assertFieldNotNull(is, "is");
278      long c = 0;
279      long i;
280      try {
281         while ((i = is.skip(1024)) != 0)
282            c += i;
283      } finally {
284         is.close();
285      }
286      return c;
287   }

309   /**
310    * Given the specified <js>"Content-Length"</js> header value, return an appropriate buffer size.
311    *
312    * <p>
313    * The maximum buffer size is 1MB.
314    *
315    * @param contentLength The value of the <js>"Content-Length"</js> header.
316    * @return The appropriate buffer size.
317    */
318   public static int getBufferSize(String contentLength) {
319      try {
320         if (! StringUtils.isEmpty(contentLength)) {
321            long l = Long.decode(contentLength);
322            if (l > 1048576)
323               return 1048576;
324            if (l <= 0)
325               return 8192;
326            return (int)l;
327         }
328      } catch (Exception e) {
329         return 8192;
330      }
331      return 8192;
332   }

394   /**
395    * Quietly close all specified input streams, output streams, readers, and writers.
396    *
397    * @param o The list of all objects to quietly close.
398    */
399   public static void closeQuietly(Object...o) {
400      for (Object o2 : o) {
401         if (o2 instanceof InputStream)
402            closeQuietly((InputStream)o2);
403         if (o2 instanceof OutputStream)
404            closeQuietly((OutputStream)o2);
405         if (o2 instanceof Reader)
406            closeQuietly((Reader)o2);
407         if (o2 instanceof Writer)
408            closeQuietly((Writer)o2);
409      }
410   }

412   /**
413    * Flushes multiple output streams and writers in a single call.
414    *
415    * @param o
416    *    The objects to flush.
417    *    <jk>null</jk> entries are ignored.
418    * @throws IOException
419    */
420   public static void flush(Object...o) throws IOException {
421      IOException ex = null;
422      for (Object o2 : o) {
423         try {
424            if (o2 instanceof OutputStream)
425               ((OutputStream)o2).flush();
426            if (o2 instanceof Writer)
427               ((Writer)o2).flush();
428         } catch (IOException e) {
429            ex = e;
430         }
431      }
432      if (ex != null)
433         throw ex;
434   }

465   /**
466    * Converts an object to an <code>InputStream</code>.
467    *
468    * @param o
469    *    The object to convert to an input stream.
470    *    Can be any of the following:
471    *    <ul>
472    *       <li>{@link InputStream}
473    *       <li>{@link Reader}
474    *       <li>{@link File}
475    *       <li>{@link CharSequence} - Converted to UTF-8 stream.
476    *       <li><code><jk>byte</jk>[]</code>
477    *       <li><code><jk>null</jk></code> - Returns null.
478    *    </ul>
479    * @return The object converted to an input stream.
480    * @throws IOException If invalid object passed in or file could not be read.
481    */
482   public static InputStream toInputStream(Object o) throws IOException {
483      if (o == null)
484         return null;
485      if (o instanceof InputStream)
486         return (InputStream)o;
487      if (o instanceof File)
488         return new FileInputStream((File)o);
489      if (o instanceof byte[])
490         return new ByteArrayInputStream((byte[])o);
491      if (o instanceof CharSequence)
492         return new ByteArrayInputStream(((CharSequence)o).toString().getBytes(UTF8));
493      if (o instanceof Reader)
494         return new ByteArrayInputStream(IOUtils.read((Reader)o).getBytes(UTF8));
495      throw new IOException("Invalid object type passed to IOUtils.toInputStream(Object): " + o.getClass().getName());
496   }
