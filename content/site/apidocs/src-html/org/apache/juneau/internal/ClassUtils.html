065   /**
066    * Converts the specified class name to a readable form when class name is a special construct like <js>"[[Z"</js>.
067    *
068    * <h5 class='section'>Example:</h5>
069    * <p class='bcode'>
070    *    <jsm>getReadableClassName</jsm>(<js>"java.lang.Object"</js>);  <jc>// Returns "java.lang.Object"</jc>
071    *    <jsm>getReadableClassName</jsm>(<js>"boolean"</js>);  <jc>// Returns "boolean"</jc>
072    *    <jsm>getReadableClassName</jsm>(<js>"[Z"</js>);  <jc>// Returns "boolean[]"</jc>
073    *    <jsm>getReadableClassName</jsm>(<js>"[[Z"</js>);  <jc>// Returns "boolean[][]"</jc>
074    *    <jsm>getReadableClassName</jsm>(<js>"[Ljava.lang.Object;"</js>);  <jc>// Returns "java.lang.Object[]"</jc>
075    *    <jsm>getReadableClassName</jsm>(<jk>null</jk>);  <jc>// Returns null</jc>
076    * </p>
077    *
078    * @param className The class name.
079    * @return A readable class type name, or <jk>null</jk> if parameter is <jk>null</jk>.
080    */
081   public static String getReadableClassName(String className) {
082      if (className == null)
083         return null;
084      if (! StringUtils.startsWith(className, '['))
085         return className;
086      int depth = 0;
087      for (int i = 0; i < className.length(); i++) {
088         if (className.charAt(i) == '[')
089            depth++;
090         else
091            break;
092      }
093      char type = className.charAt(depth);
094      String c;
095      switch (type) {
096         case 'Z': c = "boolean"; break;
097         case 'B': c = "byte"; break;
098         case 'C': c = "char"; break;
099         case 'D': c = "double"; break;
100         case 'F': c = "float"; break;
101         case 'I': c = "int"; break;
102         case 'J': c = "long"; break;
103         case 'S': c = "short"; break;
104         default: c = className.substring(depth+1, className.length()-1);
105      }
106      StringBuilder sb = new StringBuilder(c.length() + 2*depth).append(c);
107      for (int i = 0; i < depth; i++)
108         sb.append("[]");
109      return sb.toString();
110   }

175   /**
176    * Returns the signature of the specified method.
177    * For no-arg methods, the signature will be a simple string such as <js>"toString"</js>.
178    * For methods with one or more args, the arguments will be fully-qualified class names (e.g. <js>"append(java.util.StringBuilder,boolean)"</js>)
179    *
180    * @param m The methods to get the signature on.
181    * @return The methods signature.
182    */
183   public static String getMethodSignature(Method m) {
184      StringBuilder sb = new StringBuilder(m.getName());
185      Class<?>[] pt = m.getParameterTypes();
186      if (pt.length > 0) {
187         sb.append('(');
188         for (int i = 0; i < pt.length; i++) {
189            if (i > 0)
190               sb.append(',');
191            sb.append(getReadableClassName(pt[i]));
192         }
193         sb.append(')');
194      }
195      return sb.toString();
196   }

345    * Returns the specified annotation on the specified method.
346    * <p>
347    * Similar to {@link Method#getAnnotation(Class)}, but searches up the parent hierarchy for the annotation
348    * defined on parent classes and interfaces.

356    * @return The annotation, or <jk>null</jk> if it wasn't found.

362   /**
363    * Returns the specified annotation on the specified method.
364    * <p>
365    * Similar to {@link Method#getAnnotation(Class)}, but searches up the parent hierarchy for the annotation defined
366    *    on parent classes and interfaces.
367    * <p>
368    * Normally, annotations defined on methods of parent classes and interfaces are not inherited by the child methods.
369    * This utility method gets around that limitation by searching the class hierarchy for the "same" method
370    *    (i.e. the same name and arguments).
371    *
372    * @param a The annotation to search for.
373    * @param c The child class to start searching from.
374    *    Note that it can be a descendant class of the actual declaring class of the method passed in.
375    *    This allows you to find annotations on methods overridden by the method passed in.
376    * @param method The method to search.
377    * @return The annotation, or <jk>null</jk> if it wasn't found.
378    */
379   public static <T extends Annotation> T getMethodAnnotation(Class<T> a, Class<?> c, Method method) {
380      for (Method m : c.getDeclaredMethods()) {
381         if (isSameMethod(method, m)) {
382            T t = m.getAnnotation(a);
383            if (t != null)
384               return t;
385         }
386      }
387      Class<?> pc = c.getSuperclass();
388      if (pc != null) {
389         T t = getMethodAnnotation(a, pc, method);
390         if (t != null)
391            return t;
392      }
393      for (Class<?> ic : c.getInterfaces()) {
394         T t = getMethodAnnotation(a, ic, method);
395         if (t != null)
396            return t;
397      }
398      return null;
399   }

405   /**
406    * Locates the no-arg constructor for the specified class.
407    * Constructor must match the visibility requirements specified by parameter 'v'.
408    * If class is abstract, always returns <jk>null</jk>.
409    * Note that this also returns the 1-arg constructor for non-static member classes.
410    *
411    * @param c The class from which to locate the no-arg constructor.
412    * @param v The minimum visibility.
413    * @return The constructor, or <jk>null</jk> if no no-arg constructor exists with the required visibility.
414    */
415   @SuppressWarnings({"rawtypes","unchecked"})
416   public static final <T> Constructor<T> findNoArgConstructor(Class<T> c, Visibility v) {
417      int mod = c.getModifiers();
418      if (Modifier.isAbstract(mod))
419         return null;
420      boolean isMemberClass = c.isMemberClass() && ! isStatic(c);
421      for (Constructor cc : c.getConstructors()) {
422         mod = cc.getModifiers();
423         if (cc.getParameterTypes().length == (isMemberClass ? 1 : 0) && v.isVisible(mod) && isNotDeprecated(cc))
424            return v.transform(cc);
425      }
426      return null;
427   }

429   /**
430    * Finds the real parameter type of the specified class.
431    *
432    * @param c The class containing the parameters (e.g. PojoSwap&lt;T,S&gt;)
433    * @param index The zero-based index of the parameter to resolve.
434    * @param oc The class we're trying to resolve the parameter type for.
435    * @return The resolved real class.
436    */
437   public static Class<?> resolveParameterType(Class<?> c, int index, Class<?> oc) {
438
439      // We need to make up a mapping of type names.
440      Map<Type,Type> typeMap = new HashMap<Type,Type>();
441      while (c != oc.getSuperclass()) {
442         extractTypes(typeMap, oc);
443         oc = oc.getSuperclass();
444      }
445
446      ParameterizedType opt = (ParameterizedType)oc.getGenericSuperclass();
447      Type actualType = opt.getActualTypeArguments()[index];
448
449      if (typeMap.containsKey(actualType))
450         actualType = typeMap.get(actualType);
451
452      if (actualType instanceof Class) {
453         return (Class<?>)actualType;
454
455      } else if (actualType instanceof GenericArrayType) {
456         Class<?> cmpntType = (Class<?>)((GenericArrayType)actualType).getGenericComponentType();
457         return Array.newInstance(cmpntType, 0).getClass();
458
459      } else if (actualType instanceof TypeVariable) {
460         TypeVariable<?> typeVariable = (TypeVariable<?>)actualType;
461         List<Class<?>> nestedOuterTypes = new LinkedList<Class<?>>();
462         for (Class<?> ec = oc.getEnclosingClass(); ec != null; ec = ec.getEnclosingClass()) {
463            try {
464               Class<?> outerClass = oc.getClass();
465               nestedOuterTypes.add(outerClass);
466               Map<Type,Type> outerTypeMap = new HashMap<Type,Type>();
467               extractTypes(outerTypeMap, outerClass);
468               for (Map.Entry<Type,Type> entry : outerTypeMap.entrySet()) {
469                  Type key = entry.getKey(), value = entry.getValue();
470                  if (key instanceof TypeVariable) {
471                     TypeVariable<?> keyType = (TypeVariable<?>)key;
472                     if (keyType.getName().equals(typeVariable.getName()) && isInnerClass(keyType.getGenericDeclaration(), typeVariable.getGenericDeclaration())) {
473                        if (value instanceof Class)
474                           return (Class<?>)value;
475                        typeVariable = (TypeVariable<?>)entry.getValue();
476                     }
477                  }
478               }
479            } catch (Exception e) {
480               throw new RuntimeException(e);
481            }
482         }
483         throw new RuntimeException("Could not resolve type: " + actualType);
484      } else {
485         throw new RuntimeException("Invalid type found in resolveParameterType: " + actualType);
486      }
487   }

489   private static boolean isInnerClass(GenericDeclaration od, GenericDeclaration id) {
490      if (od instanceof Class && id instanceof Class) {
491         Class<?> oc = (Class<?>)od;
492         Class<?> ic = (Class<?>)id;
493         while ((ic = ic.getEnclosingClass()) != null)
494            if (ic == oc)
495               return true;
496      }
497      return false;
498   }

500   private static void extractTypes(Map<Type,Type> typeMap, Class<?> c) {
501      Type gs = c.getGenericSuperclass();
502      if (gs instanceof ParameterizedType) {
503         ParameterizedType pt = (ParameterizedType)gs;
504         Type[] typeParameters = ((Class<?>)pt.getRawType()).getTypeParameters();
505         Type[] actualTypeArguments = pt.getActualTypeArguments();
506         for (int i = 0; i < typeParameters.length; i++) {
507            if (typeMap.containsKey(actualTypeArguments[i]))
508               actualTypeArguments[i] = typeMap.get(actualTypeArguments[i]);
509            typeMap.put(typeParameters[i], actualTypeArguments[i]);
510         }
511      }
512   }

514   /**
515    * Finds a public method with the specified parameters.
516    *
517    * @param c The class to look for the method.
518    * @param name The method name.
519    * @param returnType The return type of the method.
520    * Can be a super type of the actual return type.
521    * For example, if the actual return type is <code>CharSequence</code>, then <code>Object</code> will match but <code>String</code> will not.
522    * @param parameterTypes The parameter types of the method.
523    * Can be subtypes of the actual parameter types.
524    * For example, if the parameter type is <code>CharSequence</code>, then <code>String</code> will match but <code>Object</code> will not.
525    * @return The matched method, or <jk>null</jk> if no match was found.
526    */
527   public static Method findPublicMethod(Class<?> c, String name, Class<?> returnType, Class<?>...parameterTypes) {
528      for (Method m : c.getMethods()) {
529         if (isPublic(m) && m.getName().equals(name)) {
530            Class<?> rt = m.getReturnType();
531            if (isParentClass(returnType, rt)) {
532               Class<?>[] pt = m.getParameterTypes();
533               if (pt.length == parameterTypes.length) {
534                  boolean matches = true;
535                  for (int i = 0; i < pt.length; i++) {
536                     if (! isParentClass(pt[i], parameterTypes[i])) {
537                        matches = false;
538                        break;
539                     }
540                  }
541                  if (matches)
542                     return m;
543               }
544            }
545         }
546      }
547      return null;
548   }

550   /**
551    * Finds a public constructor with the specified parameters without throwing an exception.
552    *
553    * @param c The class to search for a constructor.
554    * @param parameterTypes The parameter types in the constructor.
555    * Can be subtypes of the actual constructor argument types.
556    * @return The matching constructor, or <jk>null</jk> if constructor could not be found.
557    */
558   @SuppressWarnings("unchecked")
559   public static <T> Constructor<T> findPublicConstructor(Class<T> c, Class<?>...parameterTypes) {
560      for (Constructor<?> n : c.getConstructors()) {
561         if (isPublic(n)) {
562            Class<?>[] pt = n.getParameterTypes();
563            if (pt.length == parameterTypes.length) {
564               boolean matches = true;
565               for (int i = 0; i < pt.length; i++) {
566                  if (! isParentClass(pt[i], parameterTypes[i])) {
567                     matches = false;
568                     break;
569                  }
570               }
571               if (matches)
572                  return (Constructor<T>)n;
573            }
574         }
575      }
576      return null;
577   }

754   /**
755    * Creates an instance of the specified class without throwing exceptions.
756    *
757    * @param c The class to cast to.
758    * @param c2 The class to instantiate.
759    *    Can also be an instance of the class.
760    * @param args The arguments to pass to the constructor.
761    * @return The new class instance, or <jk>null</jk> if the class was <jk>null</jk> or is abstract or an interface.
762    * @throws RuntimeException if constructor could not be found or called.
763    */
764   @SuppressWarnings("unchecked")
765   public static <T> T newInstance(Class<T> c, Object c2, Object...args) {
766      if (c2 == null)
767         return null;
768      if (c2 instanceof Class) {
769         try {
770            Class<?> c3 = (Class<?>)c2;
771            if (c3.isInterface() || isAbstract(c3))
772               return null;
773            if (args.length == 0)
774               return (T)c3.newInstance();
775            return (T)c3.getConstructor(getClasses(args)).newInstance(args);
776         } catch (Exception e) {
777            throw new FormattedRuntimeException(e, "Could not instantiate class {0}", c.getName());
778         }
779      } else if (isParentClass(c, c2.getClass())) {
780         return (T)c2;
781      } else {
782         throw new FormattedRuntimeException("Object of type {0} found but was expecting {1}.", c2.getClass(), c.getClass());
783      }
784   }

788    *
789    * @param c The class to get all fields on.

795      return new Iterable<Field>() {
796         @Override
797         public Iterator<Field> iterator() {

799               final Iterator<Class<?>> classIterator = getParentClasses(c, parentFirst, false);

801               int fIndex = 0;

896   /**
897    * Returns a list of all the parent classes of the specified class including the class itself.
898    *
899    * @param c The class to retrieve the parent classes.
900    * @param parentFirst In parent-to-child order, otherwise child-to-parent.
901    * @param includeInterfaces Include interfaces.
902    * @return An iterator of parent classes in the class hierarchy.
903    */
904   public static Iterator<Class<?>> getParentClasses(final Class<?> c, boolean parentFirst, boolean includeInterfaces) {
905      List<Class<?>> l = getParentClasses(new ArrayList<Class<?>>(), c, parentFirst, includeInterfaces);
906      return l.iterator();
907   }
908
909   private static List<Class<?>> getParentClasses(List<Class<?>> l, Class<?> c, boolean parentFirst, boolean includeInterfaces) {
910      if (parentFirst) {
911         if (includeInterfaces)
912            for (Class<?> i : c.getInterfaces())
913               l.add(i);
914         if (c.getSuperclass() != Object.class && c.getSuperclass() != null)
915            getParentClasses(l, c.getSuperclass(), parentFirst, includeInterfaces);
916         l.add(c);
917      } else {
918         l.add(c);
919         if (c.getSuperclass() != Object.class && c.getSuperclass() != null)
920            getParentClasses(l, c.getSuperclass(), parentFirst, includeInterfaces);
921         if (includeInterfaces)
922            for (Class<?> i : c.getInterfaces())
923               l.add(i);
924      }
925      return l;
926   }
