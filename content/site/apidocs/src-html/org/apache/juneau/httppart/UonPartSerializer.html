013package org.apache.juneau.httppart;
014
015import java.io.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.uon.*;
019
020/**
021 * Serializes POJOs to values suitable for transmission as HTTP headers, query/form-data parameters, and path variables.
022 * 
023 * <p>
024 * This serializer uses UON notation for all parts by default.  This allows for arbitrary POJOs to be losslessly
025 * serialized as any of the specified HTTP types.
026 * 
027 * <h5 class='section'>Strings</h5>
028 * <p>
029 * One important note is how strings are serialized in UON notation.  
030 * In general, strings are serialized as-is except for the following cases:
031 * 
032 * <ul>
033 *    <li>String is empty.
034 *    <li>String starts with <js>'@'</js> or <js>'('</js>.
035 *    <li>String contains any of the following characters:  <js>"),=\n\t\r\b\f"</js>
036 *    <li>String is <js>"true"</js> or <js>"false"</js>.
037 *    <li>String is numeric.
038 * </ul>
039 * 
040 * <p>
041 * We quote strings in these cases to prevent mis-identification of arrays, objects, booleans, and numbers, and
042 * to preserve whitespace.
043 * 
044 * <p>
045 * The {@link #UONPART_simpleStrings} setting and {@link SimpleUonPartSerializer} class is provided to override this
046 * default behavior on strings and to always just serialize them as plain text.
047 */
048public class UonPartSerializer extends UonSerializer implements HttpPartSerializer {
049
050   //-------------------------------------------------------------------------------------------------------------------
051   // Predefined instances
052   //-------------------------------------------------------------------------------------------------------------------
053
054   /** Reusable instance of {@link UonPartSerializer}, all default settings. */
055   public static final UonPartSerializer DEFAULT = new UonPartSerializer(PropertyStore.DEFAULT);
056
057
058   //-------------------------------------------------------------------------------------------------------------------
059   // Configurable properties
060   //-------------------------------------------------------------------------------------------------------------------
061
062   private static final String PREFIX = "UonPartSerializer.";
063
064   /**
065    * <b>Configuration property:</b>  Always serialize <code>Strings</code> as plain text.
066    *
067    * <ul>
068    *    <li><b>Name:</b> <js>"UrlEncodingSerializer.simpleStrings.b"</js>
069    *    <li><b>Data type:</b> <code>Boolean</code>
070    *    <li><b>Default:</b> <jk>false</jk> for {@link UonPartSerializer}, <jk>true</jk> for {@link SimpleUonPartSerializer}
071    *    <li><b>Session-overridable:</b> <jk>false</jk>
072    * </ul>
073    *
074    * <p>
075    * Strings will never be quoted or escaped using UON notation.
076    *
077    * <p>
078    * Note that this can cause errors during parsing if you're using the URL-encoding parser to parse the results since
079    * UON constructs won't be differentiable.
080    * However, this is not an issue if you're simply creating queries or form posts against 3rd-party interfaces.
081    */
082   public static final String UONPART_simpleStrings = PREFIX + "simpleStrings.b";
083
084   
085   //-------------------------------------------------------------------------------------------------------------------
086   // Instance
087   //-------------------------------------------------------------------------------------------------------------------
088
089   /**
090    * Constructor.
091    *
092    * @param ps
093    *    The property store containing all the settings for this object.
094    */
095   public UonPartSerializer(PropertyStore ps) {
096      super(
097         ps.builder()
098            .set(UON_encodeChars, false)
099            .build() 
100      );
101   }
102
103   @Override /* Context */
104   public UonPartSerializerBuilder builder() {
105      return new UonPartSerializerBuilder(getPropertyStore());
106   }
107
108   /**
109    * Instantiates a new clean-slate {@link UonPartSerializerBuilder} object.
110    * 
111    * <p>
112    * Note that this method creates a builder initialized to all default settings, whereas {@link #builder()} copies 
113    * the settings of the object called on.
114    * 
115    * @return A new {@link UonPartSerializerBuilder} object.
116    */
117   public static UonPartSerializerBuilder create() {
118      return new UonPartSerializerBuilder();
119   }
120
121   //--------------------------------------------------------------------------------
122   // Entry point methods
123   //--------------------------------------------------------------------------------
124
125   @Override /* PartSerializer */
126   public String serialize(HttpPartType type, Object value) {
127      try {
128         // Shortcut for simple types.
129         ClassMeta<?> cm = getClassMetaForObject(value);
130         if (cm != null) {
131            if (cm.isNumber() || cm.isBoolean())
132               return value.toString();
133            if (cm.isCharSequence()) {
134               String s = value.toString();
135               boolean ptt = getParamFormat() == ParamFormat.PLAINTEXT;
136               if (ptt || s.isEmpty() || ! UonUtils.needsQuotes(s))
137                  return s;
138            }
139         }
140
141         StringWriter w = new StringWriter();
142         UonSerializerSession s = new UonSerializerSession(this, false, createDefaultSessionArgs());
143         s.serialize(value, w);
144         return w.toString();
145      } catch (Exception e) {
146         throw new RuntimeException(e);
147      }
148   }
149}
