026/**
027 * Parent class for all Juneau parsers.
028 *
029 * <h6 class='topic'>Valid data conversions</h6>
030 *
031 * Parsers can parse any parsable POJO types, as specified in the <a class="doclink"
032 * href="../../../../overview-summary.html#juneau-marshall.PojoCategories">POJO Categories</a>.
033 *
034 * <p>
035 * Some examples of conversions are shown below...
036 * </p>
037 * <table class='styled'>
038 *    <tr>
039 *       <th>Data type</th>
040 *       <th>Class type</th>
041 *       <th>JSON example</th>
042 *       <th>XML example</th>
043 *       <th>Class examples</th>
044 *    </tr>
045 *    <tr>
046 *       <td>object</td>
047 *       <td>Maps, Java beans</td>
048 *       <td class='code'>{name:<js>'John Smith'</js>,age:21}</td>
049 *       <td class='code'><xt>&lt;object&gt;
050 *    &lt;name</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt>John Smith<xt>&lt;/name&gt;
051 *    &lt;age</xt> <xa>type</xa>=<xs>'number'</xs><xt>&gt;</xt>21<xt>&lt;/age&gt;
052 * &lt;/object&gt;</xt></td>
053 *       <td class='code'>HashMap, TreeMap&lt;String,Integer&gt;</td>
054 *    </tr>
055 *    <tr>
056 *       <td>array</td>
057 *       <td>Collections, Java arrays</td>
058 *       <td class='code'>[1,2,3]</td>
059 *       <td class='code'><xt>&lt;array&gt;
060 *    &lt;number&gt;</xt>1<xt>&lt;/number&gt;
061 *    &lt;number&gt;</xt>2<xt>&lt;/number&gt;
062 *    &lt;number&gt;</xt>3<xt>&lt;/number&gt;
063 * &lt;/array&gt;</xt></td>
064 *       <td class='code'>List&lt;Integer&gt;, <jk>int</jk>[], Float[], Set&lt;Person&gt;</td>
065 *    </tr>
066 *    <tr>
067 *       <td>number</td>
068 *       <td>Numbers</td>
069 *       <td class='code'>123</td>
070 *       <td class='code'><xt>&lt;number&gt;</xt>123<xt>&lt;/number&gt;</xt></td>
071 *       <td class='code'>Integer, Long, Float, <jk>int</jk></td>
072 *    </tr>
073 *    <tr>
074 *       <td>boolean</td>
075 *       <td>Booleans</td>
076 *       <td class='code'><jk>true</jk></td>
077 *       <td class='code'><xt>&lt;boolean&gt;</xt>true<xt>&lt;/boolean&gt;</xt></td>
078 *       <td class='code'>Boolean</td>
079 *    </tr>
080 *    <tr>
081 *       <td>string</td>
082 *       <td>CharSequences</td>
083 *       <td class='code'><js>'foobar'</js></td>
084 *       <td class='code'><xt>&lt;string&gt;</xt>foobar<xt>&lt;/string&gt;</xt></td>
085 *       <td class='code'>String, StringBuilder</td>
086 *    </tr>
087 * </table>
088 *
089 * <p>
090 * In addition, any class types with {@link PojoSwap PojoSwaps} associated with them on the registered
091 * bean context can also be passed in.
092 *
093 * <p>
094 * For example, if the {@link CalendarSwap} transform is used to generalize {@code Calendar} objects to {@code String}
095 * objects.
096 * When registered with this parser, you can construct {@code Calendar} objects from {@code Strings} using the
097 * following syntax...
098 * <p class='bcode'>
099 *    Calendar c = parser.parse(<js>"'Sun Mar 03 04:05:06 EST 2001'"</js>, GregorianCalendar.<jk>class</jk>);
100 *
101 * <p>
102 * If <code>Object.<jk>class</jk></code> is specified as the target type, then the parser automatically determines the
103 * data types and generates the following object types...
104 * <table class='styled'>
105 *    <tr><th>JSON type</th><th>Class type</th></tr>
106 *    <tr><td>object</td><td>{@link ObjectMap}</td></tr>
107 *    <tr><td>array</td><td>{@link ObjectList}</td></tr>
108 *    <tr><td>number</td><td>{@link Number}<br>(depending on length and format, could be {@link Integer},
109 *       {@link Double}, {@link Float}, etc...)</td></tr>
110 *    <tr><td>boolean</td><td>{@link Boolean}</td></tr>
111 *    <tr><td>string</td><td>{@link String}</td></tr>
112 * </table>
113 *
114 * <a id='SupportedTypes'></a>
115 * <h6 class='topic'>Supported types</h6>
116 *
117 * Several of the methods below take {@link Type} parameters to identify the type of object to create.
118 * Any of the following types can be passed in to these methods...
119 * <ul>
120 *    <li>{@link ClassMeta}
121 *    <li>{@link Class}
122 *    <li>{@link ParameterizedType}
123 *    <li>{@link GenericArrayType}
124 * </ul>
125 *
126 * <p>
127 * However, {@code ParameterizedTypes} and {@code GenericArrayTypes} should not contain
128 * {@link WildcardType WildcardTypes} or {@link TypeVariable TypeVariables}.
129 *
130 * <p>
131 * Passing in <jk>null</jk> or <code>Object.<jk>class</jk></code> typically signifies that it's up to the parser
132 * to determine what object type is being parsed parsed based on the rules above.
133 */
134public abstract class Parser extends BeanContext {

158   /**
159    * <b>Configuration property:</b>  Strict mode.
160    *
161    * <ul>
162    *    <li><b>Name:</b> <js>"Parser.strict.b"</js>
163    *    <li><b>Data type:</b> <code>Boolean</code>
164    *    <li><b>Default:</b> <jk>false</jk>
165    *    <li><b>Session-overridable:</b> <jk>true</jk>
166    * </ul>
167    * <p>
168    * If <jk>true</jk>, strict mode for the parser is enabled.
169    *
170    * <p>
171    * Strict mode can mean different things for different parsers.
172    *
173    * <table class='styled'>
174    *    <tr><th>Parser class</th><th>Strict behavior</th></tr>
175    *    <tr>
176    *       <td>All reader-based parsers</td>
177    *       <td>
178    *          When enabled, throws {@link ParseException ParseExceptions} on malformed charset input.
179    *          Otherwise, malformed input is ignored.
180    *       </td>
181    *    </tr>
182    *    <tr>
183    *       <td>{@link JsonParser}</td>
184    *       <td>
185    *          When enabled, throws exceptions on the following invalid JSON syntax:
186    *          <ul>
187    *             <li>Unquoted attributes.
188    *             <li>Missing attribute values.
189    *             <li>Concatenated strings.
190    *             <li>Javascript comments.
191    *             <li>Numbers and booleans when Strings are expected.
192    *             <li>Numbers valid in Java but not JSON (e.g. octal notation, etc...)
193    *          </ul>
194    *       </td>
195    *    </tr>
196    * </table>
197    */
198   public static final String PARSER_strict = PREFIX + "strict.b";

272   // Hidden constructor to force subclass from InputStreamParser or ReaderParser.
273   Parser(PropertyStore ps, String...consumes) {
274      super(ps);
275
276      trimStrings = getProperty(PARSER_trimStrings, boolean.class, false);
277      strict = getProperty(PARSER_strict, boolean.class, false);
278      inputStreamCharset = getProperty(PARSER_inputStreamCharset, String.class, "UTF-8");
279      fileCharset = getProperty(PARSER_fileCharset, String.class, "default");
280      listener = getClassProperty(PARSER_listener, ParserListener.class, null);
281      this.consumes = new MediaType[consumes.length];
282      for (int i = 0; i < consumes.length; i++) {
283         this.consumes[i] = MediaType.forString(consumes[i]);
284      }
285   }

340   /**
341    * Parses input into the specified object type.
342    *
343    * <p>
344    * The type can be a simple type (e.g. beans, strings, numbers) or parameterized type (collections/maps).
345    *
346    * <h5 class='section'>Examples:</h5>
347    * <p class='bcode'>
348    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
349    *
350    *    <jc>// Parse into a linked-list of strings.</jc>
351    *    List l = p.parse(json, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
352    *
353    *    <jc>// Parse into a linked-list of beans.</jc>
354    *    List l = p.parse(json, LinkedList.<jk>class</jk>, MyBean.<jk>class</jk>);
355    *
356    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
357    *    List l = p.parse(json, LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
358    *
359    *    <jc>// Parse into a map of string keys/values.</jc>
360    *    Map m = p.parse(json, TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
361    *
362    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
363    *    Map m = p.parse(json, TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
364    * </p>
365    *
366    * <p>
367    * <code>Collection</code> classes are assumed to be followed by zero or one objects indicating the element type.
368    *
369    * <p>
370    * <code>Map</code> classes are assumed to be followed by zero or two meta objects indicating the key and value types.
371    *
372    * <p>
373    * The array can be arbitrarily long to indicate arbitrarily complex data structures.
374    *
375    * <h5 class='section'>Notes:</h5>
376    * <ul>
377    *    <li>Use the {@link #parse(Object, Class)} method instead if you don't need a parameterized map/collection.
378    * </ul>
379    *
380    * @param <T> The class type of the object to create.
381    * @param input
382    *    The input.
383    *    <br>Character-based parsers can handle the following input class types:
384    *    <ul>
385    *       <li><jk>null</jk>
386    *       <li>{@link Reader}
387    *       <li>{@link CharSequence}
388    *       <li>{@link InputStream} containing UTF-8 encoded text (or charset defined by
389    *          {@link #PARSER_inputStreamCharset} property value).
390    *       <li><code><jk>byte</jk>[]</code> containing UTF-8 encoded text (or charset defined by
391    *          {@link #PARSER_inputStreamCharset} property value).
392    *       <li>{@link File} containing system encoded text (or charset defined by
393    *          {@link #PARSER_fileCharset} property value).
394    *    </ul>
395    *    <br>Stream-based parsers can handle the following input class types:
396    *    <ul>
397    *       <li><jk>null</jk>
398    *       <li>{@link InputStream}
399    *       <li><code><jk>byte</jk>[]</code>
400    *       <li>{@link File}
401    *    </ul>
402    * @param type
403    *    The object type to create.
404    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType},
405    *    {@link GenericArrayType}
406    * @param args
407    *    The type arguments of the class if it's a collection or map.
408    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType},
409    *    {@link GenericArrayType}
410    *    <br>Ignored if the main type is not a map or collection.
411    * @return The parsed object.
412    * @throws ParseException
413    *    If the input contains a syntax error or is malformed, or is not valid for the specified type.
414    * @see BeanSession#getClassMeta(Type,Type...) for argument syntax for maps and collections.
415    */
416   public final <T> T parse(Object input, Type type, Type...args) throws ParseException {
417      return createSession().parse(input, type, args);
418   }

420   /**
421    * Same as {@link #parse(Object, Type, Type...)} except optimized for a non-parameterized class.
422    *
423    * <p>
424    * This is the preferred parse method for simple types since you don't need to cast the results.
425    *
426    * <h5 class='section'>Examples:</h5>
427    * <p class='bcode'>
428    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
429    *
430    *    <jc>// Parse into a string.</jc>
431    *    String s = p.parse(json, String.<jk>class</jk>);
432    *
433    *    <jc>// Parse into a bean.</jc>
434    *    MyBean b = p.parse(json, MyBean.<jk>class</jk>);
435    *
436    *    <jc>// Parse into a bean array.</jc>
437    *    MyBean[] ba = p.parse(json, MyBean[].<jk>class</jk>);
438    *
439    *    <jc>// Parse into a linked-list of objects.</jc>
440    *    List l = p.parse(json, LinkedList.<jk>class</jk>);
441    *
442    *    <jc>// Parse into a map of object keys/values.</jc>
443    *    Map m = p.parse(json, TreeMap.<jk>class</jk>);
444    * </p>
445    *
446    * @param <T> The class type of the object being created.
447    * @param input
448    *    The input.
449    *    See {@link #parse(Object, Type, Type...)} for details.
450    * @param type The object type to create.
451    * @return The parsed object.
452    * @throws ParseException
453    *    If the input contains a syntax error or is malformed, or is not valid for the specified type.
454    */
455   public final <T> T parse(Object input, Class<T> type) throws ParseException {
456      return createSession().parse(input, type);
457   }

493   /**
494    * Parses the contents of the specified reader and loads the results into the specified map.
495    *
496    * <p>
497    * Reader must contain something that serializes to a map (such as text containing a JSON object).
498    *
499    * <p>
500    * Used in the following locations:
501    * <ul class='spaced-list'>
502    *    <li>
503    *       The various character-based constructors in {@link ObjectMap} (e.g.
504    *       {@link ObjectMap#ObjectMap(CharSequence,Parser)}).
505    * </ul>
506    *
507    * @param <K> The key class type.
508    * @param <V> The value class type.
509    * @param input The input.  See {@link #parse(Object, ClassMeta)} for supported input types.
510    * @param m The map being loaded.
511    * @param keyType The class type of the keys, or <jk>null</jk> to default to <code>String.<jk>class</jk></code>.
512    * @param valueType The class type of the values, or <jk>null</jk> to default to whatever is being parsed.
513    * @return The same map that was passed in to allow this method to be chained.
514    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
515    * @throws UnsupportedOperationException If not implemented.
516    */
517   public final <K,V> Map<K,V> parseIntoMap(Object input, Map<K,V> m, Type keyType, Type valueType) throws ParseException {
518      return createSession().parseIntoMap(input, m, keyType, valueType);
519   }

521   /**
522    * Parses the contents of the specified reader and loads the results into the specified collection.
523    *
524    * <p>
525    * Used in the following locations:
526    * <ul class='spaced-list'>
527    *    <li>
528    *       The various character-based constructors in {@link ObjectList} (e.g.
529    *       {@link ObjectList#ObjectList(CharSequence,Parser)}.
530    * </ul>
531    *
532    * @param <E> The element class type.
533    * @param input The input.  See {@link #parse(Object, ClassMeta)} for supported input types.
534    * @param c The collection being loaded.
535    * @param elementType The class type of the elements, or <jk>null</jk> to default to whatever is being parsed.
536    * @return The same collection that was passed in to allow this method to be chained.
537    * @throws ParseException
538    *    If the input contains a syntax error or is malformed, or is not valid for the specified type.
539    * @throws UnsupportedOperationException If not implemented.
540    */
541   public final <E> Collection<E> parseIntoCollection(Object input, Collection<E> c, Type elementType) throws ParseException {
542      return createSession().parseIntoCollection(input, c, elementType);
543   }

545   /**
546    * Parses the specified array input with each entry in the object defined by the {@code argTypes}
547    * argument.
548    *
549    * <p>
550    * Used for converting arrays (e.g. <js>"[arg1,arg2,...]"</js>) into an {@code Object[]} that can be passed
551    * to the {@code Method.invoke(target, args)} method.
552    *
553    * <p>
554    * Used in the following locations:
555    * <ul class='spaced-list'>
556    *    <li>
557    *       Used to parse argument strings in the {@link PojoIntrospector#invokeMethod(Method, Reader)} method.
558    * </ul>
559    *
560    * @param input The input.  Subclasses can support different input types.
561    * @param argTypes Specifies the type of objects to create for each entry in the array.
562    * @return An array of parsed objects.
563    * @throws ParseException
564    *    If the input contains a syntax error or is malformed, or is not valid for the specified type.
565    */
566   public final Object[] parseArgs(Object input, Type[] argTypes) throws ParseException {
567      if (argTypes == null || argTypes.length == 0)
568         return new Object[0];
569      return createSession().parseArgs(input, argTypes);
570   }
