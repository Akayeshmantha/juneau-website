029/**
030 * Parent class for all Juneau parsers.
031 *
032 * <h6 class='topic'>@Consumes annotation</h6>
033 *
034 * The media types that this parser can handle is specified through the {@link Consumes @Consumes} annotation.
035 *
036 * <p>
037 * However, the media types can also be specified programmatically by overriding the {@link #getMediaTypes()} method.
038 *
039 * <h6 class='topic'>Valid data conversions</h6>
040 *
041 * Parsers can parse any parsable POJO types, as specified in the <a class="doclink"
042 * href="../../../../overview-summary.html#Core.PojoCategories">POJO Categories</a>.
043 *
044 * <p>
045 * Some examples of conversions are shown below...
046 * </p>
047 * <table class='styled'>
048 *    <tr>
049 *       <th>Data type</th>
050 *       <th>Class type</th>
051 *       <th>JSON example</th>
052 *       <th>XML example</th>
053 *       <th>Class examples</th>
054 *    </tr>
055 *    <tr>
056 *       <td>object</td>
057 *       <td>Maps, Java beans</td>
058 *       <td class='code'>{name:<js>'John Smith'</js>,age:21}</td>
059 *       <td class='code'><xt>&lt;object&gt;
060 *    &lt;name</xt> <xa>type</xa>=<xs>'string'</xs><xt>&gt;</xt>John Smith<xt>&lt;/name&gt;
061 *    &lt;age</xt> <xa>type</xa>=<xs>'number'</xs><xt>&gt;</xt>21<xt>&lt;/age&gt;
062 * &lt;/object&gt;</xt></td>
063 *       <td class='code'>HashMap, TreeMap&lt;String,Integer&gt;</td>
064 *    </tr>
065 *    <tr>
066 *       <td>array</td>
067 *       <td>Collections, Java arrays</td>
068 *       <td class='code'>[1,2,3]</td>
069 *       <td class='code'><xt>&lt;array&gt;
070 *    &lt;number&gt;</xt>1<xt>&lt;/number&gt;
071 *    &lt;number&gt;</xt>2<xt>&lt;/number&gt;
072 *    &lt;number&gt;</xt>3<xt>&lt;/number&gt;
073 * &lt;/array&gt;</xt></td>
074 *       <td class='code'>List&lt;Integer&gt;, <jk>int</jk>[], Float[], Set&lt;Person&gt;</td>
075 *    </tr>
076 *    <tr>
077 *       <td>number</td>
078 *       <td>Numbers</td>
079 *       <td class='code'>123</td>
080 *       <td class='code'><xt>&lt;number&gt;</xt>123<xt>&lt;/number&gt;</xt></td>
081 *       <td class='code'>Integer, Long, Float, <jk>int</jk></td>
082 *    </tr>
083 *    <tr>
084 *       <td>boolean</td>
085 *       <td>Booleans</td>
086 *       <td class='code'><jk>true</jk></td>
087 *       <td class='code'><xt>&lt;boolean&gt;</xt>true<xt>&lt;/boolean&gt;</xt></td>
088 *       <td class='code'>Boolean</td>
089 *    </tr>
090 *    <tr>
091 *       <td>string</td>
092 *       <td>CharSequences</td>
093 *       <td class='code'><js>'foobar'</js></td>
094 *       <td class='code'><xt>&lt;string&gt;</xt>foobar<xt>&lt;/string&gt;</xt></td>
095 *       <td class='code'>String, StringBuilder</td>
096 *    </tr>
097 * </table>
098 *
099 * <p>
100 * In addition, any class types with {@link PojoSwap PojoSwaps} associated with them on the registered
101 * {@link #getBeanContext() beanContext} can also be passed in.
102 *
103 * <p>
104 * For example, if the {@link CalendarSwap} transform is used to generalize {@code Calendar} objects to {@code String}
105 * objects.
106 * When registered with this parser, you can construct {@code Calendar} objects from {@code Strings} using the
107 * following syntax...
108 * <p class='bcode'>
109 *    Calendar c = parser.parse(<js>"'Sun Mar 03 04:05:06 EST 2001'"</js>, GregorianCalendar.<jk>class</jk>);
110 *
111 * <p>
112 * If <code>Object.<jk>class</jk></code> is specified as the target type, then the parser automatically determines the
113 * data types and generates the following object types...
114 * <table class='styled'>
115 *    <tr><th>JSON type</th><th>Class type</th></tr>
116 *    <tr><td>object</td><td>{@link ObjectMap}</td></tr>
117 *    <tr><td>array</td><td>{@link ObjectList}</td></tr>
118 *    <tr><td>number</td><td>{@link Number}<br>(depending on length and format, could be {@link Integer},
119 *       {@link Double}, {@link Float}, etc...)</td></tr>
120 *    <tr><td>boolean</td><td>{@link Boolean}</td></tr>
121 *    <tr><td>string</td><td>{@link String}</td></tr>
122 * </table>
123 *
124 * <a id='SupportedTypes'></a>
125 * <h6 class='topic'>Supported types</h6>
126 *
127 * Several of the methods below take {@link Type} parameters to identify the type of object to create.
128 * Any of the following types can be passed in to these methods...
129 * <ul>
130 *    <li>{@link ClassMeta}
131 *    <li>{@link Class}
132 *    <li>{@link ParameterizedType}
133 *    <li>{@link GenericArrayType}
134 * </ul>
135 *
136 * <p>
137 * However, {@code ParameterizedTypes} and {@code GenericArrayTypes} should not contain
138 * {@link WildcardType WildcardTypes} or {@link TypeVariable TypeVariables}.
139 *
140 * <p>
141 * Passing in <jk>null</jk> or <code>Object.<jk>class</jk></code> typically signifies that it's up to the parser
142 * to determine what object type is being parsed parsed based on the rules above.
143 */
144public abstract class Parser extends CoreObject {
145
146   /** General parser properties currently set on this parser. */
147   private final MediaType[] mediaTypes;
148
149   // Hidden constructor to force subclass from InputStreamParser or ReaderParser.
150   Parser(PropertyStore propertyStore) {
151      super(propertyStore);
152
153      Consumes c = getAnnotation(Consumes.class, getClass());
154      if (c == null)
155         throw new FormattedRuntimeException("Class ''{0}'' is missing the @Consumes annotation", c);
156
157      String[] mt = split(c.value());
158      this.mediaTypes = new MediaType[mt.length];
159      for (int i = 0; i < mt.length; i++) {
160         mediaTypes[i] = MediaType.forString(mt[i]);
161      }
162   }
163
164   @Override /* CoreObject */
165   public ParserBuilder builder() {
166      return new ParserBuilder(propertyStore);
167   }

199   /**
200    * Parses input into the specified object type.
201    *
202    * <p>
203    * The type can be a simple type (e.g. beans, strings, numbers) or parameterized type (collections/maps).
204    *
205    * <h5 class='section'>Examples:</h5>
206    * <p class='bcode'>
207    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
208    *
209    *    <jc>// Parse into a linked-list of strings.</jc>
210    *    List l = p.parse(json, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
211    *
212    *    <jc>// Parse into a linked-list of beans.</jc>
213    *    List l = p.parse(json, LinkedList.<jk>class</jk>, MyBean.<jk>class</jk>);
214    *
215    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
216    *    List l = p.parse(json, LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
217    *
218    *    <jc>// Parse into a map of string keys/values.</jc>
219    *    Map m = p.parse(json, TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
220    *
221    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
222    *    Map m = p.parse(json, TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
223    * </p>
224    *
225    * <p>
226    * <code>Collection</code> classes are assumed to be followed by zero or one objects indicating the element type.
227    *
228    * <p>
229    * <code>Map</code> classes are assumed to be followed by zero or two meta objects indicating the key and value types.
230    *
231    * <p>
232    * The array can be arbitrarily long to indicate arbitrarily complex data structures.
233    *
234    * <h5 class='section'>Notes:</h5>
235    * <ul>
236    *    <li>Use the {@link #parse(Object, Class)} method instead if you don't need a parameterized map/collection.
237    * </ul>
238    *
239    * @param <T> The class type of the object to create.
240    * @param input
241    *    The input.
242    *    <br>Character-based parsers can handle the following input class types:
243    *    <ul>
244    *       <li><jk>null</jk>
245    *       <li>{@link Reader}
246    *       <li>{@link CharSequence}
247    *       <li>{@link InputStream} containing UTF-8 encoded text (or charset defined by
248    *          {@link ParserContext#PARSER_inputStreamCharset} property value).
249    *       <li><code><jk>byte</jk>[]</code> containing UTF-8 encoded text (or charset defined by
250    *          {@link ParserContext#PARSER_inputStreamCharset} property value).
251    *       <li>{@link File} containing system encoded text (or charset defined by
252    *          {@link ParserContext#PARSER_fileCharset} property value).
253    *    </ul>
254    *    <br>Stream-based parsers can handle the following input class types:
255    *    <ul>
256    *       <li><jk>null</jk>
257    *       <li>{@link InputStream}
258    *       <li><code><jk>byte</jk>[]</code>
259    *       <li>{@link File}
260    *    </ul>
261    * @param type
262    *    The object type to create.
263    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType},
264    *    {@link GenericArrayType}
265    * @param args
266    *    The type arguments of the class if it's a collection or map.
267    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType},
268    *    {@link GenericArrayType}
269    *    <br>Ignored if the main type is not a map or collection.
270    * @return The parsed object.
271    * @throws ParseException
272    *    If the input contains a syntax error or is malformed, or is not valid for the specified type.
273    * @see BeanSession#getClassMeta(Type,Type...) for argument syntax for maps and collections.
274    */
275   public final <T> T parse(Object input, Type type, Type...args) throws ParseException {
276      ParserSession session = createSession();
277      try {
278         return session.parse(input, type, args);
279      } finally {
280         session.close();
281      }
282   }

284   /**
285    * Same as {@link #parse(Object, Type, Type...)} except optimized for a non-parameterized class.
286    *
287    * <p>
288    * This is the preferred parse method for simple types since you don't need to cast the results.
289    *
290    * <h5 class='section'>Examples:</h5>
291    * <p class='bcode'>
292    *    ReaderParser p = JsonParser.<jsf>DEFAULT</jsf>;
293    *
294    *    <jc>// Parse into a string.</jc>
295    *    String s = p.parse(json, String.<jk>class</jk>);
296    *
297    *    <jc>// Parse into a bean.</jc>
298    *    MyBean b = p.parse(json, MyBean.<jk>class</jk>);
299    *
300    *    <jc>// Parse into a bean array.</jc>
301    *    MyBean[] ba = p.parse(json, MyBean[].<jk>class</jk>);
302    *
303    *    <jc>// Parse into a linked-list of objects.</jc>
304    *    List l = p.parse(json, LinkedList.<jk>class</jk>);
305    *
306    *    <jc>// Parse into a map of object keys/values.</jc>
307    *    Map m = p.parse(json, TreeMap.<jk>class</jk>);
308    * </p>
309    *
310    * @param <T> The class type of the object being created.
311    * @param input
312    *    The input.
313    *    See {@link #parse(Object, Type, Type...)} for details.
314    * @param type The object type to create.
315    * @return The parsed object.
316    * @throws ParseException
317    *    If the input contains a syntax error or is malformed, or is not valid for the specified type.
318    */
319   public final <T> T parse(Object input, Class<T> type) throws ParseException {
320      ParserSession session = createSession();
321      try {
322         return session.parse(input, type);
323      } finally {
324         session.close();
325      }
326   }

370   /**
371    * Parses the contents of the specified reader and loads the results into the specified map.
372    *
373    * <p>
374    * Reader must contain something that serializes to a map (such as text containing a JSON object).
375    *
376    * <p>
377    * Used in the following locations:
378    * <ul class='spaced-list'>
379    *    <li>
380    *       The various character-based constructors in {@link ObjectMap} (e.g.
381    *       {@link ObjectMap#ObjectMap(CharSequence,Parser)}).
382    * </ul>
383    *
384    * @param <K> The key class type.
385    * @param <V> The value class type.
386    * @param input The input.  See {@link #parse(Object, ClassMeta)} for supported input types.
387    * @param m The map being loaded.
388    * @param keyType The class type of the keys, or <jk>null</jk> to default to <code>String.<jk>class</jk></code>.
389    * @param valueType The class type of the values, or <jk>null</jk> to default to whatever is being parsed.
390    * @return The same map that was passed in to allow this method to be chained.
391    * @throws ParseException If the input contains a syntax error or is malformed, or is not valid for the specified type.
392    * @throws UnsupportedOperationException If not implemented.
393    */
394   public final <K,V> Map<K,V> parseIntoMap(Object input, Map<K,V> m, Type keyType, Type valueType) throws ParseException {
395      ParserSession session = createSession();
396      try {
397         return session.parseIntoMap(input, m, keyType, valueType);
398      } finally {
399         session.close();
400      }
401   }

403   /**
404    * Parses the contents of the specified reader and loads the results into the specified collection.
405    *
406    * <p>
407    * Used in the following locations:
408    * <ul class='spaced-list'>
409    *    <li>
410    *       The various character-based constructors in {@link ObjectList} (e.g.
411    *       {@link ObjectList#ObjectList(CharSequence,Parser)}.
412    * </ul>
413    *
414    * @param <E> The element class type.
415    * @param input The input.  See {@link #parse(Object, ClassMeta)} for supported input types.
416    * @param c The collection being loaded.
417    * @param elementType The class type of the elements, or <jk>null</jk> to default to whatever is being parsed.
418    * @return The same collection that was passed in to allow this method to be chained.
419    * @throws ParseException
420    *    If the input contains a syntax error or is malformed, or is not valid for the specified type.
421    * @throws UnsupportedOperationException If not implemented.
422    */
423   public final <E> Collection<E> parseIntoCollection(Object input, Collection<E> c, Type elementType) throws ParseException {
424      ParserSession session = createSession();
425      try {
426         return session.parseIntoCollection(input, c, elementType);
427      } finally {
428         session.close();
429      }
430   }

432   /**
433    * Parses the specified array input with each entry in the object defined by the {@code argTypes}
434    * argument.
435    *
436    * <p>
437    * Used for converting arrays (e.g. <js>"[arg1,arg2,...]"</js>) into an {@code Object[]} that can be passed
438    * to the {@code Method.invoke(target, args)} method.
439    *
440    * <p>
441    * Used in the following locations:
442    * <ul class='spaced-list'>
443    *    <li>
444    *       Used to parse argument strings in the {@link PojoIntrospector#invokeMethod(Method, Reader)} method.
445    * </ul>
446    *
447    * @param input The input.  Subclasses can support different input types.
448    * @param argTypes Specifies the type of objects to create for each entry in the array.
449    * @return An array of parsed objects.
450    * @throws ParseException
451    *    If the input contains a syntax error or is malformed, or is not valid for the specified type.
452    */
453   public final Object[] parseArgs(Object input, Type[] argTypes) throws ParseException {
454      if (argTypes == null || argTypes.length == 0)
455         return new Object[0];
456      ParserSession session = createSession();
457      try {
458         return session.parseArgs(input, argTypes);
459      } finally {
460         session.close();
461      }
462   }
