013package org.apache.juneau.parser;
014
015import org.apache.juneau.*;
016import org.apache.juneau.json.*;
017
018/**
019 * Configurable properties common to all parsers.
020 *
021 *
022 * <h6 class='topic' id='ConfigProperties'>Configurable properties common to all parsers</h6>
023 * <table class='styled' style='border-collapse: collapse;'>
024 *    <tr><th>Setting name</th><th>Description</th><th>Data type</th><th>Default value</th></tr>
025 *    <tr>
026 *       <td>{@link #PARSER_debug}</td>
027 *       <td>Debug mode.</td>
028 *       <td><code>Boolean</code></td>
029 *       <td><jk>false</jk></td>
030 *    </tr>
031 *    <tr>
032 *       <td>{@link #PARSER_trimStrings}</td>
033 *       <td>Trim parsed strings.</td>
034 *       <td><code>Boolean</code></td>
035 *       <td><jk>false</jk></td>
036 *    </tr>
037 * </table>
038 *
039 *
040 * @author James Bognar (james.bognar@salesforce.com)
041 */
042public class ParserContext extends Context {
043
044   /**
045    * <b>Configuration property:</b>  Debug mode.
046    * <p>
047    * <ul>
048    *    <li><b>Name:</b> <js>"Parser.debug"</js>
049    *    <li><b>Data type:</b> <code>Boolean</code>
050    *    <li><b>Default:</b> <jk>false</jk>
051    * </ul>
052    * <p>
053    * Enables the following additional information during parsing:
054    * <ul class='spaced-list'>
055    *    <li>When bean setters throws exceptions, the exception includes the object stack information
056    *       in order to determine how that method was invoked.
057    * </ul>
058    */
059   public static final String PARSER_debug = "Parser.debug";
060
061   /**
062    * <b>Configuration property:</b>  Trim parsed strings.
063    * <p>
064    * <ul>
065    *    <li><b>Name:</b> <js>"Parser.trimStrings"</js>
066    *    <li><b>Data type:</b> <code>Boolean</code>
067    *    <li><b>Default:</b> <jk>false</jk>
068    * </ul>
069    * <p>
070    * If <jk>true</jk>, string values will be trimmed of whitespace using {@link String#trim()} before being added to the POJO.
071    */
072   public static final String PARSER_trimStrings = "Parser.trimStrings";
073
074   /**
075    * <b>Configuration property:</b>  Strict mode.
076    * <p>
077    * <ul>
078    *    <li><b>Name:</b> <js>"Parser.strict"</js>
079    *    <li><b>Data type:</b> <code>Boolean</code>
080    *    <li><b>Default:</b> <jk>false</jk>
081    * </ul>
082    * <p>
083    * If <jk>true</jk>, strict mode for the parser is enabled.
084    * <p>
085    * Strict mode can mean different things for different parsers.
086    * However, all reader-based parsers will
087    * <p>
088    * <table class='styled'>
089    *    <tr><th>Parser class</th><th>Strict behavior</td></tr>
090    *    <tr>
091    *       <td>All reader-based parsers</td>
092    *       <td>
093    *          When enabled, throws {@link ParseException ParseExceptions} on malformed charset input.
094    *          Otherwise, malformed input is ignored.
095    *       </td>
096    *    </tr>
097    *    <tr>
098    *       <td>{@link JsonParser}</td>
099    *       <td>
100    *          When enabled, throws exceptions on the following invalid JSON syntax:
101    *          <ul>
102    *             <li>Unquoted attributes.
103    *             <li>Missing attribute values.
104    *             <li>Concatenated strings.
105    *             <li>Javascript comments.
106    *             <li>Numbers and booleans when Strings are expected.
107    *             <li>Numbers valid in Java but not JSON (e.g. octal notation, etc...)
108    *          </ul>
109    *       </td>
110    *    </tr>
111    * </table>
112    */
113   public static final String PARSER_strict = "Parser.strict";
114
115   /**
116    * <b>Configuration property:</b>  Input stream charset.
117    * <p>
118    * <ul>
119    *    <li><b>Name:</b> <js>"Parser.inputStreamCharset"</js>
120    *    <li><b>Data type:</b> <code>String</code>
121    *    <li><b>Default:</b> <js>"UTF-8"</js>
122    * </ul>
123    * <p>
124    * The character set to use for converting <code>InputStreams</code> and byte arrays to readers.
125    * <p>
126    * Used when passing in input streams and byte arrays to {@link Parser#parse(Object, Class)}.
127    */
128   public static final String PARSER_inputStreamCharset = "Parser.inputStreamCharset";
129
130   /**
131    * <b>Configuration property:</b>  File charset.
132    * <p>
133    * <ul>
134    *    <li><b>Name:</b> <js>"Parser.fileCharset"</js>
135    *    <li><b>Data type:</b> <code>String</code>
136    *    <li><b>Default:</b> <js>"default"</js>
137    * </ul>
138    * <p>
139    * The character set to use for reading <code>Files</code> from the file system.
140    * <p>
141    * Used when passing in files to {@link Parser#parse(Object, Class)}.
142    * <p>
143    * <js>"default"</js> can be used to indicate the JVM default file system charset.
144    */
145   public static final String PARSER_fileCharset = "Parser.fileCharset";
146
147
148   final boolean debug, trimStrings, strict;
149   final String inputStreamCharset, fileCharset;
150
151   /**
152    * Constructor.
153    *
154    * @param cf The factory that created this context.
155    */
156   public ParserContext(ContextFactory cf) {
157      super(cf);
158      this.debug = cf.getProperty(PARSER_debug, boolean.class, false);
159      this.trimStrings = cf.getProperty(PARSER_trimStrings, boolean.class, false);
160      this.strict = cf.getProperty(PARSER_strict, boolean.class, false);
161      this.inputStreamCharset = cf.getProperty(PARSER_inputStreamCharset, String.class, "UTF-8");
162      this.fileCharset = cf.getProperty(PARSER_fileCharset, String.class, "default");
163   }
164}
