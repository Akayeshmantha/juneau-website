028/**
029 * Session object that lives for the duration of a single use of {@link Parser}.
030 * <p>
031 * This class is NOT thread safe.  It is meant to be discarded after one-time use.

042   private InputStream inputStream;
043   private Reader reader, noCloseReader;

045   private ClassMeta<?> currentClass;
046   private final ParserListener listener;

048   /**
049    * Create a new session using properties specified in the context.
050    *
051    * @param ctx The context creating this session object.
052    * The context contains all the configuration settings for this object.
053    * @param input The input.
054    * <br>For character-based parsers, this can be any of the following types:
055    * <ul>
056    *    <li><jk>null</jk>
057    *    <li>{@link Reader}
058    *    <li>{@link CharSequence}
059    *    <li>{@link InputStream} containing UTF-8 encoded text (or whatever the encoding specified by
060    *       {@link ParserContext#PARSER_inputStreamCharset}).
061    *    <li><code><jk>byte</jk>[]</code> containing UTF-8 encoded text (or whatever the encoding specified by
062    *       {@link ParserContext#PARSER_inputStreamCharset}).
063    *    <li>{@link File} containing system encoded text (or whatever the encoding specified by
064    *       {@link ParserContext#PARSER_fileCharset}).
065    * </ul>
066    * <br>For byte-based parsers, this can be any of the following types:
067    * <ul>
068    *    <li><jk>null</jk>
069    *    <li>{@link InputStream}
070    *    <li><code><jk>byte</jk>[]</code>
071    *    <li>{@link File}
072    * </ul>
073    *
074    * @param op The override properties.
075    * These override any context properties defined in the context.
076    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
077    * @param outer The outer object for instantiating top-level non-static inner classes.
078    * @param locale The session locale.
079    * If <jk>null</jk>, then the locale defined on the context is used.
080    * @param timeZone The session timezone.
081    * If <jk>null</jk>, then the timezone defined on the context is used.
082    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
083    */
084   public ParserSession(ParserContext ctx, ObjectMap op, Object input, Method javaMethod, Object outer, Locale locale,
085         TimeZone timeZone, MediaType mediaType) {
086      super(ctx, op, locale, timeZone, mediaType);
087      Class<?> listenerClass;
088      if (op == null || op.isEmpty()) {
089         trimStrings = ctx.trimStrings;
090         strict = ctx.strict;
091         inputStreamCharset = ctx.inputStreamCharset;
092         fileCharset = ctx.fileCharset;
093         listenerClass = ctx.listener;
094      } else {
095         trimStrings = op.getBoolean(PARSER_trimStrings, ctx.trimStrings);
096         strict = op.getBoolean(PARSER_strict, ctx.strict);
097         inputStreamCharset = op.getString(PARSER_inputStreamCharset, ctx.inputStreamCharset);
098         fileCharset = op.getString(PARSER_fileCharset, ctx.fileCharset);
099         listenerClass = op.get(Class.class, PARSER_listener, ctx.listener);
100      }
101      this.input = input;
102      this.javaMethod = javaMethod;
103      this.outer = outer;
104      this.listener = newInstance(ParserListener.class, listenerClass);
105   }

107   /**
108    * Wraps the specified input object inside an input stream.
109    * Subclasses can override this method to implement their own input streams.
110    *
111    * @return The input object wrapped in an input stream, or <jk>null</jk> if the object is null.
112    * @throws ParseException If object could not be converted to an input stream.
113    */
114   public InputStream getInputStream() throws ParseException {
115      try {
116         if (input == null)
117            return null;
118         if (input instanceof InputStream) {
119            if (isDebug()) {
120               byte[] b = readBytes((InputStream)input, 1024);
121               inputString = toHex(b);
122               return new ByteArrayInputStream(b);
123            }
124            return (InputStream)input;
125         }
126         if (input instanceof byte[]) {
127            if (isDebug())
128               inputString = toHex((byte[])input);
129            return new ByteArrayInputStream((byte[])input);
130         }
131         if (input instanceof String) {
132            inputString = (String)input;
133            return new ByteArrayInputStream(fromHex((String)input));
134         }
135         if (input instanceof File) {
136            if (isDebug()) {
137               byte[] b = readBytes((File)input);
138               inputString = toHex(b);
139               return new ByteArrayInputStream(b);
140            }
141            inputStream = new FileInputStream((File)input);
142            return inputStream;
143         }
144      } catch (IOException e) {
145         throw new ParseException(e);
146      }
147      throw new ParseException("Cannot convert object of type {0} to an InputStream.", input.getClass().getName());
148   }

151   /**
152    * Wraps the specified input object inside a reader.
153    * Subclasses can override this method to implement their own readers.
154    *
155    * @return The input object wrapped in a Reader, or <jk>null</jk> if the object is null.
156    * @throws Exception If object could not be converted to a reader.
157    */
158   public Reader getReader() throws Exception {
159      if (input == null)
160         return null;
161      if (input instanceof Reader) {
162         if (isDebug()) {
163            inputString = read((Reader)input);
164            return new StringReader(inputString);
165         }
166         return (Reader)input;
167      }
168      if (input instanceof CharSequence) {
169         inputString = input.toString();
170         if (reader == null)
171            reader = new ParserReader((CharSequence)input);
172         return reader;
173      }
174      if (input instanceof InputStream || input instanceof byte[]) {
175         InputStream is = (
176            input instanceof InputStream
177            ? (InputStream)input
178            : new ByteArrayInputStream((byte[])input)
179         );
180         if (noCloseReader == null) {
181            CharsetDecoder cd = (
182               "default".equalsIgnoreCase(inputStreamCharset)
183               ? Charset.defaultCharset()
184               : Charset.forName(inputStreamCharset)
185            ).newDecoder();
186            if (strict) {
187               cd.onMalformedInput(CodingErrorAction.REPORT);
188               cd.onUnmappableCharacter(CodingErrorAction.REPORT);
189            } else {
190               cd.onMalformedInput(CodingErrorAction.REPLACE);
191               cd.onUnmappableCharacter(CodingErrorAction.REPLACE);
192            }
193            noCloseReader = new InputStreamReader(is, cd);
194         }
195         if (isDebug()) {
196            inputString = read(noCloseReader);
197            return new StringReader(inputString);
198         }
199         return noCloseReader;
200      }
201      if (input instanceof File) {
202         if (reader == null) {
203            CharsetDecoder cd = (
204               "default".equalsIgnoreCase(fileCharset)
205               ? Charset.defaultCharset()
206               : Charset.forName(fileCharset)
207            ).newDecoder();
208            if (strict) {
209               cd.onMalformedInput(CodingErrorAction.REPORT);
210               cd.onUnmappableCharacter(CodingErrorAction.REPORT);
211            } else {
212               cd.onMalformedInput(CodingErrorAction.REPLACE);
213               cd.onUnmappableCharacter(CodingErrorAction.REPLACE);
214            }
215            reader = new InputStreamReader(new FileInputStream((File)input), cd);
216         }
217         if (isDebug()) {
218            inputString = read(reader);
219            return new StringReader(inputString);
220         }
221         return reader;
222      }
223      throw new ParseException("Cannot convert object of type {0} to a Reader.", input.getClass().getName());
224   }

226   /**
227    * Returns information used to determine at what location in the parse a failure occurred.
228    *
229    * @return A map, typically containing something like <code>{line:123,column:456,currentProperty:"foobar"}</code>
230    */
231   public Map<String,Object> getLastLocation() {
232      Map<String,Object> m = new LinkedHashMap<String,Object>();
233      if (currentClass != null)
234         m.put("currentClass", currentClass.toString(true));
235      if (currentProperty != null)
236         m.put("currentProperty", currentProperty);
237      return m;
238   }

333   /**
334    * Converts the specified <code>ObjectMap</code> into a bean identified by the <js>"_type"</js> property in the map.
335    *
336    * @param m The map to convert to a bean.
337    * @param pMeta The current bean property being parsed.
338    * @param eType The current expected type being parsed.
339    * @return The converted bean, or the same map if the <js>"_type"</js> entry wasn't found or didn't resolve to a
340    * bean.
341    */
342   public final Object cast(ObjectMap m, BeanPropertyMeta pMeta, ClassMeta<?> eType) {
343
344      String btpn = getBeanTypePropertyName(eType);
345
346      Object o = m.get(btpn);
347      if (o == null)
348         return m;
349      String typeName = o.toString();
350
351      ClassMeta<?> cm = getClassMeta(typeName, pMeta, eType);
352
353      if (cm != null) {
354         BeanMap<?> bm = m.getBeanSession().newBeanMap(cm.getInnerClass());
355
356         // Iterate through all the entries in the map and set the individual field values.
357         for (Map.Entry<String,Object> e : m.entrySet()) {
358            String k = e.getKey();
359            Object v = e.getValue();
360            if (! k.equals(btpn)) {
361               // Attempt to recursively cast child maps.
362               if (v instanceof ObjectMap)
363                  v = cast((ObjectMap)v, pMeta, eType);
364               bm.put(k, v);
365            }
366         }
367         return bm.getBean();
368      }
369
370      return m;
371   }

373   /**
374    * Give the specified dictionary name, resolve it to a class.
375    *
376    * @param typeName The dictionary name to resolve.
377    * @param pMeta The bean property we're currently parsing.
378    * @param eType The expected type we're currently parsing.
379    * @return The resolved class, or <jk>null</jk> if the type name could not be resolved.
380    */
381   public final ClassMeta<?> getClassMeta(String typeName, BeanPropertyMeta pMeta, ClassMeta<?> eType) {
382      BeanRegistry br = null;
383
384      // Resolve via @BeanProperty(beanDictionary={})
385      if (pMeta != null) {
386         br = pMeta.getBeanRegistry();
387         if (br != null && br.hasName(typeName))
388            return br.getClassMeta(typeName);
389      }
390
391      // Resolve via @Bean(beanDictionary={}) on the expected type where the
392      // expected type is an interface with subclasses.
393      if (eType != null) {
394         br = eType.getBeanRegistry();
395         if (br != null && br.hasName(typeName))
396            return br.getClassMeta(typeName);
397      }
398
399      // Last resort, resolve using the session registry.
400      return getBeanRegistry().getClassMeta(typeName);
401   }

403   /**
404    * Method that gets called when an unknown bean property name is encountered.
405    *
406    * @param propertyName The unknown bean property name.
407    * @param beanMap The bean that doesn't have the expected property.
408    * @param line The line number where the property was found.  <code>-1</code> if line numbers are not available.
409    * @param col The column number where the property was found.  <code>-1</code> if column numbers are not available.
410    * @throws ParseException Automatically thrown if {@link BeanContext#BEAN_ignoreUnknownBeanProperties} setting
411    * on this parser is <jk>false</jk>
412    * @param <T> The class type of the bean map that doesn't have the expected property.
413    */
414   public <T> void onUnknownProperty(String propertyName, BeanMap<T> beanMap, int line, int col) throws ParseException {
415      if (propertyName.equals(getBeanTypePropertyName(beanMap.getClassMeta())))
416         return;
417      if (! isIgnoreUnknownBeanProperties())
418         throw new ParseException(this,
419            "Unknown property ''{0}'' encountered while trying to parse into class ''{1}''", propertyName,
420            beanMap.getClassMeta());
421      if (listener != null)
422         listener.onUnknownBeanProperty(this, propertyName, beanMap.getClassMeta().getInnerClass(), beanMap.getBean(),
423            line, col);
424   }

437   /**
438    * Perform cleanup on this context object if necessary.
439    */
440   @Override
441   public boolean close() {
442      if (super.close()) {
443         try {
444            if (inputStream != null)
445               inputStream.close();
446            if (reader != null)
447               reader.close();
448         } catch (IOException e) {
449            throw new BeanRuntimeException(e);
450         }
451         return true;
452      }
453      return false;
454   }
