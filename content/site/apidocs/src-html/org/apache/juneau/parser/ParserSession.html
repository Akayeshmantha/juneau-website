013package org.apache.juneau.parser;
014
015import static org.apache.juneau.parser.ParserContext.*;
016
017import java.io.*;
018import java.lang.reflect.*;
019import java.nio.charset.*;
020import java.util.*;
021
022import org.apache.juneau.*;
023
024/**
025 * Session object that lives for the duration of a single use of {@link Parser}.
026 * <p>
027 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
028 */
029public class ParserSession extends BeanSession {
030
031   private final boolean trimStrings, strict;
032   private final String inputStreamCharset, fileCharset;
033
034   private final Method javaMethod;
035   private final Object outer;
036   private final Object input;
037   private InputStream inputStream;
038   private Reader reader, noCloseReader;
039   private BeanPropertyMeta currentProperty;
040   private ClassMeta<?> currentClass;
041
042   /**
043    * Create a new session using properties specified in the context.
044    *
045    * @param ctx The context creating this session object.
046    *    The context contains all the configuration settings for this object.
047    * @param input The input.
048    *    <br>For character-based parsers, this can be any of the following types:
049    *    <ul>
050    *       <li><jk>null</jk>
051    *       <li>{@link Reader}
052    *       <li>{@link CharSequence}
053    *       <li>{@link InputStream} containing UTF-8 encoded text (or whatever the encoding specified by {@link ParserContext#PARSER_inputStreamCharset}).
054    *       <li><code><jk>byte</jk>[]</code> containing UTF-8 encoded text (or whatever the encoding specified by {@link ParserContext#PARSER_inputStreamCharset}).
055    *       <li>{@link File} containing system encoded text (or whatever the encoding specified by {@link ParserContext#PARSER_fileCharset}).
056    *    </ul>
057    *    <br>For byte-based parsers, this can be any of the following types:
058    *    <ul>
059    *       <li><jk>null</jk>
060    *       <li>{@link InputStream}
061    *       <li><code><jk>byte</jk>[]</code>
062    *       <li>{@link File}
063    *    </ul>
064    * @param op The override properties.
065    *    These override any context properties defined in the context.
066    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
067    * @param outer The outer object for instantiating top-level non-static inner classes.
068    * @param locale The session locale.
069    *    If <jk>null</jk>, then the locale defined on the context is used.
070    * @param timeZone The session timezone.
071    *    If <jk>null</jk>, then the timezone defined on the context is used.
072    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
073    */
074   public ParserSession(ParserContext ctx, ObjectMap op, Object input, Method javaMethod, Object outer, Locale locale, TimeZone timeZone, MediaType mediaType) {
075      super(ctx, op, locale, timeZone, mediaType);
076      if (op == null || op.isEmpty()) {
077         trimStrings = ctx.trimStrings;
078         strict = ctx.strict;
079         inputStreamCharset = ctx.inputStreamCharset;
080         fileCharset = ctx.fileCharset;
081      } else {
082         trimStrings = op.getBoolean(PARSER_trimStrings, ctx.trimStrings);
083         strict = op.getBoolean(PARSER_strict, ctx.strict);
084         inputStreamCharset = op.getString(PARSER_inputStreamCharset, ctx.inputStreamCharset);
085         fileCharset = op.getString(PARSER_fileCharset, ctx.fileCharset);
086      }
087      this.input = input;
088      this.javaMethod = javaMethod;
089      this.outer = outer;
090   }
091
092   /**
093    * Wraps the specified input object inside an input stream.
094    * Subclasses can override this method to implement their own input streams.
095    *
096    * @return The input object wrapped in an input stream, or <jk>null</jk> if the object is null.
097    * @throws ParseException If object could not be converted to an input stream.
098    */
099   public InputStream getInputStream() throws ParseException {
100      if (input == null)
101         return null;
102      if (input instanceof InputStream)
103         return (InputStream)input;
104      if (input instanceof byte[])
105         return new ByteArrayInputStream((byte[])input);
106      if (input instanceof File)
107         try {
108            inputStream = new FileInputStream((File)input);
109            return inputStream;
110         } catch (FileNotFoundException e) {
111            throw new ParseException(e);
112         }
113      throw new ParseException("Cannot convert object of type {0} to an InputStream.", input.getClass().getName());
114   }
115
116
117   /**
118    * Wraps the specified input object inside a reader.
119    * Subclasses can override this method to implement their own readers.
120    *
121    * @return The input object wrapped in a Reader, or <jk>null</jk> if the object is null.
122    * @throws Exception If object could not be converted to a reader.
123    */
124   public Reader getReader() throws Exception {
125      if (input == null)
126         return null;
127      if (input instanceof Reader)
128         return (Reader)input;
129      if (input instanceof CharSequence) {
130         if (reader == null)
131            reader = new ParserReader((CharSequence)input);
132         return reader;
133      }
134      if (input instanceof InputStream || input instanceof byte[]) {
135         InputStream is = (input instanceof InputStream ? (InputStream)input : new ByteArrayInputStream((byte[])input));
136         if (noCloseReader == null) {
137            CharsetDecoder cd = ("default".equalsIgnoreCase(inputStreamCharset) ? Charset.defaultCharset() : Charset.forName(inputStreamCharset)).newDecoder();
138            if (strict) {
139               cd.onMalformedInput(CodingErrorAction.REPORT);
140               cd.onUnmappableCharacter(CodingErrorAction.REPORT);
141            } else {
142               cd.onMalformedInput(CodingErrorAction.REPLACE);
143               cd.onUnmappableCharacter(CodingErrorAction.REPLACE);
144            }
145            noCloseReader = new InputStreamReader(is, cd);
146         }
147         return noCloseReader;
148      }
149      if (input instanceof File) {
150         if (reader == null) {
151            CharsetDecoder cd = ("default".equalsIgnoreCase(fileCharset) ? Charset.defaultCharset() : Charset.forName(fileCharset)).newDecoder();
152            if (strict) {
153               cd.onMalformedInput(CodingErrorAction.REPORT);
154               cd.onUnmappableCharacter(CodingErrorAction.REPORT);
155            } else {
156               cd.onMalformedInput(CodingErrorAction.REPLACE);
157               cd.onUnmappableCharacter(CodingErrorAction.REPLACE);
158            }
159            reader = new InputStreamReader(new FileInputStream((File)input), cd);
160         }
161         return reader;
162      }
163      throw new ParseException("Cannot convert object of type {0} to a Reader.", input.getClass().getName());
164   }
165
166   /**
167    * Returns information used to determine at what location in the parse a failure occurred.
168    *
169    * @return A map, typically containing something like <code>{line:123,column:456,currentProperty:"foobar"}</code>
170    */
171   public Map<String,Object> getLastLocation() {
172      Map<String,Object> m = new LinkedHashMap<String,Object>();
173      if (currentClass != null)
174         m.put("currentClass", currentClass.toString(true));
175      if (currentProperty != null)
176         m.put("currentProperty", currentProperty);
177      return m;
178   }
179
180   /**
181    * Returns the raw input object passed into this session.
182    *
183    * @return The raw input object passed into this session.
184    */
185   protected Object getInput() {
186      return input;
187   }
188
189   /**
190    * Returns the Java method that invoked this parser.
191    * <p>
192    * When using the REST API, this is the Java method invoked by the REST call.
193    * Can be used to access annotations defined on the method or class.
194    *
195    * @return The Java method that invoked this parser.
196   */
197   public final Method getJavaMethod() {
198      return javaMethod;
199   }
200
201   /**
202    * Returns the outer object used for instantiating top-level non-static member classes.
203    * When using the REST API, this is the servlet object.
204    *
205    * @return The outer object.
206   */
207   public final Object getOuter() {
208      return outer;
209   }
210
211   /**
212    * Sets the current bean property being parsed for proper error messages.
213    * @param currentProperty The current property being parsed.
214    */
215   public void setCurrentProperty(BeanPropertyMeta currentProperty) {
216      this.currentProperty = currentProperty;
217   }
218
219   /**
220    * Sets the current class being parsed for proper error messages.
221    * @param currentClass The current class being parsed.
222    */
223   public void setCurrentClass(ClassMeta<?> currentClass) {
224      this.currentClass = currentClass;
225   }
226
227   /**
228    * Returns the {@link ParserContext#PARSER_trimStrings} setting value for this session.
229    *
230    * @return The {@link ParserContext#PARSER_trimStrings} setting value for this session.
231    */
232   public final boolean isTrimStrings() {
233      return trimStrings;
234   }
235
236   /**
237    * Returns the {@link ParserContext#PARSER_strict} setting value for this session.
238    *
239    * @return The {@link ParserContext#PARSER_strict} setting value for this session.
240    */
241   public final boolean isStrict() {
242      return strict;
243   }
244
245   /**
246    * Trims the specified object if it's a <code>String</code> and {@link #isTrimStrings()} returns <jk>true</jk>.
247    *
248    * @param o The object to trim.
249    * @return The trimmmed string if it's a string.
250    */
251   @SuppressWarnings("unchecked")
252   public final <K> K trim(K o) {
253      if (trimStrings && o instanceof String)
254         return (K)o.toString().trim();
255      return o;
256
257   }
258
259   /**
260    * Trims the specified string if {@link ParserSession#isTrimStrings()} returns <jk>true</jk>.
261    *
262    * @param s The input string to trim.
263    * @return The trimmed string, or <jk>null</jk> if the input was <jk>null</jk>.
264    */
265   public final String trim(String s) {
266      if (trimStrings && s != null)
267         return s.trim();
268      return s;
269   }
270
271   /**
272    * Perform cleanup on this context object if necessary.
273    */
274   @Override
275   public boolean close() {
276      if (super.close()) {
277         try {
278            if (inputStream != null)
279               inputStream.close();
280            if (reader != null)
281               reader.close();
282         } catch (IOException e) {
283            throw new BeanRuntimeException(e);
284         }
285         return true;
286      }
287      return false;
288   }
