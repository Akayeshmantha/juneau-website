033/**
034 * A wrapper class around the {@link Class} object that provides cached information
035 * about that class.
036 *
037 * <p>
038 *    Instances of this class can be created through the {@link BeanContext#getClassMeta(Class)} method.
039 * <p>
040 *    The {@link BeanContext} class will cache and reuse instances of this class except for the following class types:
041 * <ul>
042 *    <li>Arrays
043 *    <li>Maps with non-Object key/values.
044 *    <li>Collections with non-Object key/values.
045 * </ul>
046 * <p>
047 *    This class is tied to the {@link BeanContext} class because it's that class that makes the determination
048 *    of what is a bean.
049 *
050 * @param <T> The class type of the wrapped class.
051 */
052@Bean(properties="innerClass,classCategory,elementType,keyType,valueType,notABeanReason,initException,beanMeta")
053public final class ClassMeta<T> implements Type {
054
055   /** Class categories. */
056   enum ClassCategory {
057      MAP, COLLECTION, CLASS, NUMBER, DECIMAL, BOOLEAN, CHAR, DATE, ARRAY, ENUM, OTHER, CHARSEQ, STR, OBJ, URI, BEANMAP, READER, INPUTSTREAM
058   }
059
060   final Class<T> innerClass;                              // The class being wrapped.
061
062   private final Class<? extends T> implClass;             // The implementation class to use if this is an interface.
063   private final ClassCategory cc;                         // The class category.
064   private final Method fromStringMethod;                  // The static valueOf(String) or fromString(String) or forString(String) method (if it has one).
065   private final Constructor<? extends T>
066      noArgConstructor;                                    // The no-arg constructor for this class (if it has one).
067   private final Constructor<T>
068      stringConstructor,                                   // The X(String) constructor (if it has one).
069      numberConstructor,                                   // The X(Number) constructor (if it has one).
070      swapConstructor;                                     // The X(Swappable) constructor (if it has one).
071   private final Class<?>
072      swapMethodType,                                      // The class type of the object in the number constructor.
073      numberConstructorType;
074   private final Method
075      swapMethod,                                          // The swap() method (if it has one).
076      unswapMethod,                                        // The unswap() method (if it has one).
077      namePropertyMethod,                                  // The method to set the name on an object (if it has one).
078      parentPropertyMethod;                                // The method to set the parent on an object (if it has one).
079   private final boolean
080      isDelegate,                                          // True if this class extends Delegate.
081      isAbstract,                                          // True if this class is abstract.
082      isMemberClass;                                       // True if this is a non-static member class.
083   private final Object primitiveDefault;                  // Default value for primitive type classes.
084   private final Map<String,Method>
085      remoteableMethods,                                   // Methods annotated with @Remoteable.  Contains all public methods if class is annotated with @Remotable.
086      publicMethods;                                       // All public methods, including static methods.
087   private final PojoSwap<?,?>[] childPojoSwaps;           // Any PojoSwaps where the normal type is a subclass of this class.
088   private final ConcurrentHashMap<Class<?>,PojoSwap<?,?>>
089      childSwapMap,                                        // Maps normal subclasses to PojoSwaps.
090      childUnswapMap;                                      // Maps swap subclasses to PojoSwaps.
091   private final PojoSwap<T,?> pojoSwap;                   // The object POJO swap associated with this bean (if it has one).
092   private final BeanFilter beanFilter;                    // The bean filter associated with this bean (if it has one).
093   private final MetadataMap extMeta;                      // Extended metadata
094   private final BeanContext beanContext;                  // The bean context that created this object.
095   private final ClassMeta<?>
096      serializedClassMeta,                                 // The transformed class type (if class has swap associated with it).
097      elementType,                                         // If ARRAY or COLLECTION, the element class type.
098      keyType,                                             // If MAP, the key class type.
099      valueType;                                           // If MAP, the value class type.
100   private final BeanMeta<T> beanMeta;                     // The bean meta for this bean class (if it's a bean).
101   private final String
102      notABeanReason,                                      // If this isn't a bean, the reason why.
103      dictionaryName,                                      // The dictionary name of this class if it has one.
104      resolvedDictionaryName;                              // The name if this is an array type (e.g. "X^^").
105   private final Throwable initException;                  // Any exceptions thrown in the init() method.
106   private final InvocationHandler invocationHandler;      // The invocation handler for this class (if it has one).
107
108   private static final Boolean BOOLEAN_DEFAULT = false;
109   private static final Character CHARACTER_DEFAULT = (char)0;
110   private static final Short SHORT_DEFAULT = (short)0;
111   private static final Integer INTEGER_DEFAULT = 0;
112   private static final Long LONG_DEFAULT = 0l;
113   private static final Float FLOAT_DEFAULT = 0f;
114   private static final Double DOUBLE_DEFAULT = 0d;
115   private static final Byte BYTE_DEFAULT = (byte)0;
116
117   /**
118    * Construct a new {@code ClassMeta} based on the specified {@link Class}.
119    *
120    * @param innerClass The class being wrapped.
121    * @param beanContext The bean context that created this object.
122    * @param implClass For interfaces and abstract classes, this represents the "real" class to instantiate.
123    *    Can be <jk>null</jk>.
124    * @param beanFilter The {@link BeanFilter} programmatically associated with this class.
125    *    Can be <jk>null</jk>.
126    * @param pojoSwap The {@link PojoSwap} programmatically associated with this class.
127    *    Can be <jk>null</jk>.
128    * @param childPojoSwap The child {@link PojoSwap PojoSwaps} programmatically associated with this class.
129    *    These are the <code>PojoSwaps</code> that have normal classes that are subclasses of this class.
130    *    Can be <jk>null</jk>.
131    * @param delayedInit Don't call init() in constructor.
132    *    Used for delayed initialization when the possibility of class reference loops exist.
133    */
134   @SuppressWarnings({ "rawtypes", "unchecked" })
135   ClassMeta(Class<T> innerClass, BeanContext beanContext, Class<? extends T> implClass, BeanFilter beanFilter, PojoSwap<T,?> pojoSwap, PojoSwap<?,?>[] childPojoSwaps) {
136      this.innerClass = innerClass;
137      this.beanContext = beanContext;
138
139      // We always immediately add this class meta to the bean context cache so that we can resolve recursive references.
140      if (beanContext != null && beanContext.cmCache != null)
141         beanContext.cmCache.put(innerClass, this);
142
143      this.implClass = implClass;
144      this.childPojoSwaps = childPojoSwaps;
145      this.childSwapMap = childPojoSwaps == null ? null : new ConcurrentHashMap<Class<?>,PojoSwap<?,?>>();
146      this.childUnswapMap = childPojoSwaps == null ? null : new ConcurrentHashMap<Class<?>,PojoSwap<?,?>>();
147
148      Class<T> c = innerClass;
149      ClassCategory _cc = ClassCategory.OTHER;
150      boolean _isDelegate = false;
151      Method
152         _fromStringMethod = null,
153         _swapMethod = null,
154         _unswapMethod = null,
155         _parentPropertyMethod = null,
156         _namePropertyMethod = null;
157      Constructor<T>
158         _noArgConstructor = null,
159         _stringConstructor = null,
160         _swapConstructor = null,
161         _numberConstructor = null;
162      Class<?>
163         _swapMethodType = null,
164         _numberConstructorType = null;
165      Object _primitiveDefault = null;
166      Map<String,Method>
167         _publicMethods = new LinkedHashMap<String,Method>(),
168         _remoteableMethods = null;
169      ClassMeta<?>
170         _keyType = null,
171         _valueType = null,
172         _elementType = null,
173         _serializedClassMeta = null;
174      String
175         _notABeanReason = null,
176         _dictionaryName = null,
177         _resolvedDictionaryName = null;
178      Throwable _initException = null;
179      BeanMeta _beanMeta = null;
180      PojoSwap _pojoSwap = null;
181      InvocationHandler _invocationHandler = null;
182
183      if (c.isPrimitive()) {
184         if (c == Boolean.TYPE)
185            _cc = BOOLEAN;
186         else if (c == Byte.TYPE || c == Short.TYPE || c == Integer.TYPE || c == Long.TYPE || c == Float.TYPE || c == Double.TYPE) {
187            if (c == Float.TYPE || c == Double.TYPE)
188               _cc = DECIMAL;
189            else
190               _cc = NUMBER;
191         }
192         else if (c == Character.TYPE)
193            _cc = CHAR;
194      } else {
195         if (isParentClass(Delegate.class, c))
196            _isDelegate = true;
197
198         if (c == Object.class)
199            _cc = OBJ;
200         else if (c.isEnum())
201            _cc = ENUM;
202         else if (c.equals(Class.class))
203            _cc = CLASS;
204         else if (isParentClass(CharSequence.class, c)) {
205            if (c.equals(String.class))
206               _cc = STR;
207            else
208               _cc = CHARSEQ;
209         }
210         else if (isParentClass(Number.class, c)) {
211            if (isParentClass(Float.class, c) || isParentClass(Double.class, c))
212               _cc = DECIMAL;
213            else
214               _cc = NUMBER;
215         }
216         else if (isParentClass(Collection.class, c))
217            _cc = COLLECTION;
218         else if (isParentClass(Map.class, c)) {
219            if (isParentClass(BeanMap.class, c))
220               _cc = BEANMAP;
221            else
222               _cc = MAP;
223         }
224         else if (c == Character.class)
225            _cc = CHAR;
226         else if (c == Boolean.class)
227            _cc = BOOLEAN;
228         else if (isParentClass(Date.class, c) || isParentClass(Calendar.class, c))
229            _cc = DATE;
230         else if (c.isArray())
231            _cc = ARRAY;
232         else if (isParentClass(URL.class, c) || isParentClass(URI.class, c) || c.isAnnotationPresent(org.apache.juneau.annotation.URI.class))
233            _cc = URI;
234         else if (isParentClass(Reader.class, c))
235            _cc = READER;
236         else if (isParentClass(InputStream.class, c))
237            _cc = INPUTSTREAM;
238      }
239
240      isMemberClass = c.isMemberClass() && ! isStatic(c);
241
242      // Find static fromString(String) or equivalent method.
243      // fromString() must be checked before valueOf() so that Enum classes can create their own
244      //    specialized fromString() methods to override the behavior of Enum.valueOf(String).
245      // valueOf() is used by enums.
246      // parse() is used by the java logging Level class.
247      // forName() is used by Class and Charset
248      for (String methodName : new String[]{"fromString","valueOf","parse","parseString","forName","forString"}) {
249         if (_fromStringMethod == null) {
250            for (Method m : c.getMethods()) {
251               if (isStatic(m) && isPublic(m) && isNotDeprecated(m)) {
252                  String mName = m.getName();
253                  if (mName.equals(methodName) && m.getReturnType() == c) {
254                     Class<?>[] args = m.getParameterTypes();
255                     if (args.length == 1 && args[0] == String.class) {
256                        _fromStringMethod = m;
257                        break;
258                     }
259                  }
260               }
261            }
262         }
263      }
264
265      // Special cases
266      try {
267         if (c == TimeZone.class)
268            _fromStringMethod = c.getMethod("getTimeZone", String.class);
269         else if (c == Locale.class)
270            _fromStringMethod = LocaleAsString.class.getMethod("fromString", String.class);
271      } catch (NoSuchMethodException e1) {}
272
273      // Find swap() method if present.
274      for (Method m : c.getMethods()) {
275         if (isPublic(m) && isNotDeprecated(m) && ! isStatic(m)) {
276            String mName = m.getName();
277            if (mName.equals("swap")) {
278               Class<?>[] pt = m.getParameterTypes();
279               if (pt.length == 1 && pt[0] == BeanSession.class) {
280                  _swapMethod = m;
281                  _swapMethodType = m.getReturnType();
282                  break;
283               }
284            }
285         }
286      }
287      // Find unswap() method if present.
288      if (_swapMethod != null) {
289         for (Method m : c.getMethods()) {
290            if (isPublic(m) && isNotDeprecated(m) && isStatic(m)) {
291               String mName = m.getName();
292               if (mName.equals("unswap")) {
293                  Class<?>[] pt = m.getParameterTypes();
294                  if (pt.length == 2 && pt[0] == BeanSession.class && pt[1] == _swapMethodType) {
295                     _unswapMethod = m;
296                     break;
297                  }
298               }
299            }
300         }
301      }
302
303      // Find @NameProperty and @ParentProperty methods if present.
304      for (Method m : c.getDeclaredMethods()) {
305         if (m.isAnnotationPresent(ParentProperty.class) && m.getParameterTypes().length == 1) {
306            m.setAccessible(true);
307            _parentPropertyMethod = m;
308         }
309         if (m.isAnnotationPresent(NameProperty.class) && m.getParameterTypes().length == 1) {
310            m.setAccessible(true);
311            _namePropertyMethod = m;
312         }
313      }
314
315      // Note:  Primitive types are normally abstract.
316      isAbstract = Modifier.isAbstract(c.getModifiers()) && ! c.isPrimitive();
317
318      // Find constructor(String) method if present.
319      for (Constructor cs : c.getConstructors()) {
320         if (isPublic(cs) && isNotDeprecated(cs)) {
321            Class<?>[] args = cs.getParameterTypes();
322            if (args.length == (isMemberClass ? 1 : 0) && c != Object.class && ! isAbstract) {
323               _noArgConstructor = cs;
324            } else if (args.length == (isMemberClass ? 2 : 1)) {
325               Class<?> arg = args[(isMemberClass ? 1 : 0)];
326               if (arg == String.class)
327                  _stringConstructor = cs;
328               else if (_swapMethodType != null && _swapMethodType.isAssignableFrom(arg))
329                  _swapConstructor = cs;
330               else if (_cc != NUMBER && (Number.class.isAssignableFrom(arg) || (arg.isPrimitive() && (arg == int.class || arg == short.class || arg == long.class || arg == float.class || arg == double.class)))) {
331                  _numberConstructor = cs;
332                  _numberConstructorType = ClassUtils.getWrapperIfPrimitive(arg);
333               }
334            }
335         }
336      }
337
338      if (c.isPrimitive()) {
339         if (c == Boolean.TYPE)
340            _primitiveDefault = BOOLEAN_DEFAULT;
341         else if (c == Character.TYPE)
342            _primitiveDefault = CHARACTER_DEFAULT;
343         else if (c == Short.TYPE)
344            _primitiveDefault = SHORT_DEFAULT;
345         else if (c == Integer.TYPE)
346            _primitiveDefault = INTEGER_DEFAULT;
347         else if (c == Long.TYPE)
348            _primitiveDefault = LONG_DEFAULT;
349         else if (c == Float.TYPE)
350            _primitiveDefault = FLOAT_DEFAULT;
351         else if (c == Double.TYPE)
352            _primitiveDefault = DOUBLE_DEFAULT;
353         else if (c == Byte.TYPE)
354            _primitiveDefault = BYTE_DEFAULT;
355      } else {
356         if (c == Boolean.class)
357            _primitiveDefault = BOOLEAN_DEFAULT;
358         else if (c == Character.class)
359            _primitiveDefault = CHARACTER_DEFAULT;
360         else if (c == Short.class)
361            _primitiveDefault = SHORT_DEFAULT;
362         else if (c == Integer.class)
363            _primitiveDefault = INTEGER_DEFAULT;
364         else if (c == Long.class)
365            _primitiveDefault = LONG_DEFAULT;
366         else if (c == Float.class)
367            _primitiveDefault = FLOAT_DEFAULT;
368         else if (c == Double.class)
369            _primitiveDefault = DOUBLE_DEFAULT;
370         else if (c == Byte.class)
371            _primitiveDefault = BYTE_DEFAULT;
372      }
373
374      for (Method m : c.getMethods())
375         if (isPublic(m) && isNotDeprecated(m))
376            _publicMethods.put(ClassUtils.getMethodSignature(m), m);
377
378      if (c.getAnnotation(Remoteable.class) != null) {
379         _remoteableMethods = _publicMethods;
380      } else {
381         for (Method m : c.getMethods()) {
382            if (m.getAnnotation(Remoteable.class) != null) {
383               if (_remoteableMethods == null)
384                  _remoteableMethods = new LinkedHashMap<String,Method>();
385               _remoteableMethods.put(ClassUtils.getMethodSignature(m), m);
386            }
387         }
388      }
389
390      if (innerClass != Object.class) {
391         _noArgConstructor = (Constructor<T>)findNoArgConstructor(implClass == null ? innerClass : implClass, Visibility.PUBLIC);
392      }
393
394      if (beanFilter == null)
395         beanFilter = findBeanFilter();
396
397      if (_swapMethod != null) {
398         final Method fSwapMethod = _swapMethod;
399         final Method fUnswapMethod = _unswapMethod;
400         final Constructor<T> fSwapConstructor = _swapConstructor;
401         _pojoSwap = new PojoSwap<T,Object>(c, _swapMethod.getReturnType()) {
402            @Override
403            public Object swap(BeanSession session, Object o) throws SerializeException {
404               try {
405                  return fSwapMethod.invoke(o, session);
406               } catch (Exception e) {
407                  throw new SerializeException(e);
408               }
409            }
410            @Override
411            public T unswap(BeanSession session, Object f, ClassMeta<?> hint) throws ParseException {
412               try {
413                  if (fUnswapMethod != null)
414                     return (T)fUnswapMethod.invoke(null, session, f);
415                  if (fSwapConstructor != null)
416                     return fSwapConstructor.newInstance(f);
417                  return super.unswap(session, f, hint);
418               } catch (Exception e) {
419                  throw new ParseException(e);
420               }
421            }
422         };
423      }
424      if (_pojoSwap == null)
425         _pojoSwap = findPojoSwap();
426      if (_pojoSwap == null)
427         _pojoSwap = pojoSwap;
428
429      try {
430
431         // If this is an array, get the element type.
432         if (_cc == ARRAY)
433            _elementType = findClassMeta(innerClass.getComponentType());
434
435         // If this is a MAP, see if it's parameterized (e.g. AddressBook extends HashMap<String,Person>)
436         else if (_cc == MAP) {
437            ClassMeta[] parameters = findParameters();
438            if (parameters != null && parameters.length == 2) {
439               _keyType = parameters[0];
440               _valueType = parameters[1];
441            } else {
442               _keyType = findClassMeta(Object.class);
443               _valueType = findClassMeta(Object.class);
444            }
445         }
446
447         // If this is a COLLECTION, see if it's parameterized (e.g. AddressBook extends LinkedList<Person>)
448         else if (_cc == COLLECTION) {
449            ClassMeta[] parameters = findParameters();
450            if (parameters != null && parameters.length == 1) {
451               _elementType = parameters[0];
452            } else {
453               _elementType = findClassMeta(Object.class);
454            }
455         }
456
457         // If the category is unknown, see if it's a bean.
458         // Note that this needs to be done after all other initialization has been done.
459         else if (_cc == OTHER) {
460
461            BeanMeta newMeta = null;
462            try {
463               newMeta = new BeanMeta(this, beanContext, beanFilter, null);
464               _notABeanReason = newMeta.notABeanReason;
465            } catch (RuntimeException e) {
466               _notABeanReason = e.getMessage();
467               throw e;
468            }
469            if (_notABeanReason == null)
470               _beanMeta = newMeta;
471         }
472
473      } catch (NoClassDefFoundError e) {
474         _initException = e;
475      } catch (RuntimeException e) {
476         _initException = e;
477         throw e;
478      }
479
480      if (_beanMeta != null)
481         _dictionaryName = _resolvedDictionaryName = _beanMeta.getDictionaryName();
482
483      if (_cc == ARRAY && _elementType != null) {
484         _resolvedDictionaryName = _elementType.getResolvedDictionaryName();
485         if (_resolvedDictionaryName != null)
486            _resolvedDictionaryName += "^";
487      }
488
489      _serializedClassMeta = (_pojoSwap == null ? this : findClassMeta(_pojoSwap.getSwapClass()));
490      if (_serializedClassMeta == null)
491         _serializedClassMeta = this;
492
493      if (_beanMeta != null && beanContext != null && beanContext.useInterfaceProxies && innerClass.isInterface())
494         _invocationHandler = new BeanProxyInvocationHandler<T>(_beanMeta);
495
496      this.cc = _cc;
497      this.isDelegate = _isDelegate;
498      this.fromStringMethod = _fromStringMethod;
499      this.swapMethod = _swapMethod;
500      this.unswapMethod = _unswapMethod;
501      this.swapMethodType = _swapMethodType;
502      this.parentPropertyMethod = _parentPropertyMethod;
503      this.namePropertyMethod =_namePropertyMethod;
504      this.noArgConstructor = _noArgConstructor;
505      this.stringConstructor = _stringConstructor;
506      this.swapConstructor = _swapConstructor;
507      this.numberConstructor = _numberConstructor;
508      this.numberConstructorType = _numberConstructorType;
509      this.primitiveDefault = _primitiveDefault;
510      this.publicMethods = _publicMethods;
511      this.remoteableMethods = _remoteableMethods;
512      this.beanFilter = beanFilter;
513      this.pojoSwap = _pojoSwap;
514      this.extMeta = new MetadataMap();
515      this.keyType = _keyType;
516      this.valueType = _valueType;
517      this.elementType = _elementType;
518      this.notABeanReason = _notABeanReason;
519      this.beanMeta = _beanMeta;
520      this.initException = _initException;
521      this.dictionaryName = _dictionaryName;
522      this.resolvedDictionaryName = _resolvedDictionaryName;
523      this.serializedClassMeta = _serializedClassMeta;
524      this.invocationHandler = _invocationHandler;
525   }

571   private ClassMeta<?> findClassMeta(Class<?> c) {
572      return beanContext.getClassMeta(c);
573   }
574
575   private ClassMeta<?>[] findParameters() {
576      return beanContext.findParameters(innerClass, innerClass);
577   }

579   private BeanFilter findBeanFilter() {
580      try {
581         List<Bean> ba = ReflectionUtils.findAnnotations(Bean.class, innerClass);
582         if (! ba.isEmpty())
583            return new AnnotationBeanFilterBuilder(innerClass, ba).build();
584      } catch (Exception e) {
585         throw new RuntimeException(e);
586      }
587      return null;
588   }

590   @SuppressWarnings("unchecked")
591   private PojoSwap<T,?> findPojoSwap() {
592      try {
593         Pojo p = innerClass.getAnnotation(Pojo.class);
594         if (p != null) {
595            Class<?> c = p.swap();
596            if (c != Null.class) {
597               if (ClassUtils.isParentClass(PojoSwap.class, c))
598                  return (PojoSwap<T,?>)c.newInstance();
599               throw new RuntimeException("TODO - Surrogate classes not yet supported.");
600            }
601         }
602         return null;
603      } catch (Exception e) {
604         throw new RuntimeException(e);
605      }
606   }

681   /**
682    * Returns the {@link PojoSwap} where the specified class is the same/subclass of the normal class of
683    * one of the child pojo swaps associated with this class.
684    *
685    * @param normalClass The normal class being resolved.
686    * @return The resolved {@link PojoSwap} or <jk>null</jk> if none were found.
687    */
688   protected PojoSwap<?,?> getChildPojoSwapForSwap(Class<?> normalClass) {
689      if (childSwapMap != null) {
690         PojoSwap<?,?> s = childSwapMap.get(normalClass);
691         if (s == null) {
692            for (PojoSwap<?,?> f : childPojoSwaps)
693               if (s == null && isParentClass(f.getNormalClass(), normalClass))
694                  s = f;
695            if (s == null)
696                s = PojoSwap.NULL;
697            childSwapMap.putIfAbsent(normalClass, s);
698         }
699         if (s == PojoSwap.NULL)
700            return null;
701         return s;
702      }
703      return null;
704   }

731   /**
732    * Locates the no-arg constructor for the specified class.
733    * Constructor must match the visibility requirements specified by parameter 'v'.
734    * If class is abstract, always returns <jk>null</jk>.
735    * Note that this also returns the 1-arg constructor for non-static member classes.
736    *
737    * @param c The class from which to locate the no-arg constructor.
738    * @param v The minimum visibility.
739    * @return The constructor, or <jk>null</jk> if no no-arg constructor exists with the required visibility.
740    */
741   @SuppressWarnings({"rawtypes","unchecked"})
742   protected static <T> Constructor<? extends T> findNoArgConstructor(Class<T> c, Visibility v) {
743      int mod = c.getModifiers();
744      if (Modifier.isAbstract(mod))
745         return null;
746      boolean isMemberClass = c.isMemberClass() && ! isStatic(c);
747      for (Constructor cc : c.getConstructors()) {
748         mod = cc.getModifiers();
749         if (cc.getParameterTypes().length == (isMemberClass ? 1 : 0) && v.isVisible(mod) && isNotDeprecated(cc))
750            return v.transform(cc);
751      }
752      return null;
753   }

1476   private static class LocaleAsString {
1477      private static Method forLanguageTagMethod;
1478      static {
1479         try {
1480            forLanguageTagMethod = Locale.class.getMethod("forLanguageTag", String.class);
1481         } catch (NoSuchMethodException e) {}
1482      }
1483
1484      @SuppressWarnings("unused")
1485      public static final Locale fromString(String localeString) {
1486         if (forLanguageTagMethod != null) {
1487            if (localeString.indexOf('_') != -1)
1488               localeString = localeString.replace('_', '-');
1489            try {
1490               return (Locale)forLanguageTagMethod.invoke(null, localeString);
1491            } catch (Exception e) {
1492               throw new BeanRuntimeException(e);
1493            }
1494         }
1495         String[] v = localeString.toString().split("[\\-\\_]");
1496         if (v.length == 1)
1497            return new Locale(v[0]);
1498         else if (v.length == 2)
1499            return new Locale(v[0], v[1]);
1500         else if (v.length == 3)
1501            return new Locale(v[0], v[1], v[2]);
1502         throw new BeanRuntimeException("Could not convert string ''{0}'' to a Locale.", localeString);
1503      }
1504   }
