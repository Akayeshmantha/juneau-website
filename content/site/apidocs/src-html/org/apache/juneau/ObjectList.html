023/**
024 * Java implementation of a JSON array.
025 * <p>
026 *    An extension of {@link LinkedList}, so all methods available to in that class are also available
027 *    to this class.
028 * <p>
029 *    Note that the use of this class is optional.  The serializers will accept any objects that implement
030 *    the {@link Collection} interface.  But this class provides some useful additional functionality
031 *    when working with JSON models constructed from Java Collections Framework objects.  For example, a
032 *    constructor is provided for converting a JSON array string directly into a {@link List}.  It also contains
033 *    accessor methods for to avoid common typecasting when accessing elements in a list.
034 *
035 * <h6 class='topic'>Example:</h6>
036 * <p class='bcode'>
037 *    <jc>// Construct an empty List</jc>
038 *    List l = <jk>new</jk> ObjectList();
039 *
040 *    <jc>// Construct a list of objects using various methods</jc>
041 *    l = <jk>new</jk> ObjectList().append(<js>"foo"</js>).append(123).append(<jk>true</jk>);
042 *    l = <jk>new</jk> ObjectList().append(<js>"foo"</js>, 123, <jk>true</jk>);  <jc>// Equivalent</jc>
043 *    l = <jk>new</jk> ObjectList(<js>"foo"</js>, 123, <jk>true</jk>);  <jc>// Equivalent</jc>
044 *
045 *    <jc>// Construct a list of integers from JSON</jc>
046 *    l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
047 *
048 *    <jc>// Construct a list of generic ObjectMap objects from JSON</jc>
049 *    l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:'bing'}]"</js>);
050 *
051 *    <jc>// Construct a list of integers from XML</jc>
052 *    String xml = <js>"&lt;array&gt;&lt;number&gt;1&lt;/number&gt;&lt;number&gt;2&lt;/number&gt;&lt;number&gt;3&lt;/number&gt;&lt;/array&gt;"</js>;
053 *    l = <jk>new</jk> ObjectList(xml, DataFormat.<jsf>XML</jsf>);
054 *    l = (List)XmlParser.<jsf>DEFAULT</jsf>.parse(xml);  <jc>// Equivalent</jc>
055 *    l = (List)XmlParser.<jsf>DEFAULT</jsf>.parse(Object.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
056 *    l = XmlParser.<jsf>DEFAULT</jsf>.parse(List.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
057 *    l = XmlParser.<jsf>DEFAULT</jsf>.parse(ObjectList.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
058 *
059 *    <jc>// Construct JSON from ObjectList</jc>
060 *    l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:'bing'}]"</js>);
061 *    String json = l.toString();  <jc>// Produces "[{foo:'bar'},{baz:'bing'}]"</jc>
062 *    json = l.toString(JsonSerializer.<jsf>DEFAULT_CONDENSED</jsf>);  <jc>// Equivalent</jc>
063 *    json = JsonSerializer.<jsf>DEFAULT_CONDENSED</jsf>.serialize(l);  <jc>// Equivalent</jc>
064 *
065 *    <jc>// Get one of the entries in the list as an Integer</jc>
066 *    l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
067 *    Integer i = l.getInt(1);
068 *    i = l.get(Integer.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
069 *
070 *    <jc>// Get one of the entries in the list as an Float</jc>
071 *    l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
072 *    Float f = l.getFloat(1); <jc>// Returns 2f </jc>
073 *    f = l.get(Float.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
074 *
075 *    <jc>// Same as above, except converted to a String</jc>
076 *    l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
077 *    String s = l.getString(1); <jc>// Returns "2" </jc>
078 *    s = l.get(String.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
079 *
080 *    <jc>// Get one of the entries in the list as a bean (converted to a bean if it isn't already one)</jc>
081 *    l = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
082 *    Person p = l.get(Person.<jk>class</jk>, 0);
083 *
084 *    <jc>// Iterate over a list of beans using the elements() method</jc>
085 *    ObjectList ObjectList = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
086 *    <jk>for</jk> (Person p : ObjectList.elements(Person.<jk>class</jk>) {
087 *       <jc>// Do something with p</jc>
088 *    }
089 * </p>
090 * <p>
091 *    This class is not thread safe.
092 * </p>
093 *
094 * @author James Bognar (james.bognar@salesforce.com)
095 */

134   /**
135    * Construct a JSON array directly from text using the specified parser.
136    *
137    * @param s The string being parsed.
138    * @param p The parser to use to parse the input.
139    * @throws ParseException If the input contains a syntax error or is malformed.
140    */
141   public ObjectList(CharSequence s, Parser p) throws ParseException {
142      this(p == null ? BeanContext.DEFAULT : p.getBeanContext());
143      if (p == null)
144         p = JsonParser.DEFAULT;
145      if (s != null)
146         p.parseIntoCollection(s, this, beanContext.object());
147   }

154    */

163    * @param p The parser to use to parse the input.
164    * @throws ParseException If the input contains a syntax error or is malformed.

168      parseReader(r, p);
169   }

171   private void parseReader(Reader r, Parser p) throws ParseException {
172      if (p == null)
173         p = JsonParser.DEFAULT;
174      p.parseIntoCollection(r, this, beanContext.object());
175   }

340    * Same as {@link #get(Class,int) get(Class,int)}, but the key is a slash-delimited
341    *    path used to traverse entries in this POJO.
342    * <p>
343    *    For example, the following code is equivalent:
344    * </p>
345    * <p class='bcode'>
346    *    ObjectMap m = getObjectMap();

348    *    <jc>// Long way</jc>
349    *    <jk>long</jk> l = m.getObjectMap(<js>"foo"</js>).getObjectList(<js>"bar"</js>).getObjectMap(<js>"0"</js>).getLong(<js>"baz"</js>);
350    *
351    *    <jc>// Using this method</jc>
352    *    <jk>long</jk> l = m.getAt(<jk>long</jk>.<jk>class</jk>, <js>"foo/bar/0/baz"</js>);
353    * </p>
354    * <p>
355    *    This method uses the {@link PojoRest} class to perform the lookup, so the map can contain
356    *       any of the various class types that the {@link PojoRest} class supports (e.g. beans, collections, arrays).
357    * </p>
358    *
359    * @param <T> The class type.
360    * @param type The class type.
361    * @param path The path to the entry.
362    * @return The value, or <jk>null</jk> if the entry doesn't exist.
363    */
364   public <T> T getAt(Class<T> type, String path) {
365      return getPojoRest().get(type, path);
366   }

451   /**
452    * Creates an {@link Iterable} with elements of the specified child type.
453    * <p>
454    * Attempts to convert the child objects to the correct type if they aren't already the correct type.
455    * <p>
456    * The <code>next()</code> method on the returned iterator may throw a {@link InvalidDataConversionException} if
457    *    the next element cannot be converted to the specified type.
458    * <p>
459    * See {@link BeanContext#convertToType(Object, ClassMeta)} for a description of valid conversions.
460    *
461    * <h6 class='topic'>Example:</h6>
462    * <p class='bcode'>
463    *    <jc>// Iterate over a list of ObjectMaps.</jc>
464    *    ObjectList l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:123}]"</js>);
465    *    for (ObjectMap m : l.elements(ObjectMap.<jk>class</jk>)) {
466    *       <jc>// Do something with m.</jc>
467    *    }
468    *
469    *    <jc>// Iterate over a list of ints.</jc>
470    *    ObjectList l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
471    *    for (Integer i : l.elements(Integer.<jk>class</jk>)) {
472    *       <jc>// Do something with i.</jc>
473    *    }
474    *
475    *    <jc>// Iterate over a list of beans.</jc>
476    *    <jc>// Automatically converts to beans.</jc>
477    *    ObjectList l = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
478    *    for (Person p : l.elements(Person.<jk>class</jk>)) {
479    *       <jc>// Do something with p.</jc>
480    *    }
481    * </p>
482    *
483    * @param <E> The child object type.
484    * @param childType The child object type.
485    * @return A new <code>Iterable</code> object over this list.
486    */
487   public <E> Iterable<E> elements(final Class<E> childType) {
488      final Iterator<?> i = iterator();
489      return new Iterable<E>() {
490
491         @Override /* Iterable */
492         public Iterator<E> iterator() {
493            return new Iterator<E>() {
494
495               @Override /* Iterator */
496               public boolean hasNext() {
497                  return i.hasNext();
498               }
499
500               @Override /* Iterator */
501               public E next() {
502                  return beanContext.convertToType(i.next(), childType);
503               }
504
505               @Override /* Iterator */
506               public void remove() {
507                  i.remove();
508               }
509
510            };
511         }
512      };
513   }
