013package org.apache.juneau.remoteable;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.io.*;
019import java.lang.annotation.*;
020
021/**
022 * Annotation applied to Java methods on interface proxy classes.
023 * <p>
024 * TODO <i>(sorry)</i>
025 * <p>
026 * The return type on the Java method can be any of the following:
027 * <ul>
028 *    <li><jk>void</jk> - Don't parse any response.  Note that the method will still throw an exception if an error HTTP status is returned.
029 *    <li>Any parsable POJO - The body of the response will be converted to the POJO using the parser defined on the <code>RestClient</code>.
030 *    <li><code>HttpResponse</code> - Returns the raw <code>HttpResponse</code> returned by the inner <code>HttpClient</code>.
031 *    <li>{@link Reader} - Returns access to the raw reader of the response.
032 *    <li>{@link InputStream} - Returns access to the raw input stream of the response.
033 * </ul>
034 *
035 * <h6 class='topic'>Additional Information</h6>
036 * <ul class='doctree'>
037 *    <li class='link'><a class='doclink' href='../../../../overview-summary.html#Remoteable.3rdParty'>Interface proxies against 3rd-party REST interfaces</a>
038 *    <li class='jp'><a class='doclink' href='package-summary.html#TOC'>org.apache.juneau.remoteable</a>
039 * </ul>
040 */
041@Documented
042@Target(METHOD)
043@Retention(RUNTIME)
044@Inherited
045public @interface RemoteMethod {
046
047   /**
048    * The path to the REST service for this Java method relative to the parent proxy interface URL.
049    * <p>
050    * The default value is the Java method name (e.g. <js>"http://localhost/root-url/org.foo.MyInterface/myMethod"</js>) if
051    *    {@link Remoteable#methodPaths() @Remoteable.methodPaths()} is <js>"NAME"</js>, or the Java method signature
052    *    (e.g. <js>"http://localhost/root-url/org.foo.MyInterface/myMethod(int,boolean,java.lang.String)"</js>) if
053    *    it's <js>"SIGNATURE"</js>.
054    */
055   String path() default "";
056
057   /**
058    * Defines the HTTP method to use for REST calls.
059    * <p>
060    * Possible values:
061    * <ul>
062    *    <li><js>"POST"</js> (default) - Parameters are serialized using the serializer registered with the RestClient.
063    *    <li><js>"GET"</js> - Parameters are serialized using the UrlEncodingSerializer registered with the RestClient.
064    * </ul>
065    * <p>
066    * The default value is <js>"POST"</js>.
067    */
068   String httpMethod() default "POST";
069
070   /**
071    * The value the remoteable method returns.
072    * <p>
073    * Possible values:
074    * <ul>
075    *    <li>{@link ReturnValue#BODY} (default) - The body of the HTTP response converted to a POJO.
076    *       <br>The return type on the Java method can be any of the following:
077    *       <ul>
078    *          <li><jk>void</jk> - Don't parse any response.  Note that the method will still throw an exception if an
079    *                error HTTP status is returned.
080    *          <li>Any parsable POJO - The body of the response will be converted to the POJO using the parser defined
081    *                on the <code>RestClient</code>.
082    *          <li><code>HttpResponse</code> - Returns the raw <code>HttpResponse</code> returned by the inner
083    *                <code>HttpClient</code>.
084    *          <li>{@link Reader} - Returns access to the raw reader of the response.
085    *          <li>{@link InputStream} - Returns access to the raw input stream of the response.
086    *       </ul>
087    *    <li>{@link ReturnValue#HTTP_STATUS} - The HTTP status code on the response.
088    *       <br>The return type on the Java method can be any of the following:
089    *       <ul>
090    *          <li><jk>int</jk>/<code>Integer</code> - The HTTP response code.
091    *          <li><jk>boolean</jk>/<code>Boolean</code> - <jk>true</jk> if the response code is <code>&lt;400</code>
092    *       </ul>
093    * </ul>
094    */
095   ReturnValue returns() default ReturnValue.BODY;
096}
