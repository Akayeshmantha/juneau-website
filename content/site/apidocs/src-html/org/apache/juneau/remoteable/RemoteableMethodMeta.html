013package org.apache.juneau.remoteable;
014
015import static org.apache.juneau.internal.StringUtils.*;
016
017import java.lang.annotation.*;
018import java.lang.reflect.*;
019import java.util.*;
020
021import org.apache.juneau.internal.*;
022
023/**
024 * Contains the meta-data about a Java method on a remoteable interface.
025 * <p>
026 * Captures the information in {@link RemoteMethod @RemoteMethod} annotations for caching and reuse.
027 */
028public class RemoteableMethodMeta {
029
030   private final String httpMethod;
031   private final String url;
032   private final RemoteMethodArg[] queryArgs, headerArgs, formDataArgs;
033   private final Integer[] otherArgs;
034   private final Integer bodyArg;
035
036   /**
037    * Constructor.
038    *
039    * @param restUrl The absolute URL of the REST interface backing the interface proxy.
040    * @param m The Java method.
041    */
042   public RemoteableMethodMeta(final String restUrl, Method m) {
043      Builder b = new Builder(restUrl, m);
044      this.httpMethod = b.httpMethod;
045      this.url = b.url;
046      this.queryArgs = b.queryArgs.toArray(new RemoteMethodArg[b.queryArgs.size()]);
047      this.formDataArgs = b.formDataArgs.toArray(new RemoteMethodArg[b.formDataArgs.size()]);
048      this.headerArgs = b.headerArgs.toArray(new RemoteMethodArg[b.headerArgs.size()]);
049      this.otherArgs = b.otherArgs.toArray(new Integer[b.otherArgs.size()]);
050      this.bodyArg = b.bodyArg;
051   }
052
053   private static class Builder {
054      private String httpMethod, url;
055      private List<RemoteMethodArg>
056         queryArgs = new LinkedList<RemoteMethodArg>(),
057         headerArgs = new LinkedList<RemoteMethodArg>(),
058         formDataArgs = new LinkedList<RemoteMethodArg>();
059      private List<Integer> otherArgs = new LinkedList<Integer>();
060      private Integer bodyArg;
061
062      private Builder(String restUrl, Method m) {
063         Remoteable r = m.getDeclaringClass().getAnnotation(Remoteable.class);
064         RemoteMethod rm = m.getAnnotation(RemoteMethod.class);
065
066         httpMethod = rm == null ? "POST" : rm.httpMethod();
067         if (! isOneOf(httpMethod, "GET", "POST"))
068            throw new RemoteableMetadataException(m, "Invalid value specified for @RemoteMethod.httpMethod() annotation.  Valid values are [GET,POST].");
069
070         String path = rm == null || rm.path().isEmpty() ? null : rm.path();
071         String methodPaths = r == null ? "NAME" : r.methodPaths();
072
073         if (! isOneOf(methodPaths, "NAME", "SIGNATURE"))
074            throw new RemoteableMetadataException(m, "Invalid value specified for @Remoteable.methodPaths() annotation.  Valid values are [NAME,SIGNATURE].");
075
076         url =
077            trimSlashes(restUrl)
078            + '/'
079            + (path != null ? trimSlashes(path) : urlEncode("NAME".equals(methodPaths) ? m.getName() : ClassUtils.getMethodSignature(m)));
080
081         int index = 0;
082         for (Annotation[] aa : m.getParameterAnnotations()) {
083            boolean annotated = false;
084            for (Annotation a : aa) {
085               Class<?> ca = a.annotationType();
086               if (ca == Query.class) {
087                  Query q = (Query)a;
088                  annotated = queryArgs.add(new RemoteMethodArg(q.value(), index, false));
089               } else if (ca == QueryIfNE.class) {
090                  Query q = (Query)a;
091                  annotated = queryArgs.add(new RemoteMethodArg(q.value(), index, true));
092               } else if (ca == FormData.class) {
093                  FormData f = (FormData)a;
094                  annotated = formDataArgs.add(new RemoteMethodArg(f.value(), index, false));
095               } else if (ca == FormDataIfNE.class) {
096                  FormData f = (FormData)a;
097                  annotated = formDataArgs.add(new RemoteMethodArg(f.value(), index, true));
098               } else if (ca == Header.class) {
099                  Header h = (Header)a;
100                  annotated = headerArgs.add(new RemoteMethodArg(h.value(), index, false));
101               } else if (ca == HeaderIfNE.class) {
102                  Header h = (Header)a;
103                  annotated = headerArgs.add(new RemoteMethodArg(h.value(), index, true));
104               } else if (ca == Body.class) {
105                  annotated = true;
106                  if (bodyArg == null)
107                     bodyArg = index;
108                  else
109                     throw new RemoteableMetadataException(m, "Multiple @Body parameters found.  Only one can be specified per Java method.");
110               }
111            }
112            if (! annotated)
113               otherArgs.add(index);
114            index++;
115         }
116
117         if (bodyArg != null && otherArgs.size() > 0)
118            throw new RemoteableMetadataException(m, "@Body and non-annotated parameters found together.  Non-annotated parameters cannot be used when @Body is used.");
119      }
120   }
121
122   /**
123    * Returns the value of the {@link RemoteMethod#httpMethod()} annotation on this Java method.
124    * @return The value of the {@link RemoteMethod#httpMethod()} annotation, never <jk>null</jk>.
125    */
126   public String getHttpMethod() {
127      return httpMethod;
128   }
129
130   /**
131    * Returns the absolute URL of the REST interface invoked by this Java method.
132    * @return The absolute URL of the REST interface, never <jk>null</jk>.
133    */
134   public String getUrl() {
135      return url;
136   }
137
138   /**
139    * Returns the {@link Query @Query} annotated arguments on this Java method.
140    * @return A map of {@link Query#value() @Query.value()} names to zero-indexed argument indices.
141    */
142   public RemoteMethodArg[] getQueryArgs() {
143      return queryArgs;
144   }
145
146   /**
147    * Returns the {@link FormData @FormData} annotated arguments on this Java method.
148    * @return A map of {@link FormData#value() @FormData.value()} names to zero-indexed argument indices.
149    */
150   public RemoteMethodArg[] getFormDataArgs() {
151      return formDataArgs;
152   }
153
154   /**
155    * Returns the {@link Header @Header} annotated arguments on this Java method.
156    * @return A map of {@link Header#value() @Header.value()} names to zero-indexed argument indices.
157    */
158   public RemoteMethodArg[] getHeaderArgs() {
159      return headerArgs;
160   }
161
162   /**
163    * Returns the remaining non-annotated arguments on this Java method.
164    * @return A list of zero-indexed argument indices.
165    */
166   public Integer[] getOtherArgs() {
167      return otherArgs;
168   }
169
170   /**
171    * Returns the argument annotated with {@link Body @Body}.
172    * @return A index of the argument with the {@link Body @Body} annotation, or <jk>null</jk> if no argument exists.
173    */
174   public Integer getBodyArg() {
175      return bodyArg;
176   }
177}
