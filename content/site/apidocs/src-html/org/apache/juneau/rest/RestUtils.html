075   /**
076    * Identical to {@link HttpServletRequest#getPathInfo()} but doesn't decode encoded characters.
077    *
078    * @param req The HTTP request
079    * @return The un-decoded path info.
080    */
081   public static String getPathInfoUndecoded(HttpServletRequest req) {
082      String requestURI = req.getRequestURI();
083      String contextPath = req.getContextPath();
084      String servletPath = req.getServletPath();
085      int l = contextPath.length() + servletPath.length();
086      if (requestURI.length() == l)
087         return null;
088      return requestURI.substring(l);
089   }

091   /**
092    * Efficiently trims the path info part from a request URI.
093    *
094    * <p>
095    * The result is the URI of the servlet itself.
096    *
097    * @param requestURI The value returned by {@link HttpServletRequest#getRequestURL()}
098    * @param contextPath The value returned by {@link HttpServletRequest#getContextPath()}
099    * @param servletPath The value returned by {@link HttpServletRequest#getServletPath()}
100    * @return The same StringBuilder with remainder trimmed.
101    */
102   public static StringBuffer trimPathInfo(StringBuffer requestURI, String contextPath, String servletPath) {
103      if (servletPath.equals("/"))
104         servletPath = "";
105      if (contextPath.equals("/"))
106         contextPath = "";
107
108      try {
109         // Given URL:  http://hostname:port/servletPath/extra
110         // We want:    http://hostname:port/servletPath
111         int sc = 0;
112         for (int i = 0; i < requestURI.length(); i++) {
113            char c = requestURI.charAt(i);
114            if (c == '/') {
115               sc++;
116               if (sc == 3) {
117                  if (servletPath.isEmpty()) {
118                     requestURI.setLength(i);
119                     return requestURI;
120                  }
121
122                  // Make sure context path follows the authority.
123                  for (int j = 0; j < contextPath.length(); i++, j++)
124                     if (requestURI.charAt(i) != contextPath.charAt(j))
125                        throw new Exception("case=1");
126
127                  // Make sure servlet path follows the authority.
128                  for (int j = 0; j < servletPath.length(); i++, j++)
129                     if (requestURI.charAt(i) != servletPath.charAt(j))
130                        throw new Exception("case=2");
131
132                  // Make sure servlet path isn't a false match (e.g. /foo2 should not match /foo)
133                  c = (requestURI.length() == i ? '/' : requestURI.charAt(i));
134                  if (c == '/' || c == '?') {
135                     requestURI.setLength(i);
136                     return requestURI;
137                  }
138
139                  throw new Exception("case=3");
140               }
141            } else if (c == '?') {
142               if (sc != 2)
143                  throw new Exception("case=4");
144               if (servletPath.isEmpty()) {
145                  requestURI.setLength(i);
146                  return requestURI;
147               }
148               throw new Exception("case=5");
149            }
150         }
151         if (servletPath.isEmpty())
152            return requestURI;
153         throw new Exception("case=6");
154      } catch (Exception e) {
155         throw new RuntimeException("Could not find servlet path in request URI.  URI=["+requestURI+"], servletPath=["+servletPath+"]", e);
156      }
157   }

159   static String[] parseHeader(String s) {
160      int i = s.indexOf(':');
161      if (i == -1)
162         return null;
163      String name = s.substring(0, i).trim().toLowerCase(Locale.ENGLISH);
164      String val = s.substring(i+1).trim();
165      return new String[]{name,val};
166   }
167
168   /**
169    * Parses key/value pairs separated by either : or =
170    */
171   static String[] parseKeyValuePair(String s) {
172      int i = -1;
173      for (int j = 0; j < s.length() && i < 0; j++) { s.indexOf(':');
174         char c = s.charAt(j);
175         if (c == '=' || c == ':')
176            i = j;
177      }
178      if (i == -1)
179         return null;
180      String name = s.substring(0, i).trim();
181      String val = s.substring(i+1).trim();
182      return new String[]{name,val};
183   }
