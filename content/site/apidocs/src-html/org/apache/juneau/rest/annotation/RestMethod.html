017
018import java.lang.annotation.*;

026/**
027 * Identifies a REST Java method on a {@link RestServlet} implementation class.
028 * <p>
029 * Refer to <a class='doclink' href='../package-summary.html#TOC'>org.apache.juneau.rest</a> doc for information on using this class.
030 */
031@Documented
032@Target(METHOD)
033@Retention(RUNTIME)
034@Inherited
035public @interface RestMethod {
036
037   /**
038    * REST method name.
039    * <p>
040    * Typically <js>"GET"</js>, <js>"PUT"</js>, <js>"POST"</js>, <js>"DELETE"</js>, or <js>"OPTIONS"</js>.
041    * <p>
042    * Method names are case-insensitive (always folded to upper-case).
043    * <p>
044    * Besides the standard HTTP method names, the following can also be specified:
045    * <ul>
046    *    <li><js>"*"</js> - Denotes any method.
047    *       <br>Use this if you want to capture any HTTP methods in a single Java method.
048    *       <br>The {@link Method @Method} annotation and/or {@link RestRequest#getMethod()} method can be used
049    *       to distinguish the actual HTTP method name.
050    *    <li><js>""</js> - Auto-detect.
051    *       <br>The method name is determined based on the Java method name.
052    *       <br>For example, if the method is <code>doPost(...)</code>, then the method name is automatically detected as <js>"POST"</js>.
053    *    <li><js>"PROXY"</js> - Remote-proxy interface.
054    *       <br>This denotes a Java method that returns an object (usually an interface, often annotated with the {@link Remoteable @Remoteable} annotation)
055    *       to be used as a remote proxy using <code>RestClient.getRemoteableProxy(Class<T> interfaceClass, String url)</code>.
056    *       <br>This allows you to construct client-side interface proxies using REST as a transport medium.
057    *       <br>Conceptually, this is simply a fancy <code>POST</code> against the url <js>"/{path}/{javaMethodName}"</js> where the arguments
058    *       are marshalled from the client to the server as an HTTP body containing an array of objects,
059    *       passed to the method as arguments, and then the resulting object is marshalled back to the client.
060    *    <li>Anything else - Overloaded non-HTTP-standard names that are passed in through a <code>&amp;method=methodName</code> URL parameter.
061    * </ul>
062    */
063   String name() default "";
064
065   /**
066    * Optional path pattern for the specified method.
067    * <p>
068    * Appending <js>"/*"</js> to the end of the path pattern will make it match any remainder too.<br>
069    * Not appending <js>"/*"</js> to the end of the pattern will cause a 404 (Not found) error to occur
070    *    if the exact pattern is not found.
071    */
072   String path() default "/*";
073
074   /**
075    * URL path pattern priority.
076    * <p>
077    * To force path patterns to be checked before other path patterns, use a higher priority number.
078    * <p>
079    * By default, it's <code>0</code>, which means it will use an internal heuristic to
080    *    determine a best match.
081    */
082   int priority() default 0;
083
084   /**
085    * Method guards.
086    * <p>
087    * Associates one or more {@link RestGuard RestGuards} with a method call.
088    * These guards get called immediately before execution of the REST method.
089    * <p>
090    * Typically, guards will be used for permissions checking on the user making the request,
091    *    but it can also be used for other purposes like pre-call validation of a request.
092    */
093   Class<? extends RestGuard>[] guards() default {};
094
095   /**
096    * Method response converters.
097    * <p>
098    * Associates one or more {@link RestConverter RestConverters} with a method call.
099    * These converters get called immediately after execution of the REST method in the same
100    *       order specified in the annotation.
101    * <p>
102    * Can be used for performing post-processing on the response object before serialization.
103    * <p>
104    * Default converters are available in the <a class='doclink' href='../converters/package-summary.html#TOC'>org.apache.juneau.rest.converters</a> package.
105    */
106   Class<? extends RestConverter>[] converters() default {};
107
108   /**
109    * Method matchers.
110    * <p>
111    * Associates one more more {@link RestMatcher RestMatchers} with this method.
112    * <p>
113    * Matchers are used to allow multiple Java methods to handle requests assigned to the same
114    *    URL path pattern, but differing based on some request attribute, such as a specific header value.
115    * <p>
116    * See {@link RestMatcher} for details.
117    */
118   Class<? extends RestMatcher>[] matchers() default {};
119
120   /**
121    * Overrides the list of serializers assigned at the method level.
122    * <p>
123    * Use this annotation when the list of serializers assigned to a method differs from the list of serializers assigned at the servlet level.
124    * <p>
125    * To append to the list of serializers assigned at the servlet level, use <code>serializersInherit=<jsf>SERIALIZERS</jsf></code>.
126    *
127    * <p class='bcode'>
128    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServlet {
129    *
130    *       <ja>@RestMethod</ja>(
131    *          name=<js>"GET"</js>,
132    *          path=<js>"/foo"</js>,
133    *          serializers=MySpecialSerializer.<jk>class</jk>,
134    *          serializersInherit=<jsf>SERIALIZERS</jsf>
135    *       )
136    *       <jk>public</jk> Object doGetWithSpecialAcceptType() {
137    *          <jc>// Handle request for special Accept type</jc>
138    *       }
139    *    }
140    * </p>
141    */
142   Class<? extends Serializer>[] serializers() default {};
143
144   /**
145    * Used in conjunction with {@link #serializers()} to identify what class-level settings are inherited by the method serializer group.
146    * <p>
147    * Possible values:
148    * <ul>
149    *    <li>{@link Inherit#SERIALIZERS} - Inherit class-level serializers.
150    *    <li>{@link Inherit#PROPERTIES} - Inherit class-level properties.
151    *    <li>{@link Inherit#TRANSFORMS} - Inherit class-level transforms.
152    * </ul>
153    * <p>
154    * For example, to inherit all serializers, properties, and transforms from the servlet class:
155    * </p>
156    * <p class='bcode'>
157    *    <ja>@RestMethod</ja>(
158    *       path=<js>"/foo"</js>,
159    *       serializers=MySpecialSerializer.<jk>class</jk>,
160    *       serializersInherit={<jsf>SERIALIZERS</jsf>,<jsf>PROPERTIES</jsf>,<jsf>TRANSFORMS</jsf>}
161    *    )
162    * </p>
163    */
164   Inherit[] serializersInherit() default {};
165
166   /**
167    * Overrides the list of parsers assigned at the method level.
168    * <p>
169    * Use this annotation when the list of parsers assigned to a method differs from the list of parsers assigned at the servlet level.
170    * <p>
171    * To append to the list of serializers assigned at the servlet level, use <code>serializersInherit=<jsf>SERIALIZERS</jsf></code>.
172    *
173    * <p class='bcode'>
174    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServlet {
175    *
176    *       <ja>@RestMethod</ja>(
177    *          name=<js>"PUT"</js>,
178    *          path=<js>"/foo"</js>,
179    *          parsers=MySpecialParser.<jk>class</jk>,
180    *          parsersInherit=<jsf>PARSERS</jsf>
181    *       )
182    *       <jk>public</jk> Object doGetWithSpecialAcceptType() {
183    *          <jc>// Handle request for special Accept type</jc>
184    *       }
185    *    }
186    * </p>
187    */
188   Class<? extends Parser>[] parsers() default {};
189
190   /**
191    * Used in conjunction with {@link #parsers()} to identify what class-level settings are inherited by the method parser group.
192    * <p>
193    * Possible values:
194    * <ul>
195    *    <li>{@link Inherit#PARSERS} - Inherit class-level parsers.
196    *    <li>{@link Inherit#PROPERTIES} - Inherit class-level properties.
197    *    <li>{@link Inherit#TRANSFORMS} - Inherit class-level transforms.
198    * </ul>
199    * <p>
200    * For example, to inherit all parsers, properties, and transforms from the servlet class:
201    * <p class='bcode'>
202    *    <ja>@RestMethod</ja>(
203    *       path=<js>"/foo"</js>,
204    *       parsers=MySpecialParser.<jk>class</jk>,
205    *       parsersInherit={<jsf>PARSERS</jsf>,<jsf>PROPERTIES</jsf>,<jsf>TRANSFORMS</jsf>}
206    *    )
207    * </p>
208    */
209   Inherit[] parsersInherit() default {};
210
211   /**
212    * Appends to the list of {@link Encoder encoders} specified on the servlet.
213    * <p>
214    * Use this annotation when the list of encoders assigned to a method differs from the list of encoders assigned at the servlet level.
215    * <p>
216    * These can be used to enable various kinds of compression (e.g. <js>"gzip"</js>) on requests and responses.
217    *
218    * <p class='bcode'>
219    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServlet {
220    *
221    *       <ja>@RestMethod</ja>(
222    *          name=<js>"PUT"</js>,
223    *          path=<js>"/foo"</js>,
224    *          encoders={GzipEncoder.<jk>class</jk>}
225    *       )
226    *       <jk>public</jk> Object doGetWithSpecialEncoding() {
227    *          <jc>// Handle request with special encoding</jc>
228    *       }
229    *    }
230    * </p>
231    * <p>
232    * If you want to OVERRIDE the set of encoders specified by the servlet, combine this annotation with <code><ja>@RestMethod</ja>(inheritEncoders=<jk>false</jk>)</code>.
233    */
234   Class<? extends Encoder>[] encoders() default {};
235
236   /**
237    * Specifies whether the method should inherit encoders from the servlet.
238    */
239   boolean inheritEncoders() default true;
240
241   /**
242    * Same as {@link RestResource#properties()}, except defines property values by default when this method is called.
243    * <p>
244    * This is equivalent to simply calling <code>res.addProperties()</code> in the Java method, but is provided for convenience.
245    */
246   Property[] properties() default {};
247
248   /**
249    * Appends the specified bean filters to all serializers and parsers used by this method.
250    */
251   Class<?>[] beanFilters() default {};
252
253   /**
254    * Appends the specified POJO swaps to all serializers and parsers used by this method.
255    */
256   Class<?>[] pojoSwaps() default {};
257
258   /**
259    * Specifies default values for request headers.
260    * <p>
261    * Strings are of the format <js>"Header-Name: header-value"</js>.
262    * <p>
263    * Affects values returned by {@link RestRequest#getHeader(String)} when the header is not present on the request.
264    * <p>
265    * The most useful reason for this annotation is to provide a default <code>Accept</code> header when one is not specified
266    *    so that a particular default {@link Serializer} is picked.
267    * <p>
268    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
269    * <p>
270    * Header values specified at the method level override header values specified at the servlet level.
271    *
272    * <h5 class='section'>Example:</h5>
273    * <p class='bcode'>
274    *    <jc>// Assume "text/json" Accept value when Accept not specified</jc>
275    *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/*"</js>, defaultRequestHeaders={<js>"Accept: text/json"</js>})
276    *    <jk>public</jk> String doGet() {
277    *       ...
278    *    }
279    * </p>
280    */
281   String[] defaultRequestHeaders() default {};
282
283   /**
284    * Optional summary for the exposed API.
285    * <p>
286    * This summary is used in the following locations:
287    * <ul class='spaced-list'>
288    *    <li>The value returned by {@link RestRequest#getMethodSummary()}.
289    *    <li>The <js>"$R{methodSummary}"</js> variable.
290    *    <li>The summary of the method in the Swagger page.
291    * </ul>
292    * <p>
293    * The default value pulls the description from the <code>(className.?)[javaMethodName].summary</code> entry in the servlet resource bundle.
294    *    (e.g. <js>"MyClass.myMethod.summary = foo"</js> or <js>"myMethod.summary = foo"</js>).
295    * <p>
296    * This field value can contain variables (e.g. "$L{my.localized.variable}").
297    * <p>
298    * Corresponds to the swagger field <code>/paths/{path}/{method}/summary</code>.
299    */
300   String summary() default "";
301
302   /**
303    * Optional description for the exposed API.
304    * <p>
305    * This description is used in the following locations:
306    * <ul class='spaced-list'>
307    *    <li>The value returned by {@link RestRequest#getMethodDescription()}.
308    *    <li>The <js>"$R{methodDescription}"</js> variable.
309    *    <li>The description of the method in the Swagger page.
310    * </ul>
311    * <p>
312    * The default value pulls the description from the <code>(className.?)[javaMethodName].description</code> entry in the servlet resource bundle.
313    *    (e.g. <js>"MyClass.myMethod.description = foo"</js> or <js>"myMethod.description = foo"</js>).
314    * <p>
315    * This field value can contain variables (e.g. "$L{my.localized.variable}").
316    * <p>
317    * Corresponds to the swagger field <code>/paths/{path}/{method}/description</code>.
318    */
319   String description() default "";
320
321   /**
322    * Optional external documentation information for the exposed API.
323    * <p>
324    * Used to populate the Swagger external documentation field.
325    * <p>
326    * A simplified JSON string with the following fields:
327    * <p class='bcode'>
328    *    {
329    *       description: string,
330    *       url: string
331    *    }
332    * </p>
333    * <p>
334    * The default value pulls the description from the <code>(className.?)[javaMethodName].externalDocs</code> entry in the servlet resource bundle.
335    *    (e.g. <js>"MyClass.myMethod.externalDocs = {url:'http://juneau.apache.org'}"</js> or <js>"myMethod.externalDocs = {url:'http://juneau.apache.org'}"</js>).
336    *
337    * <h5 class='section'>Example:</h5>
338    * <p class='bcode'>
339    *    <ja>@RestMethod</ja>(externalDocs=<js>"{url:'http://juneau.apache.org'}"</js>)
340    * </p>
341    * <p>
342    * This field can contain variables (e.g. "$L{my.localized.variable}").
343    * <p>
344    * Corresponds to the swagger field <code>/paths/{path}/{method}/externalDocs</code>.
345    */
346   String externalDocs() default "";
347
348   /**
349    * Optional tagging information for the exposed API.
350    * <p>
351    * Used to populate the Swagger tags field.
352    * <p>
353    * A comma-delimited list of tags for API documentation control.
354    * Tags can be used for logical grouping of operations by resources or any other qualifier.
355    * <p>
356    * The default value pulls the description from the <code>(className.?)[javaMethodName].tags</code> entry in the servlet resource bundle.
357    *    (e.g. <js>"MyClass.myMethod.tags = foo,bar"</js> or <js>"myMethod.tags = foo,bar"</js>).
358    *
359    * <h5 class='section'>Example:</h5>

389   /**
390    * Optional parameter descriptions.
391    * <p>
392    * This annotation is provided for documentation purposes and is used to populate the method <js>"parameters"</js> column

462   /**
463    * Specifies whether this method can be called based on the client version.
464    * <p>
465    * The client version is identified via the HTTP request header identified by {@link RestResource#clientVersionHeader()} which
466    *    by default is <js>"X-Client-Version"</js>.
467    * <p>
468    * This is a specialized kind of {@link RestMatcher} that allows you to invoke different Java methods for the same method/path based
469    *    on the client version.
470    * <p>
471    * The format of the client version range is similar to that of OSGi versions.
472    * <p>
473    * In the following example, the Java methods are mapped to the same HTTP method and URL <js>"/foobar"</js>.
474    * <p class='bcode'>
475    *    <jc>// Call this method if X-Client-Version is at least 2.0.
476    *    // Note that this also matches 2.0.1.</jc>
477    *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foobar"</js>, clientVersion=<js>"2.0"</js>)
478    *    <jk>public</jk> Object method1() {
479    *       ...
480    *    }
481    *
482    *    <jc>// Call this method if X-Client-Version is at least 1.1, but less than 2.0.</jc>
483    *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foobar"</js>, clientVersion=<js>"[1.1,2.0)"</js>)
484    *    <jk>public</jk> Object method2() {
485    *       ...
486    *    }
487    *
488    *    <jc>// Call this method if X-Client-Version is less than 1.1.</jc>
489    *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foobar"</js>, clientVersion=<js>"[0,1.1)"</js>)
490    *    <jk>public</jk> Object method3() {
491    *       ...
492    *    }
493    * </p>
494    * <p>
495    * It's common to combine the client version with transforms that will convert new POJOs into older POJOs for backwards compatability.
496    * <p class='bcode'>
497    *    <jc>// Call this method if X-Client-Version is at least 2.0.</jc>
498    *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foobar"</js>, clientVersion=<js>"2.0"</js>)
499    *    <jk>public</jk> NewPojo newMethod() {
500    *       ...
501    *    }
502    *
503    *    <jc>// Call this method if X-Client-Version is at least 1.1, but less than 2.0.</jc>
504    *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foobar"</js>, clientVersion=<js>"[1.1,2.0)"</js>, transforms={NewToOldPojoSwap.<jk>class</jk>})
505    *    <jk>public</jk> NewPojo oldMethod() {
506    *       <jk>return</jk> newMethod()
507    *    }
508    * <p>
509    * Note that in the previous example, we're returning the exact same POJO, but using a transform to convert it into an older form.
510    * The old method could also just return back a completely different object.
511    * The range can be any of the following:
512    * <ul>
513    *    <li><js>"[0,1.0)"</js> = Less than 1.0.  1.0 and 1.0.0 does not match.
514    *    <li><js>"[0,1.0]"</js> = Less than or equal to 1.0.  Note that 1.0.1 will match.
515    *    <li><js>"1.0"</js> = At least 1.0.  1.0 and 2.0 will match.
516    * </ul>
517    */
