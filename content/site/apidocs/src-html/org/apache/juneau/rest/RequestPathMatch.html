024/**
025 * Contains information about the matched path on the HTTP request.
026 *
027 * <p>
028 * Provides access to the matched path variables and path match remainder.
029 */
030@SuppressWarnings("unchecked")
031public class RequestPathMatch extends TreeMap<String,String> {
032   private static final long serialVersionUID = 1L;
033
034   private HttpPartParser parser;
035   private BeanSession beanSession;
036   private String remainder;
037
038   RequestPathMatch() {
039      super(String.CASE_INSENSITIVE_ORDER);
040   }
041
042   RequestPathMatch setParser(HttpPartParser parser) {
043      this.parser = parser;
044      return this;
045   }
046
047   RequestPathMatch setBeanSession(BeanSession beanSession) {
048      this.beanSession = beanSession;
049      return this;
050   }
051
052   RequestPathMatch setRemainder(String remainder) {
053      this.remainder = remainder;
054      return this;
055   }
056
057   /**
058    * Sets a request query parameter value.
059    *
060    * @param name The parameter name.
061    * @param value The parameter value.
062    */
063   public void put(String name, Object value) {
064      super.put(name, value.toString());
065   }
066
067   /**
068    * Returns the specified path parameter converted to a POJO.
069    *
070    * <p>
071    * The type can be any POJO type convertible from a <code>String</code> (See <a class="doclink"
072    * href="package-summary.html#PojosConvertibleFromString">POJOs Convertible From Strings</a>).
073    *
074    * <h5 class='section'>Examples:</h5>
075    * <p class='bcode'>
076    *    <jc>// Parse into an integer.</jc>
077    *    <jk>int</jk> myparam = req.getPathParameter(<js>"myparam"</js>, <jk>int</jk>.<jk>class</jk>);
078    *
079    *    <jc>// Parse into an int array.</jc>
080    *    <jk>int</jk>[] myparam = req.getPathParameter(<js>"myparam"</js>, <jk>int</jk>[].<jk>class</jk>);
081
082    *    <jc>// Parse into a bean.</jc>
083    *    MyBean myparam = req.getPathParameter(<js>"myparam"</js>, MyBean.<jk>class</jk>);
084    *
085    *    <jc>// Parse into a linked-list of objects.</jc>
086    *    List myparam = req.getPathParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>);
087    *
088    *    <jc>// Parse into a map of object keys/values.</jc>
089    *    Map myparam = req.getPathParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>);
090    * </p>
091    *
092    * @param name The attribute name.
093    * @param type The class type to convert the attribute value to.
094    * @param <T> The class type to convert the attribute value to.
095    * @return The attribute value converted to the specified class type.
096    * @throws ParseException
097    */
098   public <T> T get(String name, Class<T> type) throws ParseException {
099      return parse(name, beanSession.getClassMeta(type));
100   }
101
102   /**
103    * Returns the specified path parameter converted to a POJO.
104    *
105    * <p>
106    * The type can be any POJO type convertible from a <code>String</code> (See <a class="doclink"
107    * href="package-summary.html#PojosConvertibleFromString">POJOs Convertible From Strings</a>).
108    *
109    * <p>
110    * Use this method if you want to parse into a parameterized <code>Map</code>/<code>Collection</code> object.
111    *
112    * <h5 class='section'>Examples:</h5>
113    * <p class='bcode'>
114    *    <jc>// Parse into a linked-list of strings.</jc>
115    *    List&lt;String&gt; myparam = req.getPathParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
116    *
117    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
118    *    List&lt;List&lt;String&gt;&gt; myparam = req.getPathParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
119    *
120    *    <jc>// Parse into a map of string keys/values.</jc>
121    *    Map&lt;String,String&gt; myparam = req.getPathParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
122    *
123    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
124    *    Map&lt;String,List&lt;MyBean&gt;&gt; myparam = req.getPathParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
125    * </p>
126    *
127    * @param name The attribute name.
128    * @param type
129    *    The type of object to create.
130    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType},
131    *    {@link GenericArrayType}
132    * @param args
133    *    The type arguments of the class if it's a collection or map.
134    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType},
135    *    {@link GenericArrayType}
136    *    <br>Ignored if the main type is not a map or collection.
137    * @param <T> The class type to convert the attribute value to.
138    * @return The attribute value converted to the specified class type.
139    * @throws ParseException
140    */
141   public <T> T get(String name, Type type, Type...args) throws ParseException {
142      return (T)parse(name, beanSession.getClassMeta(type, args));
143   }
144
145   /* Workhorse method */
146   <T> T parse(String name, ClassMeta<T> cm) throws ParseException {
147      Object attr = get(name);
148      T t = null;
149      if (attr != null)
150         t = parser.parse(HttpPartType.PATH, attr.toString(), cm);
151      if (t == null && cm.isPrimitive())
152         return cm.getPrimitiveDefault();
153      return t;
154   }
155
156   /**
157    * Returns the decoded remainder of the URL following any path pattern matches.
158    *
159    * <p>
160    * The behavior of path remainder is shown below given the path pattern "/foo/*":
161    * <table class='styled'>
162    *    <tr>
163    *       <th>URL</th>
164    *       <th>Path Remainder</th>
165    *    </tr>
166    *    <tr>
167    *       <td><code>/foo</code></td>
168    *       <td><jk>null</jk></td>
169    *    </tr>
170    *    <tr>
171    *       <td><code>/foo/</code></td>
172    *       <td><js>""</js></td>
173    *    </tr>
174    *    <tr>
175    *       <td><code>/foo//</code></td>
176    *       <td><js>"/"</js></td>
177    *    </tr>
178    *    <tr>
179    *       <td><code>/foo///</code></td>
180    *       <td><js>"//"</js></td>
181    *    </tr>
182    *    <tr>
183    *       <td><code>/foo/a/b</code></td>
184    *       <td><js>"a/b"</js></td>
185    *    </tr>
186    *    <tr>
187    *       <td><code>/foo//a/b/</code></td>
188    *       <td><js>"/a/b/"</js></td>
189    *    </tr>
190    *    <tr>
191    *       <td><code>/foo/a%2Fb</code></td>
192    *       <td><js>"a/b"</js></td>
193    *    </tr>
194    * </table>
195    *
196    * <h5 class='section'>Example:</h5>
197    * <p class='bcode'>
198    *    <jc>// REST method</jc>
199    *    <ja>@RestMethod</ja>(name=<jsf>GET</jsf>,path=<js>"/foo/{bar}/*"</js>)
200    *    <jk>public</jk> String doGetById(RequestPathParams pathParams, <jk>int</jk> bar) {
201    *       <jk>return</jk> pathParams.getRemainder();
202    *    }
203    *
204    *    <jc>// Prints "path/remainder"</jc>
205    *    <jk>new</jk> RestCall(servletPath + <js>"/foo/123/path/remainder"</js>).connect();
206    * </p>
207    *
208    * @return The path remainder string.
209    */
210   public String getRemainder() {
211      return urlDecode(remainder);
212   }
213
214   /**
215    * Same as {@link #getRemainder()} but doesn't decode characters.
216    *
217    * @return The un-decoded path remainder.
218    */
219   public String getRemainderUndecoded() {
220      return remainder;
221   }
