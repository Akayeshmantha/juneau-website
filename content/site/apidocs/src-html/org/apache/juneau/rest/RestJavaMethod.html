115   private static final class Builder  {
116      String httpMethod, defaultCharset, description, tags, summary, externalDocs;
117      UrlPathPattern pathPattern;
118      RestParam[] params;
119      RestGuard[] guards;
120      RestMatcher[] optionalMatchers, requiredMatchers;
121      RestConverter[] converters;
122      SerializerGroup serializers;
123      ParserGroup parsers;
124      EncoderGroup encoders;
125      HttpPartParser partParser;
126      HttpPartSerializer partSerializer;
127      BeanContext beanContext;
128      ObjectMap properties;
129      Map<String,Object> defaultRequestHeaders, defaultQuery, defaultFormData;
130      boolean deprecated;
131      long maxInput;
132      Integer priority;
133      org.apache.juneau.rest.annotation.Parameter[] parameters;
134      Response[] responses;
135      Map<String,Widget> widgets;
136      List<MediaType> supportedAcceptTypes, supportedContentTypes;
137
138      Builder(Object servlet, java.lang.reflect.Method method, RestContext context) throws RestServletException {
139         String sig = method.getDeclaringClass().getName() + '.' + method.getName();
140
141         try {
142
143            RestMethod m = method.getAnnotation(RestMethod.class);
144            if (m == null)
145               throw new RestServletException("@RestMethod annotation not found on method ''{0}''", sig);
146            
147            VarResolver vr = context.getVarResolver();
148
149            if (! m.description().isEmpty())
150               description = m.description();
151            MethodSwagger sm = m.swagger();
152            if (! sm.tags().isEmpty())
153               tags = sm.tags();
154            if (! m.summary().isEmpty())
155               summary = m.summary();
156            if (! sm.externalDocs().isEmpty())
157               externalDocs = sm.externalDocs();
158            deprecated = sm.deprecated();
159            parameters = sm.parameters();
160            responses = sm.responses();
161            serializers = context.getSerializers();
162            parsers = context.getParsers();
163            partSerializer = context.getPartSerializer();
164            partParser = context.getPartParser();
165            beanContext = context.getBeanContext();
166            encoders = context.getEncoders();
167            properties = new ObjectMap().setInner(context.getProperties());
168            defaultCharset = context.getDefaultCharset();
169            maxInput = context.getMaxInput();
170
171            if (! m.defaultCharset().isEmpty())
172               defaultCharset = vr.resolve(m.defaultCharset());
173            if (! m.maxInput().isEmpty())
174               maxInput = StringUtils.parseLongWithSuffix(vr.resolve(m.maxInput()));
175
176            HtmlDocBuilder hdb = new HtmlDocBuilder(properties);
177
178            HtmlDoc hd = m.htmldoc();
179            hdb.process(hd);
180
181            widgets = new HashMap<>(context.getWidgets());
182            for (Class<? extends Widget> wc : hd.widgets()) {
183               Widget w = beanContext.newInstance(Widget.class, wc);
184               widgets.put(w.getName(), w);
185               hdb.script("INHERIT", "$W{"+w.getName()+".script}");
186               hdb.style("INHERIT", "$W{"+w.getName()+".style}");
187            }
188
189            ASet<String> inherit = new ASet<String>().appendAll(StringUtils.split(m.inherit()));
190            if (inherit.contains("*")) 
191               inherit.appendAll("SERIALIZERS","PARSERS","TRANSFORMS","PROPERTIES","ENCODERS");
192
193            SerializerGroupBuilder sgb = null;
194            ParserGroupBuilder pgb = null;
195            ParserBuilder uepb = null;
196            BeanContextBuilder bcb = null;
197            PropertyStore cps = context.getPropertyStore();
198
199            if (m.serializers().length > 0 || m.parsers().length > 0 || m.properties().length > 0 || m.flags().length > 0
200                  || m.beanFilters().length > 0 || m.pojoSwaps().length > 0 || m.bpi().length > 0
201                  || m.bpx().length > 0) {
202               sgb = SerializerGroup.create();
203               pgb = ParserGroup.create();
204               uepb = Parser.create();
205               bcb = beanContext.builder();
206
207               if (inherit.contains("SERIALIZERS") || m.serializers().length == 0)
208                  sgb.append(cps.getArrayProperty(REST_serializers, Object.class));
209
210               if (inherit.contains("PARSERS") || m.parsers().length == 0)
211                  pgb.append(cps.getArrayProperty(REST_parsers, Object.class));
212            }
213
214            httpMethod = m.name().toUpperCase(Locale.ENGLISH);
215            if (httpMethod.equals("") && method.getName().startsWith("do"))
216               httpMethod = method.getName().substring(2).toUpperCase(Locale.ENGLISH);
217            if (httpMethod.equals(""))
218               httpMethod = "GET";
219            if (httpMethod.equals("METHOD"))
220               httpMethod = "*";
221
222            priority = m.priority();
223
224            String p = m.path();
225            converters = new RestConverter[m.converters().length];
226            for (int i = 0; i < converters.length; i++)
227               converters[i] = beanContext.newInstance(RestConverter.class, m.converters()[i]);
228
229            guards = new RestGuard[m.guards().length];
230            for (int i = 0; i < guards.length; i++)
231               guards[i] = beanContext.newInstance(RestGuard.class, m.guards()[i]);
232
233            List<RestMatcher> optionalMatchers = new LinkedList<>(), requiredMatchers = new LinkedList<>();
234            for (int i = 0; i < m.matchers().length; i++) {
235               Class<? extends RestMatcher> c = m.matchers()[i];
236               RestMatcher matcher = beanContext.newInstance(RestMatcher.class, c, true, servlet, method);
237               if (matcher.mustMatch())
238                  requiredMatchers.add(matcher);
239               else
240                  optionalMatchers.add(matcher);
241            }
242            if (! m.clientVersion().isEmpty())
243               requiredMatchers.add(new ClientVersionMatcher(context.getClientVersionHeader(), method));
244
245            this.requiredMatchers = requiredMatchers.toArray(new RestMatcher[requiredMatchers.size()]);
246            this.optionalMatchers = optionalMatchers.toArray(new RestMatcher[optionalMatchers.size()]);
247
248            PropertyStore ps = context.getPropertyStore();
249            if (! inherit.contains("TRANSFORMS"))
250               ps = ps.builder().set(BEAN_beanFilters, null).set(BEAN_pojoSwaps, null).build();
251            
252            if (sgb != null) {
253               sgb.append(m.serializers());
254            
255               if (! inherit.contains("PROPERTIES"))
256                  sgb.beanFilters((Object[])ps.getClassArrayProperty(BEAN_beanFilters)).pojoSwaps(ps.getClassArrayProperty(BEAN_pojoSwaps));
257               else
258                  sgb.apply(ps);
259               for (Property p1 : m.properties())
260                  sgb.set(p1.name(), p1.value());
261               for (String p1 : m.flags())
262                  sgb.set(p1, true);
263               if (m.bpi().length > 0) {
264                  Map<String,String> bpiMap = new LinkedHashMap<>();
265                  for (String s : m.bpi()) {
266                     for (String s2 : split(s, ';')) {
267                        int i = s2.indexOf(':');
268                        if (i == -1)
269                           throw new RestServletException(
270                              "Invalid format for @RestMethod.bpi() on method ''{0}''.  Must be in the format \"ClassName: comma-delimited-tokens\".  \nValue: {1}", sig, s);
271                        bpiMap.put(s2.substring(0, i).trim(), s2.substring(i+1).trim());
272                     }
273                  }
274                  sgb.includeProperties(bpiMap);
275               }
276               if (m.bpx().length > 0) {
277                  Map<String,String> bpxMap = new LinkedHashMap<>();
278                  for (String s : m.bpx()) {
279                     for (String s2 : split(s, ';')) {
280                        int i = s2.indexOf(':');
281                        if (i == -1)
282                           throw new RestServletException(
283                              "Invalid format for @RestMethod.bpx() on method ''{0}''.  Must be in the format \"ClassName: comma-delimited-tokens\".  \nValue: {1}", sig, s);
284                        bpxMap.put(s2.substring(0, i).trim(), s2.substring(i+1).trim());
285                     }
286                  }
287                  sgb.excludeProperties(bpxMap);
288               }
289               sgb.beanFilters((Object[])m.beanFilters());
290               sgb.pojoSwaps(m.pojoSwaps());
291            }
292
293            if (pgb != null) {
294               pgb.append(m.parsers());
295               if (! inherit.contains("PROPERTIES"))
296                  pgb.beanFilters((Object[])ps.getClassArrayProperty(BEAN_beanFilters)).pojoSwaps(ps.getClassArrayProperty(BEAN_pojoSwaps));
297               else
298                  pgb.apply(ps);
299               for (Property p1 : m.properties())
300                  pgb.set(p1.name(), p1.value());
301               for (String p1 : m.flags())
302                  pgb.set(p1, true);
303               pgb.beanFilters((Object[])m.beanFilters());
304               pgb.pojoSwaps(m.pojoSwaps());
305            }
306
307            if (uepb != null) {
308               uepb.apply(ps);
309               for (Property p1 : m.properties())
310                  uepb.set(p1.name(), p1.value());
311               for (String p1 : m.flags())
312                  uepb.set(p1, true);
313               uepb.beanFilters((Object[])m.beanFilters());
314               uepb.pojoSwaps(m.pojoSwaps());
315            }
316            
317            if (bcb != null) {
318               bcb.apply(ps);
319               for (Property p1 : m.properties())
320                  bcb.set(p1.name(), p1.value());
321               for (String p1 : m.flags())
322                  bcb.set(p1, true);
323               bcb.beanFilters((Object[])m.beanFilters());
324               bcb.pojoSwaps(m.pojoSwaps());
325            }
326            
327            if (m.properties().length > 0 || m.flags().length > 0) {
328               properties = new ObjectMap().setInner(properties);
329               for (Property p1 : m.properties())
330                  properties.put(p1.name(), p1.value());
331               for (String p1 : m.flags())
332                  properties.put(p1, true);
333            }
334
335            if (m.encoders().length > 0) {
336               EncoderGroupBuilder g = EncoderGroup.create().append(IdentityEncoder.INSTANCE);
337               if (inherit.contains("ENCODERS"))
338                  g.append(encoders);
339
340               for (Class<? extends Encoder> c : m.encoders()) {
341                  try {
342                     g.append(c);
343                  } catch (Exception e) {
344                     throw new RestServletException(
345                        "Exception occurred while trying to instantiate Encoder on method ''{0}'': ''{1}''", sig, c.getSimpleName()).initCause(e);
346                  }
347               }
348               encoders = g.build();
349            }
350
351            defaultRequestHeaders = new TreeMap<>(String.CASE_INSENSITIVE_ORDER);
352            for (String s : m.defaultRequestHeaders()) {
353               String[] h = RestUtils.parseKeyValuePair(vr.resolve(s));
354               if (h == null)
355                  throw new RestServletException(
356                     "Invalid default request header specified on method ''{0}'': ''{1}''.  Must be in the format: ''name[:=]value''", sig, s);
357               defaultRequestHeaders.put(h[0], h[1]);
358            }
359
360            defaultQuery = new LinkedHashMap<>();
361            for (String s : m.defaultQuery()) {
362               String[] h = RestUtils.parseKeyValuePair(vr.resolve(s));
363               if (h == null)
364                  throw new RestServletException(
365                     "Invalid default query parameter specified on method ''{0}'': ''{1}''.  Must be in the format: ''name[:=]value''", sig, s);
366               defaultQuery.put(h[0], h[1]);
367            }
368
369            defaultFormData = new LinkedHashMap<>();
370            for (String s : m.defaultFormData()) {
371               String[] h = RestUtils.parseKeyValuePair(vr.resolve(s));
372               if (h == null)
373                  throw new RestServletException(
374                     "Invalid default form data parameter specified on method ''{0}'': ''{1}''.  Must be in the format: ''name[:=]value''", sig, s);
375               defaultFormData.put(h[0], h[1]);
376            }
377
378            Type[] pt = method.getGenericParameterTypes();
379            Annotation[][] pa = method.getParameterAnnotations();
380            for (int i = 0; i < pt.length; i++) {
381               for (Annotation a : pa[i]) {
382                  if (a instanceof Header) {
383                     Header h = (Header)a;
384                     if (! h.def().isEmpty())
385                        defaultRequestHeaders.put(firstNonEmpty(h.name(), h.value()), h.def());
386                  } else if (a instanceof Query) {
387                     Query q = (Query)a;
388                     if (! q.def().isEmpty())
389                        defaultQuery.put(firstNonEmpty(q.name(), q.value()), q.def());
390                  } else if (a instanceof FormData) {
391                     FormData f = (FormData)a;
392                     if (! f.def().isEmpty())
393                        defaultFormData.put(firstNonEmpty(f.name(), f.value()), f.def());
394                  }
395               }
396            }
397
398            pathPattern = new UrlPathPattern(p);
399
400            if (sgb != null) 
401               serializers = sgb.build();
402            if (pgb != null)
403               parsers = pgb.build();
404            if (uepb != null && partParser instanceof Parser) {
405               Parser pp = (Parser)partParser;
406               partParser = (HttpPartParser)pp.builder().apply(uepb.getPropertyStore()).build();
407            }
408            if (bcb != null)
409               beanContext = bcb.build();
410
411            supportedAcceptTypes = 
412               m.supportedAcceptTypes().length > 0 
413               ? Collections.unmodifiableList(new ArrayList<>(Arrays.asList(MediaType.forStrings(resolveVars(vr, m.supportedAcceptTypes()))))) 
414               : serializers.getSupportedMediaTypes();
415            supportedContentTypes =
416               m.supportedContentTypes().length > 0 
417               ? Collections.unmodifiableList(new ArrayList<>(Arrays.asList(MediaType.forStrings(resolveVars(vr, m.supportedContentTypes()))))) 
418               : parsers.getSupportedMediaTypes();
419               
420            params = context.findParams(method, pathPattern, false);
421
422            // Need this to access methods in anonymous inner classes.
423            method.setAccessible(true);
424         } catch (RestServletException e) {
425            throw e;
426         } catch (Exception e) {
427            throw new RestServletException("Exception occurred while initializing method ''{0}''", sig).initCause(e);
428         }
429      }
430   }

453   /**
454    * Returns the localized Swagger for this Java method.
455    */
456   Operation getSwaggerOperation(RestRequest req) throws ParseException {
457      Operation o = operation()
458         .operationId(method.getName())
459         .description(getDescription(req))
460         .tags(getTags(req))
461         .summary(getSummary(req))
462         .externalDocs(getExternalDocs(req))
463         .parameters(getParameters(req))
464         .responses(getResponses(req));
465
466      if (isDeprecated())
467         o.deprecated(true);
468
469      if (! parsers.getSupportedMediaTypes().equals(context.getParsers().getSupportedMediaTypes()))
470         o.consumes(parsers.getSupportedMediaTypes());
471
472      if (! serializers.getSupportedMediaTypes().equals(context.getSerializers().getSupportedMediaTypes()))
473         o.produces(serializers.getSupportedMediaTypes());
474
475      return o;
476   }

478   private Operation getSwaggerOperationFromFile(RestRequest req) {
479      Swagger s = req.getSwaggerFromFile();
480      if (s != null && s.getPaths() != null && s.getPaths().get(pathPattern.getPatternString()) != null)
481         return s.getPaths().get(pathPattern.getPatternString()).get(httpMethod);
482      return null;
483   }

521      VarResolverSession vr = req.getVarResolverSession();
522      JsonParser jp = JsonParser.DEFAULT;
523      try {
524         if (tags != null)
525            return jp.parse(vr.resolve(tags), ArrayList.class, String.class);
526         String tags = context.getMessages().findFirstString(req.getLocale(), method.getName() + ".tags");
527         if (tags != null)
528            return jp.parse(vr.resolve(tags), ArrayList.class, String.class);
529         Operation o = getSwaggerOperationFromFile(req);
530         if (o != null)
531            return o.getTags();
532         return null;
533      } catch (Exception e) {
534         throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
535      }

566   /**
567    * Returns the localized Swagger parameter information for this Java method.
568    */
569   private List<ParameterInfo> getParameters(RestRequest req) throws ParseException {
570      Operation o = getSwaggerOperationFromFile(req);
571      if (o != null && o.getParameters() != null)
572         return o.getParameters();
573
574      VarResolverSession vr = req.getVarResolverSession();
575      JsonParser jp = JsonParser.DEFAULT;
576      Map<String,ParameterInfo> m = new TreeMap<>();
577
578      // First parse @RestMethod.parameters() annotation.
579      for (org.apache.juneau.rest.annotation.Parameter v : parameters) {
580         String in = vr.resolve(v.in());
581         ParameterInfo p = parameterInfo(in, vr.resolve(v.name()));
582
583         if (! v.description().isEmpty())
584            p.description(vr.resolve(v.description()));
585         if (v.required())
586            p.required(v.required());
587
588         if ("body".equals(in)) {
589            if (! v.schema().isEmpty())
590               p.schema(jp.parse(vr.resolve(v.schema()), SchemaInfo.class));
591         } else {
592            if (v.allowEmptyValue())
593               p.allowEmptyValue(v.allowEmptyValue());
594            if (! v.collectionFormat().isEmpty())
595               p.collectionFormat(vr.resolve(v.collectionFormat()));
596            if (! v._default().isEmpty())
597               p._default(vr.resolve(v._default()));
598            if (! v.format().isEmpty())
599               p.format(vr.resolve(v.format()));
600            if (! v.items().isEmpty())
601               p.items(jp.parse(vr.resolve(v.items()), Items.class));
602            p.type(vr.resolve(v.type()));
603         }
604         m.put(p.getIn() + '.' + p.getName(), p);
605      }
606
607      // Next, look in resource bundle.
608      String prefix = method.getName() + ".req";
609      for (String key : context.getMessages().keySet(prefix)) {
610         if (key.length() > prefix.length()) {
611            String value = vr.resolve(context.getMessages().getString(key));
612            String[] parts = key.substring(prefix.length() + 1).split("\\.");
613            String in = parts[0], name, field;
614            boolean isBody = "body".equals(in);
615            if (parts.length == (isBody ? 2 : 3)) {
616               if ("body".equals(in)) {
617                  name = null;
618                  field = parts[1];
619               } else {
620                  name = parts[1];
621                  field = parts[2];
622               }
623               String k2 = in + '.' + name;
624               ParameterInfo p = m.get(k2);
625               if (p == null) {
626                  p = parameterInfoStrict(in, name);
627                  m.put(k2, p);
628               }
629
630               if (field.equals("description"))
631                  p.description(value);
632               else if (field.equals("required"))

647                  else if (field.equals("items"))
648                     p.items(jp.parse(value, Items.class));
649                  else if (field.equals("type"))
650                     p.type(value);
651               }
652            } else {

654            }
655         }
656      }
657

659      for (RestParam mp : this.params) {

662            String k2 = in.toString() + '.' + (in == RestParamType.BODY ? null : mp.getName());
663            ParameterInfo p = m.get(k2);
664            if (p == null) {
665               p = parameterInfoStrict(in.toString(), mp.getName());
666               m.put(k2, p);
667            }
668         }
669      }
670
671      if (m.isEmpty())
672         return null;
673      return new ArrayList<>(m.values());
674   }

781   /**
782    * Workhorse method.
783    *
784    * @param pathInfo The value of {@link HttpServletRequest#getPathInfo()} (sorta)
785    * @return The HTTP response code.
786    */
787   int invoke(String pathInfo, RestRequest req, RestResponse res) throws RestException {
788
789      String[] patternVals = pathPattern.match(pathInfo);
790      if (patternVals == null)
791         return SC_NOT_FOUND;
792
793      String remainder = null;
794      if (patternVals.length > pathPattern.getVars().length)
795         remainder = patternVals[pathPattern.getVars().length];
796      for (int i = 0; i < pathPattern.getVars().length; i++)
797         req.getPathMatch().put(pathPattern.getVars()[i], patternVals[i]);
798      req.getPathMatch().setRemainder(remainder);
799
800      ObjectMap requestProperties = new ResolvingObjectMap(req.getVarResolverSession()).setInner(properties);
801
802      req.init(this, requestProperties);
803      res.init(this, requestProperties);
804
805      // Class-level guards
806      for (RestGuard guard : context.getGuards())
807         if (! guard.guard(req, res))
808            return SC_UNAUTHORIZED;
809
810      // If the method implements matchers, test them.
811      for (RestMatcher m : requiredMatchers)
812         if (! m.matches(req))
813            return SC_PRECONDITION_FAILED;
814      if (optionalMatchers.length > 0) {
815         boolean matches = false;
816         for (RestMatcher m : optionalMatchers)
817            matches |= m.matches(req);
818         if (! matches)
819            return SC_PRECONDITION_FAILED;
820      }
821
822      context.preCall(req, res);
823
824      Object[] args = new Object[params.length];
825      for (int i = 0; i < params.length; i++) {
826         try {
827            args[i] = params[i].resolve(req, res);
828         } catch (RestException e) {
829            throw e;
830         } catch (Exception e) {
831            throw new RestException(SC_BAD_REQUEST,
832               "Invalid data conversion.  Could not convert {0} ''{1}'' to type ''{2}'' on method ''{3}.{4}''.",
833               params[i].getParamType().name(), params[i].getName(), params[i].getType(), method.getDeclaringClass().getName(), method.getName()
834            ).initCause(e);
835         }
836      }
837
838      try {
839
840         for (RestGuard guard : guards)
841            if (! guard.guard(req, res))
842               return SC_OK;
843
844         Object output = method.invoke(context.getResource(), args);
845         if (! method.getReturnType().equals(Void.TYPE))
846            if (output != null || ! res.getOutputStreamCalled())
847               res.setOutput(output);
848
849         context.postCall(req, res);
850
851         if (res.hasOutput()) {
852            output = res.getOutput();
853            for (RestConverter converter : converters)
854               output = converter.convert(req, output, beanContext.getClassMetaForObject(output));
855            res.setOutput(output);
856         }
857      } catch (IllegalArgumentException e) {
858         throw new RestException(SC_BAD_REQUEST,
859            "Invalid argument type passed to the following method: ''{0}''.\n\tArgument types: {1}",
860            method.toString(), getReadableClassNames(args)
861         ).initCause(e);
862      } catch (InvocationTargetException e) {
863         Throwable e2 = e.getTargetException();    // Get the throwable thrown from the doX() method.
864         if (e2 instanceof RestException)
865            throw (RestException)e2;
866         if (e2 instanceof ParseException)
867            throw new RestException(SC_BAD_REQUEST, e2);
868         if (e2 instanceof InvalidDataConversionException)
869            throw new RestException(SC_BAD_REQUEST, e2);
870         throw new RestException(SC_INTERNAL_SERVER_ERROR, e2);
871      } catch (RestException e) {
872         throw e;
873      } catch (Exception e) {
874         throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
875      }
876      return SC_OK;
877   }

889   public int compareTo(RestJavaMethod o) {
890      int c;
891
892      c = priority.compareTo(o.priority);
893      if (c != 0)
894         return c;
895
896      c = pathPattern.compareTo(o.pathPattern);
897      if (c != 0)
898         return c;

900      c = compare(o.requiredMatchers.length, requiredMatchers.length);
901      if (c != 0)
902         return c;
903
904      c = compare(o.optionalMatchers.length, optionalMatchers.length);
905      if (c != 0)
906         return c;
907
908      c = compare(o.guards.length, guards.length);
909      if (c != 0)
910         return c;
911
912      return 0;
913   }
