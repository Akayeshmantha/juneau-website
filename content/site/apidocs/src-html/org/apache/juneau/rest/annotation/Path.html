013package org.apache.juneau.rest.annotation;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019
020/**
021 * Annotation that can be applied to a parameter of a {@link RestMethod} annotated method
022 *    to identify it as a variable in a URL path pattern converted to a POJO.
023 *
024 * <h5 class='section'>Example:</h5>
025 * <p class='bcode'>
026 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/myurl/{foo}/{bar}/{baz}/*"</js>)
027 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res,
028 *          <ja>@Path</ja> String foo, <ja>@Path</ja> <jk>int</jk> bar, <ja>@Path</ja> UUID baz) {
029 *       ...
030 *    }
031 * </p>
032 * <p>
033 * The <ja>@Path</ja> annotation is optional if the parameters are specified immediately
034 *    following the <code>RestRequest</code> and <code>RestResponse</code> parameters,
035 *    and are specified in the same order as the variables in the URL path pattern.
036 * The following example is equivalent to the previous example.
037 * </p>
038 * <p class='bcode'>
039 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/myurl/{foo}/{bar}/{baz}/*"</js>)
040 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res,
041 *          String foo, <jk>int</jk> bar, UUID baz) {
042 *       ...
043 *    }
044 * </p>
045 * <p>
046 * If the order of parameters is not the default order shown above, the
047 *    attribute names must be specified (since parameter names are lost during compilation).
048 * The following example is equivalent to the previous example, except
049 *    the parameter order has been switched, requiring the use of the <ja>@Path</ja>
050 *    annotations.
051 * <p>
052 * <p class='bcode'>
053 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/myurl/{foo}/{bar}/{baz}/*"</js>)
054 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res,
055 *          <ja>@Path</ja>(<js>"baz"</js>) UUID baz, <ja>@Path</ja>(<js>"foo"</js>) String foo, <ja>@Path</ja>(<js>"bar"</js>) <jk>int</jk> bar) {
056 *       ...
057 *    }
058 * </p>
059 * You can also use <code>{#}</code> notation to specify path parameters without specifying names.
060 * <p>
061 * <p class='bcode'>
062 *    <ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/myurl/{0}/{1}/{2}/*"</js>)
063 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res,
064 *          <ja>@Path</ja> String foo, <ja>@Path</ja> <jk>int</jk> bar, <ja>@Path</ja> UUID baz) {
065 *       ...
066 *    }
067 * </p>
068 */
069@Documented
070@Target(PARAMETER)
071@Retention(RUNTIME)
072@Inherited
073public @interface Path {
074
075   /**
076    * URL path variable name.
077    * <p>
078    * Optional if the attributes are specified in the same order as in the URL path pattern.
079    */
080   String name() default "";
081
082   /**
083    * A synonym for {@link #name()}.
084    * <p>
085    * Allows you to use shortened notation if you're only specifying the name.
086    */
087   String value() default "";
088}
