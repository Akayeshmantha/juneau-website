033/**
034 * Used to denote that a class is a REST resource and to associate metadata on it.
035 *
036 * <p>
037 * Usually used on a subclass of {@link RestServlet}, but can be used to annotate any class that you want to expose as
038 * a REST resource.
039 *
040 * Refer to <a class='doclink' href='../package-summary.html#TOC'>org.apache.juneau.rest</a> doc for information on
041 * using this class.
042 */
043@Documented
044@Target(TYPE)
045@Retention(RUNTIME)
046@Inherited
047public @interface RestResource {
048
049   /**
050    * Identifies the location of the resource bundle for this class.
051    *
052    * <p>
053    * This annotation is used to provide localized messages for the following methods:
054    * <ul>
055    *    <li>{@link RestRequest#getMessage(String, Object...)}
056    *    <li>{@link RestContext#getMessages()}
057    * </ul>
058    *
059    * <p>
060    * Refer to the {@link MessageBundle} class for a description of the message key formats used in the properties file.
061    *
062    * <p>
063    * The value can be a relative path like <js>"nls/Messages"</js>, indicating to look for the resource bundle
064    * <js>"com.foo.sample.nls.Messages"</js> if the resource class is in <js>"com.foo.sample"</js>, or it can be an
065    * absolute path, like <js>"com.foo.sample.nls.Messages"</js>
066    */
067   String messages() default "";
068
069   /**
070    * Class-level guards.
071    *
072    * <p>
073    * Associates one or more {@link RestGuard RestGuards} with all REST methods defined in this class.
074    * These guards get called immediately before execution of any REST method in this class.
075    *
076    * <p>
077    * Typically, guards will be used for permissions checking on the user making the request, but it can also be used
078    * for other purposes like pre-call validation of a request.
079    *
080    * <p>
081    * The programmatic equivalent to this annotation are the {@link RestConfig#addGuards(Class...)}/
082    * {@link RestConfig#addGuards(RestGuard...)} methods.
083    */
084   Class<? extends RestGuard>[] guards() default {};
085
086   /**
087    * Class-level converters.
088    *
089    * <p>
090    * Associates one or more {@link RestConverter converters} with a resource class.
091    * These converters get called immediately after execution of the REST method in the same order specified in the
092    * annotation.
093    *
094    * <p>
095    * Can be used for performing post-processing on the response object before serialization.
096    *
097    * <p>
098    * Default converter implementations are provided in the <a class='doclink'
099    * href='../converters/package-summary.html#TOC'>org.apache.juneau.rest.converters</a> package.
100    *
101    * <p>
102    * The programmatic equivalent to this annotation are the {@link RestConfig#addConverters(Class...)}/
103    * {@link RestConfig#addConverters(RestConverter...)} methods.
104    */
105   Class<? extends RestConverter>[] converters() default {};
106
107   /**
108    * Class-level bean filters.
109    *
110    * <p>
111    * Shortcut to add bean filters to the bean contexts of the objects returned by the following methods:
112    * <ul>
113    *    <li>{@link RestContext#getBeanContext()}
114    *    <li>{@link RestContext#getSerializers()}
115    *    <li>{@link RestContext#getParsers()}
116    * </ul>
117    *
118    * <p>
119    * If the specified class is an instance of {@link BeanFilterBuilder}, then a filter built from that builder is added.
120    * Any other classes are wrapped in a {@link InterfaceBeanFilterBuilder} to indicate that subclasses should be
121    * treated as the specified class type.
122    *
123    * <p>
124    * The programmatic equivalent to this annotation is the {@link RestConfig#addBeanFilters(Class...)} method.
125    */
126   Class<?>[] beanFilters() default {};
127
128   /**
129    * Class-level POJO swaps.
130    *
131    * <p>
132    * Shortcut to add POJO swaps to the bean contexts of the objects returned by the following methods:
133    * <ul>
134    *    <li>{@link RestContext#getBeanContext()}
135    *    <li>{@link RestContext#getSerializers()}
136    *    <li>{@link RestContext#getParsers()}
137    * </ul>
138    *
139    * <p>
140    * If the specified class is an instance of {@link PojoSwap}, then that swap is added.
141    * Any other classes are wrapped in a {@link SurrogateSwap}.
142    *
143    * <p>
144    * The programmatic equivalent to this annotation is the {@link RestConfig#addPojoSwaps(Class...)} method.
145    */
146   Class<?>[] pojoSwaps() default {};
147
148   /**
149    * Class-level Java method parameter resolvers.
150    *
151    * <p>
152    * By default, the Juneau framework will automatically Java method parameters of various types (e.g.
153    * <code>RestRequest</code>, <code>Accept</code>, <code>Reader</code>).
154    * This annotation allows you to provide your own resolvers for your own class types that you want resolved.
155    *
156    * <p>
157    * For example, if you want to pass in instances of <code>MySpecialObject</code> to your Java method, define
158    * the following resolver:
159    * <p class='bcode'>
160    *    <jk>public class</jk> MyRestParam <jk>extends</jk> RestParam {
161    *
162    *       <jc>// Must have no-arg constructor!</jc>
163    *       <jk>public</jk> MyRestParam() {
164    *          <jc>// First two parameters help with Swagger doc generation.</jc>
165    *          <jk>super</jk>(<jsf>QUERY</jsf>, <js>"myparam"</js>, MySpecialObject.<jk>class</jk>);
166    *       }
167    *
168    *       <jc>// The method that creates our object.
169    *       // In this case, we're taking in a query parameter and converting it to our object.</jc>
170    *       <jk>public</jk> Object resolve(RestRequest req, RestResponse res) <jk>throws</jk> Exception {
171    *          <jk>return new</jk> MySpecialObject(req.getQuery().get(<js>"myparam"</js>));
172    *       }
173    *    }
174    * </p>
175    *
176    * <p>
177    * <b>Note:</b>{@link RestParam} classes must have no-arg constructors.
178    *
179    * <p>
180    * The programmatic equivalent to this annotation is the {@link RestConfig#addParamResolvers(Class...)} method.
181    */
182   Class<? extends RestParam>[] paramResolvers() default {};
183
184   /**
185    * Class-level properties.
186    *
187    * <p>
188    * Shortcut for specifying class-level properties on this servlet to the objects returned by the following methods:
189    * <ul>
190    *    <li>{@link RestContext#getBeanContext()}
191    *    <li>{@link RestContext#getSerializers()}
192    *    <li>{@link RestContext#getParsers()}
193    * </ul>
194    * <p>
195    * Any of the following property names can be specified:
196    * <ul>
197    *    <li>{@link RestContext}
198    *    <li>{@link BeanContext}
199    *    <li>{@link SerializerContext}
200    *    <li>{@link ParserContext}
201    *    <li>{@link JsonSerializerContext}
202    *    <li>{@link XmlSerializerContext}
203    *    <li>{@link XmlParserContext}
204    * </ul>
205    *
206    * <p>
207    * Property values will be converted to the appropriate type.
208    *
209    * <p>
210    * In some cases, properties can be overridden at runtime through the
211    * {@link RestResponse#setProperty(String, Object)} method or through a {@link Properties @Properties} annotated
212    * method parameter.
213    *
214    * <p>
215    * The programmatic equivalent to this annotation are the {@link RestConfig#setProperty(String, Object)}/
216    * {@link RestConfig#setProperties(java.util.Map)} methods.
217    */
218   Property[] properties() default {};
219
220   /**
221    * Shortcut for setting {@link #properties()} of simple boolean types.
222    *
223    * <p>
224    * Setting a flag is equivalent to setting the same property to <js>"true"</js>.
225    */
226   String[] flags() default {};
227
228   /**
229    * Specifies a list of {@link Serializer} classes to add to the list of serializers available for this servlet.
230    *
231    * <p>
232    * This annotation can only be used on {@link Serializer} classes that have no-arg constructors.
233    *
234    * <p>
235    * The programmatic equivalent to this annotation are the {@link RestConfig#addSerializers(Class...)}/
236    * {@link RestConfig#addSerializers(Serializer...)} methods.
237    */
238   Class<? extends Serializer>[] serializers() default {};
239
240   /**
241    * Specifies a list of {@link Parser} classes to add to the list of parsers available for this servlet.
242    *
243    * <p>
244    * This annotation can only be used on {@link Parser} classes that have no-arg constructors.
245    *
246    * <p>
247    * The programmatic equivalent to this annotation are the {@link RestConfig#addParsers(Class...)}/
248    * {@link RestConfig#addParsers(Parser...)} methods.
249    */
250   Class<? extends Parser>[] parsers() default {};
251
252   /**
253    * Specifies a list of {@link ResponseHandler} classes that know how to convert POJOs returned by REST methods or
254    * set via {@link RestResponse#setOutput(Object)} into appropriate HTTP responses.
255    *
256    * <p>
257    * See {@link ResponseHandler} for details.
258    *
259    * <p>
260    * The programmatic equivalent to this annotation are the {@link RestConfig#addResponseHandlers(Class...)}/
261    * {@link RestConfig#addResponseHandlers(ResponseHandler...)} methods.
262    */
263   Class<? extends ResponseHandler>[] responseHandlers() default {};
264
265   /**
266    * Specifies a list of {@link Encoder} to associate with this servlet.
267    *
268    * <p>
269    * These can be used to enable various kinds of compression (e.g. <js>"gzip"</js>) on requests and responses.
270    *
271    * <p>
272    * This annotation can only be used on {@link Encoder} classes that have no-arg constructors.
273    *
274    * <h5 class='section'>Example:</h5>
275    * <p class='bcode'>
276    *    <jc>// Servlet with automated support for GZIP compression</jc>
277    *    <ja>@RestResource</ja>(encoders={GzipEncoder.<jk>class</jk>})
278    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
279    *       ...
280    *    }
281    * </p>
282    *
283    * <p>
284    * The programmatic equivalent to this annotation are the {@link RestConfig#addEncoders(Class...)}/
285    * {@link RestConfig#addEncoders(Encoder...)} methods.
286    */
287   Class<? extends Encoder>[] encoders() default {};
288
289   /**
290    * Specifies default values for request headers.
291    *
292    * <p>
293    * Strings are of the format <js>"Header-Name: header-value"</js>.
294    *
295    * <p>
296    * Affects values returned by {@link RestRequest#getHeader(String)} when the header is not present on the request.
297    *
298    * <p>
299    * The most useful reason for this annotation is to provide a default <code>Accept</code> header when one is not
300    * specified so that a particular default {@link Serializer} is picked.
301    *
302    * <p>
303    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
304    *
305    * <h5 class='section'>Example:</h5>
306    * <p class='bcode'>
307    *    <jc>// Assume "text/json" Accept value when Accept not specified</jc>
308    *    <ja>@RestResource</ja>(defaultRequestHeaders={<js>"Accept: text/json"</js>})
309    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
310    *       ...
311    *    }
312    * </p>
313    *
314    * <p>
315    * The programmatic equivalent to this annotation are the {@link RestConfig#addDefaultRequestHeader(String, Object)}/
316    * {@link RestConfig#addDefaultRequestHeaders(String...)} methods.
317    */
318   String[] defaultRequestHeaders() default {};
319
320   /**
321    * Specifies default values for response headers.
322    *
323    * <p>
324    * Strings are of the format <js>"Header-Name: header-value"</js>.
325    *
326    * <p>
327    * This is equivalent to calling {@link RestResponse#setHeader(String, String)} programmatically in each of the Java
328    * methods.
329    *
330    * <p>
331    * The header value will not be set if the header value has already been specified (hence the 'default' in the name).
332    *
333    * <p>
334    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
335    *
336    * <h5 class='section'>Example:</h5>
337    * <p class='bcode'>
338    *    <jc>// Add a version header attribute to all responses</jc>
339    *    <ja>@RestResource</ja>(defaultResponseHeaders={<js>"X-Version: 1.0"</js>})
340    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
341    *       ...
342    *    }
343    * </p>
344    *
345    * <p>
346    * The programmatic equivalent to this annotation are the {@link RestConfig#addDefaultResponseHeader(String, Object)}/
347    * {@link RestConfig#addDefaultResponseHeaders(String...)} methods.
348    */
349   String[] defaultResponseHeaders() default {};

351   /**
352    * Defines children of this resource.
353    *
354    * <p>
355    * A REST child resource is simply another servlet that is initialized as part of the parent resource and has a
356    * servlet path directly under the parent servlet path.
357    * The main advantage to defining servlets as REST children is that you do not need to define them in the
358    * <code>web.xml</code> file of the web application.
359    * This can cut down on the number of entries that show up in the <code>web.xml</code> file if you are defining
360    * large numbers of servlets.
361    *
362    * <p>
363    * Child resources must specify a value for {@link #path()} that identifies the subpath of the child resource
364    * relative to the parent path.
365    *
366    * <p>
367    * It should be noted that servlets can be nested arbitrarily deep using this technique (i.e. children can also have
368    * children).
369    *
370    * <dl>
371    *    <dt>Servlet initialization:</dt>
372    *    <dd>
373    *       <p>
374    *          A child resource will be initialized immediately after the parent servlet is initialized.
375    *          The child resource receives the same servlet config as the parent resource.
376    *          This allows configuration information such as servlet initialization parameters to filter to child
377    *          resources.
378    *       </p>
379    *    </dd>
380    *    <dt>Runtime behavior:</dt>
381    *    <dd>
382    *       <p>
383    *          As a rule, methods defined on the <code>HttpServletRequest</code> object will behave as if the child
384    *          servlet were deployed as a top-level resource under the child's servlet path.
385    *          For example, the <code>getServletPath()</code> and <code>getPathInfo()</code> methods on the
386    *          <code>HttpServletRequest</code> object will behave as if the child resource were deployed using the
387    *          child's servlet path.
388    *          Therefore, the runtime behavior should be equivalent to deploying the child servlet in the
389    *          <code>web.xml</code> file of the web application.
390    *       </p>
391    *    </dd>
392    * </dl>
393    *
394    * <p>
395    * The programmatic equivalent to this annotation are the {@link RestConfig#addChildResource(String, Object)}/
396    * {@link RestConfig#addChildResources(Class...)}/{@link RestConfig#addChildResources(Object...)} methods.
397    */
398   Class<?>[] children() default {};
399
400   /**
401    * Identifies the URL subpath relative to the parent resource.
402    *
403    * <p>
404    * Typically, this annotation is only applicable to resources defined as children through the {@link #children()}
405    * annotation.
406    * However, it may be used in other ways (e.g. defining paths for top-level resources in microservices).
407    *
408    * <p>
409    * This annotation is ignored on top-level servlets (i.e. servlets defined in <code>web.xml</code> files).
410    * Therefore, implementers can optionally specify a path value for documentation purposes.
411    *
412    * <p>
413    * The programmatic equivalent to this annotation is the {@link RestConfig#setPath(String)} method.
414    */
415   String path() default "";

417   /**
418    * Optional site name.
419    *
420    * <p>
421    * The site name is intended to be a title that can be applied to the entire site.
422    *
423    * <p>
424    * This value can be retrieved programmatically through the {@link RestRequest#getSiteName()} method.
425    *
426    * <p>
427    * One possible use is if you want to add the same title to the top of all pages by defining a header on a
428    * common parent class like so:
429    * <p class='bcode'>
430    *    htmldoc=<ja>@HtmlDoc</ja>(
431    *       header={
432    *          <js>"&lt;h1&gt;$R{siteName}&lt;/h1&gt;"</js>,
433    *          <js>"&lt;h2&gt;$R{servletTitle}&lt;/h2&gt;"</js>
434    *       }
435    *    )
436    * </p>
437    *
438    * <p>
439    * This field can contain variables (e.g. <js>"$L{my.localized.variable}"</js>).
440    * <br>See {@link RestContext#getVarResolver()} for the list of supported variables.
441    *
442    * <p>
443    * The programmatic equivalent to this annotation is the {@link RestInfoProvider#getSiteName(RestRequest)} method.
444    */
445   String siteName() default "";

448    * Optional servlet title.

450    * <p>

458    * <p>
459    * This field can contain variables (e.g. <js>"$L{my.localized.variable}"</js>).
460    * <br>See {@link RestContext#getVarResolver()} for the list of supported variables.

462    * <p>
463    * Corresponds to the swagger field <code>/info/title</code>.

500    * This field can contain variables (e.g. <js>"$L{my.localized.variable}"</js>).

508   /**
509    * The favicon to use for HTML views.
510    *
511    * <p>
512    * The name is a path to an icon file located in either the classpath or working directory.
513    * The resulting favicon becomes available in the servlet via the URL <js>"[servlet-path]/favicon.ico"</js>.
514    *
515    * <p>
516    * If the file cannot be located, the request to <js>"[servlet-path]/favicon.ico"</js> will return
517    * {@link HttpServletResponse#SC_NOT_FOUND}.
518    *
519    * <h5 class='section'>Example:</h5>
520    * <p class='bcode'>
521    *    <jk>package</jk> com.foo.mypackage;
522    *
523    *    <ja>@RestResource</ja>(
524    *       favicon=<js>"mydocs/myicon.ico"</js>
525    *    )
526    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
527    *    }
528    * </p>
529    *
530    * <p>
531    * In this example, the servlet will attempt to find the <code>myicon.ico</code> file in the following ordered
532    * locations:
533    * <ol>
534    *    <li><code>com.foo.mypackage.mydocs</code> package.
535    *    <li><code>org.apache.juneau.rest.mydocs</code> package (since <code>RestServletDefault</code> is in
536    *       <code>org.apache.juneau.rest</code>).
537    *    <li><code>[working-dir]/mydocs</code> directory.
538    * </ol>
539    *
540    * <p>
541    * The programmatic equivalent to this annotation are the {@link RestConfig#setFavIcon(Object)}/
542    * {@link RestConfig#setFavIcon(Class, String)} methods.
543    */
544   String favicon() default "";
545
546   /**
547    * Defines paths and locations of statically served files.
548    *
549    * <p>
550    * This is a JSON map of paths to packages/directories located on either the classpath or working directory.
551    *
552    * <p>
553    * Mappings are cumulative from parent to child.  Child resources can override mappings made on parent resources.
554    *
555    * <p>
556    * If the file cannot be located, the request will return {@link HttpServletResponse#SC_NOT_FOUND}.
557    *
558    * <h5 class='section'>Example:</h5>
559    * <p class='bcode'>
560    *    <jk>package</jk> com.foo.mypackage;
561    *
562    *    <ja>@RestResource</ja>(
563    *       path=<js>"/myresource"</js>,
564    *       staticFiles=<js>"{htdocs:'docs'}"</js>
565    *    )
566    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
567    *    }
568    * </p>
569    *
570    * <p>
571    * In this example, given a GET request to <code>/myresource/htdocs/foobar.html</code>, the servlet will attempt to
572    * find the <code>foobar.html</code> file in the following ordered locations:
573    * <ol>
574    *    <li><code>com.foo.mypackage.docs</code> package.
575    *    <li><code>org.apache.juneau.rest.docs</code> package (since <code>RestServletDefault</code> is in
576    *       <code>org.apache.juneau.rest</code>).
577    *    <li><code>[working-dir]/docs</code> directory.
578    * </ol>
579    *
580    * <p>
581    * The programmatic equivalent to this annotation is the {@link RestConfig#addStaticFiles(Class, String)} method.
582    */
583   String staticFiles() default "";

600   /**
601    * Specifies the resolver class to use for resolving child resources by class name.
602    *
603    * <p>
604    * The default implementation simply instantiates the class using one of the following constructors:
605    * <ul>
