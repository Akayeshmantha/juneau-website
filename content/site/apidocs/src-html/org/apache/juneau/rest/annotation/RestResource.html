033/**
034 * Optionally used to associate metadata on an instance of {@link RestServlet}.
035 * <p>
036 * Refer to <a class='doclink' href='../package-summary.html#TOC'>org.apache.juneau.rest</a> doc for information on using this class.
037 */
038@Documented
039@Target(TYPE)
040@Retention(RUNTIME)
041@Inherited
042public @interface RestResource {
043
044   /**
045    * Identifies the location of the resource bundle for this class.
046    * <p>
047    * This annotation is used to provide localized messages for the following methods:
048    * <ul>
049    *    <li>{@link RestServlet#getMessage(java.util.Locale, String, Object...)}
050    *    <li>{@link RestServlet#getTitle(RestRequest)}
051    *    <li>{@link RestServlet#getDescription(RestRequest)}
052    * </ul>
053    * <p>
054    * Refer to the {@link MessageBundle} class for a description of the message key formats
055    *    used in the properties file.
056    * <p>
057    * The value can be a relative path like <js>"nls/Messages"</js>, indicating to look for the
058    *    resource bundle <js>"com.foo.sample.nls.Messages"</js> if the resource class
059    *    is in <js>"com.foo.sample"</js>, or it can be an absolute path, like <js>"com.foo.sample.nls.Messages"</js>
060    */
061   String messages() default "";
062
063   /**
064    * Class-level guards.
065    * <p>
066    * Associates one or more {@link RestGuard RestGuards} with all REST methods defined
067    *    in this class.
068    * These guards get called immediately before execution of any REST method in this class.
069    * <p>
070    * Typically, guards will be used for permissions checking on the user making the request,
071    *    but it can also be used for other purposes like pre-call validation of a request.
072    */
073   Class<? extends RestGuard>[] guards() default {};
074
075   /**
076    * Class-level converters.
077    * <p>
078    * Associates one or more {@link RestConverter converters} with a resource class.
079    * These converters get called immediately after execution of the REST method in the same
080    *    order specified in the annotation.
081    * <p>
082    * Can be used for performing post-processing on the response object before serialization.
083    * <p>
084    * Default converter implementations are provided in the <a class='doclink' href='../converters/package-summary.html#TOC'>org.apache.juneau.rest.converters</a> package.
085    */
086   Class<? extends RestConverter>[] converters() default {};
087
088   /**
089    * Class-level bean filters.
090    * <p>
091    * Shortcut to add bean filters to the bean contexts of the objects returned by the following methods:
092    * <ul>
093    *    <li>{@link RestServlet#getBeanContext()}
094    *    <li>{@link RestServlet#getSerializers()}
095    *    <li>{@link RestServlet#getParsers()}
096    * </ul>
097    * <p>
098    * If the specified class is an instance of {@link BeanFilterBuilder}, then a filter built from that builder is added.
099    * Any other classes are wrapped in a {@link InterfaceBeanFilterBuilder} to indicate that subclasses should
100    *    be treated as the specified class type.
101    */
102   Class<?>[] beanFilters() default {};
103
104   /**
105    * Class-level POJO swaps.
106    * <p>
107    * Shortcut to add POJO swaps to the bean contexts of the objects returned by the following methods:
108    * <ul>
109    *    <li>{@link RestServlet#getBeanContext()}
110    *    <li>{@link RestServlet#getSerializers()}
111    *    <li>{@link RestServlet#getParsers()}
112    * </ul>
113    * <p>
114    * If the specified class is an instance of {@link PojoSwap}, then that swap is added.
115    * Any other classes are wrapped in a {@link SurrogateSwap}.
116    */
117   Class<?>[] pojoSwaps() default {};
118
119   /**
120    * Class-level properties.
121    * <p>
122    * Shortcut for specifying class-level properties on this servlet to the objects returned by the following methods:
123    * <ul>
124    *    <li>{@link RestServlet#getBeanContext()}
125    *    <li>{@link RestServlet#getSerializers()}
126    *    <li>{@link RestServlet#getParsers()}
127    * </ul>
128    * <p>
129    * Any of the following property names can be specified:
130    * <ul>
131    *    <li>{@link RestServletContext}
132    *    <li>{@link BeanContext}
133    *    <li>{@link SerializerContext}
134    *    <li>{@link ParserContext}
135    *    <li>{@link JsonSerializerContext}
136    *    <li>{@link RdfSerializerContext}
137    *    <li>{@link RdfParserContext}
138    *    <li>{@link RdfCommonContext}
139    *    <li>{@link XmlSerializerContext}
140    *    <li>{@link XmlParserContext}
141    * </ul>
142    * <p>
143    * Property values will be converted to the appropriate type.
144    * <p>
145    * In some cases, properties can be overridden at runtime through the {@link RestResponse#setProperty(String, Object)} method
146    *    or through a {@link Properties @Properties} annotated method parameter.
147    */
148   Property[] properties() default {};
149
150   /**
151    * Specifies a list of {@link Serializer} classes to add to the list of serializers available for this servlet.

153    * This annotation can only be used on {@link Serializer} classes that have no-arg constructors.
154    */

157   /**

161    */
162   Class<? extends Parser>[] parsers() default {};
163
164   /**
165    * Specifies a list of {@link ResponseHandler} classes that know how to convert POJOs returned

168    * <p>
169    * See {@link ResponseHandler} for details.
170    */
171   Class<? extends ResponseHandler>[] responseHandlers() default {};

173   /**
174    * Specifies a list of {@link Encoder} to associate with this servlet.
175    * <p>
176    * These can be used to enable various kinds of compression (e.g. <js>"gzip"</js>) on requests and responses.
177    * <p>
178    * This annotation can only be used on {@link Encoder} classes that have no-arg constructors.
179    *
180    * <h5 class='section'>Example:</h5>
181    * <p class='bcode'>
182    *    <jc>// Servlet with automated support for GZIP compression</jc>
183    *    <ja>@RestResource</ja>(encoders={GzipEncoder.<jk>class</jk>})
184    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
185    *       ...
186    *    }
187    * </p>
188    */
189   Class<? extends Encoder>[] encoders() default {};

191   /**
192    * Specifies default values for request headers.
193    * <p>
194    * Strings are of the format <js>"Header-Name: header-value"</js>.
195    * <p>
196    * Affects values returned by {@link RestRequest#getHeader(String)} when the header is not present on the request.
197    * <p>
198    * The most useful reason for this annotation is to provide a default <code>Accept</code> header when one is not specified
199    *    so that a particular default {@link Serializer} is picked.
200    * <p>
201    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
202    *
203    * <h5 class='section'>Example:</h5>
204    * <p class='bcode'>
205    *    <jc>// Assume "text/json" Accept value when Accept not specified</jc>
206    *    <ja>@RestResource</ja>(defaultRequestHeaders={<js>"Accept: text/json"</js>})
207    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
208    *       ...
209    *    }
210    * </p>
211    */
212   String[] defaultRequestHeaders() default {};
213
214   /**
215    * Specifies default values for response headers.
216    * <p>
217    * Strings are of the format <js>"Header-Name: header-value"</js>.
218    * <p>
219    * This is equivalent to calling {@link RestResponse#setHeader(String, String)} programmatically in each of the Java methods.
220    * <p>
221    * The header value will not be set if the header value has already been specified (hence the 'default' in the name).
222    * <p>
223    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
224    *
225    * <h5 class='section'>Example:</h5>
226    * <p class='bcode'>
227    *    <jc>// Add a version header attribute to all responses</jc>
228    *    <ja>@RestResource</ja>(defaultResponseHeaders={<js>"X-Version: 1.0"</js>})
229    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
230    *       ...
231    *    }
232    * </p>
233    */
234   String[] defaultResponseHeaders() default {};

236   /**
237    * Defines children of this resource.
238    * <p>
239    * A REST child resource is simply another servlet that is initialized as part of the parent
240    *    resource and has a servlet path directly under the parent servlet path.
241    * The main advantage to defining servlets as REST children is that you do not need
242    *    to define them in the <code>web.xml</code> file of the web application.
243    * This can cut down on the number of entries that show up in the <code>web.xml</code> file
244    *    if you are defining large numbers of servlets.
245    * <p>
246    * Child resources must specify a value for {@link #path()} that identifies the subpath of the
247    *    child resource relative to the parent path.
248    * <p>
249    * It should be noted that servlets can be nested arbitrarily deep using this technique (i.e. children can also have children).
250    *
251    * <dl>
252    *    <dt>Servlet initialization:</dt>
253    *    <dd>
254    *       <p>
255    *          A child resource will be initialized immediately after the parent servlet is initialized.  The child resource
256    *          receives the same servlet config as the parent resource.  This allows configuration information such as
257    *          servlet initialization parameters to filter to child resources.
258    *       </p>
259    *    </dd>
260    *    <dt>Runtime behavior:</dt>
261    *    <dd>
262    *       <p>
263    *          As a rule, methods defined on the <code>HttpServletRequest</code> object will behave as if
264    *          the child servlet were deployed as a top-level resource under the child's servlet path.
265    *          For example, the <code>getServletPath()</code> and <code>getPathInfo()</code> methods on the
266    *          <code>HttpServletRequest</code> object will behave as if the child resource were deployed
267    *          using the child's servlet path.
268    *          Therefore, the runtime behavior should be equivalent to deploying the child servlet in
269    *          the <code>web.xml</code> file of the web application.
270    *       </p>
271    *    </dd>
272    * </dl>
273    */
274   Class<?>[] children() default {};
275
276   /**
277    * Identifies the URL subpath relative to the parent resource.
278    * <p>
279    * Typically, this annotation is only applicable to resources defined as children through the {@link #children()}
280    *    annotation.  However, it may be used in other ways (e.g. defining paths for top-level resources in microservices).
281    * <p>
282    * This annotation is ignored on top-level servlets (i.e. servlets defined in <code>web.xml</code> files).
283    * Therefore, implementers can optionally specify a path value for documentation purposes.
284    */
285   String path() default "";

475   /**
476    * The stylesheet to use for HTML views.
477    * <p>
478    * The name is a path to a stylesheet located in either the classpath or working directory.
479    * The resulting stylesheet becomes available through the servlet via the URL <js>"[servletpath]/style.css"</js>.
480    * <p>
481    * The default set of styles located in the <code>org.apache.juneau.rest.styles</code> package are:
482    * <ul class='spaced-list'>
483    *    <li><js>"styles/juneau.css"</js> - Theme based on Jazz look-and-feel.
484    *    <li><js>"styles/devops.css"</js> - Theme based on IBM DevOps look-and-feel.
485    * </ul>
486    * <p>
487    * The classpath search starts with the child servlet class and proceeds up the class hierarchy
488    *    chain.  Since the {@link RestServlet} class is in the <code>org.apache.juneau.rest</code> package
489    *    and the predefined styles are in the <code>org.apache.juneau.rest.styles</code> package, the paths to
490    *    the predefined styles are prefixed with <js>"styles/"</js>.
491    * <p>
492    * If the stylesheet cannot be found on the classpath, an attempt to look in the working directory
493    *    for it will be made.  This allows for stylesheets to be placed on the file system in the working
494    *    directory.
495    * <p>
496    * If the file cannot be located, the request to <js>"[servletpath]/style.css"</js> will return {@link HttpServletResponse#SC_NOT_FOUND}.
497    *
498    * <h5 class='section'>Example:</h5>
499    * <p class='bcode'>
500    *    <jk>package</jk> com.foo.mypackage;
501    *
502    *    <ja>@RestResource</ja>(
503    *       stylesheet=<js>"mystyles/mycss.css"</js>
504    *    )
505    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
506    *    }
507    * </p>
508    * <p>
509    * In this example, the servlet will attempt to find the <code>mycss.css</code> file in the following ordered locations:
510    * </p>
511    * <ol>
512    *    <li><code>com.foo.mypackage.mystyles</code> package.
513    *    <li><code>org.apache.juneau.rest.mystyles</code> package (since <code>RestServletDefault</code> is in <code>org.apache.juneau.rest</code>).
514    *    <li><code>[working-dir]/mystyles</code> directory.
515    * </ol>
516    * <p>
517    * Multiple stylesheets can be specified as a comma-delimited list.
518    * When multiple stylesheets are specified, their contents will be concatenated and return in the order specified
519    * in the list.
520    */
521   String stylesheet() default "";

553   /**
554    * Defines paths and locations of statically served files.
555    * <p>
556    * This is a JSON map of paths to packages/directories located on either the classpath or working directory.
557    * <p>
558    * Mappings are cumulative from parent to child.  Child resources can override mappings made on parent resources.
559    * <p>
560    * If the file cannot be located, the request will return {@link HttpServletResponse#SC_NOT_FOUND}.
561    * <p>
562    * The media type on the response is determined by the {@link RestServlet#getMimetypesFileTypeMap()} method.
563    *
564    * <h5 class='section'>Example:</h5>
565    * <p class='bcode'>
566    *    <jk>package</jk> com.foo.mypackage;
567    *
568    *    <ja>@RestResource</ja>(
569    *       path=<js>"/myresource"</js>,
570    *       staticFiles=<js>"{htdocs:'docs'}"</js>
571    *    )
572    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
573    *    }
574    * </p>
575    * <p>
576    * In this example, given a GET request to <code>/myresource/htdocs/foobar.html</code>, the servlet will attempt to find the <code>foobar.html</code> file
577    *    in the following ordered locations:
578    * </p>
579    * <ol>
580    *    <li><code>com.foo.mypackage.docs</code> package.
581    *    <li><code>org.apache.juneau.rest.docs</code> package (since <code>RestServletDefault</code> is in <code>org.apache.juneau.rest</code>).
582    *    <li><code>[working-dir]/docs</code> directory.
583    * </ol>
584    */
585   String staticFiles() default "";
