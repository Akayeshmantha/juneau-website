031/**
032 * Used to denote that a class is a REST resource and to associate metadata on it.
033 * 
034 * <p>
035 * Usually used on a subclass of {@link RestServlet}, but can be used to annotate any class that you want to expose as
036 * a REST resource.
037 * 
038 * Refer to <a class='doclink' href='../package-summary.html#TOC'>org.apache.juneau.rest</a> doc for information on
039 * using this class.
040 */
041@Documented
042@Target(TYPE)
043@Retention(RUNTIME)
044@Inherited
045public @interface RestResource {
046
047   /**
048    * Allow body URL parameter.
049    * 
050    * <p>
051    * When enabled, the HTTP body content on PUT and POST requests can be passed in as text using the <js>"body"</js>
052    * URL parameter.
053    * <br>
054    * For example:  
055    * <p class='bcode'>
056    *  ?body=(name='John%20Smith',age=45)
057    * </p>
058    * 
059    * <p>
060    * Value can contain any of the following variables:  
061    * {@link ConfigFileVar $C} 
062    * {@link CoalesceVar $CO}
063    * {@link CoalesceAndRecurseVar $CR}
064    * {@link EnvVariablesVar $E} 
065    * {@link IfVar $IF}
066    * {@link SystemPropertiesVar $S}
067    * {@link SwitchVar $SW}
068    * 
069    * <h5 class='section'>See Also:</h5>
070    * <ul>
071    *    <li class='jf'>{@link RestContext#REST_allowBodyParam}
072    * </ul>
073    */
074   String allowBodyParam() default "";
075
076   /**
077    * Allowed method parameters.
078    * 
079    * <p>
080    * When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular
081    * GET request.
082    * <br>
083    * For example: 
084    * <p class='bcode'>
085    *  ?method=OPTIONS
086    * </p>
087    * 
088    * <p>
089    * Value can contain any of the following variables:  
090    * {@link ConfigFileVar $C} 
091    * {@link CoalesceVar $CO}
092    * {@link CoalesceAndRecurseVar $CR}
093    * {@link EnvVariablesVar $E} 
094    * {@link IfVar $IF}
095    * {@link SystemPropertiesVar $S}
096    * {@link SwitchVar $SW}
097    * 
098    * <h5 class='section'>See Also:</h5>
099    * <ul>
100    *    <li class='jf'>{@link RestContext#REST_allowedMethodParams}
101    * </ul>
102    */
103   String allowedMethodParams() default "";

107    * 
108    * <p>

351    * <p>
352    * Value can contain any of the following variables:  
353    * {@link ConfigFileVar $C} 
354    * {@link CoalesceVar $CO}
355    * {@link CoalesceAndRecurseVar $CR}
356    * {@link EnvVariablesVar $E} 
357    * {@link FileVar $F} 
358    * {@link ServletInitParamVar $I},
359    * {@link IfVar $IF}
360    * {@link LocalizationVar $L}
361    * {@link RequestAttributeVar $RA} 
362    * {@link RequestFormDataVar $RF} 
363    * {@link RequestHeaderVar $RH} 
364    * {@link RequestPathVar $RP} 
365    * {@link RequestQueryVar $RQ} 
366    * {@link RequestVar $R} 
367    * {@link SystemPropertiesVar $S}
368    * {@link SerializedRequestAttrVar $SA}
369    * {@link SwitchVar $SW}
370    * {@link UrlVar $U}
371    * {@link UrlEncodeVar $UE}
372    * {@link WidgetVar $W}
373    * 
374    * <p>
375    * Corresponds to the swagger field <code>/info/description</code>.
376    *
