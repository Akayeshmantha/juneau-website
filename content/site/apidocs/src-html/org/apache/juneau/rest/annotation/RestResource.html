035/**
036 * Used to denote that a class is a REST resource and to associate metadata on it.
037 *
038 * <p>
039 * Usually used on a subclass of {@link RestServlet}, but can be used to annotate any class that you want to expose as
040 * a REST resource.
041 *
042 * Refer to <a class='doclink' href='../package-summary.html#TOC'>org.apache.juneau.rest</a> doc for information on
043 * using this class.
044 */
045@Documented
046@Target(TYPE)
047@Retention(RUNTIME)
048@Inherited
049public @interface RestResource {
050
051   /**
052    * Identifies the location of the resource bundle for this class.
053    *
054    * <p>
055    * This annotation is used to provide localized messages for the following methods:
056    * <ul>
057    *    <li>{@link RestRequest#getMessage(String, Object...)}
058    *    <li>{@link RestContext#getMessages()}
059    * </ul>
060    *
061    * <p>
062    * Refer to the {@link MessageBundle} class for a description of the message key formats used in the properties file.
063    *
064    * <p>
065    * The value can be a relative path like <js>"nls/Messages"</js>, indicating to look for the resource bundle
066    * <js>"com.foo.sample.nls.Messages"</js> if the resource class is in <js>"com.foo.sample"</js>, or it can be an
067    * absolute path, like <js>"com.foo.sample.nls.Messages"</js>
068    */
069   String messages() default "";
070
071   /**
072    * Class-level guards.
073    *
074    * <p>
075    * Associates one or more {@link RestGuard RestGuards} with all REST methods defined in this class.
076    * These guards get called immediately before execution of any REST method in this class.
077    *
078    * <p>
079    * Typically, guards will be used for permissions checking on the user making the request, but it can also be used
080    * for other purposes like pre-call validation of a request.
081    *
082    * <p>
083    * The programmatic equivalent to this annotation are the {@link RestConfig#addGuards(Class...)}/
084    * {@link RestConfig#addGuards(RestGuard...)} methods.
085    */
086   Class<? extends RestGuard>[] guards() default {};
087
088   /**
089    * Class-level converters.
090    *
091    * <p>
092    * Associates one or more {@link RestConverter converters} with a resource class.
093    * These converters get called immediately after execution of the REST method in the same order specified in the
094    * annotation.
095    *
096    * <p>
097    * Can be used for performing post-processing on the response object before serialization.
098    *
099    * <p>
100    * Default converter implementations are provided in the <a class='doclink'
101    * href='../converters/package-summary.html#TOC'>org.apache.juneau.rest.converters</a> package.
102    *
103    * <p>
104    * The programmatic equivalent to this annotation are the {@link RestConfig#addConverters(Class...)}/
105    * {@link RestConfig#addConverters(RestConverter...)} methods.
106    */
107   Class<? extends RestConverter>[] converters() default {};
108
109   /**
110    * Class-level bean filters.
111    *
112    * <p>
113    * Shortcut to add bean filters to the bean contexts of the objects returned by the following methods:
114    * <ul>
115    *    <li>{@link RestContext#getBeanContext()}
116    *    <li>{@link RestContext#getSerializers()}
117    *    <li>{@link RestContext#getParsers()}
118    * </ul>
119    *
120    * <p>
121    * If the specified class is an instance of {@link BeanFilterBuilder}, then a filter built from that builder is added.
122    * Any other classes are wrapped in a {@link InterfaceBeanFilterBuilder} to indicate that subclasses should be
123    * treated as the specified class type.
124    *
125    * <p>
126    * The programmatic equivalent to this annotation is the {@link RestConfig#addBeanFilters(Class...)} method.
127    */
128   Class<?>[] beanFilters() default {};
129
130   /**
131    * Class-level POJO swaps.
132    *
133    * <p>
134    * Shortcut to add POJO swaps to the bean contexts of the objects returned by the following methods:
135    * <ul>
136    *    <li>{@link RestContext#getBeanContext()}
137    *    <li>{@link RestContext#getSerializers()}
138    *    <li>{@link RestContext#getParsers()}
139    * </ul>
140    *
141    * <p>
142    * If the specified class is an instance of {@link PojoSwap}, then that swap is added.
143    * Any other classes are wrapped in a {@link SurrogateSwap}.
144    *
145    * <p>
146    * The programmatic equivalent to this annotation is the {@link RestConfig#addPojoSwaps(Class...)} method.
147    */
148   Class<?>[] pojoSwaps() default {};
149
150   /**
151    * Class-level Java method parameter resolvers.
152    *
153    * <p>
154    * By default, the Juneau framework will automatically Java method parameters of various types (e.g.
155    * <code>RestRequest</code>, <code>Accept</code>, <code>Reader</code>).
156    * This annotation allows you to provide your own resolvers for your own class types that you want resolved.
157    *
158    * <p>
159    * For example, if you want to pass in instances of <code>MySpecialObject</code> to your Java method, define
160    * the following resolver:
161    * <p class='bcode'>
162    *    <jk>public class</jk> MyRestParam <jk>extends</jk> RestParam {
163    *
164    *       <jc>// Must have no-arg constructor!</jc>
165    *       <jk>public</jk> MyRestParam() {
166    *          <jc>// First two parameters help with Swagger doc generation.</jc>
167    *          <jk>super</jk>(<jsf>QUERY</jsf>, <js>"myparam"</js>, MySpecialObject.<jk>class</jk>);
168    *       }
169    *
170    *       <jc>// The method that creates our object.
171    *       // In this case, we're taking in a query parameter and converting it to our object.</jc>
172    *       <jk>public</jk> Object resolve(RestRequest req, RestResponse res) <jk>throws</jk> Exception {
173    *          <jk>return new</jk> MySpecialObject(req.getQuery().get(<js>"myparam"</js>));
174    *       }
175    *    }
176    * </p>
177    *
178    * <p>
179    * <b>Note:</b>{@link RestParam} classes must have no-arg constructors.
180    *
181    * <p>
182    * The programmatic equivalent to this annotation is the {@link RestConfig#addParamResolvers(Class...)} method.
183    */
184   Class<? extends RestParam>[] paramResolvers() default {};
185
186   /**
187    * Class-level properties.
188    *
189    * <p>
190    * Shortcut for specifying class-level properties on this servlet to the objects returned by the following methods:
191    * <ul>
192    *    <li>{@link RestContext#getBeanContext()}
193    *    <li>{@link RestContext#getSerializers()}
194    *    <li>{@link RestContext#getParsers()}
195    * </ul>
196    * <p>
197    * Any of the following property names can be specified:
198    * <ul>
199    *    <li>{@link RestContext}
200    *    <li>{@link BeanContext}
201    *    <li>{@link SerializerContext}
202    *    <li>{@link ParserContext}
203    *    <li>{@link JsonSerializerContext}
204    *    <li>{@link RdfSerializerContext}
205    *    <li>{@link RdfParserContext}
206    *    <li>{@link RdfCommonContext}
207    *    <li>{@link XmlSerializerContext}
208    *    <li>{@link XmlParserContext}
209    * </ul>
210    *
211    * <p>
212    * Property values will be converted to the appropriate type.
213    *
214    * <p>
215    * In some cases, properties can be overridden at runtime through the
216    * {@link RestResponse#setProperty(String, Object)} method or through a {@link Properties @Properties} annotated
217    * method parameter.
218    *
219    * <p>
220    * The programmatic equivalent to this annotation are the {@link RestConfig#setProperty(String, Object)}/
221    * {@link RestConfig#setProperties(java.util.Map)} methods.
222    */
223   Property[] properties() default {};
224
225   /**
226    * Shortcut for setting {@link #properties()} of simple boolean types.
227    *
228    * <p>
229    * Setting a flag is equivalent to setting the same property to <js>"true"</js>.
230    */
231   String[] flags() default {};

233   /**
234    * Specifies a list of {@link Serializer} classes to add to the list of serializers available for this servlet.
235    *
236    * <p>
237    * This annotation can only be used on {@link Serializer} classes that have no-arg constructors.
238    *
239    * <p>
240    * The programmatic equivalent to this annotation are the {@link RestConfig#addSerializers(Class...)}/
241    * {@link RestConfig#addSerializers(Serializer...)} methods.
242    */
243   Class<? extends Serializer>[] serializers() default {};
244
245   /**
246    * Specifies a list of {@link Parser} classes to add to the list of parsers available for this servlet.
247    *
248    * <p>
249    * This annotation can only be used on {@link Parser} classes that have no-arg constructors.
250    *
251    * <p>
252    * The programmatic equivalent to this annotation are the {@link RestConfig#addParsers(Class...)}/
253    * {@link RestConfig#addParsers(Parser...)} methods.
254    */
255   Class<? extends Parser>[] parsers() default {};

257   /**
258    * Specifies a list of {@link ResponseHandler} classes that know how to convert POJOs returned by REST methods or
259    * set via {@link RestResponse#setOutput(Object)} into appropriate HTTP responses.
260    *
261    * <p>
262    * See {@link ResponseHandler} for details.
263    *
264    * <p>
265    * The programmatic equivalent to this annotation are the {@link RestConfig#addResponseHandlers(Class...)}/
266    * {@link RestConfig#addResponseHandlers(ResponseHandler...)} methods.
267    */
268   Class<? extends ResponseHandler>[] responseHandlers() default {};
269
270   /**
271    * Specifies a list of {@link Encoder} to associate with this servlet.
272    *
273    * <p>
274    * These can be used to enable various kinds of compression (e.g. <js>"gzip"</js>) on requests and responses.
275    *
276    * <p>
277    * This annotation can only be used on {@link Encoder} classes that have no-arg constructors.
278    *
279    * <h5 class='section'>Example:</h5>
280    * <p class='bcode'>
281    *    <jc>// Servlet with automated support for GZIP compression</jc>
282    *    <ja>@RestResource</ja>(encoders={GzipEncoder.<jk>class</jk>})
283    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
284    *       ...
285    *    }
286    * </p>
287    *
288    * <p>
289    * The programmatic equivalent to this annotation are the {@link RestConfig#addEncoders(Class...)}/
290    * {@link RestConfig#addEncoders(Encoder...)} methods.
291    */
292   Class<? extends Encoder>[] encoders() default {};
293
294   /**
295    * Specifies default values for request headers.
296    *
297    * <p>
298    * Strings are of the format <js>"Header-Name: header-value"</js>.
299    *
300    * <p>
301    * Affects values returned by {@link RestRequest#getHeader(String)} when the header is not present on the request.
302    *
303    * <p>
304    * The most useful reason for this annotation is to provide a default <code>Accept</code> header when one is not
305    * specified so that a particular default {@link Serializer} is picked.
306    *
307    * <p>
308    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
309    *
310    * <h5 class='section'>Example:</h5>
311    * <p class='bcode'>
312    *    <jc>// Assume "text/json" Accept value when Accept not specified</jc>
313    *    <ja>@RestResource</ja>(defaultRequestHeaders={<js>"Accept: text/json"</js>})
314    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
315    *       ...
316    *    }
317    * </p>
318    *
319    * <p>
320    * The programmatic equivalent to this annotation are the {@link RestConfig#addDefaultRequestHeader(String, Object)}/
321    * {@link RestConfig#addDefaultRequestHeaders(String...)} methods.
322    */
323   String[] defaultRequestHeaders() default {};
324
325   /**
326    * Specifies default values for response headers.
327    *
328    * <p>
329    * Strings are of the format <js>"Header-Name: header-value"</js>.
330    *
331    * <p>
332    * This is equivalent to calling {@link RestResponse#setHeader(String, String)} programmatically in each of the Java
333    * methods.
334    *
335    * <p>
336    * The header value will not be set if the header value has already been specified (hence the 'default' in the name).
337    *
338    * <p>
339    * Only one header value can be specified per entry (i.e. it's not a delimited list of header entries).
340    *
341    * <h5 class='section'>Example:</h5>
342    * <p class='bcode'>
343    *    <jc>// Add a version header attribute to all responses</jc>
344    *    <ja>@RestResource</ja>(defaultResponseHeaders={<js>"X-Version: 1.0"</js>})
345    *    <jk>public</jk> MyRestServlet <jk>extends</jk> RestServlet {
346    *       ...
347    *    }
348    * </p>
349    *
350    * <p>
351    * The programmatic equivalent to this annotation are the {@link RestConfig#addDefaultResponseHeader(String, Object)}/
352    * {@link RestConfig#addDefaultResponseHeaders(String...)} methods.
353    */
354   String[] defaultResponseHeaders() default {};

356   /**
357    * Defines children of this resource.
358    *
359    * <p>
360    * A REST child resource is simply another servlet that is initialized as part of the parent resource and has a
361    * servlet path directly under the parent servlet path.
362    * The main advantage to defining servlets as REST children is that you do not need to define them in the
363    * <code>web.xml</code> file of the web application.
364    * This can cut down on the number of entries that show up in the <code>web.xml</code> file if you are defining
365    * large numbers of servlets.
366    *
367    * <p>
368    * Child resources must specify a value for {@link #path()} that identifies the subpath of the child resource
369    * relative to the parent path.
370    *
371    * <p>
372    * It should be noted that servlets can be nested arbitrarily deep using this technique (i.e. children can also have
373    * children).
374    *
375    * <dl>
376    *    <dt>Servlet initialization:</dt>
377    *    <dd>
378    *       <p>
379    *          A child resource will be initialized immediately after the parent servlet is initialized.
380    *          The child resource receives the same servlet config as the parent resource.
381    *          This allows configuration information such as servlet initialization parameters to filter to child
382    *          resources.
383    *       </p>
384    *    </dd>
385    *    <dt>Runtime behavior:</dt>
386    *    <dd>
387    *       <p>
388    *          As a rule, methods defined on the <code>HttpServletRequest</code> object will behave as if the child
389    *          servlet were deployed as a top-level resource under the child's servlet path.
390    *          For example, the <code>getServletPath()</code> and <code>getPathInfo()</code> methods on the
391    *          <code>HttpServletRequest</code> object will behave as if the child resource were deployed using the
392    *          child's servlet path.
393    *          Therefore, the runtime behavior should be equivalent to deploying the child servlet in the
394    *          <code>web.xml</code> file of the web application.
395    *       </p>
396    *    </dd>
397    * </dl>
398    *
399    * <p>
400    * The programmatic equivalent to this annotation are the {@link RestConfig#addChildResource(String, Object)}/
401    * {@link RestConfig#addChildResources(Class...)}/{@link RestConfig#addChildResources(Object...)} methods.
402    */
403   Class<?>[] children() default {};
404
405   /**
406    * Identifies the URL subpath relative to the parent resource.
407    *
408    * <p>
409    * Typically, this annotation is only applicable to resources defined as children through the {@link #children()}
410    * annotation.
411    * However, it may be used in other ways (e.g. defining paths for top-level resources in microservices).
412    *
413    * <p>
414    * This annotation is ignored on top-level servlets (i.e. servlets defined in <code>web.xml</code> files).
415    * Therefore, implementers can optionally specify a path value for documentation purposes.
416    *
417    * <p>
418    * The programmatic equivalent to this annotation is the {@link RestConfig#setPath(String)} method.
419    */
420   String path() default "";

433    * <p>
434    * This field can contain variables (e.g. "$L{my.localized.variable}").
435    *
436    * <p>
437    * Corresponds to the swagger field <code>/info/title</code>.

481   /**
482    * The favicon to use for HTML views.
483    *
484    * <p>
485    * The name is a path to an icon file located in either the classpath or working directory.
486    * The resulting favicon becomes available in the servlet via the URL <js>"[servlet-path]/favicon.ico"</js>.
487    *
488    * <p>
489    * If the file cannot be located, the request to <js>"[servlet-path]/favicon.ico"</js> will return
490    * {@link HttpServletResponse#SC_NOT_FOUND}.
491    *
492    * <h5 class='section'>Example:</h5>
493    * <p class='bcode'>
494    *    <jk>package</jk> com.foo.mypackage;
495    *
496    *    <ja>@RestResource</ja>(
497    *       favicon=<js>"mydocs/myicon.ico"</js>
498    *    )
499    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
500    *    }
501    * </p>
502    *
503    * <p>
504    * In this example, the servlet will attempt to find the <code>myicon.ico</code> file in the following ordered
505    * locations:
506    * <ol>
507    *    <li><code>com.foo.mypackage.mydocs</code> package.
508    *    <li><code>org.apache.juneau.rest.mydocs</code> package (since <code>RestServletDefault</code> is in
509    *       <code>org.apache.juneau.rest</code>).
510    *    <li><code>[working-dir]/mydocs</code> directory.
511    * </ol>
512    *
513    * <p>
514    * The programmatic equivalent to this annotation are the {@link RestConfig#setFavIcon(Object)}/
515    * {@link RestConfig#setFavIcon(Class, String)} methods.
516    */
517   String favicon() default "";
518
519   /**
520    * Defines paths and locations of statically served files.
521    *
522    * <p>
523    * This is a JSON map of paths to packages/directories located on either the classpath or working directory.
524    *
525    * <p>
526    * Mappings are cumulative from parent to child.  Child resources can override mappings made on parent resources.
527    *
528    * <p>
529    * If the file cannot be located, the request will return {@link HttpServletResponse#SC_NOT_FOUND}.
530    *
531    * <h5 class='section'>Example:</h5>
532    * <p class='bcode'>
533    *    <jk>package</jk> com.foo.mypackage;
534    *
535    *    <ja>@RestResource</ja>(
536    *       path=<js>"/myresource"</js>,
537    *       staticFiles=<js>"{htdocs:'docs'}"</js>
538    *    )
539    *    <jk>public class</jk> MyResource <jk>extends</jk> RestServletDefault {
540    *    }
541    * </p>
542    *
543    * <p>
544    * In this example, given a GET request to <code>/myresource/htdocs/foobar.html</code>, the servlet will attempt to
545    * find the <code>foobar.html</code> file in the following ordered locations:
546    * <ol>
547    *    <li><code>com.foo.mypackage.docs</code> package.
548    *    <li><code>org.apache.juneau.rest.docs</code> package (since <code>RestServletDefault</code> is in
549    *       <code>org.apache.juneau.rest</code>).
550    *    <li><code>[working-dir]/docs</code> directory.
551    * </ol>
552    *
553    * <p>
554    * The programmatic equivalent to this annotation is the {@link RestConfig#addStaticFiles(Class, String)} method.
555    */
556   String staticFiles() default "";

560    *
561    * <p>
562    * The client version is used to support backwards compatibility for breaking REST interface changes.
563    * Used in conjunction with {@link RestMethod#clientVersion()} annotation.
564    *
565    * <p>
566    * If not specified, uses <js>"X-Client-Version"</js>.
567    *
568    * <p>
569    * The programmatic equivalent to this annotation is the {@link RestConfig#setClientVersionHeader(String)} method.

573   /**

575    *
576    * <p>
577    * The default implementation simply instantiates the class using one of the following constructors:
578    * <ul>
579    *    <li><code><jk>public</jk> T(RestConfig)</code>
580    *    <li><code><jk>public</jk> T()</code>
581    * </ul>
582    *
583    * <p>
584    * The former constructor can be used to get access to the {@link RestConfig} object to get access to the config
585    * file and initialization information or make programmatic modifications to the resource before full initialization.
586    *
587    * <p>
588    * Non-<code>RestServlet</code> classes can also add the following two methods to get access to the
589    * {@link RestConfig} and {@link RestContext} objects:
590    * <ul>
591    *    <li><code><jk>public void</jk> init(RestConfig);</code>
592    *    <li><code><jk>public void</jk> init(RestContext);</code>
593    * </ul>
594    *
595    * <p>
596    * Subclasses can be used to provide customized resolution of REST resource class instances.
597    *
598    * <p>
599    * The programmatic equivalent to this annotation are the {@link RestConfig#setResourceResolver(Class)}/
600    * {@link RestConfig#setResourceResolver(RestResourceResolver)} methods.
601    */

604   /**
605    * Specifies the logger class to use for logging.
606    *
