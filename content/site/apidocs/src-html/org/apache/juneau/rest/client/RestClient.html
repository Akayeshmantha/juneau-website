036/**
037 * Utility class for interfacing with remote REST interfaces.

039 * <h6 class='topic'>Features</h6>

108      this.parser = parser;
109      this.urlEncodingSerializer = urlEncodingSerializer;
110
111      Map<String,String> h2 = new ConcurrentHashMap<String,String>(headers);
112
113      this.headers = Collections.unmodifiableMap(h2);
114      this.interceptors = interceptors.toArray(new RestCallInterceptor[interceptors.size()]);
115      this.remoteableServletUri = remoteableServletUri;
116      this.remoteableServiceUriMap = new ConcurrentHashMap<Method,String>(remoteableServiceUriMap);
117      this.rootUrl = rootUri;
118      this.retryOn = retryOn;
119      this.retries = retries;
120      this.retryInterval = retryInterval;

122      if (Boolean.getBoolean("org.apache.juneau.rest.client.RestClient.trackLifecycle"))

125         creationStack = null;

128   /**
129    * Calls {@link CloseableHttpClient#close()} on the underlying {@link CloseableHttpClient}.
130    * It's good practice to call this method after the client is no longer used.
131    *
132    * @throws IOException
133    */
134   public void close() throws IOException {
135      isClosed = true;
136      if (httpClient != null && ! keepHttpClientOpen)
137         httpClient.close();
138      if (Boolean.getBoolean("org.apache.juneau.rest.client.RestClient.trackLifecycle"))
139         closedStack = Thread.currentThread().getStackTrace();
140   }

142   /**
143    * Same as {@link #close()}, but ignores any exceptions.
144    */
145   public void closeQuietly() {
146      isClosed = true;
147      try {
148         if (httpClient != null && ! keepHttpClientOpen)
149            httpClient.close();
150      } catch (Throwable t) {}
151      if (Boolean.getBoolean("org.apache.juneau.rest.client.RestClient.trackLifecycle"))
152         closedStack = Thread.currentThread().getStackTrace();
153   }

258   /**
259    * Performs a REST call where the entire call is specified in a simple string.
260    * <p>
261    * This method is useful for performing callbacks when the target of a callback is passed in
262    * on an initial request, for example to signal when a long-running process has completed.
263    * <p>
264    * The call string can be any of the following formats:
265    * <ul class='spaced-list'>
266    *    <li><js>"[method] [url]"</js> - e.g. <js>"GET http://localhost/callback"</js>
267    *    <li><js>"[method] [url] [payload]"</js> - e.g. <js>"POST http://localhost/callback some text payload"</js>
268    *    <li><js>"[method] [headers] [url] [payload]"</js> - e.g. <js>"POST {'Content-Type':'text/json'} http://localhost/callback {'some':'json'}"</js>
269    * </ul>
270    * <p>
271    * The payload will always be sent using a simple {@link StringEntity}.
272    *
273    * @param callString The call string.
274    * @return A {@link RestCall} object that can be further tailored before executing the request
275    *    and getting the response as a parsed object.
276    * @throws RestCallException
277    */
278   public RestCall doCallback(String callString) throws RestCallException {
279      String s = callString;
280      try {
281         RestCall rc = null;
282         String method = null, uri = null, content = null;
283         ObjectMap h = null;
284         int i = s.indexOf(' ');
285         if (i != -1) {
286            method = s.substring(0, i).trim();
287            s = s.substring(i).trim();
288            if (s.length() > 0) {
289               if (s.charAt(0) == '{') {
290                  i = s.indexOf('}');
291                  if (i != -1) {
292                     String json = s.substring(0, i+1);
293                     h = JsonParser.DEFAULT.parse(json, ObjectMap.class);
294                     s = s.substring(i+1).trim();
295                  }
296               }
297               if (s.length() > 0) {
298                  i = s.indexOf(' ');
299                  if (i == -1)
300                     uri = s;
301                  else {
302                     uri = s.substring(0, i).trim();
303                     s = s.substring(i).trim();
304                     if (s.length() > 0)
305                        content = s;
306                  }
307               }
308            }
309         }
310         if (method != null && uri != null) {
311            rc = doCall(method, uri, content != null);
312            if (content != null)
313               rc.input(new StringEntity(content));
314            if (h != null)
315               for (Map.Entry<String,Object> e : h.entrySet())
316                  rc.header(e.getKey(), e.getValue());
317            return rc;
318         }
319      } catch (Exception e) {
320         throw new RestCallException(e);
321      }
322      throw new RestCallException("Invalid format for call string.");
323   }

325   /**
326    * Perform a generic REST call.
327    *
328    * @param method The HTTP method.
329    * @param url The URL of the remote REST resource.  Can be any of the following:  {@link String}, {@link URI}, {@link URL}.
330    * @param content The HTTP body content.
331    * Can be of the following types:
332    * <ul class='spaced-list'>
333    *    <li>{@link Reader} - Raw contents of {@code Reader} will be serialized to remote resource.
334    *    <li>{@link InputStream} - Raw contents of {@code InputStream} will be serialized to remote resource.
335    *    <li>{@link Object} - POJO to be converted to text using the {@link Serializer} registered with the {@link RestClient}.
336    *    <li>{@link HttpEntity} - Bypass Juneau serialization and pass HttpEntity directly to HttpClient.
337    * </ul>
338    * This parameter is IGNORED if {@link HttpMethod#hasContent()} is <jk>false</jk>.
339    * @return A {@link RestCall} object that can be further tailored before executing the request
340    *    and getting the response as a parsed object.
341    * @throws RestCallException If any authentication errors occurred.
342    */
343   public RestCall doCall(HttpMethod method, Object url, Object content) throws RestCallException {
344      RestCall rc = doCall(method.name(), url, method.hasContent());
345      if (method.hasContent())
346         rc.input(content);
347      return rc;
348   }

350   /**
351    * Perform a generic REST call.
352    *
353    * @param method The method name (e.g. <js>"GET"</js>, <js>"OPTIONS"</js>).
354    * @param url The URL of the remote REST resource.  Can be any of the following:  {@link String}, {@link URI}, {@link URL}.
355    * @param hasContent Boolean flag indicating if the specified request has content associated with it.
356    * @return A {@link RestCall} object that can be further tailored before executing the request
357    *    and getting the response as a parsed object.
358    * @throws RestCallException If any authentication errors occurred.
359    */
360   public RestCall doCall(String method, Object url, boolean hasContent) throws RestCallException {
361      if (isClosed) {
362         Exception e2 = null;
363         if (closedStack != null) {
364            e2 = new Exception("Creation stack:");
365            e2.setStackTrace(closedStack);
366            throw new RestCallException("RestClient.close() has already been called.  This client cannot be reused.").initCause(e2);
367         }
368         throw new RestCallException("RestClient.close() has already been called.  This client cannot be reused.  Closed location stack trace can be displayed by setting the system property 'org.apache.juneau.rest.client.RestClient.trackCreation' to true.");
369      }
370
371      HttpRequestBase req = null;
372      RestCall restCall = null;
373      final String methodUC = method.toUpperCase(Locale.ENGLISH);
374      if (hasContent) {
375         req = new HttpEntityEnclosingRequestBase() {
376            @Override /* HttpRequest */
377            public String getMethod() {
378               return methodUC;
379            }
380         };
381         restCall = new RestCall(this, req);
382      } else {
383         req = new HttpRequestBase() {
384            @Override /* HttpRequest */
385            public String getMethod() {
386               return methodUC;
387            }
388         };
389         restCall = new RestCall(this, req);
390      }
391      try {
392         req.setURI(toURI(url));
393      } catch (URISyntaxException e) {
394         throw new RestCallException(e);
395      }
396      for (Map.Entry<String,? extends Object> e : headers.entrySet())
397         restCall.header(e.getKey(), e.getValue());
398
399      if (parser != null && ! req.containsHeader("Accept"))
400         req.setHeader("Accept", parser.getPrimaryMediaType().toString());
401
402      return restCall;
403   }

405   /**
406    * Create a new proxy interface for the specified remoteable service interface.
407    *
408    * @param interfaceClass The interface to create a proxy for.
409    * @return The new proxy interface.
410    * @throws RuntimeException If the Remotable service URI has not been specified on this
411    *    client by calling {@link RestClientBuilder#remoteableServletUri(String)}.
412    */
413   @SuppressWarnings("unchecked")
414   public <T> T getRemoteableProxy(final Class<T> interfaceClass) {
415      if (remoteableServletUri == null)
416         throw new RuntimeException("Remoteable service URI has not been specified.");
417      return (T)Proxy.newProxyInstance(
418         interfaceClass.getClassLoader(),
419         new Class[] { interfaceClass },
420         new InvocationHandler() {
421            @Override /* InvocationHandler */
422            public Object invoke(Object proxy, Method method, Object[] args) {
423               try {
424                  String uri = remoteableServiceUriMap.get(method);
425                  if (uri == null) {
426                     // Constructing this string each time can be time consuming, so cache it.
427                     uri = remoteableServletUri + '/' + interfaceClass.getName() + '/' + ClassUtils.getMethodSignature(method);
428                     remoteableServiceUriMap.put(method, uri);
429                  }
430                  return doPost(uri, args).getResponse(method.getReturnType());
431               } catch (Exception e) {
432                  throw new RuntimeException(e);
433               }
434            }
435      });
436   }

438   private Pattern absUrlPattern = Pattern.compile("^\\w+\\:\\/\\/.*");
439
440   private URI toURI(Object url) throws URISyntaxException {
441      assertFieldNotNull(url, "url");
442      if (url instanceof URI)
443         return (URI)url;
444      if (url instanceof URL)
445         ((URL)url).toURI();
446      String s = url.toString();
447      if (rootUrl != null && ! absUrlPattern.matcher(s).matches()) {
448         if (s.isEmpty())
449            s = rootUrl;
450         else {
451            StringBuilder sb = new StringBuilder(rootUrl);
452            if (! s.startsWith("/"))
453               sb.append('/');
454            sb.append(s);
455            s = sb.toString();
456         }
457      }
458      return new URI(s);
459   }

463      if (! isClosed && ! keepHttpClientOpen) {
464         System.err.println("WARNING:  RestClient garbage collected before it was finalized.");
465         if (creationStack != null) {
466            System.err.println("Creation Stack:");
467            for (StackTraceElement e : creationStack)
468               System.err.println(e);
469         } else {
470            System.err.println("Creation stack traces can be displayed by setting the system property 'org.apache.juneau.rest.client.RestClient.trackLifecycle' to true.");
471         }
472      }
