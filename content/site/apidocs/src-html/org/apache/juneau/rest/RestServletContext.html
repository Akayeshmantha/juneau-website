023/**
024 * Configurable properties on the {@link RestServlet} class.
025 * <p>
026 *    Properties can be set on the {@link RestServlet} class using the {@link RestResource#properties} or {@link RestMethod#properties} annotations.
027 * <p>
028 *    These properties can also be passed in as servlet init parameters or system properties.
029 * <p>
030 *    Some of these properties are only applicable on the servlet class, and others can be specified on the servlet class or method.<br>
031 *    These distinctions are noted below.
032 * <p>
033 *    See {@link ContextFactory} for more information about context properties.
034 *
035 * <h6 class='topic' id='ConfigProperties'>Configurable properties on the REST servlet</h6>
036 * <table class='styled' style='border-collapse: collapse;'>
037 *    <tr><th>Setting name</th><th>Description</th><th>Data type</th><th>Default value</th></tr>
038 *    <tr>
039 *       <td>{@link #REST_allowHeaderParams}</td>
040 *       <td>Enable header URL parameters.</td>
041 *       <td><code>Boolean</code></td>
042 *       <td><jk>true</jk></td>
043 *    </tr>
044 *    <tr>
045 *       <td>{@link #REST_allowMethodParam}</td>
046 *       <td>Enable <js>"method"</js> URL parameter for specific HTTP methods.</td>
047 *       <td><code>String</code></td>
048 *       <td><js>""</js></td>
049 *    </tr>
050 *    <tr>
051 *       <td>{@link #REST_allowBodyParam}</td>
052 *       <td>Enable <js>"body"</js> URL parameter.</td>
053 *       <td><code>Boolean</code></td>
054 *       <td><jk>true</jk></td>
055 *    </tr>
056 *    <tr>
057 *       <td>{@link #REST_renderResponseStackTraces}</td>
058 *       <td>Render stack traces in HTTP response bodies when errors occur.</td>
059 *       <td><code>Boolean</code></td>
060 *       <td><jk>false</jk></td>
061 *    </tr>
062 *    <tr>
063 *       <td>{@link #REST_useStackTraceHashes}</td>
064 *       <td>Use stack trace hashes.</td>
065 *       <td><code>Boolean</code></td>
066 *       <td><jk>false</jk></td>
067 *    </tr>
068 *    <tr>
069 *       <td>{@link #REST_defaultCharset}</td>
070 *       <td>Default character encoding.</td>
071 *       <td><code>String</code></td>
072 *       <td><js>"utf-8"</js></td>
073 *    </tr>
074 *    <tr>
075 *       <td>{@link #REST_paramFormat}</td>
076 *       <td>Expected format of request parameters.</td>
077 *       <td><code>String</code></td>
078 *       <td><js>"UON"</js></td>
079 *    </tr>
080 * </table>
081 */
082public final class RestServletContext extends Context {
083
084   /**
085    * <b>Configuration property:</b>  Enable header URL parameters.
086    * <p>
087    * <ul>
088    *    <li><b>Name:</b> <js>"RestServlet.allowHeaderParams"</js>
089    *    <li><b>Data type:</b> <code>Boolean</code>
090    *    <li><b>Default:</b> <jk>true</jk>
091    * </ul>
092    * <p>
093    *    When enabled, headers such as <js>"Accept"</js> and <js>"Content-Type"</js> to be passed in as URL query parameters.
094    *    For example:  <js>"?Accept=text/json&amp;Content-Type=text/json"</js>
095    * <p>
096    *    Parameter names are case-insensitive.
097    * <p>
098    *    Useful for debugging REST interface using only a browser.
099    * <p>
100    *    Applicable to servlet class only.
101    */
102   public static final String REST_allowHeaderParams = "RestServlet.allowHeaderParams";
103
104   /**
105    * <b>Configuration property:</b>  Enable <js>"method"</js> URL parameter for specific HTTP methods.
106    * <p>
107    * <ul>
108    *    <li><b>Name:</b> <js>"RestServlet.allowMethodParam"</js>
109    *    <li><b>Data type:</b> <code>String</code>
110    *    <li><b>Default:</b> <js>""</js>
111    * </ul>
112    * <p>
113    *    When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular GET request.
114    *    For example:  <js>"?method=OPTIONS"</js>
115    * <p>
116    *    Format is a comma-delimited list of HTTP method names that can be passed in as a method parameter.
117    *    Parameter name is case-insensitive.
118    *    Use "*" to represent all methods.
119    *    For backwards compatibility, "true" also means "*".
120    * <p>
121    *    Note that per the <a href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec9.html">HTTP specification</a>, special care should
122    *    be taken when allowing non-safe (POST, PUT, DELETE) methods to be invoked through GET requests.
123    * <p>
124    *    Applicable to servlet class only.
125    * <p>
126    *    Example: <js>"HEAD,OPTIONS"</js>
127    */
128   public static final String REST_allowMethodParam = "RestServlet.allowMethodParam";
129
130   /**
131    * <b>Configuration property:</b>  Enable <js>"body"</js> URL parameter.
132    * <p>
133    * <ul>
134    *    <li><b>Name:</b> <js>"RestServlet.allowBodyParam"</js>
135    *    <li><b>Data type:</b> <code>Boolean</code>
136    *    <li><b>Default:</b> <jk>true</jk>
137    * </ul>
138    * <p>
139    *    When enabled, the HTTP body content on PUT and POST requests can be passed in as text using the <js>"body"</js> URL parameter.
140    *    For example:  <js>"?body={name:'John%20Smith',age:45}"</js>
141    * <p>
142    *    Parameter name is case-insensitive.
143    * <p>
144    *    Useful for debugging PUT and POST methods using only a browser.
145    * <p>
146    *    Applicable to servlet class only.
147    */
148   public static final String REST_allowBodyParam = "RestServlet.allowBodyParam";
149
150   /**
151    * <b>Configuration property:</b>  Render stack traces.
152    * <p>
153    * <ul>
154    *    <li><b>Name:</b> <js>"RestServlet.renderResponseStackTraces"</js>
155    *    <li><b>Data type:</b> <code>Boolean</code>
156    *    <li><b>Default:</b> <jk>false</jk>
157    * </ul>
158    * <p>
159    *    Render stack traces in HTTP response bodies when errors occur.
160    * <p>
161    *    When enabled, Java stack traces will be rendered in the output response.
162    *    Useful for debugging, although allowing stack traces to be rendered may cause security concerns.
163    * <p>
164    *    Applicable to servlet class only.
165    */
166   public static final String REST_renderResponseStackTraces = "RestServlet.renderResponseStackTraces";
167
168   /**
169    * <b>Configuration property:</b>  Use stack trace hashes.
170    * <p>
171    * <ul>
172    *    <li><b>Name:</b> <js>"RestServlet.useStackTraceHashes"</js>
173    *    <li><b>Data type:</b> <code>Boolean</code>
174    *    <li><b>Default:</b> <jk>true</jk>
175    * </ul>
176    * <p>
177    *    When enabled, the number of times an exception has occurred will be determined based on stack trace hashsums,
178    *    made available through the {@link RestException#getOccurrence()} method.
179    * <p>
180    *    Applicable to servlet class only.
181    */
182   public static final String REST_useStackTraceHashes = "RestServlet.useStackTraceHashes";
183
184   /**
185    * <b>Configuration property:</b>  Default character encoding.
186    * <p>
187    * <ul>
188    *    <li><b>Name:</b> <js>"RestServlet.defaultCharset"</js>
189    *    <li><b>Data type:</b> <code>String</code>
190    *    <li><b>Default:</b> <js>"utf-8"</js>
191    * </ul>
192    * <p>
193    *    The default character encoding for the request and response if not specified on the request.
194    * <p>
195    *    Applicable to servlet class and methods.
196    */
197   public static final String REST_defaultCharset = "RestServlet.defaultCharset";
198
199   /**
200    * <b>Configuration property:</b>  Expected format of request parameters.
201    * <p>
202    * <ul>
203    *    <li><b>Name:</b> <js>"RestServlet.paramFormat"</js>
204    *    <li><b>Data type:</b> <code>String</code>
205    *    <li><b>Default:</b> <js>"UON"</js>
206    * </ul>
207    * <p>
208    *    Possible values:
209    * <ul class='spaced-list'>
210    *    <li><js>"UON"</js> - URL-Encoded Object Notation.<br>
211    *       This notation allows for request parameters to contain arbitrarily complex POJOs.
212    *    <li><js>"PLAIN"</js> - Plain text.<br>
213    *       This treats request parameters as plain text.<br>
214    *       Only POJOs directly convertable from <l>Strings</l> can be represented in parameters when using this mode.
215    * </ul>
216    * <p>
217    *    Note that the parameter value <js>"(foo)"</js> is interpreted as <js>"(foo)"</js> when using plain mode, but
218    *    <js>"foo"</js> when using UON mode.
219    * <p>
220    *    The format can also be specified per-parameter using the {@link FormData#format() @FormData.format()} and {@link Query#format() @Query.format()}
221    *    annotations.
222    * <p>
223    *    Applicable to servlet class and methods.
224    */
225   public static final String REST_paramFormat = "RestServlet.paramFormat";

325   final boolean allowHeaderParams, allowBodyParam, renderResponseStackTraces, useStackTraceHashes;
326   final String defaultCharset, paramFormat;
327   final Set<String> allowMethodParams;
328
329   /**
330    * Constructor.
331    * <p>
332    *    Typically only called from {@link ContextFactory#getContext(Class)}.
333    *
334    * @param cf The factory that created this context.
335    */
336   public RestServletContext(ContextFactory cf) {
337      super(cf);
338      allowHeaderParams = cf.getProperty(REST_allowHeaderParams, boolean.class, true);
339      allowBodyParam = cf.getProperty(REST_allowBodyParam, boolean.class, true);
340      renderResponseStackTraces = cf.getProperty(REST_renderResponseStackTraces, boolean.class, false);
341      useStackTraceHashes = cf.getProperty(REST_useStackTraceHashes, boolean.class, true);
342      defaultCharset = cf.getProperty(REST_defaultCharset, String.class, "utf-8");
343      paramFormat = cf.getProperty(REST_paramFormat, String.class, "");
344
345      Set<String> s = new LinkedHashSet<String>();
346      for (String m : StringUtils.split(cf.getProperty(REST_allowMethodParam, String.class, ""), ','))
347         if (m.equals("true"))  // For backwards compatibility when this was a boolean field.
348            s.add("*");
349         else
350            s.add(m.toUpperCase());
351      allowMethodParams = Collections.unmodifiableSet(s);
352   }
