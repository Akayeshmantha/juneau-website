023/**
024 * Configurable properties on the {@link RestServlet} class.
025 * <p>
026 * Properties can be set on the {@link RestServlet} class using the {@link RestResource#properties} or {@link RestMethod#properties} annotations.
027 * <p>
028 * These properties can also be passed in as servlet init parameters or system properties.
029 * <p>
030 * Some of these properties are only applicable on the servlet class, and others can be specified on the servlet class or method.<br>
031 * These distinctions are noted below.
032 * <p>
033 * See {@link ContextFactory} for more information about context properties.
034 */
035public final class RestServletContext extends Context {
036
037   /**
038    * <b>Configuration property:</b>  Enable header URL parameters.
039    * <p>
040    * <ul>
041    *    <li><b>Name:</b> <js>"RestServlet.allowHeaderParams"</js>
042    *    <li><b>Data type:</b> <code>Boolean</code>
043    *    <li><b>Default:</b> <jk>true</jk>
044    * </ul>
045    * <p>
046    * When enabled, headers such as <js>"Accept"</js> and <js>"Content-Type"</js> to be passed in as URL query parameters.
047    * For example:  <js>"?Accept=text/json&amp;Content-Type=text/json"</js>
048    * <p>
049    * Parameter names are case-insensitive.
050    * <p>
051    * Useful for debugging REST interface using only a browser.
052    * <p>
053    * Applicable to servlet class only.
054    */
055   public static final String REST_allowHeaderParams = "RestServlet.allowHeaderParams";
056
057   /**
058    * <b>Configuration property:</b>  Enable <js>"method"</js> URL parameter for specific HTTP methods.
059    * <p>
060    * <ul>
061    *    <li><b>Name:</b> <js>"RestServlet.allowMethodParam"</js>
062    *    <li><b>Data type:</b> <code>String</code>
063    *    <li><b>Default:</b> <js>""</js>
064    * </ul>
065    * <p>
066    * When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular GET request.
067    * For example:  <js>"?method=OPTIONS"</js>
068    * <p>
069    * Format is a comma-delimited list of HTTP method names that can be passed in as a method parameter.
070    * Parameter name is case-insensitive.
071    * Use "*" to represent all methods.
072    * For backwards compatibility, "true" also means "*".
073    * <p>
074    * Note that per the <a class="doclink" href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec9.html">HTTP specification</a>, special care should
075    *    be taken when allowing non-safe (POST, PUT, DELETE) methods to be invoked through GET requests.
076    * <p>
077    * Applicable to servlet class only.
078    * <p>
079    * Example: <js>"HEAD,OPTIONS"</js>
080    */
081   public static final String REST_allowMethodParam = "RestServlet.allowMethodParam";
082
083   /**
084    * <b>Configuration property:</b>  Enable <js>"body"</js> URL parameter.
085    * <p>
086    * <ul>
087    *    <li><b>Name:</b> <js>"RestServlet.allowBodyParam"</js>
088    *    <li><b>Data type:</b> <code>Boolean</code>
089    *    <li><b>Default:</b> <jk>true</jk>
090    * </ul>
091    * <p>
092    * When enabled, the HTTP body content on PUT and POST requests can be passed in as text using the <js>"body"</js> URL parameter.
093    * For example:  <js>"?body={name:'John%20Smith',age:45}"</js>
094    * <p>
095    * Parameter name is case-insensitive.
096    * <p>
097    * Useful for debugging PUT and POST methods using only a browser.
098    * <p>
099    * Applicable to servlet class only.
100    */
101   public static final String REST_allowBodyParam = "RestServlet.allowBodyParam";
102
103   /**
104    * <b>Configuration property:</b>  Render stack traces.
105    * <p>
106    * <ul>
107    *    <li><b>Name:</b> <js>"RestServlet.renderResponseStackTraces"</js>
108    *    <li><b>Data type:</b> <code>Boolean</code>
109    *    <li><b>Default:</b> <jk>false</jk>
110    * </ul>
111    * <p>
112    * Render stack traces in HTTP response bodies when errors occur.
113    * <p>
114    * When enabled, Java stack traces will be rendered in the output response.
115    * Useful for debugging, although allowing stack traces to be rendered may cause security concerns.
116    * <p>
117    * Applicable to servlet class only.
118    */
119   public static final String REST_renderResponseStackTraces = "RestServlet.renderResponseStackTraces";
120
121   /**
122    * <b>Configuration property:</b>  Use stack trace hashes.
123    * <p>
124    * <ul>
125    *    <li><b>Name:</b> <js>"RestServlet.useStackTraceHashes"</js>
126    *    <li><b>Data type:</b> <code>Boolean</code>
127    *    <li><b>Default:</b> <jk>true</jk>
128    * </ul>
129    * <p>
130    * When enabled, the number of times an exception has occurred will be determined based on stack trace hashsums,
131    *    made available through the {@link RestException#getOccurrence()} method.
132    * <p>
133    * Applicable to servlet class only.
134    */
135   public static final String REST_useStackTraceHashes = "RestServlet.useStackTraceHashes";
136
137   /**
138    * <b>Configuration property:</b>  Default character encoding.
139    * <p>
140    * <ul>
141    *    <li><b>Name:</b> <js>"RestServlet.defaultCharset"</js>
142    *    <li><b>Data type:</b> <code>String</code>
143    *    <li><b>Default:</b> <js>"utf-8"</js>
144    * </ul>
145    * <p>
146    * The default character encoding for the request and response if not specified on the request.
147    * <p>
148    * Applicable to servlet class and methods.
149    */
150   public static final String REST_defaultCharset = "RestServlet.defaultCharset";
151
152   /**
153    * <b>Configuration property:</b>  Expected format of request parameters.
154    * <p>
155    * <ul>
156    *    <li><b>Name:</b> <js>"RestServlet.paramFormat"</js>
157    *    <li><b>Data type:</b> <code>String</code>
158    *    <li><b>Default:</b> <js>"UON"</js>
159    * </ul>
160    * <p>
161    * Possible values:
162    * <ul class='spaced-list'>
163    *    <li><js>"UON"</js> - URL-Encoded Object Notation.<br>
164    *       This notation allows for request parameters to contain arbitrarily complex POJOs.
165    *    <li><js>"PLAIN"</js> - Plain text.<br>
166    *       This treats request parameters as plain text.<br>
167    *       Only POJOs directly convertable from <l>Strings</l> can be represented in parameters when using this mode.
168    * </ul>
169    * <p>
170    * Note that the parameter value <js>"(foo)"</js> is interpreted as <js>"(foo)"</js> when using plain mode, but
171    *    <js>"foo"</js> when using UON mode.
172    * <p>
173    * The format can also be specified per-parameter using the {@link FormData#format() @FormData.format()} and {@link Query#format() @Query.format()}
174    *    annotations.
175    * <p>
176    * Applicable to servlet class and methods.
177    */
178   public static final String REST_paramFormat = "RestServlet.paramFormat";

278   final boolean allowHeaderParams, allowBodyParam, renderResponseStackTraces, useStackTraceHashes;
279   final String defaultCharset, paramFormat;
280   final Set<String> allowMethodParams;
281
282   /**
283    * Constructor.
284    * <p>
285    * Typically only called from {@link ContextFactory#getContext(Class)}.
286    *
287    * @param cf The factory that created this context.
288    */
289   public RestServletContext(ContextFactory cf) {
290      super(cf);
291      allowHeaderParams = cf.getProperty(REST_allowHeaderParams, boolean.class, true);
292      allowBodyParam = cf.getProperty(REST_allowBodyParam, boolean.class, true);
293      renderResponseStackTraces = cf.getProperty(REST_renderResponseStackTraces, boolean.class, false);
294      useStackTraceHashes = cf.getProperty(REST_useStackTraceHashes, boolean.class, true);
295      defaultCharset = cf.getProperty(REST_defaultCharset, String.class, "utf-8");
296      paramFormat = cf.getProperty(REST_paramFormat, String.class, "");
297
298      Set<String> s = new LinkedHashSet<String>();
299      for (String m : StringUtils.split(cf.getProperty(REST_allowMethodParam, String.class, ""), ','))
300         if (m.equals("true"))  // For backwards compatibility when this was a boolean field.
301            s.add("*");
302         else
303            s.add(m.toUpperCase());
304      allowMethodParams = Collections.unmodifiableSet(s);
305   }
