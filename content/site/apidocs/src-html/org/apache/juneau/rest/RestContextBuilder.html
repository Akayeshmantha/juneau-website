044/**
045 * Defines the initial configuration of a <code>RestServlet</code> or <code>@RestResource</code> annotated object.
046 * 
047 * <p>
048 * An extension of the {@link ServletConfig} object used during servlet initialization.
049 * 
050 * <p>
051 * Provides access to the following initialized resources:
052 * <ul>
053 *    <li>{@link #getConfigFile()} - The external configuration file for this resource.
054 *    <li>{@link #getProperties()} - The modifiable configuration properties for this resource.
055 *    <li>{@link #getVarResolverBuilder()} - The variable resolver for this resource.
056 * </ul>
057 * 
058 * <p>
059 * Methods are provided for overriding or augmenting the information provided by the <ja>@RestResource</ja> annotation.
060 * In general, most information provided in the <ja>@RestResource</ja> annotation can be specified programmatically
061 * through calls on this object.
062 * 
063 * <p>
064 * To interact with this object, simply pass it in as a constructor argument or in an INIT hook.
065 * <p class='bcode'>
066 *    <jc>// Option #1 - Pass in through constructor.</jc>
067 *    <jk>public</jk> MyResource(RestContextBuilder builder) {
068 *          builder
069 *             .pojoSwaps(CalendarSwap.<jsf>RFC2822DTZ</jsf>.<jk>class</jk>)
070 *             .set(<jsf>PARSER_debug</jsf>, <jk>true</jk>);
071 *    }
072 * 
073 *    <jc>// Option #2 - Use an INIT hook.</jc>
074 *    <ja>@RestHook</ja>(<jsf>INIT</jsf>)
075 *    <jk>public void</jk> init(RestContextBuilder builder) <jk>throws</jk> Exception {
076 *          builder
077 *             .pojoSwaps(CalendarSwap.<jsf>RFC2822DTZ</jsf>.<jk>class</jk>)
078 *             .set(<jsf>PARSER_debug</jsf>, <jk>true</jk>);
079 *    }
080 * </p>
081 * 
082 * 
083 * <h5 class='section'>Documentation:</h5>
084 * <ul>
085 *    <li><a class="doclink" href="../../../../overview-summary.html#juneau-rest-server.RestContext">Overview &gt; RestContext</a>
086 * </ul>
087 */

125   /**
126    * Constructor.
127    * 
128    * @param config The servlet config passed into the servlet by the servlet container.
129    * @param resource The class annotated with <ja>@RestResource</ja>.
130    * @throws ServletException Something bad happened.
131    */
132   RestContextBuilder(ServletConfig config, Class<?> resourceClass, RestContext parentContext) throws ServletException {
133      this.inner = config;
134      this.resourceClass = resourceClass;
135      this.parentContext = parentContext;
136      
137      properties = new RestContextProperties();
138      logger(RestLoggerDefault.class);
139      staticFileResponseHeader("Cache-Control", "max-age=86400, public");
140      encoders(IdentityEncoder.INSTANCE);
141
142      try {
143
144         ConfigFileBuilder cfb = new ConfigFileBuilder();
145
146         htmlDocBuilder = new HtmlDocBuilder(properties);
147         configFile = cfb.build();
148         varResolverBuilder = new VarResolverBuilder()
149            .vars(
150               SystemPropertiesVar.class,
151               EnvVariablesVar.class,
152               ConfigFileVar.class,
153               IfVar.class,
154               SwitchVar.class,
155               CoalesceVar.class,
156               CoalesceAndRecurseVar.class
157            );
158
159         VarResolver vr = varResolverBuilder.build();
160
161         Map<Class<?>,RestResource> restResourceAnnotationsParentFirst = findAnnotationsMapParentFirst(RestResource.class, resourceClass);
162
163         // Find our config file.  It's the last non-empty @RestResource.config().
164         String configPath = "";
165         for (RestResource r : restResourceAnnotationsParentFirst.values())
166            if (! r.config().isEmpty())
167               configPath = r.config();
168         String cf = vr.resolve(configPath);
169         if (! cf.isEmpty())
170            configFile = cfb.build(cf);
171         configFile = configFile.getResolving(vr);
172
173         // Add our config file to the variable resolver.
174         varResolverBuilder.contextObject(ConfigFileVar.SESSION_config, configFile);
175         vr = varResolverBuilder.build();
176
177         // Add the servlet init parameters to our properties.
178         for (Enumeration<String> ep = config.getInitParameterNames(); ep.hasMoreElements();) {
179            String p = ep.nextElement();
180            String initParam = config.getInitParameter(p);
181            set(vr.resolve(p), vr.resolve(initParam));
182         }
183
184         // Load stuff from parent-to-child order.
185         // This allows child settings to overwrite parent settings.
186         for (Map.Entry<Class<?>,RestResource> e : restResourceAnnotationsParentFirst.entrySet()) {
187            Class<?> c = e.getKey();
188            RestResource r = e.getValue();
189            for (Property p : r.properties())
190               set(vr.resolve(p.name()), vr.resolve(p.value()));
191            for (String p : r.flags())
192               set(p, true);
193            serializers(r.serializers());
194            parsers(r.parsers());
195            encoders(r.encoders());
196            if (r.produces().length > 0)
197               produces(false, resolveVars(vr, r.produces()));
198            if (r.consumes().length > 0)
199               consumes(false, resolveVars(vr, r.consumes()));
200            defaultRequestHeaders(resolveVars(vr, r.defaultRequestHeaders()));
201            defaultResponseHeaders(resolveVars(vr, r.defaultResponseHeaders()));
202            responseHandlers(r.responseHandlers());
203            converters(r.converters());
204            guards(reverse(r.guards()));
205            children(r.children());
206            beanFilters((Object[])r.beanFilters());
207            pojoSwaps(r.pojoSwaps());
208            paramResolvers(r.paramResolvers());
209            if (r.serializerListener() != SerializerListener.Null.class)
210               serializerListener(r.serializerListener());
211            if (r.parserListener() != ParserListener.Null.class)
212               parserListener(r.parserListener());
213            contextPath(vr.resolve(r.contextPath()));
214            for (String mapping : r.staticFiles())
215               staticFiles(c, vr.resolve(mapping));
216            if (! r.messages().isEmpty())
217               messages(c, vr.resolve(r.messages()));
218            staticFileResponseHeaders(resolveVars(vr, r.staticFileResponseHeaders()));
219            if (! r.useClasspathResourceCaching().isEmpty())
220               useClasspathResourceCaching(Boolean.valueOf(vr.resolve(r.useClasspathResourceCaching())));
221            if (r.classpathResourceFinder() != ClasspathResourceFinder.Null.class)
222               classpathResourceFinder(r.classpathResourceFinder());
223            if (! r.path().isEmpty())
224               path(vr.resolve(r.path()));
225            if (! r.clientVersionHeader().isEmpty())
226               clientVersionHeader(vr.resolve(r.clientVersionHeader()));
227            if (r.resourceResolver() != RestResourceResolver.Null.class)
228               resourceResolver(r.resourceResolver());
229            if (r.logger() != RestLogger.Null.class)
230               logger(r.logger());
231            if (r.callHandler() != RestCallHandler.Null.class)
232               callHandler(r.callHandler());
233            if (r.infoProvider() != RestInfoProvider.Null.class)
234               infoProvider(r.infoProvider());
235            if (! r.allowHeaderParams().isEmpty())
236               allowHeaderParams(Boolean.valueOf(vr.resolve(r.allowHeaderParams())));
237            if (! r.allowedMethodParams().isEmpty())
238               allowedMethodParams(vr.resolve(r.allowedMethodParams()));
239            if (! r.allowBodyParam().isEmpty())
240               allowBodyParam(Boolean.valueOf(vr.resolve(r.allowBodyParam())));
241            if (! r.renderResponseStackTraces().isEmpty())
242               renderResponseStackTraces(Boolean.valueOf(vr.resolve(r.renderResponseStackTraces())));
243            if (! r.useStackTraceHashes().isEmpty())
244               useStackTraceHashes(Boolean.valueOf(vr.resolve(r.useStackTraceHashes())));
245            if (! r.defaultCharset().isEmpty())
246               defaultCharset(vr.resolve(r.defaultCharset()));
247            if (! r.maxInput().isEmpty())
248               maxInput(vr.resolve(r.maxInput()));
249            mimeTypes(resolveVars(vr, r.mimeTypes()));
250
251            HtmlDoc hd = r.htmldoc();
252            widgets(hd.widgets());
253            htmlDocBuilder.process(hd);
254         }
255
256         responseHandlers(
257            StreamableHandler.class,
258            WritableHandler.class,
259            ReaderHandler.class,
260            InputStreamHandler.class,
261            RedirectHandler.class,
262            DefaultHandler.class
263         );
264
265      } catch (Exception e) {
266         throw new ServletException(e);
267      }
268   }

270   private static String[] resolveVars(VarResolver vr, String[] in) {
271      String[] out = new String[in.length];
272      for (int i = 0; i < in.length; i++) 
273         out[i] = vr.resolve(in[i]);
274      return out;
275   }

277   /*
278    * Calls all @RestHook(INIT) methods on the specified resource object.
279    */
280   void init(Object resource) throws ServletException {
281      this.resource = resource;
282
283      // Once we have the resource object, we can construct the Widgets.
284      // We want to do that here so that we can update the script/style properties while they're still modifiable.
285      HtmlDocBuilder hdb = getHtmlDocBuilder();
286      PropertyStore ps = getPropertyStore();
287      Widget[] widgets = ps.getInstanceArrayProperty(REST_widgets, Widget.class, new Widget[0], true, ps, resource);
288      for (Widget w : widgets) {
289         hdb.script("INHERIT", "$W{"+w.getName()+".script}");
290         hdb.style("INHERIT", "$W{"+w.getName()+".style}");
291      }
292      widgets(false, widgets);
293      
294      Map<String,Method> map = new LinkedHashMap<>();
295      for (Method m : ClassUtils.getAllMethods(this.resourceClass, true)) {
296         if (m.isAnnotationPresent(RestHook.class) && m.getAnnotation(RestHook.class).value() == HookEvent.INIT) {
297            Visibility.setAccessible(m);
298            String sig = ClassUtils.getMethodSignature(m);
299            if (! map.containsKey(sig))
300               map.put(sig, m);
301         }
302      }
303      for (Method m : map.values()) {
304         ClassUtils.assertArgsOfType(m, RestContextBuilder.class, ServletConfig.class);
305         Class<?>[] argTypes = m.getParameterTypes();
306         Object[] args = new Object[argTypes.length];
307         for (int i = 0; i < args.length; i++) {
308            if (argTypes[i] == RestContextBuilder.class)
309               args[i] = this;
310            else
311               args[i] = this.inner;
312         }
313         try {
314            m.invoke(resource, args);
315         } catch (Exception e) {
316            throw new RestServletException("Exception thrown from @RestHook(INIT) method {0}.", m).initCause(e);
317         }
318      }
319   }

344   /**
345    * Adds a var context object to this config.
346    * 
347    * <p>
348    * Var context objects are read-only objects associated with the variable resolver for vars that require external
349    * information.
350    * 
351    * <p>
352    * For example, the {@link ConfigFileVar} needs access to this resource's {@link ConfigFile} through the
353    * {@link ConfigFileVar#SESSION_config} object that can be specified as either a session object (temporary) or
354    * context object (permanent).
355    * In this case, we call the following code to add it to the context map:
356    * <p class='bcode'>
357    *    config.addVarContextObject(<jsf>SESSION_config</jsf>, configFile);
358    * </p>
359    * 
360    * @param name The context object key (i.e. the name that the Var class looks for).
361    * @param object The context object.
362    * @return This object (for method chaining).
363    */
364   public RestContextBuilder varContextObject(String name, Object object) {
365      this.varResolverBuilder.contextObject(name, object);
366      return this;
367   }
