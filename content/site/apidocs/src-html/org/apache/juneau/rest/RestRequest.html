044/**
045 * Represents an HTTP request for a REST resource.
046 * <p>
047 * Equivalent to {@link HttpServletRequest} except with some additional convenience methods.
048 * <p>
049 * For reference, given the URL <js>"http://localhost:9080/contextRoot/servletPath/foo?bar=baz#qux"</js>, the
050 *    following methods return the following values....
051 * <table class='styled'>
052 *    <tr><th>Method</th><th>Value</th></tr>
053 *    <tr><td>{@code getContextPath()}</td><td>{@code /contextRoot}</td></tr>
054 *    <tr><td>{@code getPathInfo()}</td><td>{@code /foo}</td></tr>
055 *    <tr><td>{@code getPathTranslated()}</td><td>{@code path-to-deployed-war-on-filesystem/foo}</td></tr>
056 *    <tr><td>{@code getQueryString()}</td><td>{@code bar=baz}</td></tr>
057 *    <tr><td>{@code getRequestURI()}</td><td>{@code /contextRoot/servletPath/foo}</td></tr>
058 *    <tr><td>{@code getRequestURL()}</td><td>{@code http://localhost:9080/contextRoot/servletPath/foo}</td></tr>
059 *    <tr><td>{@code getServletPath()}</td><td>{@code /servletPath}</td></tr>
060 * </table>
061 * <p>
062 * Refer to <a class="doclink" href="package-summary.html#TOC">REST Servlet API</a> for information about using this class.

092   /**
093    * Constructor.
094    */
095   RestRequest(RestContext context, HttpServletRequest req) throws ServletException {
096      super(req);
097      this.context = context;
098
099      try {
100         isPost = req.getMethod().equalsIgnoreCase("POST");
101
102         // If this is a POST, we want to parse the query parameters ourselves to prevent
103         // the servlet code from processing the HTTP body as URL-Encoded parameters.
104         if (isPost)
105            queryParams = context.getUrlEncodingParser().parseIntoSimpleMap(getQueryString());
106         else {
107            queryParams = req.getParameterMap();
108         }
109
110         // Get the HTTP method.
111         // Can be overridden through a "method" GET attribute.
112         String _method = super.getMethod();
113
114         String m = getQueryParameter("method");
115         if (context.allowMethodParam(m))
116            _method = m;
117
118         method = _method;
119
120         if (context.isAllowBodyParam()) {
121            body = getQueryParameter("body");
122            if (body != null)
123               setHeader("Content-Type", UonSerializer.DEFAULT.getResponseContentType());
124         }
125
126         defaultServletHeaders = context.getDefaultRequestHeaders();
127
128         debug = "true".equals(getQueryParameter("debug", "false"));
129
130         if (debug) {
131            context.getLogger().log(Level.INFO, toString());
132         }
133
134      } catch (RestException e) {
135         throw e;
136      } catch (Exception e) {
137         throw new ServletException(e);
138      }
139   }

221    * Returns the specified header value, or a default value if the header doesn't exist.
222    * <p>
223    * If {@code allowHeaderParams} init parameter is <jk>true</jk>, then first looks for {@code &HeaderName=x} in the URL query string.
224    *
225    * @param name The HTTP header name.
226    * @param def The default value to return if the header value isn't found.
227    * @return The header value, or the default value if the header isn't present.
228    */
229   public String getHeader(String name, String def) {
230      String h = getOverriddenHeader(name);
231      if (h != null)
232         return h;
233      h = super.getHeader(name);
234      if (h != null && ! h.isEmpty())
235         return h;
236      if (defaultMethodHeaders != null) {
237         h = defaultMethodHeaders.get(name);
238         if (h != null)
239            return h;
240      }
241      h = defaultServletHeaders.get(name);
242      if (h != null)
243         return h;
244      return def;
245   }

393   /**
394    * Returns the charset specified on the <code>Content-Type</code> header, or <js>"UTF-8"</js> if not specified.
395    */
396   @Override /* ServletRequest */
397   public String getCharacterEncoding() {
398      if (charset == null) {
399         // Determine charset
400         // NOTE:  Don't use super.getCharacterEncoding() because the spec is implemented inconsistently.
401         // Jetty returns the default charset instead of null if the character is not specified on the request.
402         String h = getHeader("Content-Type");
403         if (h != null) {
404            int i = h.indexOf(";charset=");
405            if (i > 0)
406               charset = h.substring(i+9).trim();
407         }
408         if (charset == null)
409            charset = defaultCharset;
410         if (! Charset.isSupported(charset))
411            throw new RestException(SC_UNSUPPORTED_MEDIA_TYPE, "Unsupported charset in header ''Content-Type'': ''{0}''", h);
412      }
413      return charset;
414   }

432         if (mr.length > 0) {

437         }
438      }

459   /**
460    * Returns a query parameter value.
461    * <p>
462    * Same as {@link #getParameter(String)} except only looks in the URL string, not parameters from URL-Encoded FORM posts.
463    * <p>
464    * This method can be used to retrieve a parameter without triggering the underlying servlet API to load and parse the request body.
465    *
466    * @param name The URL parameter name.
467    * @return The parameter value, or <jk>null</jk> if parameter not specified or has no value (e.g. <js>"&amp;foo"</js>.
468    */
469   public String getQueryParameter(String name) {
470      String s = null;
471      if (overriddenQueryParams != null)
472         s = overriddenQueryParams.get(name);
473      if (s != null)
474         return s;
475      String[] v = queryParams.get(name);
476      if (v == null || v.length == 0)
477         return null;
478      if (v.length == 1 && v[0] != null && v[0].isEmpty()) {
479         // Fix for behavior difference between Tomcat and WAS.
480         // getParameter("foo") on "&foo" in Tomcat returns "".
481         // getParameter("foo") on "&foo" in WAS returns null.
482         if (queryParams.containsKey(name))
483            return null;
484      }
485      return v[0];
486   }

1286   <T> T getBody(ClassMeta<T> cm) throws RestException {
1287
1288      try {
1289         if (cm.isReader())
1290            return (T)getReader();
1291
1292         if (cm.isInputStream())
1293            return (T)getInputStream();
1294
1295         TimeZone timeZone = getTimeZone();
1296         Locale locale = getLocale();
1297         ParserMatch pm = getParserMatch();
1298
1299         if (pm != null) {
1300            Parser p = pm.getParser();
1301            MediaType mediaType = pm.getMediaType();
1302            try {
1303               properties.append("mediaType", mediaType).append("characterEncoding", getCharacterEncoding());
1304               if (! p.isReaderParser()) {
1305                  InputStreamParser p2 = (InputStreamParser)p;
1306                  ParserSession session = p2.createSession(getInputStream(), properties, getJavaMethod(), context.getResource(), locale, timeZone, mediaType);
1307                  return p2.parseSession(session, cm);
1308               }
1309               ReaderParser p2 = (ReaderParser)p;
1310               ParserSession session = p2.createSession(getUnbufferedReader(), properties, getJavaMethod(), context.getResource(), locale, timeZone, mediaType);
1311               return p2.parseSession(session, cm);
1312            } catch (ParseException e) {

1314                  "Could not convert request body content to class type ''{0}'' using parser ''{1}''.",

1321            "Unsupported media-type in request header ''Content-Type'': ''{0}''\n\tSupported media-types: {1}",

1325      } catch (IOException e) {

1360      s = s.substring(1);

1913   private <T> T parseParameter(String val, ClassMeta<T> c) throws ParseException {
1914      if (val == null)
1915         return null;
1916
1917      // Shortcut - If we're returning a string and the value doesn't start with "'" or is "null", then
1918      // just return the string since it's a plain value.
1919      // This allows us to bypass the creation of a UonParserSession object.
1920      if (c.getInnerClass() == String.class && val.length() > 0) {
1921         char x = val.charAt(0);
1922         if (x != '\'' && x != 'n' && val.indexOf('~') == -1)
1923            return (T)val;
1924         if (x == 'n' && "null".equals(val))
1925            return null;
1926      }
1927
1928      return urlEncodingParser.parseParameter(val, c);
1929   }

1944   private Encoder getEncoder() {
1945      if (encoder == null) {
1946         String ce = getHeader("content-encoding");
1947         if (! (ce == null || ce.isEmpty())) {
1948            ce = ce.trim();
1949            encoder = encoders.getEncoder(ce);
1950            if (encoder == null)
1951               throw new RestException(SC_UNSUPPORTED_MEDIA_TYPE,
1952                  "Unsupported encoding in request header ''Content-Encoding'': ''{0}''\n\tSupported codings: {1}",
1953                  getHeader("content-encoding"), encoders.getSupportedEncodings()
1954               );
1955         }
1956
1957         if (encoder != null)
1958            contentLength = -1;
1959      }
1960      // Note that if this is the identity encoder, we want to return null
1961      // so that we don't needlessly wrap the input stream.
1962      if (encoder == IdentityEncoder.INSTANCE)
1963         return null;
1964      return encoder;
1965   }
