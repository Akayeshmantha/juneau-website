031/**
032 * Represents an HTTP response for a REST resource.
033 *
034 * <p>
035 * Essentially an extended {@link HttpServletResponse} with some special convenience methods that allow you to easily
036 * output POJOs as responses.
037 *
038 * <p>
039 * Since this class extends {@link HttpServletResponse}, developers are free to use these convenience methods, or
040 * revert to using lower level methods like any other servlet response.
041 *
042 * <h5 class='section'>Example:</h5>
043 * <p class='bcode'>
044 *    <ja>@RestMethod</ja>(name=<jsf>GET</jsf>)
045 *    <jk>public void</jk> doGet(RestRequest req, RestResponse res) {
046 *       res.setPageTitle(<js>"My title"</js>)
047 *          .setOutput(<js>"Simple string response"</js>);
048 *    }
049 * </p>
050 *
051 * <p>
052 * Refer to <a class="doclink" href="package-summary.html#TOC">REST Servlet API</a> for information about using this
053 * class.
054 */
055public final class RestResponse extends HttpServletResponseWrapper {
056
057   private final RestRequest request;
058   private Object output;                       // The POJO being sent to the output.
059   private boolean isNullOutput;                // The output is null (as opposed to not being set at all)
060   private ObjectMap properties;                // Response properties
061   SerializerGroup serializerGroup;
062   UrlEncodingSerializer urlEncodingSerializer; // The serializer used to convert arguments passed into Redirect objects.
063   private EncoderGroup encoders;
064   private ServletOutputStream os;
065   private PrintWriter w;
066   private HtmlDocBuilder htmlDocBuilder;
067
068   /**
069    * Constructor.
070    */
071   RestResponse(RestContext context, RestRequest req, HttpServletResponse res) {
072      super(res);
073      this.request = req;
074
075      for (Map.Entry<String,Object> e : context.getDefaultResponseHeaders().entrySet())
076         setHeader(e.getKey(), e.getValue().toString());
077
078      try {
079         String passThroughHeaders = req.getHeader("x-response-headers");
080         if (passThroughHeaders != null) {
081            PartParser p = context.getUrlEncodingParser();
082            ObjectMap m = p.parse(PartType.HEADER, passThroughHeaders, context.getBeanContext().getClassMeta(ObjectMap.class));
083            for (Map.Entry<String,Object> e : m.entrySet())
084               setHeader(e.getKey(), e.getValue().toString());
085         }
086      } catch (Exception e1) {
087         throw new RestException(SC_BAD_REQUEST, "Invalid format for header 'x-response-headers'.  Must be in URL-encoded format.").initCause(e1);
088      }
089   }

091   /*
092    * Called from RestServlet after a match has been made but before the guard or method invocation.
093    */
094   @SuppressWarnings("hiding")
095   final void init(ObjectMap properties, String defaultCharset, SerializerGroup mSerializers, UrlEncodingSerializer mUrlEncodingSerializer, EncoderGroup encoders) {
096      this.properties = properties;
097      this.serializerGroup = mSerializers;
098      this.urlEncodingSerializer = mUrlEncodingSerializer;
099      this.encoders = encoders;
100
101      // Find acceptable charset
102      String h = request.getHeader("accept-charset");
103      String charset = null;
104      if (h == null)
105         charset = defaultCharset;
106      else for (MediaTypeRange r : MediaTypeRange.parse(h)) {
107         if (r.getQValue() > 0) {
108            MediaType mt = r.getMediaType();
109            if (mt.getType().equals("*"))
110               charset = defaultCharset;
111            else if (Charset.isSupported(mt.getType()))
112               charset = mt.getType();
113            if (charset != null)
114               break;
115         }
116      }
117
118      if (charset == null)
119         throw new RestException(SC_NOT_ACCEPTABLE, "No supported charsets in header ''Accept-Charset'': ''{0}''", request.getHeader("Accept-Charset"));
120      super.setCharacterEncoding(charset);
121   }

271   /**
272    * Equivalent to {@link HttpServletResponse#getOutputStream()}, except wraps the output stream if an {@link Encoder}
273    * was found that matched the <code>Accept-Encoding</code> header.
274    *
275    * @return A negotiated output stream.
276    * @throws IOException
277    */
278   public ServletOutputStream getNegotiatedOutputStream() throws IOException {
279      if (os == null) {
280         Encoder encoder = null;
281
282         String ae = request.getHeader("Accept-Encoding");
283         if (! (ae == null || ae.isEmpty())) {
284            EncoderMatch match = encoders != null ? encoders.getEncoderMatch(ae) : null;
285            if (match == null) {
286               // Identity should always match unless "identity;q=0" or "*;q=0" is specified.
287               if (ae.matches(".*(identity|\\*)\\s*;\\s*q\\s*=\\s*(0(?!\\.)|0\\.0).*")) {
288                  throw new RestException(SC_NOT_ACCEPTABLE,
289                     "Unsupported encoding in request header ''Accept-Encoding'': ''{0}''\n\tSupported codings: {1}",
290                     ae, encoders.getSupportedEncodings()
291                  );
292               }
293            } else {
294               encoder = match.getEncoder();
295               String encoding = match.getEncoding().toString();
296
297               // Some clients don't recognize identity as an encoding, so don't set it.
298               if (! encoding.equals("identity"))
299                  setHeader("content-encoding", encoding);
300            }
301         }
302         os = getOutputStream();
303         if (encoder != null) {
304            final OutputStream os2 = encoder.getOutputStream(os);
305            os = new ServletOutputStream(){
306               @Override /* OutputStream */
307               public final void write(byte[] b, int off, int len) throws IOException {
308                  os2.write(b, off, len);
309               }
310               @Override /* OutputStream */
311               public final void write(int b) throws IOException {
312                  os2.write(b);
313               }
314               @Override /* OutputStream */
315               public final void flush() throws IOException {
316                  os2.flush();
317               }
318               @Override /* OutputStream */
319               public final void close() throws IOException {
320                  os2.close();
321               }
322               @Override /* ServletOutputStream */
323               public boolean isReady() {
324                  return true;
325               }
326               @Override /* ServletOutputStream */
327               public void setWriteListener(WriteListener arg0) {
328                  throw new NoSuchMethodError();
329               }
330            };
331         }
332      }
333      return os;
334   }

336   @Override /* ServletResponse */
337   public ServletOutputStream getOutputStream() throws IOException {
338      if (os == null)
339         os = super.getOutputStream();
340      return os;

352   /**
353    * Returns the writer to the response body.
354    *
355    * <p>
356    * This methods bypasses any specified encoders and returns a regular unbuffered writer.
357    * Use the {@link #getNegotiatedWriter()} method if you want to use the matched encoder (if any).
358    */
359   @Override /* ServletResponse */
360   public PrintWriter getWriter() throws IOException {
361      return getWriter(true);
362   }

364   /**
365    * Convenience method meant to be used when rendering directly to a browser with no buffering.
366    *
367    * <p>
368    * Sets the header <js>"x-content-type-options=nosniff"</js> so that output is rendered immediately on IE and Chrome
369    * without any buffering for content-type sniffing.
370    *
371    * @param contentType The value to set as the <code>Content-Type</code> on the response.
372    * @return The raw writer.
373    * @throws IOException
374    */
375   public PrintWriter getDirectWriter(String contentType) throws IOException {
376      setContentType(contentType);
377      setHeader("x-content-type-options", "nosniff");
378      return getWriter();
379   }

381   /**
382    * Equivalent to {@link HttpServletResponse#getWriter()}, except wraps the output stream if an {@link Encoder} was
383    * found that matched the <code>Accept-Encoding</code> header and sets the <code>Content-Encoding</code>
384    * header to the appropriate value.
385    *
386    * @return The negotiated writer.
387    * @throws IOException
388    */
389   public PrintWriter getNegotiatedWriter() throws IOException {
390      return getWriter(false);
391   }
392
393   private PrintWriter getWriter(boolean raw) throws IOException {
394      if (w != null)
395         return w;
396
397      // If plain text requested, override it now.
398      if (request.isPlainText())
399         setHeader("Content-Type", "text/plain");
400
401      try {
402         OutputStream out = (raw ? getOutputStream() : getNegotiatedOutputStream());
403         w = new PrintWriter(new OutputStreamWriter(out, getCharacterEncoding()));
404         return w;
405      } catch (UnsupportedEncodingException e) {
406         String ce = getCharacterEncoding();
407         setCharacterEncoding("UTF-8");
408         throw new RestException(SC_NOT_ACCEPTABLE, "Unsupported charset in request header ''Accept-Charset'': ''{0}''", ce);
409      }
410   }

421   /**
422    * Redirects to the specified URI.
423    *
424    * <p>
425    * Relative URIs are always interpreted as relative to the context root.
426    * This is similar to how WAS handles redirect requests, and is different from how Tomcat handles redirect requests.
427    */
428   @Override /* ServletResponse */
429   public void sendRedirect(String uri) throws IOException {
430      char c = (uri.length() > 0 ? uri.charAt(0) : 0);
431      if (c != '/' && uri.indexOf("://") == -1)
432         uri = request.getContextPath() + '/' + uri;
433      super.sendRedirect(uri);
434   }

445   @Override /* ServletResponse */
446   public void setHeader(String name, String value) {
447      // Jetty doesn't set the content type correctly if set through this method.
448      // Tomcat/WAS does.
449      if (name.equalsIgnoreCase("Content-Type"))
450         super.setContentType(value);
451      else
452         super.setHeader(name, value);
453   }

456   @Override /* ServletResponse */
457   public void flushBuffer() throws IOException {
458      if (w != null)
459         w.flush();
460      if (os != null)
461         os.flush();
462      super.flushBuffer();
463   }
