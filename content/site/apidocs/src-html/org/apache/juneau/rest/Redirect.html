013package org.apache.juneau.rest;
014
015import static org.apache.juneau.internal.StringUtils.*;
016
017import java.net.*;
018import java.text.*;
019
020import org.apache.juneau.urlencoding.*;
021
022/**
023 * REST methods can return this object as a shortcut for performing <code>HTTP 302</code> redirects.
024 * 
025 * <p>
026 * The following example shows the difference between handling redirects via the {@link RestRequest}/{@link RestResponse},
027 * and the simplified approach of using this class.
028 * <p class='bcode'>
029 *    <jc>// Redirect to "/contextPath/servletPath/foobar"</jc>
030 * 
031 *    <jc>// Using RestRequest and RestResponse</jc>
032 *    <ja>@RestMethod</ja>(name=<jsf>GET</jsf>, path=<js>"/example1"</js>)
033 *    <jk>public void</jk> example1(RestRequest req, RestResponse res) <jk>throws</jk> IOException {
034 *       res.sendRedirect(req.getServletURI() + <js>"/foobar"</js>);
035 *    }
036 * 
037 *    <jc>// Using Redirect</jc>
038 *    <ja>@RestMethod</ja>(name=<jsf>GET</jsf>, path=<js>"/example2"</js>)
039 *    <jk>public</jk> Redirect example2() {
040 *       <jk>return new</jk> Redirect(<js>"foobar"</js>);
041 *    }
042 * </p>
043 * 
044 * <p>
045 * The constructor can use a {@link MessageFormat}-style pattern with multiple arguments:
046 * <p class='bcode'>
047 *    <ja>@RestMethod</ja>(name=<jsf>GET</jsf>, path=<js>"/example3"</js>)
048 *    <jk>public</jk> Redirect example3() {
049 *       <jk>return new</jk> Redirect(<js>"foo/{0}/bar/{1}"</js>, id1, id2);
050 *    }
051 * </p>
052 * 
053 * <p>
054 * The arguments are serialized to strings using the servlet's {@link UrlEncodingSerializer}, so any filters defined on
055 * the serializer or REST method/class will be used when present.
056 * The arguments will also be automatically URL-encoded.
057 * 
058 * <p>
059 * Redirecting to the servlet root can be accomplished by simply using the no-arg constructor.
060 * <p class='bcode'>
061 *    <jc>// Simply redirect to the servlet root.
062 *    // Equivalent to res.sendRedirect(req.getServletURI()).</jc>
063 *    <ja>@RestMethod</ja>(name=<jsf>GET</jsf>, path=<js>"/example4"</js>)
064 *    <jk>public</jk> Redirect exmaple4() {
065 *       <jk>return new</jk> Redirect();
066 *    }
067 * </p>
068 * 
069 * <p>
070 * This class is handled by {@link org.apache.juneau.rest.response.RedirectHandler}, a built-in default response
071 * handler created in {@link RestContextBuilder}.
072 */
073public final class Redirect {
074
075   private final int httpResponseCode;
076   private final URI uri;
077
078   /**
079    * Redirect to the specified URL.
080    * 
081    * <p>
082    * Relative paths are interpreted as relative to the servlet path.
083    * 
084    * @param uri
085    *    The URL to redirect to.
086    *    <br>Can be any of the following:
087    *    <ul>
088    *       <li><code>URL</code>
089    *       <li><code>URI</code>
090    *       <li><code>CharSequence</code>
091    *    </ul>
092    * @param args Optional {@link MessageFormat}-style arguments.
093    */
094   public Redirect(Object uri, Object...args) {
095      this(0, uri, args);
096   }
097
098   /**
099    * Convenience method for redirecting to instance of {@link URL} and {@link URI}.
100    * 
101    * <p>
102    * Same as calling <code>toString()</code> on the object and using the other constructor.
103    * 
104    * @param uri
105    *    The URL to redirect to.
106    *    <br>Can be any of the following:
107    *    <ul>
108    *       <li><code>URL</code>
109    *       <li><code>URI</code>
110    *       <li><code>CharSequence</code>
111    *    </ul>
112    */
113   public Redirect(Object uri) {
114      this(0, uri, (Object[])null);
115   }
116
117   /**
118    * Redirect to the specified URL.
119    * 
120    * <p>
121    * Relative paths are interpreted as relative to the servlet path.
122    * 
123    * @param httpResponseCode The HTTP response code.
124    * @param url
125    *    The URL to redirect to.
126    *    <br>Can be any of the following:
127    *    <ul>
128    *       <li><code>URL</code>
129    *       <li><code>URI</code>
130    *       <li><code>CharSequence</code>
131    *    </ul>
132    * @param args Optional {@link MessageFormat}-style arguments.
133    */
134   public Redirect(int httpResponseCode, Object url, Object...args) {
135      this.httpResponseCode = httpResponseCode;
136      if (url == null)
137         url = "";
138      this.uri = toURI(format(url.toString(), args));
139   }
140
141   /**
142    * Shortcut for redirecting to the servlet root.
143    */
144   public Redirect() {
145      this(0, null, (Object[])null);
146   }
147
148   /**
149    * Returns the response code passed in through the constructor.
150    * 
151    * @return The response code passed in through the constructor, or <code>0</code> if response code wasn't specified.
152    */
153   public int getHttpResponseCode() {
154      return httpResponseCode;
155   }
156
157   /**
158    * Returns the URI to redirect to.
159    * 
160    * @return The URI to redirect to.
161    */
162   public URI getURI() {
163      return uri;
164   }
165}
