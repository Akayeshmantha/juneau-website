013package org.apache.juneau.rest.annotation;
014
015import org.apache.juneau.rest.*;
016
017/**
018 * Extended annotation for {@link RestMethod#swagger() RestMethod.swagger()}.
019 * 
020 * <h5 class='section'>See Also:</h5>
021 * <ul>
022 *    <li class='link'><a class="doclink" href="../../../../../overview-summary.html#juneau-rest-server.OptionsPages">Overview &gt; OPTIONS Pages</a>
023 * </ul>
024 */
025public @interface MethodSwagger {
026
027   /**
028    * Optional deprecated flag for the exposed API.
029    * 
030    * <p>
031    * Used to populate the Swagger deprecated field.
032    * 
033    * <h5 class='section'>Example:</h5>
034    * <p class='bcode'>
035    *    <ja>@RestMethod</ja>(
036    *       swagger=<ja>@MethodSwagger</ja>(
037    *          deprecated=<jk>true</jk>
038    *       )
039    *    )
040    * </p>
041    * 
042    * <h5 class='section'>Notes:</h5>
043    * <ul class='spaced-list'>
044    *    <li>
045    *       Corresponds to the swagger field <code>/paths/{path}/{method}/deprecated</code>.
046    * </ul>
047    * 
048    * <h5 class='section'>See Also:</h5>
049    * <ul>
050    *    <li class='jm'>{@link RestInfoProviderDefault#isDeprecated(java.lang.reflect.Method, RestRequest) RestInfoProviderDefault.isDeprecated(Method,RestRequest)}
051    * </ul>
052    */
053   boolean deprecated() default false;
054
055   /**
056    * Optional external documentation information for the exposed API.
057    * 
058    * <p>
059    * Used to populate the Swagger external documentation field.
060    * 
061    * <p>
062    * A simplified JSON string with the following fields:
063    * <p class='bcode'>
064    *    {
065    *       description: string,
066    *       url: string
067    *    }
068    * </p>
069    * 
070    * <h5 class='section'>Example:</h5>
071    * <p class='bcode'>
072    *    <ja>@RestMethod</ja>(
073    *       swagger=<ja>@MethodSwagger</ja>(
074    *          <js>"{url:'http://juneau.apache.org'}"</js>
075    *       )
076    *    )
077    * </p>
078    * 
079    * <h5 class='section'>Notes:</h5>
080    * <ul class='spaced-list'>
081    *    <li>
082    *       Supports <a class="doclink" href="../../../../../overview-summary.html#DefaultRestSvlVariables">initialization-time and request-time variables</a> 
083    *       (e.g. <js>"$L{my.localized.variable}"</js>).
084    *    <li>
085    *       Corresponds to the swagger field <code>/paths/{path}/{method}/externalDocs</code>.
086    * </ul>
087    * 
088    * <h5 class='section'>See Also:</h5>
089    * <ul>
090    *    <li class='jm'>{@link RestInfoProviderDefault#getMethodExternalDocs(java.lang.reflect.Method, RestRequest) RestInfoProviderDefault.getMethodExternalDocs(Method,RestRequest)}
091    * </ul>
092    */
093   String externalDocs() default "";
094
095   /**
096    * Optional parameter descriptions.
097    * 
098    * <p>
099    * This annotation is provided for documentation purposes and is used to populate the method <js>"parameters"</js>
100    * column on the Swagger page.
101    * 
102    * <h5 class='section'>Example:</h5>
103    * <p class='bcode'>
104    *    <ja>@RestMethod</ja>(
105    *       name=<jsf>POST</jsf>, path=<js>"/{a}"</js>,
106    *       description=<js>"This is my method."</js>,
107    *       swagger=<ja>@MethodSwagger</ja>(
108    *          parameters={
109    *             <ja>@Parameter</ja>(in=<js>"path"</js>, name=<js>"a"</js>, description=<js>"The 'a' attribute"</js>),
110    *             <ja>@Parameter</ja>(in=<js>"query"</js>, name=<js>"b"</js>, description=<js>"The 'b' parameter"</js>, required=<jk>true</jk>),
111    *             <ja>@Parameter</ja>(in=<js>"body"</js>, description=<js>"The HTTP content"</js>),
112    *             <ja>@Parameter</ja>(in=<js>"header"</js>, name=<js>"D"</js>, description=<js>"The 'D' header"</js>),
113    *          }
114    *       )
115    *    )
116    * </p>
117    * 
118    * <h5 class='section'>Notes:</h5>
119    * <ul class='spaced-list'>
120    *    <li>
121    *       Supports <a class="doclink" href="../../../../../overview-summary.html#DefaultRestSvlVariables">initialization-time and request-time variables</a> 
122    *       (e.g. <js>"$L{my.localized.variable}"</js>).
123    *    <li>
124    *       Corresponds to the swagger field <code>/paths/{path}/{method}/parameters</code>.
125    * </ul>
126    * 
127    * <h5 class='section'>See Also:</h5>
128    * <ul>
129    *    <li class='jm'>{@link RestInfoProviderDefault#getMethodParameters(java.lang.reflect.Method, RestRequest) RestInfoProviderDefault.getMethodParameters(Method,RestRequest)}
130    * </ul>
131    */
132   Parameter[] parameters() default {};
133
134   /**
135    * Optional output description.
136    * 
137    * <p>
138    * This annotation is provided for documentation purposes and is used to populate the method <js>"responses"</js>
139    * column on the Swagger page.
140    * 
141    * <h5 class='section'>Example:</h5>
142    * <p class='bcode'>
143    *    <ja>@RestMethod</ja>(
144    *       name=<jsf>GET</jsf>, path=<js>"/"</js>,
145    *       swagger=<ja>@MethodSwagger</ja>(
146    *          responses={
147    *             <ja>@Response</ja>(200),
148    *             <ja>@Response</ja>(
149    *                value=302,
150    *                description=<js>"Thing wasn't found here"</js>,
151    *                headers={
152    *                   <ja>@Parameter</ja>(name=<js>"Location"</js>, description=<js>"The place to find the thing"</js>)
153    *                }
154    *             )
155    *          }
156    *       )
157    *    )
158    * </p>
159    * 
160    * <h5 class='section'>Notes:</h5>
161    * <ul class='spaced-list'>
162    *    <li>
163    *       Supports <a class="doclink" href="../../../../../overview-summary.html#DefaultRestSvlVariables">initialization-time and request-time variables</a> 
164    *       (e.g. <js>"$L{my.localized.variable}"</js>).
165    *    <li>
166    *       Corresponds to the swagger field <code>/paths/{path}/{method}/responses</code>.
167    * </ul>
168    * 
169    * <h5 class='section'>See Also:</h5>
170    * <ul>
171    *    <li class='jm'>{@link RestInfoProviderDefault#getMethodResponses(java.lang.reflect.Method, RestRequest) RestInfoProviderDefault.getMethodResponses(Method,RestRequest)}
172    * </ul>
173    */
174   Response[] responses() default {};
175
176   /**
177    * Optional tagging information for the exposed API.
178    * 
179    * <p>
180    * Used to populate the Swagger tags field.
181    * 
182    * <p>
183    * A comma-delimited list of tags for API documentation control.
184    * <br>Tags can be used for logical grouping of operations by resources or any other qualifier.
185    * 
186    * <h5 class='section'>Example:</h5>
187    * <p class='bcode'>
188    *    <ja>@RestMethod</ja>(
189    *       swagger=<ja>@MethodSwagger</ja>(
190    *          tags=<js>"foo,bar"</js>
191    *       )
192    *    )
193    * </p>
194    * 
195    * <h5 class='section'>Notes:</h5>
196    * <ul class='spaced-list'>
197    *    <li>
198    *       Supports <a class="doclink" href="../../../../../overview-summary.html#DefaultRestSvlVariables">initialization-time and request-time variables</a> 
199    *       (e.g. <js>"$L{my.localized.variable}"</js>).
200    *    <li>
201    *       Corresponds to the swagger field <code>/paths/{path}/{method}/tags</code>.
202    * </ul>
203    * 
204    * <h5 class='section'>See Also:</h5>
205    * <ul>
206    *    <li class='jm'>{@link RestInfoProviderDefault#getMethodTags(java.lang.reflect.Method, RestRequest) RestInfoProviderDefault.getMethodTags(Method,RestRequest)}
207    * </ul>
208    */
209   String tags() default "";
210}
