200   /**
201    * Configuration property:  Children.
202    *
203    * <h5 class='section'>Property:</h5>
204    * <ul>
205    *    <li><b>Name:</b>  <js>"RestContext.children.lo"</js>
206    *    <li><b>Data type:</b>  <code>List&lt;Class | Object | {@link RestChild}&gt;</code>
207    *    <li><b>Default:</b>  empty list
208    *    <li><b>Session-overridable:</b>  <jk>false</jk>
209    *    <li><b>Annotations:</b>
210    *       <ul>
211    *          <li class='ja'>{@link RestResource#children()}
212    *       </ul>
213    *    <li><b>Methods:</b>
214    *       <ul>
215    *          <li class='jm'>{@link RestContextBuilder#child(String,Object)}
216    *          <li class='jm'>{@link RestContextBuilder#children(Class...)}
217    *          <li class='jm'>{@link RestContextBuilder#children(Object...)}
218    *       </ul>
219    * </ul>
220    * 
221    * <h5 class='section'>Description:</h5>
222    * <p>
223    * Defines children of this resource.
224    *
225    * <p>
226    * A REST child resource is simply another servlet or object that is initialized as part of the parent resource and has a
227    * servlet path directly under the parent servlet path.
228    * <br>The main advantage to defining servlets as REST children is that you do not need to define them in the
229    * <code>web.xml</code> file of the web application.
230    * <br>This can cut down on the number of entries that show up in the <code>web.xml</code> file if you are defining
231    * large numbers of servlets.
232    *
233    * <p>
234    * Child resources must specify a value for {@link RestResource#path()} that identifies the subpath of the child resource
235    * relative to the parent path.
236    *
237    * <p>
238    * Child resources can be nested arbitrarily deep using this technique (i.e. children can also have children).
239    *
240    * <dl>
241    *    <dt>Servlet initialization:</dt>
242    *    <dd>
243    *       <p>
244    *          A child resource will be initialized immediately after the parent servlet is initialized.
245    *          The child resource receives the same servlet config as the parent resource.
246    *          This allows configuration information such as servlet initialization parameters to filter to child
247    *          resources.
248    *       </p>
249    *    </dd>
250    *    <dt>Runtime behavior:</dt>
251    *    <dd>
252    *       <p>
253    *          As a rule, methods defined on the <code>HttpServletRequest</code> object will behave as if the child
254    *          servlet were deployed as a top-level resource under the child's servlet path.
255    *          For example, the <code>getServletPath()</code> and <code>getPathInfo()</code> methods on the
256    *          <code>HttpServletRequest</code> object will behave as if the child resource were deployed using the
257    *          child's servlet path.
258    *          Therefore, the runtime behavior should be equivalent to deploying the child servlet in the
259    *          <code>web.xml</code> file of the web application.
260    *       </p>
261    *    </dd>
262    * </dl>
263    * 
264    * <h5 class='section'>Notes:</h5>
265    * <ul class='spaced-list'>
266    *    <li>When defined as classes, instances are resolved using the registered {@link #REST_resourceResolver} which
267    *       by default is {@link RestResourceResolverSimple} which requires the class have one of the following
268    *       constructors:
269    *       <ul>
270    *          <li><code><jk>public</jk> T(RestContextBuilder)</code>
271    *          <li><code><jk>public</jk> T()</code>
272    *       </ul>
273    * </ul>
274    *
275    * <h5 class='section'>Documentation:</h5>
276    * <ul>
277    *    <li><a class="doclink" href="package-summary.html#RestResources.Children">org.apache.juneau.rest &gt; Children</a>
278    * </ul>
279    */
280   public static final String REST_children = PREFIX + "children.lo";

2481                  rp[i] = new RestParamDefaults.MethodObject(method, t);
