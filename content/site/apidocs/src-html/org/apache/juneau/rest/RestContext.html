062    *
063    * <ul>
064    *    <li><b>Name:</b> <js>"RestServlet.allowHeaderParams"</js>
065    *    <li><b>Data type:</b> <code>Boolean</code>
066    *    <li><b>Default:</b> <jk>true</jk>
067    * </ul>
068    *
069    * <p>
070    * When enabled, headers such as <js>"Accept"</js> and <js>"Content-Type"</js> to be passed in as URL query
071    * parameters.
072    * For example:  <js>"?Accept=text/json&amp;Content-Type=text/json"</js>
073    *
074    * <p>
075    * Parameter names are case-insensitive.
076    *
077    * <p>
078    * Useful for debugging REST interface using only a browser.
079    *
080    * <p>
081    * Applicable to servlet class only.
082    */
083   public static final String REST_allowHeaderParams = "RestServlet.allowHeaderParams";

085   /**
086    * <b>Configuration property:</b>  Enable <js>"method"</js> URL parameter for specific HTTP methods.
087    *
088    * <ul>
089    *    <li><b>Name:</b> <js>"RestServlet.allowMethodParam"</js>
090    *    <li><b>Data type:</b> <code>String</code>
091    *    <li><b>Default:</b> <js>""</js>
092    * </ul>
093    *
094    * <p>
095    * When specified, the HTTP method can be overridden by passing in a <js>"method"</js> URL parameter on a regular
096    * GET request.
097    * For example:  <js>"?method=OPTIONS"</js>
098    *
099    * <p>
100    * Format is a comma-delimited list of HTTP method names that can be passed in as a method parameter.
101    * Parameter name is case-insensitive.
102    * Use "*" to represent all methods.
103    * For backwards compatibility, "true" also means "*".
104    *
105    * <p>
106    * Note that per the <a class="doclink"
107    * href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec9.html">HTTP specification</a>, special care should
108    * be taken when allowing non-safe (POST, PUT, DELETE) methods to be invoked through GET requests.
109    *
110    * <p>
111    * Applicable to servlet class only.
112    *
113    * <p>
114    * Example: <js>"HEAD,OPTIONS"</js>
115    */
116   public static final String REST_allowMethodParam = "RestServlet.allowMethodParam";

354      super(null);
355      RestException _initException = null;
356      try {
357         this.resource = resource;
358         this.config = config;

380         this.encoders = b.encoders;
381         this.supportedContentTypes = toObjectArray(b.supportedContentTypes, MediaType.class);
382         this.supportedAcceptTypes = toObjectArray(b.supportedAcceptTypes, MediaType.class);
383         this.clientVersionHeader = b.clientVersionHeader;
384         this.defaultRequestHeaders = Collections.unmodifiableMap(b.defaultRequestHeaders);
385         this.defaultResponseHeaders = Collections.unmodifiableMap(b.defaultResponseHeaders);
386         this.beanContext = b.beanContext;
387         this.converters = b.converters.toArray(new RestConverter[b.converters.size()]);
388         this.guards = b.guards.toArray(new RestGuard[b.guards.size()]);
389         this.responseHandlers = toObjectArray(b.responseHandlers, ResponseHandler.class);
390         this.mimetypesFileTypeMap = b.mimetypesFileTypeMap;

394         this.msgs = b.messageBundle;
395         this.childResources = Collections.synchronizedMap(new LinkedHashMap<String,RestContext>());  // Not unmodifiable on purpose so that children can be replaced.
396         this.logger = b.logger;
397         this.fullPath = b.fullPath;

399         this.htmlWidgets = Collections.unmodifiableMap(b.htmlWidgets);

401         this.htmlLinks = b.htmlLinks;

412         //----------------------------------------------------------------------------------------------------
413         // Initialize the child resources.
414         // Done after initializing fields above since we pass this object to the child resources.
415         //----------------------------------------------------------------------------------------------------
416         List<String> methodsFound = new LinkedList<String>();   // Temporary to help debug transient duplicate method issue.
417         Map<String,CallRouter.Builder> routers = new LinkedHashMap<String,CallRouter.Builder>();
418         Map<String,CallMethod> _javaRestMethods = new LinkedHashMap<String,CallMethod>();
419         Map<String,Method>
420            _startCallMethods = new LinkedHashMap<String,Method>(),
421            _preCallMethods = new LinkedHashMap<String,Method>(),
422            _postCallMethods = new LinkedHashMap<String,Method>(),
423            _endCallMethods = new LinkedHashMap<String,Method>(),
424            _postInitMethods = new LinkedHashMap<String,Method>(),
425            _postInitChildFirstMethods = new LinkedHashMap<String,Method>(),
426            _destroyMethods = new LinkedHashMap<String,Method>();
427         List<RestParam[]>
428            _preCallMethodParams = new ArrayList<RestParam[]>(),
429            _postCallMethodParams = new ArrayList<RestParam[]>();
430         List<Class<?>[]>
431            _startCallMethodParams = new ArrayList<Class<?>[]>(),
432            _endCallMethodParams = new ArrayList<Class<?>[]>(),
433            _postInitMethodParams = new ArrayList<Class<?>[]>(),
434            _postInitChildFirstMethodParams = new ArrayList<Class<?>[]>(),
435            _destroyMethodParams = new ArrayList<Class<?>[]>();
436
437         for (java.lang.reflect.Method method : resource.getClass().getMethods()) {
438            if (method.isAnnotationPresent(RestMethod.class)) {
439               RestMethod a = method.getAnnotation(RestMethod.class);
440               methodsFound.add(method.getName() + "," + a.name() + "," + a.path());
441               try {
442                  if (! Modifier.isPublic(method.getModifiers()))
443                     throw new RestServletException("@RestMethod method {0}.{1} must be defined as public.", this.getClass().getName(), method.getName());
444
445                  CallMethod sm = new CallMethod(resource, method, this);
446                  String httpMethod = sm.getHttpMethod();
447
448                  // PROXY is a special case where a method returns an interface that we
449                  // can perform REST calls against.
450                  // We override the CallMethod.invoke() method to insert our logic.
451                  if ("PROXY".equals(httpMethod)) {
452
453                     final ClassMeta<?> interfaceClass = beanContext.getClassMeta(method.getGenericReturnType());
454                     final Map<String,Method> remoteableMethods = interfaceClass.getRemoteableMethods();
455                     if (remoteableMethods.isEmpty())
456                        throw new RestException(SC_INTERNAL_SERVER_ERROR, "Method {0} returns an interface {1} that doesn't define any remoteable methods.", getMethodSignature(method), interfaceClass.getReadableName());
457
458                     sm = new CallMethod(resource, method, this) {
459
460                        @Override
461                        int invoke(String pathInfo, RestRequest req, RestResponse res) throws RestException {
462
463                           int rc = super.invoke(pathInfo, req, res);
464                           if (rc != SC_OK)
465                              return rc;
466
467                           final Object o = res.getOutput();
468
469                           if ("GET".equals(req.getMethod())) {
470                              res.setOutput(getMethodInfo(remoteableMethods.values()));
471                              return SC_OK;
472
473                           } else if ("POST".equals(req.getMethod())) {
474                              if (pathInfo.indexOf('/') != -1)
475                                 pathInfo = pathInfo.substring(pathInfo.lastIndexOf('/')+1);
476                              pathInfo = urlDecode(pathInfo);
477                              java.lang.reflect.Method m = remoteableMethods.get(pathInfo);
478                              if (m != null) {
479                                 try {
480                                    // Parse the args and invoke the method.
481                                    Parser p = req.getBody().getParser();
482                                    Object input = p.isReaderParser() ? req.getReader() : req.getInputStream();
483                                    Object output = m.invoke(o, p.parseArgs(input, m.getGenericParameterTypes()));
484                                    res.setOutput(output);
485                                    return SC_OK;
486                                 } catch (Exception e) {
487                                    throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
488                                 }
489                              }
490                           }
491                           return SC_NOT_FOUND;
492                        }
493                     };
494
495                     _javaRestMethods.put(method.getName(), sm);
496                     addToRouter(routers, "GET", sm);
497                     addToRouter(routers, "POST", sm);
498
499                  } else {
500                     _javaRestMethods.put(method.getName(), sm);
501                     addToRouter(routers, httpMethod, sm);
502                  }
503               } catch (RestServletException e) {
504                  throw new RestServletException("Problem occurred trying to serialize methods on class {0}, methods={1}", this.getClass().getName(), JsonSerializer.DEFAULT_LAX.serialize(methodsFound)).initCause(e);
505               }
506            }
507         }
508
509         for (Method m : ClassUtils.getAllMethods(resource.getClass(), true)) {
510            if (ClassUtils.isPublic(m) && m.isAnnotationPresent(RestHook.class)) {
511               HookEvent he = m.getAnnotation(RestHook.class).value();
512               String sig = ClassUtils.getMethodSignature(m);
513               switch(he) {
514                  case PRE_CALL: {
515                     if (! _preCallMethods.containsKey(sig)) {
516                        Visibility.setAccessible(m);
517                        _preCallMethods.put(sig, m);
518                        _preCallMethodParams.add(findParams(m, false, null, true));
519                     }
520                     break;
521                  }
522                  case POST_CALL: {
523                     if (! _postCallMethods.containsKey(sig)) {
524                        Visibility.setAccessible(m);
525                        _postCallMethods.put(sig, m);
526                        _postCallMethodParams.add(findParams(m, false, null, true));
527                     }
528                     break;
529                  }
530                  case START_CALL: {
531                     if (! _startCallMethods.containsKey(sig)) {
532                        Visibility.setAccessible(m);
533                        _startCallMethods.put(sig, m);
534                        _startCallMethodParams.add(m.getParameterTypes());
535                        ClassUtils.assertArgsOfType(m, HttpServletRequest.class, HttpServletResponse.class);
536                     }
537                     break;
538                  }
539                  case END_CALL: {
540                     if (! _endCallMethods.containsKey(sig)) {
541                        Visibility.setAccessible(m);
542                        _endCallMethods.put(sig, m);
543                        _endCallMethodParams.add(m.getParameterTypes());
544                        ClassUtils.assertArgsOfType(m, HttpServletRequest.class, HttpServletResponse.class);
545                     }
546                     break;
547                  }
548                  case POST_INIT: {
549                     if (! _postInitMethods.containsKey(sig)) {
550                        Visibility.setAccessible(m);
551                        _postInitMethods.put(sig, m);
552                        _postInitMethodParams.add(m.getParameterTypes());
553                        ClassUtils.assertArgsOfType(m, RestContext.class);
554                     }
555                     break;
556                  }
557                  case POST_INIT_CHILD_FIRST: {
558                     if (! _postInitChildFirstMethods.containsKey(sig)) {
559                        Visibility.setAccessible(m);
560                        _postInitChildFirstMethods.put(sig, m);
561                        _postInitChildFirstMethodParams.add(m.getParameterTypes());
562                        ClassUtils.assertArgsOfType(m, RestContext.class);
563                     }
564                     break;
565                  }
566                  case DESTROY: {
567                     if (! _destroyMethods.containsKey(sig)) {
568                        Visibility.setAccessible(m);
569                        _destroyMethods.put(sig, m);
570                        _destroyMethodParams.add(m.getParameterTypes());
571                        ClassUtils.assertArgsOfType(m, RestContext.class);
572                     }
573                     break;
574                  }
575                  default: // Ignore INIT
576               }
577            }
578         }

580         this.callMethods = Collections.unmodifiableMap(_javaRestMethods);
581         this.preCallMethods = _preCallMethods.values().toArray(new Method[_preCallMethods.size()]);
582         this.postCallMethods = _postCallMethods.values().toArray(new Method[_postCallMethods.size()]);
583         this.startCallMethods = _startCallMethods.values().toArray(new Method[_startCallMethods.size()]);
584         this.endCallMethods = _endCallMethods.values().toArray(new Method[_endCallMethods.size()]);
585         this.postInitMethods = _postInitMethods.values().toArray(new Method[_postInitMethods.size()]);
586         this.postInitChildFirstMethods = _postInitChildFirstMethods.values().toArray(new Method[_postInitChildFirstMethods.size()]);
587         this.destroyMethods = _destroyMethods.values().toArray(new Method[_destroyMethods.size()]);
588         this.preCallMethodParams = _preCallMethodParams.toArray(new RestParam[_preCallMethodParams.size()][]);
589         this.postCallMethodParams = _postCallMethodParams.toArray(new RestParam[_postCallMethodParams.size()][]);
590         this.startCallMethodParams = _startCallMethodParams.toArray(new Class[_startCallMethodParams.size()][]);
591         this.endCallMethodParams = _endCallMethodParams.toArray(new Class[_endCallMethodParams.size()][]);
592         this.postInitMethodParams = _postInitMethodParams.toArray(new Class[_postInitMethodParams.size()][]);
593         this.postInitChildFirstMethodParams = _postInitChildFirstMethodParams.toArray(new Class[_postInitChildFirstMethodParams.size()][]);
594         this.destroyMethodParams = _destroyMethodParams.toArray(new Class[_destroyMethodParams.size()][]);

596         Map<String,CallRouter> _callRouters = new LinkedHashMap<String,CallRouter>();
597         for (CallRouter.Builder crb : routers.values())
598            _callRouters.put(crb.getHttpMethodName(), crb.build());
599         this.callRouters = Collections.unmodifiableMap(_callRouters);
600
601         // Initialize our child resources.
602         resourceResolver = resolve(resource, RestResourceResolver.class, b.resourceResolver);
603         for (Object o : config.childResources) {
604            String path = null;
605            Object r = null;
606            if (o instanceof Pair) {
607               Pair<String,Object> p = (Pair<String,Object>)o;
608               path = p.first();
609               r = p.second();
610            } else if (o instanceof Class<?>) {
611               Class<?> c = (Class<?>)o;
612               // Don't allow specifying yourself as a child.  Causes an infinite loop.
613               if (c == config.resourceClass)
614                  continue;
615               r = c;
616            } else {
617               r = o;
618            }
619
620            RestConfig childConfig = null;
621
622            if (o instanceof Class) {
623               Class<?> oc = (Class<?>)o;
624               childConfig = new RestConfig(config.inner, oc, this);
625               r = resourceResolver.resolve(oc, childConfig);
626            } else {
627               r = o;
628               childConfig = new RestConfig(config.inner, o.getClass(), this);
629            }
630
631            childConfig.init(r);
632            if (r instanceof RestServlet)
633               ((RestServlet)r).innerInit(childConfig);
634            RestContext rc2 = new RestContext(r, servletContext, childConfig);
635            if (r instanceof RestServlet)
636               ((RestServlet)r).setContext(rc2);
637            path = childConfig.path;
638            childResources.put(path, rc2);
639         }
640
641         callHandler = config.callHandler == null ? new RestCallHandler(this) : resolve(resource, RestCallHandler.class, config.callHandler, this);
642         infoProvider = config.infoProvider == null ? new RestInfoProvider(this) : resolve(resource, RestInfoProvider.class, config.infoProvider, this);
643
644      } catch (RestException e) {
645         _initException = e;
646         throw e;
647      } catch (Exception e) {
648         _initException = new RestException(SC_INTERNAL_SERVER_ERROR, e);
649         throw e;
650      } finally {
651         initException = _initException;
652      }
653   }

946   /**
947    * Resolve a static resource file.
948    *
949    * <p>
950    * The location of static resources are defined via one of the following:
951    * <ul>
952    *    <li>{@link RestResource#staticFiles() @RestResource.staticFiles()} annotation.
953    *    <li>{@link RestConfig#addStaticFiles(Class, String)} method.
954    * </ul>
955    *
956    * @param pathInfo The unencoded path info.
957    * @return The resource, or <jk>null</jk> if the resource could not be resolved.
958    * @throws IOException
959    */
960   public StreamResource resolveStaticFile(String pathInfo) throws IOException {
961      if (! staticFilesCache.containsKey(pathInfo)) {
962         String p = urlDecode(trimSlashes(pathInfo));
963         if (p.indexOf("..") != -1)
964            throw new RestException(SC_NOT_FOUND, "Invalid path");
965         for (Map.Entry<String,String> e : staticFilesMap.entrySet()) {
966            String key = trimSlashes(e.getKey());
967            if (p.startsWith(key)) {
968               String remainder = (p.equals(key) ? "" : p.substring(key.length()));
969               if (remainder.isEmpty() || remainder.startsWith("/")) {
970                  String p2 = trimSlashes(e.getValue()) + remainder;
971                  InputStream is = getResource(p2, null);
972                  if (is != null) {
973                     try {
974                        int i = p2.lastIndexOf('/');
975                        String name = (i == -1 ? p2 : p2.substring(i+1));
976                        String mediaType = mimetypesFileTypeMap.getContentType(name);
977                        ObjectMap headers = new ObjectMap().append("Cache-Control", "max-age=86400, public");
978                        staticFilesCache.put(pathInfo, new StreamResource(MediaType.forString(mediaType), headers, is));
979                        return staticFilesCache.get(pathInfo);
980                     } finally {
981                        is.close();
982                     }
983                  }
984               }
985            }
986         }
987      }
988      return staticFilesCache.get(pathInfo);
989   }

1551   /**
1552    * Finds the {@link RestParam} instances to handle resolving objects on the calls to the specified Java method.
1553    *
1554    * @param method The Java method being called.
1555    * @param methodPlainParams Whether plain-params setting is specified.
1556    * @param pathPattern The parsed URL path pattern.
1557    * @param isPreOrPost Whether this is a <ja>@RestMethodPre</ja> or <ja>@RestMethodPost</ja>.
1558    * @return The array of resolvers.
1559    * @throws ServletException If an annotation usage error was detected.
1560    */
1561   protected RestParam[] findParams(Method method, boolean methodPlainParams, UrlPathPattern pathPattern, boolean isPreOrPost) throws ServletException {
1562
1563      Type[] pt = method.getGenericParameterTypes();
1564      Annotation[][] pa = method.getParameterAnnotations();
1565      RestParam[] rp = new RestParam[pt.length];
1566      int attrIndex = 0;
1567
1568      for (int i = 0; i < pt.length; i++) {
1569
1570         Type t = pt[i];
1571         if (t instanceof Class) {
1572            Class<?> c = (Class<?>)t;
1573            rp[i] = paramResolvers.get(c);
1574            if (rp[i] == null)
1575               rp[i] = RestParamDefaults.STANDARD_RESOLVERS.get(c);
1576         }
1577
1578         if (rp[i] == null) {
1579            for (Annotation a : pa[i]) {
1580               if (a instanceof Header)
1581                  rp[i] = new RestParamDefaults.HeaderObject((Header)a, t);
1582               else if (a instanceof FormData)
1583                  rp[i] = new RestParamDefaults.FormDataObject(method, (FormData)a, t, methodPlainParams);
1584               else if (a instanceof Query)
1585                  rp[i] = new RestParamDefaults.QueryObject(method, (Query)a, t, methodPlainParams);
1586               else if (a instanceof HasFormData)
1587                  rp[i] = new RestParamDefaults.HasFormDataObject(method, (HasFormData)a, t);
1588               else if (a instanceof HasQuery)
1589                  rp[i] = new RestParamDefaults.HasQueryObject(method, (HasQuery)a, t);
1590               else if (a instanceof Body)
1591                  rp[i] = new RestParamDefaults.BodyObject(t);
1592               else if (a instanceof org.apache.juneau.rest.annotation.Method)
1593                  rp[i] = new RestParamDefaults.MethodObject(method, t);
1594               else if (a instanceof PathRemainder)
1595                  rp[i] = new RestParamDefaults.PathRemainderObject(method, t);
1596               else if (a instanceof Properties)
1597                  rp[i] = new RestParamDefaults.PropsObject(method, t);
1598               else if (a instanceof Messages)
1599                  rp[i] = new RestParamDefaults.MessageBundleObject();
1600            }
1601         }
1602
1603         if (rp[i] == null) {
1604
1605            if (isPreOrPost)
1606               throw new RestServletException("Invalid parameter specified for method ''{0}'' at index position {1}", method, i);
1607
1608            Path p = null;
1609            for (Annotation a : pa[i])
1610               if (a instanceof Path)
1611                  p = (Path)a;
1612
1613            String name = (p == null ? "" : firstNonEmpty(p.name(), p.value()));
1614
1615            if (isEmpty(name)) {
1616               int idx = attrIndex++;
1617               String[] vars = pathPattern.getVars();
1618               if (vars.length <= idx)
1619                  throw new RestServletException("Number of attribute parameters in method ''{0}'' exceeds the number of URL pattern variables.", method);
1620
1621               // Check for {#} variables.
1622               String idxs = String.valueOf(idx);
1623               for (int j = 0; j < vars.length; j++)
1624                  if (isNumeric(vars[j]) && vars[j].equals(idxs))
1625                     name = vars[j];
1626
1627               if (isEmpty(name))
1628                  name = pathPattern.getVars()[idx];
1629            }
1630            rp[i] = new RestParamDefaults.PathParameterObject(name, t);
1631         }
1632      }
1633
1634      return rp;
1635   }

1988         try {
1989            postInitOrDestroy(resource, destroyMethods[i], destroyMethodParams[i]);
1990         } catch (Exception e) {
1991            e.printStackTrace();
1992         }
1993      }
1994
1995      for (RestContext r : childResources.values()) {
1996         r.destroy();
1997         if (r.resource instanceof Servlet)
