346   /**
347    * Configuration property:  Children.
348    * 
349    * <h5 class='section'>Property:</h5>
350    * <ul>
351    *    <li><b>Name:</b>  <js>"RestContext.children.lo"</js>
352    *    <li><b>Data type:</b>  <code>List&lt;Class | Object | {@link RestChild}&gt;</code>
353    *    <li><b>Default:</b>  empty list
354    *    <li><b>Session-overridable:</b>  <jk>false</jk>
355    *    <li><b>Annotations:</b>
356    *       <ul>
357    *          <li class='ja'>{@link RestResource#children()}
358    *       </ul>
359    *    <li><b>Methods:</b>
360    *       <ul>
361    *          <li class='jm'>{@link RestContextBuilder#child(String,Object)}
362    *          <li class='jm'>{@link RestContextBuilder#children(Class...)}
363    *          <li class='jm'>{@link RestContextBuilder#children(Object...)}
364    *       </ul>
365    * </ul>
366    * 
367    * <h5 class='section'>Description:</h5>
368    * <p>
369    * Defines children of this resource.
370    * 
371    * <p>
372    * A REST child resource is simply another servlet or object that is initialized as part of the parent resource and has a
373    * servlet path directly under the parent servlet path.
374    * <br>The main advantage to defining servlets as REST children is that you do not need to define them in the
375    * <code>web.xml</code> file of the web application.
376    * <br>This can cut down on the number of entries that show up in the <code>web.xml</code> file if you are defining
377    * large numbers of servlets.
378    * 
379    * <p>
380    * Child resources must specify a value for {@link RestResource#path() @RestResource.path()} that identifies the subpath of the child resource
381    * relative to the parent path UNLESS you use the {@link RestContextBuilder#child(String, Object)} method to register it.
382    * 
383    * <p>
384    * Child resources can be nested arbitrarily deep using this technique (i.e. children can also have children).
385    * 
386    * <dl>
387    *    <dt>Servlet initialization:</dt>
388    *    <dd>
389    *       <p>
390    *          A child resource will be initialized immediately after the parent servlet is initialized.
391    *          <br>The child resource receives the same servlet config as the parent resource.
392    *          <br>This allows configuration information such as servlet initialization parameters to filter to child
393    *          resources.
394    *       </p>
395    *    </dd>
396    *    <dt>Runtime behavior:</dt>
397    *    <dd>
398    *       <p>
399    *          As a rule, methods defined on the <code>HttpServletRequest</code> object will behave as if the child
400    *          servlet were deployed as a top-level resource under the child's servlet path.
401    *          <br>For example, the <code>getServletPath()</code> and <code>getPathInfo()</code> methods on the
402    *          <code>HttpServletRequest</code> object will behave as if the child resource were deployed using the
403    *          child's servlet path.
404    *          <br>Therefore, the runtime behavior should be equivalent to deploying the child servlet in the
405    *          <code>web.xml</code> file of the web application.
406    *       </p>
407    *    </dd>
408    * </dl>
409    * 
410    * <h5 class='section'>Notes:</h5>
411    * <ul class='spaced-list'>
412    *    <li>When defined as classes, instances are resolved using the registered {@link #REST_resourceResolver} which
413    *       by default is {@link RestResourceResolverSimple} which requires the class have one of the following
414    *       constructors:
415    *       <ul>
416    *          <li><code><jk>public</jk> T(RestContextBuilder)</code>
417    *          <li><code><jk>public</jk> T()</code>
418    *       </ul>
419    * </ul>
420    * 
421    * <h5 class='section'>Example:</h5>
422    * <p class='bcode'>
423    *    <jc>// Our child resource.</jc>
424    *    <ja>@RestResource</ja>(path=<js>"/child"</js>)
425    *    <jk>public class</jk> MyChildResource {...}
426    * 
427    *    <jc>// Registered via annotation.</jc>
428    *    <ja>@RestResource</ja>(children={MyChildResource.<jk>class</jk>})
429    *    <jk>public class</jk> MyResource {...}
430    * 
431    *    <jc>// Registered via builder passed in through resource constructor.</jc>
432    *    <jk>public class</jk> MyResource {
433    *       <jk>public</jk> MyResource(RestContextBuilder builder) <jk>throws</jk> Exception {
434    *          
435    *          <jc>// Using method on builder.</jc>
436    *          builder.children(MyChildResource.<jk>class</jk>);
437    * 
438    *          <jc>// Same, but using property.</jc>
439    *          builder.addTo(<jsf>REST_children</jsf>, MyChildResource.<jk>class</jk>));
440    * 
441    *          <jc>// Use a pre-instantiated object instead.</jc>
442    *          builder.child(<js>"/child"</js>, <jk>new</jk> MyChildResource());
443    *       }
444    *    }
445    * 
446    *    <jc>// Registered via builder passed in through init method.</jc>
447    *    <jk>public class</jk> MyResource {
448    *       <ja>@RestHook</ja>(<jsf>INIT</jsf>)
449    *       <jk>public void</jk> init(RestContextBuilder builder) <jk>throws</jk> Exception {
450    *          builder.children(MyChildResource.<jk>class</jk>);
451    *       }
452    *    }
453    * 
454    * <h5 class='section'>Documentation:</h5>
455    * <ul>
456    *    <li><a class="doclink" href="package-summary.html#RestResources.Children">org.apache.juneau.rest &gt; Children</a>
457    * </ul>
458    */
459   public static final String REST_children = PREFIX + "children.lo";

2511    *    <li class='jf'>{@link RestContext#REST_staticFiles RestContext.REST_staticFiles}
2512    *    <li class='jc'>{@link RestContextBuilder#staticFiles(Class,String)}
2513    * </ul>
2514    * 
2515    * @param pathInfo The unencoded path info.
2516    * @return The resource, or <jk>null</jk> if the resource could not be resolved.
2517    * @throws IOException
2518    */
2519   public StreamResource resolveStaticFile(String pathInfo) throws IOException {
2520      if (! staticFilesCache.containsKey(pathInfo)) {
2521         String p = urlDecode(trimSlashes(pathInfo));
2522         if (p.indexOf("..") != -1)
2523            throw new RestException(SC_NOT_FOUND, "Invalid path");
2524         for (StaticFileMapping sfm : staticFiles) {
2525            String path = sfm.path;
2526            if (p.startsWith(path)) {
2527               String remainder = (p.equals(path) ? "" : p.substring(path.length()));
2528               if (remainder.isEmpty() || remainder.startsWith("/")) {
2529                  String p2 = sfm.location + remainder;
2530                  try (InputStream is = getClasspathResource(sfm.resourceClass, p2, null)) {
2531                     if (is != null) {
2532                        int i = p2.lastIndexOf('/');
2533                        String name = (i == -1 ? p2 : p2.substring(i+1));
2534                        String mediaType = mimetypesFileTypeMap.getContentType(name);
2535                        Map<String,Object> responseHeaders = sfm.responseHeaders != null ? sfm.responseHeaders : staticFileResponseHeaders;
2536                        staticFilesCache.put(pathInfo, new StreamResource(MediaType.forString(mediaType), responseHeaders, is));
2537                        return staticFilesCache.get(pathInfo);
2538                     }
2539                  }
2540               }
2541            }
2542         }
2543      }
2544      return staticFilesCache.get(pathInfo);
2545   }
