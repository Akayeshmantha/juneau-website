029/**
030 * Subclass of {@link RestServlet} with default serializers and parsers defined.
031 * <p>
032 *    Supports the following request <code>Accept</code> header values with the resulting response <code>Content-Type</code>:
033 * </p>
034 * <table class='styled'>
035 *    <tr>
036 *       <th>Accept</th>
037 *       <th>Content-Type</th>
038 *       <th>Serializer</th>
039 *    </tr>
040 *    <tr>
041 *       <td class='code'>application/json<br>text/json</td>
042 *       <td class='code'>application/json</td>
043 *       <td>{@link JsonSerializer}</td>
044 *    </tr>
045 *    <tr>
046 *       <td class='code'>application/json+simple<br>text/json+simple</td>
047 *       <td class='code'>application/json</td>
048 *       <td>{@link org.apache.juneau.json.JsonSerializer.Simple}</td>
049 *    </tr>
050 *       <td class='code'>application/json+schema<br>text/json+schema</td>
051 *       <td class='code'>application/json</td>
052 *       <td>{@link JsonSchemaSerializer}</td>
053 *    </tr>
054 *    <tr>
055 *       <td class='code'>text/xml</td>
056 *       <td class='code'>text/xml</td>
057 *       <td>{@link XmlDocSerializer}</td>
058 *    </tr>
059 *    <tr>
060 *       <td class='code'>text/xml+schema</td>
061 *       <td class='code'>text/xml</td>
062 *       <td>{@link XmlSchemaDocSerializer}</td>
063 *    </tr>
064 *    <tr>
065 *       <td class='code'>text/html</td>
066 *       <td class='code'>text/html</td>
067 *       <td>{@link HtmlDocSerializer}</td>
068 *    </tr>
069 *    <tr>
070 *       <td class='code'>text/html+stripped</td>
071 *       <td class='code'>text/html</td>
072 *       <td>{@link HtmlStrippedDocSerializer}</td>
073 *    </tr>
074 *    <tr>
075 *       <td class='code'>text/uon</td>
076 *       <td class='code'>text/uon</td>
077 *       <td>{@link UonSerializer}</td>
078 *    </tr>
079 *    <tr>
080 *       <td class='code'>text/uon-simple</td>
081 *       <td class='code'>text/uon</td>
082 *       <td>{@link org.apache.juneau.urlencoding.UonSerializer.Simple}</td>
083 *    </tr>
084 *    <tr>
085 *       <td class='code'>application/x-www-form-urlencoded</td>
086 *       <td class='code'>application/x-www-form-urlencoded</td>
087 *       <td>{@link UrlEncodingSerializer}</td>
088 *    </tr>
089 *    <tr>
090 *       <td class='code'>application/x-www-form-urlencoded-simple</td>
091 *       <td class='code'>application/x-www-form-urlencoded</td>
092 *       <td>{@link org.apache.juneau.urlencoding.UrlEncodingSerializer.Simple}</td>
093 *    </tr>
094 *    <tr>
095 *       <td class='code'>text/xml+soap</td>
096 *       <td class='code'>text/xml</td>
097 *       <td>{@link SoapXmlSerializer}</td>
098 *    </tr>
099 *    <tr>
100 *       <td class='code'>text/plain</td>
101 *       <td class='code'>text/plain</td>
102 *       <td>{@link PlainTextSerializer}</td>
103 *    </tr>
104 *    <tr>
105 *       <td class='code'>application/x-java-serialized-object</td>
106 *       <td class='code'>application/x-java-serialized-object</td>
107 *       <td>{@link JavaSerializedObjectSerializer}</td>
108 *    </tr>
109 * </table>
110 * <p>
111 *    Supports the following request <code>Content-Type</code> header values:
112 * </p>
113 * <table class='styled'>
114 *    <tr>
115 *       <th>Content-Type</th>
116 *       <th>Parser</th>
117 *    </tr>
118 *    <tr>
119 *       <td class='code'>application/json<br>text/json</td>
120 *       <td>{@link JsonParser}</td>
121 *    </tr>
122 *    <tr>
123 *       <td class='code'>text/xml<br>application/xml</td>
124 *       <td>{@link XmlParser}</td>
125 *    </tr>
126 *    <tr>
127 *       <td class='code'>text/html<br>text/html+stripped</td>
128 *       <td>{@link HtmlParser}</td>
129 *    </tr>
130 *    <tr>
131 *       <td class='code'>text/uon</td>
132 *       <td>{@link UonParser}</td>
133 *    </tr>
134 *    <tr>
135 *       <td class='code'>application/x-www-form-urlencoded</td>
136 *       <td>{@link UrlEncodingParser}</td>
137 *    </tr>
138 *    <tr>
139 *       <td class='code'>text/plain</td>
140 *       <td>{@link PlainTextParser}</td>
141 *    </tr>
142 * </table>
143 * <p>
144 *    It should be noted that we do NOT add {@link JavaSerializedObjectParser} to the list of parsers since this could
145 *       cause security issues.  Use caution when using this particular parser as it could inadvertantly cause
146 *       code execution security holes.
147 * </p>
148 * <p>
149 *    The list of serializers and parsers can be appended to using the {@link RestResource#serializers() @RestResource.serializers()}
150 *       and {@link RestResource#parsers() @RestResource.parsers()} annotations on subclasses.
151 * </p>
152 * <p>
153 *    This subclass also provides a default OPTIONS page by implementing a {@link #getOptions(RestRequest)} that returns a POJO consisting
154 *       of beans describing the class.
155 * </p>
156 * <img class='bordered' src='doc-files/OptionsPage.png'>
157 * <p>
158 *    The OPTIONS page can be modified or augmented by overriding this method and providing your own data.
159 * </p>
160 *
161 * <h6 class='topic'>Other Notes</h6>
162 * <ul class='spaced-list'>
163 *    <li>Provides a default HTML stylesheet by setting {@link RestResource#stylesheet() @RestResource.stylesheet()} to <js>"styles/juneau.css"</js>.
164 *    <li>Provides a default favicon by setting {@link RestResource#favicon() @RestResource.favicon()} to <js>"juneau.ico"</js>.
165 *    <li>Provides a default classpath entry "htdocs" by setting {@link RestResource#staticFiles() @RestResource.staticFiles()} to <js>"{htdocs:'htdocs'}"</js>.
166 *       This allows files inside the <code>[servletPackage].htdocs</code> package to be served up under the URL <code>/servletPath/htdocs</code>.
167 * </ul>
168 */
169@RestResource(
170   serializers={
171      HtmlDocSerializer.class, // HTML must be listed first because Internet Explore does not include text/html in their Accept header.
172      HtmlStrippedDocSerializer.class,
173      HtmlSchemaDocSerializer.class,
174      JsonSerializer.class,
175      JsonSerializer.Simple.class,
176      JsonSchemaSerializer.class,
177      XmlDocSerializer.class,
178      XmlSchemaDocSerializer.class,
179      UonSerializer.class,
180      UonSerializer.Simple.class,
181      UrlEncodingSerializer.class,
182      UrlEncodingSerializer.Simple.class,
183      MsgPackSerializer.class,
184      SoapXmlSerializer.class,
185      PlainTextSerializer.class,
186      JavaSerializedObjectSerializer.class
187   },
188   parsers={
189      JsonParser.class,
190      XmlParser.class,
191      HtmlParser.class,
192      UonParser.class,
193      UrlEncodingParser.class,
194      MsgPackParser.class,
195      PlainTextParser.class
196   },
197   properties={
198      // Allow &method parameter on safe HTTP methods.
199      @Property(name=REST_allowMethodParam, value="OPTIONS"),
200      // Provide a default title on HTML pages.
201      @Property(name=HTMLDOC_title, value="$R{servletTitle}"),
202      // Provide a default description on HTML pages.
203      @Property(name=HTMLDOC_description, value="$R{servletDescription}")
204   },
205   stylesheet="styles/juneau.css",
206   favicon="juneau.ico",
207   staticFiles="{htdocs:'htdocs'}"
208)
209public abstract class RestServletDefault extends RestServlet {
210   private static final long serialVersionUID = 1L;
211
212   /**
213    * [OPTIONS /*] - Show resource options.
214    *
215    * @param req The HTTP request.
216    * @return A bean containing the contents for the OPTIONS page.
217    */
218   @RestMethod(name="OPTIONS", path="/*",
219      properties={
220         @Property(name=HTMLDOC_links, value="{back:'$R{servletURI}'}"),
221         @Property(name=HTMLDOC_description, value="Resource options")
222      },
223      description="Resource options"
224   )
225   public Swagger getOptions(RestRequest req) {
226      return req.getSwagger();
227   }
228
229   @Override /* RestServlet */
230   public boolean hasOptionsPage() {
231      return true;
232   }
233}
