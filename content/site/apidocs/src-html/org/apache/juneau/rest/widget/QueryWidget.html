019/**
020 * Adds a <code>QUERY</code> link to the page that allows you to perform search/view/sort/paging on the page data.
021 *
022 * <p>
023 * A typical usage of the query widget is to include it as a navigation link as shown in the example below
024 * pulled from the <code>PetStoreResource</code> example:
025 * <p class='bcode'>
026 *    <ja>@RestResource</ja>(
027 *       widgets={
028 *          QueryWidget.<jk>class</jk>
029 *       },
030 *       htmldoc=<ja>@HtmlDoc</ja>(
031 *          links=<js>"{up:'...',options:'...',query:'$W{query}',source:'...'}"</js>
032 *       )
033 *    )
034 * </p>
035 *
036 * <p>
037 * In the above example, this adds a <code>QUERY</code> that displays a search popup that can be used for filtering the
038 * page results...
039 * <img class='bordered' src='doc-files/PetStore_Query.png'>
040 *
041 * <p>
042 * Tooltips are provided by hovering over the field names.
043 * <img class='bordered' src='doc-files/PetStore_Query_tooltip.png'>
044 *
045 * <p>
046 * When submitted, the form submits a GET request against the current URI with special GET search API query parameters.
047 * (e.g. <js>"?s=column1=Foo*&amp;v=column1,column2&amp;o=column1,column2-&amp;p=100&amp;l=100"</js>).
048 *
049 * <p>
050 * The search arguments can be retrieved programmatically using {@link RequestQuery#getSearchArgs()}.
051 *
052 * <p>
053 * Typically, the search functionality is implemented by applying the predefined {@link Queryable} converter on the
054 * method that's returning a 2-dimensional table of POJOs that you wish to filter:
055 * <p class='bcode'>
056 *    <ja>@RestMethod</ja>(
057 *       name=<js>"GET"</js>,
058 *       path=<js>"/"</js>,
059 *       converters=Queryable.<jk>class</jk>
060 *    )
061 *    <jk>public</jk> Collection&lt;Pet&gt; getPets() {
062 * </p>
063 *
064 * <p>
065 * The following shows various search arguments and their results on the page:
066 * <table style='width:auto'>
067 *    <tr>
068 *       <th>Search type</th><th>Query arguments</th><th>Query results</th>
069 *    </tr>
070 *    <tr>
071 *       <td>No arguments</td>
072 *       <td><img class='bordered' src='doc-files/PetStore_Query_q1.png'></td>
073 *       <td><img class='bordered' src='doc-files/PetStore_Query_r1.png'></td>
074 *    </tr>
075 *    <tr>
076 *       <td>String search</td>
077 *       <td><img class='bordered' src='doc-files/PetStore_Query_q2.png'></td>
078 *       <td><img class='bordered' src='doc-files/PetStore_Query_r2.png'></td>
079 *    </tr>
080 *    <tr>
081 *       <td>Numeric range</td>
082 *       <td><img class='bordered' src='doc-files/PetStore_Query_q3.png'></td>
083 *       <td><img class='bordered' src='doc-files/PetStore_Query_r3.png'></td>
084 *    </tr>
085 *    <tr>
086 *       <td>ANDed terms</td>
087 *       <td><img class='bordered' src='doc-files/PetStore_Query_q4.png'></td>
088 *       <td><img class='bordered' src='doc-files/PetStore_Query_r4.png'></td>
089 *    </tr>
090 *    <tr>
091 *       <td>Date range (entire year)</td>
092 *       <td><img class='bordered' src='doc-files/PetStore_Query_q8.png'></td>
093 *       <td><img class='bordered' src='doc-files/PetStore_Query_r8.png'></td>
094 *    </tr>
095 *    <tr>

099 *    </tr>
100 *    <tr>
101 *       <td>Date range</td>
102 *       <td><img class='bordered' src='doc-files/PetStore_Query_q10.png'></td>
103 *       <td><img class='bordered' src='doc-files/PetStore_Query_r10.png'></td>
104 *    </tr>
105 *    <tr>
106 *       <td>Hide columns</td>
107 *       <td><img class='bordered' src='doc-files/PetStore_Query_q5.png'></td>
108 *       <td><img class='bordered' src='doc-files/PetStore_Query_r5.png'></td>
109 *    </tr>
110 *    <tr>
111 *       <td>Sort</td>
112 *       <td><img class='bordered' src='doc-files/PetStore_Query_q6.png'></td>
113 *       <td><img class='bordered' src='doc-files/PetStore_Query_r6.png'></td>
114 *    </tr>
115 *    <tr>
116 *       <td>Sort descending</td>
117 *       <td><img class='bordered' src='doc-files/PetStore_Query_q7.png'></td>
118 *       <td><img class='bordered' src='doc-files/PetStore_Query_r7.png'></td>
119 *    </tr>
120 * </table>
121 *
122 */
123public class QueryWidget extends Widget {
124
125   @Override
126   public String getName() {
127      return "query";
128   }
129
130   @Override /* Widget */
131   public String resolve(RestRequest req) throws Exception {
132      // Note we're stripping off the license header.
133      return IOUtils.read(getClass().getResourceAsStream("QueryWidget.html")).replaceFirst("(?s)<!--(.*?)-->\\s*", "");
134   }
135}
