013package org.apache.juneau.rest.response;
014
015import static javax.servlet.http.HttpServletResponse.*;
016
017import java.io.*;
018
019import org.apache.juneau.*;
020import org.apache.juneau.annotation.*;
021import org.apache.juneau.http.*;
022import org.apache.juneau.internal.*;
023import org.apache.juneau.rest.*;
024import org.apache.juneau.serializer.*;
025
026/**
027 * Response handler for POJOs not handled by other handlers.
028 * <p>
029 * This uses the serializers defined on the response to serialize the POJO.
030 * <p>
031 * The {@link Serializer} used is based on the <code>Accept</code> header on the request.
032 * <p>
033 * The <code>Content-Type</code> header is set to the mime-type defined on the selected
034 *    serializer based on the {@link Produces#contentType() @Produces.contentType} annotation.
035 */
036public class DefaultHandler implements ResponseHandler {
037
038   @Override /* ResponseHandler */
039   public boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException {
040      SerializerGroup g = res.getSerializerGroup();
041      String accept = req.getHeaders().getFirst("Accept", "");
042      SerializerMatch sm = g.getSerializerMatch(accept);
043      if (sm != null) {
044         Serializer s = sm.getSerializer();
045         MediaType mediaType = res.getMediaType();
046         if (mediaType == null)
047            mediaType = sm.getMediaType();
048         res.setContentType(mediaType.toString());
049         ObjectMap headers = s.getResponseHeaders(res.getProperties());
050         if (headers != null)
051            for (String key : headers.keySet())
052               res.setHeader(key, headers.getString(key));
053
054         try {
055            ObjectMap p = res.getProperties();
056            if (req.isPlainText()) {
057               p.put(SerializerContext.SERIALIZER_useWhitespace, true);
058               res.setContentType("text/plain");
059            }
060            p.append("mediaType", mediaType).append("characterEncoding", res.getCharacterEncoding());
061            if (! s.isWriterSerializer()) {
062               if (req.isPlainText()) {
063                  OutputStreamSerializer s2 = (OutputStreamSerializer)s;
064                  Writer w = res.getNegotiatedWriter();
065                  ByteArrayOutputStream baos = new ByteArrayOutputStream();
066                  SerializerSession session = s.createSession(baos, p, req.getJavaMethod(), req.getLocale(), req.getHeaders().getTimeZone(), mediaType, req.getUriContext());
067                  s2.serialize(session, output);
068                  w.write(StringUtils.toHex(baos.toByteArray()));
069                  w.close();
070               } else {
071                  OutputStreamSerializer s2 = (OutputStreamSerializer)s;
072                  OutputStream os = res.getNegotiatedOutputStream();
073                  SerializerSession session = s.createSession(os, p, req.getJavaMethod(), req.getLocale(), req.getHeaders().getTimeZone(), mediaType, req.getUriContext());
074                  s2.serialize(session, output);
075                  os.close();
076               }
077            } else {
078               WriterSerializer s2 = (WriterSerializer)s;
079               Writer w = res.getNegotiatedWriter();
080               SerializerSession session = s.createSession(w, p, req.getJavaMethod(), req.getLocale(), req.getHeaders().getTimeZone(), mediaType, req.getUriContext());
081               s2.serialize(session, output);
082               w.close();
083            }
084         } catch (SerializeException e) {
085            throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
086         }
087      } else {
088         throw new RestException(SC_NOT_ACCEPTABLE,
089            "Unsupported media-type in request header ''Accept'': ''{0}''\n\tSupported media-types: {1}",
090            req.getHeaders().getFirst("Accept", ""), g.getSupportedMediaTypes()
091         );
092      }
093      return true;
094   }
095}
