059   /**
060    * Adds default entries to these query parameters.
061    * 
062    * <p>
063    * This includes the default queries defined on the servlet and method levels.
064    * 
065    * @param defaultEntries The default entries.  Can be <jk>null</jk>.
066    * @return This object (for method chaining).
067    */
068   public RequestQuery addDefault(Map<String,Object> defaultEntries) {
069      if (defaultEntries != null) {
070         for (Map.Entry<String,Object> e : defaultEntries.entrySet()) {
071            String key = e.getKey();
072            Object value = e.getValue();
073            String[] v = get(key);
074            if (v == null || v.length == 0 || StringUtils.isEmpty(v[0]))
075               put(key, new String[]{StringUtils.toString(value)});
076         }
077      }
078      return this;
079   }

091   /**
092    * Returns a query parameter value.
093    * 
094    * <p>
095    * Same as {@link HttpServletRequest#getParameter(String)} except only looks in the URL string, not parameters from
096    * URL-Encoded FORM posts.
097    * 
098    * <p>
099    * This method can be used to retrieve a parameter without triggering the underlying servlet API to load and parse
100    * the request body.
101    * 
102    * <p>
103    * If multiple query parameters have the same name, this returns only the first instance.
104    * 
105    * @param name The URL parameter name.
106    * @return The parameter value, or <jk>null</jk> if parameter not specified or has no value (e.g. <js>"&amp;foo"</js>.
107    */
108   public String getString(String name) {
109      String[] v = get(name);
110      if (v == null || v.length == 0)
111         return null;
112
113      // Fix for behavior difference between Tomcat and WAS.
114      // getParameter("foo") on "&foo" in Tomcat returns "".
115      // getParameter("foo") on "&foo" in WAS returns null.
116      if (v.length == 1 && v[0] == null)
117         return "";
118
119      return v[0];
120   }

465   /**
466    * Locates the special search query arguments in the query and returns them as a {@link SearchArgs} object.
467    * 
468    * <p>
469    * The query arguments are as follows:
470    * <ul>
471    *    <li>
472    *       <js>"&amp;s="</js> - A comma-delimited list of column-name/search-token pairs.
473    *       <br>Example: <js>"&amp;s=column1=foo*,column2=*bar"</js>
474    *    <li>
475    *       <js>"&amp;v="</js> - A comma-delimited list column names to view.
476    *       <br>Example: <js>"&amp;v=column1,column2"</js>
477    *    <li>
478    *       <js>"&amp;o="</js> - A comma-delimited list column names to sort by.
479    *       <br>Column names can be suffixed with <js>'-'</js> to indicate descending order.
480    *       <br>Example: <js>"&amp;o=column1,column2-"</js>
481    *    <li>
482    *       <js>"&amp;p="</js> - The zero-index row number of the first row to display.
483    *       <br>Example: <js>"&amp;p=100"</js>
484    *    <li>
485    *       <js>"&amp;l="</js> - The number of rows to return.
486    *       <br><code>0</code> implies return all rows.
487    *       <br>Example: <js>"&amp;l=100"</js>
488    *    <li>
489    *       <js>"&amp;i="</js> - The case-insensitive search flag.
490    *       <br>Example: <js>"&amp;i=true"</js>
491    * </ul>
492    * 
493    * <p>
494    * Whitespace is trimmed in the parameters.
495    * 
496    * @return
497    *    A new {@link SearchArgs} object initialized with the special search query arguments.
498    *    <jk>null</jk> if no search arguments were found.
499    */
500   public SearchArgs getSearchArgs() {
501      if (hasAny("s","v","o","p","l","i")) {
502         return new SearchArgs.Builder()
503            .search(getString("s"))
504            .view(getString("v"))
505            .sort(getString("o"))
506            .position(getInt("p"))
507            .limit(getInt("l"))
508            .ignoreCase(getBoolean("i"))
509            .build();
510      }
511      return null;
512   }

527   /* Workhorse method */
528   private <T> T parse(HttpPartParser parser, String name, Object def, ClassMeta<T> cm) throws ParseException {
529      String val = getString(name);
530      if (val == null)
531         return (T)def;
532      return parseValue(parser, val, cm);
533   }

535   /* Workhorse method */
536   private <T> T parse(HttpPartParser parser, String name, ClassMeta<T> cm) throws ParseException {
537      String val = getString(name);
538      if (cm.isPrimitive() && (val == null || val.isEmpty()))
539         return cm.getPrimitiveDefault();
540      return parseValue(parser, val, cm);
541   }

543   /* Workhorse method */
544   @SuppressWarnings("rawtypes")
545   private <T> T parseAll(HttpPartParser parser, String name, ClassMeta<T> cm) throws ParseException {
546      String[] p = get(name);
547      if (p == null)
548         return null;
549      if (cm.isArray()) {
550         List c = new ArrayList();
551         for (int i = 0; i < p.length; i++)
552            c.add(parseValue(parser, p[i], cm.getElementType()));
553         return (T)toArray(c, cm.getElementType().getInnerClass());
554      } else if (cm.isCollection()) {
555         try {
556            Collection c = (Collection)(cm.canCreateNewInstance() ? cm.newInstance() : new ObjectList());
557            for (int i = 0; i < p.length; i++)
558               c.add(parseValue(parser, p[i], cm.getElementType()));
559            return (T)c;
560         } catch (ParseException e) {
561            throw e;
562         } catch (Exception e) {
563            // Typically an instantiation exception.
564            throw new ParseException(e);
565         }
566      }
567      throw new ParseException("Invalid call to getQueryParameters(String, ClassMeta).  Class type must be a Collection or array.");
568   }
569
570   private <T> T parseValue(HttpPartParser parser, String val, ClassMeta<T> c) throws ParseException {
571      if (parser == null)
572         parser = this.parser;
573      return parser.parse(HttpPartType.QUERY, val, c);
574   }

576   /**
577    * Converts the query parameters to a readable string.
578    * 
579    * @param sorted Sort the query parameters by name.
580    * @return A JSON string containing the contents of the query parameters.
581    */
582   public String toString(boolean sorted) {
583      Map<String,Object> m = (sorted ? new TreeMap<String,Object>() : new LinkedHashMap<String,Object>());
584      for (Map.Entry<String,String[]> e : this.entrySet()) {
585         String[] v = e.getValue();
586         m.put(e.getKey(), v.length == 1 ? v[0] : v);
587      }
588      return JsonSerializer.DEFAULT_LAX.toString(m);
589   }

591   /**
592    * Converts this object to a query string.
593    * 
594    * <p>
595    * Returned query string does not start with <js>'?'</js>.
596    * 
597    * @return A new query string, or an empty string if this object is empty.
598    */
599   public String toQueryString() {
600      StringBuilder sb = new StringBuilder();
601      for (Map.Entry<String,String[]> e : this.entrySet()) {
602         for (int i = 0; i < e.getValue().length; i++) {
603            if (sb.length() > 0)
604               sb.append("&");
605            sb.append(XmlUtils.urlEncode(e.getKey())).append('=').append(XmlUtils.urlEncode(e.getValue()[i]));
606         }
607      }
608      return sb.toString();
609   }
