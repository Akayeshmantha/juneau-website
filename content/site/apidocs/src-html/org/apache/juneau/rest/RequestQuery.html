059   /**
060    * Adds default entries to these query parameters.
061    *
062    * <p>
063    * This includes the default queries defined on the servlet and method levels.
064    *
065    * @param defaultEntries The default entries.  Can be <jk>null</jk>.
066    * @return This object (for method chaining).
067    */
068   public RequestQuery addDefault(Map<String,String> defaultEntries) {
069      if (defaultEntries != null) {
070         for (Map.Entry<String,String> e : defaultEntries.entrySet()) {
071            String key = e.getKey(), value = e.getValue();
072            String[] v = get(key);
073            if (v == null || v.length == 0 || StringUtils.isEmpty(v[0]))
074               put(key, new String[]{value});
075         }
076      }
077      return this;
078   }

090   /**
091    * Returns a query parameter value.
092    *
093    * <p>
094    * Same as {@link HttpServletRequest#getParameter(String)} except only looks in the URL string, not parameters from
095    * URL-Encoded FORM posts.
096    *
097    * <p>
098    * This method can be used to retrieve a parameter without triggering the underlying servlet API to load and parse
099    * the request body.
100    *
101    * <p>
102    * If multiple query parameters have the same name, this returns only the first instance.
103    *
104    * @param name The URL parameter name.
105    * @return The parameter value, or <jk>null</jk> if parameter not specified or has no value (e.g. <js>"&amp;foo"</js>.
106    */
107   public String getString(String name) {
108      String[] v = get(name);
109      if (v == null || v.length == 0)
110         return null;
111
112      // Fix for behavior difference between Tomcat and WAS.
113      // getParameter("foo") on "&foo" in Tomcat returns "".
114      // getParameter("foo") on "&foo" in WAS returns null.
115      if (v.length == 1 && v[0] == null)
116         return "";
117
118      return v[0];
119   }

188   /**
189    * Returns the specified query parameter value converted to a POJO.
190    *
191    * <p>
192    * This method can be used to retrieve a parameter without triggering the underlying servlet API to load and parse
193    * the request body.
194    *
195    * <h5 class='section'>Examples:</h5>
196    * <p class='bcode'>
197    *    <jc>// Parse into an integer.</jc>
198    *    <jk>int</jk> myparam = req.getQueryParameter(<js>"myparam"</js>, <jk>int</jk>.<jk>class</jk>);
199    *
200    *    <jc>// Parse into an int array.</jc>
201    *    <jk>int</jk>[] myparam = req.getQueryParameter(<js>"myparam"</js>, <jk>int</jk>[].<jk>class</jk>);
202
203    *    <jc>// Parse into a bean.</jc>
204    *    MyBean myparam = req.getQueryParameter(<js>"myparam"</js>, MyBean.<jk>class</jk>);
205    *
206    *    <jc>// Parse into a linked-list of objects.</jc>
207    *    List myparam = req.getQueryParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>);
208    *
209    *    <jc>// Parse into a map of object keys/values.</jc>
210    *    Map myparam = req.getQueryParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>);
211    * </p>
212    *
213    * @param name The parameter name.
214    * @param type The class type to convert the parameter value to.
215    * @param <T> The class type to convert the parameter value to.
216    * @return The parameter value converted to the specified class type.
217    * @throws ParseException
218    */
219   public <T> T get(String name, Class<T> type) throws ParseException {
220      return get(name, beanSession.getClassMeta(type));
221   }

237   /**
238    * Returns the specified query parameter value converted to a POJO.
239    *
240    * <p>
241    * This method can be used to retrieve a parameter without triggering the underlying servlet API to load and parse
242    * the request body.
243    *
244    * <p>
245    * Use this method if you want to parse into a parameterized <code>Map</code>/<code>Collection</code> object.
246    *
247    * <h5 class='section'>Examples:</h5>
248    * <p class='bcode'>
249    *    <jc>// Parse into a linked-list of strings.</jc>
250    *    List&lt;String&gt; myparam = req.getQueryParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
251    *
252    *    <jc>// Parse into a linked-list of linked-lists of strings.</jc>
253    *    List&lt;List&lt;String&gt;&gt; myparam = req.getQueryParameter(<js>"myparam"</js>, LinkedList.<jk>class</jk>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
254    *
255    *    <jc>// Parse into a map of string keys/values.</jc>
256    *    Map&lt;String,String&gt; myparam = req.getQueryParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, String.<jk>class</jk>);
257    *
258    *    <jc>// Parse into a map containing string keys and values of lists containing beans.</jc>
259    *    Map&lt;String,List&lt;MyBean&gt;&gt; myparam = req.getQueryParameter(<js>"myparam"</js>, TreeMap.<jk>class</jk>, String.<jk>class</jk>, List.<jk>class</jk>, MyBean.<jk>class</jk>);
260    * </p>
261    *
262    * @param name The parameter name.
263    * @param type
264    *    The type of object to create.
265    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType},
266    *    {@link GenericArrayType}
267    * @param args
268    *    The type arguments of the class if it's a collection or map.
269    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType},
270    *    {@link GenericArrayType}
271    *    <br>Ignored if the main type is not a map or collection.
272    * @param <T> The class type to convert the parameter value to.
273    * @return The parameter value converted to the specified class type.
274    * @throws ParseException
275    */
276   public <T> T get(String name, Type type, Type...args) throws ParseException {
277      return (T)parse(name, beanSession.getClassMeta(type, args));
278   }

357   /**
358    * Locates the special search query arguments in the query and returns them as a {@link SearchArgs} object.
359    *
360    * <p>
361    * The query arguments are as follows:
362    * <ul>
363    *    <li>
364    *       <js>"&amp;s="</js> - A comma-delimited list of column-name/search-token pairs.
365    *       <br>Example: <js>"&amp;s=column1=foo*,column2=*bar"</js>
366    *    <li>
367    *       <js>"&amp;v="</js> - A comma-delimited list column names to view.
368    *       <br>Example: <js>"&amp;v=column1,column2"</js>
369    *    <li>
370    *       <js>"&amp;o="</js> - A comma-delimited list column names to sort by.
371    *       <br>Column names can be suffixed with <js>'-'</js> to indicate descending order.
372    *       <br>Example: <js>"&amp;o=column1,column2-"</js>
373    *    <li>
374    *       <js>"&amp;p="</js> - The zero-index row number of the first row to display.
375    *       <br>Example: <js>"&amp;p=100"</js>
376    *    <li>
377    *       <js>"&amp;l="</js> - The number of rows to return.
378    *       <br><code>0</code> implies return all rows.
379    *       <br>Example: <js>"&amp;l=100"</js>
380    *    <li>
381    *       <js>"&amp;i="</js> - The case-insensitive search flag.
382    *       <br>Example: <js>"&amp;i=true"</js>
383    * </ul>
384    *
385    * <p>
386    * Whitespace is trimmed in the parameters.
387    *
388    * @return
389    *    A new {@link SearchArgs} object initialized with the special search query arguments.
390    *    <jk>null</jk> if no search arguments were found.
391    */
392   public SearchArgs getSearchArgs() {
393      if (hasAny("s","v","o","p","l","i")) {
394         return new SearchArgs.Builder()
395            .search(getString("s"))
396            .view(getString("v"))
397            .sort(getString("o"))
398            .position(getInt("p"))
399            .limit(getInt("l"))
400            .ignoreCase(getBoolean("i"))
401            .build();
402      }
403      return null;
404   }

419   /* Workhorse method */
420   private <T> T parse(String name, T def, ClassMeta<T> cm) throws ParseException {
421      String val = getString(name);
422      if (val == null)
423         return def;
424      return parseValue(val, cm);
425   }
426
427   /* Workhorse method */
428   private <T> T parse(String name, ClassMeta<T> cm) throws ParseException {
429      String val = getString(name);
430      if (cm.isPrimitive() && (val == null || val.isEmpty()))
431         return cm.getPrimitiveDefault();
432      return parseValue(val, cm);
433   }
434
435   /* Workhorse method */
436   @SuppressWarnings("rawtypes")
437   private <T> T parseAll(String name, ClassMeta<T> cm) throws ParseException {
438      String[] p = get(name);
439      if (p == null)
440         return null;
441      if (cm.isArray()) {
442         List c = new ArrayList();
443         for (int i = 0; i < p.length; i++)
444            c.add(parseValue(p[i], cm.getElementType()));
445         return (T)toArray(c, cm.getElementType().getInnerClass());
446      } else if (cm.isCollection()) {
447         try {
448            Collection c = (Collection)(cm.canCreateNewInstance() ? cm.newInstance() : new ObjectList());
449            for (int i = 0; i < p.length; i++)
450               c.add(parseValue(p[i], cm.getElementType()));
451            return (T)c;
452         } catch (ParseException e) {
453            throw e;
454         } catch (Exception e) {
455            // Typically an instantiation exception.
456            throw new ParseException(e);
457         }
458      }
459      throw new ParseException("Invalid call to getQueryParameters(String, ClassMeta).  Class type must be a Collection or array.");
460   }
461
462   private <T> T parseValue(String val, ClassMeta<T> c) throws ParseException {
463      return parser.parse(PartType.QUERY, val, c);
464   }

466   /**
467    * Converts the query parameters to a readable string.
468    *
469    * @param sorted Sort the query parameters by name.
470    * @return A JSON string containing the contents of the query parameters.
471    */
472   public String toString(boolean sorted) {
473      Map<String,Object> m = (sorted ? new TreeMap<String,Object>() : new LinkedHashMap<String,Object>());
474      for (Map.Entry<String,String[]> e : this.entrySet()) {
475         String[] v = e.getValue();
476         m.put(e.getKey(), v.length == 1 ? v[0] : v);
477      }
478      return JsonSerializer.DEFAULT_LAX.toString(m);
479   }

481   /**
482    * Converts this object to a query string.
483    *
484    * <p>
485    * Returned query string does not start with <js>'?'</js>.
486    *
487    * @return A new query string, or an empty string if this object is empty.
488    */
489   public String toQueryString() {
490      StringBuilder sb = new StringBuilder();
491      for (Map.Entry<String,String[]> e : this.entrySet()) {
492         for (int i = 0; i < e.getValue().length; i++) {
493            if (sb.length() > 0)
494               sb.append("&");
495            sb.append(XmlUtils.urlEncode(e.getKey())).append('=').append(XmlUtils.urlEncode(e.getValue()[i]));
496         }
497      }
498      return sb.toString();
499   }
