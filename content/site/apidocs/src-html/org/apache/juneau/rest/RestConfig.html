039/**
040 * Defines the initial configuration of a <code>RestServlet</code> or <code>@RestResource</code> annotated object.
041 * <p>
042 * An extension of the {@link ServletConfig} object used during servlet initialization.
043 * <p>
044 * Provides access to the following initialized resources:
045 * <ul>
046 *    <li>{@link #getConfigFile()} - The external configuration file for this resource.
047 *    <li>{@link #getProperties()} - The modifiable configuration properties for this resource.
048 *    <li>{@link #getVarResolverBuilder()} - The variable resolver for this resource.
049 * </ul>
050 * <p>
051 * Methods are provided for overriding or augmenting the information provided by the <ja>@RestResource</ja> annotation.
052 * In general, most information provided in the <ja>@RestResource</ja> annotation can be specified programmatically
053 * through calls on this object.
054 * <p>
055 * To interact with this object, simply implement the following init method in your resource class:
056 * <p class='bcode'>
057 *    <jk>public synchronized void</jk> init(RestConfig config) <jk>throws</jk> Exception {
058 *       config.addPojoSwaps(CalendarSwap.<jsf>RFC2822DTZ</jsf>.<jk>class</jk>);
059 *       config.setProperty(<jsf>PARSER_debug</jsf>, <jk>true</jk>);
060 *       <jk>super</jk>.init(config); <jc>// Make sure this is the last line! (or just leave it out entirely)</jc>
061 *    }
062 * </p>
063 * <p>
064 * Note that this method is identical to {@link HttpServlet#init(ServletConfig)} except you get access to
065 * this object instead.  Also, this method can throw any exception, not just a {@link ServletException}.
066 * <p>
067 * The parent <code>init(RestServletConfig)</code> method will construct a read-only {@link RestContext} object
068 * that contains a snapshot of these settings.  If you call <code><jk>super</jk>.init(RestServletConfig)<code> before
069 * you modify this config object, you won't see the changes!
070 */
071@SuppressWarnings("hiding")
072public class RestConfig implements ServletConfig {
073
074   final ServletConfig inner;

077   // The following fields are meant to be modifiable.

113   /**
114    * Constructor.
115    * @param config The servlet config passed into the servlet by the servlet container.
116    * @param resource The class annotated with <ja>@RestResource</ja>.
117    * @throws ServletException Something bad happened.
118    */
119   RestConfig(ServletConfig config, Class<?> resourceClass, RestContext parentContext) throws ServletException {
120      this.inner = config;
121      this.parentContext = parentContext;
122      try {
123
124         ConfigFileBuilder cfb = new ConfigFileBuilder();
125
126         properties = new ObjectMap();
127         configFile = cfb.build();
128         varResolverBuilder = new VarResolverBuilder()
129            .vars(
130               SystemPropertiesVar.class,
131               EnvVariablesVar.class,
132               ConfigFileVar.class,
133               IfVar.class,
134               SwitchVar.class
135            );
136
137         VarResolver vr = varResolverBuilder.build();
138
139         Map<Class<?>,RestResource> restResourceAnnotationsParentFirst = ReflectionUtils.findAnnotationsMapParentFirst(RestResource.class, resourceClass);
140
141         // Find our config file.  It's the last non-empty @RestResource.config().
142         String configPath = "";
143         for (RestResource r : restResourceAnnotationsParentFirst.values())
144            if (! r.config().isEmpty())
145               configPath = r.config();
146         String cf = vr.resolve(configPath);
147         if (! cf.isEmpty())
148            configFile = cfb.build(cf);
149         configFile = configFile.getResolving(vr);
150
151         // Add our config file to the variable resolver.
152         varResolverBuilder.contextObject(ConfigFileVar.SESSION_config, configFile);
153         vr = varResolverBuilder.build();
154
155         // Add our servlet URI to our properties.
156         ServletContext ctx = config.getServletContext();
157         String ctxPath = ctx.getContextPath();
158         // Workaround for bug in Jetty that causes context path to always end in "null".
159         if (ctxPath.endsWith("null"))
160            ctxPath = ctxPath.substring(0, ctxPath.length()-4);
161         properties.put(SERIALIZER_relativeUriBase, ctxPath);
162
163         // Add the servlet init parameters to our properties.
164         for (Enumeration<String> ep = config.getInitParameterNames(); ep.hasMoreElements();) {
165            String p = ep.nextElement();
166            String initParam = config.getInitParameter(p);
167            properties.put(vr.resolve(p), vr.resolve(initParam));
168         }
169
170         // Load stuff from parent-to-child order.
171         // This allows child settings to overwrite parent settings.
172         for (Map.Entry<Class<?>,RestResource> e : restResourceAnnotationsParentFirst.entrySet()) {
173            Class<?> c = e.getKey();
174            RestResource r = e.getValue();
175            for (Property p : r.properties())
176               properties.append(vr.resolve(p.name()), vr.resolve(p.value()));
177            addSerializers(r.serializers());
178            addParsers(r.parsers());
179            addEncoders(r.encoders());
180            addDefaultRequestHeaders(r.defaultRequestHeaders());
181            addDefaultResponseHeaders(r.defaultResponseHeaders());
182            addResponseHandlers(r.responseHandlers());
183            addConverters(r.converters());
184            addGuards(reverse(r.guards()));
185            addChildResources(r.children());
186            addBeanFilters(r.beanFilters());
187            addPojoSwaps(r.pojoSwaps());
188            addParamResolvers(r.paramResolvers());
189            if (! r.stylesheet().isEmpty())
190               setStyleSheet(c, r.stylesheet());
191            if (! r.favicon().isEmpty())
192               setFavIcon(c, r.favicon());
193            if (! r.staticFiles().isEmpty())
194               addStaticFiles(c, r.staticFiles());
195            if (! r.path().isEmpty())
196               setPath(r.path());
197            if (! r.clientVersionHeader().isEmpty())
198               setClientVersionHeader(r.clientVersionHeader());
199            if (! r.pageTitle().isEmpty())
200               setPageTitle(r.pageTitle());
201            if (! r.pageText().isEmpty())
202               setPageText(r.pageText());
203            if (! r.pageLinks().isEmpty())
204               setPageLinks(r.pageLinks());
205            if (r.resourceResolver() != RestResourceResolver.class)
206               setResourceResolver(r.resourceResolver());
207            if (r.logger() != RestLogger.Normal.class)
208               setLogger(r.logger());
209            if (r.callHandler() != RestCallHandler.class)
210               setCallHandler(r.callHandler());
211            if (r.infoProvider() != RestInfoProvider.class)
212               setInfoProvider(r.infoProvider());
213         }
214
215         addResponseHandlers(
216            StreamableHandler.class,
217            WritableHandler.class,
218            ReaderHandler.class,
219            InputStreamHandler.class,
220            RedirectHandler.class,
221            DefaultHandler.class
222         );
223
224         addMimeTypes(
225            "text/css css CSS",
226            "text/html html htm HTML",
227            "text/plain txt text TXT",
228            "application/javascript js",
229            "image/png png",
230            "image/gif gif",
231            "application/xml xml XML",
232            "application/json json JSON"
233         );
234      } catch (Exception e) {
235         throw new ServletException(e);
236      }
237   }

240    * Adds the specified {@link Var} classes to this config.
241    * <p>
242    * These variables affect the variable resolver returned by {@link RestRequest#getVarResolverSession()} which is

592   /**
593    * Adds class-level default HTTP request headers to this resource.
594    * <p>
595    * Default request headers are default values for when HTTP requests do not specify a header value.
596    * For example, you can specify a default value for <code>Accept</code> if a request does not specify that header value.
597    * <p>
598    * This is the programmatic equivalent to the {@link RestResource#defaultRequestHeaders() @RestResource.defaultRequestHeaders()} annotation.
599    *
600    * @param headers HTTP headers of the form <js>"Name: Value"</js>.
601    * @return This object (for method chaining).
602    * @throws RestServletException If header string is not correctly formatted.
603    */
604   public RestConfig addDefaultRequestHeaders(String...headers) throws RestServletException {
605      for (String header : headers) {
606         String[] h = RestUtils.parseHeader(header);
607         if (h == null)
608            throw new RestServletException("Invalid default request header specified: ''{0}''.  Must be in the format: ''Header-Name: header-value''", header);
609         addDefaultRequestHeader(h[0], h[1]);
610      }
611      return this;
612   }

634    * Adds class-level default HTTP response headers to this resource.
635    * <p>
636    * Default response headers are headers that will be appended to all responses if those headers have not already been
637    *    set on the response object.
638    * <p>
639    * This is the programmatic equivalent to the {@link RestResource#defaultResponseHeaders() @RestResource.defaultResponseHeaders()} annotation.
640    *
641    * @param headers HTTP headers of the form <js>"Name: Value"</js>.
642    * @return This object (for method chaining).
643    * @throws RestServletException If header string is not correctly formatted.
644    */
645   public RestConfig addDefaultResponseHeaders(String...headers) throws RestServletException {
646      for (String header : headers) {
647         String[] h = RestUtils.parseHeader(header);
648         if (h == null)
649            throw new RestServletException("Invalid default response header specified: ''{0}''.  Must be in the format: ''Header-Name: header-value''", header);
650         addDefaultResponseHeader(h[0], h[1]);
651      }
652      return this;
653   }
