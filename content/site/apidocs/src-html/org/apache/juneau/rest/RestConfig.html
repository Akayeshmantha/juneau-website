042/**
043 * Defines the initial configuration of a <code>RestServlet</code> or <code>@RestResource</code> annotated object.
044 *
045 * <p>
046 * An extension of the {@link ServletConfig} object used during servlet initialization.
047 *
048 * <p>
049 * Provides access to the following initialized resources:
050 * <ul>
051 *    <li>{@link #getConfigFile()} - The external configuration file for this resource.
052 *    <li>{@link #getProperties()} - The modifiable configuration properties for this resource.
053 *    <li>{@link #getVarResolverBuilder()} - The variable resolver for this resource.
054 * </ul>
055 *
056 * <p>
057 * Methods are provided for overriding or augmenting the information provided by the <ja>@RestResource</ja> annotation.
058 * In general, most information provided in the <ja>@RestResource</ja> annotation can be specified programmatically
059 * through calls on this object.
060 *
061 * <p>
062 * To interact with this object, simply implement the following init method in your resource class:
063 * <p class='bcode'>
064 *    <jk>public synchronized void</jk> init(RestConfig config) <jk>throws</jk> Exception {
065 *       config.addPojoSwaps(CalendarSwap.<jsf>RFC2822DTZ</jsf>.<jk>class</jk>);
066 *       config.setProperty(<jsf>PARSER_debug</jsf>, <jk>true</jk>);
067 *       <jk>super</jk>.init(config); <jc>// Make sure this is the last line! (or just leave it out entirely)</jc>
068 *    }
069 * </p>
070 *
071 * <p>
072 * Note that this method is identical to {@link HttpServlet#init(ServletConfig)} except you get access to
073 * this object instead.  Also, this method can throw any exception, not just a {@link ServletException}.
074 *
075 * <p>
076 * The parent <code>init(RestServletConfig)</code> method will construct a read-only {@link RestContext} object
077 * that contains a snapshot of these settings.  If you call <code><jk>super</jk>.init(RestServletConfig)</code> before
078 * you modify this config object, you won't see the changes!
079 */
080@SuppressWarnings({"hiding"})
081public class RestConfig implements ServletConfig {
082

084

130   /**
131    * Constructor.
132    *
133    * @param config The servlet config passed into the servlet by the servlet container.
134    * @param resource The class annotated with <ja>@RestResource</ja>.
135    * @throws ServletException Something bad happened.
136    */
137   RestConfig(ServletConfig config, Class<?> resourceClass, RestContext parentContext) throws ServletException {
138      this.inner = config;
139      this.resourceClass = resourceClass;
140      this.parentContext = parentContext;
141      try {
142
143         ConfigFileBuilder cfb = new ConfigFileBuilder();
144
145         properties = new ObjectMap();
146         configFile = cfb.build();
147         varResolverBuilder = new VarResolverBuilder()
148            .vars(
149               SystemPropertiesVar.class,
150               EnvVariablesVar.class,
151               ConfigFileVar.class,
152               IfVar.class,
153               SwitchVar.class
154            );
155
156         VarResolver vr = varResolverBuilder.build();
157
158         Map<Class<?>,RestResource> restResourceAnnotationsParentFirst = findAnnotationsMapParentFirst(RestResource.class, resourceClass);
159
160         // Find our config file.  It's the last non-empty @RestResource.config().
161         String configPath = "";
162         for (RestResource r : restResourceAnnotationsParentFirst.values())
163            if (! r.config().isEmpty())
164               configPath = r.config();
165         String cf = vr.resolve(configPath);
166         if (! cf.isEmpty())
167            configFile = cfb.build(cf);
168         configFile = configFile.getResolving(vr);
169
170         // Add our config file to the variable resolver.
171         varResolverBuilder.contextObject(ConfigFileVar.SESSION_config, configFile);
172         vr = varResolverBuilder.build();
173
174         // Add the servlet init parameters to our properties.
175         for (Enumeration<String> ep = config.getInitParameterNames(); ep.hasMoreElements();) {
176            String p = ep.nextElement();
177            String initParam = config.getInitParameter(p);
178            properties.put(vr.resolve(p), vr.resolve(initParam));
179         }
180
181         // Load stuff from parent-to-child order.
182         // This allows child settings to overwrite parent settings.
183         for (Map.Entry<Class<?>,RestResource> e : restResourceAnnotationsParentFirst.entrySet()) {
184            Class<?> c = e.getKey();
185            RestResource r = e.getValue();
186            for (Property p : r.properties())
187               properties.append(vr.resolve(p.name()), vr.resolve(p.value()));
188            for (String p : r.flags())
189               properties.append(p, true);
190            addSerializers(r.serializers());
191            addParsers(r.parsers());
192            addEncoders(r.encoders());
193            addDefaultRequestHeaders(r.defaultRequestHeaders());
194            addDefaultResponseHeaders(r.defaultResponseHeaders());
195            addResponseHandlers(r.responseHandlers());
196            addConverters(r.converters());
197            addGuards(reverse(r.guards()));
198            addChildResources(r.children());
199            addBeanFilters(r.beanFilters());
200            addPojoSwaps(r.pojoSwaps());
201            addParamResolvers(r.paramResolvers());
202            serializerListener(r.serializerListener());
203            parserListener(r.parserListener());
204            if (! r.favicon().isEmpty())
205               setFavIcon(c, r.favicon());
206            if (! r.staticFiles().isEmpty())
207               addStaticFiles(c, r.staticFiles());
208            if (! r.path().isEmpty())
209               setPath(r.path());
210            if (! r.clientVersionHeader().isEmpty())
211               setClientVersionHeader(r.clientVersionHeader());
212
213            if (r.resourceResolver() != RestResourceResolver.class)
214               setResourceResolver(r.resourceResolver());
215            if (r.logger() != RestLogger.Normal.class)
216               setLogger(r.logger());
217            if (r.callHandler() != RestCallHandler.class)
218               setCallHandler(r.callHandler());
219            if (r.infoProvider() != RestInfoProvider.class)
220               setInfoProvider(r.infoProvider());
221
222            for (Class<? extends Widget> cw : r.widgets())
223               addWidget(cw);
224
225            HtmlDoc hd = r.htmldoc();
226            if (! hd.title().isEmpty())
227               setHtmlTitle(hd.title());
228            if (! hd.description().isEmpty())
229               setHtmlDescription(hd.description());
230            if (! hd.branding().isEmpty())
231               setHtmlBranding(hd.branding());
232            if (! hd.header().isEmpty())
233               setHtmlHeader(hd.header());
234            if (! hd.links().isEmpty())
235               setHtmlLinks(hd.links());
236            if (! hd.nav().isEmpty())
237               setHtmlNav(hd.nav());
238            if (! hd.aside().isEmpty())
239               setHtmlAside(hd.aside());
240            if (! hd.footer().isEmpty())
241               setHtmlFooter(hd.footer());
242            if (! hd.style().isEmpty())
243               setHtmlStyle(hd.style());
244            if (! hd.stylesheet().isEmpty())
245               setHtmlStylesheet(hd.stylesheet());
246            if (! hd.noResultsMessage().isEmpty())
247               setHtmlNoResultsMessage(hd.noResultsMessage());
248            if (hd.nowrap())
249               setHtmlNoWrap(true);
250            if (hd.template() != HtmlDocTemplate.class)
251               setHtmlTemplate(hd.template());
252         }
253
254         addResponseHandlers(
255            StreamableHandler.class,
256            WritableHandler.class,
257            ReaderHandler.class,
258            InputStreamHandler.class,
259            RedirectHandler.class,
260            DefaultHandler.class
261         );
262
263         addMimeTypes(
264            "text/css css CSS",
265            "text/html html htm HTML",
266            "text/plain txt text TXT",
267            "application/javascript js",
268            "image/png png",
269            "image/gif gif",
270            "application/xml xml XML",
271            "application/json json JSON"
272         );
273      } catch (Exception e) {
274         throw new ServletException(e);
275      }
276   }

278   /**
279    * Adds the specified {@link Var} classes to this config.
280    *
281    * <p>
282    * These variables affect the variable resolver returned by {@link RestRequest#getVarResolverSession()} which is
283    * used to resolve string variables of the form <js>"$X{...}"</js>.
284    *
285    * <p>
286    * By default, this config includes the following variables:
287    * <ul>
288    *    <li>{@link SystemPropertiesVar}
289    *    <li>{@link EnvVariablesVar}
290    *    <li>{@link ConfigFileVar}
291    *    <li>{@link IfVar}
292    *    <li>{@link SwitchVar}
293    * </ul>
294    *
295    * <p>
296    * Later during the construction of {@link RestContext}, we add the following variables:
297    * <ul>
298    *    <li>{@link LocalizationVar}
299    *    <li>{@link RequestVar}
300    *    <li>{@link SerializedRequestAttrVar}
301    *    <li>{@link ServletInitParamVar}
302    *    <li>{@link UrlVar}
303    *    <li>{@link UrlEncodeVar}
304    *    <li>{@link WidgetVar}
305    * </ul>
306    *
307    * @param vars The {@link Var} classes to add to this config.
308    * @return This object (for method chaining).
309    */
310   public RestConfig addVars(Class<?>...vars) {
311      this.varResolverBuilder.vars(vars);
312      return this;
313   }

315   /**
316    * Adds a var context object to this config.
317    *
318    * <p>
319    * Var context objects are read-only objects associated with the variable resolver for vars that require external
320    * information.
321    *
322    * <p>
323    * For example, the {@link ConfigFileVar} needs access to this resource's {@link ConfigFile} through the
324    * {@link ConfigFileVar#SESSION_config} object that can be specified as either a session object (temporary) or
325    * context object (permanent).
326    * In this case, we call the following code to add it to the context map:
327    * <p class='bcode'>
328    *    config.addVarContextObject(<jsf>SESSION_config</jsf>, configFile);
329    * </p>
330    *
331    * @param name The context object key (i.e. the name that the Var class looks for).
332    * @param object The context object.
333    * @return This object (for method chaining).
334    */
335   public RestConfig addVarContextObject(String name, Object object) {
336      this.varResolverBuilder.contextObject(name, object);
337      return this;
338   }
