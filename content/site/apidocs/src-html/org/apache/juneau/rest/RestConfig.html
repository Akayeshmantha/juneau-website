044/**
045 * Defines the initial configuration of a <code>RestServlet</code> or <code>@RestResource</code> annotated object.
046 *
047 * <p>
048 * An extension of the {@link ServletConfig} object used during servlet initialization.
049 *
050 * <p>
051 * Provides access to the following initialized resources:
052 * <ul>
053 *    <li>{@link #getConfigFile()} - The external configuration file for this resource.
054 *    <li>{@link #getProperties()} - The modifiable configuration properties for this resource.
055 *    <li>{@link #getVarResolverBuilder()} - The variable resolver for this resource.
056 * </ul>
057 *
058 * <p>
059 * Methods are provided for overriding or augmenting the information provided by the <ja>@RestResource</ja> annotation.
060 * In general, most information provided in the <ja>@RestResource</ja> annotation can be specified programmatically
061 * through calls on this object.
062 *
063 * <p>
064 * To interact with this object, simply implement the following init method in your resource class:
065 * <p class='bcode'>
066 *    <jk>public synchronized void</jk> init(RestConfig config) <jk>throws</jk> Exception {
067 *       config.addPojoSwaps(CalendarSwap.<jsf>RFC2822DTZ</jsf>.<jk>class</jk>);
068 *       config.setProperty(<jsf>PARSER_debug</jsf>, <jk>true</jk>);
069 *       <jk>super</jk>.init(config); <jc>// Make sure this is the last line! (or just leave it out entirely)</jc>
070 *    }
071 * </p>
072 *
073 * <p>
074 * Note that this method is identical to {@link HttpServlet#init(ServletConfig)} except you get access to
075 * this object instead.  Also, this method can throw any exception, not just a {@link ServletException}.
076 *
077 * <p>
078 * The parent <code>init(RestServletConfig)</code> method will construct a read-only {@link RestContext} object
079 * that contains a snapshot of these settings.  If you call <code><jk>super</jk>.init(RestServletConfig)</code> before
080 * you modify this config object, you won't see the changes!
081 */
082@SuppressWarnings({"hiding"})
083public class RestConfig implements ServletConfig {

132   /**
133    * Constructor.
134    *
135    * @param config The servlet config passed into the servlet by the servlet container.
136    * @param resource The class annotated with <ja>@RestResource</ja>.
137    * @throws ServletException Something bad happened.
138    */
139   RestConfig(ServletConfig config, Class<?> resourceClass, RestContext parentContext) throws ServletException {
140      this.inner = config;
141      this.resourceClass = resourceClass;
142      this.parentContext = parentContext;
143      try {
144
145         ConfigFileBuilder cfb = new ConfigFileBuilder();
146
147         properties = new ObjectMap();
148         configFile = cfb.build();
149         varResolverBuilder = new VarResolverBuilder()
150            .vars(
151               SystemPropertiesVar.class,
152               EnvVariablesVar.class,
153               ConfigFileVar.class,
154               IfVar.class,
155               SwitchVar.class
156            );
157
158         VarResolver vr = varResolverBuilder.build();
159
160         Map<Class<?>,RestResource> restResourceAnnotationsParentFirst = findAnnotationsMapParentFirst(RestResource.class, resourceClass);
161
162         // Find our config file.  It's the last non-empty @RestResource.config().
163         String configPath = "";
164         for (RestResource r : restResourceAnnotationsParentFirst.values())
165            if (! r.config().isEmpty())
166               configPath = r.config();
167         String cf = vr.resolve(configPath);
168         if (! cf.isEmpty())
169            configFile = cfb.build(cf);
170         configFile = configFile.getResolving(vr);
171
172         // Add our config file to the variable resolver.
173         varResolverBuilder.contextObject(ConfigFileVar.SESSION_config, configFile);
174         vr = varResolverBuilder.build();
175
176         // Add the servlet init parameters to our properties.
177         for (Enumeration<String> ep = config.getInitParameterNames(); ep.hasMoreElements();) {
178            String p = ep.nextElement();
179            String initParam = config.getInitParameter(p);
180            properties.put(vr.resolve(p), vr.resolve(initParam));
181         }
182
183         // Load stuff from parent-to-child order.
184         // This allows child settings to overwrite parent settings.
185         for (Map.Entry<Class<?>,RestResource> e : restResourceAnnotationsParentFirst.entrySet()) {
186            Class<?> c = e.getKey();
187            RestResource r = e.getValue();
188            for (Property p : r.properties())
189               properties.append(vr.resolve(p.name()), vr.resolve(p.value()));
190            for (String p : r.flags())
191               properties.append(p, true);
192            addSerializers(r.serializers());
193            addParsers(r.parsers());
194            addEncoders(r.encoders());
195            addDefaultRequestHeaders(r.defaultRequestHeaders());
196            addDefaultResponseHeaders(r.defaultResponseHeaders());
197            addResponseHandlers(r.responseHandlers());
198            addConverters(r.converters());
199            addGuards(reverse(r.guards()));
200            addChildResources(r.children());
201            addBeanFilters(r.beanFilters());
202            addPojoSwaps(r.pojoSwaps());
203            addParamResolvers(r.paramResolvers());
204            serializerListener(r.serializerListener());
205            parserListener(r.parserListener());
206            if (! r.favicon().isEmpty())
207               setFavIcon(c, r.favicon());
208            if (! r.staticFiles().isEmpty())
209               addStaticFiles(c, r.staticFiles());
210            if (! r.path().isEmpty())
211               setPath(r.path());
212            if (! r.clientVersionHeader().isEmpty())
213               setClientVersionHeader(r.clientVersionHeader());
214
215            if (r.resourceResolver() != RestResourceResolver.class)
216               setResourceResolver(r.resourceResolver());
217            if (r.logger() != RestLogger.Normal.class)
218               setLogger(r.logger());
219            if (r.callHandler() != RestCallHandler.class)
220               setCallHandler(r.callHandler());
221            if (r.infoProvider() != RestInfoProvider.class)
222               setInfoProvider(r.infoProvider());
223
224            HtmlDoc hd = r.htmldoc();
225            for (Class<? extends Widget> cw : hd.widgets())
226               addHtmlWidget(cw);
227            setHtmlHeader(resolveNewlineSeparatedAnnotation(hd.header(), htmlHeader));
228            setHtmlNav(resolveNewlineSeparatedAnnotation(hd.nav(), htmlNav));
229            setHtmlAside(resolveNewlineSeparatedAnnotation(hd.aside(), htmlAside));
230            setHtmlFooter(resolveNewlineSeparatedAnnotation(hd.footer(), htmlFooter));
231            setHtmlStyle(resolveNewlineSeparatedAnnotation(hd.style(), htmlStyle));
232            setHtmlScript(resolveNewlineSeparatedAnnotation(hd.script(), htmlScript));
233            setHtmlLinks(resolveLinks(hd.links(), htmlLinks));
234
235            if (! hd.stylesheet().isEmpty())
236               setHtmlStylesheet(hd.stylesheet());
237            if (! hd.noResultsMessage().isEmpty())
238               setHtmlNoResultsMessage(hd.noResultsMessage());
239            if (hd.nowrap())
240               setHtmlNoWrap(true);
241            if (hd.template() != HtmlDocTemplate.class)
242               setHtmlTemplate(hd.template());
243         }
244
245         addResponseHandlers(
246            StreamableHandler.class,
247            WritableHandler.class,
248            ReaderHandler.class,
249            InputStreamHandler.class,
250            RedirectHandler.class,
251            DefaultHandler.class
252         );
253
254      } catch (Exception e) {
255         throw new ServletException(e);
256      }
257   }

259   /**
260    * Adds the specified {@link Var} classes to this config.
261    *
262    * <p>
263    * These variables affect the variable resolver returned by {@link RestRequest#getVarResolverSession()} which is
264    * used to resolve string variables of the form <js>"$X{...}"</js>.
265    *
266    * <p>
267    * By default, this config includes the following variables:
268    * <ul>
269    *    <li>{@link SystemPropertiesVar}
270    *    <li>{@link EnvVariablesVar}
271    *    <li>{@link ConfigFileVar}
272    *    <li>{@link IfVar}
273    *    <li>{@link SwitchVar}
274    * </ul>
275    *
276    * <p>
277    * Later during the construction of {@link RestContext}, we add the following variables:
278    * <ul>
279    *    <li>{@link LocalizationVar}
280    *    <li>{@link RequestVar}
281    *    <li>{@link SerializedRequestAttrVar}
282    *    <li>{@link ServletInitParamVar}
283    *    <li>{@link UrlVar}
284    *    <li>{@link UrlEncodeVar}
285    *    <li>{@link WidgetVar}
286    * </ul>
287    *
288    * @param vars The {@link Var} classes to add to this config.
289    * @return This object (for method chaining).
290    */
291   public RestConfig addVars(Class<?>...vars) {
292      this.varResolverBuilder.vars(vars);
293      return this;
294   }

296   /**
297    * Adds a var context object to this config.
298    *
299    * <p>
300    * Var context objects are read-only objects associated with the variable resolver for vars that require external
301    * information.
302    *
303    * <p>
304    * For example, the {@link ConfigFileVar} needs access to this resource's {@link ConfigFile} through the
305    * {@link ConfigFileVar#SESSION_config} object that can be specified as either a session object (temporary) or
306    * context object (permanent).
307    * In this case, we call the following code to add it to the context map:
308    * <p class='bcode'>
309    *    config.addVarContextObject(<jsf>SESSION_config</jsf>, configFile);
310    * </p>
311    *
312    * @param name The context object key (i.e. the name that the Var class looks for).
313    * @param object The context object.
314    * @return This object (for method chaining).
315    */
316   public RestConfig addVarContextObject(String name, Object object) {
317      this.varResolverBuilder.contextObject(name, object);
318      return this;
319   }
