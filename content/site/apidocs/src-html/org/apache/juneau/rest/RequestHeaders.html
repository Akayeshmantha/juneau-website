060   /**
061    * Adds default entries to these headers.
062    * <p>
063    * This includes the default headers defined on the servlet and method levels.
064    *
065    * @param defaultEntries The default entries.  Can be <jk>null</jk>.
066    * @return This object (for method chaining).
067    */
068   public RequestHeaders addDefault(Map<String,String> defaultEntries) {
069      if (defaultEntries != null) {
070         for (Map.Entry<String,String> e : defaultEntries.entrySet()) {
071            String key = e.getKey(), value = e.getValue();
072            String[] v = get(key);
073            if (v == null || v.length == 0 || StringUtils.isEmpty(v[0]))
074               put(key, new String[]{value});
075         }
076      }
077      return this;
078   }
079
080   /**
081    * Adds a set of header values to this object.
082    *
083    * @param name The header name.
084    * @param values The header values.
085    * @return This object (for method chaining).
086    */
087   public RequestHeaders put(String name, Enumeration<String> values) {
088      // Optimized for enumerations of one entry, the most-common case.
089      if (values.hasMoreElements()) {
090         String v = values.nextElement();
091         String[] s = new String[]{v};
092         while (values.hasMoreElements())
093            s = append(s, values.nextElement());
094         put(name, s);
095      }
096      return this;
097   }

099   /**
100    * Returns the specified header value, or <jk>null</jk> if the header doesn't exist.
101    * <p>
102    * If {@code allowHeaderParams} init parameter is <jk>true</jk>, then first looks for {@code &HeaderName=x} in the URL query string.
103    * <p>
104    * @param name The header name.
105    * @return The header value, or <jk>null</jk> if it doesn't exist.
106    */
107   public String getString(String name) {
108      String[] v = null;
109      if (queryParams != null)
110         v = queryParams.get(name);
111      if (v == null || v.length == 0)
112         v = get(name);
113      if (v == null || v.length == 0)
114         return null;
115      return v[0];
116   }

119    * Returns the specified header value, or a default value if the header doesn't exist.

121    * If {@code allowHeaderParams} init parameter is <jk>true</jk>, then first looks for {@code &HeaderName=x} in the URL query string.
122    *
123    * @param name The HTTP header name.
124    * @param def The default value to return if the header value isn't found.
125    * @return The header value, or the default value if the header isn't present.
126    */
127   public String getString(String name, String def) {
128      String s = getString(name);
129      return StringUtils.isEmpty(s) ? def : s;
130   }

186   /**
187    * Returns the specified header value converted to a POJO.
188    * <p>
189    * The type can be any POJO type convertable from a <code>String</code>
190    * (See <a class="doclink" href="package-summary.html#PojosConvertableFromString">POJOs Convertable From Strings</a>).
191    * <p>
192    * <h5 class='section'>Examples:</h5>
193    * <p class='bcode'>
194    *    <jc>// Parse into an integer.</jc>
195    *    <jk>int</jk> myheader = req.getHeader(<js>"My-Header"</js>, <jk>int</jk>.<jk>class</jk>);
196    *
197    *    <jc>// Parse a UUID.</jc>
198    *    UUID myheader = req.getHeader(<js>"My-Header"</js>, UUID.<jk>class</jk>);
199    * </p>
200    *
201    * @param name The HTTP header name.
202    * @param type The class type to convert the header value to.
203    * @param <T> The class type to convert the header value to.
204    * @return The parameter value converted to the specified class type.
205    */
206   public <T> T get(String name, Class<T> type) {
207      String h = getString(name);
208      return beanSession.convertToType(h, type);
209   }

211   /**
212    * Same as {@link #get(String, Class)} but returns a default value if not found.
213    *
214    * @param name The HTTP header name.
215    * @param def The default value if the header was not specified or is <jk>null</jk>.
216    * @param type The class type to convert the header value to.
217    * @param <T> The class type to convert the header value to.
218    * @return The parameter value converted to the specified class type.
219    */
220   public <T> T get(String name, T def, Class<T> type) {
221      String h = getString(name);
222      if (h == null)
223         return def;
224      return beanSession.convertToType(h, type);
225   }

227   /**
228    * Returns the specified header value converted to a POJO.
229    * <p>
230    * The type can be any POJO type convertable from a <code>String</code>
231    * (See <a class="doclink" href="package-summary.html#PojosConvertableFromString">POJOs Convertable From Strings</a>).
232    * <p>
233    * <h5 class='section'>Examples:</h5>
234    * <p class='bcode'>
235    *    <jc>// Parse into a linked-list of strings.</jc>
236    *    List&lt;String&gt; myheader = req.getHeader(<js>"My-Header"</js>, LinkedList.<jk>class</jk>, String.<jk>class</jk>);
237    * </p>
238    *
239    * @param name The HTTP header name.
240    * @param type The type of object to create.
241    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
242    * @param args The type arguments of the class if it's a collection or map.
243    *    <br>Can be any of the following: {@link ClassMeta}, {@link Class}, {@link ParameterizedType}, {@link GenericArrayType}
244    *    <br>Ignored if the main type is not a map or collection.
245    * @param <T> The class type to convert the header value to.
246    * @return The parameter value converted to the specified class type.
247    * @throws ParseException If the header could not be converted to the specified type.
248    */
249   @SuppressWarnings("unchecked")
250   public <T> T get(String name, Type type, Type...args) throws ParseException {
251      String h = getString(name);
252      return (T)parser.parse(PartType.HEADER, h, beanSession.getClassMeta(type, args));
253   }

255   /**
256    * Returns a copy of this object, but only with the specified header names copied.
257    * @param headers The headers to include in the copy.
258    * @return A new headers object.
259    */
260   public RequestHeaders subset(String...headers) {
261      RequestHeaders rh2 = new RequestHeaders().setParser(parser).setBeanSession(beanSession).setQueryParams(queryParams);
262      for (String h : headers)
263         if (containsKey(h))
264            rh2.put(h, get(h));
265      return rh2;
266   }

268   /**
269    * Same as {@link #subset(String...)}, but allows you to specify header names as a comma-delimited list.
270    * @param headers The headers to include in the copy.
271    * @return A new headers object.
272    */
273   public RequestHeaders subset(String headers) {
274      return subset(split(headers));
275   }

746   /**
747    * Converts the headers to a readable string.
748    *
749    * @param sorted Sort the headers by name.
750    * @return A JSON string containing the contents of the headers.
751    */
752   public String toString(boolean sorted) {
753      Map<String,Object> m = (sorted ? new TreeMap<String,Object>() : new LinkedHashMap<String,Object>());
754      for (Map.Entry<String,String[]> e : this.entrySet()) {
755         String[] v = e.getValue();
756         m.put(e.getKey(), v.length == 1 ? v[0] : v);
757      }
758      return JsonSerializer.DEFAULT_LAX.toString(m);
759   }
