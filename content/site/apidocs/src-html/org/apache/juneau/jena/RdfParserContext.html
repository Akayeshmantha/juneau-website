013package org.apache.juneau.jena;
014
015import java.util.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.parser.*;
019import org.apache.juneau.xml.*;

021/**
022 * Configurable properties on the {@link RdfParser} class.
023 * <p>
024 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
025 * returned {@link CoreApi#getContextFactory()}.
026 * <p>
027 * The following convenience methods are also provided for setting context properties:
028 * <ul>
029 *    <li>{@link RdfParser#setProperty(String,Object)}
030 *    <li>{@link RdfParser#setProperties(ObjectMap)}
031 *    <li>{@link RdfParser#addNotBeanClasses(Class[])}
032 *    <li>{@link RdfParser#addBeanFilters(Class[])}
033 *    <li>{@link RdfParser#addPojoSwaps(Class[])}
034 *    <li>{@link RdfParser#addToDictionary(Class[])}
035 *    <li>{@link RdfParser#addImplClass(Class,Class)}
036 * </ul>
037 * <p>
038 * See {@link ContextFactory} for more information about context properties.
039 *
040 * <h6 class='topic' id='ConfigProperties'>Configurable properties on the RDF parsers</h6>
041 * <table class='styled' style='border-collapse: collapse;'>
042 *    <tr><th>Setting name</th><th>Description</th><th>Data type</th><th>Default value</th><th>Session overridable</th></tr>
043 *    <tr>
044 *       <td>{@link #RDF_trimWhitespace}</td>
045 *       <td>Trim whitespace from text elements.</td>
046 *       <td><code>Boolean</code></td>
047 *       <td><jk>false</jk></td>
048 *       <td><jk>true</jk></td>
049 *    </tr>
050 * </table>
051 *
052 * <h6 class='topic' id='ConfigProperties'>Configurable properties inherited by the RDF parsers</h6>
053 * <ul class='javahierarchy'>
054 *    <li class='c'><a class='doclink' href='../BeanContext.html#ConfigProperties'>BeanContext</a> - Properties associated with handling beans on serializers and parsers.
055 *    <ul>
056 *       <li class='c'><a class='doclink' href='../parser/ParserContext.html#ConfigProperties'>ParserContext</a> - Configurable properties common to all parsers.
057 *       <ul>
058 *          <li class='i'><a class='doclink' href='RdfCommonContext.html#ConfigProperties'>RdfCommonContext</a> - Configurable properties common to the RDF serializers and parsers.
059 *       </ul>
060 *    </ul>
061 * </ul>
062 */
063public final class RdfParserContext extends ParserContext implements RdfCommonContext {
064
065   /**
066    * <b>Configuration property:</b>  Trim whitespace from text elements.
067    * <p>
068    * <ul>
069    *    <li><b>Name:</b> <js>"RdfParser.trimWhitespace"</js>
070    *    <li><b>Data type:</b> <code>Boolean</code>
071    *    <li><b>Default:</b> <jk>false</jk>
072    *    <li><b>Session-overridable:</b> <jk>true</jk>
073    * </ul>
074    * <p>
075    * If <jk>true</jk>, whitespace in text elements will be automatically trimmed.
076    */
077   public static final String RDF_trimWhitespace = "RdfParser.trimWhitespace";
078
079   final boolean trimWhitespace, looseCollections;
080   final String rdfLanguage;
081   final Namespace juneauNs, juneauBpNs;
082   final RdfCollectionFormat collectionFormat;
083   final Map<String,Object> jenaSettings = new HashMap<String,Object>();
084
085   /**
086    * Constructor.
087    * <p>
088    * Typically only called from {@link ContextFactory#getContext(Class)}.
089    *
090    * @param cf The factory that created this context.
091    */
092   public RdfParserContext(ContextFactory cf) {
093      super(cf);
094      trimWhitespace = cf.getProperty(RDF_trimWhitespace, boolean.class, false);
095      looseCollections = cf.getProperty(RDF_looseCollections, boolean.class, false);
096      rdfLanguage = cf.getProperty(RDF_language, String.class, "RDF/XML-ABBREV");
097      juneauNs = cf.getProperty(RDF_juneauNs, Namespace.class, new Namespace("j", "http://www.apache.org/juneau/"));
098      juneauBpNs = cf.getProperty(RDF_juneauBpNs, Namespace.class, new Namespace("j", "http://www.apache.org/juneaubp/"));
099      collectionFormat = cf.getProperty(RDF_collectionFormat, RdfCollectionFormat.class, RdfCollectionFormat.DEFAULT);
100   }
101
102   @Override /* Context */
103   public ObjectMap asMap() {
104      return super.asMap()
105         .append("RdfParserContext", new ObjectMap()
106            .append("trimWhitespace", trimWhitespace)
107            .append("looseCollections", looseCollections)
108            .append("rdfLanguage", rdfLanguage)
109            .append("juneauNs", juneauNs)
110            .append("juneauBpNs", juneauBpNs)
111            .append("collectionFormat", collectionFormat)
112         );
113   }
114}
