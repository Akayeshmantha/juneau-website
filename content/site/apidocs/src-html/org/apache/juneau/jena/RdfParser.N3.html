033 * <h6 class='topic'>Configurable properties</h6>

105   @SuppressWarnings({"unchecked", "rawtypes"})
106   @Override /* ReaderParser */
107   protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
108
109      RdfParserSession s = (RdfParserSession)session;
110
111      type = session.normalizeClassMeta(type);
112
113      Model model = s.getModel();
114      RDFReader r = s.getRdfReader();
115      r.read(model, session.getReader(), null);
116
117      List<Resource> roots = getRoots(s, model);
118
119      // Special case where we're parsing a loose collection of resources.
120      if (s.isLooseCollections() && type.isCollectionOrArray()) {
121         Collection c = null;
122         if (type.isArray())
123            c = new ArrayList();
124         else
125            c = (type.canCreateNewInstance(session.getOuter()) ? (Collection<?>)type.newInstance(session.getOuter()) : new ObjectList(session));
126         for (Resource resource : roots)
127            c.add(parseAnything(s, type.getElementType(), resource, session.getOuter(), null));
128
129         if (type.isArray())
130            return (T)session.toArray(type, c);
131         return (T)c;
132      }
133
134      if (roots.isEmpty())
135         return null;
136      if (roots.size() > 1)
137         throw new ParseException(session, "Too many root nodes found in model:  {0}", roots.size());
138      Resource resource = roots.get(0);
139
140      return parseAnything(s, type, resource, session.getOuter(), null);
141   }

143   /*
144    * Finds the roots in the model using either the "root" property to identify it,
145    *    or by resorting to scanning the model for all nodes with no incoming predicates.
146    */
147   private List<Resource> getRoots(RdfParserSession session, Model m) {
148      List<Resource> l = new LinkedList<Resource>();
149
150      // First try to find the root using the "http://www.apache.org/juneau/root" property.
151      Property root = m.createProperty(session.getJuneauNsUri(), RDF_juneauNs_ROOT);
152      for (ResIterator i  = m.listResourcesWithProperty(root); i.hasNext();)
153         l.add(i.next());
154
155      if (! l.isEmpty())
156         return l;
157
158      // Otherwise, we need to find all resources that aren't objects.
159      // We want to explicitly ignore statements where the subject
160      // and object are the same node.
161      Set<RDFNode> objects = new HashSet<RDFNode>();
162      for (StmtIterator i = m.listStatements(); i.hasNext();) {
163         Statement st = i.next();
164         RDFNode subject = st.getSubject();
165         RDFNode object = st.getObject();
166         if (object.isResource() && ! object.equals(subject))
167            objects.add(object);
168      }
169      for (ResIterator i = m.listSubjects(); i.hasNext();) {
170         Resource r = i.next();
171         if (! objects.contains(r))
172            l.add(r);
173      }
174      return l;
175   }

177   private <T> BeanMap<T> parseIntoBeanMap(RdfParserSession session, Resource r2, BeanMap<T> m) throws Exception {
178      BeanMeta<T> bm = m.getMeta();
179      RdfBeanMeta rbm = bm.getExtendedMeta(RdfBeanMeta.class);
180      if (rbm.hasBeanUri() && r2.getURI() != null)
181         rbm.getBeanUriProperty().set(m, r2.getURI());
182      Property subTypeIdProperty = null;
183      BeanPropertyMeta stp = bm.getSubTypeProperty();
184      if (stp != null) {
185         subTypeIdProperty = session.getProperty(stp.getName());
186         Statement st = r2.getProperty(subTypeIdProperty);
187         if (st == null)
188            throw new ParseException(session, "Could not find subtype ID property for bean of type ''{0}''", m.getClassMeta());
189         String subTypeId = st.getLiteral().getString();
190         stp.set(m, subTypeId);
191      }
192      for (StmtIterator i = r2.listProperties(); i.hasNext();) {
193         Statement st = i.next();
194         Property p = st.getPredicate();
195         if (p.equals(subTypeIdProperty))
196            continue;
197         String key = session.decodeString(p.getLocalName());
198         BeanPropertyMeta pMeta = m.getPropertyMeta(key);
199         session.setCurrentProperty(pMeta);
200         if (pMeta != null) {
201            RDFNode o = st.getObject();
202            ClassMeta<?> cm = pMeta.getClassMeta();
203            if (cm.isCollectionOrArray() && isMultiValuedCollections(session, pMeta)) {
204               ClassMeta<?> et = cm.getElementType();
205               Object value = parseAnything(session, et, o, m.getBean(false), pMeta);
206               setName(et, value, key);
207               pMeta.add(m, value);
208            } else {
209               Object value = parseAnything(session, cm, o, m.getBean(false), pMeta);
210               setName(cm, value, key);
211               pMeta.set(m, value);
212            }
213         } else if (! (p.equals(session.getRootProperty()) || p.equals(session.getTypeProperty()) || p.equals(subTypeIdProperty))) {
214            if (bm.isSubTyped()) {
215               RDFNode o = st.getObject();
216               Object value = parseAnything(session, object(), o, m.getBean(false), pMeta);
217               m.put(key, value);
218            } else {
219               onUnknownProperty(session, key, m, -1, -1);
220            }
221         }
222         session.setCurrentProperty(null);
223      }
224      return m;
225   }

227   private boolean isMultiValuedCollections(RdfParserSession session, BeanPropertyMeta pMeta) {
228      if (pMeta != null && pMeta.getExtendedMeta(RdfBeanPropertyMeta.class).getCollectionFormat() != RdfCollectionFormat.DEFAULT)
229         return pMeta.getExtendedMeta(RdfBeanPropertyMeta.class).getCollectionFormat() == RdfCollectionFormat.MULTI_VALUED;
230      return session.getCollectionFormat() == RdfCollectionFormat.MULTI_VALUED;
231   }

233   @SuppressWarnings({ "unchecked", "rawtypes" })
234   private <T> T parseAnything(RdfParserSession session, ClassMeta<T> eType, RDFNode n, Object outer, BeanPropertyMeta pMeta) throws Exception {
235
236      if (eType == null)
237         eType = (ClassMeta<T>)object();
238      PojoSwap<T,Object> transform = (PojoSwap<T,Object>)eType.getPojoSwap();
239      ClassMeta<?> sType = eType.getSerializedClassMeta();
240      session.setCurrentClass(sType);
241      BeanRegistry breg = pMeta == null ? session.getBeanRegistry() : pMeta.getBeanRegistry();
242
243      if (! sType.canCreateNewInstance(outer)) {
244         if (n.isResource()) {
245            Statement st = n.asResource().getProperty(session.getTypeProperty());
246            if (st != null) {
247               String c = st.getLiteral().getString();
248               if (breg.hasName(c))
249                  sType = eType = (ClassMeta<T>)breg.getClassMeta(c);
250            }
251         }
252      }
253
254      Object o = null;
255      if (n.isResource() && n.asResource().getURI() != null && n.asResource().getURI().equals(RDF_NIL)) {
256         // Do nothing.  Leave o == null.
257      } else if (sType.isObject()) {
258         if (n.isLiteral()) {
259            o = n.asLiteral().getValue();
260            if (o instanceof String) {
261               o = session.decodeString(o);
262            }
263         }
264         else if (n.isResource()) {
265            Resource r = n.asResource();
266            if (session.wasAlreadyProcessed(r))
267               o = r.getURI();
268            else if (r.getProperty(session.getValueProperty()) != null) {
269               o = parseAnything(session, object(), n.asResource().getProperty(session.getValueProperty()).getObject(), outer, null);
270            } else if (isSeq(session, r)) {
271               o = new ObjectList(session);
272               parseIntoCollection(session, r.as(Seq.class), (Collection)o, sType.getElementType());
273            } else if (isBag(session, r)) {
274               o = new ObjectList(session);
275               parseIntoCollection(session, r.as(Bag.class), (Collection)o, sType.getElementType());
276            } else if (r.canAs(RDFList.class)) {
277               o = new ObjectList(session);
278               parseIntoCollection(session, r.as(RDFList.class), (Collection)o, sType.getElementType());
279            } else {
280               // If it has a URI and no child properties, we interpret this as an
281               // external resource, and convert it to just a URL.
282               String uri = r.getURI();
283               if (uri != null && ! r.listProperties().hasNext()) {
284                  o = r.getURI();
285               } else {
286                  o = new ObjectMap(session);
287                  parseIntoMap(session, r, (Map)o, null, null);
288               }
289            }
290         } else {
291            throw new ParseException(session, "Unrecognized node type ''{0}'' for object", n);
292         }
293      } else if (sType.isBoolean()) {
294         o = session.convertToType(getValue(session, n, outer), boolean.class);
295      } else if (sType.isCharSequence()) {
296         o = session.decodeString(getValue(session, n, outer));
297      } else if (sType.isChar()) {
298         o = session.decodeString(getValue(session, n, outer)).charAt(0);
299      } else if (sType.isNumber()) {
300         o = parseNumber(getValue(session, n, outer).toString(), (Class<? extends Number>)sType.getInnerClass());
301      } else if (sType.isMap()) {
302         Resource r = n.asResource();
303         if (session.wasAlreadyProcessed(r))
304            return null;
305         Map m = (sType.canCreateNewInstance(outer) ? (Map)sType.newInstance(outer) : new ObjectMap(session));
306         o = parseIntoMap(session, r, m, eType.getKeyType(), eType.getValueType());
307      } else if (sType.isCollectionOrArray()) {
308         if (sType.isArray())
309            o = new ArrayList();
310         else
311            o = (sType.canCreateNewInstance(outer) ? (Collection<?>)sType.newInstance(outer) : new ObjectList(session));
312         Resource r = n.asResource();
313         if (session.wasAlreadyProcessed(r))
314            return null;
315         if (isSeq(session, r)) {
316            parseIntoCollection(session, r.as(Seq.class), (Collection)o, sType.getElementType());
317         } else if (isBag(session, r)) {
318            parseIntoCollection(session, r.as(Bag.class), (Collection)o, sType.getElementType());
319         } else if (r.canAs(RDFList.class)) {
320            parseIntoCollection(session, r.as(RDFList.class), (Collection)o, sType.getElementType());
321         } else {
322            throw new ParseException("Unrecognized node type ''{0}'' for collection", n);
323         }
324         if (sType.isArray())
325            o = session.toArray(sType, (Collection)o);
326      } else if (sType.canCreateNewBean(outer)) {
327         Resource r = n.asResource();
328         if (session.wasAlreadyProcessed(r))
329            return null;
330         BeanMap<?> bm = session.newBeanMap(outer, sType.getInnerClass());
331         o = parseIntoBeanMap(session, r, bm).getBean();
332      } else if (sType.isUri() && n.isResource()) {
333         o = sType.newInstanceFromString(outer, session.decodeString(n.asResource().getURI()));
334      } else if (sType.canCreateNewInstanceFromString(outer)) {
335         o = sType.newInstanceFromString(outer, session.decodeString(getValue(session, n, outer)));
336      } else if (sType.canCreateNewInstanceFromNumber(outer)) {
337         o = sType.newInstanceFromNumber(session, outer, parseNumber(getValue(session, n, outer).toString(), sType.getNewInstanceFromNumberClass()));
338      } else {
339         throw new ParseException("Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
340      }
341
342      if (transform != null && o != null)
343         o = transform.unswap(session, o, eType);
344
345      if (outer != null)
346         setParent(eType, o, outer);
347
348      return (T)o;
349   }

351   private boolean isSeq(RdfParserSession session, RDFNode n) {
352      if (n.isResource()) {
353         Statement st = n.asResource().getProperty(session.getRdfTypeProperty());
354         if (st != null)
355            return RDF_SEQ.equals(st.getResource().getURI());
356      }
357      return false;
358   }
359
360   private boolean isBag(RdfParserSession session, RDFNode n) {
361      if (n.isResource()) {
362         Statement st = n.asResource().getProperty(session.getRdfTypeProperty());
363         if (st != null)
364            return RDF_BAG.equals(st.getResource().getURI());
365      }
366      return false;
367   }

369   private Object getValue(RdfParserSession session, RDFNode n, Object outer) throws Exception {
370      if (n.isLiteral())
371         return n.asLiteral().getValue();
372      if (n.isResource()) {
373         Statement st = n.asResource().getProperty(session.getValueProperty());
374         if (st != null) {
375            n = st.getObject();
376            if (n.isLiteral())
377               return n.asLiteral().getValue();
378            return parseAnything(session, object(), st.getObject(), outer, null);
379         }
380      }
381      throw new ParseException(session, "Unknown value type for node ''{0}''", n);
382   }

384   private <K,V> Map<K,V> parseIntoMap(RdfParserSession session, Resource r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType) throws Exception {
385      // Add URI as "uri" to generic maps.
386      if (r.getURI() != null) {
387         K uri = convertAttrToType(session, m, "uri", keyType);
388         V value = convertAttrToType(session, m, r.getURI(), valueType);
389         m.put(uri, value);
390      }
391      for (StmtIterator i = r.listProperties(); i.hasNext();) {
392         Statement st = i.next();
393         Property p = st.getPredicate();
394         String key = p.getLocalName();
395         if (! (key.equals("root") && p.getURI().equals(session.getJuneauNsUri()))) {
396            key = session.decodeString(key);
397            RDFNode o = st.getObject();
398            K key2 = convertAttrToType(session, m, key, keyType);
399            V value = parseAnything(session, valueType, o, m, null);
400            setName(valueType, value, key);
401            m.put(key2, value);
402         }
403
404      }
405      // TODO Auto-generated method stub
406      return m;
407   }

409   private <E> Collection<E> parseIntoCollection(RdfParserSession session, Container c, Collection<E> l, ClassMeta<E> et) throws Exception {
410      for (NodeIterator ni = c.iterator(); ni.hasNext();) {
411         E e = parseAnything(session, et, ni.next(), l, null);
412         l.add(e);
413      }
414      return l;
415   }
416
417   private <E> Collection<E> parseIntoCollection(RdfParserSession session, RDFList list, Collection<E> l, ClassMeta<E> et) throws Exception {
418      for (ExtendedIterator<RDFNode> ni = list.iterator(); ni.hasNext();) {
419         E e = parseAnything(session, et, ni.next(), l, null);
420         l.add(e);
421      }
422      return l;
423   }
