013package org.apache.juneau.jena;
014
015import java.util.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.serializer.*;
019import org.apache.juneau.xml.*;
020
021/**
022 * Configurable properties on the {@link RdfSerializer} class.
023 * <p>
024 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
025 * returned {@link CoreApi#getContextFactory()}.
026 * <p>
027 * The following convenience methods are also provided for setting context properties:
028 * <ul>
029 *    <li>{@link RdfSerializer#setProperty(String,Object)}
030 *    <li>{@link RdfSerializer#setProperties(ObjectMap)}
031 *    <li>{@link RdfSerializer#addNotBeanClasses(Class[])}
032 *    <li>{@link RdfSerializer#addBeanFilters(Class[])}
033 *    <li>{@link RdfSerializer#addPojoSwaps(Class[])}
034 *    <li>{@link RdfSerializer#addToDictionary(Class[])}
035 *    <li>{@link RdfSerializer#addImplClass(Class,Class)}
036 * </ul>
037 * <p>
038 * See {@link ContextFactory} for more information about context properties.
039 *
040 *
041 * <h6 class='topic' id='ConfigProperties'>Configurable properties on the RDF serializers</h6>
042 * <table class='styled' style='border-collapse: collapse;'>
043 *    <tr><th>Setting name</th><th>Description</th><th>Data type</th><th>Default value</th></tr>
044 *    <tr>
045 *       <td>{@link #RDF_addLiteralTypes}</td>
046 *       <td>Add XSI data types to non-<code>String</code> literals.</td>
047 *       <td><code>Boolean</code></td>
048 *       <td><jk>false</jk></td>
049 *    </tr>
050 *    <tr>
051 *       <td>{@link #RDF_addRootProperty}</td>
052 *       <td>Add RDF root identifier property to root node.</td>
053 *       <td><code>Boolean</code></td>
054 *       <td><jk>false</jk></td>
055 *    </tr>
056 *    <tr>
057 *       <td>{@link #RDF_autoDetectNamespaces}</td>
058 *       <td>Auto-detect namespace usage.</td>
059 *       <td><code>Boolean</code></td>
060 *       <td><jk>true</jk></td>
061 *    </tr>
062 *    <tr>
063 *       <td>{@link #RDF_namespaces}</td>
064 *       <td>Default namespaces.</td>
065 *       <td><code>List&lt;{@link Namespace}&gt;</code></td>
066 *       <td>empty list</td>
067 *    </tr>
068 * </table>
069 *
070 * <h6 class='topic' id='ConfigProperties'>Configurable properties inherited by the RDF serializers</h6>
071 * <ul class='javahierarchy'>
072 *    <li class='c'><a class='doclink' href='../BeanContext.html#ConfigProperties'>BeanContext</a> - Properties associated with handling beans on serializers and parsers.
073 *    <ul>
074 *       <li class='c'><a class='doclink' href='../serializer/SerializerContext.html#ConfigProperties'>SerializerContext</a> - Configurable properties common to all serializers.
075 *       <ul>
076 *          <li class='c'><a class='doclink' href='RdfCommonContext.html#ConfigProperties'>RdfCommonContext</a> - Configurable properties common to the RDF serializers and parsers.
077 *       </ul>
078 *    </ul>
079 * </ul>
080 *
081 *
082 * @author James Bognar (james.bognar@salesforce.com)
083 */
084public final class RdfSerializerContext extends SerializerContext implements RdfCommonContext {
085
086   /**
087    * <b>Configuration property:</b>  Add XSI data types to non-<code>String</code> literals.
088    * <p>
089    * <ul>
090    *    <li><b>Name:</b> <js>"RdfSerializer.addLiteralTypes"</js>
091    *    <li><b>Data type:</b> <code>Boolean</code>
092    *    <li><b>Default:</b> <jk>false</jk>
093    * </ul>
094    */
095   public static final String RDF_addLiteralTypes = "RdfSerializer.addLiteralTypes";
096
097   /**
098    * <b>Configuration property:</b>  Add RDF root identifier property to root node.
099    * <p>
100    * <ul>
101    *    <li><b>Name:</b> <js>"RdfSerializer.addRootProperty"</js>
102    *    <li><b>Data type:</b> <code>Boolean</code>
103    *    <li><b>Default:</b> <jk>false</jk>
104    * </ul>
105    * <p>
106    * When enabled an RDF property <code>http://www.ibm.com/juneau/root</code> is added with a value of <js>"true"</js>
107    *    to identify the root node in the graph.
108    * This helps locate the root node during parsing.
109    * <p>
110    * If disabled, the parser has to search through the model to find any resources without
111    *    incoming predicates to identify root notes, which can introduce a considerable performance
112    *    degradation.
113    */
114   public static final String RDF_addRootProperty = "RdfSerializer.addRootProperty";
115
116   /**
117    * <b>Configuration property:</b>  Auto-detect namespace usage.
118    * <p>
119    * <ul>
120    *    <li><b>Name:</b> <js>"RdfSerializer.autoDetectNamespaces"</js>
121    *    <li><b>Data type:</b> <code>Boolean</code>
122    *    <li><b>Default:</b> <jk>true</jk>
123    * </ul>
124    * <p>
125    * Detect namespace usage before serialization.
126    * <p>
127    * If enabled, then the data structure will first be crawled looking for
128    * namespaces that will be encountered before the root element is
129    * serialized.
130    */
131   public static final String RDF_autoDetectNamespaces = "RdfSerializer.autoDetectNamespaces";
132
133   /**
134    * <b>Configuration property:</b>  Default namespaces.
135    * <p>
136    * <ul>
137    *    <li><b>Name:</b> <js>"RdfSerializer.namespaces.list"</js>
138    *    <li><b>Data type:</b> <code>List&lt;{@link Namespace}&gt;</code>
139    *    <li><b>Default:</b> empty list
140    * </ul>
141    * <p>
142    * The default list of namespaces associated with this serializer.
143    */
144   public static final String RDF_namespaces = "RdfSerializer.namespaces.list";
145
146
147   final boolean addLiteralTypes, addRootProperty, useXmlNamespaces, looseCollections, autoDetectNamespaces;
148   final String rdfLanguage;
149   final Namespace juneauNs;
150   final Namespace juneauBpNs;
151   final RdfCollectionFormat collectionFormat;
152   final Map<String,Object> jenaSettings = new HashMap<String,Object>();
153   final Namespace[] namespaces;
154
155   /**
156    * Constructor.
157    * <p>
158    * Typically only called from {@link ContextFactory#getContext(Class)}.
159    *
160    * @param cf The factory that created this context.
161    */
162   public RdfSerializerContext(ContextFactory cf) {
163      super(cf);
164      addLiteralTypes = cf.getProperty(RDF_addLiteralTypes, boolean.class, false);
165      addRootProperty = cf.getProperty(RDF_addRootProperty, boolean.class, false);
166      useXmlNamespaces = cf.getProperty(RDF_useXmlNamespaces, boolean.class, true);
167      looseCollections = cf.getProperty(RDF_looseCollections, boolean.class, false);
168      autoDetectNamespaces = cf.getProperty(RDF_autoDetectNamespaces, boolean.class, true);
169      rdfLanguage = cf.getProperty(RDF_language, String.class, "RDF/XML-ABBREV");
170      juneauNs = cf.getProperty(RDF_juneauNs, Namespace.class, new Namespace("j", "http://www.ibm.com/juneau/"));
171      juneauBpNs = cf.getProperty(RDF_juneauBpNs, Namespace.class, new Namespace("jp", "http://www.ibm.com/juneaubp/"));
172      collectionFormat = cf.getProperty(RDF_collectionFormat, RdfCollectionFormat.class, RdfCollectionFormat.DEFAULT);
173      namespaces = cf.getProperty(RDF_namespaces, Namespace[].class, new Namespace[0]);
174   }
175}
