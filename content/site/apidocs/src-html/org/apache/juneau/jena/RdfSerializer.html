031/**
032 * Serializes POJOs to RDF.
033 *
034 * <h5 class='section'>Configurable properties:</h5>
035 * <p>
036 * Refer to <a class="doclink" href="package-summary.html#SerializerConfigurableProperties">Configurable Properties</a>
037 *    for the entire list of configurable properties.
038 *
039 * <h6 class='topic'>Behavior-specific subclasses</h6>
040 * <p>
041 * The following direct subclasses are provided for language-specific serializers:
042 * <ul>
043 *    <li>{@link RdfSerializer.Xml} - RDF/XML.
044 *    <li>{@link RdfSerializer.XmlAbbrev} - RDF/XML-ABBREV.
045 *    <li>{@link RdfSerializer.NTriple} - N-TRIPLE.
046 *    <li>{@link RdfSerializer.Turtle} - TURTLE.
047 *    <li>{@link RdfSerializer.N3} - N3.
048 * </ul>
049 *
050 * <h5 class='section'>Additional information:</h5>
051 * <p>
052 * See <a class="doclink" href="package-summary.html#TOC">RDF Overview</a> for an overview of RDF support in Juneau.
053 */

181   @Override /* Serializer */
182   protected void doSerialize(SerializerSession session, Object o) throws Exception {
183
184      RdfSerializerSession s = (RdfSerializerSession)session;
185
186      Model model = s.getModel();
187      Resource r = null;
188
189      ClassMeta<?> cm = session.getClassMetaForObject(o);
190      if (s.isLooseCollections() && cm != null && cm.isCollectionOrArray()) {
191         Collection c = s.sort(cm.isCollection() ? (Collection)o : toList(cm.getInnerClass(), o));
192         for (Object o2 : c)
193            serializeAnything(s, o2, false, object(), "root", null, null);
194      } else {
195         RDFNode n = serializeAnything(s, o, false, s.getExpectedRootType(o), "root", null, null);
196         if (n.isLiteral()) {
197            r = model.createResource();
198            r.addProperty(s.getValueProperty(), n);
199         } else {
200            r = n.asResource();
201         }
202
203         if (s.isAddRootProp())
204            r.addProperty(s.getRootProp(), "true");
205      }
206
207      s.getRdfWriter().write(model, session.getWriter(), "http://unknown/");
208   }

210   private RDFNode serializeAnything(RdfSerializerSession session, Object o, boolean isURI, ClassMeta<?> eType, String attrName, BeanPropertyMeta bpm, Resource parentResource) throws SerializeException {
211      Model m = session.getModel();
212
213      ClassMeta<?> aType = null;       // The actual type
214      ClassMeta<?> wType = null;       // The wrapped type
215      ClassMeta<?> sType = object();   // The serialized type
216
217      aType = session.push(attrName, o, eType);
218
219      if (eType == null)
220         eType = object();
221
222      // Handle recursion
223      if (aType == null) {
224         o = null;
225         aType = object();
226      }
227
228      if (o != null) {
229
230         if (aType.isDelegate()) {
231            wType = aType;
232            aType = ((Delegate)o).getClassMeta();
233         }
234
235         sType = aType.getSerializedClassMeta();
236
237         // Swap if necessary
238         PojoSwap swap = aType.getPojoSwap();
239         if (swap != null) {
240            o = swap.swap(session, o);
241
242            // If the getSwapClass() method returns Object, we need to figure out
243            // the actual type now.
244            if (sType.isObject())
245               sType = session.getClassMetaForObject(o);
246         }
247      } else {
248         sType = eType.getSerializedClassMeta();
249      }
250
251      String typeName = session.getBeanTypeName(eType, aType, bpm);
252
253      RDFNode n = null;
254
255      if (o == null || sType.isChar() && ((Character)o).charValue() == 0) {
256         if (bpm != null) {
257            if (! session.isTrimNulls()) {
258               n = m.createResource(RDF_NIL);
259            }
260         } else {
261            n = m.createResource(RDF_NIL);
262         }
263
264      } else if (sType.isUri() || isURI) {
265         n = m.createResource(getUri(session, o, null));
266
267      } else if (sType.isCharSequence() || sType.isChar()) {
268         n = m.createLiteral(session.encodeTextInvalidChars(o));
269
270      } else if (sType.isNumber() || sType.isBoolean()) {
271         if (! session.isAddLiteralTypes())
272            n = m.createLiteral(o.toString());
273         else
274            n = m.createTypedLiteral(o);
275
276      } else if (sType.isMap() || (wType != null && wType.isMap())) {
277         if (o instanceof BeanMap) {
278            BeanMap bm = (BeanMap)o;
279            Object uri = null;
280            RdfBeanMeta rbm = (RdfBeanMeta)bm.getMeta().getExtendedMeta(RdfBeanMeta.class);
281            if (rbm.hasBeanUri())
282               uri = rbm.getBeanUriProperty().get(bm, null);
283            String uri2 = getUri(session, uri, null);
284            n = m.createResource(uri2);
285            serializeBeanMap(session, bm, (Resource)n, typeName);
286         } else {
287            Map m2 = (Map)o;
288            n = m.createResource();
289            serializeMap(session, m2, (Resource)n, sType);
290         }
291
292      } else if (sType.isBean()) {
293         BeanMap bm = session.toBeanMap(o);
294         Object uri = null;
295         RdfBeanMeta rbm = (RdfBeanMeta)bm.getMeta().getExtendedMeta(RdfBeanMeta.class);
296         if (rbm.hasBeanUri())
297            uri = rbm.getBeanUriProperty().get(bm, null);
298         String uri2 = getUri(session, uri, null);
299         n = m.createResource(uri2);
300         serializeBeanMap(session, bm, (Resource)n, typeName);
301
302      } else if (sType.isCollectionOrArray() || (wType != null && wType.isCollection())) {
303         Collection c = session.sort(sType.isCollection() ? (Collection)o : toList(sType.getInnerClass(), o));
304         RdfCollectionFormat f = session.getCollectionFormat();
305         RdfClassMeta rcm = sType.getExtendedMeta(RdfClassMeta.class);
306         if (rcm.getCollectionFormat() != RdfCollectionFormat.DEFAULT)
307            f = rcm.getCollectionFormat();
308         if (bpm != null && bpm.getExtendedMeta(RdfBeanPropertyMeta.class).getCollectionFormat() != RdfCollectionFormat.DEFAULT)
309            f = bpm.getExtendedMeta(RdfBeanPropertyMeta.class).getCollectionFormat();
310         switch (f) {
311            case BAG: n = serializeToContainer(session, c, eType, m.createBag()); break;
312            case LIST: n = serializeToList(session, c, eType); break;
313            case MULTI_VALUED: serializeToMultiProperties(session, c, eType, bpm, attrName, parentResource); break;
314            default: n = serializeToContainer(session, c, eType, m.createSeq());
315         }
316      } else {
317         n = m.createLiteral(session.encodeTextInvalidChars(session.toString(o)));
318      }
319
320      session.pop();
321
322      return n;
323   }

325   private static String getUri(RdfSerializerSession session, Object uri, Object uri2) {
326      String s = null;
327      if (uri != null)
328         s = uri.toString();
329      if ((s == null || s.isEmpty()) && uri2 != null)
330         s = uri2.toString();
331      if (s == null)
332         return null;
333      if (s.indexOf("://") == -1) {
334         String aUri = session.getAbsolutePathUriBase();
335         String rUri = session.getRelativeUriBase();
336         if (StringUtils.startsWith(s, '/')) {
337            if (aUri != null)
338               return aUri + s;
339         } else {
340            if (rUri != null) {
341               if (rUri.equals("/"))
342                  return '/' + s;
343               return rUri + '/' + s;
344            }
345         }
346      }
347      return s;
348   }

350   private void serializeMap(RdfSerializerSession session, Map m, Resource r, ClassMeta<?> type) throws SerializeException {
351
352      m = session.sort(m);
353
354      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
355
356      ArrayList<Map.Entry<Object,Object>> l = new ArrayList<Map.Entry<Object,Object>>(m.entrySet());
357      Collections.reverse(l);
358      for (Map.Entry<Object,Object> me : l) {
359         Object value = me.getValue();
360
361         Object key = session.generalize(me.getKey(), keyType);
362
363         Namespace ns = session.getJuneauBpNs();
364         Model model = session.getModel();
365         Property p = model.createProperty(ns.getUri(), session.encodeElementName(session.toString(key)));
366         RDFNode n = serializeAnything(session, value, false, valueType, key == null ? null : session.toString(key), null, r);
367         if (n != null)
368            r.addProperty(p, n);
369      }
370   }
371
372   private void serializeBeanMap(RdfSerializerSession session, BeanMap<?> m, Resource r, String typeName) throws SerializeException {
373      List<BeanPropertyValue> l = m.getValues(session.isTrimNulls(), typeName != null ? session.createBeanTypeNameProperty(m, typeName) : null);
374      Collections.reverse(l);
375      for (BeanPropertyValue bpv : l) {
376         BeanPropertyMeta pMeta = bpv.getMeta();
377         ClassMeta<?> cMeta = pMeta.getClassMeta();
378
379         if (pMeta.getExtendedMeta(RdfBeanPropertyMeta.class).isBeanUri())
380            continue;
381
382         String key = bpv.getName();
383         Object value = bpv.getValue();
384         Throwable t = bpv.getThrown();
385         if (t != null)
386            session.addBeanGetterWarning(pMeta, t);
387
388         if (session.canIgnoreValue(cMeta, key, value))
389            continue;
390
391         BeanPropertyMeta bpm = bpv.getMeta();
392         Namespace ns = bpm.getExtendedMeta(RdfBeanPropertyMeta.class).getNamespace();
393         if (ns == null && session.isUseXmlNamespaces())
394            ns = bpm.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace();
395         if (ns == null)
396            ns = session.getJuneauBpNs();
397         else if (session.isAutoDetectNamespaces())
398            session.addModelPrefix(ns);
399
400         Property p = session.getModel().createProperty(ns.getUri(), session.encodeElementName(key));
401         RDFNode n = serializeAnything(session, value, pMeta.isUri(), cMeta, key, pMeta, r);
402         if (n != null)
403            r.addProperty(p, n);
404      }
405   }

408   private Container serializeToContainer(RdfSerializerSession session, Collection c, ClassMeta<?> type, Container list) throws SerializeException {
409
410      ClassMeta<?> elementType = type.getElementType();
411      for (Object e : c) {
412         RDFNode n = serializeAnything(session, e, false, elementType, null, null, null);
413         list = list.add(n);
414      }
415      return list;
416   }
417
418   private RDFList serializeToList(RdfSerializerSession session, Collection c, ClassMeta<?> type) throws SerializeException {
419      ClassMeta<?> elementType = type.getElementType();
420      List<RDFNode> l = new ArrayList<RDFNode>(c.size());
421      for (Object e : c) {
422         l.add(serializeAnything(session, e, false, elementType, null, null, null));
423      }
424      return session.getModel().createList(l.iterator());
425   }
426
427   private void serializeToMultiProperties(RdfSerializerSession session, Collection c, ClassMeta<?> sType, BeanPropertyMeta bpm, String attrName, Resource parentResource) throws SerializeException {
428      ClassMeta<?> elementType = sType.getElementType();
429      for (Object e : c) {
430         Namespace ns = null;
431         if (bpm != null) {
432            ns = bpm.getExtendedMeta(RdfBeanPropertyMeta.class).getNamespace();
433            if (ns == null && session.isUseXmlNamespaces())
434               ns = bpm.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace();
435         }
436         if (ns == null)
437            ns = session.getJuneauBpNs();
438         else if (session.isAutoDetectNamespaces())
439            session.addModelPrefix(ns);
440         RDFNode n2 = serializeAnything(session, e, false, elementType, null, null, null);
441         Property p = session.getModel().createProperty(ns.getUri(), session.encodeElementName(attrName));
442         parentResource.addProperty(p, n2);
443      }
