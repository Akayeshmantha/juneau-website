013package org.apache.juneau.msgpack;
014
015import org.apache.juneau.*;
016import org.apache.juneau.serializer.*;
017
018/**
019 * Configurable properties on the {@link MsgPackSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link PropertyStore#setProperty(String, Object)} on the property store
022 * passed into the constructor.
023 * <p>
024 * See {@link PropertyStore} for more information about context properties.
025 *
026 * <h6 class='topic'>Inherited configurable properties</h6>
027 * <ul class='doctree'>
028 *    <li class='jc'><a class="doclink" href="../BeanContext.html#ConfigProperties">BeanContext</a>
029 *       - Properties associated with handling beans on serializers and parsers.
030 *    <ul>
031 *       <li class='jc'><a class="doclink"
032 *          href="../serializer/SerializerContext.html#ConfigProperties">SerializerContext</a>
033 *          - Configurable properties common to all serializers.
034 *    </ul>
035 * </ul>
036 */
037public final class MsgPackSerializerContext extends SerializerContext {
038
039   /**
040    * <b>Configuration property:</b>  Add <js>"_type"</js> properties when needed.
041    * <p>
042    * <ul>
043    *    <li><b>Name:</b> <js>"MsgPackSerializer.addBeanTypeProperties"</js>
044    *    <li><b>Data type:</b> <code>Boolean</code>
045    *    <li><b>Default:</b> <jk>false</jk>
046    *    <li><b>Session-overridable:</b> <jk>true</jk>
047    * </ul>
048    * <p>
049    * If <jk>true</jk>, then <js>"_type"</js> properties will be added to beans if their type cannot be inferred
050    * through reflection.
051    * This is used to recreate the correct objects during parsing if the object types cannot be inferred.
052    * For example, when serializing a {@code Map<String,Object>} field, where the bean class cannot be determined from
053    * the value type.
054    * <p>
055    * When present, this value overrides the {@link SerializerContext#SERIALIZER_addBeanTypeProperties} setting and is
056    * provided to customize the behavior of specific serializers in a {@link SerializerGroup}.
057    */
058   public static final String MSGPACK_addBeanTypeProperties = "MsgPackSerializer.addBeanTypeProperties";
059
060   final boolean
061      addBeanTypeProperties;
062
063   /**
064    * Constructor.
065    * <p>
066    * Typically only called from {@link PropertyStore#getContext(Class)}.
067    *
068    * @param ps The property store that created this context.
069    */
070   public MsgPackSerializerContext(PropertyStore ps) {
071      super(ps);
072      addBeanTypeProperties = ps.getProperty(MSGPACK_addBeanTypeProperties, boolean.class,
073         ps.getProperty(SERIALIZER_addBeanTypeProperties, boolean.class, true));
074   }
075
076   @Override /* Context */
077   public ObjectMap asMap() {
078      return super.asMap()
079         .append("MsgPackSerializerContext", new ObjectMap()
080            .append("addBeanTypeProperties", addBeanTypeProperties)
081         );
082   }
083}
