023/**
024 * Serializes POJO models to MessagePack.
025 *
026 * <h5 class='section'>Media types:</h5>
027 * <p>
028 * Handles <code>Accept</code> types: <code>octal/msgpack</code>
029 * <p>
030 * Produces <code>Content-Type</code> types: <code>octal/msgpack</code>
031 *
032 * <h5 class='section'>Configurable properties:</h5>
033 * <p>
034 * This class has the following properties associated with it:
035 * <ul>
036 *    <li>{@link MsgPackSerializerContext}
037 *    <li>{@link SerializerContext}
038 *    <li>{@link BeanContext}
039 * </ul>
040 */
041@Produces("octal/msgpack")
042public class MsgPackSerializer extends OutputStreamSerializer {
043
044   /** Default serializer, all default settings.*/
045   public static final MsgPackSerializer DEFAULT = new MsgPackSerializer().lock();
046
047   /**
048    * Workhorse method. Determines the type of object, and then calls the
049    * appropriate type-specific serialization method.
050    */
051   @SuppressWarnings({ "rawtypes", "unchecked" })
052   MsgPackOutputStream serializeAnything(MsgPackSerializerSession session, MsgPackOutputStream out, Object o, ClassMeta<?> eType, String attrName, BeanPropertyMeta pMeta) throws Exception {
053
054      if (o == null)
055         return out.appendNull();
056
057      if (eType == null)
058         eType = object();
059
060      ClassMeta<?> aType;        // The actual type
061      ClassMeta<?> sType;        // The serialized type
062
063      aType = session.push(attrName, o, eType);
064      boolean isRecursion = aType == null;
065
066      // Handle recursion
067      if (aType == null) {
068         o = null;
069         aType = object();
070      }
071
072      sType = aType.getSerializedClassMeta();
073      String typeName = session.getBeanTypeName(eType, aType, pMeta);
074
075      // Swap if necessary
076      PojoSwap swap = aType.getPojoSwap();
077      if (swap != null) {
078         o = swap.swap(session, o);
079
080         // If the getSwapClass() method returns Object, we need to figure out
081         // the actual type now.
082         if (sType.isObject())
083            sType = session.getClassMetaForObject(o);
084      }
085
086      // '\0' characters are considered null.
087      if (o == null || (sType.isChar() && ((Character)o).charValue() == 0))
088         out.appendNull();
089      else if (sType.isBoolean())
090         out.appendBoolean((Boolean)o);
091      else if (sType.isNumber())
092         out.appendNumber((Number)o);
093      else if (sType.isBean())
094         serializeBeanMap(session, out, session.toBeanMap(o), typeName);
095      else if (sType.isUri() || (pMeta != null && pMeta.isUri()))
096         out.appendString(session.resolveUri(o.toString()));
097      else if (sType.isMap()) {
098         if (o instanceof BeanMap)
099            serializeBeanMap(session, out, (BeanMap)o, typeName);
100         else
101            serializeMap(session, out, (Map)o, eType);
102      }
103      else if (sType.isCollection()) {
104         serializeCollection(session, out, (Collection) o, eType);
105      }
106      else if (sType.isArray()) {
107         serializeCollection(session, out, toList(sType.getInnerClass(), o), eType);
108      } else
109         out.appendString(session.toString(o));
110
111      if (! isRecursion)
112         session.pop();
113      return out;
114   }
115
116   @SuppressWarnings({ "rawtypes", "unchecked" })
117   private void serializeMap(MsgPackSerializerSession session, MsgPackOutputStream out, Map m, ClassMeta<?> type) throws Exception {
118
119      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
120
121      m = session.sort(m);
122
123      // The map size may change as we're iterating over it, so
124      // grab a snapshot of the entries in a separate list.
125      List<SimpleMapEntry> entries = new ArrayList<SimpleMapEntry>(m.size());
126      for (Map.Entry e : (Set<Map.Entry>)m.entrySet())
127         entries.add(new SimpleMapEntry(e.getKey(), e.getValue()));
128
129      out.startMap(entries.size());
130
131      for (SimpleMapEntry e : entries) {
132         Object value = e.value;
133         Object key = session.generalize(e.key, keyType);
134
135         serializeAnything(session, out, key, keyType, null, null);
136         serializeAnything(session, out, value, valueType, null, null);
137      }
138   }
139
140   private void serializeBeanMap(MsgPackSerializerSession session, MsgPackOutputStream out, final BeanMap<?> m, String typeName) throws Exception {
141
142      List<BeanPropertyValue> values = m.getValues(session.isTrimNulls(), typeName != null ? session.createBeanTypeNameProperty(m, typeName) : null);
143
144      int size = values.size();
145      for (BeanPropertyValue p : values)
146         if (p.getThrown() != null)
147            size--;
148      out.startMap(size);
149
150      for (BeanPropertyValue p : values) {
151         BeanPropertyMeta pMeta = p.getMeta();
152         ClassMeta<?> cMeta = p.getClassMeta();
153         String key = p.getName();
154         Object value = p.getValue();
155         Throwable t = p.getThrown();
156         if (t != null)
157            session.addBeanGetterWarning(pMeta, t);
158         else {
159            serializeAnything(session, out, key, null, null, null);
160            serializeAnything(session, out, value, cMeta, key, pMeta);
161         }
162      }
163   }
164
165   private static class SimpleMapEntry {
166      final Object key;
167      final Object value;
168
169      private SimpleMapEntry(Object key, Object value) {
170         this.key = key;
171         this.value = value;
172      }
173   }
174
175   @SuppressWarnings({"rawtypes", "unchecked"})
176   private void serializeCollection(MsgPackSerializerSession session, MsgPackOutputStream out, Collection c, ClassMeta<?> type) throws Exception {
177
178      ClassMeta<?> elementType = type.getElementType();
179      List<Object> l = new ArrayList<Object>(c.size());
180
181      c = session.sort(c);
182      l.addAll(c);
183
184      out.startArray(l.size());
185
186      for (Object o : l)
187         serializeAnything(session, out, o, elementType, "<iterator>", null);
188   }

195   @Override /* Serializer */
196   public MsgPackSerializerSession createSession(Object output, ObjectMap op, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType) {
197      return new MsgPackSerializerSession(getContext(MsgPackSerializerContext.class), op, output, javaMethod, locale, timeZone, mediaType);
198   }
199
200   @Override /* Serializer */
201   protected void doSerialize(SerializerSession session, Object o) throws Exception {
202      MsgPackSerializerSession s = (MsgPackSerializerSession)session;
203      serializeAnything(s, s.getOutputStream(), o, null, "root", null);
204   }
