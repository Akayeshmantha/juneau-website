013package org.apache.juneau.msgpack;
014
015import static org.apache.juneau.msgpack.MsgPackSerializerContext.*;
016
017import java.lang.reflect.*;
018import java.util.*;
019
020import org.apache.juneau.*;
021import org.apache.juneau.http.*;
022import org.apache.juneau.json.*;
023import org.apache.juneau.serializer.*;
024
025/**
026 * Session object that lives for the duration of a single use of {@link MsgPackSerializer}.
027 * <p>
028 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
029 */
030public final class MsgPackSerializerSession extends SerializerSession {
031
032   private final boolean
033      addBeanTypeProperties;
034
035   /**
036    * Create a new session using properties specified in the context.
037    *
038    * @param ctx The context creating this session object.
039    * The context contains all the configuration settings for this object.
040    * @param output The output object.  See {@link JsonSerializerSession#getOutputStream()} for valid class types.
041    * @param op The override properties.
042    * These override any context properties defined in the context.
043    * @param javaMethod The java method that called this serializer, usually the method in a REST servlet.
044    * @param locale The session locale.
045    * If <jk>null</jk>, then the locale defined on the context is used.
046    * @param timeZone The session timezone.
047    * If <jk>null</jk>, then the timezone defined on the context is used.
048    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
049    */
050   protected MsgPackSerializerSession(MsgPackSerializerContext ctx, ObjectMap op, Object output, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType) {
051      super(ctx, op, output, javaMethod, locale, timeZone, mediaType);
052      if (op == null || op.isEmpty()) {
053         addBeanTypeProperties = ctx.addBeanTypeProperties;
054      } else {
055         addBeanTypeProperties = op.getBoolean(MSGPACK_addBeanTypeProperties, ctx.addBeanTypeProperties);
056      }
057   }
058
059   /**
060    * Returns the {@link MsgPackSerializerContext#MSGPACK_addBeanTypeProperties} setting value for this session.
061    *
062    * @return The {@link MsgPackSerializerContext#MSGPACK_addBeanTypeProperties} setting value for this session.
063    */
064   @Override /* SerializerSession */
065   public final boolean isAddBeanTypeProperties() {
066      return addBeanTypeProperties;
067   }
068
069   @Override /*SerializerSession */
070   public MsgPackOutputStream getOutputStream() throws Exception {
071      Object output = getOutput();
072      if (output instanceof MsgPackOutputStream)
073         return (MsgPackOutputStream)output;
074      return new MsgPackOutputStream(super.getOutputStream());
075   }
076}
