013package org.apache.juneau.serializer;
014
015import java.util.*;
016import java.util.concurrent.*;
017
018import org.apache.juneau.*;
019import org.apache.juneau.internal.*;
020
021/**
022 * Represents a group of {@link Serializer Serializers} that can be looked up by media type.
023 *
024 * <h5 class='section'>Description:</h5>
025 * <p>
026 * Provides the following features:
027 * <ul class='spaced-list'>
028 *    <li>Finds serializers based on HTTP <code>Accept</code> header values.
029 *    <li>Sets common properties on all serializers in a single method call.
030 *    <li>Locks all serializers in a single method call.
031 *    <li>Clones existing groups and all serializers within the group in a single method call.
032 * </ul>
033 *
034 * <h6 class='topic'>Match ordering</h6>
035 * <p>
036 * Serializers are matched against <code>Accept</code> strings in the order they exist in this group.
037 * <p>
038 * Adding new entries will cause the entries to be prepended to the group.
039 * This allows for previous serializers to be overridden through subsequent calls.
040 * <p>
041 * For example, calling <code>g.append(S1.<jk>class</jk>,S2.<jk>class</jk>).append(S3.<jk>class</jk>,S4.<jk>class</jk>)</code>
042 * will result in the order <code>S3, S4, S1, S2</code>.
043 *
044 * <h5 class='section'>Example:</h5>
045 * <p class='bcode'>
046 *    <jc>// Construct a new serializer group</jc>
047 *    SerializerGroup g = <jk>new</jk> SerializerGroupBuilder();
048 *       .append(JsonSerializer.<jk>class</jk>, XmlSerializer.<jk>class</jk>); <jc>// Add some serializers to it</jc>
049 *       .ws().pojoSwaps(CalendarSwap.ISO8601DT.<jk>class</jk>) <jc>// Change settings for all serializers in the group.</jc>
050 *       .build();
051 *
052 *    <jc>// Find the appropriate serializer by Accept type</jc>
053 *    String mediaTypeMatch = g.findMatch(<js>"text/foo, text/json;q=0.8, text/*;q:0.6, *\/*;q=0.0"</js>);
054 *    WriterSerializer s = g.getWriterSerializer(mediaTypeMatch);
055 *
056 *    <jc>// Serialize a bean to JSON text </jc>
057 *    AddressBook addressBook = <jk>new</jk> AddressBook();  <jc>// Bean to serialize.</jc>
058 *    String json = s.serialize(addressBook);
059 * </p>
060 */
061public final class SerializerGroup {
062
063   // Maps Accept headers to matching serializers.
064   private final Map<String,SerializerMatch> cache = new ConcurrentHashMap<String,SerializerMatch>();
065
066   final Serializer[] serializers;
067   private final PropertyStore propertyStore;
068
069   /**
070    * Constructor.
071    *
072    * @param propertyStore The modifiable properties that were used to initialize the serializers.
073    * A snapshot of these will be made so that we can clone and modify this group.
074    * @param serializers The serializers defined in this group.
075    * The order is important because they will be tried in reverse order (e.g.
076    *    newer first) in which they will be tried to match against media types.
077    */
078   public SerializerGroup(PropertyStore propertyStore, Serializer[] serializers) {
079      this.propertyStore = PropertyStore.create(propertyStore);
080      this.serializers = ArrayUtils.reverse(serializers);
081   }
082
083   /**
084    * Searches the group for a serializer that can handle the specified <code>Accept</code> value.
085    * <p>
086    * The <code>accept</code> value complies with the syntax described in RFC2616, Section 14.1, as described below:
087    * <p class='bcode'>
088    *    Accept         = "Accept" ":"
089    *                      #( media-range [ accept-params ] )
090    *
091    *    media-range    = ( "*\/*"
092    *                      | ( type "/" "*" )
093    *                      | ( type "/" subtype )
094    *                      ) *( ";" parameter )
095    *    accept-params  = ";" "q" "=" qvalue *( accept-extension )
096    *    accept-extension = ";" token [ "=" ( token | quoted-string ) ]
097    * </p>
098    * <p>
099    * The general idea behind having the serializer resolution be a two-step process is so that
100    *    the matched media type can be passed in to the {@link WriterSerializer#doSerialize(SerializerSession, Object)} method.
101    * For example...
102    * <p class='bcode'>
103    *    String acceptHeaderValue = request.getHeader(<js>"Accept"</js>);
104    *    String matchingMediaType = group.findMatch(acceptHeaderValue);
105    *    if (matchingMediaType == <jk>null</jk>)
106    *       <jk>throw new</jk> RestException(<jsf>SC_NOT_ACCEPTABLE</jsf>);
107    *    WriterSerializer s = (WriterSerializer)group.getSerializer(matchingMediaType);
108    *    s.serialize(getPojo(), response.getWriter(), response.getProperties(), matchingMediaType);
109    * </p>
110    *
111    * @param acceptHeader The HTTP <l>Accept</l> header string.
112    * @return The serializer and media type that matched the accept header, or <jk>null</jk> if no match was made.
113    */
114   public SerializerMatch getSerializerMatch(String acceptHeader) {
115      SerializerMatch sm = cache.get(acceptHeader);
116      if (sm != null)
117         return sm;
118
119      MediaRange[] mr = MediaRange.parse(acceptHeader);
120      if (mr.length == 0)
121         mr = MediaRange.parse("*/*");
122
123      Map<Float,SerializerMatch> m = null;
124
125      for (MediaRange a : mr) {
126         for (Serializer s : serializers) {
127            for (MediaType a2 : s.getMediaTypes()) {
128               float q = a.matches(a2);
129               if (q == 1) {
130                  sm = new SerializerMatch(a2, s);
131                  cache.put(acceptHeader, sm);
132                  return sm;
133               } else if (q > 0) {
134                  if (m == null)
135                     m = new TreeMap<Float,SerializerMatch>(Collections.reverseOrder());
136                  m.put(q, new SerializerMatch(a2, s));
137               }
138            }
139         }
140      }
141
142      return (m == null ? null : m.values().iterator().next());
143   }
144
145   /**
146    * Same as {@link #getSerializerMatch(String)} but matches using a {@link MediaType} instance.
147    *
148    * @param mediaType The HTTP media type.
149    * @return The serializer and media type that matched the media type, or <jk>null</jk> if no match was made.
150    */
151   public SerializerMatch getSerializerMatch(MediaType mediaType) {
152      return getSerializerMatch(mediaType.toString());
153   }
154
155   /**
156    * Same as {@link #getSerializerMatch(String)} but returns just the matched serializer.
157    *
158    * @param acceptHeader The HTTP <l>Accept</l> header string.
159    * @return The serializer that matched the accept header, or <jk>null</jk> if no match was made.
160    */
161   public Serializer getSerializer(String acceptHeader) {
162      SerializerMatch sm = getSerializerMatch(acceptHeader);
163      return sm == null ? null : sm.getSerializer();
164   }
165
166   /**
167    * Same as {@link #getSerializerMatch(MediaType)} but returns just the matched serializer.
168    *
169    * @param mediaType The HTTP media type.
170    * @return The serializer that matched the accept header, or <jk>null</jk> if no match was made.
171    */
172   public Serializer getSerializer(MediaType mediaType) {
173      if (mediaType == null)
174         return null;
175      return getSerializer(mediaType.toString());
176   }
177
178   /**
179    * Returns the media types that all serializers in this group can handle.
180    * <p>
181    * Entries are ordered in the same order as the serializers in the group.
182    *
183    * @return The list of media types.
184    */
185   public List<MediaType> getSupportedMediaTypes() {
186      List<MediaType> l = new ArrayList<MediaType>();
187      for (Serializer s : serializers)
188         for (MediaType mt : s.getMediaTypes())
189            if (! l.contains(mt))
190               l.add(mt);
191      return l;
192   }
193
194   /**
195    * Returns a copy of the property store that was used to create the serializers in this group.
196    * This method returns a new factory each time so is somewhat expensive.
197    *
198    * @return A new copy of the property store passed in to the constructor.
199    */
200   public PropertyStore createPropertyStore() {
201      return PropertyStore.create(propertyStore);
202   }
203
204   /**
205    * Returns a copy of the serializers in this group.
206    * This method returns a new array each time so is somewhat expensive.
207    *
208    * @return A new array containing the serializers in this group.
209    */
210   public Serializer[] getSerializers() {
211      return ArrayUtils.reverse(serializers);
212   }
213}
