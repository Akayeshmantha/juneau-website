040   /**
041    * Create an empty serializer group using the specified property store for settings.
042    * <p>
043    * Note:  Modifying the specified property store externally will also modify it here.
044    *
045    * @param propertyStore The property store containing all settings common to all serializers in this group.
046    */
047   public SerializerGroupBuilder(PropertyStore propertyStore) {
048      this.serializers = new ArrayList<Object>();
049      this.propertyStore = propertyStore;
050   }
051
052   /**
053    * Clone an existing serializer group builder.
054    * @param copyFrom The serializer group that we're copying settings and serializers from.
055    */
056   public SerializerGroupBuilder(SerializerGroup copyFrom) {
057      this.serializers = new ArrayList<Object>();
058      addReverse(serializers, copyFrom.getSerializers());
059      this.propertyStore = copyFrom.createPropertyStore();
060   }

095   /**
096    * Creates a new {@link SerializerGroup} object using a snapshot of the settings defined in this builder.
097    * <p>
098    * This method can be called multiple times to produce multiple serializer groups.
099    *
100    * @return A new {@link SerializerGroup} object.
101    */
102   @SuppressWarnings("unchecked")
103   public SerializerGroup build() {
104      List<Serializer> l = new ArrayList<Serializer>();
105      for (Object s : serializers) {
106         Class<? extends Serializer> c = null;
107         PropertyStore ps = propertyStore;
108         if (s instanceof Class) {
109            c = ((Class<? extends Serializer>)s);
110         } else {
111            // Note that if we added a serializer instance, we want a new instance with this builder's properties
112            // on top of the previous serializer's properties.
113            Serializer s2 = (Serializer)s;
114            ps = s2.createPropertyStore().copyFrom(propertyStore);
115            c = s2.getClass();
116         }
117         try {
118            l.add(c.getConstructor(PropertyStore.class).newInstance(ps));
119         } catch (Exception e) {
120            throw new RuntimeException("Could not instantiate serializer " + c.getName(), e);
121         }
122      }
123      Collections.reverse(l);
124      return new SerializerGroup(propertyStore, l.toArray(new Serializer[l.size()]));
125   }
