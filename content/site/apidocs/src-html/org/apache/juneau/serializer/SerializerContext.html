133   /**
134    * <b>Configuration property:</b>  Max serialization depth.
135    * <p>
136    * <ul>
137    *    <li><b>Name:</b> <js>"Serializer.maxDepth"</js>
138    *    <li><b>Data type:</b> <code>Integer</code>
139    *    <li><b>Default:</b> <code>100</code>
140    *    <li><b>Session-overridable:</b> <jk>true</jk>
141    * </ul>
142    * <p>
143    * Abort serialization if specified depth is reached in the POJO tree.
144    * If this depth is exceeded, an exception is thrown.
145    * This prevents stack overflows from occurring when trying to serialize models with recursive references.
146    */
147   public static final String SERIALIZER_maxDepth = "Serializer.maxDepth";

155    *    <li><b>Default:</b> <code>0</code>
156    *    <li><b>Session-overridable:</b> <jk>true</jk>
157    * </ul>
158    * <p>
159    * The initial indentation level at the root.

164   /**
165    * <b>Configuration property:</b>  Automatically detect POJO recursions.
166    * <p>
167    * <ul>
168    *    <li><b>Name:</b> <js>"Serializer.detectRecursions"</js>
169    *    <li><b>Data type:</b> <code>Boolean</code>
170    *    <li><b>Default:</b> <jk>false</jk>
171    *    <li><b>Session-overridable:</b> <jk>true</jk>
172    * </ul>
173    * <p>
174    * Specifies that recursions should be checked for during serialization.
175    * <p>
176    * Recursions can occur when serializing models that aren't true trees, but rather contain loops.
177    * <p>
178    * The behavior when recursions are detected depends on the value for {@link #SERIALIZER_ignoreRecursions}.
179    * <p>
180    * For example, if a model contains the links A-&gt;B-&gt;C-&gt;A, then the JSON generated will look like
181    *    the following when <jsf>SERIALIZER_ignoreRecursions</jsf> is <jk>true</jk>...
182    * <code>{A:{B:{C:null}}}</code><br>
183    * <p>
184    * <h5 class='section'>Notes:</h5>
185    * <ul>
186    *    <li>Checking for recursion can cause a small performance penalty.
187    * </ul>
188    */
189   public static final String SERIALIZER_detectRecursions = "Serializer.detectRecursions";
190
191   /**
192    * <b>Configuration property:</b>  Ignore recursion errors.
193    * <p>
194    * <ul>
195    *    <li><b>Name:</b> <js>"Serializer.ignoreRecursions"</js>
196    *    <li><b>Data type:</b> <code>Boolean</code>
197    *    <li><b>Default:</b> <jk>false</jk>
198    *    <li><b>Session-overridable:</b> <jk>true</jk>
199    * </ul>
200    * <p>
201    * Used in conjunction with {@link #SERIALIZER_detectRecursions}.
202    * Setting is ignored if <jsf>SERIALIZER_detectRecursions</jsf> is <jk>false</jk>.
203    * <p>
204    * If <jk>true</jk>, when we encounter the same object when serializing a tree,
205    *    we set the value to <jk>null</jk>.
206    * Otherwise, an exception is thrown.
207    */

442   /**
443    * Constructor.
444    *
445    * @param cf The factory that created this context.
446    */
447   public SerializerContext(ContextFactory cf) {
448      super(cf);
449      maxDepth = cf.getProperty(SERIALIZER_maxDepth, int.class, 100);
450      initialDepth = cf.getProperty(SERIALIZER_initialDepth, int.class, 0);
451      detectRecursions = cf.getProperty(SERIALIZER_detectRecursions, boolean.class, false);
452      ignoreRecursions = cf.getProperty(SERIALIZER_ignoreRecursions, boolean.class, false);
453      useIndentation = cf.getProperty(SERIALIZER_useIndentation, boolean.class, false);
454      addBeanTypeProperties = cf.getProperty(SERIALIZER_addBeanTypeProperties, boolean.class, false);
455      trimNulls = cf.getProperty(SERIALIZER_trimNullProperties, boolean.class, true);
456      trimEmptyCollections = cf.getProperty(SERIALIZER_trimEmptyCollections, boolean.class, false);
457      trimEmptyMaps = cf.getProperty(SERIALIZER_trimEmptyMaps, boolean.class, false);
458      trimStrings = cf.getProperty(SERIALIZER_trimStrings, boolean.class, false);
459      sortCollections = cf.getProperty(SERIALIZER_sortCollections, boolean.class, false);
460      sortMaps = cf.getProperty(SERIALIZER_sortMaps, boolean.class, false);
461      quoteChar = cf.getProperty(SERIALIZER_quoteChar, String.class, "\"").charAt(0);
462      relativeUriBase = resolveRelativeUriBase(cf.getProperty(SERIALIZER_relativeUriBase, String.class, ""));
463      absolutePathUriBase = resolveAbsolutePathUriBase(cf.getProperty(SERIALIZER_absolutePathUriBase, String.class, ""));
464   }
465
466   private String resolveRelativeUriBase(String s) {
467      if (StringUtils.isEmpty(s))
468         return null;
469      if (s.equals("/"))
470         return s;
471      else if (StringUtils.endsWith(s, '/'))
472         s = s.substring(0, s.length()-1);
473      return s;
474   }
475
476   private String resolveAbsolutePathUriBase(String s) {
477      if (StringUtils.isEmpty(s))
478         return null;
479      if (StringUtils.endsWith(s, '/'))
480         s = s.substring(0, s.length()-1);
481      return s;
482   }
