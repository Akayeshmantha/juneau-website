013package org.apache.juneau.serializer;
014
015import static org.apache.juneau.internal.StringUtils.*;
016import static org.apache.juneau.internal.ReflectionUtils.*;
017
018import java.io.*;
019
020import org.apache.juneau.*;
021import org.apache.juneau.annotation.*;
022import org.apache.juneau.http.*;
023
024/**
025 * Parent class for all Juneau serializers.
026 *
027 * <h5 class='section'>Description:</h5>
028 *
029 * Base serializer class that serves as the parent class for all serializers.
030 *
031 * <p>
032 * The purpose of this class is:
033 * <ul>
034 *    <li>Maintain a read-only configuration state of a serializer (i.e. {@link SerializerContext}).
035 *    <li>Create session objects used for serializing POJOs (i.e. {@link SerializerSession}).
036 *    <li>Provide convenience methods for serializing POJOs without having to construct session objects.
037 * </ul>
038 *
039 * <p>
040 * Subclasses should extend directly from {@link OutputStreamSerializer} or {@link WriterSerializer} depending on
041 * whether it's a stream or character based serializer.
042 *
043 * <h6 class='topic'>@Produces annotation</h6>
044 *
045 * The media types that this serializer can produce is specified through the {@link Produces @Produces} annotation.
046 * <br>
047 * However, the media types can also be specified programmatically by overriding the {@link #getMediaTypes()}
048 * and {@link #getResponseContentType()} methods.
049 */
050public abstract class Serializer extends CoreObject {
051
052   private final MediaType[] mediaTypes;
053   private final MediaType contentType;
054
055   // Hidden constructors to force subclass from OuputStreamSerializer or WriterSerializer.
056   Serializer(PropertyStore propertyStore) {
057      super(propertyStore);
058
059      Produces p = getAnnotation(Produces.class, getClass());
060      if (p == null)
061         throw new FormattedRuntimeException("Class ''{0}'' is missing the @Produces annotation", getClass());
062
063      String[] mt = split(p.value());
064      this.mediaTypes = new MediaType[mt.length];
065      for (int i = 0; i < mt.length; i++) {
066         mediaTypes[i] = MediaType.forString(mt[i]);
067      }
068
069      String ct = p.contentType().isEmpty() ? this.mediaTypes[0].toString() : p.contentType();
070      contentType = ct.isEmpty() ? null : MediaType.forString(ct);
071   }
072
073   @Override /* CoreObject */
074   public SerializerBuilder builder() {
075      return new SerializerBuilder(propertyStore);
076   }
077
078   //--------------------------------------------------------------------------------
079   // Abstract methods
080   //--------------------------------------------------------------------------------
081
082   /**
083    * Returns <jk>true</jk> if this serializer subclasses from {@link WriterSerializer}.
084    *
085    * @return <jk>true</jk> if this serializer subclasses from {@link WriterSerializer}.
086    */
087   public abstract boolean isWriterSerializer();
088
089   /**
090    * Create the session object used for actual serialization of objects.
091    *
092    * @param args
093    *    Runtime arguments.
094    *    These specify session-level information such as locale and URI context.
095    *    It also include session-level properties that override the properties defined on the bean and serializer
096    *    contexts.
097    *    <br>If <jk>null</jk>, defaults to {@link SerializerSessionArgs#DEFAULT}.
098    * @return
099    *    The new session object.
100    *    <br>Note that you must call {@link SerializerSession#close()} on this object to perform any necessary
101    *    cleanup.
102    */
103   public abstract SerializerSession createSession(SerializerSessionArgs args);
104
105
106   //--------------------------------------------------------------------------------
107   // Convenience methods
108   //--------------------------------------------------------------------------------
109
110   /**
111    * Shortcut for calling <code>createSession(<jk>null</jk>)</code>.
112    *
113    * @return
114    *    The new session object.
115    *    <br>Note that you must call {@link SerializerSession#close()} on this object to perform any necessary
116    *    cleanup.
117    */
118   public final SerializerSession createSession() {
119      return createSession(null);
120   }
121
122   /**
123    * Serializes a POJO to the specified output stream or writer.
124    *
125    * <p>
126    * Equivalent to calling <code>serializer.createSession().serialize(o, output);</code>
127    *
128    * @param o The object to serialize.
129    * @param output
130    *    The output object.
131    *    <br>Character-based serializers can handle the following output class types:
132    *    <ul>
133    *       <li>{@link Writer}
134    *       <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
135    *       <li>{@link File} - Output will be written as system-default encoded stream.
136    *       <li>{@link StringBuilder} - Output will be written to the specified string builder.
137    *    </ul>
138    *    <br>Stream-based serializers can handle the following output class types:
139    *    <ul>
140    *       <li>{@link OutputStream}
141    *       <li>{@link File}
142    *    </ul>
143    * @throws SerializeException If a problem occurred trying to convert the output.
144    */
145   public final void serialize(Object o, Object output) throws SerializeException {
146      SerializerSession s = createSession();
147      try {
148         s.serialize(o, output);
149      } finally {
150         s.close();
151      }
152   }
153
154   /**
155    * Shortcut method for serializing objects directly to either a <code>String</code> or <code><jk>byte</jk>[]</code>
156    * depending on the serializer type.
157    *
158    * @param o The object to serialize.
159    * @return
160    *    The serialized object.
161    *    <br>Character-based serializers will return a <code>String</code>
162    *    <br>Stream-based serializers will return a <code><jk>byte</jk>[]</code>
163    * @throws SerializeException If a problem occurred trying to convert the output.
164    */
165   public Object serialize(Object o) throws SerializeException {
166      SerializerSession s = createSession();
167      try {
168         return s.serialize(o);
169      } finally {
170         s.close();
171      }
172   }
173
174   //--------------------------------------------------------------------------------
175   // Other methods
176   //--------------------------------------------------------------------------------
177
178   /**
179    * Returns the media types handled based on the value of the {@link Produces} annotation on the serializer class.
180    *
181    * <p>
182    * This method can be overridden by subclasses to determine the media types programmatically.
183    *
184    * @return The list of media types.  Never <jk>null</jk>.
185    */
186   public final MediaType[] getMediaTypes() {
187      return mediaTypes;
188   }
189
190   /**
191    * Returns the first media type specified on this serializer via the {@link Produces} annotation.
192    *
193    * @return The media type.
194    */
195   public final MediaType getPrimaryMediaType() {
196      return mediaTypes == null || mediaTypes.length == 0 ? null : mediaTypes[0];
197   }
198
199   /**
200    * Optional method that returns the response <code>Content-Type</code> for this serializer if it is different from
201    * the matched media type.
202    *
203    * <p>
204    * This method is specified to override the content type for this serializer.
205    * For example, the {@link org.apache.juneau.json.JsonSerializer.Simple} class returns that it handles media type
206    * <js>"text/json+simple"</js>, but returns <js>"text/json"</js> as the actual content type.
207    * This allows clients to request specific 'flavors' of content using specialized <code>Accept</code> header values.
208    *
209    * <p>
210    * This method is typically meaningless if the serializer is being used stand-alone (i.e. outside of a REST server
211    * or client).
212    *
213    * @return The response content type.  If <jk>null</jk>, then the matched media type is used.
214    */
215   public MediaType getResponseContentType() {
216      return contentType;
217   }
218}
