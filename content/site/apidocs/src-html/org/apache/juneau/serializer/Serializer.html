027/**
028 * Parent class for all Juneau serializers.
029 *
030 * <h5 class='section'>Description:</h5>
031 *
032 * Base serializer class that serves as the parent class for all serializers.
033 *
034 * <p>
035 * Subclasses should extend directly from {@link OutputStreamSerializer} or {@link WriterSerializer}.
036 *
037 * <h6 class='topic'>@Produces annotation</h6>
038 *
039 * The media types that this serializer can produce is specified through the {@link Produces @Produces} annotation.
040 *
041 * <p>
042 * However, the media types can also be specified programmatically by overriding the {@link #getMediaTypes()}
043 * and {@link #getResponseContentType()} methods.
044 */
045public abstract class Serializer extends CoreObject {
046
047   private final MediaType[] mediaTypes;
048   private final MediaType contentType;
049   private final SerializerContext ctx;
050
051   // Hidden constructors to force subclass from OuputStreamSerializer or WriterSerializer.
052   Serializer(PropertyStore propertyStore) {
053      super(propertyStore);
054      this.ctx = createContext(SerializerContext.class);
055
056      Produces p = getAnnotation(Produces.class, getClass());
057      if (p == null)
058         throw new FormattedRuntimeException("Class ''{0}'' is missing the @Produces annotation", getClass());
059
060      String[] mt = split(p.value());
061      this.mediaTypes = new MediaType[mt.length];
062      for (int i = 0; i < mt.length; i++) {
063         mediaTypes[i] = MediaType.forString(mt[i]);
064      }
065
066      String ct = p.contentType().isEmpty() ? this.mediaTypes[0].toString() : p.contentType();
067      contentType = ct.isEmpty() ? null : MediaType.forString(ct);
068   }
069
070   @Override /* CoreObject */
071   public SerializerBuilder builder() {
072      return new SerializerBuilder(propertyStore);
073   }
074
075   /**
076    * Returns <jk>true</jk> if this serializer subclasses from {@link WriterSerializer}.
077    *
078    * @return <jk>true</jk> if this serializer subclasses from {@link WriterSerializer}.
079    */
080   public abstract boolean isWriterSerializer();

086   /**
087    * Serializes a POJO to the specified output stream or writer.
088    *
089    * <p>
090    * This method should NOT close the context object.
091    *
092    * @param session
093    *    The serializer session object return by {@link #createSession(Object, ObjectMap, Method, Locale, TimeZone,
094    *    MediaType, UriContext)}.
095    *    If <jk>null</jk>, session is created using {@link #createSession(Object)}.
096    * @param o The object to serialize.
097    * @throws Exception If thrown from underlying stream, or if the input contains a syntax error or is malformed.
098    */
099   protected abstract void doSerialize(SerializerSession session, Object o) throws Exception;
100
101   /**
102    * Shortcut method for serializing objects directly to either a <code>String</code> or <code><jk>byte</jk>[]</code>
103    * depending on the serializer type.
104    *
105    * @param o The object to serialize.
106    * @return
107    *    The serialized object.
108    *    <br>Character-based serializers will return a <code>String</code>
109    *    <br>Stream-based serializers will return a <code><jk>byte</jk>[]</code>
110    * @throws SerializeException If a problem occurred trying to convert the output.
111    */
112   public abstract Object serialize(Object o) throws SerializeException;
113
114   //--------------------------------------------------------------------------------
115   // Other methods
116   //--------------------------------------------------------------------------------
117
118   /**
119    * Serialize the specified object using the specified session.
120    *
121    * @param session
122    *    The serializer session object return by {@link #createSession(Object, ObjectMap, Method, Locale, TimeZone,
123    *    MediaType, UriContext)}.
124    *    If <jk>null</jk>, session is created using {@link #createSession(Object)}.
125    * @param o The object to serialize.
126    * @throws SerializeException If a problem occurred trying to convert the output.
127    */
128   public final void serialize(SerializerSession session, Object o) throws SerializeException {
129      try {
130         doSerialize(session, o);
131      } catch (SerializeException e) {
132         throw e;
133      } catch (StackOverflowError e) {
134         throw new SerializeException(session,
135            "Stack overflow occurred.  This can occur when trying to serialize models containing loops.  It's recommended you use the SerializerContext.SERIALIZER_detectRecursions setting to help locate the loop.").initCause(e);
136      } catch (Exception e) {
137         throw new SerializeException(session, e);
138      } finally {
139         session.close();
140      }
141   }
142
143   /**
144    * Serializes a POJO to the specified output stream or writer.
145    *
146    * <p>
147    * Equivalent to calling <code>serializer.serialize(o, out, <jk>null</jk>);</code>
148    *
149    * @param o The object to serialize.
150    * @param output
151    *    The output object.
152    *    <br>Character-based serializers can handle the following output class types:
153    *    <ul>
154    *       <li>{@link Writer}
155    *       <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
156    *       <li>{@link File} - Output will be written as system-default encoded stream.
157    *    </ul>
158    *    <br>Stream-based serializers can handle the following output class types:
159    *    <ul>
160    *       <li>{@link OutputStream}
161    *       <li>{@link File}
162    *    </ul>
163    * @throws SerializeException If a problem occurred trying to convert the output.
164    */
165   public final void serialize(Object o, Object output) throws SerializeException {
166      SerializerSession session = createSession(output);
167      serialize(session, o);
168   }
169
170   /**
171    * Create the session object that will be passed in to the serialize method.
172    *
173    * <p>
174    * It's up to implementers to decide what the session object looks like, although typically it's going to be a
175    * subclass of {@link SerializerSession}.
176    *
177    * @param output
178    *    The output object.
179    *    <br>Character-based serializers can handle the following output class types:
180    *    <ul>
181    *       <li>{@link Writer}
182    *       <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
183    *       <li>{@link File} - Output will be written as system-default encoded stream.
184    *    </ul>
185    *    <br>Stream-based serializers can handle the following output class types:
186    *    <ul>
187    *       <li>{@link OutputStream}
188    *       <li>{@link File}
189    *    </ul>
190    * @param op Optional additional properties.
191    * @param javaMethod
192    *    Java method that invoked this serializer.
193    *    When using the REST API, this is the Java method invoked by the REST call.
194    *    Can be used to access annotations defined on the method or class.
195    * @param locale
196    *    The session locale.
197    *    If <jk>null</jk>, then the locale defined on the context is used.
198    * @param timeZone
199    *    The session timezone.
200    *    If <jk>null</jk>, then the timezone defined on the context is used.
201    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
202    * @param uriContext
203    *    The URI context.
204    *    Identifies the current request URI used for resolution of URIs to absolute or root-relative form.
205    * @return The new session.
206    */
207   public SerializerSession createSession(Object output, ObjectMap op, Method javaMethod, Locale locale,
208         TimeZone timeZone, MediaType mediaType, UriContext uriContext) {
209      return new SerializerSession(ctx, op, output, javaMethod, locale, timeZone, mediaType, uriContext);
210   }
211
212   /**
213    * Create a basic session object without overriding properties or specifying <code>javaMethod</code>.
214    *
215    * <p>
216    * Equivalent to calling <code>createSession(<jk>null</jk>, <jk>null</jk>)</code>.
217    *
218    * @param output
219    *    The output object.
220    *    <br>Character-based serializers can handle the following output class types:
221    *    <ul>
222    *       <li>{@link Writer}
223    *       <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
224    *       <li>{@link File} - Output will be written as system-default encoded stream.
225    *    </ul>
226    *    <br>Stream-based serializers can handle the following output class types:
227    *    <ul>
228    *       <li>{@link OutputStream}
229    *       <li>{@link File}
230    *    </ul>
231    * @return The new session.
232    */
233   protected SerializerSession createSession(Object output) {
234      return createSession(output, null, null, null, null, getPrimaryMediaType(), null);
235   }

237   /**
238    * Converts the contents of the specified object array to a list.
239    *
240    * <p>
241    * Works on both object and primitive arrays.
242    *
243    * <p>
244    * In the case of multi-dimensional arrays, the outgoing list will contain elements of type n-1 dimension.
245    * i.e. if {@code type} is <code><jk>int</jk>[][]</code> then {@code list} will have entries of type
246    * <code><jk>int</jk>[]</code>.
247    *
248    * @param type The type of array.
249    * @param array The array being converted.
250    * @return The array as a list.
251    */
252   protected static final List<Object> toList(Class<?> type, Object array) {
253      Class<?> componentType = type.getComponentType();
254      if (componentType.isPrimitive()) {
255         int l = Array.getLength(array);
256         List<Object> list = new ArrayList<Object>(l);
257         for (int i = 0; i < l; i++)
258            list.add(Array.get(array, i));
259         return list;
260      }
261      return Arrays.asList((Object[])array);
262   }
263
264   /**
265    * Returns the media types handled based on the value of the {@link Produces} annotation on the serializer class.
266    *
267    * <p>
268    * This method can be overridden by subclasses to determine the media types programmatically.
269    *
270    * @return The list of media types.  Never <jk>null</jk>.
271    */
272   public MediaType[] getMediaTypes() {
273      return mediaTypes;
274   }
275
276   /**
277    * Returns the first media type specified on this serializer via the {@link Produces} annotation.
278    *
279    * @return The media type.
280    */
281   public MediaType getPrimaryMediaType() {
282      return mediaTypes == null || mediaTypes.length == 0 ? null : mediaTypes[0];
283   }

285   /**
286    * Optional method that specifies HTTP request headers for this serializer.
287    *
288    * <p>
289    * For example, {@link SoapXmlSerializer} needs to set a <code>SOAPAction</code> header.
290    *
291    * <p>
292    * This method is typically meaningless if the serializer is being used stand-alone (i.e. outside of a REST server
293    * or client).
294    *
295    * @param properties
296    *    Optional run-time properties (the same that are passed to {@link WriterSerializer#doSerialize(SerializerSession, Object)}.
297    *    Can be <jk>null</jk>.
298    * @return
299    *    The HTTP headers to set on HTTP requests.
300    *    Can be <jk>null</jk>.
301    */
302   public ObjectMap getResponseHeaders(ObjectMap properties) {
303      return ObjectMap.EMPTY_MAP;
304   }
305
306   /**
307    * Optional method that returns the response <code>Content-Type</code> for this serializer if it is different from
308    * the matched media type.
309    *
310    * <p>
311    * This method is specified to override the content type for this serializer.
312    * For example, the {@link org.apache.juneau.json.JsonSerializer.Simple} class returns that it handles media type
313    * <js>"text/json+simple"</js>, but returns <js>"text/json"</js> as the actual content type.
314    * This allows clients to request specific 'flavors' of content using specialized <code>Accept</code> header values.
315    *
316    * <p>
317    * This method is typically meaningless if the serializer is being used stand-alone (i.e. outside of a REST server
318    * or client).
319    *
320    * @return The response content type.  If <jk>null</jk>, then the matched media type is used.
321    */
322   public MediaType getResponseContentType() {
323      return contentType;
324   }
