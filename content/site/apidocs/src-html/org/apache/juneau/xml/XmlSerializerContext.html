013package org.apache.juneau.xml;
014
015import org.apache.juneau.*;
016import org.apache.juneau.serializer.*;
017
018/**
019 * Configurable properties on the {@link XmlSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link PropertyStore#setProperty(String, Object)} on the property store
022 * passed into the constructor.
023 * <p>
024 * See {@link PropertyStore} for more information about context properties.
025 *
026 * <h5 class='section'>Inherited configurable properties:</h5>
027 * <ul class='doctree'>
028 *    <li class='jc'><a class="doclink" href="../BeanContext.html#ConfigProperties">BeanContext</a> - Properties associated with handling beans on serializers and parsers.
029 *    <ul>
030 *       <li class='jc'><a class="doclink" href="../serializer/SerializerContext.html#ConfigProperties">SerializerContext</a> - Configurable properties common to all serializers.
031 *    </ul>
032 * </ul>
033 */
034public class XmlSerializerContext extends SerializerContext {
035
036   /**
037    * <b>Configuration property:</b>  Enable support for XML namespaces.
038    * <p>
039    * <ul>
040    *    <li><b>Name:</b> <js>"XmlSerializer.enableNamespaces"</js>
041    *    <li><b>Data type:</b> <code>Boolean</code>
042    *    <li><b>Default:</b> <jk>false</jk>
043    *    <li><b>Session-overridable:</b> <jk>true</jk>
044    * </ul>
045    * <p>
046    * If not enabled, XML output will not contain any namespaces regardless of any other settings.
047    */
048   public static final String XML_enableNamespaces = "XmlSerializer.enableNamespaces";
049
050   /**
051    * <b>Configuration property:</b>  Auto-detect namespace usage.
052    * <p>
053    * <ul>
054    *    <li><b>Name:</b> <js>"XmlSerializer.autoDetectNamespaces"</js>
055    *    <li><b>Data type:</b> <code>Boolean</code>
056    *    <li><b>Default:</b> <jk>true</jk>
057    *    <li><b>Session-overridable:</b> <jk>true</jk>
058    * </ul>
059    * <p>
060    * Detect namespace usage before serialization.
061    * <p>
062    * Used in conjunction with {@link #XML_addNamespaceUrisToRoot} to reduce
063    * the list of namespace URLs appended to the root element to only those
064    * that will be used in the resulting document.
065    * <p>
066    * If enabled, then the data structure will first be crawled looking for
067    * namespaces that will be encountered before the root element is
068    * serialized.
069    * <p>
070    * This setting is ignored if {@link #XML_enableNamespaces} is not enabled.
071    * <p>
072    * <h5 class='section'>Notes:</h5>
073    * <ul>
074    *    <li>Auto-detection of namespaces can be costly performance-wise.
075    *       In high-performance environments, it's recommended that namespace detection be
076    *       disabled, and that namespaces be manually defined through the {@link #XML_namespaces} property.
077    * </ul>
078    */
079   public static final String XML_autoDetectNamespaces = "XmlSerializer.autoDetectNamespaces";
080
081   /**
082    * <b>Configuration property:</b>  Add namespace URLs to the root element.
083    * <p>
084    * <ul>
085    *    <li><b>Name:</b> <js>"XmlSerializer.addNamespaceUrisToRoot"</js>
086    *    <li><b>Data type:</b> <code>Boolean</code>
087    *    <li><b>Default:</b> <jk>false</jk>
088    *    <li><b>Session-overridable:</b> <jk>true</jk>
089    * </ul>
090    * <p>
091    * Use this setting to add {@code xmlns:x} attributes to the root
092    * element for the default and all mapped namespaces.
093    * <p>
094    * This setting is ignored if {@link #XML_enableNamespaces} is not enabled.
095    */
096   public static final String XML_addNamespaceUrisToRoot = "XmlSerializer.addNamespaceUrisToRoot";
097
098   /**
099    * <b>Configuration property:</b>  Default namespace.
100    * <p>
101    * <ul>
102    *    <li><b>Name:</b> <js>"XmlSerializer.defaultNamespace"</js>
103    *    <li><b>Data type:</b> <code>String</code>
104    *    <li><b>Default:</b> <js>"{juneau:'http://www.apache.org/2013/Juneau'}"</js>
105    *    <li><b>Session-overridable:</b> <jk>true</jk>
106    * </ul>
107    * <p>
108    * Specifies the default namespace URI for this document.
109    */
110   public static final String XML_defaultNamespace = "XmlSerializer.defaultNamespace";
111
112   /**
113    * <b>Configuration property:</b>  XMLSchema namespace.
114    * <p>
115    * <ul>
116    *    <li><b>Name:</b> <js>"XmlSerializer.xsNamespace"</js>
117    *    <li><b>Data type:</b> {@link Namespace}
118    *    <li><b>Default:</b> <code>{name:<js>'xs'</js>,uri:<js>'http://www.w3.org/2001/XMLSchema'</js>}</code>
119    *    <li><b>Session-overridable:</b> <jk>true</jk>
120    * </ul>
121    * <p>
122    * Specifies the namespace for the <code>XMLSchema</code> namespace, used by the schema generated
123    * by the {@link XmlSchemaSerializer} class.
124    */
125   public static final String XML_xsNamespace = "XmlSerializer.xsNamespace";
126
127   /**
128    * <b>Configuration property:</b>  Default namespaces.
129    * <p>
130    * <ul>
131    *    <li><b>Name:</b> <js>"XmlSerializer.namespaces"</js>
132    *    <li><b>Data type:</b> <code>Set&lt;{@link Namespace}&gt;</code>
133    *    <li><b>Default:</b> empty set
134    *    <li><b>Session-overridable:</b> <jk>true</jk>
135    * </ul>
136    * <p>
137    * The default list of namespaces associated with this serializer.
138    */
139   public static final String XML_namespaces = "XmlSerializer.namespaces.list";
140
141   /**
142    * <b>Configuration property:</b>  Add <js>"_type"</js> properties when needed.
143    * <p>
144    * <ul>
145    *    <li><b>Name:</b> <js>"XmlSerializer.addBeanTypeProperties"</js>
146    *    <li><b>Data type:</b> <code>Boolean</code>
147    *    <li><b>Default:</b> <jk>false</jk>
148    *    <li><b>Session-overridable:</b> <jk>true</jk>
149    * </ul>
150    * <p>
151    * If <jk>true</jk>, then <js>"_type"</js> properties will be added to beans if their type cannot be inferred through reflection.
152    * This is used to recreate the correct objects during parsing if the object types cannot be inferred.
153    * For example, when serializing a {@code Map<String,Object>} field, where the bean class cannot be determined from the value type.
154    * <p>
155    * When present, this value overrides the {@link SerializerContext#SERIALIZER_addBeanTypeProperties} setting and is
156    * provided to customize the behavior of specific serializers in a {@link SerializerGroup}.
157    */
158   public static final String XML_addBeanTypeProperties = "XmlSerializer.addBeanTypeProperties";
159
160
161   final boolean
162      autoDetectNamespaces,
163      enableNamespaces,
164      addNamespaceUrlsToRoot,
165      addBeanTypeProperties;
166
167   final String defaultNamespace;
168
169   final Namespace
170      xsNamespace;
171
172   final Namespace[] namespaces;
173
174   /**
175    * Constructor.
176    * <p>
177    * Typically only called from {@link PropertyStore#getContext(Class)}.
178    *
179    * @param ps The property store that created this context.
180    */
181   public XmlSerializerContext(PropertyStore ps) {
182      super(ps);
183      autoDetectNamespaces = ps.getProperty(XML_autoDetectNamespaces, boolean.class, true);
184      enableNamespaces = ps.getProperty(XML_enableNamespaces, boolean.class, false);
185      addNamespaceUrlsToRoot = ps.getProperty(XML_addNamespaceUrisToRoot, boolean.class, false);
186      defaultNamespace = ps.getProperty(XML_defaultNamespace, String.class, "{juneau:'http://www.apache.org/2013/Juneau'}");
187      xsNamespace = ps.getProperty(XML_xsNamespace, Namespace.class, new Namespace("xs", "http://www.w3.org/2001/XMLSchema"));
188      namespaces = ps.getProperty(XML_namespaces, Namespace[].class, new Namespace[0]);
189      addBeanTypeProperties = ps.getProperty(XML_addBeanTypeProperties, boolean.class, ps.getProperty(SERIALIZER_addBeanTypeProperties, boolean.class, true));
190   }
191
192   @Override /* Context */
193   public ObjectMap asMap() {
194      return super.asMap()
195         .append("XmlSerializerContext", new ObjectMap()
196            .append("autoDetectNamespaces", autoDetectNamespaces)
197            .append("enableNamespaces", enableNamespaces)
198            .append("addNamespaceUrlsToRoot", addNamespaceUrlsToRoot)
199            .append("defaultNamespace", defaultNamespace)
200            .append("xsNamespace", xsNamespace)
201            .append("namespaces", namespaces)
202            .append("addBeanTypeProperties", addBeanTypeProperties)
203         );
204   }
205}
