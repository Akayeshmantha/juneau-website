018/**
019 * Configurable properties on the {@link XmlSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * The following convenience methods are also provided for setting context properties:
025 * <ul>
026 *    <li>{@link XmlSerializer#setProperty(String,Object)}
027 *    <li>{@link XmlSerializer#setProperties(ObjectMap)}
028 *    <li>{@link XmlSerializer#addNotBeanClasses(Class[])}
029 *    <li>{@link XmlSerializer#addBeanFilters(Class[])}
030 *    <li>{@link XmlSerializer#addPojoSwaps(Class[])}
031 *    <li>{@link XmlSerializer#addToDictionary(Class[])}
032 *    <li>{@link XmlSerializer#addImplClass(Class,Class)}
033 * </ul>
034 * <p>
035 * See {@link ContextFactory} for more information about context properties.
036 *
037 *
038 * <h6 class='topic' id='ConfigProperties'>Configurable properties on the XML serializer</h6>
039 * <table class='styled' style='border-collapse: collapse;'>
040 *    <tr><th>Setting name</th><th>Description</th><th>Data type</th><th>Default value</th></tr>

046 *    </tr>
047 *    <tr>

049 *       <td>Add JSON type attributes for strings to output.</td>
050 *       <td><code>Boolean</code></td>
051 *       <td><jk>false</jk></td>
052 *    </tr>
053 *    <tr>
054 *       <td>{@link #XML_enableNamespaces}</td>

083 *    <tr>

085 *       <td>XMLSchema-Instance namespace.</td>

105 *
106 * @author James Bognar (james.bognar@salesforce.com)
107 */
108public class XmlSerializerContext extends SerializerContext {
109
110   /**
111    * <b>Configuration property:</b>  Add JSON type attributes to output.
112    * <p>
113    * <ul>
114    *    <li><b>Name:</b> <js>"XmlSerializer.addJsonTypeAttrs"</js>
115    *    <li><b>Data type:</b> <code>Boolean</code>
116    *    <li><b>Default:</b> <jk>false</jk>
117    * </ul>
118    * <p>
119    * If <js>true</jk>, {@code type} attributes will be added to elements in the XML for number/boolean/null nodes.
120    */
121   public static final String XML_addJsonTypeAttrs = "XmlSerializer.addJsonTypeAttrs";
122
123   /**
124    * <b>Configuration property:</b>  Add JSON type attributes for strings to output.
125    * <p>
126    * <ul>
127    *    <li><b>Name:</b> <js>"XmlSerializer.addJsonStringTypeAttrs"</js>
128    *    <li><b>Data type:</b> <code>Boolean</code>
129    *    <li><b>Default:</b> <jk>false</jk>
130    * </ul>
131    * <p>
132    * If <jk>true</jk>, {@code type} attributes will be added to elements in the XML for string nodes.
133    * <p>
134    * By default, these attributes are not added, and the parser will assume that the content type
135    * of the node is string by default.
136    * <p>
137    * This feature is disabled if {@link #XML_addJsonTypeAttrs} is disabled.
138    */
139   public static final String XML_addJsonStringTypeAttrs = "XmlSerializer.addJsonStringTypeAttrs";
140
141   /**
142    * <b>Configuration property:</b>  Enable support for XML namespaces.
143    * <p>
144    * <ul>
145    *    <li><b>Name:</b> <js>"XmlSerializer.enableNamespaces"</js>
146    *    <li><b>Data type:</b> <code>Boolean</code>
147    *    <li><b>Default:</b> <jk>true</jk>
148    * </ul>
149    * <p>
150    * If not enabled, XML output will not contain any namespaces regardless of any other settings.
151    */
152   public static final String XML_enableNamespaces = "XmlSerializer.enableNamespaces";
153
154   /**
155    * <b>Configuration property:</b>  Auto-detect namespace usage.
156    * <p>
157    * <ul>
158    *    <li><b>Name:</b> <js>"XmlSerializer.autoDetectNamespaces"</js>
159    *    <li><b>Data type:</b> <code>Boolean</code>
160    *    <li><b>Default:</b> <jk>true</jk>
161    * </ul>
162    * <p>
163    * Detect namespace usage before serialization.
164    * <p>
165    * Used in conjunction with {@link #XML_addNamespaceUrisToRoot} to reduce
166    * the list of namespace URLs appended to the root element to only those
167    * that will be used in the resulting document.
168    * <p>
169    * If enabled, then the data structure will first be crawled looking for
170    * namespaces that will be encountered before the root element is
171    * serialized.
172    * <p>
173    * This setting is ignored if {@link #XML_enableNamespaces} is not enabled.
174    * <p>
175    * <b>IMPORTANT NOTE:</b>
176    * Auto-detection of namespaces can be costly performance-wise.
177    * In high-performance environments, it's recommended that namespace detection be
178    *    disabled, and that namespaces be manually defined through the {@link #XML_namespaces} property.
179    */
180   public static final String XML_autoDetectNamespaces = "XmlSerializer.autoDetectNamespaces";
181
182   /**
183    * <b>Configuration property:</b>  Add namespace URLs to the root element.
184    * <p>
185    * <ul>
186    *    <li><b>Name:</b> <js>"XmlSerializer.addNamespaceUrisToRoot"</js>
187    *    <li><b>Data type:</b> <code>Boolean</code>
188    *    <li><b>Default:</b> <jk>true</jk>
189    * </ul>
190    * <p>
191    * Use this setting to add {@code xmlns:x} attributes to the root
192    * element for the default and all mapped namespaces.
193    * <p>
194    * This setting is ignored if {@link #XML_enableNamespaces} is not enabled.
195    */
196   public static final String XML_addNamespaceUrisToRoot = "XmlSerializer.addNamespaceUrisToRoot";
197
198   /**
199    * <b>Configuration property:</b>  Default namespace URI.
200    * <p>
201    * <ul>
202    *    <li><b>Name:</b> <js>"XmlSerializer.defaultNamespaceUri"</js>
203    *    <li><b>Data type:</b> <code>String</code>
204    *    <li><b>Default:</b> <jk>null</jk>
205    * </ul>
206    * <p>
207    * Specifies the default namespace URI for this document.
208    */
209   public static final String XML_defaultNamespaceUri = "XmlSerializer.defaultNamespaceUri";
210
211   /**
212    * <b>Configuration property:</b>  XMLSchema namespace.
213    * <p>
214    * <ul>
215    *    <li><b>Name:</b> <js>"XmlSerializer.xsNamespace"</js>
216    *    <li><b>Data type:</b> {@link Namespace}
217    *    <li><b>Default:</b> <code>{name:<js>'xs'</js>,uri:<js>'http://www.w3.org/2001/XMLSchema'</js>}</code>
218    * </ul>
219    * <p>
220    * Specifies the namespace for the <code>XMLSchema</code> namespace, used by the schema generated
221    * by the {@link XmlSchemaSerializer} class.
222    */
223   public static final String XML_xsNamespace = "XmlSerializer.xsNamespace";
224
225   /**
226    * <b>Configuration property:</b>  XMLSchema-Instance namespace.
227    * <p>
228    * <ul>
229    *    <li><b>Name:</b> <js>"XmlSerializer.xsiNamespace"</js>
230    *    <li><b>Data type:</b> {@link Namespace}
231    *    <li><b>Default:</b> <code>{name:<js>'xsi'</js>,uri:<js>'http://www.w3.org/2001/XMLSchema-instance'</js>}</code>
232    * </ul>
233    * <p>
234    * Specifies the namespace of the <code>XMLSchema-instance</code> namespace used for<code>nil=<jk>true</jk></code> attributes.
235    */
236   public static final String XML_xsiNamespace = "XmlSerializer.xsiNamespace";
237
238   /**
239    * <b>Configuration property:</b>  Default namespaces.
240    * <p>
241    * <ul>
242    *    <li><b>Name:</b> <js>"XmlSerializer.namespaces"</js>
243    *    <li><b>Data type:</b> <code>Set&lt;{@link Namespace}&gt;</code>
244    *    <li><b>Default:</b> empty set
245    * </ul>
246    * <p>
247    * The default list of namespaces associated with this serializer.
248    */
249   public static final String XML_namespaces = "XmlSerializer.namespaces";
250
251
252   final boolean
253      addJsonTypeAttrs,
254      addJsonStringTypeAttrs,
255      autoDetectNamespaces,
256      enableNamespaces,
257      addNamespaceUrlsToRoot;
258
259   final String defaultNamespace;
260
261   final Namespace
262      xsiNamespace,
263      xsNamespace;
264
265   final Namespace[] namespaces;
266
267   /**
268    * Constructor.
269    * <p>
270    * Typically only called from {@link ContextFactory#getContext(Class)}.
271    *
272    * @param cf The factory that created this context.
273    */
274   public XmlSerializerContext(ContextFactory cf) {
275      super(cf);
276      addJsonTypeAttrs = cf.getProperty(XML_addJsonTypeAttrs, boolean.class, false);
277      addJsonStringTypeAttrs = cf.getProperty(XML_addJsonStringTypeAttrs, boolean.class, false);
278      autoDetectNamespaces = cf.getProperty(XML_autoDetectNamespaces, boolean.class, true);
279      enableNamespaces = cf.getProperty(XML_enableNamespaces, boolean.class, true);
280      addNamespaceUrlsToRoot = cf.getProperty(XML_addNamespaceUrisToRoot, boolean.class, true);
281      defaultNamespace = cf.getProperty(XML_defaultNamespaceUri, String.class, "{juneau:'http://www.apache.org/2013/Juneau'}");
282      xsNamespace = cf.getProperty(XML_xsNamespace, Namespace.class, new Namespace("xs", "http://www.w3.org/2001/XMLSchema"));
283      xsiNamespace = cf.getProperty(XML_xsiNamespace, Namespace.class, new Namespace("xsi", "http://www.w3.org/2001/XMLSchema-instance"));
284      namespaces = cf.getProperty(XML_namespaces, Namespace[].class, new Namespace[0]);
285   }
