013package org.apache.juneau.xml.annotation;
014
015import static java.lang.annotation.ElementType.*;
016import static java.lang.annotation.RetentionPolicy.*;
017
018import java.lang.annotation.*;
019
020/**
021 * Annotation for specifying various XML options for the XML and RDF/XML serializers.
022 * <p>
023 * Can be applied to Java packages, types, fields, and methods.
024 * <p>
025 * Can be used for the following:
026 * <ul class='spaced-list'>
027 *    <li>Override the child element name on the XML representation of collection or array properties.
028 *    <li>Specify the XML namespace on a package, class, or method.
029 *    <li>Override the XML format on a POJO.
030 * </ul>
031 */
032@Documented
033@Target({TYPE,FIELD,METHOD})
034@Retention(RUNTIME)
035@Inherited
036public @interface Xml {
037
038   /**
039    * Sets the name of the XML child elements for bean properties of type collection and array.
040    * <p>
041    * Applies only to collection and array bean properties.
042    *
043    * <h5 class='section'>Example:</h5>
044    * <p class='bcode'>
045    *    <jk>public class</jk> MyBean {
046    *       <ja>@Xml</ja>(childName=<js>"child"</js>}
047    *       <jk>public</jk> String[] <jf>children</jf> = {<js>"foo"</js>,<js>"bar"</js>};
048    *    }
049    * </p>
050    * <p>
051    * Without the <ja>@Xml</ja> annotation, serializing this bean as XML would have produced the following...
052    * </p>
053    * <p class='bcode'>
054    *    <xt>&lt;object&gt;</xt>
055    *       <xt>&lt;children&gt;</xt>
056    *          <xt>&lt;string&gt;</xt>foo<xt>&lt;/string&gt;</xt>
057    *          <xt>&lt;string&gt;</xt>bar<xt>&lt;/string&gt;</xt>
058    *       <xt>&lt;/children&gt;</xt>
059    *    <xt>&lt;/object&gt;</xt>
060    * </p>
061    * <p>
062    * With the annotations, serializing this bean as XML produces the following...
063    * </p>
064    * <p class='bcode'>
065    *    <xt>&lt;object&gt;</xt>
066    *       <xt>&lt;children&gt;</xt>
067    *          <xt>&lt;child&gt;</xt>foo<xt>&lt;/child&gt;</xt>
068    *          <xt>&lt;child&gt;</xt>bar<xt>&lt;/child&gt;</xt>
069    *       <xt>&lt;/children&gt;</xt>
070    *    <xt>&lt;/object&gt;</xt>
071    * </p>
072    */
073   String childName() default "";
074
075   /**
076    * Sets the XML prefix of this property or class.
077    * <ul class='spaced-list'>
078    *    <li>When applied to a {@link ElementType#TYPE}, namespace is applied to all properties in the class, and all
079    *       subclasses of the class.
080    *    <li>When applied to bean properties on {@link ElementType#METHOD} and {@link ElementType#FIELD}, applies
081    *       to the bean property.
082    * </ul>
083    * <p>
084    * Must either be matched to a {@link #namespace()} annotation on the same object, parent object, or a {@link XmlNs} with the same name
085    *    through the {@link XmlSchema#xmlNs()} annotation on the package.
086    * </p>
087    */
088   String prefix() default "";
089
090   /**
091    * Sets the namespace URI of this property or class.
092    * <p>
093    * Must be matched with a {@link #prefix()} annotation on this object, a parent object, or a {@link XmlNs} with the same name
094    *    through the {@link XmlSchema#xmlNs()} annotation on the package.
095    */
096   String namespace() default "";
097
098   /**
099    * The {@link XmlFormat} to use for serializing this object type.
100    *
101    * <h5 class='section'>Example:</h5>
102    * <p class='bcode'>
103    *    <jk>public class</jk> MyBean {
104    *
105    *       <jc>// Normally, bean properties would be rendered as child elements of the bean element.</jc>
106    *       <jc>// Override so that it's rendered as a "f1='123'" attribute on the bean element instead.</jc>
107    *       <ja>@Xml</ja>(format=XmlFormat.<jsf>ATTR</jsf>}
108    *       <jk>public int</jk> f1 = 123;
109    *
110    *       <jc>// Normally, bean URL properties would be rendered as XML attributes on the bean element.</jc>
111    *       <jc>// Override so that it's rendered as an &lt;href&gt;http://foo&lt;/href&gt; child element instead.</jc>
112    *       <ja>@BeanProperty</ja>(uri=<jk>true</jk>)
113    *       <ja>@Xml</ja>(format=XmlFormat.<jsf>ELEMENT</jsf>}
114    *       <jk>public</jk> URL <jf>href</jf> = <jk>new</jk> URL(<js>"http://foo"</js>);
115    *
116    *       <jc>// Normally, collection properties would be grouped under a single &lt;children&gt; child element on the bean element.</jc>
117    *       <jc>// Override so that entries are directly children of the bean element with each entry having an element name of &lt;child&gt;.</jc>
118    *       <ja>@Xml</ja>(format=XmlFormat.<jsf>COLLAPSED</jsf>, childName=<js>"child"</js>}
119    *       <jk>public</jk> String[] <jf>children</jf> = <js>"foo"</js>,<js>"bar"</js>};
120    *    }
121    * </p>
122    * <p>
123    * Without the <ja>@Xml</ja> annotations, serializing this bean as XML would have produced the following...
124    * </p>
125    * <p class='bcode'>
126    *    <xt>&lt;object</xt> <xa>href</xa>=<js>'http://foo'</js><xt>&gt;</xt>
127    *       <xt>&lt;f1&gt;</xt>123<xt>&lt;/f1&gt;</xt>
128    *       <xt>&lt;children&gt;</xt>
129    *          <xt>&lt;string&gt;</xt>foo<xt>&lt;/string&gt;</xt>
130    *          <xt>&lt;string&gt;</xt>bar<xt>&lt;/string&gt;</xt>
131    *       <xt>&lt;/children&gt;</xt>
132    *    <xt>&lt;/object&gt;</xt>
133    * </p>
134    * <p>
135    * With the annotations, serializing this bean as XML produces the following...
136    * </p>
137    * <p class='bcode'>
138    *    <xt>&lt;object</xt> <xa>f1</xa>=<js>'123'</js><xt>&gt;</xt>
139    *       <xt>&lt;href&gt;</xt>http://foo<xt>&lt;/href&gt;</xt>
140    *       <xt>&lt;child&gt;</xt>foo<xt>&lt;/child&gt;</xt>
141    *       <xt>&lt;child&gt;</xt>bar<xt>&lt;/child&gt;</xt>
142    *    <xt>&lt;/object&gt;</xt>
143    * </p>
144    */
145   XmlFormat format() default XmlFormat.DEFAULT;
146}
