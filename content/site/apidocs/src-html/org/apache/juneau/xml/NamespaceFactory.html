013package org.apache.juneau.xml;
014
015import java.util.*;
016import java.util.concurrent.*;
017
018import org.apache.juneau.*;
019import org.apache.juneau.internal.*;
020import org.apache.juneau.parser.*;
021
022/**
023 * Factory class for getting unique instances of {@link Namespace} objects.
024 * <p>
025 *    For performance reasons, {@link Namespace} objects are stored in {@link IdentityList IdentityLists}.
026 *    For this to work property, namespaces with the same name and URI must only be represented by a single
027 *    {@link Namespace} instance.
028 *    This factory class ensures this identity uniqueness.
029 *
030 * @author James Bognar (james.bognar@salesforce.com)
031 */
032public final class NamespaceFactory {
033
034   private static ConcurrentHashMap<String,Namespace> cache = new ConcurrentHashMap<String,Namespace>();
035
036   /**
037    * Get the {@link Namespace} with the specified name and URI, and create a new one
038    *    if this is the first time it's been encountered.
039    *
040    * @param name The namespace name.  See {@link Namespace#getName()}.
041    * @param uri The namespace URI.  See {@link Namespace#getUri()}.
042    * @return The namespace object.
043    */
044   public static Namespace get(String name, String uri) {
045      String key = name + "+" + uri;
046      Namespace n = cache.get(key);
047      if (n == null) {
048         n = new Namespace(name, uri);
049         Namespace n2 = cache.putIfAbsent(key, n);
050         return (n2 == null ? n : n2);
051      }
052      return n;
053   }
054
055   /**
056    * Converts the specified object into a {@link Namespace} object.
057    * <p>
058    *    Can be any of following types:
059    * <ul class='spaced-list'>
060    *    <li>A {@link Namespace} object
061    *    <li>A JSON string containing a single key/value pair indicating the name/URI mapping.
062    *    <li>A <code>Map</code> containing a single key/value pair indicating the name/URI mapping.
063    * </ul>
064    *
065    * @param o The input.
066    * @return The namespace object, or <jk>null</jk> if the input was <jk>null</jk> or an empty JSON object.
067    */
068   @SuppressWarnings("rawtypes")
069   public static Namespace parseNamespace(Object o) {
070      if (o == null)
071         return null;
072      if (o instanceof Namespace)
073         return (Namespace)o;
074      try {
075         Map<?,?> m = (o instanceof Map ? (Map)o : new ObjectMap(o.toString()));
076         if (m.size() == 0)
077            return null;
078         if (m.size() > 1)
079            throw new RuntimeException("Too many namespaces specified.  Only one allowed. '"+o+"'");
080         Map.Entry<?,?> e = m.entrySet().iterator().next();
081         return get(e.getKey().toString(), e.getValue().toString());
082      } catch (ParseException e) {
083         throw new RuntimeException(e);
084      }
085   }
086
087   /**
088    * Converts the specified object into an array of {@link Namespace} object.
089    * <p>
090    *    Can be any of following types:
091    * <ul class='spaced-list'>
092    *    <li>A {@link Namespace} array
093    *    <li>A JSON string with key/value pairs indicating name/URI pairs.
094    *    <li>A <code>Map</code> with key/value pairs indicating name/URI pairs.
095    *    <li>A <code>Collection</code> containing any of object that can be passed to {@link #parseNamespace(Object)}.
096    * </ul>
097    *
098    * @param o The input.
099    * @return The namespace objects, or <jk>null</jk> if the input was <jk>null</jk> or an empty JSON object.
100    */
101   @SuppressWarnings("rawtypes")
102   public static Namespace[] parseNamespaces(Object o) {
103      try {
104         if (o instanceof Namespace[])
105            return (Namespace[])o;
106
107         if (o instanceof CharSequence)
108            o = new ObjectMap(o.toString());
109
110         Namespace[] n;
111         int i = 0;
112         if (o instanceof Collection) {
113            Collection c = (Collection)o;
114            n = new Namespace[c.size()];
115            for (Object o2 : c)
116               n[i++] = parseNamespace(o2);
117         } else if (o instanceof Map) {
118            Map<?,?> m = (Map<?,?>)o;
119            n = new Namespace[m.size()];
120            for (Map.Entry e : m.entrySet())
121               n[i++] = get(e.getKey().toString(), e.getValue().toString());
122         } else {
123            throw new RuntimeException("Invalid type passed to NamespaceFactory.listFromObject: '"+o+"'");
124         }
125         return n;
126      } catch (ParseException e) {
127         throw new RuntimeException(e);
128      }
129   }
130}
