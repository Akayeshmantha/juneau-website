013package org.apache.juneau.xml;
014
015import static org.apache.juneau.msgpack.MsgPackSerializerContext.*;
016import static org.apache.juneau.xml.NamespaceFactory.*;
017import static org.apache.juneau.xml.XmlSerializerContext.*;
018
019import java.lang.reflect.*;
020import java.util.*;
021
022import org.apache.juneau.*;
023import org.apache.juneau.internal.*;
024import org.apache.juneau.json.*;
025import org.apache.juneau.serializer.*;
026
027/**
028 * Session object that lives for the duration of a single use of {@link XmlSerializer}.
029 * <p>
030 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
031 */
032@SuppressWarnings("hiding")
033public class XmlSerializerSession extends SerializerSession {
034
035   private final boolean
036      autoDetectNamespaces,
037      enableNamespaces,
038      addNamespaceUrlsToRoot,
039      addBeanTypeProperties;
040
041   private Namespace
042      defaultNamespace;
043   private final Namespace
044      xsNamespace;
045
046   private Namespace[] namespaces = new Namespace[0];
047
048   /**
049    * Create a new session using properties specified in the context.
050    *
051    * @param ctx The context creating this session object.
052    * The context contains all the configuration settings for this object.
053    * @param output The output object.  See {@link JsonSerializerSession#getWriter()} for valid class types.
054    * @param op The override properties.
055    * These override any context properties defined in the context.
056    * @param javaMethod The java method that called this serializer, usually the method in a REST servlet.
057    * @param locale The session locale.
058    * If <jk>null</jk>, then the locale defined on the context is used.
059    * @param timeZone The session timezone.
060    * If <jk>null</jk>, then the timezone defined on the context is used.
061    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
062    */
063   public XmlSerializerSession(XmlSerializerContext ctx, ObjectMap op, Object output, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType) {
064      super(ctx, op, output, javaMethod, locale, timeZone, mediaType);
065      if (op == null || op.isEmpty()) {
066         enableNamespaces = ctx.enableNamespaces;
067         autoDetectNamespaces = ctx.autoDetectNamespaces;
068         addNamespaceUrlsToRoot = ctx.addNamespaceUrlsToRoot;
069         addNamespaces(ctx.namespaces);
070         defaultNamespace = findDefaultNamespace(ctx.defaultNamespace);
071         xsNamespace = ctx.xsNamespace;
072         addBeanTypeProperties = ctx.addBeanTypeProperties;
073      } else {
074         enableNamespaces = op.getBoolean(XML_enableNamespaces, ctx.enableNamespaces);
075         autoDetectNamespaces = op.getBoolean(XML_autoDetectNamespaces, ctx.autoDetectNamespaces);
076         addNamespaceUrlsToRoot = op.getBoolean(XML_addNamespaceUrisToRoot, ctx.addNamespaceUrlsToRoot);
077         namespaces = (op.containsKey(XML_namespaces) ? parseNamespaces(op.get(XML_namespaces)) : ctx.namespaces);
078         defaultNamespace = findDefaultNamespace(op.containsKey(XML_defaultNamespace) ? op.getString(XML_defaultNamespace) : ctx.defaultNamespace);
079         xsNamespace = (op.containsKey(XML_xsNamespace) ? parseNamespace(op.get(XML_xsNamespace)) : ctx.xsNamespace);
080         addBeanTypeProperties = op.getBoolean(MSGPACK_addBeanTypeProperties, ctx.addBeanTypeProperties);
081      }
082   }
083
084   private static Namespace findDefaultNamespace(String s) {
085      if (s == null)
086         return null;
087      if (StringUtils.startsWith(s, '{'))
088         return parseNamespace(s);
089      if (! s.startsWith("http://"))
090         return get(s, "http://unknown");
091      return get(null, s);
092   }
093
094   private void addNamespaces(Namespace...namespaces) {
095      for (Namespace ns : namespaces)
096         addNamespace(ns);
097   }
098
099   /**
100    * Add a namespace to this session.
101    *
102    * @param ns The namespace being added.
103    */
104   public void addNamespace(Namespace ns) {
105      if (ns == defaultNamespace)
106         return;
107
108      for (Namespace n : namespaces)
109         if (n == ns)
110            return;
111
112      if (defaultNamespace != null && (ns.uri.equals(defaultNamespace.uri) || ns.name.equals(defaultNamespace.name)))
113         defaultNamespace = ns;
114      else
115         namespaces = ArrayUtils.append(namespaces, ns);
116   }
117
118   /**
119    * Returns the list of namespaces being used in the current XML serialization.
120    *
121    * @return The list of namespaces being used in the current XML serialization.
122    */
123   public Namespace[] getNamespaces() {
124      return namespaces;
125   }
126
127   /**
128    * Returns the {@link XmlSerializerContext#XML_autoDetectNamespaces} setting value in this context.
129    *
130    * @return The {@link XmlSerializerContext#XML_autoDetectNamespaces} setting value in this context.
131    */
132   public final boolean isAutoDetectNamespaces() {
133      return enableNamespaces && autoDetectNamespaces;
134   }
135
136   /**
137    * Returns the {@link XmlSerializerContext#XML_enableNamespaces} setting value in this context.
138    *
139    * @return The {@link XmlSerializerContext#XML_enableNamespaces} setting value in this context.
140    */
141   public final boolean isEnableNamespaces() {
142      return enableNamespaces;
143   }
144
145   /**
146    * Returns the {@link XmlSerializerContext#XML_addNamespaceUrisToRoot} setting value in this context.
147    *
148    * @return The {@link XmlSerializerContext#XML_addNamespaceUrisToRoot} setting value in this context.
149    */
150   public final boolean isAddNamespaceUrlsToRoot() {
151      return addNamespaceUrlsToRoot;
152   }
153
154   /**
155    * Returns the {@link XmlSerializerContext#XML_addBeanTypeProperties} setting value for this session.
156    *
157    * @return The {@link XmlSerializerContext#XML_addBeanTypeProperties} setting value for this session.
158    */
159   @Override /* SerializerSession */
160   public boolean isAddBeanTypeProperties() {
161      return addBeanTypeProperties;
162   }
163
164   /**
165    * Returns the {@link XmlSerializerContext#XML_defaultNamespace} setting value in this context.
166    *
167    * @return The {@link XmlSerializerContext#XML_defaultNamespace} setting value in this context.
168    */
169   public final Namespace getDefaultNamespace() {
170      return defaultNamespace;
171   }
172
173   /**
174    * Returns the {@link XmlSerializerContext#XML_xsNamespace} setting value in this context.
175    *
176    * @return The {@link XmlSerializerContext#XML_xsNamespace} setting value in this context.
177    */
178   public final Namespace getXsNamespace() {
179      return xsNamespace;
180   }
181
182   /**
183    * Returns <jk>true</jk> if we're serializing HTML.
184    * <p>
185    * The difference in behavior is how empty non-void elements are handled.
186    * The XML serializer will produce a collapsed tag, whereas the HTML serializer
187    * will produce a start and end tag.
188    *
189    * @return <jk>true</jk> if we're generating HTML.
190    */
191   public boolean isHtmlMode() {
192      return false;
193   }
194
195   @Override /* SerializerSession */
196   public XmlWriter getWriter() throws Exception {
197      Object output = getOutput();
198      if (output instanceof XmlWriter)
199         return (XmlWriter)output;
200      return new XmlWriter(super.getWriter(), isUseWhitespace(), isTrimStrings(), getQuoteChar(), getRelativeUriBase(), getAbsolutePathUriBase(), isEnableNamespaces(), getDefaultNamespace());
201   }
202}
