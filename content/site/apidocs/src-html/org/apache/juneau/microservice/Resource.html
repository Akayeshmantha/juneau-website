013package org.apache.juneau.microservice;
014
015import static org.apache.juneau.html.HtmlDocSerializerContext.*;
016
017import org.apache.juneau.server.*;
018import org.apache.juneau.server.annotation.*;
019import org.apache.juneau.svl.*;
020import org.apache.juneau.svl.vars.*;
021
022/**
023 * Superclass for all REST resources.
024 * <p>
025 * In additional to the functionality of the {@link RestServletDefault} group,
026 * augments the {@link #createVarResolver()} method with the following additional variable types:
027 * <ul class='spaced-list'>
028 *    <li><code class='snippet'>$ARG{...}</code> - Command line arguments pulled from {@link Microservice#getArgs()}.<br>
029 *       <h6 class='figure'>Example:</h6>
030 *       <p class='bcode'>
031 *          String firstArg = request.getVarResolver().resolve(<js>"$ARG{0}"</js>);  <jc>// First argument.</jc>
032 *          String namedArg = request.getVarResolver().resolve(<js>"$ARG{myarg}"</js>);  <jc>// Named argument (e.g. "myarg=foo"). </jc>
033 *       </p>
034 *    <li><code class='snippet'>$MF{...}</code> - Manifest file entries pulled from {@link Microservice#getManifest()}.<br>
035 *       <h6 class='figure'>Example:</h6>
036 *       <p class='bcode'>
037 *          String mainClass = request.getVarResolver().resolve(<js>"$MF{Main-Class}"</js>);  <jc>// Main class. </jc>
038 *       </p>
039 * </ul>
040 *
041 * @author James Bognar (james.bognar@salesforce.com)
042 */
043@SuppressWarnings("serial")
044@RestResource(
045   properties={
046      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS'}")
047   },
048   config="$S{juneau.configFile}",
049   stylesheet="$C{REST/stylesheet,styles/juneau.css}"
050)
051public abstract class Resource extends RestServletDefault {
052
053   /**
054    * Adds $ARG and $MF variables to variable resolver defined on {@link RestServlet#createVarResolver()}.
055    */
056   @Override
057   protected VarResolver createVarResolver() {
058      return super.createVarResolver()
059         .addVars(ArgsVar.class, ManifestFileVar.class)
060         .setContextObject(ArgsVar.SESSION_args, Microservice.getArgs())
061         .setContextObject(ManifestFileVar.SESSION_manifest, Microservice.getManifest());
062   }
063}
