038/**
039 * Entry point for Juneau microservice that implements a REST interface using Jetty on a single port.
040 *
041 * <h6 class='topic'>Jetty Server Details</h6>
042 * <p>
043 * The Jetty server is created by the {@link #createServer()} method and started with the {@link #startServer()} method.
044 * These methods can be overridden to provided customized behavior.
045 * <p>
046 *
047 * <h6 class='topic'>Defining REST Resources</h6>
048 * <p>
049 * Top-level REST resources are defined by the {@link #getResourceMap()} method.
050 * This method can be overridden to provide a customized list of REST resources.
051 * <p>
052 *
053 * <h6 class='topic'>Logging</h6>
054 * <p>
055 * Logging is initialized by the {@link #initLogging()} method.
056 * This method can be overridden to provide customized logging behavior.
057 *
058 * <h6 class='topic'>Lifecycle Listener Methods</h6>
059 * Subclasses can optionally implement the following event listener methods:
060 * <ul class='spaced-list'>
061 *    <li>{@link #onStart()} - Gets executed before {@link #start()}.
062 *    <li>{@link #onStop()} - Gets executed before {@link #stop()}.
063 *    <li>{@link #onCreateServer()} - Gets executed before {@link #createServer()}.
064 *    <li>{@link #onStartServer()} - Gets executed before {@link #startServer()}.
065 *    <li>{@link #onPostStartServer()} - Gets executed after {@link #startServer()}.
066 *    <li>{@link #onStopServer()} - Gets executed before {@link #stop()}.
067 *    <li>{@link #onPostStopServer()} - Gets executed after {@link #stop()}.
068 * </ul>
069 */

118   @Override /* Microservice */
119   public RestMicroservice stop() {
120      Thread t = new Thread() {
121         @Override /* Thread */
122         public void run() {
123            try {
124               if (server.isStopping() || server.isStopped())
125                  return;
126               onStopServer();
127               logger.warning("Stopping server.");
128               server.stop();
129               logger.warning("Server stopped.");
130               onPostStopServer();
131            } catch (Exception e) {
132               logger.log(Level.SEVERE, e.getLocalizedMessage(), e);
133            }
134         }
135      };
136      t.start();
137      try {
138         t.join();
139      } catch (InterruptedException e) {
140         e.printStackTrace();
141      }
142      super.stop();
143      return this;
144   }

159   /**
160    * Returns the URI where this microservice is listening on.
161    * @return The URI where this microservice is listening on.
162    */
163   public URI getURI() {
164      String scheme = getConfig().getBoolean("REST/useSsl") ? "https" : "http";
165      String hostname = "localhost";
166      String ctx = "/".equals(contextPath) ? null : contextPath;
167      try {
168         hostname = InetAddress.getLocalHost().getHostName();
169      } catch (UnknownHostException e) {}
170      try {
171         return new URI(scheme, null, hostname, port, ctx, null, null);
172      } catch (URISyntaxException e) {
173         throw new RuntimeException(e);
174      }
175   }

177   /**
178    * Initialize the logging for this microservice.
179    * <p>
180    * Subclasses can override this method to provide customized logging.
181    * <p>
182    * The default implementation uses the <cs>Logging</cs> section in the config file to set up logging:
183    * <p class='bcode'>
184    *    <cc>#================================================================================
185    *    # Logger settings
186    *    # See FileHandler Java class for details.
187    *    #================================================================================</cc>
188    *    <cs>[Logging]</cs>
189    *
190    *    <cc># The directory where to create the log file.
191    *    # Default is ".".</cc>
192    *    <ck>logDir</ck> = logs
193    *
194    *    <cc># The name of the log file to create for the main logger.
195    *    # The logDir and logFile make up the pattern that's passed to the FileHandler
196    *    # constructor.
197    *    # If value is not specified, then logging to a file will not be set up.</cc>
198    *    <ck>logFile</ck> = microservice.%g.log
199    *
200    *    <cc># Whether to append to the existing log file or create a new one.
201    *    # Default is false.</cc>
202    *    <ck>append</ck> =
203    *
204    *    <cc># The SimpleDateFormat format to use for dates.
205    *    # Default is "yyyy.MM.dd hh:mm:ss".</cc>
206    *    <ck>dateFormat</ck> =
207    *
208    *    <cc># The log message format.
209    *    # The value can contain any of the following variables:
210    *    #  {date} - The date, formatted per dateFormat.
211    *    #  {class} - The class name.
212    *    #  {method} - The method name.
213    *    #  {logger} - The logger name.
214    *    #  {level} - The log level name.
215    *    #  {msg} - The log message.
216    *    #  {threadid} - The thread ID.
217    *    #  {exception} - The localized exception message.
218    *    # Default is "[{date} {level}] {msg}%n".</cc>
219    *    <ck>format</ck> =
220    *
221    *    <cc># The maximum log file size.
222    *    # Suffixes available for numbers.
223    *    # See ConfigFile.getInt(String,int) for details.
224    *    # Default is 1M.</cc>
225    *    <ck>limit</ck> = 10M
226    *
227    *    <cc># Max number of log files.
228    *    # Default is 1.</cc>
229    *    <ck>count</ck> = 5
230    *
231    *    <cc># Default log levels.
232    *    # Keys are logger names.
233    *    # Values are serialized Level POJOs.</cc>
234    *    <ck>levels</ck> = { org.apache.juneau:'INFO' }
235    *
236    *    <cc># Only print unique stack traces once and then refer to them by a simple 8 character hash identifier.
237    *    # Useful for preventing log files from filling up with duplicate stack traces.
238    *    # Default is false.</cc>
239    *    <ck>useStackTraceHashes</ck> = true
240    *
241    *    <cc># The default level for the console logger.
242    *    # Default is WARNING.</cc>
243    *    <ck>consoleLevel</ck> = WARNING
244    * </p>
245    *
246    * @throws Exception
247    */
248   protected void initLogging() throws Exception {
249      ConfigFile cf = getConfig();
250      logger = Logger.getLogger("");
251      String logFile = cf.getString("Logging/logFile");
252      if (! StringUtils.isEmpty(logFile)) {
253         LogManager.getLogManager().reset();
254         String logDir = cf.getString("Logging/logDir", ".");
255         FileUtils.mkdirs(new File(logDir), false);
256         boolean append = cf.getBoolean("Logging/append");
257         int limit = cf.getInt("Logging/limit", 1024*1024);
258         int count = cf.getInt("Logging/count", 1);
259         FileHandler fh = new FileHandler(logDir + '/' + logFile, limit, count, append);
260
261         boolean useStackTraceHashes = cf.getBoolean("Logging/useStackTraceHashes");
262         String format = cf.getString("Logging/format", "[{date} {level}] {msg}%n");
263         String dateFormat = cf.getString("Logging/dateFormat", "yyyy.MM.dd hh:mm:ss");
264         fh.setFormatter(new LogEntryFormatter(format, dateFormat, useStackTraceHashes));
265         logger.addHandler(fh);
266
267         ConsoleHandler ch = new ConsoleHandler();
268         ch.setLevel(Level.parse(cf.getString("Logging/consoleLevel", "WARNING")));
269         ch.setFormatter(new LogEntryFormatter(format, dateFormat, false));
270         logger.addHandler(ch);
271      }
272      ObjectMap loggerLevels = cf.getObject("Logging/levels", ObjectMap.class);
273      if (loggerLevels != null)
274      for (String l : loggerLevels.keySet())
275         Logger.getLogger(l).setLevel(loggerLevels.get(Level.class, l));
276   }

278   /**
279    * Method used to create (but not start) an instance of a Jetty server.
280    * <p>
281    * Subclasses can override this method to customize the Jetty server before it is started.
282    * <p>
283    * The default implementation is configured by the following values in the config file:
284    * <p>
285    * <p class='bcode'>
286    *    <cc>#================================================================================
287    *    # REST settings
288    *    #================================================================================</cc>
289    *    <cs>[REST]</cs>
290    *
291    *    <cc># The HTTP port number to use.
292    *    # Default is Rest-Port setting in manifest file, or 8000.
293    *    # Can also specify a comma-delimited lists of ports to try, including 0 meaning
294    *    # try a random port.</cc>
295    *    <ck>port</ck> = 10000
296    *
297    *    <cc># The context root of the Jetty server.
298    *    # Default is Rest-ContextPath in manifest file, or "/".</cc>
299    *    <ck>contextPath</ck> =
300    *
301    *    <cc># Authentication:  NONE, BASIC.
302    *    # Default is Rest-AuthType in manifest file, or NONE.</cc>
303    *    <ck>authType</ck> = NONE
304    *
305    *    <cc># The BASIC auth username.
306    *    # Default is Rest-LoginUser in manifest file.</cc>
307    *    <ck>loginUser</ck> =
308    *
309    *    <cc># The BASIC auth password.
310    *    # Default is Rest-LoginPassword in manifest file.</cc>
311    *    <ck>loginPassword</ck> =
312    *
313    *    <cc># The BASIC auth realm.
314    *    # Default is Rest-AuthRealm in manifest file.</cc>
315    *    <ck>authRealm</ck> =
316    *
317    *    <cc># Enable SSL support.</cc>
318    *    <ck>useSsl</ck> = false
319    *
320    *    <cc>#================================================================================
321    *    # Bean properties on the org.eclipse.jetty.util.ssl.SslSocketFactory class
322    *    #--------------------------------------------------------------------------------
323    *    # Ignored if REST/useSsl is false.
324    *    #================================================================================</cc>
325    *    <cs>[REST-SslContextFactory]</cs>
326    *    <ck>keyStorePath</ck> = client_keystore.jks
327    *    <ck>keyStorePassword*</ck> = {HRAaRQoT}
328    *    <ck>excludeCipherSuites</ck> = TLS_DHE.*, TLS_EDH.*
329    *    <ck>excludeProtocols</ck> = SSLv3
330    *    <ck>allowRenegotiate</ck> = false
331    * </p>
332    *
333    * @return The newly-created server.
334    * @throws Exception
335    */
336   protected Server createServer() throws Exception {
337      onCreateServer();
338
339      ConfigFile cf = getConfig();
340      ObjectMap mf = getManifest();
341      
342      int[] ports = cf.getObjectWithDefault("REST/port", mf.get(int[].class, "Rest-Port", new int[]{8000}), int[].class);
343
344      port = findOpenPort(ports);
345      if (port == 0) {
346         System.err.println("Open port not found.  Tried " + JsonSerializer.DEFAULT_LAX.toString(ports));
347         System.exit(1);
348      }
349         
350      contextPath = cf.getString("REST/contextPath", mf.getString("Rest-ContextPath", "/"));
351
352      if (cf.getBoolean("REST/useSsl")) {
353
354         SslContextFactory sslContextFactory = new SslContextFactory();
355
356         // Write the properties in REST-SslContextFactory to the bean setters on sslContextFactory.
357         // Throws an exception if section contains unknown properties.
358         // Only look for bean properties of type String/String/boolean/int since class has multiple
359         //    setters with the same name (e.g. setKeyStore(KeyStore) and setKeyStore(String)).
360         ObjectMap m = cf.writeProperties("REST-SslContextFactory", sslContextFactory, false, String.class, String[].class, boolean.class, int.class);
361
362         // We're using Jetty 8 that doesn't allow regular expression matching in SslContextFactory.setExcludeCipherSuites(),
363         // so to prevent having the config file list all old cipher suites, exclude the known bad ones.
364         String[] excludeCipherSuites = ArrayUtils.combine(
365            StringUtils.split("SSL_RSA_WITH_DES_CBC_SHA,SSL_DHE_RSA_WITH_DES_CBC_SHA,SSL_DHE_DSS_WITH_DES_CBC_SHA,SSL_RSA_EXPORT_WITH_RC4_40_MD5,SSL_RSA_EXPORT_WITH_DES40_CBC_SHA,SSL_DHE_RSA_EXPORT_WITH_DES40_CBC_SHA,SSL_DHE_RSA_WITH_3DES_EDE_CBC_SHA,SSL_DHE_DSS_WITH_3DES_EDE_CBC_SHA,TLS_DHE_RSA_WITH_AES_256_CBC_SHA256,TLS_DHE_DSS_WITH_AES_256_CBC_SHA256,TLS_DHE_RSA_WITH_AES_256_CBC_SHA,TLS_DHE_DSS_WITH_AES_256_CBC_SHA,TLS_DHE_RSA_WITH_AES_128_CBC_SHA256,TLS_DHE_DSS_WITH_AES_128_CBC_SHA256,TLS_DHE_RSA_WITH_AES_128_CBC_SHA,TLS_DHE_DSS_WITH_AES_128_CBC_SHA", ','),
366            sslContextFactory.getExcludeCipherSuites()
367         );
368         sslContextFactory.setExcludeCipherSuites(excludeCipherSuites);
369
370         logger.log(Level.WARNING, "SSL properties set: {0}", JsonSerializer.DEFAULT_LAX.toString(m));
371
372         SslSocketConnector connector = new SslSocketConnector(sslContextFactory);
373         connector.setPort(port);
374
375         server = new Server();
376         server.setConnectors(new Connector[] { connector });
377
378      } else {
379         server = new Server(port);
380      }
381
382      ServletContextHandler context = new ServletContextHandler(ServletContextHandler.SESSIONS);
383
384      String authType = cf.getString("REST/authType", mf.getString("Rest-AuthType", "NONE"));
385      if (authType.equals("BASIC"))
386         context.setSecurityHandler(basicAuth(cf, mf));
387
388      context.setContextPath(contextPath);
389      server.setHandler(context);
390
391      for (Map.Entry<String,Class<? extends Servlet>> e : getResourceMap().entrySet())
392         context.addServlet(e.getValue(), e.getKey()).setInitOrder(0);
393
394      return server;
395   }

397   private static int findOpenPort(int[] ports) {
398      for (int port : ports) {
399         try {
400            // If port is 0, try a random port between ports[0] and 32767.
401            if (port == 0)
402               port = new Random().nextInt(32767 - ports[0] + 1) + ports[0];
403            ServerSocket ss = new ServerSocket(port);
404            ss.close();
405            return port;
406         } catch (IOException e) {}
407      }
408      return 0;
409   }

427   /**
428    * Returns the resource map to use for this microservice.
429    * <p>
430    * <p>
431    * Subclasses can override this method to programmatically specify their resources.
432    * <p>
433    * The default implementation is configured by the following values in the config file:
434    * <p>
435    * <p class='bcode'>
436    *
437    *    <cc>#================================================================================
438    *    # REST settings
439    *    #================================================================================</cc>
440    *    <cs>[REST]</cs>
441    *
442    *    <cc># A JSON map of servlet paths to servlet classes.
443    *    # Example:
444    *    #  resourceMap = {'/*':'com.foo.MyServlet'}
445    *    # Either resourceMap or resources must be specified if it's not defined in
446    *    # the manifest file.</cc>
447    *    <ck>resourceMap</ck> =
448    *
449    *    <cc># A comma-delimited list of names of classes that extend from Servlet.
450    *    # Resource paths are pulled from @RestResource.path() annotation, or
451    *    #  "/*" if annotation not specified.
452    *    # Example:
453    *    #  resources = com.foo.MyServlet
454    *     *    # Default is Rest-Resources in manifest file.
455    *    # Either resourceMap or resources must be specified if it's not defined in
456    *    # the manifest file.</cc>
457    *    <ck>resources</ck> =
458    * </p>
459    * <p>
460    * In most cases, the rest resources will be specified in the manifest file since
461    *    it's not likely to be a configurable property:
462    * <p>
463    * <p class='bcode'>
464    *    <mk>Rest-Resources:</mk> org.apache.juneau.microservice.sample.RootResources
465    * </p>
466    *
467    * @return The map of REST resources.
468    * @throws ClassNotFoundException
469    * @throws ParseException
470    */
471   @SuppressWarnings("unchecked")
472   protected Map<String,Class<? extends Servlet>> getResourceMap() throws ClassNotFoundException, ParseException {
473      ConfigFile cf = getConfig();
474      ObjectMap mf = getManifest();
475      Map<String,Class<? extends Servlet>> rm = new LinkedHashMap<String,Class<? extends Servlet>>();
476
477      ObjectMap resourceMap = cf.getObject("REST/resourceMap", ObjectMap.class);
478      String[] resources = cf.getStringArray("REST/resources", mf.getStringArray("Rest-Resources"));
479
480      if (resourceMap != null && ! resourceMap.isEmpty()) {
481         for (Map.Entry<String,Object> e : resourceMap.entrySet()) {
482            Class<?> c = Class.forName(e.getValue().toString());
483            if (! ClassUtils.isParentClass(Servlet.class, c))
484               throw new ClassNotFoundException("Invalid class specified as resource.  Must be a Servlet.  Class='"+c.getName()+"'");
485            rm.put(e.getKey(), (Class<? extends Servlet>)c);
486         }
487      } else if (resources.length > 0) {
488         for (String resource : resources) {
489            Class<?> c = Class.forName(resource);
490            if (! ClassUtils.isParentClass(Servlet.class, c))
491               throw new ClassNotFoundException("Invalid class specified as resource.  Must be a Servlet.  Class='"+c.getName()+"'");
492            RestResource rr = c.getAnnotation(RestResource.class);
493            String path = rr == null ? "/*" : rr.path();
494            if (! path.endsWith("*"))
495               path += (path.endsWith("/") ? "*" : "/*");
496            rm.put(path, (Class<? extends Servlet>)c);
497         }
498      }
499      return rm;
500   }

502   /**
503    * Called when {@link ConfigFile#save()} is called on the config file.
504    * <p>
505    * The default behavior is configured by the following value in the config file:
506    * <p>
507    * <p class='bcode'>
508    *    <cs>[REST]</cs>
509    *
510    *    <cc># What to do when the config file is saved.
511    *    # Possible values:
512    *    #  NOTHING - Don't do anything. (default)
513    *    #  RESTART_SERVER - Restart the Jetty server.
514    *    #  RESTART_SERVICE - Shutdown and exit with code '3'.</cc>
515    *    <ck>saveConfigAction</ck> = RESTART_SERVER
516    * </p>
517    */
518   @Override /* Microservice */
519   protected void onConfigSave(ConfigFile cf) {
520      try {
521         String saveConfigAction = cf.getString("REST/saveConfigAction", "NOTHING");
522         if (saveConfigAction.equals("RESTART_SERVER")) {
523            new Thread() {
524               @Override /* Thread */
525               public void run() {
526                  try {
527                     RestMicroservice.this.stop();
528                     RestMicroservice.this.start();
529                  } catch (Exception e) {
530                     logger.log(Level.SEVERE, e.getLocalizedMessage(), e);
531                  }
532               }
533            }.start();
534         } else if (saveConfigAction.equals("RESTART_SERVICE")) {
535            stop();
536            System.exit(3);
537         }
538      } catch (Exception e) {
539         throw new RuntimeException(e);
540      }
541   }

628   private static final SecurityHandler basicAuth(ConfigFile cf, ObjectMap mf) {
629
630      HashLoginService l = new HashLoginService();
631      String user = cf.getString("REST/loginUser", mf.getString("Rest-LoginUser"));
632      String pw = cf.getString("REST/loginPassword", mf.getString("Rest-LoginPassword"));
633      String realm = cf.getString("REST/authRealm", mf.getString("Rest-AuthRealm", ""));
634      String ctx = cf.getString("REST/contextPath", mf.getString("Rest-ContextPath", "/"));
635
636      l.putUser(user, Credential.getCredential(pw), new String[] { "user" });
637      l.setName(realm);
638
639      Constraint constraint = new Constraint();
640      constraint.setName(Constraint.__BASIC_AUTH);
641      constraint.setRoles(new String[] { "user" });
642      constraint.setAuthenticate(true);
643
644      ConstraintMapping cm = new ConstraintMapping();
645      cm.setConstraint(constraint);
646      cm.setPathSpec(ctx);
647
648      ConstraintSecurityHandler csh = new ConstraintSecurityHandler();
649      csh.setAuthenticator(new BasicAuthenticator());
650      csh.setRealmName("myrealm");
651      csh.addConstraintMapping(cm);
652      csh.setLoginService(l);
653
654      return csh;
655   }
