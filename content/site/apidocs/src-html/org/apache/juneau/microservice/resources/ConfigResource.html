013package org.apache.juneau.microservice.resources;
014
015import static javax.servlet.http.HttpServletResponse.*;
016import static org.apache.juneau.html.HtmlDocSerializerContext.*;
017
018import java.io.*;
019import java.util.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.ini.*;
023import org.apache.juneau.microservice.*;
024import org.apache.juneau.server.*;
025import org.apache.juneau.server.annotation.*;
026
027/**
028 * Shows contents of the microservice configuration file.
029 */
030@RestResource(
031   path="/config",
032   title="Configuration",
033   description="Contents of configuration file.",
034   properties={
035      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS',edit:'$R{servletURI}/edit'}"),
036   }
037)
038public class ConfigResource extends Resource {
039   private static final long serialVersionUID = 1L;
040
041   /**
042    * [GET /] - Show contents of config file.
043    *
044    * @return The config file.
045    * @throws Exception
046    */
047   @RestMethod(name="GET", path="/", description="Show contents of config file.")
048   public ConfigFile getConfigContents() throws Exception {
049      return getConfig();
050   }
051
052   /**
053    * [GET /edit] - Show config file edit page.
054    *
055    * @param req The HTTP request.
056    * @return The config file as a reader resource.
057    * @throws Exception
058    */
059   @RestMethod(name="GET", path="/edit", description="Show config file edit page.")
060   public ReaderResource getConfigEditPage(RestRequest req) throws Exception {
061      // Note that we don't want variables in the config file to be resolved,
062      // so we need to escape any $ characters we see.
063      req.setAttribute("contents", getConfig().toString().replaceAll("\\$", "\\\\\\$"));
064      return req.getReaderResource("ConfigEdit.html", true);
065   }
066
067   /**
068    * [GET /{section}] - Show config file section.
069    *
070    * @param section The section name.
071    * @return The config file section.
072    * @throws Exception
073    */
074   @RestMethod(name="GET", path="/{section}",
075      description="Show config file section.",
076      parameters={
077         @Parameter(in="path", name="section", description="Section name.")
078      }
079   )
080   public ObjectMap getConfigSection(@Path("section") String section) throws Exception {
081      return getSection(section);
082   }
083
084   /**
085    * [GET /{section}/{key}] - Show config file entry.
086    *
087    * @param section The section name.
088    * @param key The section key.
089    * @return The value of the config file entry.
090    * @throws Exception
091    */
092   @RestMethod(name="GET", path="/{section}/{key}",
093      description="Show config file entry.",
094      parameters={
095         @Parameter(in="path", name="section", description="Section name."),
096         @Parameter(in="path", name="key", description="Entry name.")
097      }
098   )
099   public String getConfigEntry(@Path("section") String section, @Path("key") String key) throws Exception {
100      return getSection(section).getString(key);
101   }
102
103   /**
104    * [POST /] - Sets contents of config file from a FORM post.
105    *
106    * @param contents The new contents of the config file.
107    * @return The new config file contents.
108    * @throws Exception
109    */
110   @RestMethod(name="POST", path="/",
111      description="Sets contents of config file from a FORM post.",
112      parameters={
113         @Parameter(in="formData", name="contents", description="New contents in INI file format.")
114      }
115   )
116   public ConfigFile setConfigContentsFormPost(@FormData("contents") String contents) throws Exception {
117      return setConfigContents(new StringReader(contents));
118   }
119
120   /**
121    * [PUT /] - Sets contents of config file.
122    *
123    * @param contents The new contents of the config file.
124    * @return The new config file contents.
125    * @throws Exception
126    */
127   @RestMethod(name="PUT", path="/",
128      description="Sets contents of config file.",
129      parameters={
130         @Parameter(in="body", description="New contents in INI file format.")
131      }
132   )
133   public ConfigFile setConfigContents(@Body Reader contents) throws Exception {
134      ConfigFile cf2 = ConfigMgr.DEFAULT.create().load(contents);
135      return getConfig().merge(cf2).save();
136   }
137
138   /**
139    * [PUT /{section}] - Add or overwrite a config file section.
140    *
141    * @param section The section name.
142    * @param contents The new contents of the config file section.
143    * @return The new section.
144    * @throws Exception
145    */
146   @RestMethod(name="PUT", path="/{section}",
147      description="Add or overwrite a config file section.",
148      parameters={
149         @Parameter(in="path", name="section", description="Section name."),
150         @Parameter(in="body", description="New contents for section as a simple map with string keys and values.")
151      }
152   )
153   public ObjectMap setConfigSection(@Path("section") String section, @Body Map<String,String> contents) throws Exception {
154      getConfig().setSection(section, contents);
155      return getSection(section);
156   }
157
158   /**
159    * [PUT /{section}/{key}] - Add or overwrite a config file entry.
160    *
161    * @param section The section name.
162    * @param key The section key.
163    * @param value The new value.
164    * @return The new value.
165    * @throws Exception
166    */
167   @RestMethod(name="PUT", path="/{section}/{key}",
168      description="Add or overwrite a config file entry.",
169      parameters={
170         @Parameter(in="path", name="section", description="Section name."),
171         @Parameter(in="path", name="key", description="Entry name."),
172         @Parameter(in="body", description="New value as a string.")
173      }
174   )
175   public String setConfigSection(@Path("section") String section, @Path("key") String key, @Body String value) throws Exception {
176      getConfig().put(section, key, value, false);
177      return getSection(section).getString(key);
178   }
179
180   private ObjectMap getSection(String name) {
181      ObjectMap m = getConfig().getSectionMap(name);
182      if (m == null)
183         throw new RestException(SC_NOT_FOUND, "Section not found.");
184      return m;
185   }
186}
