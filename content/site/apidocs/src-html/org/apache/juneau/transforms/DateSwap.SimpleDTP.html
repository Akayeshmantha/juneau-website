025/**
026 * Transforms {@link Date Dates} to {@link String Strings}.
027 *
028 *
029 * <h6 class='topic'>Behavior-specific subclasses</h6>
030 * <p>
031 *    The following direct subclasses are provided for convenience:
032 * <ul class='spaced-list'>
033 *    <li>{@link ToString} - Transforms to {@link String Strings} using the {@code Date.toString()} method.
034 *    <li>{@link ISO8601DT} - Transforms to ISO8601 date-time strings.
035 *    <li>{@link ISO8601DTP} - Transforms to ISO8601 date-time strings with millisecond precision.
036 *    <li>{@link ISO8601DTZ} - Same as {@link ISO8601DT}, except always serializes in GMT.
037 *    <li>{@link ISO8601DTZP} - Same as {@link ISO8601DTZ}, except with millisecond precision.
038 *    <li>{@link RFC2822DT} - Transforms to RFC2822 date-time strings.
039 *    <li>{@link RFC2822DTZ} - Same as {@link RFC2822DT}, except always serializes in GMT.
040 *    <li>{@link RFC2822D} - Transforms to RFC2822 date strings.
041 *    <li>{@link SimpleDT} - Transforms to simple <js>"yyyy/MM/dd HH:mm:ss"</js> strings.
042 *    <li>{@link SimpleT} - Transforms to simple <js>"yyyy/MM/dd HH:mm:ss"</js> strings.
043 *    <li>{@link Medium} - Transforms to {@link DateFormat#MEDIUM} strings.
044 * </ul>
045 *
046 *
047 * @author James Bognar (james.bognar@salesforce.com)
048 */
049public class DateSwap extends PojoSwap<Date,String> {

066   /**
067    * Transforms {@link Date Dates} to ISO8601 date-time strings.
068    *
069    * <h6 class='topic'>Example output:</h6>
070    * <ul>
071    *    <li><js>"2001-07-04T15:30:45-05:00"</js>
072    *    <li><js>"2001-07-04T15:30:45Z"</js>
073    * </ul>
074    *
075    * <h6 class='topic'>Example input:</h6>
076    * <ul>
077    *    <li><js>"2001-07-04T15:30:45-05:00"</js>
078    *    <li><js>"2001-07-04T15:30:45Z"</js>
079    *    <li><js>"2001-07-04T15:30:45.1Z"</js>
080    *    <li><js>"2001-07-04T15:30Z"</js>
081    *    <li><js>"2001-07-04"</js>
082    *    <li><js>"2001-07"</js>
083    *    <li><js>"2001"</js>
084    * </ul>
085    */
086   public static class ISO8601DT extends DateSwap {
087      private SimpleDateFormat tzFormat = new SimpleDateFormat("Z");
088
089      /** Constructor */
090      public ISO8601DT() {
091         this("yyyy-MM-dd'T'HH:mm:ss");
092      }
093
094      /**
095       * Constructor with specific pattern.
096       *
097       * @param pattern The {@link MessageFormat}-style format string.
098       */
099      protected ISO8601DT(String pattern) {
100         super(pattern);
101      }
102
103      @Override /* PojoSwap */
104      public Date unswap(String o, ClassMeta<?> hint, BeanContext bc) throws ParseException {
105         try {
106            if (StringUtils.isEmpty(o))
107               return null;
108            return convert(DatatypeConverter.parseDateTime(o).getTime(), hint);
109         } catch (ParseException e) {
110            throw e;
111         } catch (Exception e) {
112            throw new ParseException(e);
113         }
114      }
115
116      @Override /* PojoSwap */
117      public String swap(Date o) {
118         String s = super.swap(o);
119         String tz = tzFormat.format(o);
120         if (tz.equals("+0000"))
121            return s + "Z";
122         return s + tz.substring(0,3) + ':' + tz.substring(3);
123      }
124   }

287   /** The formatter to convert dates to Strings. */
288   private DateFormat format;
289
290   /**
291    * Construct a transform using the specified date format string that will be
292    *    used to construct a {@link SimpleDateFormat} that will be used to convert
293    *    dates to strings.
294    *
295    * @param simpleDateFormat The {@link SimpleDateFormat} pattern.
296    */
297   public DateSwap(String simpleDateFormat) {
298      this(new SimpleDateFormat(simpleDateFormat));
299   }
300
301   /**
302    * Construct a transform using the specified date format string that will be
303    *    used to construct a {@link SimpleDateFormat} that will be used to convert
304    *    dates to strings.
305    *
306    * @param simpleDateFormat The {@link SimpleDateFormat} pattern.
307    * @param timeZone The time zone to associate with the date pattern.
308    */
309   public DateSwap(String simpleDateFormat, String timeZone) {
310      this(new SimpleDateFormat(simpleDateFormat));
311      format.setTimeZone(TimeZone.getTimeZone(timeZone));
312   }
313
314   /**
315    * Construct a transform using the specified {@link DateFormat} that will be used to convert
316    *    dates to strings.
317    *
318    * @param format The format to use to convert dates to strings.
319    */
320   public DateSwap(DateFormat format) {
321      super();
322      this.format = format;
323   }
324
325   /**
326    * Converts the specified {@link Date} to a {@link String}.
327    */
328   @Override /* PojoSwap */
329   public String swap(Date o) {
330      return format.format(o);
331   }
332
333   /**
334    * Converts the specified {@link String} to a {@link Date}.
335    */
336   @Override /* PojoSwap */
337   public Date unswap(String o, ClassMeta<?> hint, BeanContext bc) throws ParseException {
338      try {
339         if (StringUtils.isEmpty(o))
340            return null;
341         Date d = format.parse(o);
342         return convert(d, hint);
343      } catch (ParseException e) {
344         throw e;
345      } catch (Exception e) {
346         throw new ParseException(e);
347      }
348   }
349
350   private static Date convert(Date in, ClassMeta<?> hint) throws Exception {
351      if (in == null)
352         return null;
353      if (hint == null || hint.isInstance(in))
354         return in;
355      Class<?> c = hint.getInnerClass();
356      if (c == java.util.Date.class)
357         return in;
358      if (c == java.sql.Date.class)
359         return new java.sql.Date(in.getTime());
360      if (c == java.sql.Time.class)
361         return new java.sql.Time(in.getTime());
362      if (c == java.sql.Timestamp.class)
363         return new java.sql.Timestamp(in.getTime());
364      throw new ParseException("DateSwap is unable to narrow object of type ''{0}''", c);
365   }
