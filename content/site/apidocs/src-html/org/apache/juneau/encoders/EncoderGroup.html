013package org.apache.juneau.encoders;
014
015import static org.apache.juneau.internal.ArrayUtils.*;
016
017import java.util.*;
018import java.util.concurrent.*;
019
020import org.apache.juneau.*;
021import org.apache.juneau.internal.*;
022import org.apache.juneau.serializer.*;
023
024/**
025 * Represents the group of {@link Encoder encoders} keyed by codings.
026 *
027 * <h6 class='topic'>Description</h6>
028 * <p>
029 *    Maintains a set of encoders and the codings that they can handle.
030 * <p>
031 *    The {@link #getEncoderMatch(String)} and {@link #getEncoder(String)} methods are then
032 *       used to find appropriate encoders for specific <code>Accept-Encoding</code>
033 *       and <code>Content-Encoding</code> header values.
034 *
035 * <h6 class='topic'>Match ordering</h6>
036 * <p>
037 *    Encoders are matched against <code>Accept-Encoding</code> strings in the order they exist in this group.
038 * <p>
039 *    Adding new entries will cause the entries to be prepended to the group.
040 *    This allows for previous encoders to be overridden through subsequent calls.
041 * <p>
042 *    For example, calling <code>g.append(E1.<jk>class</jk>,E2.<jk>class</jk>).append(E3.<jk>class</jk>,E4.<jk>class</jk>)</code>
043 *    will result in the order <code>E3, E4, E1, E2</code>.
044 *
045 * <h6 class='topic'>Example:</h6>
046 * <p class='bcode'>
047 *    <jc>// Create an encoder group with support for gzip compression.</jc>
048 *    EncoderGroup g = <jk>new</jk> EncoderGroup().append(GzipEncoder.<jk>class</jk>);
049 *
050 *    <jc>// Should return "gzip"</jc>
051 *    String matchedCoding = g.findMatch(<js>"compress;q=1.0, gzip;q=0.8, identity;q=0.5, *;q=0"</js>);
052 *
053 *    <jc>// Get the encoder</jc>
054 *    IEncoder encoder = g.getEncoder(matchedCoding);
055 * </p>
056 */
057public final class EncoderGroup extends Lockable {
058
059   // Maps Accept-Encoding headers to matching encoders.
060   private final Map<String,EncoderMatch> cache = new ConcurrentHashMap<String,EncoderMatch>();
061
062   private final CopyOnWriteArrayList<Encoder> encoders = new CopyOnWriteArrayList<Encoder>();
063
064   /**
065    * Adds the specified encoder to the beginning of this group.
066    *
067    * @param e - The encoder to add to this group.
068    * @return This object (for method chaining).
069    */
070   public EncoderGroup append(Encoder e) {
071      checkLock();
072      synchronized(encoders) {
073         cache.clear();
074         encoders.add(0, e);
075      }
076      return this;
077   }
078
079   /**
080    * Registers the specified encoders with this group.
081    *
082    * @param e The encoders to append to this group.
083    * @return This object (for method chaining).
084    * @throws Exception Thrown if {@link Encoder} could not be constructed.
085    */
086   public EncoderGroup append(Class<? extends Encoder>...e) throws Exception {
087      for (Class<? extends Encoder> ee : ArrayUtils.reverse(e))
088         append(ee);
089      return this;
090   }
091
092   /**
093    * Same as {@link #append(Class[])}, except specify a single class to avoid unchecked compile warnings.
094    *
095    * @param e The encoder to append to this group.
096    * @return This object (for method chaining).
097    * @throws Exception Thrown if {@link Serializer} could not be constructed.
098    */
099   public EncoderGroup append(Class<? extends Encoder> e) throws Exception {
100      try {
101         append(e.newInstance());
102      } catch (NoClassDefFoundError x) {
103         // Ignore if dependent library not found (e.g. Jena).
104         System.err.println(e);
105      }
106      return this;
107   }
108
109   /**
110    * Adds the encoders in the specified group to this group.
111    *
112    * @param g The group containing the encoders to add to this group.
113    * @return This object (for method chaining).
114    */
115   public EncoderGroup append(EncoderGroup g) {
116      for (Encoder e : reverse(g.encoders.toArray(new Encoder[g.encoders.size()])))
117         append(e);
118      return this;
119   }
120
121   /**
122    * Returns the coding string for the matching encoder that can handle the specified <code>Accept-Encoding</code>
123    *    or <code>Content-Encoding</code> header value.
124    * <p>
125    *    Returns <jk>null</jk> if no encoders can handle it.
126    * <p>
127    *    This method is fully compliant with the RFC2616/14.3 and 14.11 specifications.
128    *
129    * @param acceptEncoding The <code>Accept-Encoding</code> or <code>Content-Encoding</code> value.
130    * @return The coding value (e.g. <js>"gzip"</js>).
131    */
132   public EncoderMatch getEncoderMatch(String acceptEncoding) {
133      if (encoders.size() == 0)
134         return null;
135
136      EncoderMatch em = cache.get(acceptEncoding);
137      if (em != null)
138         return em;
139
140      MediaRange[] ae = MediaRange.parse(acceptEncoding);
141
142      if (ae.length == 0)
143         ae = MediaRange.parse("*/*");
144
145      Map<Float,EncoderMatch> m = null;
146
147      for (MediaRange a : ae) {
148         for (Encoder e : encoders) {
149            for (String c : e.getCodings()) {
150               MediaType mt = MediaType.forString(c);
151               float q = a.matches(mt);
152               if (q == 1) {
153                  em = new EncoderMatch(mt, e);
154                  cache.put(acceptEncoding, em);
155                  return em;
156               } else if (q > 0) {
157                  if (m == null)
158                     m = new TreeMap<Float,EncoderMatch>(Collections.reverseOrder());
159                  m.put(q, new EncoderMatch(mt, e));
160               }
161            }
162         }
163      }
164      return (m == null ? null : m.values().iterator().next());
165   }
166
167   /**
168    * Returns the encoder registered with the specified coding (e.g. <js>"gzip"</js>).
169    *
170    * @param coding The coding string.
171    * @return The encoder, or <jk>null</jk> if encoder isn't registered with that coding.
172    */
173   public Encoder getEncoder(String coding) {
174      EncoderMatch em = getEncoderMatch(coding);
175      return (em == null ? null : em.getEncoder());
176   }
177
178   /**
179    * Returns the set of codings supported by all encoders in this group.
180    *
181    * @return The set of codings supported by all encoders in this group.  Never <jk>null</jk>.
182    */
183   public List<String> getSupportedEncodings() {
184      List<String> l = new ArrayList<String>();
185      for (Encoder e : encoders)
186         for (String enc : e.getCodings())
187            if (! l.contains(enc))
188               l.add(enc);
189      return l;
190   }
191}
