013package org.apache.juneau.encoders;
014
015import java.util.*;
016import java.util.concurrent.*;
017
018import org.apache.juneau.*;
019
020/**
021 * Represents the group of {@link Encoder encoders} keyed by codings.
022 *
023 * <h5 class='section'>Description:</h5>
024 * <p>
025 * Maintains a set of encoders and the codings that they can handle.
026 * <p>
027 * The {@link #getEncoderMatch(String)} and {@link #getEncoder(String)} methods are then
028 *    used to find appropriate encoders for specific <code>Accept-Encoding</code>
029 *    and <code>Content-Encoding</code> header values.
030 *
031 * <h6 class='topic'>Match ordering</h6>
032 * <p>
033 * Encoders are matched against <code>Accept-Encoding</code> strings in the order they exist in this group.
034 * <p>
035 * Adding new entries will cause the entries to be prepended to the group.
036 * This allows for previous encoders to be overridden through subsequent calls.
037 * <p>
038 * For example, calling <code>groupBuilder.append(E1.<jk>class</jk>,E2.<jk>class</jk>).append(E3.<jk>class</jk>,E4.<jk>class</jk>)</code>
039 *    will result in the order <code>E3, E4, E1, E2</code>.
040 *
041 * <h5 class='section'>Example:</h5>
042 * <p class='bcode'>
043 *    <jc>// Create an encoder group with support for gzip compression.</jc>
044 *    EncoderGroup g = <jk>new</jk> EncoderGroupBuilder().append(GzipEncoder.<jk>class</jk>).build();
045 *
046 *    <jc>// Should return "gzip"</jc>
047 *    String matchedCoding = g.findMatch(<js>"compress;q=1.0, gzip;q=0.8, identity;q=0.5, *;q=0"</js>);
048 *
049 *    <jc>// Get the encoder</jc>
050 *    IEncoder encoder = g.getEncoder(matchedCoding);
051 * </p>
052 */
053public final class EncoderGroup {
054
055   // Maps Accept-Encoding headers to matching encoders.
056   private final Map<String,EncoderMatch> cache = new ConcurrentHashMap<String,EncoderMatch>();
057
058   final Encoder[] encoders;
059
060   /**
061    * Constructor
062    *
063    * @param encoders The encoders to add to this group.
064    */
065   public EncoderGroup(Encoder[] encoders) {
066      this.encoders = Arrays.copyOf(encoders, encoders.length);
067   }
068
069
070   /**
071    * Returns the coding string for the matching encoder that can handle the specified <code>Accept-Encoding</code>
072    *    or <code>Content-Encoding</code> header value.
073    * <p>
074    * Returns <jk>null</jk> if no encoders can handle it.
075    * <p>
076    * This method is fully compliant with the RFC2616/14.3 and 14.11 specifications.
077    *
078    * @param acceptEncoding The <code>Accept-Encoding</code> or <code>Content-Encoding</code> value.
079    * @return The coding value (e.g. <js>"gzip"</js>).
080    */
081   public EncoderMatch getEncoderMatch(String acceptEncoding) {
082      if (encoders.length == 0)
083         return null;
084
085      EncoderMatch em = cache.get(acceptEncoding);
086      if (em != null)
087         return em;
088
089      MediaRange[] ae = MediaRange.parse(acceptEncoding);
090
091      if (ae.length == 0)
092         ae = MediaRange.parse("*/*");
093
094      Map<Float,EncoderMatch> m = null;
095
096      for (MediaRange a : ae) {
097         for (Encoder e : encoders) {
098            for (String c : e.getCodings()) {
099               MediaType mt = MediaType.forString(c);
100               float q = a.matches(mt);
101               if (q == 1) {
102                  em = new EncoderMatch(mt, e);
103                  cache.put(acceptEncoding, em);
104                  return em;
105               } else if (q > 0) {
106                  if (m == null)
107                     m = new TreeMap<Float,EncoderMatch>(Collections.reverseOrder());
108                  m.put(q, new EncoderMatch(mt, e));
109               }
110            }
111         }
112      }
113      return (m == null ? null : m.values().iterator().next());
114   }
115
116   /**
117    * Returns the encoder registered with the specified coding (e.g. <js>"gzip"</js>).
118    *
119    * @param coding The coding string.
120    * @return The encoder, or <jk>null</jk> if encoder isn't registered with that coding.
121    */
122   public Encoder getEncoder(String coding) {
123      EncoderMatch em = getEncoderMatch(coding);
124      return (em == null ? null : em.getEncoder());
125   }
126
127   /**
128    * Returns the set of codings supported by all encoders in this group.
129    *
130    * @return The set of codings supported by all encoders in this group.  Never <jk>null</jk>.
131    */
132   public List<String> getSupportedEncodings() {
133      List<String> l = new ArrayList<String>();
134      for (Encoder e : encoders)
135         for (String enc : e.getCodings())
136            if (! l.contains(enc))
137               l.add(enc);
138      return l;
139   }
140}
