013package org.apache.juneau.http;
014
015import java.util.*;
016import java.util.concurrent.*;
017
018import org.apache.juneau.internal.*;

020/**
021 * Represents a parsed <code>Accept:</code> HTTP header.
022 * <p>
023 * The formal RFC2616 header field definition is as follows:
024 * <p class='bcode'>
025 *    14.1 Accept
026 *
027 *    The Accept request-header field can be used to specify certain media
028 *    types which are acceptable for the response. Accept headers can be
029 *    used to indicate that the request is specifically limited to a small
030 *    set of desired types, as in the case of a request for an in-line
031 *    image.
032 *
033 *     Accept         = "Accept" ":
034 *                      #( media-range [ accept-params ] )
035 *
036 *     media-range    = ( "* /*"
037 *                      | ( type "/" "*" )
038 *                      | ( type "/" subtype )
039 *                      ) *( ";" parameter )
040 *     accept-params  = ";" "q" "=" qvalue *( accept-extension )
041 *     accept-extension = ";" token [ "=" ( token | quoted-string ) ]
042 *
043 *    The asterisk "*" character is used to group media types into ranges,
044 *    with "* /*" indicating all media types and "type/*" indicating all
045 *    subtypes of that type. The media-range MAY include media type
046 *    parameters that are applicable to that range.
047 *
048 *    Each media-range MAY be followed by one or more accept-params,
049 *    beginning with the "q" parameter for indicating a relative quality
050 *    factor. The first "q" parameter (if any) separates the media-range
051 *    parameter(s) from the accept-params. Quality factors allow the user
052 *    or user agent to indicate the relative degree of preference for that
053 *    media-range, using the qvalue scale from 0 to 1 (section 3.9). The
054 *    default value is q=1.
055 *
056 *    Note: Use of the "q" parameter name to separate media type
057 *    parameters from Accept extension parameters is due to historical
058 *    practice. Although this prevents any media type parameter named
059 *    "q" from being used with a media range, such an event is believed
060 *    to be unlikely given the lack of any "q" parameters in the IANA
061 *    media type registry and the rare usage of any media type
062 *    parameters in Accept. Future media types are discouraged from
063 *    registering any parameter named "q".
064 *
065 *    The example
066 *
067 *       Accept: audio/*; q=0.2, audio/basic
068 *
069 *    SHOULD be interpreted as "I prefer audio/basic, but send me any audio
070 *    type if it is the best available after an 80% mark-down in quality."
071 *
072 *    If no Accept header field is present, then it is assumed that the
073 *    client accepts all media types. If an Accept header field is present,
074 *    and if the server cannot send a response which is acceptable
075 *    according to the combined Accept field value, then the server SHOULD
076 *    send a 406 (not acceptable) response.
077 *
078 *    A more elaborate example is
079 *
080 *        Accept: text/plain; q=0.5, text/html,
081 *                text/x-dvi; q=0.8, text/x-c
082 *
083 *    Verbally, this would be interpreted as "text/html and text/x-c are
084 *    the preferred media types, but if they do not exist, then send the
085 *    text/x-dvi entity, and if that does not exist, send the text/plain
086 *    entity."
087 *
088 *    Media ranges can be overridden by more specific media ranges or
089 *    specific media types. If more than one media range applies to a given
090 *    type, the most specific reference has precedence. For example,
091 *
092 *        Accept: text/ *, text/html, text/html;level=1, * /*
093 *
094 *    have the following precedence:
095 *
096 *        1) text/html;level=1
097 *        2) text/html
098 *        3) text/*
099 *        4) * /*
100 *
101 *    The media type quality factor associated with a given type is
102 *    determined by finding the media range with the highest precedence
103 *    which matches that type. For example,
104 *
105 *        Accept: text/*;q=0.3, text/html;q=0.7, text/html;level=1,
106 *                text/html;level=2;q=0.4, * /*;q=0.5
107 *
108 *    would cause the following values to be associated:
109 *
110 *        text/html;level=1         = 1
111 *        text/html                 = 0.7
112 *        text/plain                = 0.3
113 *        image/jpeg                = 0.5
114 *        text/html;level=2         = 0.4
115 *        text/html;level=3         = 0.7
116 *
117 *       Note: A user agent might be provided with a default set of quality
118 *       values for certain media ranges. However, unless the user agent is
119 *       a closed system which cannot interact with other rendering agents,
120 *       this default set ought to be configurable by the user.
121 * </p>
122 */
123public final class Accept {
124
125   private static final boolean nocache = Boolean.getBoolean("juneau.http.Accept.nocache");
126   private static final ConcurrentHashMap<String,Accept> cache = new ConcurrentHashMap<String,Accept>();
127
128   private final MediaTypeRange[] mediaRanges;
129   private final List<MediaTypeRange> mediaRangesList;
130
131   /**
132    * Returns a parsed <code>Accept</code> header.
133    *
134    * @param s The <code>Accept</code> header string.
135    * @return The parsed <code>Accept</code> header, or <jk>null</jk> if the string was null.
136    */
137   public static Accept forString(String s) {
138      if (s == null)
139         return null;
140
141      // Prevent OOM in case of DDOS
142      if (cache.size() > 1000)
143         cache.clear();
144
145      while (true) {
146         Accept a = cache.get(s);
147         if (a != null)
148            return a;
149         a = new Accept(s);
150         if (nocache)
151            return a;
152         cache.putIfAbsent(s, a);
153      }
154   }
155
156   private Accept(String raw) {
157      this.mediaRanges = MediaTypeRange.parse(raw);
158      this.mediaRangesList = Collections.unmodifiableList(Arrays.asList(mediaRanges));
159   }
160
161   /**
162    * Returns the list of the media ranges that make up this header.
163    * <p>
164    * The media ranges in the list are sorted by their q-value in descending order.
165    *
166    * @return An unmodifiable list of media ranges.
167    */
168   public List<MediaTypeRange> getMediaRanges() {
169      return mediaRangesList;
170   }
171
172   /**
173    * Given a list of media types, returns the best match for this <code>Accept</code> header.
174    * <p>
175    * Note that fuzzy matching is allowed on the media types where the <code>Accept</code> header may
176    * contain additional subtype parts.
177    * <br>For example, given identical q-values and an <code>Accept</code> value of <js>"text/json+activity"</js>,
178    * the media type <js>"text/json"</js> will match if <js>"text/json+activity"</js> or <js>"text/activity+json"</js>
179    * isn't found.
180    * <br>The purpose for this is to allow serializers to match when artifacts such as <code>id</code> properties are present
181    * in the header.
182    * <p>
183    * See <a class='doclink' href='https://www.w3.org/TR/activitypub/#retrieving-objects'>ActivityPub / Retrieving Objects</a>
184    * <p>
185    *
186    * @param mediaTypes The media types to match against.
187    * @return The index into the array of the best match, or <code>-1</code> if no suitable matches could be found.
188    */
189   public int findMatch(MediaType[] mediaTypes) {
190      int matchQuant = 0, matchIndex = -1;
191      float q = 0f;
192
193      // Media ranges are ordered by 'q'.
194      // So we only need to search until we've found a match.
195      for (MediaTypeRange mr : mediaRanges) {
196         float q2 = mr.getQValue();
197
198         if (q2 < q || q2 == 0)
199            break;
200
201         for (int i = 0; i < mediaTypes.length; i++) {
202            MediaType mt = mediaTypes[i];
203            int matchQuant2 = mt.match(mr.getMediaType());
204            if (matchQuant2 > matchQuant) {
205               matchIndex = i;
206               matchQuant = matchQuant2;
207               q = q2;
208            }
209         }
210      }
211
212      return matchIndex;
213   }
214
215   /**
216    * Convenience method for searching through all of the subtypes of all the media ranges in this header
217    * for the presence of a subtype fragment.
218    * <p>
219    * For example, given the header <js>"text/json+activity"</js>, calling <code>hasSubtypePart(<js>"activity"</js>)</code> returns <jk>true</jk>.
220    *
221    * @param part The media type subtype fragment.
222    * @return <jk>true</jk> if subtype fragement exists.
223    */
224   public boolean hasSubtypePart(String part) {
225
226      for (MediaTypeRange mr : this.mediaRanges)
227         if (mr.getQValue() > 0 && mr.getMediaType().getSubTypes().indexOf(part) >= 0)
228            return true;
229
230      return false;
231   }
232
233   @Override /* Object */
234   public String toString() {
235      return StringUtils.join(mediaRanges, ',');
236   }
237}
