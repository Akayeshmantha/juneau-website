022/**
023 * Represents a parsed <l>Accept</l> HTTP request header.
024 * <p>
025 * Content-Types that are acceptable for the response.
026 *
027 * <h6 class='figure'>Example</h6>
028 * <p class='bcode'>
029 *    Accept: text/plain
030 * </p>
031 *
032 * <h6 class='topic'>RFC2616 Specification</h6>
033 *
034 * The Accept request-header field can be used to specify certain media types which are acceptable for the response.
035 * Accept headers can be used to indicate that the request is specifically limited to a small set of desired types, as
036 * in the case of a request for an in-line image.
037 *
038 * <p class='bcode'>
039 *     Accept         = "Accept" ":
040 *                      #( media-range [ accept-params ] )
041 *
042 *     media-range    = ( "* /*"
043 *                      | ( type "/" "*" )
044 *                      | ( type "/" subtype )
045 *                      ) *( ";" parameter )
046 *     accept-params  = ";" "q" "=" qvalue *( accept-extension )
047 *     accept-extension = ";" token [ "=" ( token | quoted-string ) ]
048 * </p>
049 * <p>
050 * The asterisk "*" character is used to group media types into ranges, with "* /*" indicating all media types and
051 * "type/*" indicating all subtypes of that type.
052 * The media-range MAY include media type parameters that are applicable to that range.
053 * <p>
054 * Each media-range MAY be followed by one or more accept-params, beginning with the "q" parameter for indicating a
055 * relative quality factor.
056 * The first "q" parameter (if any) separates the media-range parameter(s) from the accept-params.
057 * Quality factors allow the user or user agent to indicate the relative degree of preference for that media-range,
058 * using the qvalue scale from 0 to 1 (section 3.9).
059 * The default value is q=1.
060 * <p>
061 * Note: Use of the "q" parameter name to separate media type parameters from Accept extension parameters is due to
062 * historical practice.
063 * Although this prevents any media type parameter named "q" from being used with a media range, such an event is
064 * believed to be unlikely given the lack of any "q" parameters in the IANA
065 * media type registry and the rare usage of any media type parameters in Accept.
066 * Future media types are discouraged from registering any parameter named "q".
067 * <p>
068 * The example
069 * <p class='bcode'>
070 *    Accept: audio/*; q=0.2, audio/basic
071 * </p>
072 * <p>
073 * SHOULD be interpreted as "I prefer audio/basic, but send me any audio type if it is the best available after an 80%
074 * mark-down in quality."
075 * <p>
076 * If no Accept header field is present, then it is assumed that the client accepts all media types.
077 * <p>
078 * If an Accept header field is present, and if the server cannot send a response which is acceptable according to the
079 * combined Accept field value, then the server SHOULD send a 406 (not acceptable) response.
080 * <p>
081 * A more elaborate example is
082 * <p class='bcode'>
083 *    Accept: text/plain; q=0.5, text/html,
084 *            text/x-dvi; q=0.8, text/x-c
085 * </p>
086 * <p>
087 * Verbally, this would be interpreted as "text/html and text/x-c are the preferred media types, but if they do not
088 * exist, then send the
089 * text/x-dvi entity, and if that does not exist, send the text/plain entity."
090 * <p>
091 * Media ranges can be overridden by more specific media ranges or specific media types.
092 * If more than one media range applies to a given type, the most specific reference has precedence.
093 * For example,
094 * <p class='bcode'>
095 *    Accept: text/ *, text/html, text/html;level=1, * /*
096 * </p>
097 * <p>
098 * have the following precedence:
099 * <ol>
100 *    <li>text/html;level=1
101 *    <li>text/html
102 *    <li>text/*
103 *    <li>* /*
104 * </ol>
105 * <p>
106 * The media type quality factor associated with a given type is determined by finding the media range with the highest
107 * precedence which matches that type.
108 * For example,
109 * <p class='bcode'>
110 *    Accept: text/*;q=0.3, text/html;q=0.7, text/html;level=1,
111 *            text/html;level=2;q=0.4, * /*;q=0.5
112 * </p>
113 * <p>
114 * would cause the following values to be associated:
115 * <p class='bcode'>
116 *    text/html;level=1         = 1
117 *    text/html                 = 0.7
118 *    text/plain                = 0.3
119 *    image/jpeg                = 0.5
120 *    text/html;level=2         = 0.4
121 *    text/html;level=3         = 0.7
122 * </p>
123 * <p>
124 * Note: A user agent might be provided with a default set of quality values for certain media ranges.
125 * However, unless the user agent is a closed system which cannot interact with other rendering agents, this default
126 * set ought to be configurable by the user.
127 *
128 * <h6 class='topic'>Additional Information</h6>
129 * <ul class='doctree'>
130 *    <li class='jp'><a class='doclink' href='package-summary.html#TOC'>org.apache.juneau.http</a>
131 *    <li class='extlink'><a class='doclink' href='https://www.w3.org/Protocols/rfc2616/rfc2616.html'>
132 *       Hypertext Transfer Protocol -- HTTP/1.1</a>
133 * </ul>
134 */
135public final class Accept {
136
137   private static final Cache<String,Accept> cache = new Cache<String,Accept>(NOCACHE, CACHE_MAX_SIZE);
138
139   /**
140    * Returns a parsed <code>Accept</code> header.
141    *
142    * @param value The <code>Accept</code> header string.
143    * @return The parsed <code>Accept</code> header, or <jk>null</jk> if the string was null.
144    */
145   public static Accept forString(String value) {
146      if (value == null)
147         return null;
148      Accept a = cache.get(value);
149      if (a == null)
150         a = cache.put(value, new Accept(value));
151      return a;
152   }
153
154
155   private final MediaTypeRange[] mediaRanges;
156   private final List<MediaTypeRange> mediaRangesList;
157
158   private Accept(String value) {
159      this.mediaRanges = MediaTypeRange.parse(value);
160      this.mediaRangesList = Collections.unmodifiableList(Arrays.asList(mediaRanges));
161   }
162
163   /**
164    * Returns the list of the media ranges that make up this header.
165    * <p>
166    * The media ranges in the list are sorted by their q-value in descending order.
167    *
168    * @return An unmodifiable list of media ranges.
169    */
170   public List<MediaTypeRange> asRanges() {
171      return mediaRangesList;
172   }
173
174   /**
175    * Given a list of media types, returns the best match for this <code>Accept</code> header.
176    * <p>
177    * Note that fuzzy matching is allowed on the media types where the <code>Accept</code> header may
178    * contain additional subtype parts.
179    * <br>
180    * For example, given identical q-values and an <code>Accept</code> value of <js>"text/json+activity"</js>,
181    * the media type <js>"text/json"</js> will match if <js>"text/json+activity"</js> or <js>"text/activity+json"</js>
182    * isn't found.
183    * <br>
184    * The purpose for this is to allow serializers to match when artifacts such as <code>id</code> properties are
185    * present in the header.
186    * <p>
187    * See <a class='doclink' href='https://www.w3.org/TR/activitypub/#retrieving-objects'>
188    * ActivityPub / Retrieving Objects</a>
189    * <p>
190    *
191    * @param mediaTypes The media types to match against.
192    * @return The index into the array of the best match, or <code>-1</code> if no suitable matches could be found.
193    */
194   public int findMatch(MediaType[] mediaTypes) {
195      int matchQuant = 0, matchIndex = -1;
196      float q = 0f;
197
198      // Media ranges are ordered by 'q'.
199      // So we only need to search until we've found a match.
200      for (MediaTypeRange mr : mediaRanges) {
201         float q2 = mr.getQValue();
202
203         if (q2 < q || q2 == 0)
204            break;
205
206         for (int i = 0; i < mediaTypes.length; i++) {
207            MediaType mt = mediaTypes[i];
208            int matchQuant2 = mt.match(mr.getMediaType());
209            if (matchQuant2 > matchQuant) {
210               matchIndex = i;
211               matchQuant = matchQuant2;
212               q = q2;
213            }
214         }
215      }
216
217      return matchIndex;
218   }
219
220   /**
221    * Convenience method for searching through all of the subtypes of all the media ranges in this header
222    * for the presence of a subtype fragment.
223    * <p>
224    * For example, given the header <js>"text/json+activity"</js>, calling
225    * <code>hasSubtypePart(<js>"activity"</js>)</code> returns <jk>true</jk>.
226    *
227    * @param part The media type subtype fragment.
228    * @return <jk>true</jk> if subtype fragment exists.
229    */
230   public boolean hasSubtypePart(String part) {
231
232      for (MediaTypeRange mr : this.mediaRanges)
233         if (mr.getQValue() > 0 && mr.getMediaType().getSubTypes().indexOf(part) >= 0)
234            return true;
235
236      return false;
237   }
238
239   @Override /* Object */
240   public String toString() {
241      return join(mediaRanges, ',');
242   }
243}
