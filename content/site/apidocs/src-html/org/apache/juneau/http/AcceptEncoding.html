013package org.apache.juneau.http;
014
015import java.util.*;
016import java.util.concurrent.*;
017
018import org.apache.juneau.internal.*;
019
020/**
021 * Represents a parsed <code>Accept-Encoding:</code> HTTP header.
022 * <p>
023 * The formal RFC2616 header field definition is as follows:
024 * <p class='bcode'>
025 *    The Accept-Encoding request-header field is similar to Accept, but restricts
026 *    the content-codings (section 3.5) that are acceptable in the response.
027 *
028 *       Accept-Encoding  = "Accept-Encoding" ":"
029 *                          1#( codings [ ";" "q" "=" qvalue ] )
030 *       codings          = ( content-coding | "*" )
031 *
032 *    Examples of its use are:
033 *
034 *       Accept-Encoding: compress, gzip
035 *       Accept-Encoding:
036 *       Accept-Encoding: *
037 *       Accept-Encoding: compress;q=0.5, gzip;q=1.0
038 *       Accept-Encoding: gzip;q=1.0, identity; q=0.5, *;q=0
039 *
040 *    A server tests whether a content-coding is acceptable, according to an
041 *    Accept-Encoding field, using these rules:
042 *
043 *       1. If the content-coding is one of the content-codings listed in
044 *          the Accept-Encoding field, then it is acceptable, unless it is
045 *          accompanied by a qvalue of 0. (As defined in section 3.9, a
046 *          qvalue of 0 means "not acceptable.")
047 *       2. The special "*" symbol in an Accept-Encoding field matches any
048 *          available content-coding not explicitly listed in the header
049 *          field.
050 *       3. If multiple content-codings are acceptable, then the acceptable
051 *          content-coding with the highest non-zero qvalue is preferred.
052 *       4. The "identity" content-coding is always acceptable, unless
053 *          specifically refused because the Accept-Encoding field includes
054 *          "identity;q=0", or because the field includes "*;q=0" and does
055 *          not explicitly include the "identity" content-coding. If the
056 *          Accept-Encoding field-value is empty, then only the "identity"
057 *          encoding is acceptable.
058 *
059 *    If an Accept-Encoding field is present in a request, and if the server cannot
060 *    send a response which is acceptable according to the Accept-Encoding header,
061 *    then the server SHOULD send an error response with the 406 (Not Acceptable) status code.
062 *
063 *    If no Accept-Encoding field is present in a request, the server MAY assume
064 *    that the client will accept any content coding. In this case, if "identity"
065 *    is one of the available content-codings, then the server SHOULD use the "identity"
066 *    content-coding, unless it has additional information that a different content-coding
067 *    is meaningful to the client.
068 *
069 *          Note: If the request does not include an Accept-Encoding field,
070 *          and if the "identity" content-coding is unavailable, then
071 *          content-codings commonly understood by HTTP/1.0 clients (i.e.,
072 *          "gzip" and "compress") are preferred; some older clients
073 *          improperly display messages sent with other content-codings.  The
074 *          server might also make this decision based on information about
075 *          the particular user-agent or client.
076 *          Note: Most HTTP/1.0 applications do not recognize or obey qvalues
077 *          associated with content-codings. This means that qvalues will not
078 *          work and are not permitted with x-gzip or x-compress.
079 * </p>
080 */
081public final class AcceptEncoding {
082
083   private static final boolean nocache = Boolean.getBoolean("juneau.http.AcceptEncoding.nocache");
084   private static final ConcurrentHashMap<String,AcceptEncoding> cache = new ConcurrentHashMap<String,AcceptEncoding>();
085
086   private final TypeRange[] typeRanges;
087   private final List<TypeRange> typeRangesList;
088
089   /**
090    * Returns a parsed <code>Accept-Encoding</code> header.
091    *
092    * @param s The <code>Accept-Encoding</code> header string.
093    * @return The parsed <code>Accept-Encoding</code> header, or <jk>null</jk> if the string was null.
094    */
095   public static AcceptEncoding forString(String s) {
096      if (s == null)
097         return null;
098
099      // Prevent OOM in case of DDOS
100      if (cache.size() > 1000)
101         cache.clear();
102
103      while (true) {
104         AcceptEncoding a = cache.get(s);
105         if (a != null)
106            return a;
107         a = new AcceptEncoding(s);
108         if (nocache)
109            return a;
110         cache.putIfAbsent(s, a);
111      }
112   }
113
114   private AcceptEncoding(String raw) {
115      this.typeRanges = TypeRange.parse(raw);
116      this.typeRangesList = Collections.unmodifiableList(Arrays.asList(typeRanges));
117   }
118
119   /**
120    * Returns the list of the types ranges that make up this header.
121    * <p>
122    * The types ranges in the list are sorted by their q-value in descending order.
123    *
124    * @return An unmodifiable list of type ranges.
125    */
126   public List<TypeRange> getTypeRanges() {
127      return typeRangesList;
128   }
129
130   /**
131    * Given a list of content codings, returns the best match for this <code>Accept-Encoding</code> header.
132    * <p>
133    *
134    * @param contentCodings The codings to match against.
135    * @return The index into the array of the best match, or <code>-1</code> if no suitable matches could be found.
136    */
137   public int findMatch(String[] contentCodings) {
138
139      // Type ranges are ordered by 'q'.
140      // So we only need to search until we've found a match.
141      for (TypeRange mr : typeRanges)
142         for (int i = 0; i < contentCodings.length; i++)
143            if (mr.matches(contentCodings[i]))
144               return i;
145
146      return -1;
147   }
148
149   @Override /* Object */
150   public String toString() {
151      return StringUtils.join(typeRanges, ',');
152   }
153}
