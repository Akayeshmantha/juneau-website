023/**
024 * Describes a single media type used in content negotiation between an HTTP client and server, as described in
025 * Section 14.1 and 14.7 of RFC2616 (the HTTP/1.1 specification).
026 *
027 * <h6 class='topic'>Additional Information</h6>
028 * <ul class='doctree'>
029 *    <li class='jp'><a class='doclink' href='package-summary.html#TOC'>org.apache.juneau.http</a>
030  *   <li class='extlink'><a class='doclink' href='https://www.w3.org/Protocols/rfc2616/rfc2616.html'>Hypertext Transfer Protocol -- HTTP/1.1</a>
031* </ul>
032 */
033@BeanIgnore
034@SuppressWarnings("unchecked")
035public class MediaType implements Comparable<MediaType> {
036
037   private static final boolean nocache = Boolean.getBoolean("juneau.nocache");
038   private static final ConcurrentHashMap<String,MediaType> cache = new ConcurrentHashMap<String,MediaType>();
039
040   /** Reusable predefined media type */
041   @SuppressWarnings("javadoc")
042   public static final MediaType
043      CSV = forString("text/csv"),
044      HTML = forString("text/html"),
045      JSON = forString("application/json"),
046      MSGPACK = forString("octal/msgpack"),
047      PLAIN = forString("text/plain"),
048      UON = forString("text/uon"),
049      URLENCODING = forString("application/x-www-form-urlencoded"),
050      XML = forString("text/xml"),
051      XMLSOAP = forString("text/xml+soap"),
052
053      RDF = forString("text/xml+rdf"),
054      RDFABBREV = forString("text/xml+rdf+abbrev"),
055      NTRIPLE = forString("text/n-triple"),
056      TURTLE = forString("text/turtle"),
057      N3 = forString("text/n3")
058   ;
059
060   private final String mediaType;
061   private final String type;                           // The media type (e.g. "text" for Accept, "utf-8" for Accept-Charset)
062   private final String subType;                        // The media sub-type (e.g. "json" for Accept, not used for Accept-Charset)
063   private final String[] subTypes;                     // The media sub-type (e.g. "json" for Accept, not used for Accept-Charset)
064   private final List<String> subTypesList;             // The media sub-type (e.g. "json" for Accept, not used for Accept-Charset)
065   private final Map<String,Set<String>> parameters;    // The media type parameters (e.g. "text/html;level=1").  Does not include q!
066
067
068   /**
069    * Returns the media type for the specified string.
070    * The same media type strings always return the same objects so that these objects
071    * can be compared for equality using '=='.
072    * <p>
073    * <h5 class='section'>Notes:</h5>
074    * <ul>
075    *    <li>Spaces are replaced with <js>'+'</js> characters.
076    *       This gets around the issue where passing media type strings with <js>'+'</js> as HTTP GET parameters
077    *       get replaced with spaces by your browser.  Since spaces aren't supported by the spec, this
078    *       is doesn't break anything.
079    *    <li>Anything including and following the <js>';'</js> character is ignored (e.g. <js>";charset=X"</js>).
080    * </ul>
081    *
082    * @param s The media type string.  Will be lowercased.
083    *    <br>Returns <jk>null</jk> if input is null.
084    * @return A cached media type object.
085    */
086   public static MediaType forString(String s) {
087      if (s == null)
088         return null;
089      MediaType mt = cache.get(s);
090      if (mt == null) {
091         mt = new MediaType(s);
092         if (nocache)
093            return mt;
094         cache.putIfAbsent(s, mt);
095      }
096      return cache.get(s);
097   }
098
099   MediaType(String mt) {
100      Builder b = new Builder(mt);
101      this.mediaType = b.mediaType;
102      this.type = b.type;
103      this.subType = b.subType;
104      this.subTypes = b.subTypes;
105      this.subTypesList = Collections.unmodifiableList(Arrays.asList(subTypes));
106      this.parameters = (b.parameters == null ? Collections.EMPTY_MAP : Collections.unmodifiableMap(b.parameters));
107   }
108
109   private static class Builder {
110      private String mediaType, type, subType;
111      private String[] subTypes;
112      private Map<String,Set<String>> parameters;
113
114      private Builder(String mt) {
115         mt = mt.trim();
116
117         int i = mt.indexOf(';');
118         if (i == -1) {
119            this.parameters = Collections.EMPTY_MAP;
120         } else {
121            this.parameters = new TreeMap<String,Set<String>>();
122            String[] tokens = mt.substring(i+1).split(";");
123
124            for (int j = 0; j < tokens.length; j++) {
125               String[] parm = tokens[j].split("=");
126               if (parm.length == 2) {
127                  String k = parm[0].trim(), v = parm[1].trim();
128                  if (! parameters.containsKey(k))
129                     parameters.put(k, new TreeSet<String>());
130                  parameters.get(k).add(v);
131               }
132            }
133
134            mt = mt.substring(0, i);
135         }
136
137         this.mediaType = mt;
138         if (mt != null) {
139            mt = mt.replace(' ', '+');
140            i = mt.indexOf('/');
141            type = (i == -1 ? mt : mt.substring(0, i));
142            subType = (i == -1 ? "*" : mt.substring(i+1));
143         }
144         this.subTypes = StringUtils.split(subType, '+');
145      }
146   }

150    *
151    * @return The media type.
152    */
153   public String getType() {
154      return type;
155   }
156
157   /**
158    * Returns the <js>'subType'</js> fragment of the <js>'type/subType'</js> string.
159    *
160    * @return The media subtype.
161    */
162   public String getSubType() {
163      return subType;
164   }
165
166   /**
167    * Returns the subtypes broken down by fragments delimited by <js>"'"</js>.
168    * For example, the media type <js>"text/foo+bar"</js> will return a list of
169    * <code>[<js>'foo'</js>,<js>'bar'</js>]</code>
170    *
171    * @return An unmodifiable list of subtype fragments.  Never <jk>null</jk>.
172    */
173   public List<String> getSubTypes() {
174      return subTypesList;
175   }
176
177   /**
178    * Returns <jk>true</jk> if this media type is a match for the specified media type.
179    * <p>
180    * Matches if any of the following is true:
181    * <ul>
182    *    <li>Both type and subtype are the same.
183    *    <li>One or both types are <js>'*'</js> and the subtypes are the same.
184    *    <li>One or both subtypes are <js>'*'</js> and the types are the same.
185    *    <li>Either is <js>'*\/*'</js>.
186    * </ul>
187    *
188    * @param o The media type to compare with.
189    * @return <jk>true</jk> if the media types match.
190    */
191   public final boolean matches(MediaType o) {
192      return match(o) > 0;
193   }
194
195   /**
196    * Returns a match metric against the specified media type where a larger number represents a better match.
197    * <p>
198    * <ul>
199    *    <li>Exact matches (e.g. <js>"text/json"<js>/</js>"text/json"</js>) should match
200    *       better than metacharacter matches (e.g. <js>"text/*"<js>/</js>"text/json"</js>)
201    *    <li>The comparison media type can have additional subtype tokens (e.g. <js>"text/json+foo"</js>)
202    *       that will not prevent a match.  The reverse is not true, e.g. the comparison media type
203    *       must contain all subtype tokens found in the comparing media type.
204    *       <ul>
205    *          <li>We want the {@link JsonSerializer} (<js>"text/json"</js>) class to be able to handle requests for <js>"text/json+foo"</js>.
206    *          <li>We want to make sure {@link org.apache.juneau.json.JsonSerializer.Simple} (<js>"text/json+simple"</js>) does not handle
207    *             requests for <js>"text/json"</js>.
208    *       </ul>
209    *       More token matches should result in a higher match number.
210    * </ul>
211    *
212    * @param o The media type to compare with.
213    * @return <jk>true</jk> if the media types match.
214    */
215   public final int match(MediaType o) {
216
217      // Perfect match
218      if (this == o || (type.equals(o.type) && subType.equals(o.subType)))
219         return Integer.MAX_VALUE;
220
221      int c1 = 0, c2 = 0;
222
223      if (type.equals(o.type))
224         c1 += 10000;
225      else if ("*".equals(type) || "*".equals(o.type))
226         c1 += 5000;
227
228      if (c1 == 0)
229         return 0;
230
231      // Give type slightly higher comparison value than subtype simply for deterministic results.
232      if (subType.equals(o.subType))
233         return c1 + 9999;
234
235      int c3 = 0;
236
237      for (String st1 : subTypes) {
238         if ("*".equals(st1))
239            c1++;
240         else if (ArrayUtils.contains(st1, o.subTypes))
241            c1 += 100;
242         else if (ArrayUtils.contains("*", o.subTypes))
243            c1 += 10;
244         else
245            return 0;
246      }
247
248      return c1 + c2 + c3;
249   }
250
251   /**
252    * Returns the additional parameters on this media type.
253    * <p>
254    * For example, given the media type string <js>"text/html;level=1"</js>, will return a map
255    * with the single entry <code>{level:[<js>'1'</js>]}</code>.
256    *
257    * @return The map of additional parameters, or an empty map if there are no parameters.
258    */
259   public Map<String,Set<String>> getParameters() {
260      return parameters;
261   }
262
263   @Override /* Object */
264   public String toString() {
265      if (parameters.isEmpty())
266         return mediaType;
267      StringBuilder sb = new StringBuilder(mediaType);
268      for (Map.Entry<String,Set<String>> e : parameters.entrySet())
269         for (String value : e.getValue())
270            sb.append(';').append(e.getKey()).append('=').append(value);
271      return sb.toString();
272   }
273
274   @Override /* Object */
275   public int hashCode() {
276      return mediaType.hashCode();
277   }
278
279   @Override /* Object */
280   public boolean equals(Object o) {
281      return this == o;
282   }
283
284   @Override
285   public int compareTo(MediaType o) {
286      return mediaType.compareTo(o.mediaType);
287   }
