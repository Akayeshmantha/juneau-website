027 * Represents a mutable property.
028 */
029public abstract class Property {

035   /**
036    * Creates a new property with the specified name and value.
037    * 
038    * @param name The property name.
039    * @return A new property object.
040    */
041   public static Property create(String name) {
042      int i1 = name.indexOf('.');
043      if (i1 == -1)
044         throw new ConfigException("Invalid property name: ''{0}''", name);
045      int i2 = name.indexOf('.', i1);
046      if (i2 == -1)
047         throw new ConfigException("Invalid property name: ''{0}''", name);
048      String type = name.substring(i1, i2);
049      
050      switch (type) {
051         case "s":  return new SimpleProperty(name, "String", stringConverter);
052         case "b":  return new SimpleProperty(name, "Boolean", booleanConverter);
053         case "i":  return new SimpleProperty(name, "Integer", integerConverter);
054         case "c":  return new SimpleProperty(name, "Class", classConverter);
055         case "ss": return new SetProperty(name, "Set<String>", stringConverter);
056         case "si": return new SetProperty(name, "Set<Integer>", integerConverter);
057         case "sc": return new SetProperty(name, "Set<Class>", classConverter);
058         case "ls": return new ListProperty(name, "List<String>", stringConverter);
059         case "li": return new ListProperty(name, "List<Integer>", integerConverter);
060         case "lc": return new ListProperty(name, "List<Class>", classConverter);
061         case "ms": return new MapProperty(name, "Map<String,String>", stringConverter);
062         case "mi": return new MapProperty(name, "Map<String,Integer>", integerConverter);
063         case "mc": return new MapProperty(name, "Map<String,Class>", classConverter);
064         default: throw new ConfigException("Invalid property name: ''{0}''", name);
065      }
066   }

068   //-------------------------------------------------------------------------------------------------------------------
069   // Converters
070   //-------------------------------------------------------------------------------------------------------------------
071   
072   static interface IConverter<T> {
073      T convert(Object o);
074   }
075   
076   static final IConverter<String> stringConverter = new IConverter<String>() {
077      @Override
078      public String convert(Object o) {
079         return StringUtils.toString(o);
080      }
081   };
082
083   static final IConverter<Integer> integerConverter = new IConverter<Integer>() {
084      @Override
085      public Integer convert(Object o) {
086         try {
087            if (o == null)
088               return 0;
089            if (o instanceof Integer)
090               return (Integer)o;
091            return Integer.valueOf(o.toString());
092         } catch (Exception e) {
093            throw new ConfigException("Value ''{0}'' cannot be converted to an Integer.", o);
094         }
095      }
096   };
097   
098   static final IConverter<Boolean> booleanConverter = new IConverter<Boolean>() {
099      @Override
100      public Boolean convert(Object o) {
101         try {
102            if (o == null)
103               return false;
104            if (o instanceof Boolean)
105               return (Boolean)o;
106            return Boolean.valueOf(o.toString());
107         } catch (Exception e) {
108            throw new ConfigException("Value ''{0}'' cannot be converted to a Boolean.", o);
109         }
110      }
111   };
112   
113   static final IConverter<Class<?>> classConverter = new IConverter<Class<?>>() {
114      @Override
115      public Class<?> convert(Object o) {
116         try {
117            if (o == null)
118               return null;
119            if (o instanceof Class)
120               return (Class<?>)o;
121            throw new ConfigException("Value ''{0}'' cannot be converted to a Class.", o);
122         } catch (Exception e) {
123            throw new ConfigException("Value ''{0}'' cannot be converted to a Class.", o);
124         }
125      }
126   };

133   final String name, type;
134   final IConverter<?> converter;
135   
136   Property(String name, String type, IConverter<?> converter) {
137      this.name = name;
138      this.type = type;
139      this.converter = converter;
140   }
141   
142   abstract Object get();
143   
144   abstract Property set(Object val);
145
146   Property add(Object val) {
147      throw new ConfigException("Cannot add value {0} ({1}) to property ''{2}'' ({3}).",
148         JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type);
149   }
150
151   Property remove(Object val) {
152      throw new ConfigException("Cannot remove value {0} ({1}) from property ''{2}'' ({3}).",
153         JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type);
154   }
155
156   Property put(String key, Object val) {
157      throw new ConfigException("Cannot put value {0}({1})->{2}({3}) to property ''{4}'' ({5}).",
158         JsonSerializer.DEFAULT_LAX.toString(key), getReadableClassNameForObject(key),
159         JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type);
160   }
161   
162   Object convert(Object val) {
163      return converter.convert(val);
164   }

170   static class SimpleProperty extends Property {
171      volatile Object val;
172      volatile int hashCode;
173
174      SimpleProperty(String name, String type, IConverter<?> converter) {
175         super(name, type, converter);
176      }
177      
178      @Override /* Property */
179      synchronized Object get() {
180         return val;
181      }
182
183      @Override /* Property */
184      synchronized Property set(Object val) {
185         this.val = convert(val);
186         this.hashCode = val.hashCode();
187         return this;
188      }
189      
190      @Override /* Object */
191      public synchronized int hashCode() {
192         return hashCode;
193      }
194      
195      @Override /* Object */
196      public synchronized boolean equals(Object o) {
197         return val == o || (val != null && val.equals(o));
198      }
199   }

202   // SetProperty
203   //-------------------------------------------------------------------------------------------------------------------
204   
205   @SuppressWarnings("unchecked")
206   static class SetProperty extends Property {
207      volatile Set<Object> val = new ConcurrentSkipListSet<>();
208      volatile int hashCode;
209
210      SetProperty(String name, String type, IConverter<?> converter) {
211         super(name, type, converter);
212      }
213      
214      @Override /* Property */
215      synchronized Object get() {
216         return Collections.unmodifiableSet(new TreeSet<>(val));
217      }
218
219      @Override /* Property */
220      synchronized Property set(Object val) {
221         this.val.clear();
222         return add(val);
223      }
224      
225      @Override /* Property */
226      synchronized Property add(Object val) {
227         if (val.getClass().isArray()) {
228            for (int i = 0; i < Array.getLength(val); i++)
229               add(Array.get(val, i));
230         } else if (val instanceof Collection) {
231            for (Object o : (Collection<Object>)val)
232               add(o);
233         } else if (isObjectList(val)) {
234            try {
235               add(new ObjectList(val.toString()));
236            } catch (Exception e) {
237               throw new ConfigException(
238                  "Cannot add value {0} ({1}) to property ''{2}'' ({3}).",
239                  JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type
240               );
241            }
242         } else {
243            this.val.add(converter.convert(val));
244         }
245         hashCode = this.val.hashCode();
246         return this;
247      }
248      
249      @Override /* Property */
250      synchronized Property remove(Object val) {
251         if (val.getClass().isArray()) {
252            for (int i = 0; i < Array.getLength(val); i++)
253               remove(Array.get(val, i));
254         } else if (val instanceof Collection) {
255            for (Object o : (Collection<Object>)val)
256               remove(o);
257         } else if (isObjectList(val)) {
258            try {
259               remove(new ObjectList(val.toString()));
260            } catch (Exception e) {
261               throw new ConfigException(
262                  "Cannot remove value {0} ({1}) from property ''{2}'' ({3}).",
263                  JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type
264               );
265            }              
266         } else {
267            this.val.add(converter.convert(val));
268         }
269         hashCode = this.val.hashCode();
270         return this;
271      }
272      
273      @Override /* Object */
274      public synchronized int hashCode() {
275         return hashCode;
276      }
277      
278      @Override /* Object */
279      public synchronized boolean equals(Object o) {
280         return val.equals(o);
281      }
282   }

288   @SuppressWarnings("unchecked")
289   static class ListProperty extends Property {
290      volatile List<Object> val = new CopyOnWriteArrayList<>();
291      volatile int hashCode;
292
293      ListProperty(String name, String type, IConverter<?> converter) {
294         super(name, type, converter);
295      }
296      
297      @Override /* Property */
298      synchronized Object get() {
299         return Collections.unmodifiableList(new ArrayList<>(val));
300      }
301
302      @Override /* Property */
303      synchronized Property set(Object val) {
304         this.val.clear();
305         add(val);
306         return this;
307      }
308      
309      @Override /* Property */
310      synchronized Property add(Object val) {
311         if (val.getClass().isArray()) {
312            for (int i = Array.getLength(val) - 1; i >= 0; i--)
313               add(Array.get(val, i));
314         } else if (val instanceof List) {
315            List<Object> l = (List<Object>)val;
316            for (ListIterator<Object> i = l.listIterator(l.size()); i.hasPrevious();)
317               add(i.previous());
318         } else if (val instanceof Collection) {
319            List<Object> l = new ArrayList<>((Collection<Object>)val);
320            for (ListIterator<Object> i = l.listIterator(l.size()); i.hasPrevious();)
321               add(i.previous());
322         } else if (isObjectList(val)) {
323            try {
324               add(new ObjectList(val.toString()));
325            } catch (Exception e) {
326               throw new ConfigException(
327                  "Cannot remove add {0} ({1}) to property ''{2}'' ({3}).",
328                  JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type
329               );
330            }
331         } else {
332            this.val.remove(val);
333            this.val.add(val);
334         }
335         this.hashCode = this.val.hashCode();
336         return this;
337      }
338      
339      @Override /* Object */
340      public synchronized int hashCode() {
341         synchronized(this) {
342            return hashCode;
343         }
344      }
345      
346      @Override /* Object */
347      public synchronized boolean equals(Object o) {
348         return val.equals(o);
349      }
350   }

356   @SuppressWarnings("rawtypes")
357   static class MapProperty extends Property {
358      volatile Map<String,Object> val = new ConcurrentHashMap<>();
359      volatile int hashCode;
360      
361      MapProperty(String name, String type, IConverter<?> converter) {
362         super(name, type, converter);
363      }
364      
365      @Override /* Property */
366      synchronized Object get() {
367         return Collections.unmodifiableMap(new TreeMap<>(val));
368      }
369
370      @Override /* Property */
371      synchronized Property set(Object val) {
372         this.val.clear();
373         add(val);
374         return this;
375      }
376
377      @Override /* Property */
378      synchronized Property add(Object val) {
379         if (val instanceof Map) {
380            Map m = (Map)val;
381            for (Map.Entry e : (Set<Map.Entry>)m.entrySet())
382               put(e.getKey().toString(), e.getValue());
383            return this;
384         } else if (isObjectMap(val)) {
385            try {
386               add(new ObjectMap(val.toString()));
387            } catch (Exception e) {
388               throw new ConfigException(
389                  "Cannot remove add {0} ({1}) to property ''{2}'' ({3}).",
390                  JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type
391               );
392            }
393         } else {
394            throw new ConfigException(
395               "Cannot add {0} ({1}) to property ''{2}'' ({3}).",
396               JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type
397            );
398         }
399         this.hashCode = this.val.hashCode();
400         return this;
401      }
402
403      @Override /* Property */
404      synchronized Property put(String key, Object val) {
405         for (Entry<String,Object> e : this.val.entrySet()) {
406            if (e.getKey().equals(val)) {
407               e.setValue(converter.convert(val));
408               return this;
409            }
410         }
411         this.val.put(key, converter.convert(val));
412         this.hashCode = this.val.hashCode();
413         return this;
414      }
415      
416      @Override /* Object */
417      public synchronized int hashCode() {
418         return hashCode;
419      }
420      
421      @Override /* Object */
422      public synchronized boolean equals(Object o) {
423         return val.equals(o);
424      }
425   }

431   static boolean isObjectList(Object o) {
432      if (o instanceof CharSequence) {
433         String s = o.toString();
434         return (s.startsWith("[") && s.endsWith("]"));
435      }
436      return false;
437   }
438
439   static boolean isObjectMap(Object o) {
440      if (o instanceof CharSequence) {
441         String s = o.toString();
442         return (s.startsWith("{") && s.endsWith("}"));
443      }
444      return false;
445   }
