016import static org.apache.juneau.serializer.SerializerContext.*;
017
018import java.lang.reflect.*;
019import java.util.*;
020
021import org.apache.juneau.*;
022import org.apache.juneau.annotation.*;
023import org.apache.juneau.serializer.*;
024import org.apache.juneau.transform.*;

026/**
027 * Serializes POJO metamodels to HTML.
028 *
029 * <h6 class='topic'>Media types</h6>
030 * <p>
031 *    Handles <code>Accept</code> types: <code>text/html+schema</code>
032 * <p>
033 *    Produces <code>Content-Type</code> types: <code>text/html</code>
034 *
035 * <h6 class='topic'>Description</h6>
036 * <p>
037 *    Essentially the same as {@link HtmlSerializer}, except serializes the POJO metamodel
038 *       instead of the model itself.
039 * <p>
040 *    Produces output that describes the POJO metamodel similar to an XML schema document.
041 * <p>
042 *    The easiest way to create instances of this class is through the {@link HtmlSerializer#getSchemaSerializer()},
043 *       which will create a schema serializer with the same settings as the originating serializer.
044 */
045@Produces(value="text/html+schema", contentType="text/html")
046public final class HtmlSchemaDocSerializer extends HtmlDocSerializer {
047
048   /**
049    * Constructor.
050    */
051   public HtmlSchemaDocSerializer() {
052      setProperty(SERIALIZER_detectRecursions, true);
053      setProperty(SERIALIZER_ignoreRecursions, true);
054   }
055
056   /**
057    * Constructor.
058    *
059    * @param cf The context factory to use for creating the context for this serializer.
060    */
061   public HtmlSchemaDocSerializer(ContextFactory cf) {
062      getContextFactory().copyFrom(cf);
063      setProperty(SERIALIZER_detectRecursions, true);
064      setProperty(SERIALIZER_ignoreRecursions, true);
065   }
066
067   @Override /* Serializer */
068   public HtmlDocSerializerSession createSession(Object output, ObjectMap op, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType) {
069      return new HtmlDocSerializerSession(getContext(HtmlDocSerializerContext.class), op, output, javaMethod, locale, timeZone, mediaType);
070   }
071
072   @Override /* ISchemaSerializer */
073   protected void doSerialize(SerializerSession session, Object o) throws Exception {
074      HtmlSerializerSession s = (HtmlSerializerSession)session;
075      ObjectMap schema = getSchema(s, session.getClassMetaForObject(o), "root", null);
076      super.doSerialize(s, schema);
077   }
078
079   /*
080    * Creates a schema representation of the specified class type.
081    *
082    * @param eType The class type to get the schema of.
083    * @param ctx Serialize context used to prevent infinite loops.
084    * @param attrName The name of the current attribute.
085    * @return A schema representation of the specified class.
086    * @throws SerializeException If a problem occurred trying to convert the output.
087    */
088   @SuppressWarnings({ "unchecked", "rawtypes" })
089   private ObjectMap getSchema(HtmlSerializerSession session, ClassMeta<?> eType, String attrName, String[] pNames) throws Exception {
090
091      ObjectMap out = new ObjectMap();
092
093      ClassMeta<?> aType;        // The actual type (will be null if recursion occurs)
094      ClassMeta<?> sType;        // The serialized type
095
096      aType = session.push(attrName, eType, null);
097
098      sType = eType.getSerializedClassMeta();
099      String type = null;
100
101      if (sType.isEnum() || sType.isCharSequence() || sType.isChar())
102         type = "string";
103      else if (sType.isNumber())
104         type = "number";
105      else if (sType.isBoolean())
106         type = "boolean";
107      else if (sType.isMapOrBean())
108         type = "object";
109      else if (sType.isCollectionOrArray())
110         type = "array";
111      else
112         type = "any";
113
114      out.put("type", type);
115      out.put("class", eType.toString());
116      PojoSwap t = eType.getPojoSwap();
117      if (t != null)
118         out.put("transform", t);
119
120      if (aType != null) {
121         if (sType.isEnum())
122            out.put("enum", getEnumStrings((Class<Enum<?>>)sType.getInnerClass()));
123         else if (sType.isCollectionOrArray()) {
124            ClassMeta componentType = sType.getElementType();
125            if (sType.isCollection() && isParentClass(Set.class, sType.getInnerClass()))
126               out.put("uniqueItems", true);
127            out.put("items", getSchema(session, componentType, "items", pNames));
128         } else if (sType.isBean()) {
129            ObjectMap properties = new ObjectMap();
130            BeanMeta bm = session.getBeanMeta(sType.getInnerClass());
131            if (pNames != null)
132               bm = new BeanMetaFiltered(bm, pNames);
133            for (Iterator<BeanPropertyMeta> i = bm.getPropertyMetas().iterator(); i.hasNext();) {
134               BeanPropertyMeta p = i.next();
135               properties.put(p.getName(), getSchema(session, p.getClassMeta(), p.getName(), p.getProperties()));
136            }
137            out.put("properties", properties);
138         }
139      }
140      session.pop();
141      return out;
142   }
143
144   @SuppressWarnings({ "unchecked", "rawtypes" })
145   private List<String> getEnumStrings(Class<? extends Enum> c) {
146      List<String> l = new LinkedList<String>();
147      for (Object e : EnumSet.allOf(c))
148         l.add(e.toString());
149      return l;
150   }
151}
