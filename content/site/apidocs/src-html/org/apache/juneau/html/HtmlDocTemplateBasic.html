013package org.apache.juneau.html;
014
015import org.apache.juneau.internal.*;
016
017/**
018 * A basic template for the HTML doc serializer.
019 *
020 * <p>
021 * This class can be subclassed to customize page rendering.
022 */
023public class HtmlDocTemplateBasic implements HtmlDocTemplate {
024
025   @Override /* HtmlDocTemplate */
026   public void head(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
027      if (hasStyle(session)) {
028         w.sTag(2, "style").nl(2);
029         style(session, w, s, o);
030         w.ie(2).eTag("style").nl(2);
031      }
032      if (hasScript(session)) {
033         w.sTag(2, "script").nl(2);
034         script(session, w, s, o);
035         w.ie(2).eTag("script").nl(2);
036      }
037   }
038
039   @Override /* HtmlDocTemplate */
040   public void style(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
041
042      String[] stylesheet = session.getStylesheet();
043      if (! ArrayUtils.contains("NONE", stylesheet))
044         for (String ss : stylesheet)
045            w.append(3, "@import ").q().append(session.resolveUri(ss)).q().appendln("; ");
046
047      if (session.isNoWrap())
048         w.appendln(3, "div.data * {white-space:nowrap;} ");
049
050      if (session.getStyle() != null)
051         for (String style : session.getStyle())
052            w.append(3, style).appendln(" ");
053   }
054
055   @Override /* HtmlDocTemplate */
056   public void script(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
057
058      if (session.getScript() != null)
059         for (String script : session.getScript())
060            w.append(3, script);
061   }
062
063   @Override /* HtmlDocTemplate */
064   public void body(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
065
066      if (hasHeader(session)) {
067         w.sTag(2, "header").nl(2);
068         header(session, w, s, o);
069         w.ie(2).eTag("header").nl(2);
070      }
071
072      if (hasNav(session)) {
073         w.sTag(2, "nav").nl(2);
074         nav(session, w, s, o);
075         w.ie(2).eTag("nav").nl(2);
076      }
077
078      w.sTag(2, "section").nl(2);
079
080      w.sTag(3, "article").nl(3);
081      article(session, w, s, o);
082      w.ie(3).eTag("article").nl(3);
083
084      if (hasAside(session)) {
085         w.sTag(3, "aside").nl(3);
086         aside(session, w, s, o);
087         w.ie(3).eTag("aside").nl(3);
088      }
089
090      w.ie(2).eTag("section").nl(2);
091
092      if (hasFooter(session)) {
093         w.sTag(2, "footer").nl(2);
094         footer(session, w, s, o);
095         w.ie(2).eTag("footer").nl(2);
096      }
097   }
098
099   @Override /* HtmlDocTemplate */
100   public void header(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
101      // Write the title of the page.
102      String header = session.getHeader();
103      if (header != null) {
104         if (exists(header))
105            w.append(3, header).nl(3);
106      } else {
107         String title = session.getTitle();
108         String description = session.getDescription();
109         String branding = session.getBranding();
110         if (exists(title))
111            w.oTag(3, "h1").append('>').append(title).eTag("h1").nl(3);
112         if (exists(description))
113            w.oTag(3, "h2").append('>').append(description).eTag("h2").nl(3);
114         if (exists(branding))
115            w.append(3, branding).nl(3);
116      }
117   }
118
119
120   @Override /* HtmlDocTemplate */
121   public void nav(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
122      String nav = session.getNav();
123      if (nav != null) {
124         if (exists(nav))
125            w.append(3, nav).nl(3);
126      } else {
127         String[] links = session.getLinks();
128         if (links.length > 0) {
129            w.sTag(3, "ol").nl(3);
130            for (String l : links) {
131               w.sTag(4, "li");
132               if (l.matches("(?s)\\S+\\:.*")) {
133                  int i = l.indexOf(':');
134                  String key = l.substring(0, i);
135                  String val = l.substring(i+1).trim();
136                  if (val.startsWith("<"))
137                     w.nl(4).appendln(5, val);
138                  else
139                     w.oTag("a").attr("href", session.resolveUri(val), true).cTag().text(key, true).eTag("a");
140                  w.eTag("li").nl(4);
141               } else {
142                  w.nl(4).appendln(5, l);
143                  w.eTag(4, "li").nl(4);
144               }
145            }
146            w.eTag(3, "ol").nl(3);
147         }
148      }
149   }
150
151   @Override /* HtmlDocTemplate */
152   public void aside(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
153      String aside = session.getAside();
154      if (exists(aside))
155         w.append(4, aside);
156   }
157
158   @Override /* HtmlDocTemplate */
159   public void article(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
160      // To allow for page formatting using CSS, we encapsulate the data inside two div tags:
161      // <div class='outerdata'><div class='data' id='data'>...</div></div>
162      w.oTag(4, "div").attr("class","outerdata").append('>').nl(4);
163      w.oTag(5, "div").attr("class","data").attr("id", "data").append('>').nl(5);
164
165      if (o == null) {
166         w.append(6, "<null/>").nl(6);
167      } else if (ObjectUtils.isEmpty(o)){
168         String m = session.getNoResultsMessage();
169         if (exists(m))
170            w.append(6, m).nl(6);
171      } else {
172         session.indent = 6;
173         s.parentSerialize(session, o);
174      }
175
176      w.ie(5).eTag("div").nl(5);
177      w.ie(4).eTag("div").nl(4);
178   }
179
180   @Override /* HtmlDocTemplate */
181   public void footer(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
182      String footer = session.getFooter();
183      if (exists(footer))
184         w.append(3, footer).nl(3);
185   }
186
187   @Override /* HtmlDocTemplate */
188   public boolean hasStyle(HtmlDocSerializerSession session) {
189      return true;
190   }
191
192   @Override /* HtmlDocTemplate */
193   public boolean hasScript(HtmlDocSerializerSession session) {
194      return true;
195   }
196
197   @Override /* HtmlDocTemplate */
198   public boolean hasHeader(HtmlDocSerializerSession session) {
199      return exists(session.getHeader()) || exists(session.getTitle()) || exists(session.getDescription());
200   }
201
202   @Override /* HtmlDocTemplate */
203   public boolean hasNav(HtmlDocSerializerSession session) {
204      return exists(session.getNav()) || session.getLinks().length > 0;
205   }
206
207   @Override /* HtmlDocTemplate */
208   public boolean hasAside(HtmlDocSerializerSession session) {
209      return exists(session.getAside());
210   }
211
212   @Override /* HtmlDocTemplate */
213   public boolean hasFooter(HtmlDocSerializerSession session) {
214      return exists(session.getFooter());
215   }
216
217   private static boolean exists(String s) {
218      return s != null && ! "NONE".equals(s);
219   }
