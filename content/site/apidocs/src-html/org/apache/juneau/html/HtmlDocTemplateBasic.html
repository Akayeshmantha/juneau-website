013package org.apache.juneau.html;
014
015import java.util.Map;
016
017import org.apache.juneau.internal.*;
018
019/**
020 * A basic template for the HTML doc serializer.
021 * <p>
022 * This class can be subclassed to customize page rendering.
023 */
024public class HtmlDocTemplateBasic implements HtmlDocTemplate {
025
026   @Override /* HtmlDocTemplate */
027   public void head(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
028      if (hasCss(session)) {
029         w.oTag(1, "style").attr("type", "text/css").appendln(">").nl(1);
030         css(session, w, s, o);
031         w.ie(1).eTag("style").nl(1);
032      }
033   }
034
035   @Override /* HtmlDocTemplate */
036   public void css(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
037
038      String cssUrl = session.getCssUrl();
039      if (cssUrl == null)
040         cssUrl = "servlet:/style.css";
041      cssUrl = session.resolveUri(cssUrl);
042
043      w.append(2, "@import ").q().append(cssUrl).q().appendln(";");
044      if (session.isNoWrap())
045         w.appendln("\n* {white-space:nowrap;}");
046      if (session.getCss() != null)
047         for (String css : session.getCss())
048            w.appendln(css);
049   }
050
051   @Override /* HtmlDocTemplate */
052   public void body(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
053
054      if (hasHeader(session)) {
055         w.sTag(1, "header").nl(1);
056         header(session, w, s, o);
057         w.ie(1).eTag("header").nl(1);
058      }
059
060      if (hasNav(session)) {
061         w.sTag(1, "nav").nl(1);
062         nav(session, w, s, o);
063         w.ie(1).eTag("nav").nl(1);
064      }
065
066      w.sTag(1, "section").nl(1);
067
068      w.sTag(2, "article").nl(2);
069      article(session, w, s, o);
070      w.ie(2).eTag("article").nl(2);
071
072      if (hasAside(session)) {
073         w.sTag(2, "aside").nl(2);
074         aside(session, w, s, o);
075         w.ie(2).eTag("aside").nl(2);
076      }
077
078      w.ie(1).eTag("section").nl(1);
079
080      if (hasFooter(session)) {
081         w.sTag(1, "footer").nl(1);
082         footer(session, w, s, o);
083         w.ie(1).eTag("footer").nl(1);
084      }
085   }
086
087   @Override /* HtmlDocTemplate */
088   public void header(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
089      // Write the title of the page.
090      String header = session.getHeader();
091      if (header != null) {
092         if (exists(header))
093            w.append(2, header).nl(2);
094      } else {
095         String title = session.getTitle();
096         String description = session.getDescription();
097         String branding = session.getBranding();
098         if (exists(title))
099            w.oTag(3, "h3").attr("class", "title").append('>').text(title).eTag("h3").nl(3);
100         if (exists(description))
101            w.oTag(3, "h5").attr("class", "description").append('>').text(description).eTag("h5").nl(3);
102         if (exists(branding))
103            w.append(3, branding).nl(3);
104      }
105   }
106
107
108   @Override /* HtmlDocTemplate */
109   public void nav(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
110      String nav = session.getNav();
111      if (nav != null) {
112         if (exists(nav))
113            w.append(2, nav).nl(2);
114      } else {
115         Map<String,String> htmlLinks = session.getLinks();
116         boolean first = true;
117         if (htmlLinks != null) {
118            for (Map.Entry<String,String> e : htmlLinks.entrySet()) {
119               if (! first)
120                  w.append(3, " - ").nl(3);
121               first = false;
122               w.oTag("a").attr("class", "link").attr("href", session.resolveUri(e.getValue()), true).cTag().text(e.getKey(), true).eTag("a");
123            }
124         }
125      }
126   }
127
128   @Override /* HtmlDocTemplate */
129   public void aside(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
130      String aside = session.getAside();
131      if (exists(aside))
132         w.append(aside);
133   }
134
135   @Override /* HtmlDocTemplate */
136   public void article(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
137      // To allow for page formatting using CSS, we encapsulate the data inside two div tags:
138      // <div class='outerdata'><div class='data' id='data'>...</div></div>
139      w.oTag(3, "div").attr("class","outerdata").append('>').nl(3);
140      w.oTag(4, "div").attr("class","data").attr("id", "data").append('>').nl(4);
141
142      if (o == null) {
143         w.append(5, "<null/>").nl(5);
144      } else if (ObjectUtils.isEmpty(o)){
145         String m = session.getNoResultsMessage();
146         if (exists(m))
147            w.append(5, m).nl(5);
148      } else {
149         s.parentSerialize(session, o);
150      }
151
152      w.ie(4).eTag("div").nl(4);
153      w.ie(4).eTag("div").nl(3);
154   }
155
156   @Override /* HtmlDocTemplate */
157   public void footer(HtmlDocSerializerSession session, HtmlWriter w, HtmlDocSerializer s, Object o) throws Exception {
158      String footer = session.getFooter();
159      if (exists(footer))
160         w.append(footer);
161   }
162
163   @Override /* HtmlDocTemplate */
164   public boolean hasCss(HtmlDocSerializerSession session) {
165      return true;
166   }
167
168   @Override /* HtmlDocTemplate */
169   public boolean hasHeader(HtmlDocSerializerSession session) {
170      return exists(session.getHeader()) || exists(session.getTitle()) || exists(session.getDescription());
171   }
172
173   @Override /* HtmlDocTemplate */
174   public boolean hasNav(HtmlDocSerializerSession session) {
175      return exists(session.getNav()) || session.getLinks() != null;
176   }
177
178   @Override /* HtmlDocTemplate */
179   public boolean hasAside(HtmlDocSerializerSession session) {
180      return exists(session.getAside());
181   }
182
183   @Override /* HtmlDocTemplate */
184   public boolean hasFooter(HtmlDocSerializerSession session) {
185      return exists(session.getFooter());
186   }
187
188   private static boolean exists(String s) {
189      return s != null && ! "NONE".equals(s);
190   }
191}
