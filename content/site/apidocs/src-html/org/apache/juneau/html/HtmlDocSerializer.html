024/**
025 * Serializes POJOs to HTTP responses as HTML documents.
026 *
027 * <h6 class='topic'>Media types</h6>
028 * <p>
029 *    Handles <code>Accept</code> types: <code>text/html</code>
030 * <p>
031 *    Produces <code>Content-Type</code> types: <code>text/html</code>
032 *
033 * <h6 class='topic'>Description</h6>
034 * <p>
035 *    Same as {@link HtmlSerializer}, except wraps the response in <code><xt>&lt;html&gt;</code>, <code><xt>&lt;head&gt;</code>,
036 *    and <code><xt>&lt;body&gt;</code> tags so that it can be rendered in a browser.
037 *
038 * <h6 class='topic'>Configurable properties</h6>
039 * <p>
040 *    This class has the following properties associated with it:
041 * <ul>
042 *    <li>{@link HtmlDocSerializerContext}
043 *    <li>{@link BeanContext}
044 * </ul>
045 */
046@Produces("text/html")
047@SuppressWarnings("hiding")
048public class HtmlDocSerializer extends HtmlStrippedDocSerializer {
049
050   // Properties defined in RestServletProperties
051   private static final String
052      REST_method = "RestServlet.method",
053      REST_relativeServletURI = "RestServlet.relativeServletURI";
054
055
056   /** Default serializer, all default settings. */
057   public static final HtmlDocSerializer DEFAULT = new HtmlDocSerializer().lock();
058
059   //--------------------------------------------------------------------------------
060   // Overridden methods
061   //--------------------------------------------------------------------------------

063   @Override /* Serializer */
064   public HtmlDocSerializerSession createSession(Object output, ObjectMap op, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType) {
065      return new HtmlDocSerializerSession(getContext(HtmlDocSerializerContext.class), op, output, javaMethod, locale, timeZone, mediaType);
066   }
067
068   @Override /* Serializer */
069   protected void doSerialize(SerializerSession session, Object o) throws Exception {
070
071      HtmlDocSerializerSession s = (HtmlDocSerializerSession)session;
072      HtmlWriter w = s.getWriter();
073
074      ObjectMap op = s.getProperties();
075
076      boolean isOptionsPage = op.containsKey(REST_method) && op.getString(REST_method).equalsIgnoreCase("OPTIONS");
077
078      // Render the header.
079      w.sTag("html").nl();
080      w.sTag("head").nl();
081
082      String cssUrl = s.getCssUrl();
083      if (cssUrl == null)
084         cssUrl = op.getString(REST_relativeServletURI) + "/style.css";
085
086      w.oTag(1, "style")
087         .attr("type", "text/css")
088         .appendln(">")
089         .append(2, "@import ").q().append(cssUrl).q().appendln(";");
090      if (s.isNoWrap())
091         w.appendln("\n* {white-space:nowrap;}");
092      if (s.getCssImports() != null)
093         for (String cssImport : s.getCssImports())
094            w.append(2, "@import ").q().append(cssImport).q().appendln(";");
095      w.eTag(1, "style").nl();
096      w.eTag("head").nl();
097      w.sTag("body").nl();
098      // Write the title of the page.
099      String title = s.getTitle();
100      if (title == null && isOptionsPage)
101         title = "Options";
102      String description = s.getDescription();
103      if (title != null)
104         w.oTag(1, "h3").attr("class", "title").append('>').text(title).eTag("h3").nl();
105      if (description != null)
106         w.oTag(1, "h5").attr("class", "description").append('>').text(description).eTag("h5").nl();
107
108      // Write the action links that render above the results.
109      List<Link> actions = new LinkedList<Link>();
110
111      // If this is an OPTIONS request, provide a 'back' link to return to the GET request page.
112      if (! isOptionsPage) {
113         Map<String,String> htmlLinks = s.getLinks();
114         if (htmlLinks != null) {
115            for (Map.Entry<String,String> e : htmlLinks.entrySet()) {
116               String uri = e.getValue();
117               if (uri.indexOf("://") == -1 && ! StringUtils.startsWith(uri, '/')) {
118                  StringBuilder sb = new StringBuilder(op.getString(REST_relativeServletURI));
119                  if (! (uri.isEmpty() || uri.charAt(0) == '?' || uri.charAt(0) == '/'))
120                     sb.append('/');
121                  sb.append(uri);
122                  uri = sb.toString();
123               }
124
125               actions.add(new Link(e.getKey(), uri));
126            }
127         }
128      }
129
130      if (actions.size() > 0) {
131         w.oTag(1, "p").attr("class", "links").append('>').nl();
132         for (Iterator<Link> i = actions.iterator(); i.hasNext();) {
133            Link h = i.next();
134            w.oTag(2, "a").attr("class", "link").attr("href", h.getHref(), true).append('>').append(h.getName()).eTag("a").nl();
135            if (i.hasNext())
136               w.append(3, " - ").nl();
137         }
138         w.eTag(1, "p").nl();
139      }
140
141      s.indent = 3;
142
143      // To allow for page formatting using CSS, we encapsulate the data inside two div tags:
144      // <div class='outerdata'><div class='data' id='data'>...</div></div>
145      w.oTag(1, "div").attr("class","outerdata").append('>').nl();
146      w.oTag(2, "div").attr("class","data").attr("id", "data").append('>').nl();
147      if (isEmptyList(o))
148         w.oTag(3, "p").append('>').append("no results").eTag("p");
149      else
150         super.doSerialize(s, o);
151      w.eTag(2, "div").nl();
152      w.eTag(1, "div").nl();
153
154      w.eTag("body").nl().eTag("html").nl();
155   }
156
157   private boolean isEmptyList(Object o) {
158      if (o == null)
159         return false;
160      if (o instanceof Collection && ((Collection<?>)o).size() == 0)
161         return true;
162      if (o.getClass().isArray() && Array.getLength(o) == 0)
163         return true;
164      return false;
165   }
