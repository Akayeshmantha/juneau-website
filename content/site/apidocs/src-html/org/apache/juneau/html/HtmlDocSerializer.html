013package org.apache.juneau.html;
014
015import java.lang.reflect.*;
016import java.util.*;
017
018import org.apache.juneau.*;
019import org.apache.juneau.annotation.*;
020import org.apache.juneau.http.*;
021import org.apache.juneau.serializer.*;
022
023/**
024 * Serializes POJOs to HTTP responses as HTML documents.
025 *
026 * <h5 class='section'>Media types:</h5>
027 * <p>
028 * Handles <code>Accept</code> types: <code>text/html</code>
029 * <p>
030 * Produces <code>Content-Type</code> types: <code>text/html</code>
031 *
032 * <h5 class='section'>Description:</h5>
033 * <p>
034 * Same as {@link HtmlSerializer}, except wraps the response in <code><xt>&lt;html&gt;</code>, <code><xt>&lt;head&gt;</code>,
035 *    and <code><xt>&lt;body&gt;</code> tags so that it can be rendered in a browser.
036 *
037 * <h5 class='section'>Configurable properties:</h5>
038 * <p>
039 * This class has the following properties associated with it:
040 * <ul>
041 *    <li>{@link HtmlDocSerializerContext}
042 *    <li>{@link BeanContext}
043 * </ul>
044 */
045@Produces("text/html")
046@SuppressWarnings("hiding")
047public class HtmlDocSerializer extends HtmlStrippedDocSerializer {
048
049   /** Default serializer, all default settings. */
050   public static final HtmlDocSerializer DEFAULT = new HtmlDocSerializer(PropertyStore.create());
051
052
053   final HtmlDocSerializerContext ctx;
054
055   /**
056    * Constructor.
057    * @param propertyStore The property store containing all the settings for this object.
058    */
059   public HtmlDocSerializer(PropertyStore propertyStore) {
060      super(propertyStore);
061      this.ctx = createContext(HtmlDocSerializerContext.class);
062   }
063
064
065   //--------------------------------------------------------------------------------
066   // Overridden methods
067   //--------------------------------------------------------------------------------
068
069   @Override /* Serializer */
070   public HtmlDocSerializerSession createSession(Object output, ObjectMap op, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType, UriContext uriContext) {
071      return new HtmlDocSerializerSession(ctx, op, output, javaMethod, locale, timeZone, mediaType, uriContext);
072   }
073
074   @Override /* Serializer */
075   protected void doSerialize(SerializerSession session, Object o) throws Exception {
076
077      HtmlDocSerializerSession s = (HtmlDocSerializerSession)session;
078      HtmlWriter w = s.getWriter();
079      HtmlDocTemplate t = s.getTemplate();
080
081      w.sTag("html").nl();
082      w.sTag(1, "head").nl();
083      t.head(s, w, this, o);
084      w.eTag(1, "head").nl();
085      w.sTag(1, "body").nl();
086      t.body(s, w, this, o);
087      w.eTag(1, "body").nl();
088      w.eTag("html").nl();
089   }
090
091   /**
092    * Calls the parent {@link #doSerialize(SerializerSession, Object)} method which invokes just the HTML serializer.
093    * @param session The serializer session.
094    * @param o The object being serialized.
095    * @throws Exception
096    */
097   public void parentSerialize(SerializerSession session, Object o) throws Exception {
098      super.doSerialize(session, o);
099   }
100}
