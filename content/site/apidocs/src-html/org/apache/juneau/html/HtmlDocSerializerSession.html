027/**
028 * Context object that lives for the duration of a single serialization of {@link HtmlSerializer} and its subclasses.
029 * <p>
030 * See {@link SerializerContext} for details.
031 * <p>
032 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
033 */
034public final class HtmlDocSerializerSession extends HtmlSerializerSession {
035
036   private final String title, description, branding, header, nav, aside, footer, cssUrl, noResultsMessage;
037   private final String[] css;
038   private final Map<String,Object> links;
039   private final boolean nowrap;
040   private final HtmlDocTemplate template;
041
042   /**
043    * Create a new session using properties specified in the context.
044    *
045    * @param ctx The context creating this session object.
046    *    The context contains all the configuration settings for this object.
047    * @param output The output object.  See {@link JsonSerializerSession#getWriter()} for valid class types.
048    * @param op The override properties.
049    *    These override any context properties defined in the context.
050    * @param javaMethod The java method that called this serializer, usually the method in a REST servlet.
051    * @param locale The session locale.
052    *    If <jk>null</jk>, then the locale defined on the context is used.
053    * @param timeZone The session timezone.
054    *    If <jk>null</jk>, then the timezone defined on the context is used.
055    * @param mediaType The session media type (e.g. <js>"application/json"</js>).
056    * @param uriContext The URI context.
057    *    Identifies the current request URI used for resolution of URIs to absolute or root-relative form.
058    */
059   @SuppressWarnings({ "unchecked", "rawtypes" })
060   protected HtmlDocSerializerSession(HtmlDocSerializerContext ctx, ObjectMap op, Object output, Method javaMethod, Locale locale, TimeZone timeZone, MediaType mediaType, UriContext uriContext) {
061      super(ctx, op, output, javaMethod, locale, timeZone, mediaType, uriContext);
062      if (op == null || op.isEmpty()) {
063         title = ctx.title;
064         description = ctx.description;
065         branding = ctx.branding;
066         header = ctx.header;
067         nav = ctx.nav;
068         aside = ctx.aside;
069         footer = ctx.footer;
070         links = ctx.links;
071         cssUrl = ctx.cssUrl;
072         css = ctx.css;
073         nowrap = ctx.nowrap;
074         noResultsMessage = ctx.noResultsMessage;
075         template = ClassUtils.newInstance(HtmlDocTemplate.class, ctx.template);
076      } else {
077         title = op.getString(HTMLDOC_title, ctx.title);
078         description = op.getString(HTMLDOC_description, ctx.description);
079         branding = op.getString(HTMLDOC_branding, ctx.branding);
080         header = op.getString(HTMLDOC_header, ctx.nav);
081         nav = op.getString(HTMLDOC_nav, ctx.nav);
082         aside = op.getString(HTMLDOC_aside, ctx.aside);
083         footer = op.getString(HTMLDOC_footer, ctx.footer);
084         Map m = op.getMap(HTMLDOC_links, ctx.links);
085         links = ObjectUtils.isEmpty(m) ? null : new LinkedHashMap(m);
086         cssUrl = op.getString(HTMLDOC_cssUrl, ctx.cssUrl);
087         css = split(op.getString(HTMLDOC_css, null));
088         nowrap = op.getBoolean(HTMLDOC_nowrap, ctx.nowrap);
089         noResultsMessage = op.getString(HTMLDOC_noResultsMessage, ctx.noResultsMessage);
090         template = ClassUtils.newInstance(HtmlDocTemplate.class, op.get(HTMLDOC_template, ctx.template));
091      }
092   }
093
094   /**
095    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_cssUrl} setting value in this context.
096    * @return The {@link HtmlDocSerializerContext#HTMLDOC_cssUrl} setting value in this context.
097    *    <jk>null</jk> if not specified.  Never an empty string.
098    */
099   public final String getCssUrl() {
100      return cssUrl;
101   }
102
103   /**
104    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_css} setting value in this context.
105    * @return The {@link HtmlDocSerializerContext#HTMLDOC_css} setting value in this context.
106    *    <jk>null</jk> if not specified.  Never an empty array.
107    */
108   public final String[] getCss() {
109      return css;
110   }
111
112   /**
113    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_nowrap} setting value in this context.
114    * @return The {@link HtmlDocSerializerContext#HTMLDOC_nowrap} setting value in this context.
115    */
116   public final boolean isNoWrap() {
117      return nowrap;
118   }
119
120   /**
121    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_title} setting value in this context.
122    * @return The {@link HtmlDocSerializerContext#HTMLDOC_title} setting value in this context.
123    *    <jk>null</jk> if not specified.  Never an empty string.
124    */
125   public final String getTitle() {
126      return title;
127   }
128
129   /**
130    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_description} setting value in this context.
131    * @return The {@link HtmlDocSerializerContext#HTMLDOC_description} setting value in this context.
132    *    <jk>null</jk> if not specified.  Never an empty string.
133    */
134   public final String getDescription() {
135      return description;
136   }
137
138   /**
139    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_branding} setting value in this context.
140    * @return The {@link HtmlDocSerializerContext#HTMLDOC_branding} setting value in this context.
141    *    <jk>null</jk> if not specified.  Never an empty string.
142    */
143   public final String getBranding() {
144      return branding;
145   }
146
147   /**
148    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_header} setting value in this context.
149    * @return The {@link HtmlDocSerializerContext#HTMLDOC_header} setting value in this context.
150    *    <jk>null</jk> if not specified.  Never an empty string.
151    */
152   public final String getHeader() {
153      return header;
154   }
155
156   /**
157    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_links} setting value in this context.
158    * @return The {@link HtmlDocSerializerContext#HTMLDOC_links} setting value in this context.
159    *    <jk>null</jk> if not specified.  Never an empty map.
160    */
161   public final Map<String,Object> getLinks() {
162      return links;
163   }
164
165   /**
166    * Returns the template to use for generating the HTML page.
167    * @return The HTML page generator.
168    *    Never <jk>null</jk>.
169    */
170   public final HtmlDocTemplate getTemplate() {
171      return template;
172   }
173
174   /**
175    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_nav} setting value in this context.
176    * @return The {@link HtmlDocSerializerContext#HTMLDOC_nav} setting value in this context.
177    *    <jk>null</jk> if not specified.  Never an empty string.
178    */
179   public final String getNav() {
180      return nav;
181   }
182
183   /**
184    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_aside} setting value in this context.
185    * @return The {@link HtmlDocSerializerContext#HTMLDOC_aside} setting value in this context.
186    *    <jk>null</jk> if not specified.  Never an empty string.
187    */
188   public final String getAside() {
189      return aside;
190   }
191
192   /**
193    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_footer} setting value in this context.
194    * @return The {@link HtmlDocSerializerContext#HTMLDOC_footer} setting value in this context.
195    *    <jk>null</jk> if not specified.  Never an empty string.
196    */
197   public final String getFooter() {
198      return footer;
199   }
200
201   /**
202    * Returns the {@link HtmlDocSerializerContext#HTMLDOC_noResultsMessage} setting value in this context.
203    * @return The {@link HtmlDocSerializerContext#HTMLDOC_noResultsMessage} setting value in this context.
204    *    <jk>null</jk> if not specified.  Never an empty string.
205    */
206   public final String getNoResultsMessage() {
207      return noResultsMessage;
208   }
209
210   @Override /* XmlSerializerSession */
211   public HtmlWriter getWriter() throws Exception {
212      Object output = getOutput();
213      if (output instanceof HtmlWriter)
214         return (HtmlWriter)output;
215      return new HtmlWriter(super.getWriter(), isUseWhitespace(), getMaxIndent(), isTrimStrings(), getQuoteChar(), getUriResolver());
216   }
