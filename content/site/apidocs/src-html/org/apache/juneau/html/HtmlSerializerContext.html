013package org.apache.juneau.html;
014
015import org.apache.juneau.*;
016import org.apache.juneau.xml.*;
017
018/**
019 * Configurable properties on the {@link HtmlSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * See {@link ContextFactory} for more information about context properties.
025 *
026 * <h5 class='section'>Inherited configurable properties:</h5>
027 * <ul class='javahierarchy'>
028 *    <li class='c'><a class="doclink" href="../BeanContext.html#ConfigProperties">BeanContext</a> - Properties associated with handling beans on serializers and parsers.
029 *    <ul>
030 *       <li class='c'><a class="doclink" href="../serializer/SerializerContext.html#ConfigProperties">SerializerContext</a> - Configurable properties common to all serializers.
031 *    </ul>
032 * </ul>
033 */
034public class HtmlSerializerContext extends XmlSerializerContext {
035
036   /**
037    * <b>Configuration property:</b>  Anchor text source.
038    * <p>
039    * <ul>
040    *    <li><b>Name:</b> <js>"HtmlSerializer.uriAnchorText"</js>
041    *    <li><b>Data type:</b> <code>String</code>
042    *    <li><b>Default:</b> <js>"toString"</js>
043    *    <li><b>Session-overridable:</b> <jk>true</jk>
044    * </ul>
045    * <p>
046    * When creating anchor tags (e.g. <code><xt>&lt;a</xt> <xa>href</xa>=<xs>'...'</xs><xt>&gt;</xt>text<xt>&lt;/a&gt;</xt></code>)
047    *    in HTML, this setting defines what to set the inner text to.
048    * <p>
049    * Possible values:
050    * <ul class='spaced-list'>
051    *    <li>{@link #TO_STRING} / <js>"toString"</js> - Set to whatever is returned by {@link #toString()} on the object.
052    *    <li>{@link #URI} / <js>"uri"</js> - Set to the URI value.
053    *    <li>{@link #LAST_TOKEN} / <js>"lastToken"</js> - Set to the last token of the URI value.
054    *    <li>{@link #PROPERTY_NAME} / <js>"propertyName"</js> - Set to the bean property name.
055    *    <li>{@link #URI_ANCHOR} / <js>"uriAnchor"</js> - Set to the anchor of the URL.  (e.g. <js>"http://localhost:9080/foobar#anchorTextHere"</js>)
056    * </ul>
057    */
058   public static final String HTML_uriAnchorText = "HtmlSerializer.uriAnchorText";
059
060   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
061   public static final String PROPERTY_NAME = "PROPERTY_NAME";
062   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
063   public static final String TO_STRING = "TO_STRING";
064   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
065   public static final String URI = "URI";
066   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
067   public static final String LAST_TOKEN = "LAST_TOKEN";
068   /** Constant for {@link HtmlSerializerContext#HTML_uriAnchorText} property. */
069   public static final String URI_ANCHOR = "URI_ANCHOR";
070
071
072   /**
073    * <b>Configuration property:</b>  Look for URLs in {@link String Strings}.
074    * <p>
075    * <ul>
076    *    <li><b>Name:</b> <js>"HtmlSerializer.detectLinksInStrings"</js>
077    *    <li><b>Data type:</b> <code>Boolean</code>
078    *    <li><b>Default:</b> <jk>true</jk>
079    *    <li><b>Session-overridable:</b> <jk>true</jk>
080    * </ul>
081    * <p>
082    * If a string looks like a URL (e.g. starts with <js>"http://"</js> or <js>"https://"</js>, then treat it like a URL
083    *    and make it into a hyperlink based on the rules specified by {@link #HTML_uriAnchorText}.
084    */
085   public static final String HTML_detectLinksInStrings = "HtmlSerializer.detectLinksInStrings";
086
087   /**
088    * <b>Configuration property:</b>  Look for link labels in the <js>"label"</js> parameter of the URL.
089    * <p>
090    * <ul>
091    *    <li><b>Name:</b> <js>"HtmlSerializer.lookForLabelParameters"</js>
092    *    <li><b>Data type:</b> <code>Boolean</code>
093    *    <li><b>Default:</b> <jk>true</jk>
094    *    <li><b>Session-overridable:</b> <jk>true</jk>
095    * </ul>
096    * <p>
097    * If the URL has a label parameter (e.g. <js>"?label=foobar"</js>), then use that as the anchor text of the link.
098    * <p>
099    * The parameter name can be changed via the {@link #HTML_labelParameter} property.
100    */
101   public static final String HTML_lookForLabelParameters = "HtmlSerializer.lookForLabelParameters";
102
103   /**
104    * <b>Configuration property:</b>  The parameter name to use when using {@link #HTML_lookForLabelParameters}.
105    * <p>
106    * <ul>
107    *    <li><b>Name:</b> <js>"HtmlSerializer.labelParameter"</js>
108    *    <li><b>Data type:</b> <code>String</code>
109    *    <li><b>Default:</b> <js>"label"</js>
110    *    <li><b>Session-overridable:</b> <jk>true</jk>
111    * </ul>
112    * <p>
113    */
114   public static final String HTML_labelParameter = "HtmlSerializer.labelParameter";
115
116   /**
117    * <b>Configuration property:</b>  Add key/value headers on bean/map tables.
118    * <p>
119    * <ul>
120    *    <li><b>Name:</b> <js>"HtmlSerializer.addKeyValueTableHeaders"</js>
121    *    <li><b>Data type:</b> <code>Boolean</code>
122    *    <li><b>Default:</b> <jk>false</jk>
123    *    <li><b>Session-overridable:</b> <jk>true</jk>
124    * </ul>
125    * <p>
126    */
127   public static final String HTML_addKeyValueTableHeaders = "HtmlSerializer.addKeyValueTableHeaders";
128
129   final String uriAnchorText;
130   final boolean lookForLabelParameters, detectLinksInStrings, addKeyValueTableHeaders;
131   final String labelParameter;
132
133   /**
134    * Constructor.
135    * <p>
136    * Typically only called from {@link ContextFactory#getContext(Class)}.
137    *
138    * @param cf The factory that created this context.
139    */
140   public HtmlSerializerContext(ContextFactory cf) {
141      super(cf);
142      uriAnchorText = cf.getProperty(HTML_uriAnchorText, String.class, TO_STRING);
143      lookForLabelParameters = cf.getProperty(HTML_lookForLabelParameters, Boolean.class, true);
144      detectLinksInStrings = cf.getProperty(HTML_detectLinksInStrings, Boolean.class, true);
145      labelParameter = cf.getProperty(HTML_labelParameter, String.class, "label");
146      addKeyValueTableHeaders = cf.getProperty(HTML_addKeyValueTableHeaders, Boolean.class, false);
147   }
148
149   @Override /* Context */
150   public ObjectMap asMap() {
151      return super.asMap()
152         .append("HtmlSerializerContext", new ObjectMap()
153            .append("uriAnchorText", uriAnchorText)
154            .append("lookForLabelParameters", lookForLabelParameters)
155            .append("detectLinksInStrings", detectLinksInStrings)
156            .append("labelParameter", labelParameter)
157            .append("addKeyValueTableHeaders", addKeyValueTableHeaders)
158         );
159   }
160}
