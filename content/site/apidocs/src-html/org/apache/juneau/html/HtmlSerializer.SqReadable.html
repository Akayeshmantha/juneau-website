029/**
030 * Serializes POJO models to HTML.
031 *
032 * <h5 class='section'>Media types:</h5>
033 * <p>
034 * Handles <code>Accept</code> types: <code>text/html</code>
035 * <p>
036 * Produces <code>Content-Type</code> types: <code>text/html</code>
037 *
038 * <h5 class='section'>Description:</h5>
039 * <p>
040 * The conversion is as follows...
041 *    <ul class='spaced-list'>
042 *       <li>{@link Map Maps} (e.g. {@link HashMap}, {@link TreeMap}) and beans are converted to HTML tables with 'key' and 'value' columns.
043 *       <li>{@link Collection Collections} (e.g. {@link HashSet}, {@link LinkedList}) and Java arrays are converted to HTML ordered lists.
044 *       <li>{@code Collections} of {@code Maps} and beans are converted to HTML tables with keys as headers.
045 *       <li>Everything else is converted to text.
046 *    </ul>
047 * <p>
048 * This serializer provides several serialization options.  Typically, one of the predefined <jsf>DEFAULT</jsf> serializers will be sufficient.
049 * However, custom serializers can be constructed to fine-tune behavior.
050 * <p>
051 * The {@link HtmlLink} annotation can be used on beans to add hyperlinks to the output.
052 *
053 * <h5 class='section'>Configurable properties:</h5>
054 * <p>
055 * This class has the following properties associated with it:
056 * <ul class='spaced-list'>
057 *    <li>{@link HtmlSerializerContext}
058 * </ul>
059 *
060 * <h6 class='topic'>Behavior-specific subclasses</h6>
061 * <p>
062 * The following direct subclasses are provided for convenience:
063 * <ul class='spaced-list'>
064 *    <li>{@link Sq} - Default serializer, single quotes.
065 *    <li>{@link SqReadable} - Default serializer, single quotes, whitespace added.
066 * </ul>
067 *
068 * <h5 class='section'>Example:</h5>
069 * <p class='bcode'>
070 *    <jc>// Use one of the default serializers to serialize a POJO</jc>
071 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(someObject);
072 *
073 *       <jc>// Create a custom serializer that doesn't use whitespace and newlines</jc>
074 *       HtmlSerializer serializer = <jk>new</jk> HtmlSerializerBuider().ws().build();
075 *
076 *       <jc>// Same as above, except uses cloning</jc>
077 *       HtmlSerializer serializer = HtmlSerializer.<jsf>DEFAULT</jsf>.builder().ws().build();
078 *
079 *       <jc>// Serialize POJOs to HTML</jc>
080 *
081 *       <jc>// Produces: </jc>
082 *       <jc>// &lt;ul&gt;&lt;li&gt;1&lt;li&gt;2&lt;li&gt;3&lt;/ul&gt;</jc>
083 *       List l = new ObjectList(1, 2, 3);
084 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(l);
085 *
086 *       <jc>// Produces: </jc>
087 *       <jc>//    &lt;table&gt; </jc>
088 *       <jc>//       &lt;tr&gt;&lt;th&gt;firstName&lt;/th&gt;&lt;th&gt;lastName&lt;/th&gt;&lt;/tr&gt; </jc>
089 *       <jc>//       &lt;tr&gt;&lt;td&gt;Bob&lt;/td&gt;&lt;td&gt;Costas&lt;/td&gt;&lt;/tr&gt; </jc>
090 *       <jc>//       &lt;tr&gt;&lt;td&gt;Billy&lt;/td&gt;&lt;td&gt;TheKid&lt;/td&gt;&lt;/tr&gt; </jc>
091 *       <jc>//       &lt;tr&gt;&lt;td&gt;Barney&lt;/td&gt;&lt;td&gt;Miller&lt;/td&gt;&lt;/tr&gt; </jc>
092 *       <jc>//    &lt;/table&gt; </jc>
093 *       l = <jk>new</jk> ObjectList();
094 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Bob',lastName:'Costas'}"</js>));
095 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Billy',lastName:'TheKid'}"</js>));
096 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Barney',lastName:'Miller'}"</js>));
097 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(l);
098 *
099 *       <jc>// Produces: </jc>
100 *       <jc>//    &lt;table&gt; </jc>
101 *       <jc>//       &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
102 *       <jc>//       &lt;tr&gt;&lt;td&gt;foo&lt;/td&gt;&lt;td&gt;bar&lt;/td&gt;&lt;/tr&gt; </jc>
103 *       <jc>//       &lt;tr&gt;&lt;td&gt;baz&lt;/td&gt;&lt;td&gt;123&lt;/td&gt;&lt;/tr&gt; </jc>
104 *       <jc>//    &lt;/table&gt; </jc>
105 *       Map m = <jk>new</jk> ObjectMap(<js>"{foo:'bar',baz:123}"</js>);
106 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(m);
107 *
108 *       <jc>// HTML elements can be nested arbitrarily deep</jc>
109 *       <jc>// Produces: </jc>
110 *       <jc>//   &lt;table&gt; </jc>
111 *       <jc>//      &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
112 *       <jc>//      &lt;tr&gt;&lt;td&gt;foo&lt;/td&gt;&lt;td&gt;bar&lt;/td&gt;&lt;/tr&gt; </jc>
113 *       <jc>//      &lt;tr&gt;&lt;td&gt;baz&lt;/td&gt;&lt;td&gt;123&lt;/td&gt;&lt;/tr&gt; </jc>
114 *       <jc>//      &lt;tr&gt;&lt;td&gt;someNumbers&lt;/td&gt;&lt;td&gt;&lt;ul&gt;&lt;li&gt;1&lt;li&gt;2&lt;li&gt;3&lt;/ul&gt;&lt;/td&gt;&lt;/tr&gt; </jc>
115 *       <jc>//      &lt;tr&gt;&lt;td&gt;someSubMap&lt;/td&gt;&lt;td&gt; </jc>
116 *       <jc>//         &lt;table&gt; </jc>
117 *       <jc>//            &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
118 *       <jc>//            &lt;tr&gt;&lt;td&gt;a&lt;/td&gt;&lt;td&gt;b&lt;/td&gt;&lt;/tr&gt; </jc>
119 *       <jc>//         &lt;/table&gt; </jc>
120 *       <jc>//      &lt;/td&gt;&lt;/tr&gt; </jc>
121 *       <jc>//   &lt;/table&gt; </jc>
122 *       Map m = <jk>new</jk> ObjectMap(<js>"{foo:'bar',baz:123}"</js>);
123 *       m.put("someNumbers", new ObjectList(1, 2, 3));
124 *       m.put(<js>"someSubMap"</js>, new ObjectMap(<js>"{a:'b'}"</js>));
125 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(m);
126 * </p>
127 */

195   /**
196    * Main serialization routine.
197    * @param session The serialization context object.
198    * @param o The object being serialized.
199    * @param w The writer to serialize to.
200    *
201    * @return The same writer passed in.
202    * @throws IOException If a problem occurred trying to send output to the writer.
203    */
204   private HtmlWriter doSerialize(HtmlSerializerSession session, Object o, HtmlWriter w) throws Exception {
205      serializeAnything(session, w, o, null, null, session.getInitialDepth()-1, null, true);
206      return w;
207   }

209   /**
210    * Serialize the specified object to the specified writer.
211    *
212    * @param session The context object that lives for the duration of this serialization.
213    * @param out The writer.
214    * @param o The object to serialize.
215    * @param eType The expected type of the object if this is a bean property.
216    * @param name The attribute name of this object if this object was a field in a JSON object (i.e. key of a {@link java.util.Map.Entry} or property name of a bean).
217    * @param indent The current indentation value.
218    * @param pMeta The bean property being serialized, or <jk>null</jk> if we're not serializing a bean property.
219    * @param isRoot <jk>true</jk> if this is the root element of the document.
220    * @return The type of content encountered.  Either simple (no whitespace) or normal (elements with whitespace).
221    * @throws Exception If a problem occurred trying to convert the output.
222    */
223   @SuppressWarnings({ "rawtypes", "unchecked" })
224   protected ContentResult serializeAnything(HtmlSerializerSession session, HtmlWriter out, Object o, ClassMeta<?> eType, String name, int indent, BeanPropertyMeta pMeta, boolean isRoot) throws Exception {
225
226      ClassMeta<?> aType = null;       // The actual type
227      ClassMeta<?> wType = null;     // The wrapped type (delegate)
228      ClassMeta<?> sType = object();   // The serialized type
229
230      if (eType == null)
231         eType = object();
232
233      aType = session.push(name, o, eType);
234
235      // Handle recursion
236      if (aType == null) {
237         o = null;
238         aType = object();
239      }
240
241      session.indent += indent;
242
243      ContentResult cr = CR_NORMAL;
244
245      // Determine the type.
246      if (o == null || (aType.isChar() && ((Character)o).charValue() == 0)) {
247         out.tag("null");
248         cr = ContentResult.CR_SIMPLE;
249
250      } else {
251
252         if (aType.isDelegate()) {
253            wType = aType;
254            aType = ((Delegate)o).getClassMeta();
255         }
256
257         sType = aType.getSerializedClassMeta();
258         String typeName = null;
259         if (session.isAddBeanTypeProperties() && ! eType.equals(aType))
260            typeName = aType.getResolvedDictionaryName();
261
262         // Swap if necessary
263         PojoSwap swap = aType.getPojoSwap();
264         if (swap != null) {
265            o = swap.swap(session, o);
266
267            // If the getSwapClass() method returns Object, we need to figure out
268            // the actual type now.
269            if (sType.isObject())
270               sType = session.getClassMetaForObject(o);
271         }
272
273         HtmlClassMeta html = sType.getExtendedMeta(HtmlClassMeta.class);
274
275         if (html.isAsXml() || (pMeta != null && pMeta.getExtendedMeta(HtmlBeanPropertyMeta.class).isAsXml())) {
276            super.serializeAnything(session, out, o, null, null, null, false, XmlFormat.MIXED, false, false, null);
277
278         } else if (html.isAsPlainText() || (pMeta != null && pMeta.getExtendedMeta(HtmlBeanPropertyMeta.class).isAsPlainText())) {
279            out.write(o == null ? "null" : o.toString());
280            cr = CR_SIMPLE;
281
282         } else if (o == null || (sType.isChar() && ((Character)o).charValue() == 0)) {
283            out.tag("null");
284            cr = CR_SIMPLE;
285
286         } else if (sType.isNumber()) {
287            if (eType.isNumber())
288               out.append(o);
289            else
290               out.sTag("number").append(o).eTag("number");
291            cr = CR_SIMPLE;
292
293         } else if (sType.isBoolean()) {
294            if (eType.isBoolean())
295               out.append(o);
296            else
297               out.sTag("boolean").append(o).eTag("boolean");
298            cr = CR_SIMPLE;
299
300         } else if (sType.isMap() || (wType != null && wType.isMap())) {
301            out.nlIf(! isRoot);
302            if (o instanceof BeanMap)
303               serializeBeanMap(session, out, (BeanMap)o, eType, pMeta);
304            else
305               serializeMap(session, out, (Map)o, sType, eType.getKeyType(), eType.getValueType(), typeName, pMeta);
306
307         } else if (sType.isBean()) {
308            BeanMap m = session.toBeanMap(o);
309            Class<?> c = o.getClass();
310            if (c.isAnnotationPresent(HtmlLink.class)) {
311               HtmlLink h = o.getClass().getAnnotation(HtmlLink.class);
312               Object urlProp = m.get(h.hrefProperty());
313               Object nameProp = m.get(h.nameProperty());
314               out.oTag("a").attrUri("href", urlProp).append('>').text(nameProp).eTag("a");
315               cr = CR_SIMPLE;
316            } else {
317               out.nlIf(! isRoot);
318               serializeBeanMap(session, out, m, eType, pMeta);
319            }
320
321         } else if (sType.isCollection() || sType.isArray() || (wType != null && wType.isCollection())) {
322            out.nlIf(! isRoot);
323            serializeCollection(session, out, o, sType, eType, name, pMeta);
324
325         } else if (session.isUri(sType, pMeta, o)) {
326            String label = session.getAnchorText(pMeta, o);
327            out.oTag("a").attrUri("href", o).append('>');
328            out.append(label);
329            out.eTag("a");
330            cr = CR_SIMPLE;
331
332         } else {
333            if (isRoot)
334               out.sTag("string").text(session.toString(o)).eTag("string");
335            else
336               out.text(session.toString(o));
337            cr = CR_SIMPLE;
338         }
339      }
340      session.pop();
341      session.indent -= indent;
342      return cr;
343   }

345   /**
346    * Identifies what the contents were of a serialized bean.
347    */
348   static enum ContentResult {
349      CR_SIMPLE,    // Simple content.  Shouldn't use whitespace.
350      CR_NORMAL     // Normal content.  Use whitespace.
351   }

353   @SuppressWarnings({ "rawtypes", "unchecked" })
354   private void serializeMap(HtmlSerializerSession session, HtmlWriter out, Map m, ClassMeta<?> sType, ClassMeta<?> eKeyType, ClassMeta<?> eValueType, String typeName, BeanPropertyMeta ppMeta) throws Exception {
355
356      ClassMeta<?> keyType = eKeyType == null ? sType.getKeyType() : eKeyType;
357      ClassMeta<?> valueType = eValueType == null ? sType.getValueType() : eValueType;
358      ClassMeta<?> aType = session.getClassMetaForObject(m);       // The actual type
359
360      int i = session.getIndent();
361
362      out.oTag(i, "table");
363
364      if (typeName != null && ppMeta != null && ppMeta.getClassMeta() != aType)
365         out.attr(session.getBeanTypePropertyName(), typeName);
366
367      out.appendln(">");
368      if (session.isAddKeyValueTableHeaders() && ! (aType.getExtendedMeta(HtmlClassMeta.class).isNoTableHeaders() || (ppMeta != null && ppMeta.getExtendedMeta(HtmlBeanPropertyMeta.class).isNoTableHeaders()))) {
369         out.sTag(i+1, "tr").nl();
370         out.sTag(i+2, "th").append("key").eTag("th").nl();
371         out.sTag(i+2, "th").append("value").eTag("th").nl();
372         out.eTag(i+1, "tr").nl();
373      }
374      for (Map.Entry e : (Set<Map.Entry>)m.entrySet()) {
375
376         Object key = session.generalize(e.getKey(), keyType);
377         Object value = null;
378         try {
379            value = e.getValue();
380         } catch (StackOverflowError t) {
381            throw t;
382         } catch (Throwable t) {
383            session.addWarning("Could not call getValue() on property ''{0}'', {1}", e.getKey(), t.getLocalizedMessage());
384         }
385
386         out.sTag(i+1, "tr").nl();
387         out.sTag(i+2, "td");
388         ContentResult cr = serializeAnything(session, out, key, keyType, null, 2, null, false);
389         if (cr == CR_NORMAL)
390            out.i(i+2);
391         out.eTag("td").nl();
392         out.sTag(i+2, "td");
393         cr = serializeAnything(session, out, value, valueType, (key == null ? "_x0000_" : key.toString()), 2, null, false);
394         if (cr == CR_NORMAL)
395            out.i(i+2);
396         out.eTag("td").nl();
397         out.eTag(i+1, "tr").nl();
398      }
399      out.eTag(i, "table").nl();
400   }

402   private void serializeBeanMap(HtmlSerializerSession session, HtmlWriter out, BeanMap<?> m, ClassMeta<?> eType, BeanPropertyMeta ppMeta) throws Exception {
403      int i = session.getIndent();
404
405      out.oTag(i, "table");
406
407      String typeName = m.getMeta().getDictionaryName();
408      if (typeName != null && eType != m.getClassMeta())
409         out.attr(session.getBeanTypePropertyName(), typeName);
410
411      out.append('>').nl();
412      if (session.isAddKeyValueTableHeaders() && ! (m.getClassMeta().getExtendedMeta(HtmlClassMeta.class).isNoTableHeaders() || (ppMeta != null && ppMeta.getExtendedMeta(HtmlBeanPropertyMeta.class).isNoTableHeaders()))) {
413         out.sTag(i+1, "tr").nl();
414         out.sTag(i+2, "th").append("key").eTag("th").nl();
415         out.sTag(i+2, "th").append("value").eTag("th").nl();
416         out.eTag(i+1, "tr").nl();
417      }
418
419      for (BeanPropertyValue p : m.getValues(session.isTrimNulls())) {
420         BeanPropertyMeta pMeta = p.getMeta();
421         ClassMeta<?> cMeta = p.getClassMeta();
422
423         String key = p.getName();
424         Object value = p.getValue();
425         Throwable t = p.getThrown();
426         if (t != null)
427            session.addBeanGetterWarning(pMeta, t);
428
429         if (session.canIgnoreValue(cMeta, key, value))
430            continue;
431
432         out.sTag(i+1, "tr").nl();
433         out.sTag(i+2, "td").text(key).eTag("td").nl();
434         out.sTag(i+2, "td");
435         try {
436            ContentResult cr = serializeAnything(session, out, value, cMeta, key, 2, pMeta, false);
437            if (cr == CR_NORMAL)
438               out.i(i+2);
439         } catch (SerializeException e) {
440            throw e;
441         } catch (Error e) {
442            throw e;
443         } catch (Throwable e) {
444            e.printStackTrace();
445            session.addBeanGetterWarning(pMeta, e);
446         }
447         out.eTag("td").nl();
448         out.eTag(i+1, "tr").nl();
449      }
450      out.eTag(i, "table").nl();
451   }

453   @SuppressWarnings({ "rawtypes", "unchecked" })
454   private void serializeCollection(HtmlSerializerSession session, HtmlWriter out, Object in, ClassMeta<?> sType, ClassMeta<?> eType, String name, BeanPropertyMeta ppMeta) throws Exception {
455
456      ClassMeta<?> seType = sType.getElementType();
457      if (seType == null)
458         seType = session.object();
459
460      Collection c = (sType.isCollection() ? (Collection)in : toList(sType.getInnerClass(), in));
461
462      int i = session.getIndent();
463      if (c.isEmpty()) {
464         out.appendln(i, "<ul></ul>");
465         return;
466      }
467
468      String type2 = null;
469      if (sType != eType)
470         type2 = sType.getResolvedDictionaryName();
471      if (type2 == null)
472         type2 = "array";
473
474      c = session.sort(c);
475
476      HtmlBeanPropertyMeta hbpMeta = (ppMeta == null ? null : ppMeta.getExtendedMeta(HtmlBeanPropertyMeta.class));
477      String btpn = session.getBeanTypePropertyName();
478
479      // Look at the objects to see how we're going to handle them.  Check the first object to see how we're going to handle this.
480      // If it's a map or bean, then we'll create a table.
481      // Otherwise, we'll create a list.
482      String[] th = getTableHeaders(session, c, hbpMeta);
483
484      if (th != null) {
485
486         out.oTag(i, "table").attr(btpn, type2).append('>').nl();
487         out.sTag(i+1, "tr").nl();
488         for (String key : th)
489            out.sTag(i+2, "th").append(key).eTag("th").nl();
490         out.eTag(i+1, "tr").nl();
491
492         for (Object o : c) {
493            ClassMeta<?> cm = session.getClassMetaForObject(o);
494
495            if (cm != null && cm.getPojoSwap() != null) {
496               PojoSwap f = cm.getPojoSwap();
497               o = f.swap(session, o);
498               cm = cm.getSerializedClassMeta();
499            }
500
501            out.oTag(i+1, "tr");
502            String typeName = (cm == null ? null : cm.getDictionaryName());
503            if (typeName != null && sType.getElementType() != cm)
504               out.attr(btpn, typeName);
505            out.cTag().nl();
506
507            if (cm == null) {
508               serializeAnything(session, out, o, null, null, 1, null, false);
509
510            } else if (cm.isMap() && ! (cm.isBeanMap())) {
511               Map m2 = session.sort((Map)o);
512
513               for (String k : th) {
514                  out.sTag(i+2, "td");
515                  ContentResult cr = serializeAnything(session, out, m2.get(k), seType, k, 2, null, false);
516                  if (cr == CR_NORMAL)
517                     out.i(i+2);
518                  out.eTag("td").nl();
519               }
520            } else {
521               BeanMap m2 = null;
522               if (o instanceof BeanMap)
523                  m2 = (BeanMap)o;
524               else
525                  m2 = session.toBeanMap(o);
526
527               for (String k : th) {
528                  BeanMapEntry p = m2.getProperty(k);
529                  BeanPropertyMeta pMeta = p.getMeta();
530                  out.sTag(i+2, "td");
531                  ContentResult cr = serializeAnything(session, out, p.getValue(), pMeta.getClassMeta(), p.getKey().toString(), 2, pMeta, false);
532                  if (cr == CR_NORMAL)
533                     out.i(i+2);
534                  out.eTag("td").nl();
535               }
536            }
537            out.eTag(i+1, "tr").nl();
538         }
539         out.eTag(i, "table").nl();
540

557   /*
558    * Returns the table column headers for the specified collection of objects.
559    * Returns null if collection should not be serialized as a 2-dimensional table.
560    * 2-dimensional tables are used for collections of objects that all have the same set of property names.
561    */
562   @SuppressWarnings({ "rawtypes", "unchecked" })
563   private static String[] getTableHeaders(SerializerSession session, Collection c, HtmlBeanPropertyMeta hbpMeta) throws Exception {
564      if (c.size() == 0)
565         return null;
566      c = session.sort(c);
567      String[] th;
568      Set<String> s = new TreeSet<String>();
569      Set<ClassMeta> prevC = new HashSet<ClassMeta>();
570      Object o1 = null;
571      for (Object o : c)
572         if (o != null) {
573            o1 = o;
574            break;
575         }
576      if (o1 == null)
577         return null;
578      ClassMeta<?> cm = session.getClassMetaForObject(o1);
579      if (cm.getPojoSwap() != null) {
580         PojoSwap f = cm.getPojoSwap();
581         o1 = f.swap(session, o1);
582         cm = cm.getSerializedClassMeta();
583      }
584      if (cm == null || ! cm.isMapOrBean())
585         return null;
586      if (cm.getInnerClass().isAnnotationPresent(HtmlLink.class))
587         return null;
588      HtmlClassMeta h = cm.getExtendedMeta(HtmlClassMeta.class);
589      if (h.isNoTables() || (hbpMeta != null && hbpMeta.isNoTables()))
590         return null;
591      if (h.isNoTableHeaders() || (hbpMeta != null && hbpMeta.isNoTableHeaders()))
592         return new String[0];
593      if (session.canIgnoreValue(cm, null, o1))
594         return null;
595      if (cm.isMap() && ! cm.isBeanMap()) {
596         Set<String> set = new LinkedHashSet<String>();
597         for (Object o : c) {
598            if (! session.canIgnoreValue(cm, null, o)) {
599               if (! cm.isInstance(o))
600                  return null;
601               Map m = session.sort((Map)o);
602               for (Map.Entry e : (Set<Map.Entry>)m.entrySet()) {
603                  if (e.getValue() != null)
604                     set.add(e.getKey() == null ? null : e.getKey().toString());
605               }
606            }
607         }
608         th = set.toArray(new String[set.size()]);
609      } else {
610         Set<String> set = new LinkedHashSet<String>();
611         for (Object o : c) {
612            if (! session.canIgnoreValue(cm, null, o)) {
613               if (! cm.isInstance(o))
614                  return null;
615               BeanMap<?> bm = (o instanceof BeanMap ? (BeanMap)o : session.toBeanMap(o));
616               for (Map.Entry<String,Object> e : bm.entrySet()) {
617                  if (e.getValue() != null)
618                     set.add(e.getKey());
619               }
620            }
621         }
622         th = set.toArray(new String[set.size()]);
623      }
624      prevC.add(cm);
625      s.addAll(Arrays.asList(th));
626
627      for (Object o : c) {
628         if (o == null)
629            continue;
630         cm = session.getClassMetaForObject(o);
631         if (cm != null && cm.getPojoSwap() != null) {
632            PojoSwap f = cm.getPojoSwap();
633            o = f.swap(session, o);
634            cm = cm.getSerializedClassMeta();
635         }
636         if (prevC.contains(cm))
637            continue;
638         if (cm == null || ! (cm.isMap() || cm.isBean()))
639            return null;
640         if (cm.getInnerClass().isAnnotationPresent(HtmlLink.class))
641            return null;
642         if (session.canIgnoreValue(cm, null, o))
643            return null;
644         if (cm.isMap() && ! cm.isBeanMap()) {
645            Map m = (Map)o;
646            if (th.length != m.keySet().size())
647               return null;
648            for (Object k : m.keySet())
649               if (! s.contains(k.toString()))
650                  return null;
651         } else {
652            BeanMap<?> bm = (o instanceof BeanMap ? (BeanMap)o : session.toBeanMap(o));
653            int l = 0;
654            for (String k : bm.keySet()) {
655               if (! s.contains(k))
656                  return null;
657               l++;
658            }
659            if (s.size() != l)
660               return null;
661         }
662      }
663      return th;
664   }
