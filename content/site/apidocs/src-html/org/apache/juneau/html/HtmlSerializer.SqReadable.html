013package org.apache.juneau.html;
014
015import static org.apache.juneau.serializer.SerializerContext.*;
016
017import java.util.*;
018
019import org.apache.juneau.*;
020import org.apache.juneau.annotation.*;
021import org.apache.juneau.serializer.*;
022import org.apache.juneau.xml.*;

024/**
025 * Serializes POJO models to HTML.
026 *
027 * <h5 class='section'>Media types:</h5>
028 *
029 * Handles <code>Accept</code> types: <code>text/html</code>
030 *
031 * <p>
032 * Produces <code>Content-Type</code> types: <code>text/html</code>
033 *
034 * <h5 class='section'>Description:</h5>
035 *
036 * The conversion is as follows...
037 * <ul class='spaced-list'>
038 *    <li>
039 *       {@link Map Maps} (e.g. {@link HashMap}, {@link TreeMap}) and beans are converted to HTML tables with
040 *       'key' and 'value' columns.
041 *    <li>
042 *       {@link Collection Collections} (e.g. {@link HashSet}, {@link LinkedList}) and Java arrays are converted
043 *       to HTML ordered lists.
044 *    <li>
045 *       {@code Collections} of {@code Maps} and beans are converted to HTML tables with keys as headers.
046 *    <li>
047 *       Everything else is converted to text.
048 * </ul>
049 *
050 * <p>
051 * This serializer provides several serialization options.  Typically, one of the predefined <jsf>DEFAULT</jsf>
052 * serializers will be sufficient.
053 * However, custom serializers can be constructed to fine-tune behavior.
054 *
055 * <p>
056 * The {@link HtmlLink} annotation can be used on beans to add hyperlinks to the output.
057 *
058 * <h5 class='section'>Configurable properties:</h5>
059 *
060 * This class has the following properties associated with it:
061 * <ul class='spaced-list'>
062 *    <li>
063 *       {@link HtmlSerializerContext}
064 * </ul>
065 *
066 * <h6 class='topic'>Behavior-specific subclasses</h6>
067 *
068 * The following direct subclasses are provided for convenience:
069 * <ul class='spaced-list'>
070 *    <li>
071 *       {@link Sq} - Default serializer, single quotes.
072 *    <li>
073 *       {@link SqReadable} - Default serializer, single quotes, whitespace added.
074 * </ul>
075 *
076 * <h5 class='section'>Example:</h5>
077 * <p class='bcode'>
078 *    <jc>// Use one of the default serializers to serialize a POJO</jc>
079 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(someObject);
080 *
081 *       <jc>// Create a custom serializer that doesn't use whitespace and newlines</jc>
082 *       HtmlSerializer serializer = <jk>new</jk> HtmlSerializerBuider().ws().build();
083 *
084 *       <jc>// Same as above, except uses cloning</jc>
085 *       HtmlSerializer serializer = HtmlSerializer.<jsf>DEFAULT</jsf>.builder().ws().build();
086 *
087 *       <jc>// Serialize POJOs to HTML</jc>
088 *
089 *       <jc>// Produces: </jc>
090 *       <jc>// &lt;ul&gt;&lt;li&gt;1&lt;li&gt;2&lt;li&gt;3&lt;/ul&gt;</jc>
091 *       List l = new ObjectList(1, 2, 3);
092 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(l);
093 *
094 *       <jc>// Produces: </jc>
095 *       <jc>//    &lt;table&gt; </jc>
096 *       <jc>//       &lt;tr&gt;&lt;th&gt;firstName&lt;/th&gt;&lt;th&gt;lastName&lt;/th&gt;&lt;/tr&gt; </jc>
097 *       <jc>//       &lt;tr&gt;&lt;td&gt;Bob&lt;/td&gt;&lt;td&gt;Costas&lt;/td&gt;&lt;/tr&gt; </jc>
098 *       <jc>//       &lt;tr&gt;&lt;td&gt;Billy&lt;/td&gt;&lt;td&gt;TheKid&lt;/td&gt;&lt;/tr&gt; </jc>
099 *       <jc>//       &lt;tr&gt;&lt;td&gt;Barney&lt;/td&gt;&lt;td&gt;Miller&lt;/td&gt;&lt;/tr&gt; </jc>
100 *       <jc>//    &lt;/table&gt; </jc>
101 *       l = <jk>new</jk> ObjectList();
102 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Bob',lastName:'Costas'}"</js>));
103 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Billy',lastName:'TheKid'}"</js>));
104 *       l.add(<jk>new</jk> ObjectMap(<js>"{firstName:'Barney',lastName:'Miller'}"</js>));
105 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(l);
106 *
107 *       <jc>// Produces: </jc>
108 *       <jc>//    &lt;table&gt; </jc>
109 *       <jc>//       &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
110 *       <jc>//       &lt;tr&gt;&lt;td&gt;foo&lt;/td&gt;&lt;td&gt;bar&lt;/td&gt;&lt;/tr&gt; </jc>
111 *       <jc>//       &lt;tr&gt;&lt;td&gt;baz&lt;/td&gt;&lt;td&gt;123&lt;/td&gt;&lt;/tr&gt; </jc>
112 *       <jc>//    &lt;/table&gt; </jc>
113 *       Map m = <jk>new</jk> ObjectMap(<js>"{foo:'bar',baz:123}"</js>);
114 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(m);
115 *
116 *       <jc>// HTML elements can be nested arbitrarily deep</jc>
117 *       <jc>// Produces: </jc>
118 *       <jc>//   &lt;table&gt; </jc>
119 *       <jc>//      &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
120 *       <jc>//      &lt;tr&gt;&lt;td&gt;foo&lt;/td&gt;&lt;td&gt;bar&lt;/td&gt;&lt;/tr&gt; </jc>
121 *       <jc>//      &lt;tr&gt;&lt;td&gt;baz&lt;/td&gt;&lt;td&gt;123&lt;/td&gt;&lt;/tr&gt; </jc>
122 *       <jc>//      &lt;tr&gt;&lt;td&gt;someNumbers&lt;/td&gt;&lt;td&gt;&lt;ul&gt;&lt;li&gt;1&lt;li&gt;2&lt;li&gt;3&lt;/ul&gt;&lt;/td&gt;&lt;/tr&gt; </jc>
123 *       <jc>//      &lt;tr&gt;&lt;td&gt;someSubMap&lt;/td&gt;&lt;td&gt; </jc>
124 *       <jc>//         &lt;table&gt; </jc>
125 *       <jc>//            &lt;tr&gt;&lt;th&gt;key&lt;/th&gt;&lt;th&gt;value&lt;/th&gt;&lt;/tr&gt; </jc>
126 *       <jc>//            &lt;tr&gt;&lt;td&gt;a&lt;/td&gt;&lt;td&gt;b&lt;/td&gt;&lt;/tr&gt; </jc>
127 *       <jc>//         &lt;/table&gt; </jc>
128 *       <jc>//      &lt;/td&gt;&lt;/tr&gt; </jc>
129 *       <jc>//   &lt;/table&gt; </jc>
130 *       Map m = <jk>new</jk> ObjectMap(<js>"{foo:'bar',baz:123}"</js>);
131 *       m.put("someNumbers", new ObjectList(1, 2, 3));
132 *       m.put(<js>"someSubMap"</js>, new ObjectMap(<js>"{a:'b'}"</js>));
133 *       String html = HtmlSerializer.<jsf>DEFAULT</jsf>.serialize(m);
134 * </p>
135 */
136@Produces("text/html")
137@SuppressWarnings("hiding")
138public class HtmlSerializer extends XmlSerializer {
139
140   /** Default serializer, all default settings. */
141   public static final HtmlSerializer DEFAULT = new HtmlSerializer(PropertyStore.create());
142
143   /** Default serializer, single quotes. */
144   public static final HtmlSerializer DEFAULT_SQ = new HtmlSerializer.Sq(PropertyStore.create());
145
146   /** Default serializer, single quotes, whitespace added. */
147   public static final HtmlSerializer DEFAULT_SQ_READABLE = new HtmlSerializer.SqReadable(PropertyStore.create());
148
149
150   /** Default serializer, single quotes. */
151   public static class Sq extends HtmlSerializer {
152
153      /**
154       * Constructor.
155       *
156       * @param propertyStore The property store containing all the settings for this object.
157       */
158      public Sq(PropertyStore propertyStore) {
159         super(propertyStore.copy().append(SERIALIZER_quoteChar, '\''));
160      }
161   }
162
163   /** Default serializer, single quotes, whitespace added. */
164   public static class SqReadable extends HtmlSerializer {
165
166      /**
167       * Constructor.
168       *
169       * @param propertyStore The property store containing all the settings for this object.
170       */
171      public SqReadable(PropertyStore propertyStore) {
172         super(propertyStore.copy().append(SERIALIZER_quoteChar, '\'').append(SERIALIZER_useWhitespace, true));
173      }
174   }
175
176
177   final HtmlSerializerContext ctx;
178   private volatile HtmlSchemaDocSerializer schemaSerializer;
179
180   /**
181    * Constructor.
182    *
183    * @param propertyStore The property store containing all the settings for this object.
184    */
185   public HtmlSerializer(PropertyStore propertyStore) {
186      super(propertyStore);
187      this.ctx = createContext(HtmlSerializerContext.class);
188   }
189
190   @Override /* CoreObject */
191   public HtmlSerializerBuilder builder() {
192      return new HtmlSerializerBuilder(propertyStore);
193   }
194
195   @Override /* XmlSerializer */
196   public HtmlSerializer getSchemaSerializer() {
197      if (schemaSerializer == null)
198         schemaSerializer = new HtmlSchemaDocSerializer(propertyStore);
199      return schemaSerializer;
200   }
201
202   @Override /* Serializer */
203   public WriterSerializerSession createSession(SerializerSessionArgs args) {
204      return new HtmlSerializerSession(ctx, args);
205   }
206}
