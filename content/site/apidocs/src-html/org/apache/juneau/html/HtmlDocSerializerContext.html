013package org.apache.juneau.html;
014
015import static org.apache.juneau.html.HtmlDocSerializer.*;
016
017import org.apache.juneau.*;
018
019/**
020 * Contains a snapshot-in-time read-only copy of the settings on the {@link HtmlDocSerializer} class.
021 */
022public final class HtmlDocSerializerContext extends HtmlSerializerContext {
023
024   final String[] style, stylesheet, script, navlinks, head, header, nav, aside, footer;
025   final String noResultsMessage;
026   final boolean nowrap;
027   final HtmlDocTemplate template;
028
029   /**
030    * Constructor.
031    *
032    * <p>
033    * Typically only called from {@link PropertyStore#getContext(Class)}.
034    *
035    * @param ps The property store that created this context.
036    */
037   public HtmlDocSerializerContext(PropertyStore ps) {
038      super(ps);
039      style = ps.getProperty(HTMLDOC_style, String[].class, new String[0]);
040      stylesheet = ps.getProperty(HTMLDOC_stylesheet, String[].class, new String[0]);
041      script = ps.getProperty(HTMLDOC_script, String[].class, new String[0]);
042      head = ps.getProperty(HTMLDOC_head, String[].class, new String[0]);
043      header = ps.getProperty(HTMLDOC_header, String[].class, new String[0]);
044      nav = ps.getProperty(HTMLDOC_nav, String[].class, new String[0]);
045      aside = ps.getProperty(HTMLDOC_aside, String[].class, new String[0]);
046      footer = ps.getProperty(HTMLDOC_footer, String[].class, new String[0]);
047      nowrap = ps.getProperty(HTMLDOC_nowrap, boolean.class, false);
048      navlinks = ps.getProperty(HTMLDOC_navlinks, String[].class, new String[0]);
049      noResultsMessage = ps.getProperty(HTMLDOC_noResultsMessage, String.class, "<p>no results</p>");
050      template = ps.getTypedProperty(HTMLDOC_template, HtmlDocTemplate.class, HtmlDocTemplateBasic.class);
051   }
052
053   @Override /* Context */
054   public ObjectMap asMap() {
055      return super.asMap()
056         .append("HtmlDocSerializerContext", new ObjectMap()
057            .append("header", header)
058            .append("nav", nav)
059            .append("navlinks", navlinks)
060            .append("aside", aside)
061            .append("footer", footer)
062            .append("style", style)
063            .append("head", head)
064            .append("stylesheet", stylesheet)
065            .append("nowrap", nowrap)
066            .append("template", template)
067            .append("noResultsMessage", noResultsMessage)
068         );
069   }
070}
