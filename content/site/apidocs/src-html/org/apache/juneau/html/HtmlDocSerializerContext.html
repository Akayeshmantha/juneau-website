019/**
020 * Properties associated with the {@link HtmlDocSerializer} class.
021 * <p>
022 *    These are typically specified via <ja>@RestResource.properties()</ja> and <ja>@RestMethod.properties()</ja> annotations,
023 *       although they can also be set programmatically via the <code>RestResponse.setProperty()</code> method.
024 *
025 * <h6 class='topic'>Example:</h6>
026 * <p class='bcode'>
027 *    <ja>@RestResource</ja>(
028 *       messages=<js>"nls/AddressBookResource"</js>,
029 *       properties={
030 *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_title</jsf>, value=<js>"$L{title}"</js>),
031 *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_description</jsf>, value=<js>"$L{description}"</js>),
032 *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_links</jsf>, value=<js>"{options:'?method=OPTIONS',doc:'doc'}"</js>)
033 *       }
034 *    )
035 *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
036 * </p>
037 * <p>
038 *    The <code>$L{...}</code> variable represent localized strings pulled from the resource bundle identified by the <code>messages</code> annotation.
039 *    These variables are replaced at runtime based on the HTTP request locale.
040 *    Several built-in runtime variable types are defined, and the API can be extended to include user-defined variables.
041 * </p>
042 *
043 * <h6 class='topic' id='ConfigProperties'>Configurable properties on the HTML document serializer</h6>
044 * <table class='styled' style='border-collapse: collapse;'>
045 *    <tr><th>Setting name</th><th>Description</th><th>Data type</th><th>Default value</th><th>Session overridable</th></tr>
046 *    <tr>
047 *       <td>{@link #HTMLDOC_title}</td>
048 *       <td>Page title.</td>
049 *       <td><code>String</code></td>
050 *       <td><jk>null</jk></td>
051 *       <td><jk>true</jk></td>
052 *    </tr>
053 *    <tr>
054 *       <td>{@link #HTMLDOC_description}</td>
055 *       <td>Page description.</td>
056 *       <td><code>String</code></td>
057 *       <td><jk>null</jk></td>
058 *       <td><jk>true</jk></td>
059 *    </tr>
060 *    <tr>
061 *       <td>{@link #HTMLDOC_links}<br>{@link #HTMLDOC_links_put}</td>
062 *       <td>Page links.</td>
063 *       <td><code>Map&lt;String,String&gt;</code></td>
064 *       <td>empty map</td>
065 *       <td><jk>true</jk></td>
066 *    </tr>
067 *    <tr>
068 *       <td>{@link #HTMLDOC_cssUrl}</td>
069 *       <td>Stylesheet URL.</td>
070 *       <td><code>String</code></td>
071 *       <td><js>"style.css"</js></td>
072 *       <td><jk>true</jk></td>
073 *    </tr>
074 *    <tr>
075 *       <td>{@link #HTMLDOC_cssImports}<br>{@link #HTMLDOC_cssImports_add}</td>
076 *       <td>CSS imports.</td>
077 *       <td><code>List&lt;String&gt;</code></td>
078 *       <td>empty list</td>
079 *       <td><jk>true</jk></td>
080 *    </tr>
081 *    <tr>
082 *       <td>{@link #HTMLDOC_nowrap}</td>
083 *       <td>Prevent word wrap on page.</td>
084 *       <td><code>Boolean</code></td>
085 *       <td><jk>false</jk></td>
086 *       <td><jk>true</jk></td>
087 *    </tr>
088 * </table>
089 *
090 * <h6 class='topic'>Configurable properties inherited from parent classes</h6>
091 * <ul class='javahierarchy'>
092 *    <li class='c'><a class='doclink' href='../BeanContext.html#ConfigProperties'>BeanContext</a> - Properties associated with handling beans on serializers and parsers.
093 *    <ul>
094 *       <li class='c'><a class='doclink' href='../serializer/SerializerContext.html#ConfigProperties'>SerializerContext</a> - Configurable properties common to all serializers.
095 *       <ul>
096 *          <li class='c'><a class='doclink' href='../html/HtmlSerializerContext.html#ConfigProperties'>HtmlSerializerContext</a> - Configurable properties on the HTML serializer.
097 *       </ul>
098 *    </ul>
099 * </ul>
100 */

177   /**
178    * <b>Configuration property:</b>  Page links.
179    * <p>
180    * <ul>
181    *    <li><b>Name:</b> <js>"HtmlDocSerializer.links.map"</js>
182    *    <li><b>Data type:</b> <code>Map&lt;String,String&gt;</code>
183    *    <li><b>Default:</b> empty map
184    *    <li><b>Session-overridable:</b> <jk>true</jk>
185    * </ul>
186    * <p>
187    * Adds a list of hyperlinks immediately under the title and description but above the content of the page.
188    * <p>
189    *    This can be used to provide convenient hyperlinks when viewing the REST interface from a browser.
190    * <p>
191    *    The value is a JSON object string where the keys are anchor text and the values are URLs.
192    * <p>
193    *    Relative URLs are considered relative to the servlet path.
194    *    For example, if the servlet path is <js>"http://localhost/myContext/myServlet"</js>, and the
195    *       URL is <js>"foo"</js>, the link becomes <js>"http://localhost/myContext/myServlet/foo"</js>.
196    *    Absolute (<js>"/myOtherContext/foo"</js>) and fully-qualified (<js>"http://localhost2/foo"</js>) URLs
197    *       can also be used.
198    *
199    * <h6 class='topic'>Example:</h6>
200    * <p>
201    *    The <code>AddressBookResource</code> sample class uses this property...
202    * </p>
203    * <p class='bcode'>
204    *    <ja>@RestResource</ja>(
205    *       messages=<js>"nls/AddressBookResource"</js>,
206    *       properties={
207    *          <ja>@Property</ja>(name=HtmlDocSerializerContext.<jsf>HTMLDOC_links</jsf>, value=<js>"{options:'?method=OPTIONS',doc:'doc'}"</js>)
208    *       }
209    *    )
210    *    <jk>public class</jk> AddressBookResource <jk>extends</jk> RestServletJenaDefault {
211    * </p>
212    * <p>
213    *    ...to produce this list of links on the HTML page...
214    * </p>
215    * <img class='bordered' src='doc-files/HTML_LINKS.png'>
216    */
217   public static final String HTMLDOC_links = "HtmlDocSerializer.links.map";
218
219   /**
220    * <b>Configuration property:</b>  Add to the {@link #HTMLDOC_links} property.
221    */
222   public static final String HTMLDOC_links_put = "HtmlDocSerializer.links.map.put";
223
224   /**
225    * <b>Configuration property:</b>  Stylesheet URL.
226    * <p>
227    * <ul>
228    *    <li><b>Name:</b> <js>"HtmlDocSerializer.cssUrl"</js>
229    *    <li><b>Data type:</b> <code>String</code>
230    *    <li><b>Default:</b> <js>"style.css"</js>
231    *    <li><b>Session-overridable:</b> <jk>true</jk>
232    * </ul>
233    * <p>
234    * Adds a link to the specified stylesheet URL.
235    * <p>
236    *    If not specified, defaults to the built-in stylesheet located at <js>"style.css"</js>.
237    *    Note that this stylesheet is controlled by the <code><ja>@RestResource</ja>.stylesheet()</code> annotation.
238    */
239   public static final String HTMLDOC_cssUrl = "HtmlDocSerializer.cssUrl";
240
241   /**
242    * <b>Configuration property:</b>  CSS imports.
243    * <p>
244    * <ul>
245    *    <li><b>Name:</b> <js>"HtmlDocSerializer.cssImports.list"</js>
246    *    <li><b>Data type:</b> <code>List&lt;String&gt;</code>
247    *    <li><b>Default:</b> empty list
248    *    <li><b>Session-overridable:</b> <jk>true</jk>
249    * </ul>
250    * <p>
251    * Imports the specified CSS page URLs into the page.
252    */
253   public static final String HTMLDOC_cssImports = "HtmlDocSerializer.cssImports.list";
254
255   /**
256    * <b>Configuration property:</b>  Add to the {@link #HTMLDOC_cssImports} property.
257    */
258   public static final String HTMLDOC_cssImports_add = "HtmlDocSerializer.cssImports.list.add";
259
260   /**
261    * <b>Configuration property:</b>  Prevent word wrap on page.
262    * <p>
263    * <ul>
264    *    <li><b>Name:</b> <js>"HtmlDocSerializer.nowrap"</js>
265    *    <li><b>Data type:</b> <code>Boolean</code>
266    *    <li><b>Default:</b> <jk>false</jk>
267    *    <li><b>Session-overridable:</b> <jk>true</jk>
268    * </ul>
269    * <p>
270    * Adds <js>"* {white-space:nowrap}"</js> to the style header to prevent word wrapping.
271    */
272   public static final String HTMLDOC_nowrap = "HtmlDocSerializer.nowrap";
273
274   final String[] cssImports;
275   final Map<String,String> links;
276   final String title, description, cssUrl;
277   final boolean nowrap;
278
279   /**
280    * Constructor.
281    * <p>
282    * Typically only called from {@link ContextFactory#getContext(Class)}.
283    *
284    * @param cf The factory that created this context.
285    */
286   public HtmlDocSerializerContext(ContextFactory cf) {
287      super(cf);
288      cssImports = cf.getProperty(HTMLDOC_cssImports, String[].class, new String[0]);
289      title = cf.getProperty(HTMLDOC_title, String.class, null);
290      description = cf.getProperty(HTMLDOC_description, String.class, null);
291      cssUrl = cf.getProperty(HTMLDOC_cssUrl, String.class, null);
292      nowrap = cf.getProperty(HTMLDOC_nowrap, boolean.class, false);
293      links = cf.getMap(HTMLDOC_links, String.class, String.class, Collections.<String,String>emptyMap());
294   }
295
296   @Override /* Context */
297   public ObjectMap asMap() {
298      return super.asMap()
299         .append("HtmlDocSerializerContext", new ObjectMap()
300            .append("cssImports", cssImports)
301            .append("title", title)
302            .append("description", description)
303            .append("cssUrl", cssUrl)
304            .append("nowrap", nowrap)
305            .append("links", links)
306         );
307   }
