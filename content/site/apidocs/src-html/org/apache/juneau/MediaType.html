013package org.apache.juneau;
014
015import java.util.*;
016import java.util.concurrent.*;
017
018import org.apache.juneau.annotation.*;
019
020
021/**
022 * Describes a single media type used in content negotiation between an HTTP client and server, as described in
023 * Section 14.1 and 14.7 of RFC2616 (the HTTP/1.1 specification).
024 */
025@BeanIgnore
026public final class MediaType {
027
028   private static final ConcurrentHashMap<String,MediaType> cache = new ConcurrentHashMap<String,MediaType>();
029
030   /** Reusable predefined media type */
031   @SuppressWarnings("javadoc")
032   public static final MediaType
033      CSV = forString("text/csv"),
034      HTML = forString("text/html"),
035      JSON = forString("application/json"),
036      MSGPACK = forString("octal/msgpack"),
037      PLAIN = forString("text/plain"),
038      UON = forString("text/uon"),
039      URLENCODING = forString("application/x-www-form-urlencoded"),
040      XML = forString("text/xml"),
041      XMLSOAP = forString("text/xml+soap"),
042
043      RDF = forString("text/xml+rdf"),
044      RDFABBREV = forString("text/xml+rdf+abbrev"),
045      NTRIPLE = forString("text/n-triple"),
046      TURTLE = forString("text/turtle"),
047      N3 = forString("text/n3")
048   ;
049
050   private final String mediaType;
051   private final String type;                      // The media type (e.g. "text" for Accept, "utf-8" for Accept-Charset)
052   private final String subType;                   // The media sub-type (e.g. "json" for Accept, not used for Accept-Charset)
053
054   /**
055    * Returns the media type for the specified string.
056    * The same media type strings always return the same objects so that these objects
057    * can be compared for equality using '=='.
058    * <p>
059    * <h5 class='section'>Notes:</h5>
060    * <ul>
061    *    <li>Spaces are replaced with <js>'+'</js> characters.
062    *       This gets around the issue where passing media type strings with <js>'+'</js> as HTTP GET parameters
063    *       get replaced with spaces by your browser.  Since spaces aren't supported by the spec, this
064    *       is doesn't break anything.
065    *    <li>Anything including and following the <js>';'</js> character is ignored (e.g. <js>";charset=X"</js>).
066    * </ul>
067    *
068    * @param s - The media type string.  Will be lowercased.
069    *    <br>Returns <jk>null</jk> if input is null.
070    * @return A cached media type object.
071    */
072   public static MediaType forString(String s) {
073      if (s == null)
074         return null;
075      MediaType mt = cache.get(s);
076      if (mt == null) {
077         mt = new MediaType(s);
078         cache.putIfAbsent(s, mt);
079      }
080      return cache.get(s);
081   }
082
083   private MediaType(String mt) {
084      int i = mt.indexOf(';');
085      if (i != -1)
086         mt = mt.substring(0, i);
087
088      mt = mt.toLowerCase(Locale.ENGLISH);
089      this.mediaType = mt;
090      String _type = null, _subType = null;
091      if (mt != null) {
092         mt = mt.replace(' ', '+');
093         i = mt.indexOf('/');
094         _type = (i == -1 ? mt : mt.substring(0, i));
095         _subType = (i == -1 ? "*" : mt.substring(i+1));
096      }
097      this.type = _type;
098      this.subType = _subType;
099   }
100
101   /**
102    * Returns the <js>'type'</js> fragment of the <js>'type/subType'</js> string.
103    *
104    * @return The media type.
105    */
106   public String getType() {
107      return type;
108   }
109
110   /**
111    * Returns the <js>'subType'</js> fragment of the <js>'type/subType'</js> string.
112    *
113    * @return The media subtype.
114    */
115   public String getSubType() {
116      return subType;
117   }
118
119   /**
120    * Returns <jk>true</jk> if this media type is a match for the specified media type.
121    * <p>
122    * Matches if any of the following is true:
123    * <ul>
124    *    <li>Both type and subtype are the same.
125    *    <li>One or both types are <js>'*'</js> and the subtypes are the same.
126    *    <li>One or both subtypes are <js>'*'</js> and the types are the same.
127    *    <li>Either is <js>'*\/*'</js>.
128    * </ul>
129    *
130    * @param o The media type to compare with.
131    * @return <jk>true</jk> if the media types match.
132    */
133   public final boolean matches(MediaType o) {
134      if (this == o)
135         return true;
136
137      if (type.equals(o.type) || (type.equals("*")) || (o.type.equals("*")))
138         if (subType.equals(o.subType) || subType.equals("*") || o.subType.equals("*"))
139            return true;
140
141      return false;
142   }
143
144   @Override /* Object */
145   public String toString() {
146      return mediaType;
147   }
148
149   @Override /* Object */
150   public int hashCode() {
151      return mediaType.hashCode();
152   }
153
154   @Override /* Object */
155   public boolean equals(Object o) {
156      return this == o;
157   }
158}
