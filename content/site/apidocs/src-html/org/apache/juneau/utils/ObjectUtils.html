013package org.apache.juneau.utils;
014
015import org.apache.juneau.*;
016import org.apache.juneau.transform.*;
017
018/**
019 * Utility class for efficiently converting objects between types.
020 * <p>
021 * If the value isn't an instance of the specified type, then converts the value if possible.
022 * <p>
023 * The following conversions are valid:
024 * <table class='styled'>
025 *    <tr><th>Convert to type</th><th>Valid input value types</th><th>Notes</th></tr>
026 *    <tr>
027 *       <td>
028 *          A class that is the normal type of a registered {@link PojoSwap}.
029 *       </td>
030 *       <td>
031 *          A value whose class matches the transformed type of that registered {@link PojoSwap}.
032 *       </td>
033 *       <td>&nbsp;</td>
034 *    </tr>
035 *    <tr>
036 *       <td>
037 *          A class that is the transformed type of a registered {@link PojoSwap}.
038 *       </td>
039 *       <td>
040 *          A value whose class matches the normal type of that registered {@link PojoSwap}.
041 *       </td>
042 *       <td>&nbsp;</td>
043 *    </tr>
044 *    <tr>
045 *       <td>
046 *          {@code Number} (e.g. {@code Integer}, {@code Short}, {@code Float},...)
047 *          <br>
048 *          <code>Number.<jsf>TYPE</jsf></code> (e.g. <code>Integer.<jsf>TYPE</jsf></code>,
049 *          <code>Short.<jsf>TYPE</jsf></code>, <code>Float.<jsf>TYPE</jsf></code>,...)
050 *       </td>
051 *       <td>
052 *          {@code Number}, {@code String}, <jk>null</jk>
053 *       </td>
054 *       <td>
055 *          For primitive {@code TYPES}, <jk>null</jk> returns the JVM default value for that type.
056 *       </td>
057 *    </tr>
058 *    <tr>
059 *       <td>
060 *          {@code Map} (e.g. {@code Map}, {@code HashMap}, {@code TreeMap}, {@code ObjectMap})
061 *       </td>
062 *       <td>
063 *          {@code Map}
064 *       </td>
065 *       <td>
066 *          If {@code Map} is not constructible, a {@code ObjectMap} is created.
067 *       </td>
068 *    </tr>
069 *    <tr>
070 *       <td>
071 *       {@code Collection} (e.g. {@code List}, {@code LinkedList}, {@code HashSet}, {@code ObjectList})
072 *       </td>
073 *       <td>
074 *          {@code Collection<Object>}
075 *          <br>
076 *          {@code Object[]}
077 *       </td>
078 *       <td>
079 *          If {@code Collection} is not constructible, a {@code ObjectList} is created.
080 *       </td>
081 *    </tr>
082 *    <tr>
083 *       <td>
084 *          {@code X[]} (array of any type X)
085 *          <br>
086 *       </td>
087 *       <td>
088 *          {@code List<X>}
089 *       </td>
090 *       <td>&nbsp;</td>
091 *    </tr>
092 *    <tr>
093 *       <td>
094 *          {@code X[][]} (multi-dimensional arrays)
095 *       </td>
096 *       <td>
097 *          {@code List<List<X>>}
098 *          <br>
099 *          {@code List<X[]>}
100 *          <br>
101 *          {@code List[]<X>}
102 *       </td>
103 *       <td>&nbsp;</td>
104 *    </tr>
105 *    <tr>
106 *       <td>
107 *          {@code Enum}
108 *       </td>
109 *       <td>
110 *          {@code String}
111 *       </td>
112 *       <td>&nbsp;</td>
113 *    </tr>
114 *    <tr>
115 *       <td>
116 *          Bean
117 *       </td>
118 *       <td>
119 *          {@code Map}
120 *       </td>
121 *       <td>&nbsp;</td>
122 *    </tr>
123 *    <tr>
124 *       <td>
125 *          {@code String}
126 *       </td>
127 *       <td>
128 *          Anything
129 *       </td>
130 *       <td>
131 *          Arrays are converted to JSON arrays
132 *       </td>
133 *    </tr>
134 *    <tr>
135 *       <td>
136 *          Anything with one of the following methods:
137 *          <br>
138 *          <code><jk>public static</jk> T fromString(String)</code>
139 *          <br>
140 *          <code><jk>public static</jk> T valueOf(String)</code>
141 *          <br>
142 *          <code><jk>public</jk> T(String)</code>
143 *       </td>
144 *       <td>
145 *          <code>String</code>
146 *       </td>
147 *       <td>
148 *          <br>
149 *       </td>
150 *    </tr>
151 * </table>
152 */
153public class ObjectUtils {
154
155   // Session objects are usually not thread safe, but we're not using any feature
156   // of bean sessions that would cause thread safety issues.
157   private static final BeanSession session = BeanContext.DEFAULT.createSession();
158
159   /**
160    * Converts the specified object to the specified type.
161    *
162    * @param <T> The class type to convert the value to.
163    * @param value The value to convert.
164    * @param type The class type to convert the value to.
165    * @throws InvalidDataConversionException If the specified value cannot be converted to the specified type.
166    * @return The converted value.
167    */
168   public static <T> T convertToType(Object value, Class<T> type) {
169      return session.convertToType(value, type);
170   }
171
172   /**
173    * Converts the specified object to the specified type.
174    *
175    * @param <T> The class type to convert the value to.
176    * @param outer If class is a member class, this is the instance of the containing class.
177    * Should be <jk>null</jk> if not a member class.
178    * @param value The value to convert.
179    * @param type The class type to convert the value to.
180    * @throws InvalidDataConversionException If the specified value cannot be converted to the specified type.
181    * @return The converted value.
182    */
183   public static <T> T convertToType(Object outer, Object value, Class<T> type) {
184      return session.convertToType(outer, value, type);
185   }
186
187}
