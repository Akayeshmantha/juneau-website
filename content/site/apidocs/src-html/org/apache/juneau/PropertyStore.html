026/**
027 * Represents an immutable collection of properties.
028 * 
029 * <p>
030 * The general idea behind a property store is to serve as a reusable configuration of an artifact (e.g. a Serializer)
031 * such that the artifact can be cached and reused if the property stores are 'equal'.
032 * 
033 * <p>
034 * For example, two serializers of the same type created with the same configuration will always end up being
035 * the same serializer: 
036 * 
037 * <p class='bcode'>
038 *    WriterSerializer s1 = JsonSerializer.<jsm>create</jsm>().pojoSwaps(MySwap.<jk>class</jk>).simple().build();
039 *    WriterSerializer s2 = JsonSerializer.<jsm>create</jsm>().simple().pojoSwaps(MySwap.<jk>class</jk>).build();
040 *    <jk>assert</jk>(s1 == s2);
041 * </p>
042 * 
043 * <p>
044 * This has the effect of significantly improving performance, especially if you're creating many Serializers and 
045 * Parsers.
046 * 
047 * <p>
048 * Property names must have the following format...
049 * <p class='bcode'>
050 *    <js>"{group}.{name}.{type}"</js>
051 * </p>
052 * ...where the parts consist of the following...
053 * <ul>
054 *    <li><js>"{group}"</js> - The group name of the property (e.g. <js>"JsonSerializer"</js>).
055 *       <br>It's always going to be the simple class name of the class it's associated with.
056 *    <li><js>"{name}"</js> - The property name (e.g. <js>"useWhitespace"</js>).
057 *    <li><js>"{type}"</js> - The property data type.
058 *       <br>A 1 or 2 character string that identifies the data type of the property.
059 *       <br>Valid values are:
060 *       <ul>
061 *          <li><js>"s"<js>:  <code>String</code>
062 *          <li><js>"b"<js>:  <code>Boolean</code>
063 *          <li><js>"i"<js>:  <code>Integer</code>
064 *          <li><js>"c"<js>:  <code>Class</code>
065 *          <li><js>"o"<js>:  <code>Object</code>
066 *          <li><js>"ss"<js>:  <code>TreeSet&lt;String&gt;</code>
067 *          <li><js>"si"<js>:  <code>TreeSet&lt;Integer&gt;</code>
068 *          <li><js>"sc"<js>:  <code>TreeSet&lt;Class&gt;</code>
069 *          <li><js>"ls"<js>:  <code>ArrayList&lt;String&gt;</code>
070 *          <li><js>"li"<js>:  <code>ArrayList&lt;Integer&gt;</code>
071 *          <li><js>"lc"<js>:  <code>ArrayList&lt;Class&gt;</code>
072 *          <li><js>"lo"<js>:  <code>ArrayList&lt;Object&gt;</code>
073 *          <li><js>"ms"<js>:  <code>TreeMap&lt;String,String&gt;</code>
074 *          <li><js>"mi"<js>:  <code>TreeMap&lt;String,Integer&gt;</code>
075 *          <li><js>"mc"<js>:  <code>TreeMap&lt;String,Class&gt;</code>
076 *          <li><js>"mo"<js>:  <code>TreeMap&lt;String,Object&gt;</code>
077 *       </ul>
078 * </ul>
079 * 
080 * <p>
081 * For example, the {@link JsonSerializer} extends from {@link Serializer} and {@link BeanContext}, and therefore
082 * is configured by properties defined in the <js>"JsonSerializer"</js>, <js>"Serializer"</js>, and <js>"BeanContext"</js>
083 * groups.  Since serializers and bean contexts can be relatively expensive to instantiate, we can cache and
084 * reuse <code>JsonSerializer</code> instances if the property store used to instantiate them have equivalent
085 * <js>"JsonSerializer"</js>, <js>"Serializer"</js>, and <js>"BeanContext"</js> groups.
086 * 
087 * <p>
088 * Set and list properties have the additional convenience 'command' names for adding and removing entries:
089 * <p class='bcode'>
090 *    <js>"{group}.{name}.{type}/add"</js>  <jc>// Add a value to the set/list.</jc>
091 *    <js>"{group}.{name}.{type}/remove"</js>  <jc>// Remove a value from the set/list.</jc>
092 * </p>
093 * 
094 * <p>
095 * Set properties are typically used in cases where the order of the value is not important.
096 * A <code>TreeSet</code> is used so that the order in which you add elements does not affect the resulting order
097 * of the property.
098 * 
099 * <p>
100 * List properties are typically used in cases where the order in which entries are added is important.
101 * Adding to a list property will cause the new entries to be added to the BEGINNING of the list.
102 * This ensures that the resulting order of the list is in most-to-least importance.
103 * For example, multiple calls to <code>pojoSwaps()</code> causes new entries to be added to the beginning of the list
104 * so that previous values can be 'overridden':
105 * <p class='bcode'>
106 *    <jc>// Swap order:  [MySwap2.class, MySwap1.class]</jc>
107 *    JsonSerializer.create().pojoSwaps(MySwap1.class).pojoSwaps(MySwap2.class).build();
108 * </p>
109 * <p>
110 * Note that the order is different when passing multiple values into the <code>pojoSwaps()</code> method, in which
111 * case the order should be first-match-wins:
112 * <p class='bcode'>
113 *    <jc>// Swap order:  [MySwap1.class, MySwap2.class]</jc>
114 *    JsonSerializer.create().pojoSwaps(MySwap1.class,MySwap2.class).build();
115 * </p>
116 * 
117 * <p>
118 * Map properties have the additional convenience property name for adding and removing map entries:
119 * <p class='bcode'>
120 *    <js>"{group}.{name}.{type}/add.{key}"</js>  <jc>// Add a map entry (or delete if the value is null).</jc>
121 * </p>
122 * 
123 * <p>
124 * Property stores are entirely immutable which means their hashcode is calculated exactly once, meaning they
125 * are particularly good for use as hashmap keys.
126 */

139   PropertyStore(Map<String,PropertyGroupBuilder> propertyMaps) {
140      TreeMap<String,PropertyGroup> m = new TreeMap<>();
141      for (Map.Entry<String,PropertyGroupBuilder> p : propertyMaps.entrySet()) 
142         m.put(p.getKey(), p.getValue().build());
143      this.groups = Collections.unmodifiableSortedMap(m);
144      this.hashCode = groups.hashCode();
145   }

165   private Property findProperty(String key) {
166      String g = group(key);
167      String k = key.substring(g.length()+1);
168      PropertyGroup pm = groups.get(g);
169      
170      if (pm != null) {
171         Property p = pm.get(k);
172         if (p != null)
173            return p;
174      }
175
176      String s = System.getProperty(key);
177      if (s != null)
178         return PropertyStoreBuilder.MutableProperty.create(k, s).build();
179      
180      return null;
181   }

407         return (T)def;
408      throw new ConfigException("Could not instantiate property ''{0}'' as type ''{1}'' with default value ''{2}''", key, type, def);
409   }

468   /**
469    * Compares two property stores, but only based on the specified group names.
470    * 
471    * @param ps The property store to compare to.
472    * @param groups The groups to compare.
473    * @return <jk>true</jk> if the two property stores are equal in the specified groups.
474    */
475   public boolean equals(PropertyStore ps, String...groups) {
476      if (this == ps)
477         return true;
478      for (String p : groups) {
479         if (p != null) {
480            PropertyGroup pg1 = this.groups.get(p), pg2 = ps.groups.get(p);
481            if (pg1 == null && pg2 == null)
482               continue;
483            if (pg1 == null || pg2 == null)
484               return false;
485            if (! pg1.equals(pg2))
486               return false;
487         }
488      }
489      return true;
490   }

554   static class Property {
555      private final String name;
556      final Object value;
557      private final int hashCode; 
558      private final PropertyType type;
559
560      Property(String name, Object value, PropertyType type) {
561         this.name = name;
562         this.value = value;
563         this.type = type;
564         this.hashCode = value.hashCode();
565      }
566      
567      MutableProperty mutable() {
568         switch(type) {
569            case STRING:  
570            case BOOLEAN:
571            case INTEGER:
572            case CLASS:  
573            case OBJECT: return new MutableSimpleProperty(name, type, value);
574            case SET_STRING:
575            case SET_INTEGER:
576            case SET_CLASS: return new MutableSetProperty(name, type, value);
577            case LIST_STRING:
578            case LIST_INTEGER:
579            case LIST_CLASS: 
580            case LIST_OBJECT: return new MutableListProperty(name, type, value);
581            case MAP_STRING: 
582            case MAP_INTEGER: 
583            case MAP_CLASS: 
584            case MAP_OBJECT: return new MutableMapProperty(name, type, value);
585         }
586         throw new ConfigException("Invalid type specified: ''{0}''", type);
587      }
588      
589      public <T> T as(Class<T> c) {
590         Class<?> c2 = ClassUtils.getPrimitiveWrapper(c);
591         if (c2 != null)
592            c = (Class<T>)c2;
593         if (c.isInstance(value))
594            return (T)value;
595         if (c.isArray() && value instanceof Collection) 
596            return (T)asArray(c.getComponentType());
597         if (type == STRING) {
598            T t = ClassUtils.fromString(c, value.toString());
599            if (t != null)
600               return t;
601         }
602         throw new ConfigException("Invalid property conversion ''{0}'' to ''{1}'' on property ''{2}''", type, c, name);
603      }
604      
605      public <T> T[] asArray(Class<T> eType) {
606         if (value instanceof Collection) {
607            Collection<?> l = (Collection<?>)value;
608            Object t = Array.newInstance(eType, l.size());
609            int i = 0;
610            for (Object o : l) {
611               Object o2 = null;
612               if (eType.isInstance(o))
613                  o2 = o;
614               else if (type == SET_STRING || type == LIST_STRING) {
615                  o2 = ClassUtils.fromString(eType, o.toString());
616                  if (o2 == null)
617                     throw new ConfigException("Invalid property conversion ''{0}'' to ''{1}[]'' on property ''{2}''", type, eType, name);
618               } else {
619                  throw new ConfigException("Invalid property conversion ''{0}'' to ''{1}[]'' on property ''{2}''", type, eType, name);
620               }
621               Array.set(t, i++, o2);
622            }
623            return (T[])t;
624         }
625         throw new ConfigException("Invalid property conversion ''{0}'' to ''{1}[]'' on property ''{2}''", type, eType, name);
626      }
627      
628      public <T> Set<T> asSet(Class<T> eType) {
629         if (type == SET_STRING && eType == String.class || type == SET_INTEGER && eType == Integer.class || type == SET_CLASS && eType == Class.class) {
630            return (Set<T>)value;
631         } else if (type == SET_STRING) {
