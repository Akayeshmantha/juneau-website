028/**
029 * A store for instantiating {@link Context} objects.
030 *
031 * <p>
032 * The hierarchy of these objects are...
033 * <ul class='spaced-list'>
034 *    <li>
035 *       {@link PropertyStore} - A thread-safe, modifiable context property store.
036 *       <br>Used to create {@link Context} objects.
037 *    <li>
038 *       {@link Context} - A reusable, cacheable, thread-safe, read-only context with configuration properties copied
039 *       from the store.
040 *       <br>Often used to create {@link Session} objects.
041 *    <li>
042 *       {@link Session} - A one-time-use non-thread-safe object.
043 *       <br>Used by serializers and parsers to retrieve context properties and to be used as scratchpads.
044 * </ul>
045 *
046 * <h6 class='topic'>PropertyStore objects</h6>
047 *
048 * Property stores can be thought of as consisting of the following:
049 * <ul>
050 *    <li>A <code>Map&lt;String,Object&gt;</code> of context properties.
051 *    <li>A <code>Map&lt;Class,Context&gt;</code> of context instances.
052 * </ul>
053 *
054 * <p>
055 * Property stores are used to create and cache {@link Context} objects using the {@link #getContext(Class)} method.
056 *
057 * <p>
058 * As a general rule, {@link PropertyStore} objects are 'slow'.
059 * <br>Setting and retrieving properties on a store can involve relatively slow data conversion and synchronization.
060 * <br>However, the {@link #getContext(Class)} method is fast, and will return cached context objects if the context
061 * properties have not changed.
062 *
063 * <p>
064 * Property stores can be used to store context properties for a variety of contexts.
065 * <br>For example, a single store can store context properties for the JSON serializer, XML serializer, HTML serializer
066 * etc... and can thus be used to retrieve context objects for those serializers.
067 *
068 * <h6 class='topic'>Context properties</h6>
069 *
070 * Context properties are 'settings' for serializers and parsers.
071 * <br>For example, the {@link BeanContext#BEAN_sortProperties} context property defines whether bean properties should be
072 * serialized in alphabetical order.
073 *
074 * <p>
075 * Each {@link Context} object should contain the context properties that apply to it as static fields
076 * (e.g {@link BeanContext#BEAN_sortProperties}).
077 *
078 * <p>
079 * Context properties can be of the following types:
080 * <ul class='spaced-list'>
081 *    <li>
082 *       <l>SIMPLE</l> - A simple property.
083 *       <br>Examples include:  booleans, integers, Strings, Classes, etc...
084 *       <br>An example of this would be the {@link BeanContext#BEAN_sortProperties} property.
085 *       <br>It's name is simply <js>"BeanContext.sortProperties"</js>.
086 *    <li>
087 *       <l>SET</l> - A sorted set of objects.
088 *       <br>These are denoted by appending <js>".set"</js> to the property name.
089 *       <br>Objects can be of any type, even complex types.
090 *       <br>Sorted sets use tree sets to maintain the value in alphabetical order.
091 *       <br>For example, the {@link BeanContext#BEAN_notBeanClasses} property is used to store classes that should not be
092 *       treated like beans.
093 *       <br>It's name is <js>"BeanContext.notBeanClasses.set"</js>.
094 *    <li>
095 *       <l>LIST</l> - A list of unique objects.
096 *       <br>These are denoted by appending <js>".list"</js> to the property name.
097 *       <br>Objects can be of any type, even complex types.
098 *       <br>Use lists if the ordering of the values in the set is important (similar to how the order of entries in a
099 *       classpath is important).
100 *       <br>For example, the {@link BeanContext#BEAN_beanFilters} property is used to store bean filters.
101 *       <br>It's name is <js>"BeanContext.transforms.list"</js>.
102 *    <li>
103 *       <l>MAP</l> - A sorted map of key-value pairs.
104 *       <br>These are denoted by appending <js>".map"</js> to the property name.
105 *       <br>Keys can be any type directly convertible to and from Strings.
106 *       Values can be of any type, even complex types.
107 *       <br>For example, the {@link BeanContext#BEAN_implClasses} property is used to specify the names of implementation
108 *       classes for interfaces.
109 *       <br>It's name is <js>"BeanContext.implClasses.map"</js>.
110 * </ul>
111 *
112 * <p>
113 * All context properties are set using the {@link #setProperty(String, Object)} method.
114 *
115 * <p>
116 * Default values for context properties can be specified globally as system properties.
117 * <br>Example: <code>System.<jsm>setProperty</jsm>(<jsf>BEAN_sortProperties</jsf>, <jk>true</jk>);</code>
118 *
119 * <p>
120 * SET and LIST properties can be added to using the {@link #addToProperty(String, Object)} method and removed from
121 * using the {@link #removeFromProperty(String, Object)} method.
122 *
123 * <p>
124 * SET and LIST properties can also be added to and removed from by appending <js>".add"</js> or <js>".remove"</js> to
125 * the property name and using the {@link #setProperty(String, Object)} method.
126 *
127 * <p>
128 * The following shows the two different ways to append to a set or list property:
129 * <p class='bcode'>
130 *    PropertyStore ps = <jk>new</jk> PropertyStore().setProperty(<js>"BeanContext.notBeanClasses.set"</js>,
131 *       Collections.<jsm>emptySet</jsm>());
132 *
133 *    <jc>// Append to set property using addTo().</jc>
134 *    ps.addToProperty(<js>"BeanContext.notBeanClasses.set"</js>, MyNotBeanClass.<jk>class</jk>);
135 *
136 *    <jc>// Append to set property using set().</jc>
137 *    ps.setProperty(<js>"BeanContext.notBeanClasses.set.add"</js>, MyNotBeanClass.<jk>class</jk>);
138 * </p>
139 *
140 * <p>
141 * SET and LIST properties can also be set and manipulated using JSON strings.
142 * <p class='bcode'>
143 *    PropertyStore ps = PropertyStore.<jsm>create</jsm>();
144 *
145 *    <jc>// Set SET value using JSON array.
146 *    ps.setProperty(<js>"BeanContext.notBeanClasses.set"</js>, <js>"['com.my.MyNotBeanClass1']"</js>);
147 *
148 *    <jc>// Add to SET using simple string.
149 *    ps.addToProperty(<js>"BeanContext.notBeanClasses.set"</js>, <js>"com.my.MyNotBeanClass2"</js>);
150 *
151 *    <jc>// Add an array of values as a JSON array..
152 *    ps.addToProperty(<js>"BeanContext.notBeanClasses.set"</js>, <js>"['com.my.MyNotBeanClass3']"</js>);
153 *
154 *    <jc>// Remove an array of values as a JSON array..
155 *    ps.removeFromProperty(<js>"BeanContext.notBeanClasses.set"</js>, <js>"['com.my.MyNotBeanClass3']"</js>);
156 * </p>
157 *
158 * <p>
159 * MAP properties can be added to using the {@link #putToProperty(String, Object, Object)} and
160 * {@link #putToProperty(String, Object)} methods.
161 * <br>MAP property entries can be removed by setting the value to <jk>null</jk>
162 * (e.g. <code>putToProperty(<js>"BEAN_implClasses"</js>, MyNotBeanClass.<jk>class</jk>, <jk>null</jk>);</code>.
163 * <br>MAP properties can also be added to by appending <js>".put"</js> to the property name and using the
164 * {@link #setProperty(String, Object)} method.
165 *
166 * <p>
167 * The following shows the two different ways to append to a set property:
168 * <p class='bcode'>
169 *    PropertyStore ps = PropertyStore.<jsm>create</jsm>().setProperty(<js>"BeanContext.implClasses.map"</js>,
170 *       Collections.<jsm>emptyMap</jsm>());
171 *
172 *    <jc>// Append to map property using putTo().</jc>
173 *    ps.putToProperty(<js>"BeanContext.implClasses.map"</js>, MyInterface.<jk>class</jk>, MyInterfaceImpl.<jk>class</jk>);
174 *
175 *    <jc>// Append to map property using set().</jc>
176 *    Map m = <jk>new</jk> AMap().append(MyInterface.<jk>class</jk>,MyInterfaceImpl.<jk>class</jk>);
177 *    ps.setProperty(<js>"BeanContext.implClasses.map.put"</js>, m);
178 * </p>
179 *
180 * <p>
181 * MAP properties can also be set and manipulated using JSON strings.
182 * <p class='bcode'>
183 *    PropertyStore ps = PropertyStore.<jsm>create</jsm>();
184 *
185 *    <jc>// Set MAP value using JSON object.</jc>
186 *    ps.setProperty(<js>"BeanContext.implClasses.map"</js>,
187 *       <js>"{'com.my.MyInterface1':'com.my.MyInterfaceImpl1'}"</js>);
188 *
189 *    <jc>// Add to MAP using JSON object.</jc>
190 *    ps.putToProperty(<js>"BeanContext.implClasses.map"</js>,
191 *       <js>"{'com.my.MyInterface2':'com.my.MyInterfaceImpl2'}"</js>);
192 *
193 *    <jc>// Remove from MAP using JSON object.</jc>
194 *    ps.putToProperty(<js>"BeanContext.implClasses.map"</js>, <js>"{'com.my.MyInterface2':null}"</js>);
195 * </p>
196 *
197 * <p>
198 * Context properties are retrieved from this store using the following 3 methods:
199 * <ul class='spaced-list'>
200 *    <li>
201 *       {@link #getProperty(String, Class, Object)} - Retrieve a SIMPLE or SET property converted to the specified
202 *       class type.
203 *    <li>
204 *       {@link #getMap(String, Class, Class, Map)} - Retrieve a MAP property with keys/values converted to the
205 *       specified class types.
206 *    <li>
207 *       {@link #getPropertyMap(String)} - Retrieve a map of all context properties with the specified prefix
208 *       (e.g. <js>"BeanContext"</js> for {@link BeanContext} properties).
209 * </ul>
210 *
211 * <p>
212 * As a general rule, only {@link Context} objects will use these read methods.
213 *
214 * <h6 class='topic'>Context objects</h6>
215 *
216 * A Context object can be thought of as unmodifiable snapshot of a store.
217 * <br>They should be 'fast' by avoiding synchronization by using final fields whenever possible.
218 * <br>However, they MUST be thread safe.
219 *
220 * <p>
221 * Context objects are created using the {@link #getContext(Class)} method.
222 * <br>As long as the properties on a store have not been modified, the store will return a cached copy of a context.
223 * <p class='bcode'>
224 *    PropertyStore ps = PropertyStore.<jsm>create</jsm>();
225 *
226 *    <jc>// Get BeanContext with default store settings.</jc>
227 *    BeanContext bc = ps.getContext(BeanContext.<jk>class</jk>);
228 *
229 *    <jc>// Get another one.  This will be the same one.</jc>
230 *    BeanContext bc2 = ps.getContext(BeanContext.<jk>class</jk>);
231 *    <jsm>assertTrue</jsm>(bc1 == bc2);
232 *
233 *    <jc>// Set a property.</jc>
234 *    ps.setProperty(<jsf>BEAN_sortProperties</jsf>, <jk>true</jk>);
235 *
236 *    <jc>// Get another one.  This will be different!</jc>
237 *    bc2 = f.getContext(BeanContext.<jk>class</jk>);
238 *    <jsm>assertFalse</jsm>(bc1 == bc2);
239 * </p>
240 *
241 * <h6 class='topic'>Session objects</h6>
242 *
243 * Session objects are created through {@link Context} objects, typically through a <code>createContext()</code> method.
244 * <br>Unlike context objects, they are NOT reusable and NOT thread safe.
245 * <br>They are meant to be used one time and then thrown away.
246 * <br>They should NEVER need to use synchronization.
247 *
248 * <p>
249 * Session objects are also often used as scratchpads for information such as keeping track of call stack information
250 * to detect recursive loops when serializing beans.
251 */
252public final class PropertyStore {

254   // All configuration properties in this object.
255   // Keys are property prefixes (e.g. 'BeanContext').
256   // Values are maps containing properties for that specific prefix.
257   private Map<String,PropertyMap> properties = new ConcurrentSkipListMap<>();
258
259   // Context cache.
260   // This gets cleared every time any properties change on this object.
261   private final Map<Class<? extends Context>,Context> contexts = new ConcurrentHashMap<>();
262
263   // Global Context cache.
264   // Property stores that are the 'same' will use the same maps from this cache.
265   // 'same' means the context properties are all the same when converted to strings.
266   private static final ConcurrentHashMap<Integer, ConcurrentHashMap<Class<? extends Context>,Context>>
267      globalContextCache = new ConcurrentHashMap<>();

270   private Lock rl = lock.readLock(), wl = lock.writeLock();

314    *
315    * <p>
316    * Properties of type set/list/collection will be appended to the existing properties if they already exist.
317    *
318    * @param ps The store to copy from.
319    * @return This object (for method chaining).
320    */
321   public PropertyStore copyFrom(PropertyStore ps) {
322      if (ps != null) {
323         // TODO - Needs more testing.
324         for (Map.Entry<String,PropertyMap> e : ps.properties.entrySet())
325            this.properties.put(e.getKey(), new PropertyMap(this.properties.get(e.getKey()), e.getValue()));
326         this.classLoader = ps.classLoader;
327         this.defaultParser = ps.defaultParser;
328      }
329      return this;
330   }

341   /**
342    * Sets a configuration property value on this object.
343    *
344    * <p>
345    * A typical usage is to set or overwrite configuration values like so...
346    * <p class='bcode'>
347    *    PropertyStore ps = PropertyStore.<jsm>create</jsm>();
348    *    ps.setProperty(<jsf>BEAN_sortProperties</jsf>, <jk>true</jk>);
349    * </p>
350    *
351    * <p>
352    * The possible class types of the value depend on the property type:
353    * <table class='styled'>
354    *    <tr>
355    *       <th>Property type</th>
356    *       <th>Example</th>
357    *       <th>Allowed value type</th>
358    *    </tr>
359    *    <tr>
360    *       <td>Set <l>SIMPLE</l></td>
361    *       <td><js>"Foo.x"</js></td>
362    *       <td>Any object type.</td>
363    *    </tr>
364    *    <tr>
365    *       <td>Set <l>SET/LIST</l></td>
366    *       <td><js>"Foo.x.set"</js></td>
367    *       <td>Any collection or array of any objects, or a String containing a JSON array.</td>
368    *    </tr>
369    *    <tr>
370    *       <td>Add/Remove <l>SET/LIST</l></td>
371    *       <td><js>"Foo.x.set.add"</js></td>
372    *       <td>If a collection, adds or removes the entries in the collection.  Otherwise, adds/removes a single
373    *          entry.</td>
374    *    </tr>
375    *    <tr>
376    *       <td>Set <l>MAP</l></td>
377    *       <td><js>"Foo.x.map"</js></td>
378    *       <td>A map, or a String containing a JSON object.  Entries overwrite existing map.</td>
379    *    </tr>
380    *    <tr>
381    *       <td>Put <l>MAP</l></td>
382    *       <td><js>"Foo.x.map.put"</js></td>
383    *       <td>A map, or a String containing a JSON object.  Entries are added to existing map.</td>
384    *    </tr>
385    * </table>
386    *
387    * @param name
388    *    The configuration property name.
389    *    <br>If name ends with <l>.add</l>, then the specified value is added to the existing property value as an entry
390    *    in a SET or LIST property.
391    *    <br>If name ends with <l>.put</l>, then the specified value is added to the existing property value as a
392    *    key/value pair in a MAP property.
393    *    <br>If name ends with <l>.remove</l>, then the specified value is removed from the existing property property
394    *    value in a SET or LIST property.
395    * @param value
396    *    The new value.
397    *    If <jk>null</jk>, the property value is deleted.
398    *    In general, the value type can be anything.
399    * @return This object (for method chaining).
400    */
401   public PropertyStore setProperty(String name, Object value) {
402      String prefix = prefix(name);
403
404      if (name.endsWith(".add"))
405         return addToProperty(name.substring(0, name.lastIndexOf('.')), value);
406
407      if (name.endsWith(".put"))
408         return putToProperty(name.substring(0, name.lastIndexOf('.')), value);
409
410      if (name.endsWith(".remove"))
411         return removeFromProperty(name.substring(0, name.lastIndexOf('.')), value);
412
413      wl.lock();
414      try {
415         contexts.clear();
416         if (! properties.containsKey(prefix))
417            properties.put(prefix, new PropertyMap(prefix));
418         properties.get(prefix).set(name, value);
419      } finally {
420         wl.unlock();
421      }
422      return this;
423   }

425   /**
426    * Synonym for {@link #setProperty(String, Object)}.
427    *
428    * @param name
429    *    The configuration property name.
430    *    <br>If name ends with <l>.add</l>, then the specified value is added to the existing property value as an entry
431    *    in a SET or LIST property.
432    *    <br>If name ends with <l>.put</l>, then the specified value is added to the existing property value as a
433    *    key/value pair in a MAP property.
434    *    <br>If name ends with <l>.remove</l>, then the specified value is removed from the existing property property
435    *    value in a SET or LIST property.
436    * @param value
437    *    The new value.
438    *    If <jk>null</jk>, the property value is deleted.
439    *    In general, the value type can be anything.
440    * @return This object (for method chaining).
441    */
442   public PropertyStore append(String name, Object value) {
443      return setProperty(name, value);
444   }

446   /**
447    * Convenience method for setting multiple properties in one call.
448    *
449    * <p>
450    * This appends to any previous configuration properties set on this store.
451    *
452    * @param newProperties The new properties to set.
453    * @return This object (for method chaining).
454    */
455   public PropertyStore setProperties(Map<String,Object> newProperties) {
456      if (newProperties == null || newProperties.isEmpty())
457         return this;
458      wl.lock();
459      try {
460         contexts.clear();
461         for (Map.Entry<String,Object> e : newProperties.entrySet()) {
462            String name = e.getKey().toString();
463            Object value = e.getValue();
464            String prefix = prefix(name);
465            if (name.endsWith(".add"))
466               addToProperty(name.substring(0, name.lastIndexOf('.')), value);
467            else if (name.endsWith(".remove"))
468               removeFromProperty(name.substring(0, name.lastIndexOf('.')), value);
469            else {
470               if (! properties.containsKey(prefix))
471                  properties.put(prefix, new PropertyMap(prefix));
472               properties.get(prefix).set(name, value);
473            }
474         }
475
476      } finally {
477         wl.unlock();
478      }
479      return this;
480   }

482   /**
483    * Adds several properties to this store.
484    *
485    * @param properties The properties to add to this store.
486    * @return This object (for method chaining).
487    */
488   public PropertyStore addProperties(Map<String,Object> properties) {
489      if (properties != null)
490         for (Map.Entry<String,Object> e : properties.entrySet())
491            setProperty(e.getKey(), e.getValue());
492      return this;
493   }

562   /**
563    * Removes a value from a SET property.
564    *
565    * @param name The property name.
566    * @param value The property value in the SET property.
567    * @return This object (for method chaining).
568    * @throws ConfigException If property is not a SET property.

571      String prefix = prefix(name);
572      wl.lock();
573      try {
574         contexts.clear();
575         if (properties.containsKey(prefix))
576            properties.get(prefix).removeFrom(name, value);
577      } finally {
578         wl.unlock();
579      }
580      return this;

583   /**
584    * Returns an instance of the specified context initialized with the properties in this store.
585    *
586    * <p>
587    * Multiple calls to this method for the same store class will return the same cached value as long as the
588    * properties on this store are not touched.
589    *
590    * <p>
591    * As soon as any properties are modified on this store, all cached entries are discarded and recreated as needed.
592    *
593    * @param c The context class to instantiate.
594    * @return The context instance.
595    */
596   @SuppressWarnings("unchecked")
597   public <T extends Context> T getContext(Class<T> c) {
598      rl.lock();
599      try {
600         try {
601            if (! contexts.containsKey(c)) {
602
603               // Try to get it from the global cache.
604               Integer key = hashCode();
605               if (! globalContextCache.containsKey(key))
606                  globalContextCache.putIfAbsent(key, new ConcurrentHashMap<Class<? extends Context>,Context>());
607               ConcurrentHashMap<Class<? extends Context>, Context> cacheForThisConfig = globalContextCache.get(key);
608
609               if (! cacheForThisConfig.containsKey(c))
610                  cacheForThisConfig.putIfAbsent(c, newInstance(c, c, this));
611
612               contexts.put(c, cacheForThisConfig.get(c));
613            }
614            return (T)contexts.get(c);
615         } catch (Exception e) {
616            throw new ConfigException("Could not instantiate context class ''{0}''", className(c)).initCause(e);
617         }
618      } finally {
619         rl.unlock();
620      }
621   }

704    * a subclass of the type to be instantiated.
705    *

707    * @param type The class type to convert the property value to.
708    * @param def The type to instantiate if the property is not set.

711    * @throws ConfigException If property has a value that cannot be converted to a boolean.

714   public <T> T getTypedProperty(String name, Class<T> type, Class<? extends T> def, Object...args) {
715      rl.lock();
716      try {
717         Object o = null;
718         PropertyMap pm = getPropertyMap(prefix(name));
719         if (pm != null)

721         if (o == null && def != null)

723         if (o == null)

726            return (T)o;
727         throw new FormattedRuntimeException(
728            "Invalid object of type {0} found in call to PropertyStore.getTypeProperty({1},{2},{3},...)",
729            o.getClass(), name, type, def);
730      } finally {
731         rl.unlock();
732      }

920   /**
921    * Contains all the properties for a particular property prefix (e.g. <js>'BeanContext'</js>)
922    *
923    * <p>
924    * Instances of this map are immutable from outside this class.
925    *
926    * <p>
927    * The {@link PropertyMap#hashCode()} and {@link PropertyMap#equals(Object)} methods can be used to compare with
928    * other property maps.
929    */
930   final class PropertyMap {
931
932      private final Map<String,Property> map = new ConcurrentSkipListMap<>();
933      private volatile int hashCode = 0;
934      private final ReadWriteLock lock = new ReentrantReadWriteLock();
935      private final Lock rl = lock.readLock(), wl = lock.writeLock();
936      private final String prefix;
937
938      PropertyMap(String prefix) {
939         this.prefix = prefix;
940         prefix = prefix + '.';
941         Properties p = System.getProperties();
942         for (Map.Entry<Object,Object> e : p.entrySet())
943            if (e.getKey().toString().startsWith(prefix))
944               set(e.getKey().toString(), e.getValue());
945      }
946
947      /**
948       * Copy constructor.
949       */
950      PropertyMap(PropertyMap orig, PropertyMap apply) {
951         this.prefix = apply.prefix;
952         if (orig != null)
953            for (Map.Entry<String,Property> e : orig.map.entrySet())
954               this.map.put(e.getKey(), Property.create(e.getValue().name, e.getValue().value()));
955         for (Map.Entry<String,Property> e : apply.map.entrySet()) {
956            Property prev = this.map.get(e.getKey());
957            if (prev == null || "SIMPLE".equals(prev.type))
958               this.map.put(e.getKey(), Property.create(e.getValue().name, e.getValue().value()));
959            else {
960               if ("SET".equals(prev.type) || "LIST".equals(prev.type))
961                  prev.add(e.getValue().value());
962               else if ("MAP".equals(prev.type))
963                  prev.put(e.getValue().value());
964            }
965         }
966      }
967
968      /**
969       * Returns the specified property as the specified class type.
970       *
971       * @param name The property name.
972       * @param type The type of object to convert the value to.
973       * @param def The default value if the specified property is not set.
974       * @return The property value.
975       */
976      <T> T get(String name, Class<T> type, T def) {
977         rl.lock();
978         try {
979            Property p = map.get(name);
980            if (p == null || type == null)
981               return def;
982            try {
983               if (! isBeanSessionAvailable())
984                  return def;
985               return getBeanSession().convertToType(p.value, type);
986            } catch (InvalidDataConversionException e) {
987               throw new ConfigException("Could not retrieve property store property ''{0}''.  {1}", p.name,
988                  e.getMessage());
989            }
990         } finally {
991            rl.unlock();
992         }
993      }
994
995      /**
996       * Returns the specified property as a map with the specified key and value types.
997       *
998       * <p>
999       * The map returned is an instance of {@link LinkedHashMap}.
1000       *
1001       * @param name The property name.
1002       * @param keyType The class type of the keys of the map.
1003       * @param valueType The class type of the values of the map.
1004       * @param def The default value if the specified property is not set.
1005       * @return The property value.
1006       */
1007      @SuppressWarnings("unchecked")
1008      <K,V> Map<K,V> getMap(String name, Class<K> keyType, Class<V> valueType, Map<K,V> def) {
1009         rl.lock();
1010         try {
1011            Property p = map.get(name);
1012            if (p == null || keyType == null || valueType == null)
1013               return def;
1014            try {
1015               if (isBeanSessionAvailable()) {
1016                  BeanSession session = getBeanSession();
1017                  return (Map<K,V>)session.convertToType(p.value,
1018                     session.getClassMeta(LinkedHashMap.class, keyType, valueType));
1019               }
1020               return def;
1021            } catch (InvalidDataConversionException e) {
1022               throw new ConfigException("Could not retrieve property store property ''{0}''.  {1}", p.name,
1023                  e.getMessage());
1024            }
1025         } finally {
1026            rl.unlock();
1027         }
1028      }
1029
1030      /**
1031       * Convenience method for returning all values in this property map as a simple map.
1032       *
1033       * <p>
1034       * Primarily useful for debugging.
1035       *
1036       * @return A new {@link LinkedHashMap} with all values in this property map.
1037       */
1038      Map<String,Object> asMap() {
1039         rl.lock();
1040         try {
1041            Map<String,Object> m = new LinkedHashMap<>();
1042            for (Property p : map.values())
1043               m.put(p.name, p.value);
1044            return m;
1045         } finally {
1046            rl.unlock();
1047         }
1048      }
1049
1050      void set(String name, Object value) {
1051         wl.lock();
1052         hashCode = 0;
1053         try {
1054            if (value == null)
1055               map.remove(name);
1056            else
1057               map.put(name, Property.create(name, value));
1058         } finally {
1059            wl.unlock();
1060         }
1061      }
1062
1063      void addTo(String name, Object value) {
1064         wl.lock();
1065         hashCode = 0;
1066         try {
1067            if (! map.containsKey(name))
1068               map.put(name, Property.create(name, Collections.emptyList()));
1069            map.get(name).add(value);
1070         } finally {
1071            wl.unlock();
1072         }
1073      }
1074
1075      void putTo(String name, Object key, Object value) {
1076         wl.lock();
1077         hashCode = 0;
1078         try {
1079            if (! map.containsKey(name))
1080               map.put(name, Property.create(name, Collections.emptyMap()));
1081            map.get(name).put(key, value);
1082         } finally {
1083            wl.unlock();
1084         }
1085      }
1086
1087      void putTo(String name, Object value) {
1088         wl.lock();
1089         hashCode = 0;
1090         try {
1091            if (! map.containsKey(name))
1092               map.put(name, Property.create(name, Collections.emptyMap()));
1093            map.get(name).put(value);
1094         } finally {
1095            wl.unlock();
1096         }
1097      }
1098
1099      void removeFrom(String name, Object value) {
1100         wl.lock();
1101         hashCode = 0;
1102         try {
1103            if (map.containsKey(name))
1104               map.get(name).remove(value);
1105         } finally {
1106            wl.unlock();
1107         }
1108      }

1110      @Override
1111      public int hashCode() {
1112         rl.lock();
1113         try {
1114            if (hashCode == 0) {
1115               HashCode c = new HashCode().add(prefix);
1116               for (Property p : map.values())
1117                  c.add(p);
1118               this.hashCode = c.get();
1119            }
1120            return hashCode;
1121         } finally {
1122            rl.unlock();
1123         }
1124      }

1128         rl.lock();
1129         try {
1130            if (o instanceof PropertyMap) {
1131               PropertyMap m = (PropertyMap)o;
1132               if (m.hashCode() != hashCode())
1133                  return false;
1134               return this.map.equals(m.map);
1135            }
1136            return false;
1137         } finally {
1138            rl.unlock();
1139         }
1140      }

1148   private abstract static class Property {
1149      final String name, type;
1150      final Object value;
1151
1152      static final Property create(String name, Object value) {
1153         if (name.endsWith(".set"))
1154            return new SetProperty(name, value);
1155         else if (name.endsWith(".list"))
1156            return new ListProperty(name, value);
1157         else if (name.endsWith(".map"))
1158            return new MapProperty(name, value);
1159         return new SimpleProperty(name, value);
1160      }
1161
1162      Property(String name, String type, Object value) {
1163         this.name = name;
1164         this.type = type;
1165         this.value = value;
1166      }
1167
1168      void add(Object val) {
1169         throw new ConfigException("Cannot add value {0} ({1}) to property ''{2}'' ({3}).",
1170            JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type);
1171      }
1172
1173      void remove(Object val) {
1174         throw new ConfigException("Cannot remove value {0} ({1}) from property ''{2}'' ({3}).",
1175            JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type);
1176      }
1177
1178      void put(Object val) {
1179         throw new ConfigException("Cannot put value {0} ({1}) to property ''{2}'' ({3}).",
1180            JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type);
1181      }
1182
1183      void put(Object key, Object val) {
1184         throw new ConfigException("Cannot put value {0}({1})->{2}({3}) to property ''{4}'' ({5}).",
1185            JsonSerializer.DEFAULT_LAX.toString(key), getReadableClassNameForObject(key),
1186            JsonSerializer.DEFAULT_LAX.toString(val), getReadableClassNameForObject(val), name, type);
1187      }
1188
1189      protected Object value() {
1190         return value;
1191      }
1192
1193      @Override /* Object */
1194      public int hashCode() {
1195         HashCode c = new NormalizingHashCode().add(name);
1196         if (value instanceof Map) {
1197            for (Map.Entry<?,?> e : ((Map<?,?>)value).entrySet())
1198               c.add(e.getKey()).add(e.getValue());
1199         } else if (value instanceof Collection) {
1200            for (Object o : (Collection<?>)value)
1201               c.add(o);
1202         } else {
1203            c.add(value);
1204         }
1205         return c.get();
1206      }

1208      @Override
1209      public String toString() {
1210         return "Property(name="+name+",type="+type+")";
1211      }

1221   @SuppressWarnings({"unchecked"})
1222   private static final class SetProperty extends Property {
1223      private final Set<Object> value;
1224
1225      SetProperty(String name, Object value) {
1226         super(name, "SET", new ConcurrentSkipListSet<>(PROPERTY_COMPARATOR));
1227         this.value = (Set<Object>)value();
1228         add(value);
1229      }
1230
1231      @Override
1232      void add(Object val) {
1233         if (val.getClass().isArray())
1234            for (int i = 0; i < Array.getLength(val); i++)
1235               add(Array.get(val, i));
1236         else if (val instanceof Collection)
1237            for (Object o : (Collection<Object>)val)
1238               add(o);
1239         else {
1240            if (val instanceof String) {
1241               String s = val.toString();
1242               if (s.startsWith("[") && s.endsWith("]")) {
1243                  try {
1244                     add(new ObjectList(s));
1245                     return;
1246                  } catch (Exception e) {}
1247               }
1248            }
1249            for (Object o : value)
1250               if (same(val, o))
1251                  return;
1252            value.add(val);
1253         }
1254      }
1255
1256      @Override
1257      void remove(Object val) {
1258         if (val.getClass().isArray())
1259            for (int i = 0; i < Array.getLength(val); i++)
1260               remove(Array.get(val, i));
1261         else if (val instanceof Collection)
1262            for (Object o : (Collection<Object>)val)
1263               remove(o);
1264         else {
1265            if (val instanceof String) {
1266               String s = val.toString();
1267               if (s.startsWith("[") && s.endsWith("]")) {
1268                  try {
1269                     remove(new ObjectList(s));
1270                     return;
1271                  } catch (Exception e) {}
1272               }
1273            }
1274            for (Iterator<Object> i = value.iterator(); i.hasNext();)
1275               if (same(i.next(), val))
1276                  i.remove();
1277         }
1278      }

1281   @SuppressWarnings({"unchecked"})
1282   private static final class ListProperty extends Property {
1283      private final LinkedList<Object> value;
1284
1285      ListProperty(String name, Object value) {
1286         super(name, "LIST", new LinkedList<>());
1287         this.value = (LinkedList<Object>)value();
1288         add(value);
1289      }
1290
1291      @Override
1292      void add(Object val) {
1293         if (val.getClass().isArray()) {
1294            for (int i = Array.getLength(val) - 1; i >= 0; i--)
1295               add(Array.get(val, i));
1296         } else if (val instanceof List) {
1297            List<Object> l = (List<Object>)val;
1298            for (ListIterator<Object> i = l.listIterator(l.size()); i.hasPrevious();)
1299               add(i.previous());
1300         } else if (val instanceof Collection) {
1301            List<Object> l = new ArrayList<>((Collection<Object>)val);
1302            for (ListIterator<Object> i = l.listIterator(l.size()); i.hasPrevious();)
1303               add(i.previous());
1304         } else {
1305            String s = val.toString();
1306            if (s.startsWith("[") && s.endsWith("]")) {
1307               try {
1308                  add(new ObjectList(s));
1309                  return;
1310               } catch (Exception e) {}
1311            }
1312            for (Iterator<Object> i = value.iterator(); i.hasNext(); )
1313               if (same(val, i.next()))
1314                  i.remove();
1315            value.addFirst(val);
1316         }
1317      }
1318
1319      @Override
1320      void remove(Object val) {
1321         if (val.getClass().isArray())
1322            for (int i = 0; i < Array.getLength(val); i++)
1323               remove(Array.get(val, i));
1324         else if (val instanceof Collection)
1325            for (Object o : (Collection<Object>)val)
1326               remove(o);
1327         else {
1328            String s = val.toString();
1329            if (s.startsWith("[") && s.endsWith("]")) {
1330               try {
1331                  remove(new ObjectList(s));
1332                  return;
1333               } catch (Exception e) {}
1334            }
1335            for (Iterator<Object> i = value.iterator(); i.hasNext();)
1336               if (same(i.next(), val))
1337                  i.remove();
1338         }
1339      }
1340   }

1342   @SuppressWarnings({"unchecked","rawtypes"})
1343   private static final class MapProperty extends Property {
1344      final Map<Object,Object> value;
1345
1346      MapProperty(String name, Object value) {
1347         // ConcurrentSkipListMap doesn't support Map.Entry.remove(), so use TreeMap instead.
1348         super(name, "MAP", Collections.synchronizedMap(new TreeMap<>(PROPERTY_COMPARATOR)));
1349         this.value = (Map<Object,Object>)value();
1350         put(value);
1351      }
1352
1353      @Override
1354      void put(Object val) {
1355         try {
1356            if (isBeanSessionAvailable() && ! (val instanceof Map))
1357               val = getBeanSession().convertToType(val, Map.class);
1358            if (val instanceof Map) {
1359               Map m = (Map)val;
1360               for (Map.Entry e : (Set<Map.Entry>)m.entrySet())
1361                  put(e.getKey(), e.getValue());
1362               return;
1363            }
1364         } catch (Exception e) {}
1365         super.put(val);
1366      }
1367
1368      @Override
1369      void put(Object key, Object val) {
1370         // ConcurrentSkipListMap doesn't support Map.Entry.remove().
1371         for (Map.Entry<Object,Object> e : value.entrySet()) {
1372            if (same(e.getKey(), key)) {
1373               e.setValue(val);
1374               return;
1375            }
1376         }
1377         value.put(key, val);
1378      }
1379   }

1412   /*
1413    * Compares two objects for "string"-equality.
1414    * Basically mean both objects are equal if they're the same when converted to strings.
1415    */
1416   @SuppressWarnings({ "rawtypes" })
1417   static boolean same(Object o1, Object o2) {
1418      if (o1 == o2)
1419         return true;
1420      if (o1 instanceof Map) {
1421         if (o2 instanceof Map) {
1422            Map m1 = (Map)o1, m2 = (Map)o2;
1423            if (m1.size() == m2.size()) {
1424               Set<Map.Entry> s1 = m1.entrySet(), s2 = m2.entrySet();
1425               for (Iterator<Map.Entry> i1 = s1.iterator(), i2 = s2.iterator(); i1.hasNext();) {
1426                  Map.Entry e1 = i1.next(), e2 = i2.next();
1427                  if (! same(e1.getKey(), e2.getKey()))
1428                     return false;
1429                  if (! same(e1.getValue(), e2.getValue()))
1430                     return false;
1431               }
1432               return true;
1433            }
1434         }
1435         return false;
1436      } else if (o1 instanceof Collection) {
1437         if (o2 instanceof Collection) {
1438            Collection c1 = (Collection)o1, c2 = (Collection)o2;
1439            if (c1.size() == c2.size()) {
1440               for (Iterator i1 = c1.iterator(), i2 = c2.iterator(); i1.hasNext();) {
1441                  if (! same(i1.next(), i2.next()))
1442                     return false;
1443               }
1444               return true;
1445            }
1446         }
1447         return false;
1448      } else {
1449         return unswap(o1).equals(unswap(o2));
1450      }
1451   }
