013package org.apache.juneau.plaintext;
014
015import org.apache.juneau.*;
016import org.apache.juneau.serializer.*;
017import org.apache.juneau.transform.*;
018
019/**
020 * Serializes POJOs to plain text using just the <code>toString()</code> method on the serialized object.
021 *
022 * <h5 class='section'>Media types:</h5>
023 *
024 * Handles <code>Accept</code> types: <code>text/plain</code>
025 *
026 * <p>
027 * Produces <code>Content-Type</code> types: <code>text/plain</code>
028 *
029 * <h5 class='section'>Description:</h5>
030 *
031 * Essentially converts POJOs to plain text using the <code>toString()</code> method.
032 *
033 * <p>
034 * Also serializes objects using a transform if the object class has an {@link PojoSwap PojoSwap&lt;?,String&gt;}
035 * transform defined on it.
036 */
037public class PlainTextSerializer extends WriterSerializer {
038
039   //-------------------------------------------------------------------------------------------------------------------
040   // Predefined instances
041   //-------------------------------------------------------------------------------------------------------------------
042
043   /** Default serializer, all default settings.*/
044   public static final PlainTextSerializer DEFAULT = new PlainTextSerializer(PropertyStore.create());
045
046
047   //-------------------------------------------------------------------------------------------------------------------
048   // Instance
049   //-------------------------------------------------------------------------------------------------------------------
050
051   /**
052    * Constructor.
053    *
054    * @param propertyStore
055    *    The property store containing all the settings for this object.
056    */
057   public PlainTextSerializer(PropertyStore propertyStore) {
058      this(propertyStore, "text/plain");
059   }
060
061   /**
062    * Constructor.
063    *
064    * @param propertyStore
065    *    The property store containing all the settings for this object.
066    * @param produces
067    *    The media type that this serializer produces.
068    * @param accept
069    *    The accept media types that the serializer can handle.
070    *    <p>
071    *    Can contain meta-characters per the <code>media-type</code> specification of
072    *    <a class="doclink" href="http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1">RFC2616/14.1</a>
073    *    <p>
074    *    If empty, then assumes the only media type supported is <code>produces</code>.
075    *    <p>
076    *    For example, if this serializer produces <js>"application/json"</js> but should handle media types of
077    *    <js>"application/json"</js> and <js>"text/json"</js>, then the arguments should be:
078    *    <br><code><jk>super</jk>(propertyStore, <js>"application/json"</js>, <js>"application/json"</js>, <js>"text/json"</js>);</code>
079    *    <br>...or...
080    *    <br><code><jk>super</jk>(propertyStore, <js>"application/json"</js>, <js>"*&#8203;/json"</js>);</code>
081    */
082   public PlainTextSerializer(PropertyStore propertyStore, String produces, String...accept) {
083      super(propertyStore, produces, accept);
084   }
085
086
087   @Override /* CoreObject */
088   public PlainTextSerializerBuilder builder() {
089      return new PlainTextSerializerBuilder(getPropertyStore());
090   }
091
092   /**
093    * Instantiates a new clean-slate {@link PlainTextSerializerBuilder} object.
094    * 
095    * <p>
096    * This is equivalent to simply calling <code><jk>new</jk> PlainTextSerializerBuilder()</code>.
097    * 
098    * <p>
099    * Note that this method creates a builder initialized to all default settings, whereas {@link #builder()} copies 
100    * the settings of the object called on.
101    * 
102    * @return A new {@link PlainTextSerializerBuilder} object.
103    */
104   public static PlainTextSerializerBuilder create() {
105      return new PlainTextSerializerBuilder();
106   }
107
108   @Override /* Serializer */
109   public WriterSerializerSession createSession(SerializerSessionArgs args) {
110      return new PlainTextSerializerSession(this, args);
111   }
112}
