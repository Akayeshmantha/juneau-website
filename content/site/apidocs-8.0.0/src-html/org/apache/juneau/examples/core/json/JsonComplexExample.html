003import org.apache.juneau.examples.core.pojo.*;
004import org.apache.juneau.json.JsonParser;
005import org.apache.juneau.json.JsonSerializer;
006
007import java.io.ByteArrayOutputStream;

009import java.util.ArrayList;

011import java.util.List;
012
013/**

016public class JsonComplexExample {
017    /**
018     * TODO
019     *
020     * @param args
021     * @throws Exception
022     */
023    public static void main(String[] args) throws Exception{
024        // Juneau provides static constants with the most commonly used configurations
025        // Get a reference to a serializer - converting POJO to flat format
026        JsonSerializer jsonSerializer = JsonSerializer.DEFAULT;
027        // Get a reference to a parser - converts that flat format back into the POJO
028        JsonParser jsonParser = JsonParser.DEFAULT;
029
030        // Fill some data to a PojoComplex bean
031        HashMap<String, List<Pojo>> values = new HashMap<>();
032        ArrayList<Pojo> setOne = new ArrayList<>();
033        setOne.add(new Pojo("1.1", "name1"));
034        setOne.add(new Pojo("1.1", "name2"));
035        ArrayList<Pojo> setTwo = new ArrayList<>();
036        setTwo.add(new Pojo("1.2", "name1"));
037        setTwo.add(new Pojo("1.2", "name2"));
038        values.put("setOne", setOne);
039        values.put("setTwo", setTwo);
040        PojoComplex pojoc = new PojoComplex("pojo", new Pojo("1.0", "name0"), values);
041
042        String flat = jsonSerializer.serialize(pojoc);
043
044        // Print out the created POJO in JSON format.
045        System.out.println(flat);
046
047        PojoComplex parse = jsonParser.parse(flat, PojoComplex.class);
048
049        assert parse.getId().equals(pojoc.getId());
050        assert parse.getInnerPojo().getName().equals(pojoc.getInnerPojo().getName());
051        assert parse.getInnerPojo().getId().equals(pojoc.getInnerPojo().getId());
052
053        // The object above can be parsed thanks to the @BeanConstructor(properties = id,name) annotation on Pojo
054        // Using this approach, you can keep your POJOs immutable, and still serialize and deserialize them.
055    }
056}
