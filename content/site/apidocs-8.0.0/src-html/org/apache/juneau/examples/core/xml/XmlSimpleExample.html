024     */
025    public static void main(String[] args) throws SerializeException, ParseException {
026
027        // Fill some data to a Pojo bean
028        Pojo pojo = new Pojo("id","name");
029
030        // Serialize to human readable XML and print
031        String serial = XmlSerializer.DEFAULT_SQ_READABLE.serialize(pojo);
032        System.out.println(serial);
033
034        // Deserialize back to PojoComplex instance
035        Pojo obj = XmlParser.DEFAULT.parse(serial, Pojo.class);
036
037        assert obj.getId().equals(pojo.getId());
038        assert obj.getName().equals(pojo.getName());
039
040        // The object above can be parsed thanks to the @BeanConstructor annotation on PojoComplex
041        // Using this approach, you can keep your POJOs immutable, and still serialize and deserialize them.
042
043    }
