13  package org.apache.juneau;
14  
15  import static org.apache.juneau.TestUtils.*;
16  import static org.junit.Assert.*;
17  
18  import org.apache.juneau.annotation.*;
19  import org.junit.*;
20  
21  
22  @SuppressWarnings({"rawtypes","javadoc"})
23  public class AnnotationsTest {
24  
25  	//====================================================================================================
26  	// Bean with explicitly specified properties.
27  	//====================================================================================================
28  	@Test
29  	public void testBeanWithExplicitProperties() throws Exception {
30  		BeanContext bc = BeanContext.DEFAULT;
31  		BeanMap bm = null;
32  
33  		// Basic test
34  		bm = bc.newBeanMap(Person1.class).load("{age:21,name:'foobar'}");
35  		assertNotNull(bm);
36  		assertNotNull(bm.getBean());
37  		assertEquals(bm.get("age"), 21);
38  		assertEquals(bm.get("name"), "foobar");
39  
40  		bm.put("age", 65);
41  		bm.put("name", "futbol");
42  		assertEquals(bm.get("age"), 65);
43  		assertEquals(bm.get("name"), "futbol");
44  	}
45  
46  	/** Class with explicitly specified properties */
47  	@Bean(properties = { "age", "name" })
48  	public static class Person1 {
49  		public int age;
50  		private String name;
51  		public String getName() {
52  			return name;
53  		}
54  		public void setName(String name) {
55  			this.name = name;
56  		}
57  	}
58  
59  	//====================================================================================================
60  	// Private/protected/default fields should be ignored.
61  	//====================================================================================================
62  	@Test
63  	public void testForOnlyPublicFields() throws Exception {
64  		BeanContext bc = BeanContext.DEFAULT;
65  		BeanMap bm = null;
66  
67  		// Make sure only public fields are detected
68  		bm = bc.newBeanMap(A.class).load("{publicField:123}");
69  		assertNotNull("F1", bm);
70  		assertNotNull("F2", bm.getBean());
71  		assertObjectEquals("{publicField:123}", bm.getBean());
72  
73  	}
74  
75  	public static class A {
76  		public int publicField;
77  		protected int protectedField;
78  		@SuppressWarnings("unused")
79  		private int privateField;
80  		int defaultField;
81  	}
82  }
