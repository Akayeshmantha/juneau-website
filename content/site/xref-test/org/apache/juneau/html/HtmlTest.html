33  	// Verifies that lists of maps/beans are converted to tables correctly.
34  	//====================================================================================================
35  	@Test
36  	public void testTables1() throws Exception {
37  		HtmlSerializer s = HtmlSerializer.DEFAULT_SQ;
38  		Object[] t;
39  		String html;
40  
41  		t = new Object[] {new A1(), new A1()};
42  		html = s.serialize(t);
43  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
44  
45  		t = new Object[] {new A1(), new A2()};
46  		html = s.serialize(t);
47  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
48  
49  		t = new Object[] {new A1(), new ObjectMap("{f1:'f1'}")};
50  		html = s.serialize(t);
51  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
52  
53  		t = new Object[] {new ObjectMap("{f1:'f1'}"), new A1()};
54  		html = s.serialize(t);
55  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
56  
57  		// This should be serialized as a list since the objects have different properties.
58  		t = new Object[] {new A1(), new ObjectMap("{f2:'f2'}")};
59  		html = s.serialize(t);
60  		assertEquals("<ul><li><table _type='object'><tr><th><string>key</string></th><th><string>value</string></th></tr><tr><td><string>f1</string></td><td><string>f1</string></td></tr></table></li><li><table _type='object'><tr><th><string>key</string></th><th><string>value</string></th></tr><tr><td><string>f2</string></td><td><string>f2</string></td></tr></table></li></ul>", html);
61  
62  		// Tables with some beans with @Bean#properties annotations.
63  		t = new Object[] {new A1(), new A3()};
64  		html = s.serialize(t);
65  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
66  
67  		t = new Object[] {new A3(), new A1()};
68  		html = s.serialize(t);
69  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
70  
71  		// Tables with some beans with @Bean#transforms annotations.
72  		t = new Object[] {new A4(), new A1()};
73  		html = s.serialize(t);
74  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
75  
76  		t = new Object[] {new A1(), new A4()};
77  		html = s.serialize(t);
78  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
79  
80  		t = new Object[] {new A5(), new A1()};
81  		html = s.serialize(t);
82  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
83  
84  		t = new Object[] {new A1(), new A5()};
85  		html = s.serialize(t);
86  		assertEquals("<table _type='array'><tr><th>f1</th></tr><tr><td><string>f1</string></td></tr><tr><td><string>f1</string></td></tr></table>", html);
87  	}

301 	private String strip(String html) {
302 		return html
