191 	public RoundTripTest(String label, Serializer s, Parser p, int flags) throws Exception {
192 		this.s = s.clone().addBeanFilters(getBeanFilters()).addPojoSwaps(getPojoSwaps()).addToDictionary(getDictionary());
193 		this.p = p == null ? null : p.clone().addBeanFilters(getBeanFilters()).addPojoSwaps(getPojoSwaps()).addToDictionary(getDictionary());
194 		this.label = label;
195 
196 		Map<Class<Object>, Class<? extends Object>> m = getImplClasses();
197 		if (m != null) {
198 			for (Entry<Class<Object>, Class<? extends Object>> e : m.entrySet()) {
199 				this.s.addImplClass(e.getKey(), e.getValue());
200 				if (this.p != null)
201 					this.p.addImplClass(e.getKey(), e.getValue());
202 			}
203 		}
204 		this.validateXmlWhitespace = (flags & CHECK_XML_WHITESPACE) > 0;
205 		this.validateXml = (flags & VALIDATE_XML) > 0;
206 		this.returnOriginalObject = (flags & RETURN_ORIGINAL_OBJECT) > 0;
207 	}

226 	public <T> T roundTrip(T object, ClassMeta<? extends T> t) throws Exception {
227 		Object out = serialize(object, this.s);
228 		if (p == null)
229 			return object;
230 		T o = this.p.parse(out, t);
231 		return (returnOriginalObject ? object : o);
232 	}
233 	public <T> T roundTrip(T object, Class<? extends T> c) throws Exception {
234 		Object out = serialize(object, this.s);
235 		if (p == null)
236 			return object;
237 		T o = this.p.parse(out, p.getBeanContext().getClassMeta(c));
238 		return (returnOriginalObject ? object : o);
239 	}

261 	public <T> T roundTrip(T object, Serializer serializer, Parser parser) throws Exception {
262 		Object out = serialize(object, serializer);
263 		if (parser == null)
264 			return object;
265 		T o = (T)parser.parse(out,  object == null ? Object.class : object.getClass());
266 		return (returnOriginalObject ? object : o);
267 	}

299 	public <T> Object serialize(T object, Serializer s) throws Exception {
300 
301 		Object out = null;
302 		if (s.isWriterSerializer())
303 			out = ((WriterSerializer)s).serialize(object);
304 		else {
305 			out = ((OutputStreamSerializer)s).serialize(object);
306 		}
307 
308 		if (debug)
309 			System.err.println("Serialized contents from ["+label+"]...\n---START---\n" + (out instanceof byte[] ? TestUtils.toReadableBytes((byte[])out) : out) + "\n---END---\n");
310 
311 		if (validateXmlWhitespace)
312 			TestUtils.checkXmlWhitespace(out.toString());
313 
314 		if (validateXml)
315 			TestUtils.validateXml(object, (XmlSerializer)s);
316 
317 		return out;
318 	}
