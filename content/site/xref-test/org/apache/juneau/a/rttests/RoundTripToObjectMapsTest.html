24  /**
25   * Tests designed to serialize and parse objects to make sure we end up
26   * with the same objects for all serializers and parsers.
27   */
28  @SuppressWarnings("javadoc")
29  public class RoundTripToObjectMapsTest extends RoundTripTest {
30  
31  	public RoundTripToObjectMapsTest(String label, Serializer s, Parser p, int flags) throws Exception {
32  		super(label, s, p, flags);
33  	}
34  
35  	//====================================================================================================
36  	// Class with X(ObjectMap) constructor and toObjectMap() method.
37  	//====================================================================================================
38  	@SuppressWarnings({ "serial", "unchecked" })
39  	@Test
40  	public void test() throws Exception {
41  		A a = new A(new ObjectMap("{f1:'a',f2:2}"));
42  		a = roundTrip(a, A.class);
43  		assertEquals("a", a.f1);
44  		assertEquals(2, a.f2);
45  
46  		A[] aa = new A[]{a};
47  		aa = roundTrip(aa, A[].class);
48  		assertEquals(1, aa.length);
49  		assertEquals("a", aa[0].f1);
50  		assertEquals(2, aa[0].f2);
51  
52  		List<A> a2 = new ArrayList<A>(){{add(new A(new ObjectMap("{f1:'a',f2:2}")));}};
53  		a2 = roundTrip(a2, BeanContext.DEFAULT.getCollectionClassMeta(List.class, A.class));
54  		assertEquals(1, a2.size());
55  		assertEquals("a", a2.get(0).f1);
56  		assertEquals(2, a2.get(0).f2);
57  
58  		Map<String,A> a3 = new LinkedHashMap<String,A>(){{put("a", new A(new ObjectMap("{f1:'a',f2:2}")));}};
59  		a3 = roundTrip(a3, BeanContext.DEFAULT.getMapClassMeta(Map.class, String.class, A.class));
60  		assertEquals(1, a3.size());
61  		assertEquals("a", a3.get("a").f1);
62  		assertEquals(2, a3.get("a").f2);
63  	}
64  
65  	public static class A {
66  		private String f1;
67  		private int f2;
68  		public A(ObjectMap m) {
69  			this.f1 = m.getString("f1");
70  			this.f2 = m.getInt("f2");
71  		}
72  		public ObjectMap toObjectMap() {
73  			return new ObjectMap().append("f1",f1).append("f2",f2);
74  		}
75  	}
76  
77  }
