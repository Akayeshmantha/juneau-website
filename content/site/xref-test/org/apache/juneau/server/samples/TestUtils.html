70  	 * Verifies that two objects are equivalent.
71  	 * Does this by doing a string comparison after converting both to JSON.
72  	 * @param sort If <jk>true</jk> sort maps and collections before comparison.
73  	 */
74  	public static void assertEqualObjects(Object o1, Object o2, boolean sort) throws SerializeException {
75  		JsonSerializer s = (sort ? jsSorted : js);
76  		String s1 = s.serialize(o1);
77  		String s2 = s.serialize(o2);
78  		if (s1.equals(s2))
79  			return;
80  		throw new ComparisonFailure(null, s1, s2);
81  	}

83  	/**
84  	 * Validates that the whitespace is correct in the specified XML.
85  	 */
86  	public static void checkXmlWhitespace(String out) throws SerializeException {
87  		if (out.indexOf('\u0000') != -1) {
88  			for (String s : out.split("\u0000"))
89  				checkXmlWhitespace(s);
90  			return;
91  		}
92  
93  		int indent = -1;
94  		Pattern startTag = Pattern.compile("^(\\s*)<[^/>]+(\\s+\\S+=['\"]\\S*['\"])*\\s*>$");
95  		Pattern endTag = Pattern.compile("^(\\s*)</[^>]+>$");
96  		Pattern combinedTag = Pattern.compile("^(\\s*)<[^>/]+(\\s+\\S+=['\"]\\S*['\"])*\\s*/>$");
97  		Pattern contentOnly = Pattern.compile("^(\\s*)[^\\s\\<]+$");
98  		Pattern tagWithContent = Pattern.compile("^(\\s*)<[^>]+>.*</[^>]+>$");
99  		String[] lines = out.split("\n");
100 		try {
101 			for (int i = 0; i < lines.length; i++) {
102 				String line = lines[i];
103 				Matcher m = startTag.matcher(line);
104 				if (m.matches()) {
105 					indent++;
106 					if (m.group(1).length() != indent)
107 						throw new SerializeException("Wrong indentation detected on start tag line ''{0}''", i+1);
108 					continue;
109 				}
110 				m = endTag.matcher(line);
111 				if (m.matches()) {
112 					if (m.group(1).length() != indent)
113 						throw new SerializeException("Wrong indentation detected on end tag line ''{0}''", i+1);
114 					indent--;
115 					continue;
116 				}
117 				m = combinedTag.matcher(line);
118 				if (m.matches()) {
119 					indent++;
120 					if (m.group(1).length() != indent)
121 						throw new SerializeException("Wrong indentation detected on combined tag line ''{0}''", i+1);
122 					indent--;
123 					continue;
124 				}
125 				m = contentOnly.matcher(line);
126 				if (m.matches()) {
127 					indent++;
128 					if (m.group(1).length() != indent)
129 						throw new SerializeException("Wrong indentation detected on content-only line ''{0}''", i+1);
130 					indent--;
131 					continue;
132 				}
133 				m = tagWithContent.matcher(line);
134 				if (m.matches()) {
135 					indent++;
136 					if (m.group(1).length() != indent)
137 						throw new SerializeException("Wrong indentation detected on tag-with-content line ''{0}''", i+1);
138 					indent--;
139 					continue;
140 				}
141 				throw new SerializeException("Unmatched whitespace line at line number ''{0}''", i+1);
142 			}
143 			if (indent != -1)
144 				throw new SerializeException("Possible unmatched tag.  indent=''{0}''", indent);
145 		} catch (SerializeException e) {
146 			printLines(lines);
147 			throw e;
148 		}
149 	}

153 			System.err.println(String.format("%4s:" + lines[i], i+1));

156 	/**
157 	 * Validates that the specified XML conforms to the specified schema.
158 	 */
159 	private static void validateXml(String xml, String xmlSchema) throws Exception {
160 		// parse an XML document into a DOM tree
161 		DocumentBuilderFactory f = DocumentBuilderFactory.newInstance();
162 		f.setNamespaceAware(true);
163 		DocumentBuilder documentBuilder = f.newDocumentBuilder();
164 		Document document = documentBuilder.parse(new InputSource(new StringReader(xml)));
165 
166 		// create a SchemaFactory capable of understanding WXS schemas
167 		SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
168 
169 		if (xmlSchema.indexOf('\u0000') != -1) {
170 
171 			// Break it up into a map of namespaceURI->schema document
172 			final Map<String,String> schemas = new HashMap<String,String>();
173 			String[] ss = xmlSchema.split("\u0000");
174 			xmlSchema = ss[0];
175 			for (String s : ss) {
176 				Matcher m = pTargetNs.matcher(s);
177 				if (m.find())
178 					schemas.put(m.group(1), s);
179 			}
180 
181 			// Create a custom resolver
182 			factory.setResourceResolver(
183 				new LSResourceResolver() {
184 
185 					@Override /* LSResourceResolver */
186 					public LSInput resolveResource(String type, String namespaceURI, String publicId, String systemId, String baseURI) {
187 
188 						String schema = schemas.get(namespaceURI);
189 						if (schema == null)
190 							throw new RuntimeException(MessageFormat.format("No schema found for namespaceURI ''{0}''", namespaceURI));
191 
192 						try {
193 							DOMImplementationRegistry registry = DOMImplementationRegistry.newInstance();
194 							DOMImplementationLS domImplementationLS = (DOMImplementationLS)registry.getDOMImplementation("LS 3.0");
195 							LSInput in = domImplementationLS.createLSInput();
196 							in.setCharacterStream(new StringReader(schema));
197 							in.setSystemId(systemId);
198 							return in;
199 
200 						} catch (Exception e) {
201 							throw new RuntimeException(e);
202 						}
203 					}
204 				}
205 			);
206 		}
207 
208 		Schema schema = factory.newSchema(new StreamSource(new StringReader(xmlSchema)));
209 
210 		// create a Validator instance, which can be used to validate an instance document
211 		Validator validator = schema.newValidator();
212 
213 		// validate the DOM tree
214 		validator.validate(new DOMSource(document));
215 	}

224 	 * Test whitespace and generated schema.
225 	 */
226 	public static void validateXml(Object o, XmlSerializer s) throws Exception {
227 		s = s.clone().setProperty(SERIALIZER_useIndentation, true).setProperty(XML_enableNamespaces, true).setProperty(XML_addNamespaceUrisToRoot, true);
228 		String xml = s.serialize(o);
229 
230 		String xmlSchema = null;
231 		try {
232 			xmlSchema = s.getSchemaSerializer().serialize(o);
233 			TestUtils.checkXmlWhitespace(xml);
234 			TestUtils.checkXmlWhitespace(xmlSchema);
235 			TestUtils.validateXml(xml, xmlSchema);
236 		} catch (Exception e) {
237 			System.err.println("---XML---");
238 			System.err.println(xml);
239 			System.err.println("---XMLSchema---");
240 			System.err.println(xmlSchema);
241 			throw e;
242 		}
243 	}

255 	final protected static char[] hexArray = "0123456789ABCDEF".toCharArray();
256 	public static String toHex(byte b) {
257 	    char[] c = new char[2];
258 	    int v = b & 0xFF;
259 	    c[0] = hexArray[v >>> 4];
260 	    c[1] = hexArray[v & 0x0F];
261 	    return new String(c);
262 	}

272 	/**
273 	 * Sort an XML document by element and attribute names.
274 	 * This method is primarily meant for debugging purposes.
275 	 */
276 	private static final String sortXml(String xml) throws Exception {
277 
278 		xml = xml.replaceAll("\\w+\\:", "");  // Strip out all namespaces.
279 
280 		DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
281 		dbf.setIgnoringElementContentWhitespace(true);
282 		dbf.setNamespaceAware(false);
283 		DocumentBuilder db = dbf.newDocumentBuilder();
284 		Document doc = db.parse(new InputSource(new StringReader(xml)));
285 
286 		DOMSource s = new DOMSource(doc);
287 
288 		StringWriter sw = new StringWriter();
289 		StreamResult sr = new StreamResult(sw);
290 		XML_SORT_TRANSFORMER.transform(s, sr);
291 		return sw.toString().replace('"', '\'').replace("\r", "");
292 	}

303 	private static Transformer XML_SORT_TRANSFORMER;
304 	static {
305 		try {
306 			String xsl = ""
307 				+ "	<xsl:stylesheet version='1.0'"
308 				+ "	 xmlns:xsl='http://www.w3.org/1999/XSL/Transform'>"
309 				+ "	 <xsl:output omit-xml-declaration='yes' indent='yes'/>"
310 				+ "	 <xsl:strip-space elements='*'/>"
311 				+ "	 <xsl:template match='node()|@*'>"
312 				+ "	  <xsl:copy>"
313 				+ "	   <xsl:apply-templates select='@*'>"
314 				+ "	    <xsl:sort select='name()'/>"
315 				+ "	   </xsl:apply-templates>"
316 				+ "	   <xsl:apply-templates select='node()'>"
317 				+ "	    <xsl:sort select='name()'/>"
318 				+ "	    <xsl:sort select='text()'/>"
319 				+ "	   </xsl:apply-templates>"
320 				+ "	  </xsl:copy>"
321 				+ "	 </xsl:template>"
322 				+ "	</xsl:stylesheet>";
323 			TransformerFactory tf = TransformerFactory.newInstance();
324 			StreamSource ss = new StreamSource(new StringReader(xsl));
325 			XML_SORT_TRANSFORMER = tf.newTransformer(ss);
326 		} catch (Exception e) {
327 			throw new RuntimeException(e);
328 		}
329 	}
