13  package org.apache.juneau.server.samples;
14  
15  import static org.junit.Assert.*;
16  
17  import org.apache.juneau.client.*;
18  import org.apache.juneau.json.*;
19  import org.apache.juneau.samples.addressbook.*;
20  import org.apache.juneau.transforms.*;
21  import org.apache.juneau.urlencoding.*;
22  import org.apache.juneau.xml.*;
23  import org.junit.*;
24  
25  public class SampleRemoteableServicesResourceTest {
26  
27  	static RestClient[] clients;
28  
29  	@BeforeClass
30  	public static void beforeClass() throws Exception {
31  		clients = new RestClient[] {
32  			new SamplesRestClient(JsonSerializer.class, JsonParser.class),
33  			new SamplesRestClient(XmlSerializer.class, XmlParser.class),
34  //	TODO - broken?		new TestRestClient(HtmlSerializer.class, HtmlParser.class).setAccept("text/html+stripped"),
35  			new SamplesRestClient(UonSerializer.class, UonParser.class),
36  		};
37  		for (RestClient c : clients) {
38  			c.addPojoSwaps(CalendarSwap.Medium.class);
39  			c.setRemoteableServletUri("/remoteable");
40  			c.setProperty(XmlSerializerContext.XML_autoDetectNamespaces, true);
41  		}
42  	}
43  
44  	@AfterClass
45  	public static void afterClass() {
46  		for (RestClient c : clients) {
47  			c.closeQuietly();
48  		}
49  	}
50  
51  	//====================================================================================================
52  	// Get AddressBookResource as JSON
53  	//====================================================================================================
54  	@Test
55  	public void testBasic() throws Exception {
56  		for (RestClient client : clients) {
57  			IAddressBook ab = client.getRemoteableProxy(IAddressBook.class);
58  			Person p = ab.createPerson(
59  				new CreatePerson("Test Person",
60  					AddressBook.toCalendar("Aug 1, 1999"),
61  					new CreateAddress("Test street", "Test city", "Test state", 12345, true))
62  			);
63  			assertEquals(
64  				"{id:x,name:'Test Person',birthDate:'Aug 1, 1999',addresses:[{id:x,street:'Test street',city:'Test city',state:'Test state',zip:12345,isCurrent:true}],age:x}",
65  				JsonSerializer.DEFAULT_LAX.toString(p).replaceAll("id:\\d+", "id:x").replaceAll("age:\\d+", "age:x"));
66  		}
67  	}
68  
69  }
