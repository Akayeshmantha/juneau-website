13  package org.apache.juneau.server.test;
14  
15  import java.net.*;
16  
17  
18  public class Constants {
19  
20  	private static String juneauSampleUrl = System.getProperty("JUNO_SAMPLE_URL", "http://localhost:10000");
21  	private static URI juneauSampleUri = (juneauSampleUrl == null ? null : URI.create(juneauSampleUrl));
22  
23  	/**
24  	 * Returns the value of the "JUNO_SAMPLE_URL" system property, or throws a {@link RuntimeException}
25  	 * if it's not set.
26  	 */
27  	public static String getJuneauSamplesUrl() {
28  		if (juneauSampleUrl == null)
29  			throw new RuntimeException("'JUNO_SAMPLE_URL' system property not set to URL of juneau.sample.war location.");
30  		return juneauSampleUrl;
31  	}
32  
33  	public static URI getJuneauSamplesUri() {
34  		if (juneauSampleUri == null)
35  			throw new RuntimeException("'JUNO_SAMPLE_URL' system property not set to URL of juneau.sample.war location.");
36  		return juneauSampleUri;
37  	}
38  
39  	private static String juneauServerTestUrl = System.getProperty("JUNO_SERVER_TEST_URL", "http://localhost:10001");
40  	private static URI juneauServerTestUri = (juneauServerTestUrl == null ? null : URI.create(juneauServerTestUrl));
41  
42  	public static String getServerTestUrl() {
43  		if (juneauServerTestUrl == null)
44  			throw new RuntimeException("'JUNO_SERVER_TEST_URL' system property not set to URL of juneau.sample.war location.");
45  		return juneauServerTestUrl;
46  	}
47  
48  	public static URI getServerTestUri() {
49  		if (juneauServerTestUri == null)
50  			throw new RuntimeException("'JUNO_SERVER_TEST_URL' system property not set to URL of juneau.sample.war location.");
51  		return juneauServerTestUri;
52  	}
53  }
