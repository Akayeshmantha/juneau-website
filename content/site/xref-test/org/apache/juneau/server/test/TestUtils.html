13  package org.apache.juneau.server.test;
14  
15  import java.text.*;
16  
17  import org.apache.juneau.client.*;
18  import org.apache.juneau.json.*;
19  import org.apache.juneau.serializer.*;
20  import org.apache.juneau.transforms.*;
21  import org.junit.Assert;
22  
23  import junit.framework.*;
24  
25  public class TestUtils {
26  
27  	private static JsonSerializer js2 = new JsonSerializer.Simple()
28  		.addPojoSwaps(IteratorSwap.class, EnumerationSwap.class);
29  
30  	/**
31  	 * Assert that the object equals the specified string after running it through JsonSerializer.DEFAULT_LAX.toString().
32  	 */
33  	public static void assertObjectEquals(String s, Object o) {
34  		assertObjectEquals(s, o, js2);
35  	}
36  
37  	/**
38  	 * Assert that the object equals the specified string after running it through ws.toString().
39  	 */
40  	public static void assertObjectEquals(String s, Object o, WriterSerializer ws) {
41  		Assert.assertEquals(s, ws.toString(o));
42  	}
43  
44  	public static void checkErrorResponse(boolean debug, RestCallException e, int status, String...contains) throws AssertionFailedError {
45  		String r = e.getResponseMessage();
46  		if (debug) {
47  			System.err.println(r);
48  			e.printStackTrace();
49  		}
50  		if (status != e.getResponseCode())
51  			throw new AssertionFailedError(MessageFormat.format("Response status code was not correct.  Expected: ''{0}''.  Actual: ''{1}''", status, e.getResponseCode()));
52  		for (String s : contains) {
53  			if (r == null || ! r.contains(s)) {
54  				if (! debug)
55  					System.err.println(r);
56  				throw new AssertionFailedError(MessageFormat.format("Response did not have the following expected text: ''{0}''", s));
57  			}
58  		}
59  	}
60  }
