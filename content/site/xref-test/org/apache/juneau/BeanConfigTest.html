178 		protected String street;
179 		protected String city;
180 		protected String state;
181 		protected String zip;
182 
183 		public Address(String street, String city, String state, String zip) {
184 			this.street = street;
185 			this.city = city;
186 			this.state = state;
187 			this.zip = zip;
188 		}
189 
190 		public String getStreet() {
191 			return this.street;
192 		}
193 
194 		public String getCity() {
195 			return this.city;
196 		}
197 
198 		public String getState() {
199 			return this.state;
200 		}
201 
202 		public String getZip() {
203 			return this.zip;
204 		}
205 
206 		@Override /* Object */
207 		public boolean equals(Object o) {
208 			if (o == null)
209 				return false;
210 			if (this == o)
211 				return true;
212 			if (this.getClass() != o.getClass())
213 				return false;
214 			Address a = (Address) o;
215 
216 			String v1 = this.getStreet();
217 			String v2 = a.getStreet();
218 			if ((v1 == null) ? (v2 != null) : (!v1.equals(v2)))
219 				return false;
220 
221 			v1 = this.getCity();
222 			v2 = a.getCity();
223 			if ((v1 == null) ? (v2 != null) : (!v1.equals(v2)))
224 				return false;
225 
226 			v1 = this.getState();
227 			v2 = a.getState();
228 			if ((v1 == null) ? (v2 != null) : (!v1.equals(v2)))
229 				return false;
230 
231 			v1 = this.getZip();
232 			v2 = a.getZip();
233 			return ((v1 == null) ? (v2 == null) : (v1.equals(v2)));
234 		}
235 
236 		@Override /* Object */
237 		public int hashCode() {
238 			int code = 0;
239 			if (this.street != null)
240 				code ^= this.street.hashCode();
241 			if (this.city != null)
242 				code ^= this.city.hashCode();
243 			if (this.state != null)
244 				code ^= this.state.hashCode();
245 			if (this.zip != null)
246 				code ^= this.zip.hashCode();
247 			return code;
248 		}
249 
250 		@Override /* Object */
251 		public String toString() {
252 			return ("Address(street: " + this.getStreet() + ", city: "
253 					+ this.getCity() + ", state: " + this.getState()
254 					+ ", zip: " + this.getZip() + ")");
255 		}

316 		assertEquals(Boolean.FALSE, bc.convertToType(o, Boolean.TYPE));

320 		assertEquals(new Integer(1), bc.convertToType(o, Integer.TYPE));
321 		assertEquals(new Short((short) 1), bc.convertToType(o, Short.TYPE));

323 		assertEquals(new Float(1), bc.convertToType(o, Float.TYPE));
324 		assertEquals(new Double(1), bc.convertToType(o, Double.TYPE));

379 
380 		// String to generic Map
381 		assertEquals("foo", bc.convertToType(o, Map.class).values().iterator().next());
382 
383 		// Array to String
384 		o = new Object[] { "a", 1, false };
385 		assertEquals("['a',1,false]", bc.convertToType(o, String.class));
386 		o = new Object[][] { { "a", 1, false } };
387 		assertEquals("[['a',1,false]]", bc.convertToType(o, String.class));

461 		BeanContext bc = ContextFactory.create().setClassLoader(A.class.getClassLoader()).getBeanContext();

465 				new AHandler());
466 
467 		BeanMap bm1 = bc.forBean(f1);
468 		if (bm1 == null) {
469 			fail("Failed to obtain bean adapter for proxy: " + f1);
470 			return;
471 		}
472 
473 		BeanMap bm2 = bc.newBeanMap(A.class);

475 			fail("Failed to create dynamic proxy bean for interface: " + A.class.getName());

479 		bm2.put("b", new Integer(50));
480 		f1.setA("Hello");
481 		f1.setB(50);
482

484 			fail("Failed to set string property 'a' on dynamic proxy bean.  " + bm2);

486 		if (!bm2.get("b").equals(new Integer(50)))
487 			fail("Failed to set string property 'b' on dynamic proxy bean.  " + bm2);
488 
489 		if (!bm1.equals(bm2))
490 			fail("Failed equality test of dynamic proxies beans: " + bm1 + " / " + bm2);

506 	public static class AHandler implements InvocationHandler {
507 		private Map map;
508 
509 		public AHandler() {
510 			this.map = new HashMap();
511 			this.map.put("a", "");
512 			this.map.put("b", new Integer(0));
513 		}
514 
515 		@Override /* InvocationHandler */
516 		public Object invoke(Object proxy, Method method, Object[] args)
517 				throws Throwable {
518 			String methodName = method.getName();
519 			if (methodName.equals("getA")) {
520 				return this.map.get("a");
521 			}
522 			if (methodName.equals("setA")) {
523 				this.map.put("a", args[0]);
524 				return null;
525 			}
526 			if (methodName.equals("getB")) {
527 				return this.map.get("b");
528 			}
529 			if (methodName.equals("setB")) {
530 				this.map.put("b", args[0]);
531 				return null;
532 			}
533 			if (methodName.equals("toString")) {
534 				return this.map.toString();
535 			}
536 			return null;
537 		}
538 	}

851 
852 		// Blanks get interpreted as the default value for primitives and null for boxed objects.
853 		assertEquals(0, (int)bc.convertToType("", int.class));
854 		assertNull(bc.convertToType("", Integer.class));
