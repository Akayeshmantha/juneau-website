37  	@Test
38  	public void testContentFormat() throws Exception {
39  		A t = A.newInstance(), t2;
40  		XmlSerializer s1 = XmlSerializer.DEFAULT_SIMPLE_SQ,
41  			s2 = new XmlSerializer().setProperty(SERIALIZER_quoteChar, '\'').setProperty(SERIALIZER_useIndentation, true).setProperty(XML_enableNamespaces, false);
42  		XmlParser p = XmlParser.DEFAULT;
43  		XmlSerializerSession session;
44  		String r;
45  		StringWriter sw;
46  
47  		//----------------------------------------------------------------
48  		// Null
49  		//----------------------------------------------------------------
50  		t.f2 = null;
51  
52  		sw = new StringWriter();
53  		session = s1.createSession(sw, new ObjectMap("{"+SERIALIZER_trimNullProperties+":false}"), null);
54  		s1.serialize(session, t);
55  		r = sw.toString();
56  		assertEquals("<A f1='f1'>_x0000_</A>", r);
57  		t2 = p.parse(r, A.class);
58  		assertEqualObjects(t, t2);
59  
60  		sw = new StringWriter();
61  		session = s2.createSession(sw, new ObjectMap("{"+SERIALIZER_trimNullProperties+":false}"), null);
62  		s2.serialize(session, t);
63  		r = sw.toString();
64  		assertEquals("<A f1='f1'>\n\t_x0000_\n</A>\n", r);
65  		t2 = p.parse(r, A.class);
66  		assertEqualObjects(t, t2);
67  
68  		//----------------------------------------------------------------
69  		// Normal text
70  		//----------------------------------------------------------------
71  		t.f2 = "foobar";
72  
73  		r = s1.serialize(t);
74  		assertEquals("<A f1='f1'>foobar</A>", r);
75  		t2 = p.parse(r, A.class);
76  		assertEqualObjects(t, t2);
77  
78  		r = s2.serialize(t);
79  		assertEquals("<A f1='f1'>\n\tfoobar\n</A>\n", r);
80  		t2 = p.parse(r, A.class);
81  		assertEqualObjects(t, t2);
82  
83  		//----------------------------------------------------------------
84  		// Special characters
85  		//----------------------------------------------------------------
86  		t.f2 = "~!@#$%^&*()_+`-={}|[]\\:\";'<>?,.\n\r\t\b";
87  
88  		r = s1.serialize(t);
89  		assertEquals("<A f1='f1'>~!@#$%^&amp;*()_+`-={}|[]\\:\";'&lt;&gt;?,.&#x000a;&#x000d;&#x0009;_x0008_</A>", r);
90  		t2 = p.parse(r, A.class);
91  		assertEqualObjects(t, t2);
92  
93  		r = s2.serialize(t);
94  		assertEquals("<A f1='f1'>\n\t~!@#$%^&amp;*()_+`-={}|[]\\:\";'&lt;&gt;?,.&#x000a;&#x000d;&#x0009;_x0008_\n</A>\n", r);
95  		t2 = p.parse(r, A.class);
96  		assertEqualObjects(t, t2);
97  
98  		//----------------------------------------------------------------
99  		// Leading spaces
100 		//----------------------------------------------------------------
101 		t.f2 = "  foobar";
102 
103 		r = s1.serialize(t);
104 		assertEquals("<A f1='f1'>_x0020_ foobar</A>", r);
105 		t2 = p.parse(r, A.class);
106 		assertEqualObjects(t, t2);
107 
108 		r = s2.serialize(t);
109 		assertEquals("<A f1='f1'>\n\t_x0020_ foobar\n</A>\n", r);
110 		t2 = p.parse(r, A.class);
111 		assertEqualObjects(t, t2);
112 
113 		//----------------------------------------------------------------
114 		// Trailing spaces
115 		//----------------------------------------------------------------
116 		t.f2 = "foobar  ";
117 
118 		r = s1.serialize(t);
119 		assertEquals("<A f1='f1'>foobar _x0020_</A>", r);
120 		t2 = p.parse(r, A.class);
121 		assertEqualObjects(t, t2);
122 
123 		r = s2.serialize(t);
124 		assertEquals("<A f1='f1'>\n\tfoobar _x0020_\n</A>\n", r);
125 		t2 = p.parse(r, A.class);
126 		assertEqualObjects(t, t2);
127 	}

146 	public void testXmlContentFormat() throws Exception {
147 		B t = B.newInstance(), t2;
148 		XmlSerializer s1 = XmlSerializer.DEFAULT_SIMPLE_SQ,
149 			s2 = new XmlSerializer().setProperty(SERIALIZER_quoteChar, '\'').setProperty(SERIALIZER_useIndentation, true).setProperty(XML_enableNamespaces, false);

151 		XmlSerializerSession session;
152 		String r;
153 		StringWriter sw;
154 
155 		//----------------------------------------------------------------
156 		// Null
157 		//----------------------------------------------------------------
158 		t.f2 = null;
159 
160 		sw = new StringWriter();
161 		session = s1.createSession(sw, new ObjectMap("{"+SERIALIZER_trimNullProperties+":false}"), null);
162 		s1.serialize(session, t);
163 		r = sw.toString();
164 		assertEquals("<A f1='f1'>_x0000_</A>", r);
165 		t2 = p.parse(r, B.class);
166 		assertEqualObjects(t, t2);
167 
168 		sw = new StringWriter();
169 		session = s2.createSession(sw, new ObjectMap("{"+SERIALIZER_trimNullProperties+":false}"), null);
170 		s2.serialize(session, t);
171 		r = sw.toString();
172 		assertEquals("<A f1='f1'>\n\t_x0000_\n</A>\n", r);
173 		t2 = p.parse(r, B.class);
174 		assertEqualObjects(t, t2);
175 
176 		//----------------------------------------------------------------
177 		// Normal text
178 		//----------------------------------------------------------------
179 		t.f2 = "foobar";
180 
181 		r = s1.serialize(t);
182 		assertEquals("<A f1='f1'>foobar</A>", r);
183 		t2 = p.parse(r, B.class);
184 		assertEqualObjects(t, t2);
185 
186 		r = s2.serialize(t);
187 		assertEquals("<A f1='f1'>\n\tfoobar\n</A>\n", r);
188 		t2 = p.parse(r, B.class);
189 		assertEqualObjects(t, t2);
190 
191 		//----------------------------------------------------------------
192 		// Normal XML
193 		//----------------------------------------------------------------
194 		t.f2 = "<xxx>foobar<yyy>baz</yyy>foobar</xxx>";
195 
196 		r = s1.serialize(t);
197 		assertEquals("<A f1='f1'><xxx>foobar<yyy>baz</yyy>foobar</xxx></A>", r);
198 		t2 = p.parse(r, B.class);
199 		assertEqualObjects(t, t2);
200 
201 		r = s2.serialize(t);
202 		assertEquals("<A f1='f1'>\n\t<xxx>foobar<yyy>baz</yyy>foobar</xxx>\n</A>\n", r);
203 		t2 = p.parse(r, B.class);
204 		assertEqualObjects(t, t2);
205 
206 		//----------------------------------------------------------------
207 		// Normal XML with leading and trailing space
208 		//----------------------------------------------------------------
209 		t.f2 = "  <xxx>foobar<yyy>baz</yyy>foobar</xxx>  ";
210 
211 		r = s1.serialize(t);
212 		assertEquals("<A f1='f1'>_x0020_ <xxx>foobar<yyy>baz</yyy>foobar</xxx> _x0020_</A>", r);
213 		t2 = p.parse(r, B.class);
214 		assertEqualObjects(t, t2);
215 
216 		r = s2.serialize(t);
217 		assertEquals("<A f1='f1'>\n\t_x0020_ <xxx>foobar<yyy>baz</yyy>foobar</xxx> _x0020_\n</A>\n", r);
218 		t2 = p.parse(r, B.class);
219 		assertEqualObjects(t, t2);
220 
221 		//----------------------------------------------------------------
222 		// XML with attributes
223 		//----------------------------------------------------------------
224 		t.f2 = "<xxx x=\"x\">foobar</xxx>";
225 
226 		r = s1.serialize(t);
227 		assertEquals("<A f1='f1'><xxx x=\"x\">foobar</xxx></A>", r);
228 		t2 = p.parse(r, B.class);
229 		assertEqualObjects(t, t2);
230 
231 		r = s2.serialize(t);
232 		assertEquals("<A f1='f1'>\n\t<xxx x=\"x\">foobar</xxx>\n</A>\n", r);
233 		t2 = p.parse(r, B.class);
234 		assertEqualObjects(t, t2);
235 
236 		//----------------------------------------------------------------
237 		// XML with embedded entities
238 		//----------------------------------------------------------------
239 		t.f2 = "<xxx x=\"x\">foo&lt;&gt;bar</xxx>";
240 
241 		r = s1.serialize(t);
242 		assertEquals("<A f1='f1'><xxx x=\"x\">foo&lt;&gt;bar</xxx></A>", r);

252 	@Bean(typeName="A")
253 	public static class B {
254 		@Xml(format=ATTR) public String f1;
255 		@Xml(format=CONTENT, contentHandler=BContentHandler.class) public String f2;
256 
257 		public static B newInstance() {
258 			B t = new B();
259 			t.f1 = "f1";
260 			t.f2 = null;
261 			return t;
262 		}
263 	}
264 
265 	public static class BContentHandler implements XmlContentHandler<B> {
266 
267 		@Override /* XmlContentHandler */
268 		public void parse(XMLStreamReader r, B b) throws Exception {
269 			b.f2 = decode(readXmlContents(r).trim());
270 		}
271 
272 		@Override /* XmlContentHandler */
273 		public void serialize(XmlWriter w, B b) throws Exception {
274 			w.encodeTextInvalidChars(b.f2);
275 		}

292 		try {
293 			s.serialize(new C1());
294 			fail("Exception expected");
295 		} catch (Exception e) {
296 			assertTrue(e.getMessage().contains("Multiple instances of CONTENT properties defined on class"));
297 		}
