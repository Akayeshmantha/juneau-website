30  	@Test
31  	public void testFromSerializer() throws Exception {
32  		ReaderParser p = XmlParser.DEFAULT;
33  
34  		Map m = null;
35  		m = (Map)p.parse("<object><a _type='number'>1</a></object>", Object.class);
36  		assertEquals(1, m.get("a"));
37  		m = (Map)p.parse("<object><a _type='number'>1</a><b _type='string'>foo bar</b></object>", Object.class);
38  		assertEquals(1, m.get("a"));
39  		assertEquals("foo bar", m.get("b"));
40  		m = (Map)p.parse("<object><a _type='number'>1</a><b _type='string'>foo bar</b><c _type='boolean'>false</c></object>", Object.class);
41  		assertEquals(1, m.get("a"));
42  		assertEquals(false, m.get("c"));
43  		m = (Map)p.parse("   <object>	<a _type='number'>	1	</a>	<b _type='string'>	foo	</b>	<c _type='boolean'>	false 	</c>	</object>	", Object.class);
44  		assertEquals(1, m.get("a"));
45  		assertEquals("foo", m.get("b"));
46  		assertEquals(false, m.get("c"));
47  
48  		m = (Map)p.parse("<object><x _type='string'>org.apache.juneau.test.Person</x><addresses _type='array'><object><x _type='string'>org.apache.juneau.test.Address</x><city _type='string'>city A</city><state _type='string'>state A</state><street _type='string'>street A</street><zip _type='number'>12345</zip></object></addresses></object>", Object.class);
49  		assertEquals("org.apache.juneau.test.Person", m.get("x"));
50  		List l = (List)m.get("addresses");
51  		assertNotNull(l);
52  		m = (Map)l.get(0);
53  		assertNotNull(m);
54  		assertEquals("org.apache.juneau.test.Address", m.get("x"));
55  		assertEquals("city A", m.get("city"));
56  		assertEquals("state A", m.get("state"));
57  		assertEquals("street A", m.get("street"));
58  		assertEquals(12345, m.get("zip"));
59  
60  		ObjectList jl = (ObjectList)p.parse("<array><object><attribute _type='string'>value</attribute></object><object><attribute _type='string'>value</attribute></object></array>", Object.class);
61  		assertEquals("value", jl.getObjectMap(0).getString("attribute"));
62  		assertEquals("value", jl.getObjectMap(1).getString("attribute"));
63  
64  		try {
65  			jl = (ObjectList)p.parse("<array><object><attribute _type='string'>value</attribute></object><object><attribute _type='string'>value</attribute></object></array>", Object.class);
66  			assertEquals("value", jl.getObjectMap(0).getString("attribute"));
67  			assertEquals("value", jl.getObjectMap(1).getString("attribute"));
68  		} catch (Exception e) {
69  			fail(e.getLocalizedMessage());
70  		}
71  
72  		A1 t1 = new A1();
73  		A2 t2 = new A2();
74  		t2.add(new A3("name0","value0"));
75  		t2.add(new A3("name1","value1"));
76  		t1.list = t2;
77  		String r = XmlSerializer.DEFAULT.serialize(t1);
78  		t1 = p.parse(r, A1.class);
79  		assertEquals("value1", t1.list.get(1).value);
80  
81  		r = XmlSerializer.DEFAULT.serialize(t1);
82  		t1 = p.parse(r, A1.class);
83  		assertEquals("value1", t1.list.get(1).value);
84  	}

105 	@Test
106 	public void testCorrectHandlingOfUnknownProperties() throws Exception {
107 		ReaderParser p = new XmlParser().setProperty(BEAN_ignoreUnknownBeanProperties, true);
108 		B t;
109 
110 		String in =  "<object><a>1</a><unknown>foo</unknown><b>2</b></object>";
111 		t = p.parse(in, B.class);
112 		assertEquals(t.a, 1);
113 		assertEquals(t.b, 2);
114 
115 		in =  "<object><a>1</a><unknown><object><a _type='string'>foo</a></object></unknown><b>2</b></object>";
116 		t = p.parse(in, B.class);
117 		assertEquals(t.a, 1);
118 		assertEquals(t.b, 2);
119 
120 
121 		try {
122 			p = new XmlParser();
123 			p.parse(in, B.class);
124 			fail("Exception expected");
125 		} catch (ParseException e) {}
126 	}

134 	//====================================================================================================
135 	@Test
136 	public void testCollectionPropertiesWithNoSetters() throws Exception {
137 
138 		ReaderParser p = XmlParser.DEFAULT;
139 
140 		String in = "<object><ints _type='array'><number>1</number><number>2</number><number>3</number></ints><beans _type='array'><object><a _type='number'>1</a><b _type='number'>2</b></object></beans></object>";
141 		C t = p.parse(in, C.class);
142 		assertEquals(t.getInts().size(), 3);
143 		assertEquals(t.getBeans().get(0).b, 2);
144 	}
145

147 		private Collection<Integer> ints = new LinkedList<Integer>();
148 		private List<B> beans = new LinkedList<B>();
149 		public Collection<Integer> getInts() {
150 			return ints;
151 		}
152 		public List<B> getBeans() {
153 			return beans;
154 		}
155 	}

157 	//====================================================================================================
158 	// Parser listeners.
159 	//====================================================================================================
160 	@Test
161 	public void testParserListeners() throws Exception {
162 		final List<String> events = new LinkedList<String>();
163 		XmlParser p = new XmlParser().setProperty(BEAN_ignoreUnknownBeanProperties, true);
164 		p.addListener(
165 			new ParserListener() {
166 				@Override /* ParserListener */
167 				public <T> void onUnknownProperty(String propertyName, Class<T> beanClass, T bean, int line, int col) {
168 					events.add(propertyName + "," + line + "," + col);
169 				}
170 			}
171 		);
172 
173 		String in = "<object><a _type='number'>1</a><unknownProperty _type='string'>foo</unknownProperty><b _type='number'>2</b></object>";
174 		p.parse(in, B.class);
175 		assertEquals(1, events.size());
176 		// XML parser may or may not support line numbers.
177 		assertTrue(events.get(0).startsWith("unknownProperty,"));
178 	}
