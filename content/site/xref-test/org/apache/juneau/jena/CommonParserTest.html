30  	private String wrap(String in) {
31  		return ""
32  			+ "<rdf:RDF"
33  		    + " xmlns:rdf='http://www.w3.org/1999/02/22-rdf-syntax-ns#'"
34  		    + " xmlns:j='http://www.ibm.com/juneau/'"
35  		    + " xmlns:jp='http://www.ibm.com/juneaubp/'>"
36  		    + in
37  		    + "</rdf:RDF>";
38  	}
39  
40  	private String strip(String s) {
41  		return s.replaceFirst("<rdf:RDF[^>]+>\\s*", "").replaceAll("</rdf:RDF>$", "").trim().replaceAll("[\\r\\n]", "");
42  	}

44  	private RdfSerializer getBasicSerializer() {
45  		return new RdfSerializer()
46  			.setProperty(SERIALIZER_quoteChar, '\'')
47  			.setProperty(RDF_addLiteralTypes, true)
48  			.setProperty(SERIALIZER_useIndentation, false)
49  			.setProperty(RDF_rdfxml_allowBadUris, true)
50  			.setProperty(RDF_rdfxml_showDoctypeDeclaration, false)
51  			.setProperty(RDF_rdfxml_showXmlDeclaration, false);
52  	}

57  	@Test
58  	public void testFromSerializer() throws Exception {
59  		WriterSerializer s = getBasicSerializer();
60  		ReaderParser p = new RdfParser.Xml().setProperty(RdfParserContext.RDF_trimWhitespace, true).setClassLoader(getClass().getClassLoader());
61  		Map m = null;
62  		String in;
63  		Integer one = Integer.valueOf(1);
64  
65  		in = wrap("<rdf:Description><jp:a rdf:datatype='http://www.w3.org/2001/XMLSchema#int'>1</jp:a></rdf:Description>");
66  		m = (Map)p.parse(in, Object.class);
67  		assertEquals(one, m.get("a"));
68  
69  		in = wrap("<rdf:Description><jp:a rdf:datatype='http://www.w3.org/2001/XMLSchema#int'>1</jp:a><jp:b>foo bar</jp:b><jp:c rdf:datatype='http://www.w3.org/2001/XMLSchema#boolean'>false</jp:c></rdf:Description>");
70  		m = (Map)p.parse(in, Object.class);
71  		assertEquals(one, m.get("a"));
72  		assertEquals("foo bar", m.get("b"));
73  		in = wrap("<rdf:Description><jp:a rdf:datatype='http://www.w3.org/2001/XMLSchema#int'> 1 </jp:a><jp:b> foo bar </jp:b><jp:c rdf:datatype='http://www.w3.org/2001/XMLSchema#boolean'> false </jp:c></rdf:Description>");
74  		m = (Map)p.parse(in, Object.class);
75  		assertEquals(one, m.get("a"));
76  		assertEquals("foo bar", m.get("b"));
77  		assertEquals(false, m.get("c"));
78  
79  		in = wrap("<rdf:Description><jp:x>org.apache.juneau.test.Person</jp:x><jp:addresses><rdf:Seq><rdf:li rdf:parseType='Resource'><jp:x>org.apache.juneau.test.Address</jp:x><jp:city>city A</jp:city><jp:state>state A</jp:state><jp:street>street A</jp:street><jp:zip rdf:datatype='http://www.w3.org/2001/XMLSchema#int'>12345</jp:zip></rdf:li></rdf:Seq></jp:addresses></rdf:Description>");
80  		m = (Map)p.parse(in, Object.class);
81  		assertEquals("org.apache.juneau.test.Person", m.get("x"));
82  		List l = (List)m.get("addresses");
83  		assertNotNull(l);
84  		m = (Map)l.get(0);
85  		assertNotNull(m);
86  		assertEquals("org.apache.juneau.test.Address", m.get("x"));
87  		assertEquals("city A", m.get("city"));
88  		assertEquals("state A", m.get("state"));
89  		assertEquals("street A", m.get("street"));
90  		assertEquals(12345, m.get("zip"));
91  
92  		in = wrap("<rdf:Seq><rdf:li rdf:parseType='Resource'><jp:attribute>value</jp:attribute></rdf:li><rdf:li rdf:parseType='Resource'><jp:attribute>value</jp:attribute></rdf:li></rdf:Seq>");
93  		ObjectList jl = (ObjectList)p.parse(in, Object.class);
94  		assertEquals("value", jl.getObjectMap(0).getString("attribute"));
95  		assertEquals("value", jl.getObjectMap(1).getString("attribute"));
96  
97  		// Verify that all the following return null.
98  		assertNull(p.parse((CharSequence)null, Object.class));
99  		assertNull(p.parse(wrap(""), Object.class));
100 		assertNull(p.parse(wrap("   "), Object.class));
101 		assertNull(p.parse(wrap("   \t"), Object.class));
102 		assertNull(p.parse(wrap("   <!--foo-->"), Object.class));
103 		assertNull(p.parse(wrap("   <!--foo-->   "), Object.class));
104 		assertNull(p.parse(wrap("   //foo   "), Object.class));
105 
106 
107 		A1 t1 = new A1();
108 		A2 t2 = new A2();
109 		t2.add(new A3("name0","value0"));
110 		t2.add(new A3("name1","value1"));
111 		t1.list = t2;
112 
113 		s.setProperty(SERIALIZER_addBeanTypeProperties, true);
114 		in = strip(s.serialize(t1));
115 		assertEquals("<rdf:Description><j:class>"+A1.class.getName()+"</j:class><jp:list><rdf:Seq><rdf:li rdf:parseType='Resource'><jp:name>name0</jp:name><jp:value>value0</jp:value></rdf:li><rdf:li rdf:parseType='Resource'><jp:name>name1</jp:name><jp:value>value1</jp:value></rdf:li></rdf:Seq></jp:list></rdf:Description>", in);
116 		in = wrap(in);
117 		t1 = (A1)p.parse(in, Object.class);
118 		assertEquals("value1", t1.list.get(1).value);
119 		t1 = p.parse(in, A1.class);
120 		assertEquals("value1", t1.list.get(1).value);
121 	}

166 	@Test
167 	public void testCollectionPropertiesWithNoSetters() throws Exception {
168 		RdfParser p = new RdfParser.Xml();
169 		String in = wrap("<rdf:Description><jp:ints><rdf:Seq><rdf:li>1</rdf:li><rdf:li>2</rdf:li></rdf:Seq></jp:ints><jp:beans><rdf:Seq><rdf:li rdf:parseType='Resource'><jp:a>1</jp:a><jp:b>2</jp:b></rdf:li></rdf:Seq></jp:beans></rdf:Description>");
170 		C t = p.parse(in, C.class);
171 		assertEquals(t.getInts().size(), 2);
172 		assertEquals(t.getBeans().get(0).b, 2);
173 	}

186 	//====================================================================================================
187 	// Parser listeners.
188 	//====================================================================================================
189 	@Test
190 	public void testParserListeners() throws Exception {
191 		final List<String> events = new LinkedList<String>();
192 		RdfParser p = new RdfParser.Xml().setProperty(BEAN_ignoreUnknownBeanProperties, true);
193 		p.addListener(
194 			new ParserListener() {
195 				@Override /* ParserListener */
196 				public <T> void onUnknownProperty(String propertyName, Class<T> beanClass, T bean, int line, int col) {
197 					events.add(propertyName + "," + line + "," + col);
198 				}
199 			}
200 		);
201 
202 		String in = wrap("<rdf:Description><jp:a rdf:datatype='http://www.w3.org/2001/XMLSchema#int'>1</jp:a><jp:unknownProperty>foo</jp:unknownProperty><jp:b rdf:datatype='http://www.w3.org/2001/XMLSchema#int'>2</jp:b></rdf:Description>");
203 		p.parse(in, B.class);
204 		assertEquals(1, events.size());
205 		assertEquals("unknownProperty,-1,-1", events.get(0));
206 	}
