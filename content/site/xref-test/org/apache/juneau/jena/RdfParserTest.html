32  public class RdfParserTest {
33  
34  	@Test
35  	public void testParseIntoGenericPojos() throws Exception {
36  		A a = new A().init();
37  
38  		// Create a new serializer with readable output.
39  		RdfSerializer s = new RdfSerializer.XmlAbbrev()
40  		   .setProperty(RDF_rdfxml_tab, 3)
41  		   .setProperty(SERIALIZER_quoteChar, '\'')
42  		   .setProperty(RDF_addRootProperty, true);
43  
44  		String expected =
45  			 "<rdf:RDF a='http://ns/' a1='http://ns2/' j='http://www.ibm.com/juneau/' jp='http://www.ibm.com/juneaubp/' rdf='http://www.w3.org/1999/02/22-rdf-syntax-ns#'>"
46  			 + "\n   <rdf:Description about='http://test/a'>"
47  			 + "\n      <a:f1>1</a:f1>"
48  			 + "\n      <a:f2>f2</a:f2>"
49  			 + "\n      <a:f4a rdf:resource='http://test/a'/>"
50  			 + "\n      <a:f4b rdf:resource='http://test/external'/>"
51  			 + "\n      <a:f5>1999-01-01T00:00:00Z</a:f5>"
52  			 + "\n      <a:f6>"
53  			 + "\n         <rdf:Seq>"
54  			 + "\n            <rdf:li>"
55  			 + "\n               <rdf:Description about='http://test/a/a1'>"
56  			 + "\n                  <a1:f1>1</a1:f1>"
57  			 + "\n                  <a1:f2>f2</a1:f2>"
58  			 + "\n                  <a1:f4a rdf:resource='http://test/a'/>"
59  			 + "\n                  <a1:f4b rdf:resource='http://test/external'/>"
60  			 + "\n                  <a1:f5>1999-01-01T00:00:00Z</a1:f5>"
61  			 + "\n               </rdf:Description>"
62  			 + "\n            </rdf:li>"
63  			 + "\n         </rdf:Seq>"
64  			 + "\n      </a:f6>"
65  			 + "\n      <j:root>true</j:root>"
66  			 + "\n   </rdf:Description>"
67  			 + "\n</rdf:RDF>";
68  
69  		String rdfXml = s.serialize(a);
70  		assertXmlEquals(expected, rdfXml);
71  
72  		A a2 = RdfParser.DEFAULT_XML.parse(rdfXml, A.class);
73  
74  		assertEqualObjects(a, a2);
75  
76  		ObjectMap m = RdfParser.DEFAULT_XML.parse(rdfXml, ObjectMap.class);
77  		String json = JsonSerializer.DEFAULT_LAX_READABLE.serialize(m);
78  
79  		String e = ""
80  			+ "{\n"
81  			+ "	uri: 'http://test/a', \n"
82  			+ "	f6: [\n"
83  			+ "		{\n"
84  			+ "			uri: 'http://test/a/a1', \n"
85  			+ "			f5: '1999-01-01T00:00:00Z', \n"
86  			+ "			f4b: 'http://test/external', \n"
87  			+ "			f4a: 'http://test/a', \n"
88  			+ "			f2: 'f2', \n"
89  			+ "			f1: '1'\n"
90  			+ "		}\n"
91  			+ "	], \n"
92  			+ "	f5: '1999-01-01T00:00:00Z', \n"
93  			+ "	f4b: 'http://test/external', \n"
94  			+ "	f4a: 'http://test/a', \n"
95  			+ "	f2: 'f2', \n"
96  			+ "	f1: '1', \n"
97  			+ "	root: 'true'\n"
98  			+ "}";
99  		assertEquals(e, json.replace("\r", ""));
100 
101 	}

103 	@Rdf(prefix="a", namespace="http://ns/")
104 	public static class A {
105 		public int f1;
106 		public String f2;
107 		@Rdf(beanUri=true) public URI f3;
108 		public URI f4a, f4b;
109 		@BeanProperty(swap=CalendarSwap.ISO8601DTZ.class) public Calendar f5;
110 		public LinkedList<A1> f6 = new LinkedList<A1>();
111 
112 		public A init() throws Exception {
113 			f1 = 1;
114 			f2 = "f2";
115 			f3 = new URI("http://test/a"); // Bean URI.
116 			f4a = new URI("http://test/a"); // Points to itself.
117 			f4b = new URI("http://test/external");
118 			f5 = new GregorianCalendar();
119 			DateFormat df = DateFormat.getDateInstance(DateFormat.MEDIUM);
120 			df.setTimeZone(TimeZone.getTimeZone("GMT"));
121 			f5.setTime(df.parse("Jan 1, 1999"));
122 			f6 = new LinkedList<A1>();
123 			f6.add(new A1().init());
124 			return this;
125 		}
126 	}
127 
128 	@Rdf(prefix="a1", namespace="http://ns2/")
129 	public static class A1 {
130 		public int f1;
131 		public String f2;
132 		@Rdf(beanUri=true) public URI f3;
133 		public URI f4a, f4b;
134 		@BeanProperty(swap=CalendarSwap.ISO8601DTZ.class) public Calendar f5;
135 
136 		public A1 init() throws Exception {
137 			f1 = 1;
138 			f2 = "f2";
139 			f3 = new URI("http://test/a/a1");
140 			f4a = new URI("http://test/a");
141 			f4b = new URI("http://test/external");
142 			f5 = new GregorianCalendar();
143 			DateFormat df = DateFormat.getDateInstance(DateFormat.MEDIUM);
144 			df.setTimeZone(TimeZone.getTimeZone("GMT"));
145 			f5.setTime(df.parse("Jan 1, 1999"));
146 			return this;
147 		}
148 	}
