45  	//====================================================================================================
46  	@Test
47  	public void testGet() throws Exception {
48  		File f;
49  		ConfigMgr cm = new ConfigMgr(false, new XorEncoder(), JsonSerializer.DEFAULT, JsonParser.DEFAULT, Charset.defaultCharset(), new String[]{tempDir.getAbsolutePath()});
50  
51  		ConfigFile cf = cm.get("TestGet.cfg", true);
52  		cf.put("Test/A", "a");
53  
54  		f = new File(tempDir, "TestGet.cfg");
55  		assertTrue(f.exists());
56  
57  		cf.save();
58  		assertTextEquals("[Test]|A = a|", IOUtils.read(f));
59  
60  		cf = cm.get("TestGet.cfg");
61  		assertObjectEquals("{'default':{},Test:{A:'a'}}", cf);
62  
63  		ConfigFile cf2 = cm.get(tempDir.getAbsolutePath() + "/TestGet.cfg");
64  		assertObjectEquals("{'default':{},Test:{A:'a'}}", cf2);
65  		assertTrue(cf == cf2);  // Relative and absolute paths must resolve to same config file.
66  
67  		try { cm.get("TestGet2.cfg"); fail(); } catch (FileNotFoundException e) {}
68  		try { cm.get(tempDir.getAbsolutePath() + "TestGet2.cfg"); fail(); } catch (FileNotFoundException e) {}
69  
70  		cm.get(tempDir.getAbsolutePath() + "TestGet2.cfg", true);
71  
72  		ConfigMgr cm2 = new ConfigMgr(false, new XorEncoder(), JsonSerializer.DEFAULT, JsonParser.DEFAULT, Charset.defaultCharset(), new String[]{tempDir.getAbsolutePath()});
73  		cf = cm2.get("TestGet.cfg");
74  		assertObjectEquals("{'default':{},Test:{A:'a'}}", cf);
75  
76  		cm2 = new ConfigMgr(false, new XorEncoder(), JsonSerializer.DEFAULT, JsonParser.DEFAULT, Charset.defaultCharset(), null);
77  		try { cf = cm2.get("TestGet.cfg"); fail(); } catch (FileNotFoundException e) {}
78  
79  		String NL = System.getProperty("line.separator");
80  		cf = cm2.create(new StringReader(("[Test]"+NL+"A = a"+NL)));
81  		assertObjectEquals("{'default':{},Test:{A:'a'}}", cf);
82  
83  		ConfigMgr cm3 = new ConfigMgr(false, new XorEncoder(), JsonSerializer.DEFAULT, JsonParser.DEFAULT, IOUtils.UTF8, new String[]{tempDir.getAbsolutePath()});
84  		cf = cm3.get("TestGet.cfg");
85  		assertObjectEquals("{'default':{},Test:{A:'a'}}", cf);
86  
87  		cm.deleteAll();
88  		cm2.deleteAll();
89  		cm3.deleteAll();
90  	}

95  	@Test
96  	public void testLoadIfModified() throws Exception {
97  		ConfigMgr cm = new ConfigMgr(false, new XorEncoder(), JsonSerializer.DEFAULT, JsonParser.DEFAULT, Charset.defaultCharset(), new String[]{tempDir.getAbsolutePath()});
98  		File f;
99  		ConfigFile cf = cm.get("TestGet.cfg", true);
100 		cf.put("Test/A", "a");
101 
102 		f = new File(tempDir, "TestGet.cfg");
103 		String NL = System.getProperty("line.separator");
104 		IOUtils.write(f, new StringReader("[Test]"+NL+"A = b"+NL));
105 		FileUtils.modifyTimestamp(f);
106 
107 		cm.loadIfModified();
108 		assertEquals("b", cf.getString("Test/A"));
109 		cm.loadIfModified();
110 		assertEquals("b", cf.getString("Test/A"));
111 
112 		// Config file with no backing file.
113 		cf = cm.create();
114 		cf.put("Test/B", "b");
115 		cm.loadIfModified();
116 		cf.loadIfModified();
117 		assertEquals("b", cf.getString("Test/B"));
118 
119 		cm.deleteAll();
120 	}

125 	@Test
126 	public void testReadOnly() throws Exception {
127 		ConfigMgr cm = new ConfigMgr(true, new XorEncoder(), JsonSerializer.DEFAULT, JsonParser.DEFAULT, Charset.defaultCharset(), new String[]{tempDir.getAbsolutePath()});
128 		ConfigFile cf = cm.get("TestGet.cfg", true);
129 
130 		// All these should fail.
131 		try { cf.loadIfModified(); fail(); } catch (UnsupportedOperationException e) {}
132 		try { cf.load(); fail(); } catch (UnsupportedOperationException e) {}
133 		try { cf.load(new StringReader("")); fail(); } catch (UnsupportedOperationException e) {}
134 		try { cf.put("A","b"); fail(); } catch (UnsupportedOperationException e) {}
135 		try { cf.put("A","b",true); fail(); } catch (UnsupportedOperationException e) {}
136 		try { cf.put("A","b"); fail(); } catch (UnsupportedOperationException e) {}
137 		try { cf.put("A","b",true); fail(); } catch (UnsupportedOperationException e) {}
138 		try { cf.removeString("A"); fail(); } catch (UnsupportedOperationException e) {}
139 		try { cf.addLines("A","b=c"); fail(); } catch (UnsupportedOperationException e) {}
140 		try { cf.addHeaderComments("A", "b=c"); fail(); } catch (UnsupportedOperationException e) {}
141 		try { cf.clearHeaderComments("A"); fail(); } catch (UnsupportedOperationException e) {}
142 		try { cf.addSection("A"); fail(); } catch (UnsupportedOperationException e) {}
143 		try { cf.setSection("A",null); fail(); } catch (UnsupportedOperationException e) {}
144 		try { cf.removeSection("A"); fail(); } catch (UnsupportedOperationException e) {}
145 		try { cf.save(); fail(); } catch (UnsupportedOperationException e) {}
146 		try { cf.merge(cf); fail(); } catch (UnsupportedOperationException e) {}
147 		try { cf.addListener(new ConfigFileListener(){}); fail(); } catch (UnsupportedOperationException e) {}
148 
149 		// All these should succeed.
150 		cf.getObject(String.class, "A");
151 		cf.getObject(String.class, "A", "a");
152 		cf.getString("A");
153 		cf.getString("A","a");
154 		cf.getObject(String.class, "A");
155 		cf.getObject(String.class, "A", "a");
156 		cf.getObject(String[].class, "A");
157 		cf.getStringArray("A");
158 		cf.getStringArray("A", null);
159 		cf.getInt("A");
160 		cf.getInt("A", 0);
161 		cf.getBoolean("A");
162 		cf.getBoolean("A", true);
163 		cf.containsNonEmptyValue("A");
164 		cf.getSectionMap("A");
165 		cf.serializeTo(new StringWriter());
166 		cf.serializeTo(new StringWriter(), ConfigFileFormat.INI);
167 		cf.getResolving(VarResolver.DEFAULT);
168 		cf.toWritable();
169 	}

174 	@Test
175 	public void testMain() throws Exception {
176 		System.setProperty("exit.2", "0");
177 		ConfigMgr cm = new ConfigMgr(false, new XorEncoder(), JsonSerializer.DEFAULT, JsonParser.DEFAULT, Charset.defaultCharset(), new String[]{tempDir.getAbsolutePath()});
178 
179 		ConfigFile cf = cm.get("Test.cfg", true)
180 			.addLines(null, "# c1", "\t# c2", " c3 ", "  ", "x1=1", "x2=true", "x3=null")
181 			.addLines("s1", "#c4", "k1=1", "#c5 foo=bar", "k2 = true", "k3  = \tnull");
182 		cf.save();
183 
184 		File configFile = new File(tempDir, "Test.cfg");
185 		File envFile = new File(tempDir, "Test.bat");
186 
187 		ConfigMgr.main(new String[]{"createBatchEnvFile", "-configFile", configFile.getAbsolutePath(), "-envFile", envFile.getAbsolutePath()});
188 		String expected = "rem c1|rem c2|rem c3||set x1 = 1|set x2 = true|set x3 = null|rem c4|set s1_k1 = 1|rem c5 foo=bar|set s1_k2 = true|set s1_k3 = null|";
189 		String actual = IOUtils.read(envFile);
190 		assertTextEquals(expected, actual);
191 
192 		ConfigMgr.main(new String[]{"createShellEnvFile", "-configFile", configFile.getAbsolutePath(), "-envFile", envFile.getAbsolutePath()});
193 		expected = "# c1|# c2|# c3||export x1=\"1\"|export x2=\"true\"|export x3=\"null\"|# c4|export s1_k1=\"1\"|# c5 foo=bar|export s1_k2=\"true\"|export s1_k3=\"null\"|";
194 		actual = IOUtils.read(envFile);
195 		assertTextEquals(expected, actual);
196 
197 		ConfigMgr.main(new String[]{"setVals", "-configFile", configFile.getAbsolutePath(), "-vals", "x1=2", "s1/k1=2", "s2/k1=3"});
198 		FileUtils.modifyTimestamp(configFile);
199 		cf.loadIfModified();
200 		assertObjectEquals("{'default':{x1:'2',x2:'true',x3:'null'},s1:{k1:'2',k2:'true',k3:'null'},s2:{k1:'3'}}", cf);
201 
202 		ConfigMgr.main(new String[]{});
203 	}
