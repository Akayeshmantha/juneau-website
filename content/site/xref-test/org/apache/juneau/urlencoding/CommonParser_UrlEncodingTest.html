36  	@Test
37  	public void testFromSerializer() throws Exception {
38  		Map m = null;
39  		String in;
40  
41  		in = "a=$n(1)";
42  		m = (Map)p.parse(in, Object.class);
43  		assertEquals(1, m.get("a"));
44  
45  		in = "a=$n(1)&b=foo+bar";
46  		m = (Map)p.parse(in, Object.class);
47  		assertEquals(1, m.get("a"));
48  		assertEquals("foo bar", m.get("b"));
49  
50  		in = "a=$n(1)&b=foo+bar&c=$b(false)";
51  		m = (Map)p.parse(in, Object.class);
52  		assertEquals(1, m.get("a"));
53  		assertEquals("foo bar", m.get("b"));
54  		assertEquals(false, m.get("c"));
55  
56  		in = "a=$n(1)&b=foo%20bar&c=$b(false)";
57  		m = (Map)p.parse(in, Object.class);
58  		assertEquals(1, m.get("a"));
59  		assertEquals("foo bar", m.get("b"));
60  		assertEquals(false, m.get("c"));
61  
62  		ObjectMap jm = (ObjectMap)p.parse("x=$a($o(attribute=value),$o(attribute='value'))", Object.class);
63  		assertEquals("value", jm.getObjectList("x").getObjectMap(0).getString("attribute"));
64  		assertEquals("'value'", jm.getObjectList("x").getObjectMap(1).getString("attribute"));
65  
66  		ObjectList jl = (ObjectList)p.parse("_value=$a($o(attribute=value),$o(attribute='value'))", Object.class);
67  		assertEquals("value", jl.getObjectMap(0).getString("attribute"));
68  		assertEquals("'value'", jl.getObjectMap(1).getString("attribute"));
69  
70  		A1 b = new A1();
71  		A2 tl = new A2();
72  		tl.add(new A3("name0","value0"));
73  		tl.add(new A3("name1","value1"));
74  		b.list = tl;
75  
76  		in = new UrlEncodingSerializer().setProperty(SERIALIZER_addBeanTypeProperties, true).serialize(b);
77  		b = (A1)p.parse(in, Object.class);
78  		assertEquals("value1", b.list.get(1).value);
79  
80  		in = UrlEncodingSerializer.DEFAULT.serialize(b);
81  		b = p.parse(in, A1.class);
82  		assertEquals("value1", b.list.get(1).value);
83  	}

153 	//====================================================================================================
154 	@Test
155 	public void testParserListeners() throws Exception {
156 		final List<String> events = new LinkedList<String>();
157 		UonParser p = new UrlEncodingParser().setProperty(BEAN_ignoreUnknownBeanProperties, true);
158 		p.addListener(
159 			new ParserListener() {
160 				@Override /* ParserListener */
161 				public <T> void onUnknownProperty(String propertyName, Class<T> beanClass, T bean, int line, int col) {
162 					events.add(propertyName + "," + line + "," + col);
163 				}
164 			}
165 		);
166 
167 		String in = "a=1&unknownProperty=foo&b=2";
168 		p.parse(in, B.class);
169 		assertEquals(1, events.size());
170 		assertEquals("unknownProperty,1,4", events.get(0));
171 	}
