67  		assertEquals("b", m.get("a"));

72  		assertTrue(m.get("e") instanceof Boolean);
73  		assertEquals(Boolean.TRUE, m.get("e"));
74  		assertNull(m.get("f"));
75

79  		assertTrue(m.get("c") instanceof Number);
80  		assertEquals(123, m.get("c"));
81  		assertTrue(m.get("d") instanceof Boolean);

486 	// Unicode character test
487 	//====================================================================================================
488 	@Test
489 	public void testUnicodeChars() throws Exception {
490 		String t;
491 		Map m;
492 
493 		// 2-byte UTF-8 character
494 		// Top level
495 		t = "_value=¢";
496 		assertEquals("¢", p.parse(t, Object.class));
497 		assertEquals("¢", p.parse(t, String.class));
498 		t = "_value=%C2%A2";
499 		assertEquals("¢", p.parse(t, Object.class));
500 		assertEquals("¢", p.parse(t, String.class));
501 		t = "¢";
502 		assertEquals("¢", p.parseParameter(t, Object.class));
503 		assertEquals("¢", p.parseParameter(t, String.class));
504 		t = "%C2%A2";
505 		assertEquals("%C2%A2", p.parseParameter(t, Object.class));
506 		assertEquals("%C2%A2", p.parseParameter(t, String.class));
507 
508 		// 2nd level
509 		t = "?%C2%A2=%C2%A2";
510 		m = (Map)p.parse(t, Object.class);
511 		assertEquals("¢", m.get("¢"));
512 
513 		// 3rd level
514 		t = "?%C2%A2=$o(%C2%A2=%C2%A2)";
515 		m = (Map)p.parse(t, Object.class);
516 		assertEquals("¢", ((Map)m.get("¢")).get("¢"));
517 
518 		// 3-byte UTF-8 character
519 		// Top level
520 		t = "_value=€";
521 		assertEquals("€", p.parse(t, Object.class));
522 		assertEquals("€", p.parse(t, String.class));
523 		t = "_value=%E2%82%AC";
524 		assertEquals("€", p.parse(t, Object.class));
