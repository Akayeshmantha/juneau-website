13  package org.apache.juneau.dto;
14  
15  import java.sql.*;
16  import java.util.*;
17  
18  import org.apache.juneau.internal.*;
19  
20  /**
21   * Transforms an SQL {@link ResultSet ResultSet} into a list of maps.
22   * <p>
23   * 	Loads the entire result set into an in-memory data structure, and then closes the result set object.
24   *
25   * @author James Bognar (james.bognar@salesforce.com)
26   */
27  public final class ResultSetList extends LinkedList<Map<String,Object>> {
28  
29  	private static final long serialVersionUID = 1L;
30  
31  	/**
32  	 * Constructor.
33  	 *
34  	 * @param rs The result set to load into this DTO.
35  	 * @param pos The start position (zero-indexed).
36  	 * @param limit The maximum number of rows to retrieve.
37  	 * @param includeRowNums Make the first column be the row number.
38  	 * @throws SQLException Database error.
39  	 */
40  	public ResultSetList(ResultSet rs, int pos, int limit, boolean includeRowNums) throws SQLException {
41  		try {
42  			int rowNum = pos;
43  
44  			// Get the column names.
45  			ResultSetMetaData rsmd = rs.getMetaData();
46  			int offset = (includeRowNums ? 1 : 0);
47  			int cc = rsmd.getColumnCount();
48  			String[] columns = new String[cc + offset];
49  			if (includeRowNums)
50  				columns[0] = "ROW";
51  			int[] colTypes = new int[cc];
52  
53  			for (int i = 0; i < cc; i++) {
54  				columns[i+offset] = rsmd.getColumnName(i+1);
55  				colTypes[i] = rsmd.getColumnType(i+1);
56  			}
57  
58  			while (--pos > 0 && rs.next()) {}
59  
60  			// Get the rows.
61  			while (limit-- > 0 && rs.next()) {
62  				Object[] row = new Object[cc + offset];
63  				if (includeRowNums)
64  					row[0] = rowNum++;
65  				for (int i = 0; i < cc; i++) {
66  					Object o = readEntry(rs, i+1, colTypes[i]);
67  					row[i+offset] = o;
68  				}
69  				add(new SimpleMap(columns, row));
70  			}
71  		} finally {
72  			rs.close();
73  		}
74  	}
75  
76  	/**
77  	 * Reads the specified column from the current row in the result set.
78  	 * Subclasses can override this method to handle specific data types in special ways.
79  	 *
80  	 * @param rs The result set to read from.
81  	 * @param col The column number (indexed by 1).
82  	 * @param dataType The {@link Types type} of the entry.
83  	 * @return The entry as an Object.
84  	 */
85  	protected Object readEntry(ResultSet rs, int col, int dataType) {
86  		try {
87  			switch (dataType) {
88  				case Types.BLOB:
89  					Blob b = rs.getBlob(col);
90  					return "blob["+b.length()+"]";
91  				case Types.CLOB:
92  					Clob c = rs.getClob(col);
93  					return "clob["+c.length()+"]";
94  				case Types.LONGVARBINARY:
95  					return "longvarbinary["+IOUtils.count(rs.getBinaryStream(col))+"]";
96  				case Types.LONGVARCHAR:
97  					return "longvarchar["+IOUtils.count(rs.getAsciiStream(col))+"]";
98  				case Types.LONGNVARCHAR:
99  					return "longnvarchar["+IOUtils.count(rs.getCharacterStream(col))+"]";
100 				case Types.TIMESTAMP:
101 					return rs.getTimestamp(col);  // Oracle returns com.oracle.TIMESTAMP objects from getObject() which isn't a Timestamp.
102 				default:
103 					return rs.getObject(col);
104 			}
105 		} catch (Exception e) {
106 			return e.getLocalizedMessage();
107 		}
108 	}
109 }
