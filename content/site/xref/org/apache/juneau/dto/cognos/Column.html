13  package org.apache.juneau.dto.cognos;
14  
15  import org.apache.juneau.annotation.*;
16  import org.apache.juneau.transform.*;
17  import org.apache.juneau.xml.annotation.*;
18  
19  /**
20   * Represents a meta-data column in a Cognos dataset.
21   * <p>
22   * 	When serialized to XML, creates the following construct:
23   * <p class='bcode'>
24   * 	<xt>&lt;item</xt> <xa>name</xa>=<xs>'name'</xs> <xa>type</xa>=<xs>'xs:String'</xs> <xa>length</xa>=<xs>'255'</xs>/&gt;
25   * </p>
26   *
27   * @author James Bognar (james.bognar@salesforce.com)
28   */
29  @SuppressWarnings({"rawtypes","hiding"})
30  @Bean(typeName="item",properties={"name","type","length"})
31  public class Column {
32  
33  	private String name, type;
34  	private Integer length;
35  	PojoSwap pojoSwap;
36  
37  	/** Bean constructor. */
38  	public Column() {}
39  
40  	/**
41  	 * Constructor.
42  	 *
43  	 * @param name The column name.
44  	 * @param type The column type (e.g. <js>"xs:String"</js>).
45  	 */
46  	public Column(String name, String type) {
47  		this(name, type, null);
48  	}
49  
50  	/**
51  	 * Constructor.
52  	 *
53  	 * @param name The column name.
54  	 * @param type The column type (e.g. <js>"xs:String"</js>).
55  	 * @param length The column length (e.g. <code>255</code>).
56  	 */
57  	public Column(String name, String type, Integer length) {
58  		this.name = name;
59  		this.type = type;
60  		this.length = length;
61  	}
62  
63  	/**
64  	 * Associates a POJO swap with this column.
65  	 * <p>
66  	 * 	Typically used to define columns that don't exist on the underlying beans being serialized.
67  	 * <p>
68  	 * 	For example, the <code>AddressBookResource</code> sample defined the following POJO swap
69  	 * 		to define an additional <js>"numAddresses"</js> column even though no such property exists
70  	 * 		on the serialized beans.
71  	 * <p class='bcode'>
72  	 * 	Column c = <jk>new</jk> Column(<js>"numAddresses"</js>, <js>"xs:int"</js>)
73  	 * 		.addPojoSwaps(
74  	 * 			<jk>new</jk> PojoSwap<Person,Integer>() {
75  	 * 				<ja>@Override</ja>
76  	 * 				<jk>public</jk> Integer swap(Person p) {
77  	 * 					<jk>return</jk> p.<jf>addresses</jf>.size();
78  	 * 				}
79  	 * 			}
80  	 * 		);
81  	 * </p>
82  	 *
83  	 * @param pojoSwap The POJO swap to associate with the column.
84  	 * @return This object (for method chaining).
85  	 */
86  	public Column addPojoSwap(PojoSwap pojoSwap) {
87  		this.pojoSwap = pojoSwap;
88  		return this;
89  	}
90  
91  	//--------------------------------------------------------------------------------
92  	// Bean properties
93  	//--------------------------------------------------------------------------------
94  
95  	/**
96  	 * Bean property getter:  <property>name</property>.
97  	 *
98  	 * @return The value of the <property>name</property> property on this bean, or <jk>null</jk> if it is not set.
99  	 */
100 	@Xml(format=XmlFormat.ATTR)
101 	public String getName() {
102 		return name;
103 	}
104 
105 	/**
106 	 * Bean property setter:  <property>name</property>.
107 	 *
108 	 * @param name The new value for the <property>name</property> property on this bean.
109 	 * @return This object (for method chaining).
110 	 */
111 	public Column setName(String name) {
112 		this.name = name;
113 		return this;
114 	}
115 
116 	/**
117 	 * Bean property getter:  <property>type</property>.
118 	 *
119 	 * @return The value of the <property>type</property> property on this bean, or <jk>null</jk> if it is not set.
120 	 */
121 	@Xml(format=XmlFormat.ATTR)
122 	public String getType() {
123 		return type;
124 	}
125 
126 	/**
127 	 * Bean property setter:  <property>type</property>.
128 	 *
129 	 * @param type The new value for the <property>type</property> property on this bean.
130 	 * @return This object (for method chaining).
131 	 */
132 	public Column setType(String type) {
133 		this.type = type;
134 		return this;
135 	}
136 
137 	/**
138 	 * Bean property getter:  <property>length</property>.
139 	 *
140 	 * @return The value of the <property>length</property> property on this bean, or <jk>null</jk> if length is not applicable for the specified type.
141 	 */
142 	@Xml(format=XmlFormat.ATTR)
143 	public Integer getLength() {
144 		return length;
145 	}
146 
147 	/**
148 	 * Bean property setter:  <property>length</property>.
149 	 *
150 	 * @param length The new value for the <property>length</property> property on this bean.
151 	 * Can be <jk>null</jk> if length is not applicable for the specified type.
152 	 * @return This object (for method chaining).
153 	 */
154 	public Column setLength(Integer length) {
155 		this.length = length;
156 		return this;
157 	}
158 }
