21  /**
22   * Represents a Cognos dataset.
23   * <p>
24   * 	When serialized to XML, creates the following construct (example pulled from <code>AddressBookResource</code>):
25   * <p class='bcode'>
26   * 	<xt>&lt;?xml</xt> <xa>version</xa>=<xs>'1.0'</xs> <xa>encoding</xa>=<xs>'UTF-8'</xs><xt>?&gt;</xt>
27   * 	<xt>&lt;c:dataset <xa>xmlns:c</xa>=<xs>'http://developer.cognos.com/schemas/xmldata/1/'</xs>&gt;</xt>;
28   * 		<xt>&lt;c:metadata&gt;</xt>
29   * 			<xt>&lt;c:item</xt> <xa>name</xa>=<xs>'name'</xs> <xa>type</xa>=<xs>'xs:String'</xs> <xa>length</xa>=<xs>'255'</xs><xt>/&gt;</xt>
30   * 			<xt>&lt;c:item</xt> <xa>name</xa>=<xs>'age'</xs> <xa>type</xa>=<xs>'xs:int'</xs><xt>/&gt;</xt>
31   * 			<xt>&lt;c:item</xt> <xa>name</xa>=<xs>'numAddresses'</xs> <xa>type</xa>=<xs>'xs:int'</xs><xt>/&gt;</xt>
32   * 		<xt>&lt;/c:metadata&gt;</xt>
33   * 		<xt>&lt;c:data&gt;</xt>
34   * 			<xt>&lt;c:row&gt;</xt>
35   * 				<xt>&lt;c:value&gt;</xt>Barack Obama<xt>&lt;/c:value&gt;</xt>
36   * 				<xt>&lt;c:value&gt;</xt>52<xt>&lt;/c:value&gt;</xt>
37   * 				<xt>&lt;c:value&gt;</xt>2<xt>&lt;/c:value&gt;</xt>
38   * 			<xt>&lt;/c:row&gt;</xt>
39   * 			<xt>&lt;c:row&gt;</xt>
40   * 				<xt>&lt;c:value&gt;</xt>George Walker Bush<xt>&lt;/c:value&gt;</xt>
41   * 				<xt>&lt;c:value&gt;</xt>67<xt>&lt;/c:value&gt;</xt>
42   * 				<xt>&lt;c:value&gt;</xt>2<xt>&lt;/c:value&gt;</xt>
43   * 			<xt>&lt;/c:row&gt;</xt>
44   * 		<xt>&lt;/c:data&gt;</xt>
45   * 	<xt>&lt;/c:dataset&gt;</xt>
46   * </p>
47   * <p>
48   * 	Only 2-dimentional POJOs (arrays or collections of maps or beans) can be serialized to Cognos.
49   *
50   * <h6 class='topic'>Example</h6>
51   * <p>
52   * 	The construct shown above is a serialized <code>AddressBook</code> object which is a subclass of <code>LinkedList&lt;Person&gt;</code>.
53   * 	The code for generating the XML is as follows...
54   * </p>
55   * <p class='bcode'>
56   * 	Column[] items = {
57   * 		<jk>new</jk> Column(<js>"name"</js>, <js>"xs:String"</js>, 255),
58   * 		<jk>new</jk> Column(<js>"age"</js>, <js>"xs:int"</js>),
59   * 		<jk>new</jk> Column(<js>"numAddresses"</js>, <js>"xs:int"</js>)
60   * 			.addPojoSwap(
61   * 				<jk>new</jk> PojoSwap&ltPerson,Integer&gt;() {
62   * 					<ja>@Override</ja>
63   * 					<jk>public</jk> Integer swap(Person p) {
64   * 						<jk>return</jk> p.<jf>addresses</jf>.size();
65   * 					}
66   * 				}
67   * 			)
68   * 	};
69   *
70   * 	DataSet ds = <jk>new</jk> DataSet(items, <jsf>addressBook</jsf>, BeanContext.<jsf>DEFAULT</jsf>);
71   *
72   * 	String xml = XmlSerializer.<jsf>DEFAULT_SQ</jsf>.serialize(ds);
73   * </p>
74   *
75   * @author James Bognar (james.bognar@salesforce.com)
76   */
77  @SuppressWarnings("unchecked")
78  @Bean(typeName="dataset", properties={"metadata","data"})
79  public class DataSet {
80  
81  	private Column[] metaData;
82  	private List<Row> data;
83  
84  	/** Bean constructor. */
85  	public DataSet() {}
86  
87  	/**
88  	 * Constructor.
89  	 *
90  	 * @param columns The meta-data that represents the columns in the dataset.
91  	 * @param o The POJO being serialized to Cognos.
92  	 * 	Must be an array/collection of beans/maps.
93  	 * @param beanContext The bean context used to convert POJOs to strings.
94  	 * @throws Exception An error occurred trying to serialize the POJO.
95  	 */
96  	public DataSet(Column[] columns, Object o, BeanContext beanContext) throws Exception {
97  		metaData = columns;
98  		data = new LinkedList<Row>();
99  		if (o != null) {
100 			if (o.getClass().isArray())
101 				o = Arrays.asList((Object[])o);
102 			if (o instanceof Collection) {
103 				Collection<?> c = (Collection<?>)o;
104 				for (Object o2 : c) {
105 					Row r = new Row();
106 					Map<?,?> m = null;
107 					if (o2 instanceof Map)
108 						m = (Map<?,?>)o2;
109 					else
110 						m = beanContext.forBean(o2);
111 					for (Column col : columns) {
112 						Object v;
113 						if (col.pojoSwap != null)
114 							v = col.pojoSwap.swap(o2, beanContext);
115 						else
116 							v = m.get(col.getName());
117 						r.add(v == null ? null : v.toString());
118 					}
119 					data.add(r);
120 				}
121 			}
122 		}
123 	}
124 
125 	/**
126 	 * Represents a row of data.
127 	 * <p>
128 	 * 	When serialized to XML, creates the following construct (example pulled from <code>AddressBookResource</code>):
129 	 * <p class='bcode'>
130 	 * 	<xt>&lt;row&gt;</xt>
131 	 * 		<xt>&lt;value&gt;</xt>Barack Obama<xt>&lt;/value&gt;</xt>
132 	 * 		<xt>&lt;value&gt;</xt>52<xt>&lt;/value&gt;</xt>
133 	 * 		<xt>&lt;value&gt;</xt>2<xt>&lt;/value&gt;</xt>
134 	 * 	<xt>&lt;/row&gt;</xt>
135 	 * </p>
136 	 *
137 	 * @author James Bognar (james.bognar@salesforce.com)
138 	 */
139 	@Bean(typeName="row")
140 	public static class Row {
141 		private List<String> values = new LinkedList<String>();
142 
143 		private void add(String value) {
144 			values.add(value);
145 		}
146 
147 		/**
148 		 * Returns the values in this row.
149 		 *
150 		 * @return The values in this row.
151 		 */
152 		@Xml(format=XmlFormat.COLLAPSED, childName="value")
153 		public List<String> getValues() {
154 			return values;
155 		}
156 	}
157

164 	 * Bean property getter:  <property>metadata</property>.
165 	 *
166 	 * @return The value of the <property>metadata</property> property on this bean, or <jk>null</jk> if it is not set.
167 	 */
168 	@BeanProperty(name="metadata")
169 	public Column[] getMetaData() {
170 		return metaData;
171 	}
172 
173 	/**
174 	 * Bean property setter:  <property>metadata</property>.
175 	 *
176 	 * @param metaData The new value for the <property>metadata</property> property on this bean.
177 	 * @return This object (for method chaining).
178 	 */
179 	@BeanProperty(name="metadata")
180 	public DataSet setMetaData(Column[] metaData) {
181 		this.metaData = metaData;
182 		return this;
183 	}
184 
185 	/**
186 	 * Bean property getter:  <property>data</property>.
187 	 *
188 	 * @return The value of the <property>data</property> property on this bean, or <jk>null</jk> if it is not set.
189 	 */
190 	@BeanProperty(name="data")
191 	public List<Row> getData() {
192 		return data;
193 	}
194 
195 	/**
196 	 * Bean property setter:  <property>data</property>.
197 	 *
198 	 * @param data The new value for the <property>data</property> property on this bean.
199 	 * @return This object (for method chaining).
200 	 */
201 	@BeanProperty(name="data")
202 	public DataSet setData(List<Row> data) {
203 		this.data = data;
204 		return this;
205 	}
206 }
