13  package org.apache.juneau.dto.jsonschema;
14  
15  import java.io.*;
16  import java.net.*;
17  
18  import org.apache.juneau.internal.*;
19  import org.apache.juneau.json.*;
20  
21  @SuppressWarnings("serial")
22  class Sample {
23  
24  	public static void main(String[] args) {
25  
26  		// Create a SchemaMap for looking up schemas.
27  		SchemaMap schemaMap = new SchemaMap() {
28  
29  			@Override /* SchemaMap */
30  			public Schema load(URI uri) {
31  				Reader r = null;
32  				try {
33  					r = new InputStreamReader(uri.toURL().openStream(), IOUtils.UTF8);
34  					Schema s = JsonParser.DEFAULT.parse(r, Schema.class);
35  					return s;
36  				} catch (Exception e) {
37  					throw new RuntimeException(e);
38  				} finally {
39  					if (r != null) {
40  						try {
41  							r.close();
42  						} catch (IOException e) {
43  						}
44  					}
45  				}
46  			}
47  		};
48  
49  		// Get schema from the schema map.
50  		Schema purchaseOrderSchema = schemaMap.get("http://www.ibm.com/purchase-order/PurchaseOrder#");
51  
52  		JsonType streetType = purchaseOrderSchema
53  			.getProperty("address",true)                         // Get "address" property, resolved to Address schema.
54  			.getProperty("street")                               // Get "street" property.
55  			.getTypeAsJsonType();                                // Get data type.
56  		System.err.println("streetType=" + streetType);         // Prints "streetType=string"
57  
58  		JsonType productIdType = purchaseOrderSchema
59  			.getProperty("product")                              // Get "product" property
60  			.getItemsAsSchemaArray()                             // Get "items".
61  			.get(0)                                              // Get first entry.
62  			.resolve()                                           // Resolve to Product schema.
63  			.getProperty("productId")                            // Get "productId" property.
64  			.getTypeAsJsonType();                                // Get data type.
65  		System.err.println("productIdType=" + productIdType);   // Prints "productIdType=number"
66  	}
67  }
