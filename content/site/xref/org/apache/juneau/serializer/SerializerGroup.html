23  /**
24   * Represents a group of {@link Serializer Serializers} that can be looked up by media type.
25   *
26   *
27   * <h6 class='topic'>Description</h6>
28   * <p>
29   * 	Provides the following features:
30   * <ul class='spaced-list'>
31   * 	<li>Finds serializers based on HTTP <code>Accept</code> header values.
32   * 	<li>Sets common properties on all serializers in a single method call.
33   * 	<li>Locks all serializers in a single method call.
34   * 	<li>Clones existing groups and all serializers within the group in a single method call.
35   * </ul>
36   *
37   *
38   * <h6 class='topic'>Match ordering</h6>
39   * <p>
40   * 	Serializers are matched against <code>Accept</code> strings in the order they exist in this group.
41   * <p>
42   * 	Adding new entries will cause the entries to be prepended to the group.
43   *  	This allows for previous serializers to be overridden through subsequent calls.
44   * <p>
45   * 	For example, calling <code>g.append(S1.<jk>class</jk>,S2.<jk>class</jk>).append(S3.<jk>class</jk>,S4.<jk>class</jk>)</code>
46   * 	will result in the order <code>S3, S4, S1, S2</code>.
47   *
48   *
49   * <h6 class='topic'>Examples</h6>
50   * <p class='bcode'>
51   * 	<jc>// Construct a new serializer group</jc>
52   * 	SerializerGroup g = <jk>new</jk> SerializerGroup();
53   *
54   * 	<jc>// Add some serializers to it</jc>
55   * 	g.append(JsonSerializer.<jk>class</jk>, XmlSerializer.<jk>class</jk>);
56   *
57   * 	<jc>// Change settings for all serializers in the group and lock it.</jc>
58   * 	g.setProperty(SerializerContext.<jsf>SERIALIZER_useIndentation</jsf>, <jk>true</jk>)
59   * 		.addPojoSwaps(CalendarSwap.ISO8601DT.<jk>class</jk>)
60   * 		.lock();
61   *
62   * 	<jc>// Find the appropriate serializer by Accept type</jc>
63   * 	String mediaTypeMatch = g.findMatch(<js>"text/foo, text/json;q=0.8, text/*;q:0.6, *\/*;q=0.0"</js>);
64   * 	WriterSerializer s = (WriterSerializer)g.getSerializer(mediaTypeMatch);
65   *
66   * 	<jc>// Serialize a bean to JSON text </jc>
67   * 	AddressBook addressBook = <jk>new</jk> AddressBook();  <jc>// Bean to serialize.</jc>
68   * 	String json = s.serialize(addressBook);
69   * </p>
70   *
71   * @author James Bognar (james.bognar@salesforce.com)
72   */
73  public final class SerializerGroup extends Lockable {
74  
75  	// Maps media-types to serializers.
76  	private final Map<String,Serializer> serializerMap = new ConcurrentHashMap<String,Serializer>();
77  
78  	// Maps Accept headers to matching media types.
79  	private final Map<String,String> mediaTypeMappings = new ConcurrentHashMap<String,String>();
80  
81  	private final CopyOnWriteArrayList<Serializer> serializers = new CopyOnWriteArrayList<Serializer>();
82  
83  	private final ReadWriteLock lock = new ReentrantReadWriteLock();
84  	private final Lock rl = lock.readLock(), wl = lock.writeLock();
85  
86  
87  	/**
88  	 * Registers the specified serializers with this group.
89  	 *
90  	 * @param s The serializers to append to this group.
91  	 * @return This object (for method chaining).
92  	 * @throws Exception Thrown if {@link Serializer} could not be constructed.
93  	 */
94  	public SerializerGroup append(Class<? extends Serializer>...s) throws Exception {
95  		checkLock();
96  		wl.lock();
97  		try {
98  			serializerMap.clear();
99  			mediaTypeMappings.clear();
100 			for (Class<? extends Serializer> ss : reverse(s)) {
101 				try {
102 					append(ss);
103 				} catch (NoClassDefFoundError e) {
104 					// Ignore if dependent library not found (e.g. Jena).
105 					System.err.println(e);
106 				}
107 			}
108 		} finally {
109 			wl.unlock();
110 		}
111 		return this;
112 	}
113 
114 	/**
115 	 * Same as {@link #append(Class[])}, except specify a single class to avoid unchecked compile warnings.
116 	 *
117 	 * @param c The serializer to append to this group.
118 	 * @return This object (for method chaining).
119 	 * @throws Exception Thrown if {@link Serializer} could not be constructed.
120 	 */
121 	public SerializerGroup append(Class<? extends Serializer> c) throws Exception {
122 		checkLock();
123 		wl.lock();
124 		try {
125 			serializerMap.clear();
126 			mediaTypeMappings.clear();
127 			serializers.add(0, c.newInstance());
128 		} catch (NoClassDefFoundError e) {
129 			// Ignore if dependent library not found (e.g. Jena).
130 			System.err.println(e);
131 		} finally {
132 			wl.unlock();
133 		}
134 		return this;
135 	}

137 	/**
138 	 * Returns the serializer registered to handle the specified media type.
139 	 * <p>
140 	 * The media-type string must not contain any parameters or q-values.
141 	 *
142 	 * @param mediaType The media-type string (e.g. <js>"text/json"</js>
143 	 * @return The serializer that handles the specified accept content type, or <jk>null</jk> if
144 	 * 		no serializer is registered to handle it.
145 	 */
146 	public Serializer getSerializer(String mediaType) {
147 		Serializer s = serializerMap.get(mediaType);
148 		if (s == null)
149 			s = serializerMap.get(findMatch(mediaType));
150 		return s;
151 	}
152 
153 	/**
154 	 * Searches the group for a serializer that can handle the specified <code>Accept</code> value.
155 	 * <p>
156 	 * 	The <code>accept</code> value complies with the syntax described in RFC2616, Section 14.1, as described below:
157 	 * <p class='bcode'>
158 	 * 	Accept         = "Accept" ":"
159 	 * 	                  #( media-range [ accept-params ] )
160 	 *
161 	 * 	media-range    = ( "*\/*"
162 	 * 	                  | ( type "/" "*" )
163 	 * 	                  | ( type "/" subtype )
164 	 * 	                  ) *( ";" parameter )
165 	 * 	accept-params  = ";" "q" "=" qvalue *( accept-extension )
166 	 * 	accept-extension = ";" token [ "=" ( token | quoted-string ) ]
167 	 * </p>
168 	 * <p>
169 	 * 	The general idea behind having the serializer resolution be a two-step process is so that
170 	 * 	the matched media type can be passed in to the {@link WriterSerializer#doSerialize(SerializerSession, Object)} method.
171 	 * 	For example...
172 	 * <p class='bcode'>
173 	 * 	String acceptHeaderValue = request.getHeader(<js>"Accept"</js>);
174 	 * 	String matchingMediaType = group.findMatch(acceptHeaderValue);
175 	 * 	if (matchingMediaType == <jk>null</jk>)
176 	 * 		<jk>throw new</jk> RestException(<jsf>SC_NOT_ACCEPTABLE</jsf>);
177 	 * 	WriterSerializer s = (WriterSerializer)group.getSerializer(matchingMediaType);
178 	 *  s.serialize(getPojo(), response.getWriter(), response.getProperties(), matchingMediaType);
179 	 * </p>
180 	 *
181 	 * @param acceptHeader The HTTP <l>Accept</l> header string.
182 	 * @return The media type registered by one of the parsers that matches the <code>accept</code> string,
183 	 * 	or <jk>null</jk> if no media types matched.
184 	 */
185 	public String findMatch(String acceptHeader) {
186 		rl.lock();
187 		try {
188 			String mt = mediaTypeMappings.get(acceptHeader);
189 			if (mt != null)
190 				return mt;
191 
192 			MediaRange[] mr = MediaRange.parse(acceptHeader);
193 			if (mr.length == 0)
194 				mr = MediaRange.parse("*/*");
195 
196 			for (MediaRange a : mr) {
197 				for (Serializer s : serializers) {
198 					for (MediaRange a2 : s.getMediaRanges()) {
199 						if (a.matches(a2)) {
200 							mt = a2.getMediaType();
201 							mediaTypeMappings.put(acceptHeader, mt);
202 							serializerMap.put(mt, s);
203 							return mt;
204 						}
205 					}
206 				}
207 			}
208 			return null;
209 		} finally {
210 			rl.unlock();
211 		}
212 	}

214 	/**
215 	 * Returns the media types that all parsers in this group can handle
216 	 * <p>
217 	 * Entries are ordered in the same order as the parsers in the group.
218 	 *
219 	 * @return The list of media types.
220 	 */
221 	public List<String> getSupportedMediaTypes() {
222 		List<String> l = new ArrayList<String>();
223 		for (Serializer s : serializers)
224 			for (String mt : s.getMediaTypes())
225 				if (! l.contains(mt))
226 					l.add(mt);
227 		return l;
228 	}

340 	/**
341 	 * Locks this group and all serializers in this group.
342 	 */
343 	@Override /* Lockable */
344 	public SerializerGroup lock() {
345 		super.lock();
346 		for (Serializer s : serializers)
347 			s.lock();
348 		return this;
349 	}
350 
351 	/**
352 	 * Clones this group and all serializers in this group.
353 	 */
354 	@Override /* Lockable */
355 	public SerializerGroup clone() throws CloneNotSupportedException {
356 		SerializerGroup g = new SerializerGroup();
357 
358 		List<Serializer> l = new ArrayList<Serializer>(serializers.size());
359 		for (Serializer s : serializers)
360 			l.add(s.clone());
361 
362 		g.serializers.addAll(l);
363 
364 		return g;
365 	}
