13  package org.apache.juneau.server;
14  
15  import org.apache.juneau.server.annotation.*;
16  
17  /**
18   * Class used for defining method-level matchers using the {@link RestMethod#matchers()} annotation.
19   * <p>
20   * Matchers are used to allow multiple Java methods to handle requests assigned to the same
21   * 	URL path pattern, but differing based on some request attribute, such as a specific header value.
22   * For example, matchers can be used to provide two different methods for handling requests
23   * 	from two different client versions.
24   * <p>
25   * Java methods with matchers associated with them are always attempted before Java methods
26   * 	without matchers.
27   * This allows a 'default' method to be defined to handle requests where no matchers match.
28   * <p>
29   * When multiple matchers are specified on a method, only one matcher is required to match.
30   * This is opposite from the {@link RestMethod#guards()} annotation, where all guards
31   * 	are required to match in order to execute the method.
32   *
33   * <h6 class='topic'>Example</h6>
34   * <p class='bcode'>
35   * 	<jk>public class</jk> MyResource <jk>extends</jk> RestServlet {
36   *
37   * 		<ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foo"</js>, matchers=IsDNT.<jk>class</jk>)
38   * 		<jk>public</jk> Object doGetWithDNT() {
39   * 			<jc>// Handle request with Do-Not-Track specified</jc>
40   * 		}
41   *
42   * 		<ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/foo"</js>)
43   * 		<jk>public</jk> Object doGetWithoutDNT() {
44   * 			<jc>// Handle request without Do-Not-Track specified</jc>
45   * 		}
46   * 	}
47   *
48   * 	<jk>public class</jk> IsDNT <jk>extends</jk> RestMatcher {
49   * 		<ja>@Override</ja>
50   * 		<jk>public boolean</jk> matches(RestRequest req) {
51   * 			<jk>return</jk> req.getHeader(<jk>int</jk>.<jk>class</jk>, <js>"DNT"</js>, 0) == 1;
52   * 		}
53   * 	}
54   * </p>
55   */
56  public abstract class RestMatcher {
57  
58  	/**
59  	 * Returns <jk>true</jk> if the specified request matches this matcher.
60  	 *
61  	 * @param req The servlet request.
62  	 * @return <jk>true</jk> if the specified request matches this matcher.
63  	 */
64  	public abstract boolean matches(RestRequest req);
65  
66  	/**
67  	 * Returns <jk>true</jk> if this matcher is required to match in order for the method to be invoked.
68  	 * <p>
69  	 * If <jk>false</jk>, then only one of the matchers must match.
70  	 *
71  	 * @return <jk>true</jk> if this matcher is required to match in order for the method to be invoked.
72  	 */
73  	public boolean mustMatch() {
74  		return false;
75  	}
76  }
