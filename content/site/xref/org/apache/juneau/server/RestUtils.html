80  	/**
81  	 * Trims <js>'/'</js> characters from both the start and end of the specified string.
82  	 *
83  	 * @param s The string to trim.
84  	 * @return A new trimmed string, or the same string if no trimming was necessary.
85  	 */
86  	public static String trimSlashes(String s) {
87  		if (s == null)
88  			return null;
89  		while (StringUtils.endsWith(s, '/'))
90  			s = s.substring(0, s.length()-1);
91  		while (s.length() > 0 && s.charAt(0) == '/')
92  			s = s.substring(1);
93  		return s;
94  	}

99  	 * @param s The string to trim.

103 		if (s == null)
104 			return null;

124    /**
125     * Decodes a <code>application/x-www-form-urlencoded</code> string using <code>UTF-8</code> encoding scheme.
126     *
127 	 * @param s The string to decode.
128 	 * @return The decoded string, or <jk>null</jk> if input is <jk>null</jk>.
129     */
130 	public static String decode(String s) {
131 		if (s == null)
132 			return s;
133 		boolean needsDecode = false;
134 		for (int i = 0; i < s.length() && ! needsDecode; i++) {
135 			char c = s.charAt(i);
136 			if (c == '+' || c == '%')
137 				needsDecode = true;
138 		}
139 		if (needsDecode)
140 		try {
141 				return URLDecoder.decode(s, "UTF-8");
142 			} catch (UnsupportedEncodingException e) {/* Won't happen */}
143 		return s;
144 	}

146 	// Characters that do not need to be URL-encoded
147 	private static final AsciiSet unencodedChars = new AsciiSet("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789-_.!~*'()\\");
148 
149    /**
150     * Encodes a <code>application/x-www-form-urlencoded</code> string using <code>UTF-8</code> encoding scheme.
151     *
152     * @param s The string to encode.
153     * @return The encoded string, or <jk>null</jk> if input is <jk>null</jk>.
154     */
155 	public static String encode(String s) {
156 		if (s == null)
157 			return null;
158 		boolean needsEncode = false;
159 		for (int i = 0; i < s.length() && ! needsEncode; i++)
160 			needsEncode |= (! unencodedChars.contains(s.charAt(i)));
161 		if (needsEncode)
162 		try {
163 				return URLEncoder.encode(s, "UTF-8");
164 			} catch (UnsupportedEncodingException e) {/* Won't happen */}
165 		return s;
166 	}

168    /**
169     * Identical to {@link HttpServletRequest#getPathInfo()} but doesn't decode encoded characters.
170     *
171     * @param req The HTTP request
172     * @return The undecoded path info.
173     */
174 	public static String getPathInfoUndecoded(HttpServletRequest req) {
175 		String requestURI = req.getRequestURI();
176 		String contextPath = req.getContextPath();
177 		String servletPath = req.getServletPath();
178 		int l = contextPath.length() + servletPath.length();
179 		if (requestURI.length() == l)
180 			return null;
181 		return requestURI.substring(l);
182 	}

184 	/**
185 	 * Efficiently trims the path info part from a request URI.
186 	 * <p>
187 	 * The result is the URI of the servlet itself.
188 	 *
189 	 * @param requestURI The value returned by {@link HttpServletRequest#getRequestURL()}
190 	 * @param contextPath The value returned by {@link HttpServletRequest#getContextPath()}
191 	 * @param servletPath The value returned by {@link HttpServletRequest#getServletPath()}
192 	 * @return The same StringBuilder with remainder trimmed.
193 	 */
194 	public static StringBuffer trimPathInfo(StringBuffer requestURI, String contextPath, String servletPath) {
195 		if (servletPath.equals("/"))
196 			servletPath = "";
197 		if (contextPath.equals("/"))
198 			contextPath = "";
199 
200 		try {
201 			// Given URL:  http://hostname:port/servletPath/extra
202 			// We want:    http://hostname:port/servletPath
203 			int sc = 0;
204 			for (int i = 0; i < requestURI.length(); i++) {
205 				char c = requestURI.charAt(i);
206 				if (c == '/') {
207 					sc++;
208 					if (sc == 3) {
209 						if (servletPath.isEmpty()) {
210 							requestURI.setLength(i);
211 							return requestURI;
212 						}
213 
214 						// Make sure context path follows the authority.
215 						for (int j = 0; j < contextPath.length(); i++, j++)
216 							if (requestURI.charAt(i) != contextPath.charAt(j))
217 								throw new Exception("case=1");
218 
219 						// Make sure servlet path follows the authority.
220 						for (int j = 0; j < servletPath.length(); i++, j++)
221 							if (requestURI.charAt(i) != servletPath.charAt(j))
222 								throw new Exception("case=2");
223 
224 						// Make sure servlet path isn't a false match (e.g. /foo2 should not match /foo)
225 						c = (requestURI.length() == i ? '/' : requestURI.charAt(i));
226 						if (c == '/' || c == '?') {
227 							requestURI.setLength(i);
228 							return requestURI;
229 						}
230 
231 						throw new Exception("case=3");
232 					}
233 				} else if (c == '?') {
234 					if (sc != 2)
235 						throw new Exception("case=4");
236 					if (servletPath.isEmpty()) {
237 						requestURI.setLength(i);
238 						return requestURI;
239 					}
240 					throw new Exception("case=5");
241 				}
242 			}
243 			if (servletPath.isEmpty())
244 				return requestURI;
245 			throw new Exception("case=6");
246 		} catch (Exception e) {
247 			throw new RuntimeException("Could not find servlet path in request URI.  URI=["+requestURI+"], servletPath=["+servletPath+"]", e);
248 		}
249 	}
