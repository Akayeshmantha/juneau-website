153 	@Override /* Servlet */
154 	public synchronized void init(ServletConfig servletConfig) throws ServletException {
155 		try {
156 			log(FINE, "Servlet {0} init called.", getClass().getName());
157 			this.servletConfig = servletConfig;
158 
159 			if (isInitialized)
160 				return;
161 
162 			super.init(servletConfig);
163 
164 			// Find resource resource bundle location.
165 			for (Map.Entry<Class<?>,RestResource> e : restResourceAnnotationsChildFirst.entrySet()) {
166 				Class<?> c = e.getKey();
167 				RestResource r = e.getValue();
168 				if (! r.messages().isEmpty()) {
169 					if (msgs == null)
170 						msgs = new MessageBundle(c, r.messages());
171 					else
172 						msgs.addSearchPath(c, r.messages());
173 				}
174 				if (label.isEmpty())
175 					label = r.label();
176 				if (description.isEmpty())
177 					description = r.description();
178 				if (clientVersionHeader.isEmpty())
179 					clientVersionHeader = r.clientVersionHeader();
180 			}
181 			if (msgs == null)
182 				msgs = new MessageBundle(this.getClass(), "");
183 			if (clientVersionHeader.isEmpty())
184 				clientVersionHeader = "X-Client-Version";
185 
186 			styleSheet = createStyleSheet();
187 			favIcon = createFavIcon();
188 			staticFilesMap = Collections.unmodifiableMap(createStaticFilesMap());
189 			staticFilesPrefixes = staticFilesMap.keySet().toArray(new String[0]);
190 
191 			properties = createProperties();
192 			beanFilters = createBeanFilters();
193 			pojoSwaps = createPojoSwaps();
194 			context = ContextFactory.create().setProperties(properties).getContext(RestServletContext.class);
195 			beanContext = createBeanContext(properties, beanFilters, pojoSwaps);
196 			urlEncodingSerializer = createUrlEncodingSerializer(properties, beanFilters, pojoSwaps).lock();
197 			urlEncodingParser = createUrlEncodingParser(properties, beanFilters, pojoSwaps).lock();
198 			serializers = createSerializers(properties, beanFilters, pojoSwaps).lock();
199 			parsers = createParsers(properties, beanFilters, pojoSwaps).lock();
200 			converters = createConverters(properties);
201 			encoders = createEncoders(properties);
202 			guards = createGuards(properties);
203 			mimetypesFileTypeMap = createMimetypesFileTypeMap(properties);
204 			defaultRequestHeaders = new TreeMap<String,String>(String.CASE_INSENSITIVE_ORDER);
205 			defaultRequestHeaders.putAll(createDefaultRequestHeaders(properties));
206 			defaultResponseHeaders = createDefaultResponseHeaders(properties);
207 			responseHandlers = createResponseHandlers(properties);
208 
209 			// Discover the @RestMethod methods available on the resource.
210 			List<String> methodsFound = new LinkedList<String>();   // Temporary to help debug transient duplicate method issue.
211 			for (java.lang.reflect.Method method : this.getClass().getMethods()) {
212 				if (method.isAnnotationPresent(RestMethod.class)) {
213 					RestMethod a = method.getAnnotation(RestMethod.class);
214 					methodsFound.add(method.getName() + "," + a.name() + "," + a.path());
215 					try {
216 						if (! Modifier.isPublic(method.getModifiers()))
217 							throw new RestServletException("@RestMethod method {0}.{1} must be defined as public.", this.getClass().getName(), method.getName());
218 
219 						MethodMeta sm = new MethodMeta(method);
220 						javaRestMethods.put(method.getName(), sm);
221 						ResourceMethod rm = restMethods.get(sm.httpMethod);
222 						if (rm == null)
223 							restMethods.put(sm.httpMethod, sm);
224 						else if (rm instanceof MultiMethod)
225 							((MultiMethod)rm).addSimpleMethod(sm);
226 						else
227 							restMethods.put(sm.httpMethod, new MultiMethod((MethodMeta)rm, sm));
228 					} catch (RestServletException e) {
229 						throw new RestServletException("Problem occurred trying to serialize methods on class {0}, methods={1}", this.getClass().getName(), JsonSerializer.DEFAULT_LAX.serialize(methodsFound)).initCause(e);
230 					}
231 				}
232 			}
233 
234 			for (ResourceMethod m : restMethods.values())
235 				m.complete();
236 
237 			// Discover the child resources.
238 			childResources.putAll(createChildrenMap());
239 
240 			for (RestServlet child : childResources.values())
241 				child.init(servletConfig);
242 
243 			varResolver.addVars(
244 				LocalizationVar.class,
245 				RequestAttrVar.class,
246 				RequestParamVar.class,
247 				RequestVar.class,
248 				SerializedRequestAttrVar.class,
249 				SerializedRequestParamVar.class,
250 				ServletInitParamVar.class,
251 				UrlEncodeVar.class
252 			);
253 
254 		} catch (RestException e) {
255 			// Thrown RestExceptions are simply caught and rethrown on subsequent calls to service().
256 			initException = e;
257 			log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
258 			label = String.valueOf(initException.getLocalizedMessage());
259 		} catch (ServletException e) {
260 			initException = e;
261 			log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
262 			label = String.valueOf(initException.getLocalizedMessage());
263 			throw e;
264 		} catch (Exception e) {
265 			initException = e;
266 			log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
267 			label = String.valueOf(initException.getLocalizedMessage());
268 			throw new ServletException(e);
269 		} catch (Throwable e) {
270 			initException = new Exception(e);
271 			log(SEVERE, e, "Servlet init error on class ''{0}''", getClass().getName());
272 			label = String.valueOf(initException.getLocalizedMessage());
273 			throw new ServletException(e);
274 		} finally {
275 			isInitialized = true;
276 		}
277 	}

284 	 * Creates the child resources of this resource.
285 	 * <p>
286 	 * 	Default implementation calls {@link #createChildren()} and uses the {@link RestResource#path() @RestResource.path()} annotation
287 	 * 		on each child to identify the subpath for the resource which become the keys in this map.
288 	 * 	It then calls the {@link #setParent(RestServlet)} method on the child resource.
289 	 * </p>
290 	 * <p>
291 	 * 	Subclasses can override this method to programatically create child resources
292 	 * 		without using the {@link RestResource#children() @RestResource.children()} annotation.
293 	 * 	When overridding this method, you are responsible for calling {@link #setParent(RestServlet)} on the
294 	 * 		child resources.
295 	 * </p>
296 	 *
297 	 * @return The new mutable list of child resource instances.
298 	 * @throws Exception If an error occurred during servlet instantiation.
299 	 */
300 	protected Map<String,RestServlet> createChildrenMap() throws Exception {
301 		Map<String,RestServlet> m = new LinkedHashMap<String,RestServlet>();
302 		for (RestServlet r : createChildren()) {
303 			r.setParent(this);
304 			String p = r.findPath();
305 			if (p == null)
306 				throw new RestServletException("Child resource ''{0}'' does not define a ''@RestResource.path'' attribute.", r.getClass().getName());
307 			m.put(p, r);
308 		}
309 		return m;
310 	}

313 	 * Creates instances of child resources for this servlet.
314 	 * <p>
315 	 * 	Default implementation uses the {@link RestResource#children() @RestResource.children()} annotation to identify and
316 	 * 		instantiate children.
317 	 * </p>
318 	 * <p>
319 	 * 	Subclasses can override this method to programatically create child resources
320 	 * 		without using the {@link RestResource#children() @RestResource.children()} annotation.
321 	 * </p>
322 	 *
323 	 * @return The new mutable list of child resource instances.
324 	 * @throws Exception If an error occurred during servlet instantiation.
325 	 */
326 	protected List<RestServlet> createChildren() throws Exception {
327 		List<RestServlet> l = new LinkedList<RestServlet>();
328 		for (Class<?> c : getChildClasses()) {
329 			if (isParentClass(RestServlet.class, c))
330 				l.add((RestServlet)c.newInstance());
331 			else
332 				l.add(resolveChild(c));
333 		}
334 		return l;
335 	}

358 	/**
359 	 * Creates the class-level properties associated with this servlet.
360 	 * <p>
361 	 * 	Subclasses can override this method to provide their own class-level properties for this servlet, typically
362 	 * 		by calling <code><jk>super</jk>.createProperties()</code> and appending to the map.
363 	 *	 However, in most cases, the existing set of properties can be added to by overridding {@link #getProperties()}
364 	 * 		and appending to the map returned by <code><jk>super</jk>.getProperties()</code>
365 	 * </p>
366 	 * <p>
367 	 * 	By default, the map returned by this method contains the following:
368 	 * </p>
369 	 * <ul class='spaced-list'>
370 	 * 	<li>Servlet-init parameters.
371 	 * 	<li>{@link RestResource#properties()} annotations in parent-to-child order.
372 	 * 	<li>{@link SerializerContext#SERIALIZER_relativeUriBase} from {@link ServletConfig#getServletContext()}.
373 	 * </ul>
374 	 *
375 	 * @return The resource properties as an {@link ObjectMap}.
376 	 */
377 	protected ObjectMap createProperties() {
378 		ObjectMap m = new ObjectMap();
379 
380 		ServletContext ctx = servletConfig.getServletContext();
381 
382 		// Workaround for bug in Jetty that causes context path to always end in "null".
383 		String ctxPath = ctx.getContextPath();
384 		if (ctxPath.endsWith("null"))
385 			ctxPath = ctxPath.substring(0, ctxPath.length()-4);
386 		m.put(SERIALIZER_relativeUriBase, ctxPath);
387 
388 		// Get the initialization parameters.
389 		for (Enumeration ep = servletConfig.getInitParameterNames(); ep.hasMoreElements();) {
390 			String p = (String)ep.nextElement();
391 			String initParam = servletConfig.getInitParameter(p);
392 			m.put(p, initParam);
393 		}
394 
395 		// Properties are loaded in parent-to-child order to allow overrides.
396 		for (RestResource r : restResourceAnnotationsParentFirst.values())
397 			for (Property p : r.properties())
398 				m.append(getVarResolver().resolve(p.name()), getVarResolver().resolve(p.value()));
399 
400 		return m;
401 	}

494 	 * <p>
495 	 * Subclasses can override this method to provide their own set of serializers for this servlet.
496 	 * They can do this by either creating a new {@link SerializerGroup} from scratch, or appending to the
497 	 * 	group returned by <code><jk>super</jk>.createSerializers()</code>.
498 	 * <p>
499 	 * By default, returns the serializers defined through {@link RestResource#serializers() @RestResource.serializers()} on this class
500 	 * 	and all parent classes.
501 	 *
502 	 * @param properties Servlet-level properties returned by {@link #createProperties()}.
503 	 * @param beanFilters Servlet-level bean filters returned by {@link #createBeanFilters()}.
504 	 * @param pojoSwaps Servlet-level POJO swaps returned by {@link #createPojoSwaps()}.
505 	 * @return The group of serializers.
506 	 * @throws Exception If serializer group could not be constructed for any reason.
507 	 */
508 	protected SerializerGroup createSerializers(ObjectMap properties, Class<?>[] beanFilters, Class<?>[] pojoSwaps) throws Exception {
509 		SerializerGroup g = new SerializerGroup();
510 
511 		// Serializers are loaded in parent-to-child order to allow overrides.
512 		for (RestResource r : restResourceAnnotationsParentFirst.values())
513 			for (Class<? extends Serializer> c : reverse(r.serializers()))
514 				try {
515 					g.append(c);
516 				} catch (Exception e) {
517 					throw new RestServletException("Exception occurred while trying to instantiate Serializer ''{0}''", c.getSimpleName()).initCause(e);
518 				}
519 
520 		g.setProperties(properties);
521 		g.addBeanFilters(beanFilters).addPojoSwaps(pojoSwaps);
522 		return g;
523 	}

2752 	 * @return An input stream of the resource, or <jk>null</jk> if the resource could not be found.
2753 	 * @throws IOException
2754 	 */
2755 	protected InputStream getResource(String name) throws IOException {
2756 		if (! resourceStreams.containsKey(name)) {
2757 			InputStream is = ReflectionUtils.getResource(getClass(), name);
2758 			if (is == null && name.indexOf("..") == -1) {
2759 				File f = new File(name);
2760 				if (f.exists() && f.canRead())
2761 					is = new FileInputStream(f);
2762 			}
2763 			if (is != null) {
2764 				try {
2765 					resourceStreams.put(name, ByteArrayCache.DEFAULT.cache(is));
2766 				} finally {
2767 					is.close();
2768 				}
2769 			}
2770 		}
2771 		byte[] b = resourceStreams.get(name);
2772 		return b == null ? null : new ByteArrayInputStream(b);
2773 	}

2793 	 * Reads the input stream from {@link #getResource(String)} and parses it into a POJO
2794 	 * 	using the parser matched by the specified media type.
2795 	 * <p>

2797 	 * </p>
2798 	 *
2799 	 * @param c The class type of the POJO to create.
2800 	 * @param mediaType The media type of the data in the stream (e.g. <js>"text/json"</js>)
2801 	 * @param name The resource name (e.g. "htdocs/styles.css").
2802 	 * @return The parsed resource, or <jk>null</jk> if the resource could not be found.
2803 	 * @throws IOException
2804 	 * @throws ServletException
2805 	 */

2807 		InputStream is = getResource(name);
2808 		if (is == null)
2809 			return null;
2810 		try {
2811 			Parser p = getParsers().getParser(mediaType);
2812 			if (p != null) {
2813 				try {
2814 					if (p.isReaderParser())
2815 						return p.parse(new InputStreamReader(is, IOUtils.UTF8), c);
2816 					return p.parse(is, c);
2817 				} catch (ParseException e) {
2818 					throw new ServletException("Could not parse resource '' as media type '"+mediaType+"'.");
2819 				}
2820 			}
2821 			throw new ServletException("Unknown media type '"+mediaType+"'");
2822 		} catch (Exception e) {
2823 			throw new ServletException("Could not parse resource with name '"+name+"'", e);
2824 		}
