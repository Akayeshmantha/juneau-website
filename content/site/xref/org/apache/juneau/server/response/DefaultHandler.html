13  package org.apache.juneau.server.response;
14  
15  import static javax.servlet.http.HttpServletResponse.*;
16  
17  import java.io.*;
18  
19  import org.apache.juneau.*;
20  import org.apache.juneau.annotation.*;
21  import org.apache.juneau.serializer.*;
22  import org.apache.juneau.server.*;
23  
24  /**
25   * Response handler for POJOs not handled by other handlers.
26   * <p>
27   * This uses the serializers defined on the response to serialize the POJO.
28   * <p>
29   * The {@link Serializer} used is based on the <code>Accept</code> header on the request.
30   * <p>
31   * The <code>Content-Type</code> header is set to the mime-type defined on the selected
32   * 	serializer based on the {@link Produces#contentType() @Produces.contentType} annotation.
33   * <p>
34   * This handler is registered by default on {@link RestServlet RestServlets} via the
35   * 	default implementation of the {@link RestServlet#createResponseHandlers} method.
36   *
37   * @author James Bognar (james.bognar@salesforce.com)
38   */
39  public class DefaultHandler implements ResponseHandler {
40  
41  	@Override /* ResponseHandler */
42  	public boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException {
43  		SerializerGroup g = res.getSerializerGroup();
44  		String accept = req.getHeader("Accept", "");
45  		String matchingAccept = g.findMatch(accept);
46  		if (matchingAccept != null) {
47  			Serializer s = g.getSerializer(matchingAccept);
48  			String contentType = s.getResponseContentType();
49  			if (contentType == null)
50  				contentType = res.getContentType();
51  			if (contentType == null)
52  				contentType = matchingAccept;
53  			res.setContentType(contentType);
54  			ObjectMap headers = s.getResponseHeaders(res.getProperties());
55  			if (headers != null)
56  				for (String key : headers.keySet())
57  					res.setHeader(key, headers.getString(key));
58  
59  			try {
60  				ObjectMap p = res.getProperties();
61  				if (req.isPlainText()) {
62  					p.put(SerializerContext.SERIALIZER_useIndentation, true);
63  					res.setContentType("text/plain");
64  				}
65  				p.append("mediaType", matchingAccept).append("characterEncoding", res.getCharacterEncoding());
66  				if (! s.isWriterSerializer()) {
67  					OutputStreamSerializer s2 = (OutputStreamSerializer)s;
68  					OutputStream os = res.getNegotiatedOutputStream();
69  					SerializerSession session = s.createSession(os, p, req.getJavaMethod());
70  					s2.serialize(session, output);
71  					os.close();
72  				} else {
73  					WriterSerializer s2 = (WriterSerializer)s;
74  					Writer w = res.getNegotiatedWriter();
75  					SerializerSession session = s.createSession(w, p, req.getJavaMethod());
76  					s2.serialize(session, output);
77  					w.close();
78  				}
79  			} catch (SerializeException e) {
80  				throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
81  			}
82  		} else {
83  			throw new RestException(SC_NOT_ACCEPTABLE,
84  				"Unsupported media-type in request header ''Accept'': ''{0}''\n\tSupported media-types: {1}",
85  				req.getHeader("Accept", ""), g.getSupportedMediaTypes()
86  			);
87  		}
88  		return true;
89  	}
90  }
