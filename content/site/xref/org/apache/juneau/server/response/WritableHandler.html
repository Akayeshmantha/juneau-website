13  package org.apache.juneau.server.response;
14  
15  import java.io.*;
16  import java.util.*;
17  
18  import org.apache.juneau.*;
19  import org.apache.juneau.server.*;
20  
21  /**
22   * Response handler for {@link Writable} and {@link ReaderResource} objects.
23   * <p>
24   * Uses the {@link Writable#writeTo(Writer)} method to send the contents to the {@link RestResponse#getNegotiatedWriter()} writer.
25   * <p>
26   * This handler is registered by default on {@link RestServlet RestServlets} via the
27   * 	default implementation of the {@link RestServlet#createResponseHandlers} method.
28   *
29   * @author James Bognar (james.bognar@salesforce.com)
30   */
31  public final class WritableHandler implements ResponseHandler {
32  
33  	@Override /* ResponseHandler */
34  	public boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException {
35  		if (output instanceof Writable) {
36  			if (output instanceof ReaderResource) {
37  				ReaderResource r = (ReaderResource)output;
38  				String mediaType = r.getMediaType();
39  				if (mediaType != null)
40  					res.setContentType(mediaType);
41  				for (Map.Entry<String,String> h : r.getHeaders().entrySet())
42  					res.setHeader(h.getKey(), h.getValue());
43  			}
44  			Writer w = res.getNegotiatedWriter();
45  			((Writable)output).writeTo(w);
46  			w.flush();
47  			w.close();
48  			return true;
49  		}
50  		return false;
51  	}
52  }
