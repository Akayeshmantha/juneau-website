13  package org.apache.juneau.server.response;
14  
15  import java.io.*;
16  
17  import org.apache.juneau.server.*;
18  import org.apache.juneau.utils.*;
19  
20  /**
21   * Response handler for {@link Reader} objects.
22   * <p>
23   * Simply pipes the contents of the {@link Reader} to {@link RestResponse#getNegotiatedWriter()}.
24   * <p>
25   * This handler is registered by default on {@link RestServlet RestServlets} via the
26   * 	default implementation of the {@link RestServlet#createResponseHandlers} method.
27   *
28   * @author James Bognar (james.bognar@salesforce.com)
29   */
30  public final class ReaderHandler implements ResponseHandler {
31  
32  	@Override /* ResponseHandler */
33  	public boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException {
34  		if (output instanceof Reader) {
35  			Writer w = res.getNegotiatedWriter();
36  			IOPipe.create(output, w).closeOut().run();
37  			return true;
38  		}
39  		return false;
40  	}
41  }
