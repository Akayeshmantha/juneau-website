13  package org.apache.juneau.server.samples;
14  
15  import static org.apache.juneau.html.HtmlDocSerializerContext.*;
16  
17  import java.lang.Object;
18  
19  import org.apache.juneau.*;
20  import org.apache.juneau.client.*;
21  import org.apache.juneau.dto.Link;
22  import org.apache.juneau.html.dto.proto.*;
23  import org.apache.juneau.json.*;
24  import org.apache.juneau.microservice.*;
25  import org.apache.juneau.server.annotation.*;
26  
27  @RestResource(
28  	path="/tumblrParser",
29  	messages="nls/TumblrParserResource",
30  	properties={
31  		@Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.TumblrParserResource)'}"),
32  		@Property(name=HTMLDOC_title, value="Tumblr parser service"),
33  		@Property(name=HTMLDOC_description, value="Specify a URL to a Tumblr blog and parse the results.")
34  	}
35  )
36  public class TumblrParserResource extends Resource {
37  	private static final long serialVersionUID = 1L;
38  
39  	@RestMethod(name="GET", path="/")
40  	public String getInstructions() throws Exception {
41  		return "Append the Tumblr blog name to the URL above (e.g. /juneau/sample/tumblrParser/mytumblrblog)";
42  	}
43  
44  	@RestMethod(name="GET", path="/{blogName}")
45  	public ObjectList parseBlog(@Attr String blogName) throws Exception {
46  		ObjectList l = new ObjectList();
47  		RestClient rc = new RestClient(JsonSerializer.class, JsonParser.class);
48  		try {
49  			String site = "http://" + blogName + ".tumblr.com/api/read/json";
50  			ObjectMap m = rc.doGet(site).getResponse(ObjectMap.class);
51  			int postsTotal = m.getInt("posts-total");
52  			for (int i = 0; i < postsTotal; i += 20) {
53  				m = rc.doGet(site + "?start=" + i + "&num=20&filter=text").getResponse(ObjectMap.class);
54  				ObjectList ol = m.getObjectList("posts");
55  				for (int j = 0; j < ol.size(); j++) {
56  					ObjectMap om = ol.getObjectMap(j);
57  					String type = om.getString("type");
58  					Entry e = new Entry();
59  					e.date = om.getString("date");
60  					if (type.equals("link"))
61  						e.entry = new Link(om.getString("link-text"), om.getString("link-url"));
62  					else if (type.equals("audio"))
63  						e.entry = new ObjectMap().append("type","audio").append("audio-caption", om.getString("audio-caption"));
64  					else if (type.equals("video"))
65  						e.entry = new ObjectMap().append("type","video").append("video-caption", om.getString("video-caption"));
66  					else if (type.equals("quote"))
67  						e.entry = new ObjectMap().append("type","quote").append("quote-source", om.getString("quote-source")).append("quote-text", om.getString("quote-text"));
68  					else if (type.equals("regular"))
69  						e.entry = om.getString("regular-body");
70  					else if (type.equals("photo"))
71  						e.entry = new Img(om.getString("photo-url-250"));
72  					else
73  						e.entry = new ObjectMap().append("type", type);
74  					l.add(e);
75  				}
76  			}
77  		} finally {
78  			rc.closeQuietly();
79  		}
80  		return l;
81  	}
82  
83  	public static class Entry {
84  		public String date;
85  		public Object entry;
86  	}
87  }
