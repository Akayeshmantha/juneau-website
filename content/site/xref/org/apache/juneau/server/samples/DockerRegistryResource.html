28  /**
29   * Sample resource that shows how to mirror query results from a Docker registry.
30   */
31  @RestResource(
32  	path="/docker",
33  	label="Sample Docker resource",
34  	properties={
35  		@Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.AtomFeedResource)'}")
36  	}
37  )
38  public class DockerRegistryResource extends Resource {
39  	private static final long serialVersionUID = 1L;
40  
41  	// Get registry URL from samples.cfg file.
42  	private String registryUrl = getConfig().getString("DockerRegistry/url");
43  
44  	RestClient rc;
45  
46  	@Override /* Servlet */
47  	public void init() throws ServletException {
48  		super.init();
49  		rc = new RestClient(JsonSerializer.DEFAULT, JsonParser.DEFAULT);
50  	}
51  
52  	@Override /* Servlet */
53  	public void destroy() {
54  		rc.closeQuietly();
55  		super.destroy();
56  	}
57  
58  	/** [GET /] - Show child resources. */
59  	@SuppressWarnings("nls")
60  	@RestMethod(name="GET", path="/")
61  	public ResourceDescription[] getChildren(RestRequest req) {
62  		return new ResourceDescription[] {
63  			new ResourceDescription(req, "search", "Search Registry")
64  		};
65  	}
66  
67  	/**
68  	 * PUT request handler.
69  	 * Replaces the feed with the specified content, and then mirrors it as the response.
70  	 */
71  	@RestMethod(name="GET", path="/search")
72  	public QueryResults query(@Param("q") String q) throws Exception {
73  		String url = registryUrl + "/search" + (q == null ? "" : "?q=" + q);
74  		synchronized(rc) {
75  			return rc.doGet(url).getResponse(QueryResults.class);
76  		}
77  	}
78  
79  	public static class QueryResults {
80  		public int num_results;
81  		public String query;
82  		public List<DockerImage> results;
83  	}
84  
85  	public static class DockerImage {
86  		public String name, description;
87  	}
88  }
