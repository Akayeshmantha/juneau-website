13  package org.apache.juneau.server.samples;
14  
15  import static org.apache.juneau.html.HtmlDocSerializerContext.*;
16  import static org.apache.juneau.server.RestServletContext.*;
17  
18  import java.util.*;
19  
20  import org.apache.juneau.samples.addressbook.*;
21  import org.apache.juneau.server.annotation.*;
22  import org.apache.juneau.server.remoteable.*;
23  
24  /**
25   * Class showing the functionality of the RemoteableServlet class.
26   *
27   * @author James Bognar (james.bognar@salesforce.com)
28   */
29  @SuppressWarnings("serial")
30  @RestResource(
31  	path="/remoteable",
32  	messages="nls/SampleRemoteableServlet",
33  	properties={
34  		@Property(name=HTMLDOC_title, value="Remoteable Service Proxy API"),
35  		@Property(name=HTMLDOC_description, value="Sample class showing how to use remoteable proxies.  The list below are exposed services that can be retrieved using RestClient.getProxyInterface(Class)."),
36  		@Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.SampleRemoteableServlet)'}"),
37  		// Allow us to use method=POST from a browser.
38  		@Property(name=REST_allowMethodParam, value="*")
39  	},
40  	stylesheet="styles/devops.css"
41  )
42  public class SampleRemoteableServlet extends RemoteableServlet {
43  
44  	AddressBook addressBook = new AddressBook();
45  
46  	@Override /* RemoteableServlet */
47  	protected Map<Class<?>,Object> getServiceMap() throws Exception {
48  		Map<Class<?>,Object> m = new LinkedHashMap<Class<?>,Object>();
49  
50  		// In this simplified example, we expose the same POJO service under two different interfaces.
51  		// One is IAddressBook which only exposes methods defined on that interface, and
52  		// the other is AddressBook itself which exposes all methods defined on the class itself.
53  		m.put(IAddressBook.class, addressBook);
54  		m.put(AddressBook.class, addressBook);
55  		return m;
56  	}
57  }
