13  package org.apache.juneau.server.test;
14  
15  import static java.lang.String.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.annotation.*;
19  import org.apache.juneau.serializer.*;
20  import org.apache.juneau.server.*;
21  import org.apache.juneau.server.annotation.*;
22  
23  /**
24   * JUnit automated testcase resource.
25   */
26  @RestResource(
27  	path="/testProperties",
28  	properties={
29  		@Property(name="A1",value="a1"),
30  		@Property(name="A2",value="a2"),
31  		@Property(name="foo",value="bar"),
32  		@Property(name="bar",value="baz"),
33  		@Property(name="R1a",value="$R{requestURI}"),
34  		@Property(name="R1b",value="$R{requestParentURI}"),
35  		@Property(name="R2",value="$R{foo}"),
36  		@Property(name="R3",value="$R{$R{foo}}"),
37  		@Property(name="R4",value="$R{A1}"),
38  		@Property(name="R5",value="$R{A2}"),
39  		@Property(name="R6",value="$R{C}"),
40  	}
41  )
42  public class PropertiesResource extends RestServletDefault {
43  	private static final long serialVersionUID = 1L;
44  
45  	//====================================================================================================
46  	// Properties defined on method.
47  	//====================================================================================================
48  	@RestMethod(name="GET", path="/testPropertiesDefinedOnMethod",
49  		properties={
50  			@Property(name="B1",value="b1"),
51  			@Property(name="B2",value="b2")
52  		},
53  		serializers=PropertySerializer1.class
54  	)
55  	public void testPropertiesDefinedOnMethod(RestResponse res) {
56  		res.setProperty("A2", "c");
57  		res.setProperty("B2", "c");
58  		res.setProperty("C", "c");
59  		res.setOutput(null);
60  	}
61  
62  	@Produces({"application/json","text/json"})
63  	public static class PropertySerializer1 extends WriterSerializer {
64  		@Override /* Serializer */
65  		protected void doSerialize(SerializerSession session, Object output) throws Exception {
66  			ObjectMap p = session.getProperties();
67  			session.getWriter().write(format("A1=%s,A2=%s,B1=%s,B2=%s,C=%s,R1a=%s,R1b=%s,R2=%s,R3=%s,R4=%s,R5=%s,R6=%s",
68  				p.get("A1"), p.get("A2"), p.get("B1"), p.get("B2"), p.get("C"),
69  				p.get("R1a"), p.get("R1b"), p.get("R2"), p.get("R3"), p.get("R4"), p.get("R5"), p.get("R6")));
70  		}
71  	}
72  
73  	//====================================================================================================
74  	// Make sure attributes/parameters/headers are available through ctx.getProperties().
75  	//====================================================================================================
76  	@RestMethod(name="GET", path="/testProperties/{A}", serializers=PropertySerializer2.class)
77  	public void testProperties(RestResponse res) {
78  		res.setOutput(null);
79  	}
80  
81  	@Produces({"application/json","text/json"})
82  	public static class PropertySerializer2 extends WriterSerializer {
83  		@Override /* Serializer */
84  		protected void doSerialize(SerializerSession session, Object output) throws Exception {
85  			ObjectMap p = session.getProperties();
86  			session.getWriter().write(format("A=%s,P=%s,H=%s", p.get("A"), p.get("P"), p.get("h")));
87  		}
88  	}
89  
90  }
