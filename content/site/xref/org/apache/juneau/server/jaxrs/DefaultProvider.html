18  import org.apache.juneau.html.*;

20  import org.apache.juneau.json.*;

26  /**
27   * JAX-RS provider for the same serialize/parse support provided by the {@link RestServletDefault} class.
28   *
29   * @author James Bognar (james.bognar@salesforce.com)
30   */
31  @Provider
32  @Produces({
33  	"application/json", "text/json",                 // JsonSerializer
34  	"application/json+simple", "text/json+simple",   // JsonSerializer.Simple
35  	"application/json+schema",                       // JsonSchemaSerializer
36  	"text/xml",                                      // XmlDocSerializer
37  	"text/xml+simple",                               // XmlDocSerializer.Simple
38  	"text/xml+schema",                               // XmlSchemaDocSerializer
39  	"text/html",                                     // HtmlDocSerializer
40  	"application/x-www-form-urlencoded",             // UrlEncodingSerializer
41  	"text/xml+soap",                                 // SoapXmlSerializer
42  	"application/x-java-serialized-object"           // JavaSerializedObjectSerializer
43  })
44  @Consumes({
45  	"application/json", "text/json",                 // JsonParser
46  	"text/xml",                                      // XmlParser
47  	"text/html",                                     // HtmlParser
48  	"application/x-www-form-urlencoded",             // UrlEncodingParser
49  	"application/x-java-serialized-object"           // JavaSerializedObjectParser
50  })
51  @JuneauProvider(
52  	serializers={
53  		JsonSerializer.class,
54  		JsonSerializer.Simple.class,
55  		JsonSchemaSerializer.class,
56  		XmlDocSerializer.class,
57  		XmlDocSerializer.Simple.class,
58  		XmlSchemaDocSerializer.class,
59  		HtmlDocSerializer.class,
60  		UrlEncodingSerializer.class,
61  		SoapXmlSerializer.class,
62  		JavaSerializedObjectSerializer.class
63  	},
64  	parsers={
65  		JsonParser.class,
66  		XmlParser.class,
67  		HtmlParser.class,
68  		UrlEncodingParser.class,
69  		JavaSerializedObjectParser.class
70  	}
71  )
72  public final class DefaultProvider extends BaseProvider {}
