28  /**
29   * JAX-RS provider for the same serialize/parse support provided by the {@link RestServletJenaDefault} class.
30   *
31   * @author James Bognar (james.bognar@salesforce.com)
32   */
33  @Provider
34  @Produces({
35  	"application/json", "text/json",                 // JsonSerializer
36  	"application/json+simple","text/json+simple",    // JsonSerializer.Simple
37  	"application/json+schema","text/json+schema",    // JsonSchemaSerializer
38  	"text/xml",                                      // XmlDocSerializer
39  	"text/xml+simple",                               // XmlDocSerializer.Simple
40  	"text/xml+schema",                               // XmlSchemaDocSerializer
41  	"text/html",                                     // HtmlDocSerializer
42  	"application/x-www-form-urlencoded",             // UrlEncodingSerializer
43  	"text/xml+soap",                                 // SoapXmlSerializer
44  	"text/xml+rdf",                                  // RdfSerializer.Xml
45  	"text/xml+rdf+abbrev",                           // RdfSerializer.XmlAbbrev
46  	"text/n-triple",                                 // RdfSerializer.NTriple
47  	"text/turtle",                                   // RdfSerializer.Turtle
48  	"text/n3",                                       // RdfSerializer.N3
49  	"application/x-java-serialized-object"           // JavaSerializedObjectSerializer
50  })
51  @Consumes({
52  	"application/json", "text/json",                 // JsonParser
53  	"text/xml",                                      // XmlParser
54  	"text/html",                                     // HtmlParser
55  	"application/x-www-form-urlencoded",             // UrlEncodingParser
56  	"text/xml+rdf",                                  // RdfParser.Xml
57  	"text/n-triple",                                 // RdfParser.NTriple
58  	"text/turtle",                                   // RdfParser.Turtle
59  	"text/n3",                                       // RdfParser.N3
60  	"application/x-java-serialized-object"           // JavaSerializedObjectParser
61  })
62  @JuneauProvider(
63  	serializers={
64  		JsonSerializer.class,
65  		JsonSerializer.Simple.class,
66  		JsonSchemaSerializer.class,
67  		XmlDocSerializer.class,
68  		XmlDocSerializer.Simple.class,
69  		XmlSchemaDocSerializer.class,
70  		HtmlDocSerializer.class,
71  		UrlEncodingSerializer.class,
72  		SoapXmlSerializer.class,
73  		RdfSerializer.Xml.class,
74  		RdfSerializer.XmlAbbrev.class,
75  		RdfSerializer.NTriple.class,
76  		RdfSerializer.Turtle.class,
77  		RdfSerializer.N3.class,
78  		JavaSerializedObjectSerializer.class
79  	},
80  	parsers={
81  		JsonParser.class,
82  		XmlParser.class,
83  		HtmlParser.class,
84  		UrlEncodingParser.class,
85  		RdfParser.Xml.class,
86  		RdfParser.NTriple.class,
87  		RdfParser.Turtle.class,
88  		RdfParser.N3.class,
89  		JavaSerializedObjectParser.class,
90  	}
91  )
92  public final class DefaultJenaProvider extends BaseProvider {}
