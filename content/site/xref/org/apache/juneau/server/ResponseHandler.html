13  package org.apache.juneau.server;
14  
15  import java.io.*;
16  
17  import javax.servlet.http.*;
18  
19  import org.apache.juneau.*;
20  import org.apache.juneau.server.annotation.*;
21  import org.apache.juneau.server.response.*;
22  
23  /**
24   * Defines the interface for handlers that convert POJOs to appropriate HTTP responses.
25   * <p>
26   * The {@link RestServlet} API uses the concept of registered response handlers for
27   * 	converting objects returned by REST methods or set through {@link RestResponse#setOutput(Object)}
28   * 	into appropriate HTTP responses.
29   * <p>
30   * Response handlers can be associated with {@link RestServlet RestServlets} through the following ways:
31   * <ul class='spaced-list'>
32   * 	<li>Through the {@link RestResource#responseHandlers @RestResource.responseHandlers} annotation.
33   * 	<li>By overriding {@link RestServlet#createResponseHandlers(ObjectMap)} and augmenting or creating your
34   * 		own list of handlers.
35   * </ul>
36   * <p>
37   * By default, {@link RestServlet RestServlets} are registered with the following response handlers:
38   * <ul class='spaced-list'>
39   * 	<li>{@link DefaultHandler} - Serializes POJOs using the Juneau serializer API.
40   * 	<li>{@link ReaderHandler} - Pipes the output of {@link Reader Readers} to the response writer ({@link RestResponse#getWriter()}).
41   * 	<li>{@link InputStreamHandler} - Pipes the output of {@link InputStream InputStreams} to the response output stream ({@link RestResponse#getOutputStream()}).
42   * 	<li>{@link RedirectHandler} - Handles {@link Redirect} objects.
43   * </ul>
44   * <p>
45   * Response handlers can be used to process POJOs that cannot normally be handled through Juneau serializers, or
46   * 	because it's simply easier to define response handlers for special cases.
47   * <p>
48   * The following example shows how to create a response handler to handle special <code>Foo</code> objects outside the normal
49   * 	Juneau architecture.
50   * <p class='bcode'>
51   * 	<ja>@RestResource</ja>(
52   * 		path=<js>"/example"</js>,
53   * 		responseHandlers=FooHandler.<jk>class</jk>
54   * 	)
55   * 	<jk>public class</jk> Example <jk>extends</jk> RestServlet {
56   *
57   * 		<ja>@RestMethod</ja>(name=<js>"GET"</js>, path=<js>"/"</js>)
58   * 		<jk>public</jk> Foo test1() {
59   * 			<jk>return new</jk> Foo(<js>"123"</js>);
60   * 		}
61   *
62   * 		<jk>public static class</jk> FooHandler <jk>implements</jk> ResponseHandler {
63   * 			<ja>@Override</ja>
64   * 			<jk>public boolean</jk> handle(RestRequest req, RestResponse res, Object output) <jk>throws</jk> IOException, RestException {
65   * 				<jk>if</jk> (output <jk>instanceof</jk> Foo) {
66   * 					Foo foo = (Foo)output;
67   * 					<jc>// Set some headers and body content.</jc>
68   * 					res.setHeader(<js>"Foo-ID"</js>, foo.getId());
69   * 					res.getWriter().write(<js>"foo.id="</js> + foo.getId());
70   * 					<jk>return true</jk>;  <jc>// We handled it.</jc>
71   * 				}
72   * 				<jk>return false</jk>;  <jc>// We didn't handle it.</jc>
73   * 			}
74   * 		}
75   * 	}
76   * </p>
77   */
78  public interface ResponseHandler {
79  
80  	/**
81  	 * Process this response if possible.
82  	 * This method should return <jk>false</jk> if it wasn't able to process the response.
83  	 *
84  	 * @param req The HTTP servlet request.
85  	 * @param res The HTTP servlet response;
86  	 * @param output The POJO returned by the REST method that now needs to be sent to the response.
87  	 * @return true If this handler handled the response.
88  	 * @throws IOException - If low-level exception occurred on output stream.  Results in a {@link HttpServletResponse#SC_INTERNAL_SERVER_ERROR} error.
89  	 * @throws RestException - If some other exception occurred.  Can be used to provide an appropriate HTTP response code and message.
90  	 */
91  	boolean handle(RestRequest req, RestResponse res, Object output) throws IOException, RestException;
92  }
