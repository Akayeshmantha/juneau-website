13  package org.apache.juneau.server.annotation;
14  
15  import static java.lang.annotation.ElementType.*;
16  import static java.lang.annotation.RetentionPolicy.*;
17  
18  import java.lang.annotation.*;
19  
20  import org.apache.juneau.server.*;
21  
22  /**
23   * Identical to {@link Param @Param}, but only retrieves the parameter from the
24   * 	URL string, not URL-encoded form posts.
25   * <p>
26   * Unlike {@link Param @Param}, using this annotation does not result in the servlet reading the contents
27   * 	of URL-encoded form posts.
28   * Therefore, this annotation can be used in conjunction with the {@link Content @Content} annotation
29   * 	or {@link RestRequest#getInput(Class)} method for <code>application/x-www-form-urlencoded POST</code> calls.
30   *
31   * <h6 class='topic'>Example</h6>
32   * <p class='bcode'>
33   * 	<ja>@RestMethod</ja>(name=<js>"GET"</js>)
34   * 	<jk>public void</jk> doGet(RestRequest req, RestResponse res,
35   * 				<ja>@QParam</ja>(<js>"p1"</js>) <jk>int</jk> p1, <ja>@QParam</ja>(<js>"p2"</js>) String p2, <ja>@QParam</ja>(<js>"p3"</js>) UUID p3) {
36   * 		...
37   * 	}
38   * </p>
39   * <p>
40   * 	This is functionally equivalent to the following code...
41   * </p>
42   * <p class='bcode'>
43   * 	<ja>@RestMethod</ja>(name=<js>"GET"</js>)
44   * 	<jk>public void</jk> doGet(RestRequest req, RestResponse res) {
45   * 		<jk>int</jk> p1 = req.getQueryParameter(<jk>int</jk>.<jk>class</jk>, <js>"p1"</js>, 0);
46   * 		String p2 = req.getQueryParameter(String.<jk>class</jk>, <js>"p2"</js>);
47   * 		UUID p3 = req.getQueryParameter(UUID.<jk>class</jk>, <js>"p3"</js>);
48   * 		...
49   * 	}
50   * </p>
51   *
52   * @author James Bognar (james.bognar@salesforce.com)
53   */
54  @Documented
55  @Target(PARAMETER)
56  @Retention(RUNTIME)
57  @Inherited
58  public @interface QParam {
59  
60  	/**
61  	 * URL parameter name.
62  	 */
63  	String value();
64  
65  	/**
66  	 * Specify <jk>true</jk> if using multi-part parameters to represent collections and arrays.
67  	 * <p>
68  	 * 	Normally, we expect single parameters to be specified in UON notation for representing
69  	 * 	collections of values (e.g. <js>"&key=(1,2,3)"</js>.
70  	 * 	This annotation allows the use of multi-part parameters to represent collections
71  	 * 	(e.g. <js>"&key=1&key=2&key=3"</js>.
72  	 * <p>
73  	 *		This setting should only be applied to Java parameters of type array or Collection.
74  	 */
75  	boolean multipart() default false;
76  
77  	/**
78  	 * The expected format of the request parameter.
79  	 * <p>
80  	 * Possible values:
81  	 * <ul class='spaced-list'>
82  	 * 	<li><js>"UON"</js> - URL-Encoded Object Notation.<br>
83  	 *			This notation allows for request parameters to contain arbitrarily complex POJOs.
84  	 * 	<li><js>"PLAIN"</js> - Plain text.<br>
85  	 *			This treats request parameters as plain text.<br>
86  	 *			Only POJOs directly convertable from <l>Strings</l> can be represented in parameters when using this mode.
87  	 * 	<li><js>"INHERIT"</js> (default) - Inherit from the {@link RestServletContext#REST_paramFormat} property on the servlet method or class.
88  	 * </ul>
89  	 * <p>
90  	 * Note that the parameter value <js>"(foo)"</js> is interpreted as <js>"(foo)"</js> when using plain mode, but
91  	 * 	<js>"foo"</js> when using UON mode.
92  	 */
93  	String format() default "INHERIT";
94  }
