13  package org.apache.juneau.html;
14  
15  import static java.lang.annotation.ElementType.*;
16  import static java.lang.annotation.RetentionPolicy.*;
17  
18  import java.lang.annotation.*;
19  
20  /**
21   * Used in conjunction with the {@link HtmlSerializer} class to define hyperlinks.
22   * <p>
23   * 	This annotation is applied to classes.
24   * <p>
25   * 	Annotation that can be used to specify that a class has a URL associated with it.
26   * <p>
27   * 	When rendered using the {@link org.apache.juneau.html.HtmlSerializer HtmlSerializer} class, this class will get rendered as a hyperlink like so...
28   * <p class='code'>
29   * 	<xt>&lt;a</xt> <xa>href</xa>=<xs>'hrefProperty'</xs><xt>&gt;</xt>nameProperty<xt>&lt;/a&gt;</xt>
30   * </p>
31   *
32   * @author James Bognar (james.bognar@salesforce.com)
33   */
34  @Documented
35  @Target(TYPE)
36  @Retention(RUNTIME)
37  @Inherited
38  public @interface HtmlLink {
39  
40  	/**
41  	 * The bean property whose value becomes the name in the hyperlink.
42  	 */
43  	String nameProperty() default "";
44  
45  	/**
46  	 * The bean property whose value becomes the url in the hyperlink.
47  	 */
48  	String hrefProperty() default "";
49  }
