13  package org.apache.juneau.html;
14  
15  import java.lang.reflect.*;
16  import java.util.*;
17  
18  import org.apache.juneau.annotation.*;
19  import org.apache.juneau.serializer.*;
20  
21  /**
22   * Serializes POJOs to HTTP responses as stripped HTML.
23   *
24   *
25   * <h6 class='topic'>Media types</h6>
26   * <p>
27   * 	Handles <code>Accept</code> types: <code>text/html+stripped</code>
28   * <p>
29   * 	Produces <code>Content-Type</code> types: <code>text/html</code>
30   *
31   *
32   * <h6 class='topic'>Description</h6>
33   * <p>
34   * 	Produces the same output as {@link HtmlDocSerializer}, but without the header and body tags and page title and description.
35   * 	Used primarily for JUnit testing the {@link HtmlDocSerializer} class.
36   *
37   *
38   * @author James Bognar (james.bognar@salesforce.com)
39   */
40  @Produces(value="text/html+stripped",contentType="text/html")
41  public class HtmlStrippedDocSerializer extends HtmlSerializer {
42  
43  	//---------------------------------------------------------------------------
44  	// Overridden methods
45  	//---------------------------------------------------------------------------
46  
47  	@Override /* Serializer */
48  	protected void doSerialize(SerializerSession session, Object o) throws Exception {
49  		HtmlSerializerSession s = (HtmlSerializerSession)session;
50  		HtmlWriter w = s.getWriter();
51  		if (o == null
52  			|| (o instanceof Collection && ((Collection<?>)o).size() == 0)
53  			|| (o.getClass().isArray() && Array.getLength(o) == 0))
54  			w.sTag(1, "p").append("No Results").eTag("p").nl();
55  		else
56  			super.doSerialize(s, o);
57  	}
58  }
