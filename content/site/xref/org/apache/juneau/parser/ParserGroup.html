23  /**
24   * Represents a group of {@link Parser Parsers} that can be looked up by media type.
25   *
26   *
27   * <h6 class='topic'>Description</h6>
28   * <p>
29   * 	Provides the following features:
30   * <ul class='spaced-list'>
31   * 	<li>Finds parsers based on HTTP <code>Content-Type</code> header values.
32   * 	<li>Sets common properties on all parsers in a single method call.
33   * 	<li>Locks all parsers in a single method call.
34   * 	<li>Clones existing groups and all parsers within the group in a single method call.
35   * </ul>
36   *
37   *
38   * <h6 class='topic'>Match ordering</h6>
39   * <p>
40   * 	Parsers are matched against <code>Content-Type</code> strings in the order they exist in this group.
41   * <p>
42   * 	Adding new entries will cause the entries to be prepended to the group.
43   *  	This allows for previous parsers to be overridden through subsequent calls.
44   * <p>
45   * 	For example, calling <code>g.append(P1.<jk>class</jk>,P2.<jk>class</jk>).append(P3.<jk>class</jk>,P4.<jk>class</jk>)</code>
46   * 	will result in the order <code>P3, P4, P1, P2</code>.
47   *
48   *
49   * <h6 class='topic'>Examples</h6>
50   * <p class='bcode'>
51   * 	<jc>// Construct a new parser group</jc>
52   * 	ParserGroup g = <jk>new</jk> ParserGroup();
53   *
54   * 	<jc>// Add some parsers to it</jc>
55   * 	g.append(JsonParser.<jk>class</jk>, XmlParser.<jk>class</jk>);
56   *
57   * 	<jc>// Change settings on parsers simultaneously</jc>
58   * 	g.setProperty(BeanContext.<jsf>BEAN_beansRequireSerializable</jsf>, <jk>true</jk>)
59   * 		.addPojoSwaps(CalendarSwap.ISO8601DT.<jk>class</jk>)
60   * 		.lock();
61   *
62   * 	<jc>// Find the appropriate parser by Content-Type</jc>
63   * 	ReaderParser p = (ReaderParser)g.getParser(<js>"text/json"</js>);
64   *
65   * 	<jc>// Parse a bean from JSON</jc>
66   * 	String json = <js>"{...}"</js>;
67   * 	AddressBook addressBook = p.parse(json, AddressBook.<jk>class</jk>);
68   * </p>
69   *
70   * @author James Bognar (james.bognar@salesforce.com)
71   */
72  public final class ParserGroup extends Lockable {
73  
74  	// Maps media-types to parsers.
75  	private final Map<String,Parser> parserMap = new ConcurrentHashMap<String,Parser>();
76  
77  	// Maps Content-Type headers to matching media types.
78  	private final Map<String,String> mediaTypeMappings = new ConcurrentHashMap<String,String>();
79  
80  	private final CopyOnWriteArrayList<Parser> parsers = new CopyOnWriteArrayList<Parser>();
81  
82  	private final ReadWriteLock lock = new ReentrantReadWriteLock();
83  	private final Lock rl = lock.readLock(), wl = lock.writeLock();
84  
85  
86  	/**
87  	 * Registers the specified parsers with this group.
88  	 *
89  	 * @param p The parsers to append to this group.
90  	 * @return This object (for method chaining).
91  	 * @throws Exception Thrown if {@link Parser} could not be constructed.
92  	 */
93  	public ParserGroup append(Class<? extends Parser>...p) throws Exception {
94  		checkLock();
95  		wl.lock();
96  		try {
97  			for (Class<? extends Parser> c : reverse(p)) {
98  				parserMap.clear();
99  				mediaTypeMappings.clear();
100 				try {
101 					append(c);
102 				} catch (NoClassDefFoundError e) {
103 					// Ignore if dependent library not found (e.g. Jena).
104 					System.err.println(e);
105 				}
106 			}
107 		} finally {
108 			wl.unlock();
109 		}
110 		return this;
111 	}
112 
113 	/**
114 	 * Same as {@link #append(Class[])}, except specify a single class to avoid unchecked compile warnings.
115 	 *
116 	 * @param p The parser to append to this group.
117 	 * @return This object (for method chaining).
118 	 * @throws Exception Thrown if {@link Parser} could not be constructed.
119 	 */
120 	public ParserGroup append(Class<? extends Parser> p) throws Exception {
121 		checkLock();
122 		wl.lock();
123 		try {
124 			parserMap.clear();
125 			mediaTypeMappings.clear();
126 			parsers.add(0, p.newInstance());
127 		} catch (NoClassDefFoundError e) {
128 			// Ignore if dependent library not found (e.g. Jena).
129 			System.err.println(e);
130 		} finally {
131 			wl.unlock();
132 		}
133 		return this;
134 	}

136 	/**
137 	 * Returns the parser registered to handle the specified media type.
138 	 * <p>
139 	 * The media-type string must not contain any parameters such as <js>";charset=X"</js>.
140 	 *
141 	 * @param mediaType The media-type string (e.g. <js>"text/json"</js>).
142 	 * @return The REST parser that handles the specified request content type, or <jk>null</jk> if
143 	 * 		no parser is registered to handle it.
144 	 */
145 	public Parser getParser(String mediaType) {
146 		Parser p = parserMap.get(mediaType);
147 		if (p == null) {
148 			String mt = findMatch(mediaType);
149 			if (mt != null)
150 				p = parserMap.get(mt);
151 		}
152 		return p;
153 	}
154 
155 	/**
156 	 * Searches the group for a parser that can handle the specified <l>Content-Type</l> header value.
157 	 *
158 	 * @param contentTypeHeader The HTTP <l>Content-Type</l> header value.
159 	 * @return The media type registered by one of the parsers that matches the <code>mediaType</code> string,
160 	 * 	or <jk>null</jk> if no media types matched.
161 	 */
162 	public String findMatch(String contentTypeHeader) {
163 		rl.lock();
164 		try {
165 			String mt = mediaTypeMappings.get(contentTypeHeader);
166 			if (mt != null)
167 				return mt;
168 
169 			MediaRange[] mr = MediaRange.parse(contentTypeHeader);
170 			if (mr.length == 0)
171 				mr = MediaRange.parse("*/*");
172 
173 			for (MediaRange a : mr) {
174 				for (Parser p : parsers) {
175 					for (MediaRange a2 : p.getMediaRanges()) {
176 						if (a.matches(a2)) {
177 							mt = a2.getMediaType();
178 							mediaTypeMappings.put(contentTypeHeader, mt);
179 							parserMap.put(mt, p);
180 							return mt;
181 						}
182 					}
183 				}
184 			}
185 			return null;
186 		} finally {
187 			rl.unlock();
188 		}
189 	}
190 
191 	/**
192 	 * Returns the media types that all parsers in this group can handle
193 	 * <p>
194 	 * Entries are ordered in the same order as the parsers in the group.
195 	 *
196 	 * @return The list of media types.
197 	 */
198 	public List<String> getSupportedMediaTypes() {
199 		List<String> l = new ArrayList<String>();
200 		for (Parser p : parsers)
201 			for (String mt : p.getMediaTypes())
202 				if (! l.contains(mt))
203 					l.add(mt);
204 		return l;
205 	}

316 	/**
317 	 * Locks this group and all parsers in this group.
318 	 */
319 	@Override /* Lockable */
320 	public ParserGroup lock() {
321 		super.lock();
322 		for (Parser p : parsers)
323 			p.lock();
324 		return this;
325 	}
326 
327 	/**
328 	 * Clones this group and all parsers in this group.
329 	 */
330 	@Override /* Lockable */
331 	public ParserGroup clone() throws CloneNotSupportedException {
332 		ParserGroup g = new ParserGroup();
333 
334 		List<Parser> l = new ArrayList<Parser>(parsers.size());
335 		for (Parser p : parsers)
336 			l.add(p.clone());
337 
338 		g.parsers.addAll(l);
339 
340 		return g;
341 	}
