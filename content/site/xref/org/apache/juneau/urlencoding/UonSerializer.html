26  /**
27   * Serializes POJO models to UON (a notation for URL-encoded query parameter values).
28   *
29   *
30   * <h6 class='topic'>Media types</h6>
31   * <p>
32   * 	Handles <code>Accept</code> types: <code>text/uon</code>
33   * <p>
34   * 	Produces <code>Content-Type</code> types: <code>text/uon</code>
35   *
36   *
37   * <h6 class='topic'>Description</h6>
38   * <p>
39   * 	This serializer provides several serialization options.  Typically, one of the predefined DEFAULT serializers will be sufficient.
40   * 	However, custom serializers can be constructed to fine-tune behavior.
41   *
42   *
43   * <h6 class='topic'>Configurable properties</h6>
44   * <p>
45   * 	This class has the following properties associated with it:
46   * <ul>
47   * 	<li>{@link UonSerializerContext}
48   * 	<li>{@link BeanContext}
49   * </ul>
50   * <p>
51   * 	The following shows a sample object defined in Javascript:
52   * </p>
53   * <p class='bcode'>
54   * 	{
55   * 		id: 1,
56   * 		name: <js>'John Smith'</js>,
57   * 		uri: <js>'http://sample/addressBook/person/1'</js>,
58   * 		addressBookUri: <js>'http://sample/addressBook'</js>,
59   * 		birthDate: <js>'1946-08-12T00:00:00Z'</js>,
60   * 		otherIds: <jk>null</jk>,
61   * 		addresses: [
62   * 			{
63   * 				uri: <js>'http://sample/addressBook/address/1'</js>,
64   * 				personUri: <js>'http://sample/addressBook/person/1'</js>,
65   * 				id: 1,
66   * 				street: <js>'100 Main Street'</js>,
67   * 				city: <js>'Anywhereville'</js>,
68   * 				state: <js>'NY'</js>,
69   * 				zip: 12345,
70   * 				isCurrent: <jk>true</jk>,
71   * 			}
72   * 		]
73   * 	}
74   * </p>
75   * <p>
76   * 	Using the "strict" syntax defined in this document, the equivalent
77   * 		UON notation would be as follows:
78   * </p>
79   * <p class='bcode'>
80   * 	$o(
81   * 		<xa>id</xa>=$n(<xs>1</xs>),
82   * 		<xa>name</xa>=<xs>John+Smith</xs>,
83   * 		<xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
84   * 		<xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
85   * 		<xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
86   * 		<xa>otherIds</xa>=<xs>%00</xs>,
87   * 		<xa>addresses</xa>=$a(
88   * 			$o(
89   * 				<xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
90   * 				<xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
91   * 				<xa>id</xa>=$n(<xs>1</xs>),
92   * 				<xa>street</xa>=<xs>100+Main+Street</xs>,
93   * 				<xa>city</xa>=<xs>Anywhereville</xs>,
94   * 				<xa>state</xa>=<xs>NY</xs>,
95   * 				<xa>zip</xa>=$n(<xs>12345</xs>),
96   * 				<xa>isCurrent</xa>=$b(<xs>true</xs>)
97   * 			)
98   * 		)
99   * 	)
100  * </p>
101  * <p>
102  * 	A secondary "lax" syntax is available when the data type of the
103  * 		values are already known on the receiving end of the transmission:
104  * </p>
105  * <p class='bcode'>
106  * 	(
107  * 		<xa>id</xa>=<xs>1</xs>,
108  * 		<xa>name</xa>=<xs>John+Smith</xs>,
109  * 		<xa>uri</xa>=<xs>http://sample/addressBook/person/1</xs>,
110  * 		<xa>addressBookUri</xa>=<xs>http://sample/addressBook</xs>,
111  * 		<xa>birthDate</xa>=<xs>1946-08-12T00:00:00Z</xs>,
112  * 		<xa>otherIds</xa>=<xs>%00</xs>,
113  * 		<xa>addresses</xa>=(
114  * 			(
115  * 				<xa>uri</xa>=<xs>http://sample/addressBook/address/1</xs>,
116  * 				<xa>personUri</xa>=<xs>http://sample/addressBook/person/1</xs>,
117  * 				<xa>id</xa>=<xs>1</xs>,
118  * 				<xa>street</xa>=<xs>100+Main+Street</xs>,
119  * 				<xa>city</xa>=<xs>Anywhereville</xs>,
120  * 				<xa>state</xa>=<xs>NY</xs>,
121  * 				<xa>zip</xa>=<xs>12345</xs>,
122  * 				<xa>isCurrent</xa>=<xs>true</xs>
123  * 			)
124  * 		)
125  * 	)
126  * </p>
127  *
128  *
129  * <h6 class='topic'>Examples</h6>
130  * <p class='bcode'>
131  * 	<jc>// Serialize a Map</jc>
132  * 	Map m = <jk>new</jk> ObjectMap(<js>"{a:'b',c:1,d:false,e:['f',1,false],g:{h:'i'}}"</js>);
133  *
134  * 	<jc>// Serialize to value equivalent to JSON.</jc>
135  * 	<jc>// Produces "$o(a=b,c=$n(1),d=$b(false),e=$a(f,$n(1),$b(false)),g=$o(h=i))"</jc>
136  * 	String s = UonSerializer.<jsf>DEFAULT</jsf>.serialize(s);
137  *
138  * 	<jc>// Serialize to simplified value (for when data type is already known by receiver).</jc>
139  * 	<jc>// Produces "(a=b,c=1,d=false,e=(f,1,false),g=(h=i))"</jc>
140  * 	String s = UonSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
141  *
142  * 	<jc>// Serialize a bean</jc>
143  * 	<jk>public class</jk> Person {
144  * 		<jk>public</jk> Person(String s);
145  * 		<jk>public</jk> String getName();
146  * 		<jk>public int</jk> getAge();
147  * 		<jk>public</jk> Address getAddress();
148  * 		<jk>public boolean</jk> deceased;
149  * 	}
150  *
151  * 	<jk>public class</jk> Address {
152  * 		<jk>public</jk> String getStreet();
153  * 		<jk>public</jk> String getCity();
154  * 		<jk>public</jk> String getState();
155  * 		<jk>public int</jk> getZip();
156  * 	}
157  *
158  * 	Person p = <jk>new</jk> Person(<js>"John Doe"</js>, 23, <js>"123 Main St"</js>, <js>"Anywhere"</js>, <js>"NY"</js>, 12345, <jk>false</jk>);
159  *
160  * 	<jc>// Produces "$o(name=John Doe,age=23,address=$o(street=123 Main St,city=Anywhere,state=NY,zip=$n(12345)),deceased=$b(false))"</jc>
161  * 	String s = UonSerializer.<jsf>DEFAULT</jsf>.serialize(s);
162  *
163  * 	<jc>// Produces "(name=John Doe,age=23,address=(street=123 Main St,city=Anywhere,state=NY,zip=12345),deceased=false)"</jc>
164  * 	String s = UonSerializer.<jsf>DEFAULT_SIMPLE</jsf>.serialize(s);
165  * </p>
166  *
167  * @author James Bognar (james.bognar@salesforce.com)
168  */
169 @Produces("text/uon")
170 public class UonSerializer extends WriterSerializer {

232 	/**
233 	 * Workhorse method. Determines the type of object, and then calls the
234 	 * appropriate type-specific serialization method.
235 	 * @param session The context that exist for the duration of a serialize.
236 	 * @param out The writer to serialize to.
237 	 * @param o The object being serialized.
238 	 * @param eType The expected type of the object if this is a bean property.
239 	 * @param attrName The bean property name if this is a bean property.  <jk>null</jk> if this isn't a bean property being serialized.
240 	 * @param pMeta The bean property metadata.
241 	 * @param quoteEmptyStrings <jk>true</jk> if this is the first entry in an array.
242 	 * @param isTop If we haven't recursively called this method.
243 	 *
244 	 * @return The same writer passed in.
245 	 * @throws Exception
246 	 */
247 	@SuppressWarnings({ "rawtypes", "unchecked" })
248 	protected SerializerWriter serializeAnything(UonSerializerSession session, UonWriter out, Object o, ClassMeta<?> eType,
249 			String attrName, BeanPropertyMeta pMeta, boolean quoteEmptyStrings, boolean isTop) throws Exception {
250 		BeanContext bc = session.getBeanContext();
251 
252 		if (o == null) {
253 			out.appendObject(null, false, false, isTop);
254 			return out;
255 		}
256 
257 		if (eType == null)
258 			eType = object();
259 
260 		boolean addTypeProperty;		// Add "_type" attribute to element?
261 		ClassMeta<?> aType;			// The actual type
262 		ClassMeta<?> sType;			// The serialized type
263 
264 		aType = session.push(attrName, o, eType);
265 		boolean isRecursion = aType == null;
266 
267 		// Handle recursion
268 		if (aType == null) {
269 			o = null;
270 			aType = object();
271 		}
272 
273 		sType = aType.getSerializedClassMeta();
274 		addTypeProperty = (session.isAddBeanTypeProperties() && ! eType.equals(aType));
275 
276 		// Swap if necessary
277 		PojoSwap swap = aType.getPojoSwap();
278 		if (swap != null) {
279 			o = swap.swap(o, bc);
280 
281 			// If the getSwapClass() method returns Object, we need to figure out
282 			// the actual type now.
283 			if (sType.isObject())
284 				sType = bc.getClassMetaForObject(o);
285 		}
286 
287 		// '\0' characters are considered null.
288 		if (o == null || (sType.isChar() && ((Character)o).charValue() == 0))
289 			out.appendObject(null, false, false, isTop);
290 		else if (sType.hasToObjectMapMethod())
291 			serializeMap(session, out, sType.toObjectMap(o), eType);
292 		else if (sType.isBean())
293 			serializeBeanMap(session, out, bc.forBean(o), addTypeProperty);
294 		else if (sType.isUri() || (pMeta != null && pMeta.isUri()))
295 			out.appendUri(o, isTop);
296 		else if (sType.isMap()) {
297 			if (o instanceof BeanMap)
298 				serializeBeanMap(session, out, (BeanMap)o, addTypeProperty);
299 			else
300 				serializeMap(session, out, (Map)o, eType);
301 		}
302 		else if (sType.isCollection()) {
303 			serializeCollection(session, out, (Collection) o, eType);
304 		}
305 		else if (sType.isArray()) {
306 			serializeCollection(session, out, toList(sType.getInnerClass(), o), eType);
307 		}
308 		else {
309 			out.appendObject(o, quoteEmptyStrings, false, isTop);
310 		}
311 
312 		if (! isRecursion)
313 			session.pop();
314 		return out;
315 	}

317 	@SuppressWarnings({ "rawtypes", "unchecked" })
318 	private SerializerWriter serializeMap(UonSerializerSession session, UonWriter out, Map m, ClassMeta<?> type) throws Exception {
319 
320 		m = session.sort(m);
321 
322 		ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
323 
324 		int depth = session.getIndent();
325 		out.startFlag('o');
326 
327 		Iterator mapEntries = m.entrySet().iterator();
328 
329 		while (mapEntries.hasNext()) {
330 			Map.Entry e = (Map.Entry) mapEntries.next();
331 			Object value = e.getValue();
332 			Object key = session.generalize(e.getKey(), keyType);
333 			out.cr(depth).appendObject(key, session.isUseWhitespace(), false, false).append('=');
334 			serializeAnything(session, out, value, valueType, (key == null ? null : session.toString(key)), null, session.isUseWhitespace(), false);
335 			if (mapEntries.hasNext())
336 				out.append(',');
337 		}
338 
339 		if (m.size() > 0)
340 			out.cr(depth-1);
341 		out.append(')');
342 
343 		return out;
344 	}

346 	private SerializerWriter serializeBeanMap(UonSerializerSession session, UonWriter out, BeanMap<?> m, boolean addTypeProperty) throws Exception {
347 		int depth = session.getIndent();
348 
349 		out.startFlag('o');
350 
351 		boolean addComma = false;
352 
353 		for (BeanPropertyValue p : m.getValues(session.isTrimNulls(), addTypeProperty ? session.createBeanTypeNameProperty(m) : null)) {
354 			BeanPropertyMeta pMeta = p.getMeta();
355 			ClassMeta<?> cMeta = p.getClassMeta();
356 
357 			String key = p.getName();
358 			Object value = p.getValue();
359 			Throwable t = p.getThrown();
360 			if (t != null)
361 				session.addBeanGetterWarning(pMeta, t);
362 
363 			if (session.canIgnoreValue(cMeta, key, value))
364 				continue;
365 
366 			if (addComma)
367 				out.append(',');
368 
369 			out.cr(depth).appendObject(key, false, false, false).append('=');
370 
371 			serializeAnything(session, out, value, cMeta, key, pMeta, false, false);
372 
373 			addComma = true;
374 		}
375 
376 		if (m.size() > 0)
377 			out.cr(depth-1);
378 		out.append(')');
379 
380 		return out;
381 	}

383 	@SuppressWarnings({ "rawtypes", "unchecked" })
384 	private SerializerWriter serializeCollection(UonSerializerSession session, UonWriter out, Collection c, ClassMeta<?> type) throws Exception {
385 
386 		ClassMeta<?> elementType = type.getElementType();
387 
388 		c = session.sort(c);
389 
390 		out.startFlag('a');
391 
392 		int depth = session.getIndent();
393 		boolean quoteEmptyString = (c.size() == 1 || session.isUseWhitespace());
394 
395 		for (Iterator i = c.iterator(); i.hasNext();) {
396 			out.cr(depth);
397 			serializeAnything(session, out, i.next(), elementType, "<iterator>", null, quoteEmptyString, false);
398 			if (i.hasNext())
399 				out.append(',');
400 		}
401 
402 		if (c.size() > 0)
403 			out.cr(depth-1);
404 		out.append(')');
405 
406 		return out;
407 	}

418 	@Override /* Serializer */
419 	protected void doSerialize(SerializerSession session, Object o) throws Exception {
420 		UonSerializerSession s = (UonSerializerSession)session;
421 		serializeAnything(s, s.getWriter(), o, null, "root", null, false, true);
422 	}
