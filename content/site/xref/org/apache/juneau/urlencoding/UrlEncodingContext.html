13  package org.apache.juneau.urlencoding;
14  
15  /**
16   * Configurable properties on the {@link UrlEncodingSerializer} and {@link UrlEncodingParser} classes.
17   * <p>
18   * 	Use the {@link UrlEncodingSerializer#setProperty(String, Object)} and
19   * 	{@link UrlEncodingParser#setProperty(String, Object)} methods to set property values.
20   *
21   * @author James Bognar (james.bognar@salesforce.com)
22   */
23  public final class UrlEncodingContext implements Cloneable {
24  
25  	/**
26  	 * Serialize bean property collections/arrays as separate key/value pairs ({@link Boolean}, default=<jk>false</jk>).
27  	 * <p>
28  	 * 	If <jk>false</jk>, serializing the array <code>[1,2,3]</code> results in <code>?key=$a(1,2,3)</code>.
29  	 * 	If <jk>true</jk>, serializing the same array results in <code>?key=1&key=2&key=3</code>.
30  	 * <p>
31  	 * 	Example:
32  	 * <p class='bcode'>
33  	 * 	<jk>public class</jk> A {
34  	 * 		<jk>public</jk> String[] f1 = {<js>"a"</js>,<js>"b"</js>};
35  	 * 		<jk>public</jk> List&lt;String&gt; f2 = <jk>new</jk> LinkedList&lt;String&gt;(Arrays.<jsm>asList</jsm>(<jk>new</jk> String[]{<js>"c"</js>,<js>"d"</js>}));
36  	 * 	}
37  	 *
38  	 * 	UrlEncodingSerializer s1 = <jk>new</jk> UrlEncodingParser();
39  	 * 	UrlEncodingSerializer s2 = <jk>new</jk> UrlEncodingParser().setProperty(UrlEncodingContext.<jsf>URLENC_expandedParams</jsf>, <jk>true</jk>);
40  	 *
41  	 * 	String s1 = p1.serialize(<jk>new</jk> A()); <jc>// Produces "f1=(a,b)&f2=(c,d)"</jc>
42  	 * 	String s2 = p2.serialize(<jk>new</jk> A()); <jc>// Produces "f1=a&f1=b&f2=c&f2=d"</jc>
43  	 * </p>
44  	 * <p>
45  	 * 	<b>Important note:</b>  If parsing multi-part parameters, it's highly recommended to use Collections or Lists
46  	 * 	as bean property types instead of arrays since arrays have to be recreated from scratch every time a value
47  	 * 	is added to it.
48  	 * <p>
49  	 * 	This option only applies to beans.
50  	 */
51  	public static final String URLENC_expandedParams = "UrlEncoding.expandedParams";
52  
53  	boolean
54  		expandedParams = false;
55  
56  	//--------------------------------------------------------------------------------
57  	// Overridden methods
58  	//--------------------------------------------------------------------------------
59  
60  	@Override /* Cloneable */
61  	public UrlEncodingContext clone() {
62  		try {
63  			return (UrlEncodingContext)super.clone();
64  		} catch (CloneNotSupportedException e) {
65  			throw new RuntimeException(e); // Shouldn't happen
66  		}
67  	}
68  }
