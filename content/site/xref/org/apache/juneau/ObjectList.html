23  /**
24   * Java implementation of a JSON array.
25   * <p>
26   * 	An extension of {@link LinkedList}, so all methods available to in that class are also available
27   * 	to this class.
28   * <p>
29   * 	Note that the use of this class is optional.  The serializers will accept any objects that implement
30   * 	the {@link Collection} interface.  But this class provides some useful additional functionality
31   * 	when working with JSON models constructed from Java Collections Framework objects.  For example, a
32   * 	constructor is provided for converting a JSON array string directly into a {@link List}.  It also contains
33   * 	accessor methods for to avoid common typecasting when accessing elements in a list.
34   *
35   * <h6 class='topic'>Examples</h6>
36   * <p class='bcode'>
37   * 	<jc>// Construct an empty List</jc>
38   * 	List l = <jk>new</jk> ObjectList();
39   *
40   * 	<jc>// Construct a list of objects using various methods</jc>
41   * 	l = <jk>new</jk> ObjectList().append(<js>"foo"</js>).append(123).append(<jk>true</jk>);
42   * 	l = <jk>new</jk> ObjectList().append(<js>"foo"</js>, 123, <jk>true</jk>);  <jc>// Equivalent</jc>
43   * 	l = <jk>new</jk> ObjectList(<js>"foo"</js>, 123, <jk>true</jk>);  <jc>// Equivalent</jc>
44   *
45   * 	<jc>// Construct a list of integers from JSON</jc>
46   * 	l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
47   *
48   * 	<jc>// Construct a list of generic ObjectMap objects from JSON</jc>
49   * 	l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:'bing'}]"</js>);
50   *
51   * 	<jc>// Construct a list of integers from XML</jc>
52   * 	String xml = <js>"&lt;array&gt;&lt;number&gt;1&lt;/number&gt;&lt;number&gt;2&lt;/number&gt;&lt;number&gt;3&lt;/number&gt;&lt;/array&gt;"</js>;
53   * 	l = <jk>new</jk> ObjectList(xml, DataFormat.<jsf>XML</jsf>);
54   * 	l = (List)XmlParser.<jsf>DEFAULT</jsf>.parse(xml);  <jc>// Equivalent</jc>
55   * 	l = (List)XmlParser.<jsf>DEFAULT</jsf>.parse(Object.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
56   * 	l = XmlParser.<jsf>DEFAULT</jsf>.parse(List.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
57   * 	l = XmlParser.<jsf>DEFAULT</jsf>.parse(ObjectList.<jk>class</jk>, xml);  <jc>// Equivalent</jc>
58   *
59   * 	<jc>// Construct JSON from ObjectList</jc>
60   * 	l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:'bing'}]"</js>);
61   * 	String json = l.toString();  <jc>// Produces "[{foo:'bar'},{baz:'bing'}]"</jc>
62   * 	json = l.toString(JsonSerializer.<jsf>DEFAULT_CONDENSED</jsf>);  <jc>// Equivalent</jc>
63   * 	json = JsonSerializer.<jsf>DEFAULT_CONDENSED</jsf>.serialize(l);  <jc>// Equivalent</jc>
64   *
65   * 	<jc>// Get one of the entries in the list as an Integer</jc>
66   * 	l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
67   * 	Integer i = l.getInt(1);
68   * 	i = l.get(Integer.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
69   *
70   * 	<jc>// Get one of the entries in the list as an Float</jc>
71   * 	l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
72   * 	Float f = l.getFloat(1); <jc>// Returns 2f </jc>
73   * 	f = l.get(Float.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
74   *
75   * 	<jc>// Same as above, except converted to a String</jc>
76   * 	l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
77   * 	String s = l.getString(1); <jc>// Returns "2" </jc>
78   * 	s = l.get(String.<jk>class</jk>, 1);  <jc>// Equivalent</jc>
79   *
80   * 	<jc>// Get one of the entries in the list as a bean (converted to a bean if it isn't already one)</jc>
81   * 	l = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
82   * 	Person p = l.get(Person.<jk>class</jk>, 0);
83   *
84   * 	<jc>// Iterate over a list of beans using the elements() method</jc>
85   * 	ObjectList ObjectList = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
86   * 	<jk>for</jk> (Person p : ObjectList.elements(Person.<jk>class</jk>) {
87   * 		<jc>// Do something with p</jc>
88   * 	}
89   * </p>
90   * <p>
91   * 	This class is not thread safe.
92   * </p>
93   *
94   * @author James Bognar (james.bognar@salesforce.com)
95   */

134 	/**
135 	 * Construct a JSON array directly from text using the specified parser.
136 	 *
137 	 * @param s The string being parsed.
138 	 * @param p The parser to use to parse the input.
139 	 * @throws ParseException If the input contains a syntax error or is malformed.
140 	 */
141 	public ObjectList(CharSequence s, Parser p) throws ParseException {
142 		this(p == null ? BeanContext.DEFAULT : p.getBeanContext());
143 		if (p == null)
144 			p = JsonParser.DEFAULT;
145 		if (s != null)
146 			p.parseIntoCollection(s, this, beanContext.object());
147 	}

154 	 */

160 	 * Construct a JSON array directly from a reader using the specified parser.

163 	 * @param p The parser to use to parse the input.
164 	 * @throws ParseException If the input contains a syntax error or is malformed.
165 	 * @throws IOException If a problem occurred trying to read from the reader.

168 		parseReader(r, p);
169 	}

171 	private void parseReader(Reader r, Parser p) throws ParseException {
172 		if (p == null)
173 			p = JsonParser.DEFAULT;
174 		p.parseIntoCollection(r, this, beanContext.object());
175 	}

340 	 * Same as {@link #get(Class,int) get(Class,int)}, but the key is a slash-delimited
341 	 * 	path used to traverse entries in this POJO.
342 	 * <p>
343 	 * 	For example, the following code is equivalent:
344 	 * </p>
345 	 * <p class='bcode'>
346 	 * 	ObjectMap m = getObjectMap();
347 	 *
348 	 * 	<jc>// Long way</jc>
349 	 * 	<jk>long</jk> l = m.getObjectMap(<js>"foo"</js>).getObjectList(<js>"bar"</js>).getObjectMap(<js>"0"</js>).getLong(<js>"baz"</js>);
350 	 *
351 	 * 	<jc>// Using this method</jc>
352 	 * 	<jk>long</jk> l = m.getAt(<jk>long</jk>.<jk>class</jk>, <js>"foo/bar/0/baz"</js>);
353 	 * </p>
354 	 * <p>
355 	 * 	This method uses the {@link PojoRest} class to perform the lookup, so the map can contain
356 	 * 		any of the various class types that the {@link PojoRest} class supports (e.g. beans, collections, arrays).
357 	 * </p>
358 	 *
359 	 * @param <T> The class type.
360 	 * @param type The class type.
361 	 * @param path The path to the entry.
362 	 * @return The value, or <jk>null</jk> if the entry doesn't exist.
363 	 */
364 	public <T> T getAt(Class<T> type, String path) {
365 		return getPojoRest().get(type, path);
366 	}

451 	/**
452 	 * Creates an {@link Iterable} with elements of the specified child type.
453 	 * <p>
454 	 * Attempts to convert the child objects to the correct type if they aren't already the correct type.
455 	 * <p>
456 	 * The <code>next()</code> method on the returned iterator may throw a {@link InvalidDataConversionException} if
457 	 * 	the next element cannot be converted to the specified type.
458 	 * <p>
459 	 * See {@link BeanContext#convertToType(Object, ClassMeta)} for a description of valid conversions.
460 	 *
461 	 * <dl>
462 	 * 	<dt>Example:</dt>
463 	 * 	<dd>
464 	 * 		<p class='bcode'>
465 	 * 	<jc>// Iterate over a list of ObjectMaps.</jc>
466 	 * 	ObjectList l = <jk>new</jk> ObjectList(<js>"[{foo:'bar'},{baz:123}]"</js>);
467 	 * 	for (ObjectMap m : l.elements(ObjectMap.<jk>class</jk>)) {
468 	 * 		<jc>// Do something with m.</jc>
469 	 * 	}
470 	 *
471 	 * 	<jc>// Iterate over a list of ints.</jc>
472 	 * 	ObjectList l = <jk>new</jk> ObjectList(<js>"[1,2,3]"</js>);
473 	 * 	for (Integer i : l.elements(Integer.<jk>class</jk>)) {
474 	 * 		<jc>// Do something with i.</jc>
475 	 * 	}
476 	 *
477 	 * 	<jc>// Iterate over a list of beans.</jc>
478 	 * 	<jc>// Automatically converts to beans.</jc>
479 	 * 	ObjectList l = <jk>new</jk> ObjectList(<js>"[{name:'John Smith',age:45}]"</js>);
480 	 * 	for (Person p : l.elements(Person.<jk>class</jk>)) {
481 	 * 		<jc>// Do something with p.</jc>
482 	 * 	}
483 	 * 		</p>
484 	 * 	</dd>
485 	 * </dl>
486 	 *
487 	 * @param <E> The child object type.
488 	 * @param childType The child object type.
489 	 * @return A new <code>Iterable</code> object over this list.
490 	 */
491 	public <E> Iterable<E> elements(final Class<E> childType) {
492 		final Iterator<?> i = iterator();
493 		return new Iterable<E>() {
494 
495 			@Override /* Iterable */
496 			public Iterator<E> iterator() {
497 				return new Iterator<E>() {
498 
499 					@Override /* Iterator */
500 					public boolean hasNext() {
501 						return i.hasNext();
502 					}
503 
504 					@Override /* Iterator */
505 					public E next() {
506 						return beanContext.convertToType(i.next(), childType);
507 					}
508 
509 					@Override /* Iterator */
510 					public void remove() {
511 						i.remove();
512 					}
513 
514 				};
515 			}
516 		};
517 	}
