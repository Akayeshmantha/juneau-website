25  /**
26   * Parses a MessagePack stream into a POJO model.
27   *
28   *
29   * <h6 class='topic'>Media types</h6>
30   * <p>
31   * 	Handles <code>Content-Type</code> types: <code>octal/msgpack</code>
32   *
33   *
34   * <h6 class='topic'>Configurable properties</h6>
35   * <p>
36   * 	This class has the following properties associated with it:
37   * <ul>
38   * 	<li>{@link MsgPackParserContext}
39   * </ul>
40   *
41   *
42   * @author James Bognar (james.bognar@salesforce.com)
43   */
44  @SuppressWarnings({ "rawtypes", "unchecked" })
45  @Consumes({"octal/msgpack"})
46  public final class MsgPackParser extends InputStreamParser {
47  
48  	/** Default parser, all default settings.*/
49  	public static final MsgPackParser DEFAULT = new MsgPackParser().lock();

51  	/**
52  	 * Workhorse method.
53  	 */
54  	private <T> T parseAnything(MsgPackParserSession session, ClassMeta<T> eType, MsgPackInputStream is, Object outer, BeanPropertyMeta pMeta) throws Exception {
55  
56  		BeanContext bc = session.getBeanContext();
57  		if (eType == null)
58  			eType = (ClassMeta<T>)object();
59  		PojoSwap<T,Object> transform = (PojoSwap<T,Object>)eType.getPojoSwap();
60  		ClassMeta<?> sType = eType.getSerializedClassMeta();
61  		session.setCurrentClass(sType);
62  		BeanDictionary bd = (pMeta == null ? bc.getBeanDictionary() : pMeta.getBeanDictionary());
63  
64  		Object o = null;
65  		DataType dt = is.readDataType();
66  		int length = (int)is.readLength();
67  
68  		if (dt != DataType.NULL) {
69  			if (dt == BOOLEAN)
70  				o = is.readBoolean();
71  			else if (dt == INT)
72  				o = is.readInt();
73  			else if (dt == LONG)
74  				o = is.readLong();
75  			else if (dt == FLOAT)
76  				o = is.readFloat();
77  			else if (dt == DOUBLE)
78  				o = is.readDouble();
79  			else if (dt == STRING)
80  				o = session.trim(is.readString());
81  			else if (dt == BIN)
82  				o = is.readBinary();
83  			else if (dt == ARRAY && sType.isObject()) {
84  				ObjectList ol = new ObjectList(bc);
85  				for (int i = 0; i < length; i++)
86  					ol.add(parseAnything(session, object(), is, outer, pMeta));
87  				o = ol;
88  			} else if (dt == MAP && sType.isObject()) {
89  				ObjectMap om = new ObjectMap(bc);
90  				for (int i = 0; i < length; i++)
91  					om.put(parseAnything(session, string(), is, outer, pMeta), parseAnything(session, object(), is, om, pMeta));
92  				o = bd.cast(om);
93  			}
94  
95  			if (sType.isObject()) {
96  				// Do nothing.
97  			} else if (sType.isBoolean() || sType.isCharSequence() || sType.isChar() || sType.isNumber()) {
98  				o = bc.convertToType(o, sType);
99  			} else if (sType.isMap()) {
100 				if (dt == MAP) {
101 					Map m = (sType.canCreateNewInstance(outer) ? (Map)sType.newInstance(outer) : new ObjectMap(bc));
102 					for (int i = 0; i < length; i++) {
103 						Object key = parseAnything(session, sType.getKeyType(), is, outer, pMeta);
104 						ClassMeta<?> vt = sType.getValueType();
105 						Object value = parseAnything(session, vt, is, m, pMeta);
106 						setName(vt, value, key);
107 						m.put(key, value);
108 					}
109 					o = m;
110 				} else {
111 					throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, sType);
112 				}
113 			} else if (sType.canCreateNewInstanceFromObjectMap(outer)) {
114 				ObjectMap m = new ObjectMap(bc);
115 				for (int i = 0; i < length; i++)
116 					m.put(parseAnything(session, string(), is, outer, pMeta), parseAnything(session, object(), is, m, pMeta));
117 				o = sType.newInstanceFromObjectMap(outer, m);
118 			} else if (sType.canCreateNewBean(outer)) {
119 				if (dt == MAP) {
120 					BeanMap m = bc.newBeanMap(outer, sType.getInnerClass());
121 					for (int i = 0; i < length; i++) {
122 						String pName = parseAnything(session, string(), is, m.getBean(false), null);
123 						BeanPropertyMeta bpm = m.getPropertyMeta(pName);
124 						if (bpm == null) {
125 							if (pName.equals(bc.getBeanTypePropertyName()))
126 								parseAnything(session, bc.string(), is, null, null);
127 							else
128 								onUnknownProperty(session, pName, m, 0, is.getPosition());
129 						} else {
130 							ClassMeta<?> cm = bpm.getClassMeta();
131 							Object value = parseAnything(session, cm, is, m.getBean(false), bpm);
132 							setName(cm, value, pName);
133 							bpm.set(m, value);
134 						}
135 					}
136 					o = m.getBean();
137 				} else {
138 					throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, sType);
139 				}
140 			} else if (sType.canCreateNewInstanceFromString(outer) && dt == STRING) {
141 				o = sType.newInstanceFromString(outer, o == null ? "" : o.toString());
142 			} else if (sType.canCreateNewInstanceFromNumber(outer) && dt.isOneOf(INT, LONG, FLOAT, DOUBLE)) {
143 				o = sType.newInstanceFromNumber(outer, (Number)o);
144 			} else if (sType.isCollection()) {
145 				if (dt == MAP) {
146 					ObjectMap m = new ObjectMap(bc);
147 					for (int i = 0; i < length; i++)
148 						m.put(parseAnything(session, string(), is, outer, pMeta), parseAnything(session, object(), is, m, pMeta));
149 					o = bd.cast(m);
150 				} else if (dt == ARRAY) {
151 					Collection l = (sType.canCreateNewInstance(outer) ? (Collection)sType.newInstance() : new ObjectList(bc));
152 					for (int i = 0; i < length; i++)
153 						l.add(parseAnything(session, sType.getElementType(), is, l, pMeta));
154 					o = l;
155 				} else {
156 					throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, sType);
157 				}
158 			} else if (sType.isArray()) {
159 				if (dt == MAP) {
160 					ObjectMap m = new ObjectMap(bc);
161 					for (int i = 0; i < length; i++)
162 						m.put(parseAnything(session, string(), is, outer, pMeta), parseAnything(session, object(), is, m, pMeta));
163 					o = bd.cast(m);
164 				} else if (dt == ARRAY) {
165 					Collection l = (sType.isCollection() && sType.canCreateNewInstance(outer) ? (Collection)sType.newInstance() : new ObjectList(bc));
166 					for (int i = 0; i < length; i++)
167 						l.add(parseAnything(session, sType.getElementType(), is, l, pMeta));
168 					o = bc.toArray(sType, l);
169 				} else {
170 					throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, sType);
171 				}
172 			} else if (dt == MAP) {
173 				ObjectMap m = new ObjectMap(bc);
174 				for (int i = 0; i < length; i++)
175 					m.put(parseAnything(session, string(), is, outer, pMeta), parseAnything(session, object(), is, m, pMeta));
176 				if (m.containsKey(bc.getBeanTypePropertyName()))
177 					o = bd.cast(m);
178 				else
179 					throw new ParseException(session, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
180 			} else {
181 				throw new ParseException(session, "Invalid data type {0} encountered for parse type {1}", dt, sType);
182 			}
183 		}
184 
185 		if (transform != null && o != null)
186 			o = transform.unswap(o, eType, bc);
187 
188 		if (outer != null)
189 			setParent(eType, o, outer);
190 
191 		return (T)o;
192 	}

198 	@Override /* Parser */
199 	public MsgPackParserSession createSession(Object input, ObjectMap op, Method javaMethod, Object outer) {
200 		return new MsgPackParserSession(getContext(MsgPackParserContext.class), getBeanContext(), input, op, javaMethod, outer);
201 	}
202 
203 	@Override /* Parser */
204 	protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
205 		MsgPackParserSession s = (MsgPackParserSession)session;
206 		type = s.getBeanContext().normalizeClassMeta(type);
207 		MsgPackInputStream is = s.getInputStream();
208 		T o = parseAnything(s, type, is, s.getOuter(), null);
209 		return o;
210 	}
