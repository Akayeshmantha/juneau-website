93  	@Override /* Map */
94  	public void clear() {
95  		Set<String> changes = createChanges();
96  		writeLock();
97  		try {
98  			if (changes != null)
99  				for (String k : keySet())
100 					changes.add(getFullKey(name, k));
101 			entries.clear();
102 			lines.clear();
103 			headerComments.clear();
104 		} finally {
105 			writeUnlock();
106 		}
107 		signalChanges(changes);
108 	}

243 	String put(String key, String value, boolean encoded, Set<String> changes) {
244 		writeLock();
245 		try {
246 			addLine(key, encoded);
247 			String prev = entries.put(key, value);
248 			addChange(changes, key, prev, value);
249 			return prev;
250 		} finally {
251 			writeUnlock();
252 		}

271 	String remove(Object key, Set<String> changes) {
272 		writeLock();
273 		try {
274 			String prev = entries.remove(key);
275 			addChange(changes, key.toString(), prev, null);
276 			removeLine(key.toString());
277 			return prev;
278 		} finally {
279 			writeUnlock();
280 		}
281 	}

283 	private void removeLine(String key) {
284 		for (Iterator<String> i = lines.iterator(); i.hasNext();) {
285 			String k = i.next();
286 			if (k.startsWith("*") || k.startsWith(">")) {
287 				if (k.substring(1).equals(key)) {
288 					i.remove();
289 					break;
290 				}
291 			}
292 		}
293 	}

310 	 * Returns <jk>true</jk> if the specified entry is encoded.
311 	 *
312 	 * @param key The key.
313 	 * @return <jk>true</jk> if the specified entry is encoded.
314 	 */
315 	public boolean isEncoded(String key) {
316 		readLock();
317 		try {
318 			for (String s : lines)
319 				if (s.length() > 1)
320 					if (s.substring(1).equals(key))
321 						return s.charAt(0) == '*';
322 			return false;
323 		} finally {
324 			readUnlock();
325 		}
326 	}

329 	 * Adds header comments to this section.
330 	 * @see ConfigFile#addHeaderComments(String, String...) for a description.
331 	 * @param comments The comment lines to add to this section.
332 	 * @return This object (for method chaining).
333 	 */
334 	public Section addHeaderComments(List<String> comments) {
335 		writeLock();
336 		try {
337 			for (String c : comments) {
338 				if (c == null)
339 					c = "";
340 				if (! c.startsWith("#"))
341 					c = "#" + c;
342 				this.headerComments.add(c);
343 			}
344 			return this;
345 		} finally {
346 			writeUnlock();
347 		}
348 	}

362 	/**
363 	 * Serialize this section.
364 	 * @param out What to serialize to.
365 	 * @param format The format (e.g. INI, BATCH, SHELL).
366 	 */
367 	public void writeTo(PrintWriter out, ConfigFileFormat format) {
368 		readLock();
369 		try {
370 			if (format == INI) {
371 				for (String s : headerComments)
372 					out.append(s).println();
373 				if (! name.equals("default"))
374 					out.append('[').append(name).append(']').println();
375 				for (String l : lines) {
376 					char c = (l.length() > 0 ? l.charAt(0) : 0);
377 					if (c == '>' || c == '*'){
378 						boolean encode = c == '*';
379 						String key = l.substring(1);
380 						String val = entries.get(key);
381 						if (val.indexOf('\n') != -1)
382 							val = val.replaceAll("(\\r?\\n)", "$1\t");
383 						if (val.indexOf('=') != -1)
384 							val = val.replace("=", "\\u003D");
385 						if (val.indexOf('#') != -1)
386 							val = val.replace("#", "\\u0023");
387 						out.append(key);
388 						if (encode)
389 							out.append('*');
390 						out.append(" = ");
391 						if (encode)
392 							out.append('{').append(configFile.getEncoder().encode(key, val)).append('}');
393 						else
394 							out.append(val);
395 						out.println();
396 					} else {
397 						out.append(l).println();
398 					}
399 				}
400 
401 			} else if (format == BATCH) {
402 				String section = name.replaceAll("\\.\\/", "_");
403 				for (String l : headerComments) {
404 					l = trimComment(l);
405 					if (! l.isEmpty())
406 						out.append("rem ").append(l);
407 					out.println();
408 				}
409 				for (String l : lines) {
410 					char c = (l.length() > 0 ? l.charAt(0) : 0);
411 					if (c == '>' || c == '*') {
412 						String key = l.substring(1);
413 						String val = entries.get(key);
414 						out.append("set ");
415 						if (! name.equals("default"))
416 							out.append(section).append('_');
417 						out.append(key.replaceAll("\\.\\/", "_")).append(" = ").append(val).println();
418 					} else {
419 						l = trimComment(l);
420 						if (! l.isEmpty())
421 							out.append("rem ").append(l);
422 						out.println();
423 					}
424 				}
425 
426 			} else if (format == SHELL) {
427 				String section = name.replaceAll("\\.\\/", "_");
428 				for (String l : headerComments) {
429 					l = trimComment(l);
430 					if (! l.isEmpty())
431 						out.append("# ").append(l);
432 					out.println();
433 				}
434 				for (String l : lines) {
435 					char c = (l.length() > 0 ? l.charAt(0) : 0);
436 					if (c == '>' || c == '*'){
437 						String key = l.substring(1);
438 						String val = entries.get(key).replaceAll("\\\\", "\\\\\\\\");
439 						out.append("export ");
440 						if (! name.equals("default"))
441 							out.append(section).append('_');
442 						out.append(key.replaceAll("\\.\\/", "_")).append('=').append('"').append(val).append('"').println();
443 					} else {
444 						l = trimComment(l);
445 						if (! l.isEmpty())
446 							out.append("# ").append(l);
447 						out.println();
448 					}
449 				}
450 			}
451 		} finally {
452 			readUnlock();
453 		}
454 	}

460 	/*
461 	 * Add lines to this section.
462 	 */
463 	Section addLines(Set<String> changes, String...l) {
464 		writeLock();
465 		try {
466 			if (l == null)
467 				l = new String[0];
468 			for (int i = 0; i < l.length; i++) {
469 				String line = l[i];
470 				if (line == null)
471 					line = "";
472 				if (isComment(line))
473 					this.lines.add(line);
474 				else if (isAssignment(line)) {
475 					// Key/value pairs are stored as either ">key" or "*key";
476 					String key = StringUtils.replaceUnicodeSequences(line.substring(0, line.indexOf('=')).trim());
477 					String val = StringUtils.replaceUnicodeSequences(line.substring(line.indexOf('=')+1).trim());
478 					boolean encoded = key.length() > 1 && key.endsWith("*");
479 					if (encoded) {
480 						key = key.substring(0, key.lastIndexOf('*'));
481 						String v = val.toString().trim();
482 						if (v.startsWith("{") && v.endsWith("}"))
483 							val = configFile.getEncoder().decode(key, v.substring(1, v.length()-1));
484 						else
485 							configFile.setHasBeenModified();
486 					}
487 					if (containsKey(key)) {
488 						entries.remove(key);
489 						lines.remove('*' + key);
490 						lines.remove('>' + key);
491 					}
492 					lines.add((encoded ? '*' : '>') + key);
493 					addChange(changes, key, entries.put(key, val), val);
494 				} else {
495 					this.lines.add(line);
496 				}
497 			}
498 			return this;
499 		} finally {
500 			writeUnlock();
501 		}
502 	}

504 	/*
505 	 * Remove all "#*" lines at the end of this section so they can
506 	 * be associated with the next section.
507 	 */
508 	List<String> removeTrailingComments() {
509 		LinkedList<String> l = new LinkedList<String>();
510 		while ((! lines.isEmpty()) && lines.getLast().startsWith("#"))
511 			l.addFirst(lines.removeLast());
512 		return l;
513 	}

519 	private void addLine(String key, boolean encoded) {
520 		for (Iterator<String> i = lines.iterator(); i.hasNext();) {
521 			String k = i.next();
522 			if ((k.startsWith("*") || k.startsWith(">")) && k.substring(1).equals(key)) {
523 				if (k.startsWith("*") && encoded || k.startsWith(">") && ! encoded)
524 					return;
525 				i.remove();
526 			}
527 		}
528 		lines.add((encoded ? "*" : ">") + key);
529 	}

531 	private void readLock() {
532 		lock.readLock().lock();
533 	}
534 
535 	private void readUnlock() {
536 		lock.readLock().unlock();
537 	}
538 
539 	private void writeLock() {
540 		if (readOnly)
541 			throw new UnsupportedOperationException("Cannot modify read-only ConfigFile.");
542 		lock.writeLock().lock();
543 	}
544 
545 	private void writeUnlock() {
546 		lock.writeLock().unlock();
547 	}
548 
549 	private String trimComment(String s) {
550 		return s.replaceAll("^\\s*\\#\\s*", "").trim();
551 	}
552 
553 	private Set<String> createChanges() {
554 		return (configFile != null && configFile.getListeners().size() > 0 ? new LinkedHashSet<String>() : null);
555 	}
556 
557 	private void signalChanges(Set<String> changes) {
558 		if (changes != null && ! changes.isEmpty())
559 			for (ConfigFileListener l : configFile.getListeners())
560 				l.onChange(configFile, changes);
561 	}
562 
563 	private void addChange(Set<String> changes, String key, String oldVal, String newVal) {
564 		if (changes != null)
565 			if (! StringUtils.isEquals(oldVal, newVal))
566 				changes.add(getFullKey(name, key));
567 	}
