13  package org.apache.juneau.ini;
14  
15  /**
16   * Internal utility methods.
17   *
18   * @author James Bognar (james.bognar@salesforce.com)
19   */
20  public class ConfigUtils {
21  
22  	static final String getSectionName(String key) {
23  		int i = key.indexOf('/');
24  		if (i == -1)
25  			return "default";
26  		return key.substring(0, i);
27  	}
28  
29  	static final String getSectionKey(String key) {
30  		int i = key.indexOf('/');
31  		if (i == -1)
32  			return key;
33  		return key.substring(i+1);
34  	}
35  
36  	static final String getFullKey(String section, String key) {
37  		if (section.equals("default"))
38  			return key;
39  		return section + '/' + key;
40  	}
41  
42  	static final boolean isComment(String line) {
43  		for (int i = 0; i < line.length(); i++) {
44  			char c = line.charAt(i);
45  			if (! Character.isWhitespace(c))
46  				return c == '#';
47  		}
48  		return false;
49  	}
50  
51  	static final boolean isAssignment(String line) {
52  		int S1 = 1; // Looking for char;
53  		int S2 = 2; // Found char, looking for whitespace or =
54  		int S3 = 3; // Found whitespace, looking for =
55  		int state = S1;
56  		for (int i = 0; i < line.length(); i++) {
57  			char c = line.charAt(i);
58  			if (state == S1) {
59  				if (! Character.isWhitespace(c))
60  					state = S2;
61  			} else if (state == S2) {
62  				if (c == '=')
63  					return true;
64  				if (Character.isWhitespace(c))
65  					state = S3;
66  			} else if (state == S3) {
67  				if (c == '=')
68  					return true;
69  			}
70  		}
71  		return false;
72  	}
73  
74  	static final boolean isSection(String line) {
75  		int S1 = 1; // Looking for [;
76  		int S2 = 2; // Found [, looking for ]
77  		int state = S1;
78  		for (int i = 0; i < line.length(); i++) {
79  			char c = line.charAt(i);
80  			if (state == S1) {
81  				if (! Character.isWhitespace(c)) {
82  					if (c == '[')
83  						state = S2;
84  					else
85  						return false;
86  				}
87  			} else if (state == S2) {
88  				if (c == ']')
89  					return true;
90  			}
91  		}
92  		return false;
93  	}
94  }
