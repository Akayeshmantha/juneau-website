13  package org.apache.juneau.ini;
14  
15  import java.io.*;
16  
17  import org.apache.juneau.*;
18  
19  /**
20   * Wraps a {@link ConfigFile} in a {@link Writable} to be rendered as plain text.
21   */
22  class ConfigFileWritable implements Writable {
23  
24  	private ConfigFileImpl cf;
25  
26  	protected ConfigFileWritable(ConfigFileImpl cf) {
27  		this.cf = cf;
28  	}
29  
30  	@Override /* Writable */
31  	public void writeTo(Writer out) throws IOException {
32  		cf.readLock();
33  		try {
34  			cf.serializeTo(out);
35  		} finally {
36  			cf.readUnlock();
37  		}
38  	}
39  
40  	@Override /* Writable */
41  	public String getMediaType() {
42  		return "text/plain";
43  	}
44  }
