35  /**
36   * Serializes POJO metadata to HTTP responses as XML.
37   *
38   *
39   * <h6 class='topic'>Media types</h6>
40   * <p>
41   * 	Handles <code>Accept</code> types: <code>text/xml+schema</code>
42   * <p>
43   * 	Produces <code>Content-Type</code> types: <code>text/xml</code>
44   *
45   *
46   * <h6 class='topic'>Description</h6>
47   * <p>
48   * 	Produces the XML-schema representation of the XML produced by the {@link XmlSerializer} class with the same properties.
49   *
50   *
51   * <h6 class='topic'>Configurable properties</h6>
52   * <p>
53   * 	This class has the following properties associated with it:
54   * <ul>
55   * 	<li>{@link XmlSerializerContext}
56   * 	<li>{@link BeanContext}
57   * </ul>
58   *
59   * @author James Bognar (james.bognar@salesforce.com)
60   */
61  @Produces(value="text/xml+schema",contentType="text/xml")
62  public class XmlSchemaSerializer extends XmlSerializer {

65  	 * Constructor
66  	 */

68  		super();
69  	}

80  	@Override /* XmlSerializer */
81  	protected void doSerialize(SerializerSession session, Object o) throws Exception {
82  		XmlSerializerSession s = (XmlSerializerSession)session;
83  
84  		if (s.isEnableNamespaces() && s.isAutoDetectNamespaces())
85  			findNsfMappings(s, o);
86  
87  		Namespace xs = s.getXsNamespace();
88  		Namespace[] allNs = ArrayUtils.append(new Namespace[]{s.getDefaultNamespace()}, s.getNamespaces());
89  
90  		Schemas schemas = new Schemas(s, xs, s.getDefaultNamespace(), allNs);
91  		schemas.process(s, o);
92  		schemas.serializeTo(session.getWriter());
93  	}

95  	/**
96  	 * Returns an XML-Schema validator based on the output returned by {@link #doSerialize(SerializerSession, Object)};
97  	 *
98  	 * @param session The serializer session object return by {@link #createSession(Object, ObjectMap, Method)}.<br>
99  	 * 	Can be <jk>null</jk>.
100 	 * @param o The object to serialize.
101 	 * @return The new validator.
102 	 * @throws Exception If a problem was detected in the XML-Schema output produced by this serializer.
103 	 */
104 	public Validator getValidator(SerializerSession session, Object o) throws Exception {
105 		doSerialize(session, o);
106 		String xmlSchema = session.getWriter().toString();
107 
108 		// create a SchemaFactory capable of understanding WXS schemas
109 		SchemaFactory factory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
110 
111 		if (xmlSchema.indexOf('\u0000') != -1) {
112 
113 			// Break it up into a map of namespaceURI->schema document
114 			final Map<String,String> schemas = new HashMap<String,String>();
115 			String[] ss = xmlSchema.split("\u0000");
116 			xmlSchema = ss[0];
117 			for (String s : ss) {
118 				Matcher m = pTargetNs.matcher(s);
119 				if (m.find())
120 					schemas.put(m.group(1), s);
121 			}
122 
123 			// Create a custom resolver
124 			factory.setResourceResolver(
125 				new LSResourceResolver() {
126 
127 					@Override /* LSResourceResolver */
128 					public LSInput resolveResource(String type, String namespaceURI, String publicId, String systemId, String baseURI) {
129 
130 						String schema = schemas.get(namespaceURI);
131 						if (schema == null)
132 							throw new RuntimeException(MessageFormat.format("No schema found for namespaceURI ''{0}''", namespaceURI));
133 
134 						try {
135 							DOMImplementationRegistry registry = DOMImplementationRegistry.newInstance();
136 							DOMImplementationLS domImplementationLS = (DOMImplementationLS)registry.getDOMImplementation("LS 3.0");
137 							LSInput in = domImplementationLS.createLSInput();
138 							in.setCharacterStream(new StringReader(schema));
139 							in.setSystemId(systemId);
140 							return in;
141 
142 						} catch (Exception e) {
143 							throw new RuntimeException(e);
144 						}
145 					}
146 				}
147 			);
148 		}
149 		return factory.newSchema(new StreamSource(new StringReader(xmlSchema))).newValidator();
150 	}

155 	/* An instance of a global element, global attribute, or XML type to be serialized. */
156 	private static class QueueEntry {
157 		Namespace ns;
158 		String name;
159 		ClassMeta<?> cm;
160 		QueueEntry(Namespace ns, String name, ClassMeta<?> cm) {
161 			this.ns = ns;
162 			this.name = name;
163 			this.cm = cm;
164 		}
165 	}

167 	/* An encapsulation of all schemas present in the metamodel of the serialized object. */
168 	private class Schemas extends LinkedHashMap<Namespace,Schema> {
169 
170 		private static final long serialVersionUID = 1L;
171 
172 		private Namespace defaultNs;
173 		private LinkedList<QueueEntry>
174 			elementQueue = new LinkedList<QueueEntry>(),
175 			attributeQueue = new LinkedList<QueueEntry>(),
176 			typeQueue = new LinkedList<QueueEntry>();
177 
178 		private Schemas(XmlSerializerSession session, Namespace xs, Namespace defaultNs, Namespace[] allNs) throws IOException {
179 			this.defaultNs = defaultNs;
180 			for (Namespace ns : allNs)
181 				put(ns, new Schema(session, this, xs, ns, defaultNs, allNs));
182 		}
183 
184 		private Schema getSchema(Namespace ns) {
185 			if (ns == null)
186 				ns = defaultNs;
187 			Schema s = get(ns);
188 			if (s == null)
189 				throw new RuntimeException("No schema defined for namespace '"+ns+"'");
190 			return s;
191 		}
192 
193 		private void process(SerializerSession session, Object o) throws IOException {
194 			ClassMeta<?> cm = session.getBeanContext().getClassMetaForObject(o);
195 			Namespace ns = defaultNs;
196 			if (cm == null)
197 				queueElement(ns, "null", object());
198 			else {
199 				XmlClassMeta xmlMeta = cm.getExtendedMeta(XmlClassMeta.class);
200 				if (cm.getDictionaryName() != null && xmlMeta.getNamespace() != null)
201 					ns = xmlMeta.getNamespace();
202 				queueElement(ns, cm.getDictionaryName(), cm);
203 			}
204 			processQueue();
205 		}
206 
207 
208 		private void processQueue() throws IOException {
209 			boolean b;
210 			do {
211 				b = false;
212 				while (! elementQueue.isEmpty()) {
213 					QueueEntry q = elementQueue.removeFirst();
214 					b |= getSchema(q.ns).processElement(q.name, q.cm);
215 				}
216 				while (! typeQueue.isEmpty()) {
217 					QueueEntry q = typeQueue.removeFirst();
218 					b |= getSchema(q.ns).processType(q.name, q.cm);
219 				}
220 				while (! attributeQueue.isEmpty()) {
221 					QueueEntry q = attributeQueue.removeFirst();
222 					b |= getSchema(q.ns).processAttribute(q.name, q.cm);
223 				}
224 			} while (b);
225 		}
226 
227 		private void queueElement(Namespace ns, String name, ClassMeta<?> cm) {
228 			elementQueue.add(new QueueEntry(ns, name, cm));
229 		}
230 
231 		private void queueType(Namespace ns, String name, ClassMeta<?> cm) {
232 			if (name == null)
233 				name = XmlUtils.encodeElementName(cm);
234 			typeQueue.add(new QueueEntry(ns, name, cm));
235 		}
236 
237 		private void queueAttribute(Namespace ns, String name, ClassMeta<?> cm) {
238 			attributeQueue.add(new QueueEntry(ns, name, cm));
239 		}
240 
241 		private void serializeTo(Writer w) throws IOException {
242 			boolean b = false;
243 			for (Schema s : values()) {
244 				if (b)
245 					w.append('\u0000');
246 				w.append(s.toString());
247 				b = true;
248 			}
249 		}
250 	}

252 	/* An encapsulation of a single schema. */
253 	private class Schema {
254 		private StringWriter sw = new StringWriter();
255 		private XmlWriter w;
256 		private XmlSerializerSession session;
257 		private Namespace defaultNs, targetNs;
258 		private Schemas schemas;
259 		private Set<String>
260 			processedTypes = new HashSet<String>(),
261 			processedAttributes = new HashSet<String>(),
262 			processedElements = new HashSet<String>();
263 
264 		public Schema(XmlSerializerSession session, Schemas schemas, Namespace xs, Namespace targetNs, Namespace defaultNs, Namespace[] allNs) throws IOException {
265 			this.schemas = schemas;
266 			this.defaultNs = defaultNs;
267 			this.targetNs = targetNs;
268 			this.session = session;
269 			w = new XmlWriter(sw, session.isUseIndentation(), session.isTrimStrings(), session.getQuoteChar(), null, null, true, null);
270 			int i = session.getIndent();
271 			w.oTag(i, "schema");
272 			w.attr("xmlns", xs.getUri());
273 			w.attr("targetNamespace", targetNs.getUri());
274 			w.attr("elementFormDefault", "qualified");
275 			if (targetNs != defaultNs)
276 				w.attr("attributeFormDefault", "qualified");
277 			for (Namespace ns2 : allNs)
278 				w.attr("xmlns", ns2.name, ns2.uri);
279 			w.append('>').nl();
280 			for (Namespace ns : allNs) {
281 				if (ns != targetNs) {
282 					w.oTag(i+1, "import")
283 						.attr("namespace", ns.getUri())
284 						.attr("schemaLocation", ns.getName()+".xsd")
285 						.append("/>").nl();
286 				}
287 			}
288 		}
289 
290 		private boolean processElement(String name, ClassMeta<?> cm) throws IOException {
291 			if (processedElements.contains(name))
292 				return false;
293 			processedElements.add(name);
294 
295 			ClassMeta<?> ft = cm.getSerializedClassMeta();
296 			int i = session.getIndent() + 1;
297 			if (name == null)
298 				name = getElementName(ft);
299 			Namespace ns = first(ft.getExtendedMeta(XmlClassMeta.class).getNamespace(), defaultNs);
300 			String type = getXmlType(ns, ft);
301 
302 			w.oTag(i, "element")
303 				.attr("name", XmlUtils.encodeElementName(name))
304 				.attr("type", type)
305 				.append('/').append('>').nl();
306 
307 			schemas.queueType(ns, null, ft);
308 			schemas.processQueue();
309 			return true;
310 		}
311 
312 		private boolean processAttribute(String name, ClassMeta<?> cm) throws IOException {
313 			if (processedAttributes.contains(name))
314 				return false;
315 			processedAttributes.add(name);
316 
317 			int i = session.getIndent() + 1;
318 			String type = getXmlAttrType(cm);
319 
320 			w.oTag(i, "attribute")
321 				.attr("name", name)
322 				.attr("type", type)
323 				.append('/').append('>').nl();
324 
325 			return true;
326 		}
327 
328 		private boolean processType(String name, ClassMeta<?> cm) throws IOException {
329 			if (processedTypes.contains(name))
330 				return false;
331 			processedTypes.add(name);
332 
333 			int i = session.getIndent() + 1;
334 
335 			cm = cm.getSerializedClassMeta();
336 			BeanContext bc = cm.getBeanContext();
337 
338 			w.oTag(i, "complexType")
339 				.attr("name", name);
340 
341 			// This element can have mixed content if:
342 			// 	1) It's a generic Object (so it can theoretically be anything)
343 			//		2) The bean has a property defined with @XmlFormat.CONTENT.
344 			if ((cm.isBean() && cm.getBeanMeta().getExtendedMeta(XmlBeanMeta.class).getXmlContentProperty() != null) || cm.isObject())
345 				w.attr("mixed", "true");
346 
347 			w.cTag().nl();
348 
349 			if (! (cm.isMap() || cm.isBean() || cm.hasToObjectMapMethod() || cm.isCollection() || cm.isArray() || (cm.isAbstract() && ! cm.isNumber()) || cm.isObject())) {
350 				String base = getXmlAttrType(cm);
351 				w.sTag(i+1, "simpleContent").nl();
352 				w.oTag(i+2, "extension")
353 					.attr("base", base);
354 				if (session.isAddJsonTypeAttrs() || (session.isAddJsonStringTypeAttrs() && base.equals("string"))) {
355 					w.cTag().nl();
356 					w.oTag(i+3, "attribute")
357 						.attr("name", bc.getBeanTypePropertyName())
358 						.attr("type", "string")
359 						.ceTag().nl();
360 					w.eTag(i+2, "extension").nl();
361 				} else {
362 					w.ceTag().nl();
363 				}
364 				w.eTag(i+1, "simpleContent").nl();
365 
366 			} else {
367 
368 				//----- Bean -----
369 				if (cm.isBean()) {
370 					BeanMeta<?> bm = cm.getBeanMeta();
371 
372 					boolean hasChildElements = false;
373 
374 					for (BeanPropertyMeta pMeta : bm.getPropertyMetas())
375 						if (pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getXmlFormat() != XmlFormat.ATTR && pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getXmlFormat() != XmlFormat.CONTENT)
376 							hasChildElements = true;
377 
378 					if (bm.getExtendedMeta(XmlBeanMeta.class).getXmlContentProperty() != null) {
379 						w.sTag(i+1, "sequence").nl();
380 						w.oTag(i+2, "any")
381 							.attr("processContents", "skip")
382 							.attr("minOccurs", 0)
383 							.ceTag().nl();
384 						w.eTag(i+1, "sequence").nl();
385 
386 					} else if (hasChildElements) {
387 						w.sTag(i+1, "sequence").nl();
388 
389 						boolean hasOtherNsElement = false;
390 
391 						for (BeanPropertyMeta pMeta : bm.getPropertyMetas()) {
392 							XmlBeanPropertyMeta xmlMeta = pMeta.getExtendedMeta(XmlBeanPropertyMeta.class);
393 							if (xmlMeta.getXmlFormat() != XmlFormat.ATTR) {
394 								boolean isCollapsed = xmlMeta.getXmlFormat() == COLLAPSED;
395 								ClassMeta<?> ct2 = pMeta.getClassMeta();
396 								String childName = pMeta.getName();
397 								if (isCollapsed) {
398 									if (xmlMeta.getChildName() != null)
399 										childName = xmlMeta.getChildName();
400 									ct2 = pMeta.getClassMeta().getElementType();
401 								}
402 								Namespace cNs = first(xmlMeta.getNamespace(), ct2.getExtendedMeta(XmlClassMeta.class).getNamespace(), cm.getExtendedMeta(XmlClassMeta.class).getNamespace(), defaultNs);
403 								if (xmlMeta.getNamespace() == null) {
404 									w.oTag(i+2, "element")
405 										.attr("name", XmlUtils.encodeElementName(childName), true)
406 										.attr("type", getXmlType(cNs, ct2));
407 									if (isCollapsed) {
408 										w.attr("minOccurs", 0);
409 										w.attr("maxOccurs", "unbounded");
410 									} else {
411 										if (! session.isTrimNulls())
412 											w.attr("nillable", true);
413 										else
414 											w.attr("minOccurs", 0);
415 									}
416 
417 									w.ceTag().nl();
418 								} else {
419 									// Child element is in another namespace.
420 									schemas.queueElement(cNs, pMeta.getName(), ct2);
421 									hasOtherNsElement = true;
422 								}
423 
424 							}
425 						}
426 
427 						// If this bean has any child elements in another namespace,
428 						// we need to add an <any> element.
429 						if (hasOtherNsElement)
430 							w.oTag(i+2, "any")
431 								.attr("minOccurs", 0)
432 								.attr("maxOccurs", "unbounded")
433 								.ceTag().nl();
434 						w.eTag(i+1, "sequence").nl();
435 					}
436 
437 					for (BeanPropertyMeta pMeta : bm.getExtendedMeta(XmlBeanMeta.class).getXmlAttrProperties().values()) {
438 						Namespace pNs = pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace();
439 						if (pNs == null)
440 							pNs = defaultNs;
441 
442 						// If the bean attribute has a different namespace than the bean, then it needs to
443 						// be added as a top-level entry in the appropriate schema file.
444 						if (pNs != targetNs) {
445 							schemas.queueAttribute(pNs, pMeta.getName(), pMeta.getClassMeta());
446 							w.oTag(i+1, "attribute")
447 							//.attr("name", pMeta.getName(), true)
448 							.attr("ref", pNs.getName() + ':' + pMeta.getName())
449 							.ceTag().nl();
450 						}
451 
452 						// Otherwise, it's just a plain attribute of this bean.
453 						else {
454 							w.oTag(i+1, "attribute")
455 								.attr("name", pMeta.getName(), true)
456 								.attr("type", getXmlAttrType(pMeta.getClassMeta()))
457 								.ceTag().nl();
458 						}
459 					}
460 
461 				//----- Collection -----
462 				} else if (cm.isCollection() || cm.isArray()) {
463 					ClassMeta<?> elementType = cm.getElementType();
464 					if (elementType.isObject()) {
465 						w.sTag(i+1, "sequence").nl();
466 						w.oTag(i+2, "any")
467 							.attr("processContents", "skip")
468 							.attr("maxOccurs", "unbounded")
469 							.attr("minOccurs", "0")
470 							.ceTag().nl();
471 						w.eTag(i+1, "sequence").nl();
472 					} else {
473 						Namespace cNs = first(elementType.getExtendedMeta(XmlClassMeta.class).getNamespace(), cm.getExtendedMeta(XmlClassMeta.class).getNamespace(), defaultNs);
474 						schemas.queueType(cNs, null, elementType);
475 						w.sTag(i+1, "sequence").nl();
476 						w.oTag(i+2, "any")
477 							.attr("processContents", "skip")
478 							.attr("maxOccurs", "unbounded")
479 							.attr("minOccurs", "0")
480 							.ceTag().nl();
481 						w.eTag(i+1, "sequence").nl();
482 					}
483 
484 				//----- Map -----
485 				} else if (cm.isMap() || cm.hasToObjectMapMethod() || cm.isAbstract() || cm.isObject()) {
486 					w.sTag(i+1, "sequence").nl();
487 					w.oTag(i+2, "any")
488 						.attr("processContents", "skip")
489 						.attr("maxOccurs", "unbounded")
490 						.attr("minOccurs", "0")
491 						.ceTag().nl();
492 					w.eTag(i+1, "sequence").nl();
493 				}
494 
495 				if (session.isAddBeanTypeProperties() || session.isAddJsonTypeAttrs()) {
496 					w.oTag(i+1, "attribute")
497 						.attr("name", bc.getBeanTypePropertyName())
498 						.attr("type", "string")
499 						.ceTag().nl();
500 				}
501 			}
502 
503 			w.eTag(i, "complexType").nl();
504 			schemas.processQueue();
505 
506 			return true;
507 		}

509 		private String getElementName(ClassMeta<?> cm) {
510 			cm = cm.getSerializedClassMeta();
511 			String name = cm.getDictionaryName();
512 
513 			if (name == null) {
514 				if (cm.isBoolean())
515 					name = "boolean";
516 				else if (cm.isNumber())
517 					name = "number";
518 				else if (cm.isArray() || cm.isCollection())
519 					name = "array";
520 				else if (! (cm.isMap() || cm.hasToObjectMapMethod() || cm.isBean() || cm.isCollection() || cm.isArray() || cm.isObject() || cm.isAbstract()))
521 					name = "string";
522 				else
523 					name = "object";
524 			}
525 			return name;
526 		}

528 		@Override /* Object */
529 		public String toString() {
530 			try {
531 				w.eTag(session.getIndent(), "schema").nl();
532 			} catch (IOException e) {
533 				throw new RuntimeException(e); // Shouldn't happen.
534 			}
535 			return sw.toString();
536 		}
537 
538 		private String getXmlType(Namespace currentNs, ClassMeta<?> cm) {
539 			String name = null;
540 			cm = cm.getSerializedClassMeta();
541 			if (currentNs == targetNs && ! session.isAddJsonTypeAttrs()) {
542 				if (cm.isBoolean())
543 					name = "boolean";
544 				else if (cm.isNumber()) {
545 					if (cm.isDecimal())
546 						name = "decimal";
547 					else
548 						name = "integer";
549 				}
550 				if (name == null && ! session.isAddJsonStringTypeAttrs()) {
551 					if (! (cm.isMap() || cm.hasToObjectMapMethod() || cm.isBean() || cm.isCollection() || cm.isArray() || cm.isObject() || cm.isAbstract()))
552 						name = "string";
553 				}
554 			}
555 			if (name == null) {
556 				name = XmlUtils.encodeElementName(cm);
557 				schemas.queueType(currentNs, name, cm);
558 				return currentNs.getName() + ":" + name;
559 			}
560 
561 			return name;
562 		}

565 	private <T> T first(T...tt) {
566 		for (T t : tt)
567 			if (t != null)
568 				return t;
569 		return null;
570 	}

573 	private static String getXmlAttrType(ClassMeta<?> cm) {
574 		if (cm.isBoolean())
575 			return "boolean";
576 		if (cm.isNumber()) {
577 			if (cm.isDecimal())
578 				return "decimal";
579 			return "integer";
580 		}
581 		return "string";
582 	}

584 	@Override /* Serializer */
585 	public XmlSerializerSession createSession(Object output, ObjectMap properties, Method javaMethod) {
586 		// This serializer must always have namespaces enabled.
587 		if (properties == null)
588 			properties = new ObjectMap();
589 		properties.put(XmlSerializerContext.XML_enableNamespaces, true);
590 		return new XmlSerializerSession(getContext(XmlSerializerContext.class), getBeanContext(), output, properties, javaMethod);
591 	}
