13  package org.apache.juneau.xml;
14  
15  import javax.xml.stream.*;
16  
17  import org.apache.juneau.dto.atom.*;
18  import org.apache.juneau.xml.annotation.*;
19  
20  /**
21   * Customization class that allows a bean (or parts of a bean) to be serialized as XML text or mixed content.
22   * <p>
23   * 	For example, the ATOM specification allows text elements (e.g. title, subtitle...)
24   * 		to be either plain text or XML depending on the value of a <xa>type</xa> attribute.
25   * 	The behavior of text escaping thus depends on that attribute.
26   *
27   * <p class='bcode'>
28   * 	<xt>&lt;feed</xt> <xa>xmlns</xa>=<xs>"http://www.w3.org/2005/Atom"</xs><xt>&gt;</xt>
29   * 		<xt>&lt;title</xt> <xa>type</xa>=<xs>"html"</xs><xt>&gt;</xt>
30   * 			&amp;lt;p&amp;gt;&amp;lt;i&amp;gt;This is the title&amp;lt;/i&amp;gt;&amp;lt;/p&amp;gt;
31   * 		<xt>&lt;/title&gt;</xt>
32   * 		<xt>&lt;title</xt> <xa>type</xa>=<xs>"xhtml"</xs><xt>&gt;</xt>
33   * 			<xt>&lt;div</xt> <xa>xmlns</xa>=<xs>"http://www.w3.org/1999/xhtml"</xs><xt>&gt;</xt>
34   * 				<xt>&lt;p&gt;&lt;i&gt;</xt>This is the subtitle<xt>&lt;/i&gt;&lt;/p&gt;</xt>
35   * 			<xt>&lt;/div&gt;</xt>
36   * 		<xt>&lt;/title&gt;</xt>
37   * 	<xt>&lt;/feed&gt;</xt>
38   * </p>
39   *
40   * <p>
41   * 	The ATOM {@link Text} class (the implementation for both the <xt>&lt;title&gt;</xt> and <xt>&lt;subtitle&gt;</xt>
42   * 		tags shown above) then associates a content handler through the {@link Xml#contentHandler()} annotation
43   * 		on the bean property containing the text, like so...
44   *
45   * <p class='bcode'>
46   * 	<ja>@Xml</ja>(format=<jsf>ATTR</jsf>)
47   * 	<jk>public</jk> String getType() {
48   * 		<jk>return</jk> <jf>type</jf>;
49   * 	}
50   *
51   * 	<ja>@Xml</ja>(format=<jsf>CONTENT</jsf>, contentHandler=TextContentHandler.<jk>class</jk>)
52   * 	<jk>public</jk> String getText() {
53   * 		<jk>return</jk> <jf>text</jf>;
54   * 	}
55   *
56   * 	<jk>public void</jk> setText(String text) {
57   * 		<jk>this</jk>.<jf>text</jf> = text;
58   * 	}
59   * </p>
60   *
61   * <p>
62   * 	The content handler that transforms the output is shown below...
63   *
64   * <p class='bcode'>
65   * 	<jk>public static class</jk> TextContentHandler <jk>implements</jk> XmlContentHandler&lt;Text&gt; {
66   *
67   * 		<ja>@Override</ja>
68   * 		<jk>public void</jk> parse(XMLStreamReader r, Text text) <jk>throws</jk> Exception {
69   * 			String type = text.<jf>type</jf>;
70   * 			<jk>if</jk> (type != <jk>null</jk> && type.equals(<js>"xhtml"</js>))
71   * 				text.<jf>text</jf> = <jsm>decode</jsm>(readXmlContents(r).trim());
72   * 			<jk>else</jk>
73   * 				text.<jf>text</jf> = <jsm>decode</jsm>(r.getElementText().trim());
74   * 		}
75   *
76   * 		<ja>@Override</ja>
77   * 		<jk>public void</jk> serialize(XmlSerializerWriter w, Text text) <jk>throws</jk> Exception {
78   * 			String type = text.<jf>type</jf>;
79   * 			String content = text.<jf>text</jf>;
80   * 			<jk>if</jk> (type != <jk>null</jk> && type.equals(<js>"xhtml"</js>))
81   * 				w.encodeTextInvalidChars(content);
82   * 			<jk>else</jk>
83   * 				w.encodeText(content);
84   * 		}
85   * 	}
86   * </p>
87   *
88   * <h6 class='topic'>Notes</h6>
89   * <ul class='spaced-list'>
90   * 	<li>The {@link Xml#contentHandler()} annotation can only be specified on a bean class, or a bean property
91   * 		of format {@link XmlFormat#CONTENT}.
92   * </ul>
93   *
94   *
95   * @author James Bognar (james.bognar@salesforce.com)
96   * @param <T> The class type of the bean
97   */
98  public interface XmlContentHandler<T> {
99  
100 	/**
101 	 * Represents <jk>null</jk> on the {@link Xml#contentHandler()} annotation.
102 	 */
103 	public static interface NULL extends XmlContentHandler<Object> {}
104 
105 	/**
106 	 * Reads XML element content the specified reader and sets the appropriate value on the specified bean.
107 	 * <p>
108 	 * 	When this method is called, the attributes have already been parsed and set on the bean.
109 	 * 	Therefore, if the content handling is different based on some XML attribute (e.g.
110 	 * 		<code><xa>type</xa>=<xs>"text/xml"</xs></code> vs <code><xa>type</xa>=<xs>"text/plain"</xs></code>)
111 	 * 		then that attribute value can be obtained via the set bean property.
112 	 *
113 	 * @param r The XML stream reader.
114 	 * 	When called, the reader is positioned on the element containing the text to read.
115 	 * 	For example, calling <code>r.getElementText()</code> can be called immediately
116 	 * 	to return the element text if the element contains only characters and whitespace.
117 	 * 	However typically, the stream is going to contain XML elements that need to
118 	 * 	be handled special (otherwise you wouldn't need to use an <code>XmlContentHandler</code>
119 	 * 	to begin with).
120 	 * @param bean The bean where the parsed contents are going to be placed.
121 	 * 	Subclasses determine how the content maps to values in the bean.
122 	 * 	However, typically the contents map to a single property on the bean.
123 	 * @throws Exception If any problem occurs.  Causes parse to fail.
124 	 */
125 	public void parse(XMLStreamReader r, T bean) throws Exception;
126 
127 	/**
128 	 * Writes XML element content from values in the specified bean.
129 	 *
130 	 * @param w The XML output writer.
131 	 * 	When called, the XML element/attributes and
132 	 * 		whitespace/indentation (if enabled) have already been written to the stream.
133 	 * 	Subclasses must simply write the contents of the element.
134 	 * @param bean The bean whose values will be converted to XML content.
135 	 * @throws Exception If any problems occur.  Causes serialize to fail.
136 	 */
137 	public void serialize(XmlWriter w, T bean) throws Exception;
138 
139 }
