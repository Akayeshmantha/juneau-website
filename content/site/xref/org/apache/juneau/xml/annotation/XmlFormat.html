13  package org.apache.juneau.xml.annotation;
14  
15  /**
16   * XML format to use when serializing a POJO.
17   *
18   * @author James Bognar (james.bognar@salesforce.com)
19   */
20  public enum XmlFormat {
21  
22  	/**
23  	 * Normal formatting (default)
24  	 */
25  	NORMAL,
26  
27  	/**
28  	 * Render property as an attribute instead of an element.
29  	 * <p>
30  	 * 	Can only be applied to properties (methods/fields) of class types that can be convertible to <code>Strings</code>.
31  	 */
32  	ATTR,
33  
34  	/**
35  	 * Render property as attributes instead of an element.
36  	 * <p>
37  	 * 	Can only be applied to properties (methods/fields) of class type <code>Map&lt;Object,Object&gt;</code> where both
38  	 * 	objects are convertible to <code>Strings</code>.
39  	 */
40  	ATTRS,
41  
42  	/**
43  	 * Render property as an element instead of an attribute.
44  	 * <p>
45  	 * 	Can be applied to URL and ID bean properties that would normally be rendered as attributes.
46  	 */
47  	ELEMENT,
48  
49  	/**
50  	 * Prevents collections and arrays from being enclosed in <xt>&lt;array&gt;</xt> elements.
51  	 * <p>
52  	 * 	Can only be applied to properties (methods/fields) of type collection or array, or collection classes.
53  	 */
54  	COLLAPSED,
55  
56  	/**
57  	 * Render property value directly as content of element.
58  	 * <p>
59  	 * 	By default, content is converted to plain text.
60  	 * <p>
61  	 * 	Can be used in combination with {@link Xml#contentHandler()} to produce something other
62  	 * 	than plain text, such as embedded XML.
63  	 */
64  	CONTENT,
65  
66  	/**
67  	 * Render a collection/array as mixed child content of the element.
68  	 */
69  	MIXED
70  }
