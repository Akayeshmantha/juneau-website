13  package org.apache.juneau.transforms;
14  
15  import java.util.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.parser.*;
19  import org.apache.juneau.transform.*;
20  
21  /**
22   * Transforms {@link Calendar Calendars} to {@link Map Maps} of the format <code>{time:long,timeZone:string}</code>.
23   *
24   * @author James Bognar (james.bognar@salesforce.com)
25   */
26  @SuppressWarnings("rawtypes")
27  public class CalendarMapSwap extends PojoSwap<Calendar,Map> {
28  
29  	/**
30  	 * Converts the specified {@link Calendar} to a {@link Map}.
31  	 */
32  	@Override /* PojoSwap */
33  	public Map swap(Calendar o) {
34  		ObjectMap m = new ObjectMap();
35  		m.put("time", o.getTime().getTime());
36  		m.put("timeZone", o.getTimeZone().getID());
37  		return m;
38  	}
39  
40  	/**
41  	 * Converts the specified {@link Map} to a {@link Calendar}.
42  	 */
43  	@Override /* PojoSwap */
44  	@SuppressWarnings("unchecked")
45  	public Calendar unswap(Map o, ClassMeta<?> hint, BeanContext bc) throws ParseException {
46  		ClassMeta<? extends Calendar> tt;
47  		try {
48  			if (hint == null || ! hint.canCreateNewInstance())
49  				hint = bc.getClassMeta(GregorianCalendar.class);
50  			tt = (ClassMeta<? extends Calendar>)hint;
51  			long time = Long.parseLong(o.get("time").toString());
52  			String timeZone = o.get("timeZone").toString();
53  			Date d = new Date(time);
54  			Calendar c = tt.newInstance();
55  			c.setTime(d);
56  			c.setTimeZone(TimeZone.getTimeZone(timeZone));
57  			return c;
58  		} catch (Exception e) {
59  			throw new ParseException(e);
60  		}
61  	}
62  }
