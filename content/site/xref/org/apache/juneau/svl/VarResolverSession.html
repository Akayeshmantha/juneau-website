23  /**
24   * A var resolver session that combines a {@link VarResolver} with one or more session objects.
25   * <p>
26   * Instances of this class are considered light-weight and fast to construct, use, and discard.
27   * <p>
28   * This class contains the workhorse code for var resolution.
29   * <p>
30   * Instances of this class are created through the {@link VarResolver#createSession()} and {@link VarResolver#createSession(Map)}
31   * 	methods.
32   * <p>
33   * Instances of this class are NOT guaranteed to be thread safe.
34   *
35   * @see org.apache.juneau.svl
36   * @author James Bognar (james.bognar@salesforce.com)
37   */
38  public class VarResolverSession extends Session {
39  
40  	private final Map<String,Object> sessionObjects;
41  	private final VarResolverContext context;
42  
43  	/**
44  	 * Constructor.
45  	 *
46  	 * @param context The {@link VarResolver} context object that contains the {@link Var Vars} and
47  	 * 	context objects associated with that resolver.
48  	 * @param sessionObjects
49  	 */
50  	public VarResolverSession(VarResolverContext context, Map<String,Object> sessionObjects) {
51  		super(context);
52  		this.context = context;
53  		this.sessionObjects = (sessionObjects == null ? new TreeMap<String,Object>() : sessionObjects);
54  	}

56  	/**
57  	 * Adds a session object to this session.
58  	 *
59  	 * @param name The name of the session object.
60  	 * @param o The session object.
61  	 * @return This method (for method chaining).
62  	 */
63  	public VarResolverSession setSessionObject(String name, Object o) {
64  		this.sessionObjects.put(name, o);
65  		return this;
66  	}

68  	/**
69  	 * Resolve all variables in the specified string.
70  	 *
71  	 * @param s The string to resolve variables in.
72  	 * @return The new string with all variables resolved, or the same string if no variables were found.
73  	 * 	Null input results in a blank string.
74  	 */
75  	public String resolve(String s) {
76  
77  		if (s == null)
78  			return "";
79  		if (s.indexOf('$') == -1 && s.indexOf('\\') == -1)
80  			return s;
81  
82  		// Special case where value consists of a single variable with no embedded variables (e.g. "$X{...}").
83  		// This is a common case, so we want an optimized solution that doesn't involve string builders.
84  		if (isSimpleVar(s)) {
85  			String var = s.substring(1, s.indexOf('{'));
86  			String val = s.substring(s.indexOf('{')+1, s.length()-1);
87  			Var v = getVar(var);
88  			if (v != null) {
89  				if (v.streamed) {
90  					StringWriter sw = new StringWriter();
91  					v.resolveTo(this, sw, val);
92  					return sw.toString();
93  				}
94  				s = v.doResolve(this, val);
95  				if (s == null)
96  					s = "";
97  				return resolve(s);
98  			}
99  			return s;
100 		}
101 
102 		try {
103 			return resolveTo(s, new StringWriter()).toString();
104 		} catch (IOException e) {
105 			throw new RuntimeException(e); // Never happens.
106 		}
107 	}
108 
109 	/**
110 	 * Checks to see if string is of the simple form "$X{...}" with no embedded variables.
111 	 * This is a common case, and we can avoid using StringWriters.
112 	 */
113 	private boolean isSimpleVar(String s) {
114 		int S1 = 1;	   // Not in variable, looking for $
115 		int S2 = 2;    // Found $, Looking for {
116 		int S3 = 3;    // Found {, Looking for }
117 		int S4 = 4;    // Found }
118 
119 		int length = s.length();
120 		int state = S1;
121 		for (int i = 0; i < length; i++) {
122 			char c = s.charAt(i);
123 			if (state == S1) {
124 				if (c == '$') {
125 					state = S2;
126 				} else {
127 					return false;
128 				}
129 			} else if (state == S2) {
130 				if (c == '{') {
131 					state = S3;
132 				} else if (c < 'A' || c > 'z' || (c > 'Z' && c < 'a')) {   // False trigger "$X "
133 					return false;
134 				}
135 			} else if (state == S3) {
136 				if (c == '}')
137 					state = S4;
138 				else if (c == '{' || c == '$')
139 					return false;
140 			} else if (state == S4) {
141 				return false;
142 			}
143 		}
144 		return state == S4;
145 	}

147 	/**
148 	 * Resolves variables in the specified string and sends the output to the specified writer.
149 	 * More efficient than first parsing to a string and then serializing to the writer since this
150 	 * method doesn't need to construct a large string.
151 	 *
152 	 * @param s The string to resolve variables in.
153 	 * @param out The writer to write to.
154 	 * @return The same writer.
155 	 * @throws IOException
156 	 */
157 	public Writer resolveTo(String s, Writer out) throws IOException {
158 
159 		int S1 = 1;	   // Not in variable, looking for $
160 		int S2 = 2;    // Found $, Looking for {
161 		int S3 = 3;    // Found {, Looking for }
162 
163 		int state = S1;
164 		boolean isInEscape = false;
165 		boolean hasInternalVar = false;
166 		boolean hasInnerEscapes = false;
167 		String varType = null;
168 		String varVal = null;
169 		int x = 0, x2 = 0;
170 		int depth = 0;
171 		int length = s.length();
172 		for (int i = 0; i < length; i++) {
173 			char c = s.charAt(i);
174 			if (state == S1) {
175 				if (isInEscape) {
176 					if (c == '\\' || c == '$') {
177 						out.append(c);
178 					} else {
179 						out.append('\\').append(c);
180 					}
181 					isInEscape = false;
182 				} else if (c == '\\') {
183 					isInEscape = true;
184 				} else if (c == '$') {
185 					x = i;
186 					x2 = i;
187 					state = S2;
188 				} else {
189 					out.append(c);
190 				}
191 			} else if (state == S2) {
192 				if (isInEscape) {
193 					isInEscape = false;
194 				} else if (c == '\\') {
195 					hasInnerEscapes = true;
196 					isInEscape = true;
197 				} else if (c == '{') {
198 					varType = s.substring(x+1, i);
199 					x = i;
200 					state = S3;
201 				} else if (c < 'A' || c > 'z' || (c > 'Z' && c < 'a')) {  // False trigger "$X "
202 					if (hasInnerEscapes)
203 						out.append(unEscapeChars(s.substring(x, i+1), new char[]{'\\','{'}));
204 					else
205 						out.append(s, x, i+1);
206 					x = i + 1;
207 					state = S1;
208 					hasInnerEscapes = false;
209 				}
210 			} else if (state == S3) {
211 				if (isInEscape) {
212 					isInEscape = false;
213 				} else if (c == '\\') {
214 					isInEscape = true;
215 					hasInnerEscapes = true;
216 				} else if (c == '{') {
217 					depth++;
218 					hasInternalVar = true;
219 				} else if (c == '}') {
220 					if (depth > 0) {
221 						depth--;
222 					} else {
223 						varVal = s.substring(x+1, i);
224 						varVal = (hasInternalVar ? resolve(varVal) : varVal);
225 						Var r = getVar(varType);
226 						if (r == null) {
227 							if (hasInnerEscapes)
228 								out.append(unEscapeChars(s.substring(x2, i+1), new char[]{'\\','$','{','}'}));
229 							else
230 								out.append(s, x2, i+1);
231 							x = i+1;
232 						} else {
233 							if (r.streamed)
234 								r.resolveTo(this, out, varVal);
235 							else {
236 								String replacement = r.doResolve(this, varVal);
237 								if (replacement == null)
238 									replacement = "";
239 								// If the replacement also contains variables, replace them now.
240 								if (replacement.indexOf('$') != -1)
241 									replacement = resolve(replacement);
242 								out.append(replacement);
243 							}
244 							x = i+1;
245 						}
246 						state = 1;
247 						hasInnerEscapes = false;
248 					}
249 				}
250 			}
251 		}
252 		if (isInEscape)
253 			out.append('\\');
254 		else if (state == S2)
255 			out.append('$').append(unEscapeChars(s.substring(x+1), new char[]{'{', '\\'}));
256 		else if (state == S3)
257 			out.append('$').append(varType).append('{').append(unEscapeChars(s.substring(x+1), new char[]{'\\','$','{','}'}));
258 		return out;
259 	}

262 	/**
263 	 * Returns the session object with the specified name.
264 	 * Casts it to the specified class type for you.
265 	 *
266 	 * @param c The class type to cast to.
267 	 * @param name The name of the session object.
268 	 * @return The session object.  Never <jk>null</jk>.
269 	 * @throws RuntimeException If session object with specified name does not exist.
270 	 */
271 	@SuppressWarnings("unchecked")
272 	public <T> T getSessionObject(Class<T> c, String name) {
273 		T t = null;
274 		try {
275 			t = (T)sessionObjects.get(name);
276 			if (t == null) {
277 				sessionObjects.put(name, this.context.getContextObject(name));
278 				t = (T)sessionObjects.get(name);
279 			}
280 		} catch (Exception e) {
281 			throw new RuntimeException(format("Session object ''{0}'' or context object ''SvlContext.{0}'' could not be converted to type ''{1}''.", name, c.getName()));
282 		}
283 		if (t == null)
284 			throw new RuntimeException(format("Session object ''{0}'' or context object ''SvlContext.{0}'' not found.", name));
285 		return t;
286 	}

289 	 * Returns the {@link Var} with the specified name.
290 	 *
291 	 * @param name The var name (e.g. <js>"S"</js>).

295 		return this.context.getVars().get(name);
