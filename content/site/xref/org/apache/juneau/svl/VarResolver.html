25  /**
26   * Utility class for resolving variables of the form <js>"$X{key}"</js> in strings.
27   * <p>
28   * Variables are of the form <code>$X{key}</code>, where <code>X</code> can consist of zero or more ASCII characters.<br>
29   * 	The variable key can contain anything, even nested variables that get recursively resolved.
30   * <p>
31   * Variables are defined through the {@link #addVars(Class[])} method.
32   * <p>
33   * The {@link Var} interface defines how variables are converted to values.
34   * <p>
35   * <h6 class='topic'>Example:</h6>
36   * <p class='bcode'>
37   * 	<jk>public class</jk> SystemPropertiesVar <jk>extends</jk> SimpleVar {
38   *
39   * 		<jc>// Must have a no-arg constructor!</jc>
40   * 		<jk>public</jk> SystemPropertiesVar() {
41   * 			<jk>super</jk>(<js>"S"</js>);
42   * 		}
43   *
44   * 		<ja>@Override</ja>
45   * 		<jk>public</jk> String resolve(VarResolverSession session, String varVal) {
46   *				<jk>return</jk> System.<jsm>getProperty</jsm>(varVal);
47   * 		}
48   * 	}
49   *
50   * 	<jc>// Create a variable resolver that resolves system properties (e.g. "$S{java.home}")</jc>
51   * 	VarResolver r = <jk>new</jk> VarResolver().addVars(SystemPropertiesVar.<js>class</js>);
52   *
53   * 	<jc>// Use it!</jc>
54   * 	System.<jsf>out</jsf>.println(r.resolve(<js>"java.home is set to $S{java.home}"</js>));
55   * </p>
56   *
57   * <h6 class='topic'>Context objects</h6>
58   * <p>
59   * Var resolvers can have zero or more context objects associated with them.
60   * <p>
61   * Context objects are arbitrary objects associated with this var resolver, such as
62   * 	a {@link ConfigFile} object.
63   * They are set through the {@link #setContextObject(String, Object)} method.
64   * They can be any class type.
65   * <p>
66   * Context objects can be retrieved by {@link Var} classes through the {@link VarResolverSession#getSessionObject(Class, String)} method.
67   *
68   * <h6 class='topic'>Session objects</h6>
69   * <p>
70   * Session objects are considered more ephemeral than context objects.
71   * While a context object is unlikely to ever change, a session object may change on every
72   * 	use of the var resolver.
73   * For example, the server API defines various <code>Var</code> objects that use the <code>RestRequest</code>
74   * 	object as a session object for the duration of a single HTTP request.
75   * <p>
76   * Session objects are used by calling the {@link #createSession()} or {@link #createSession(Map)} methods to create an instance
77   * 	of a {@link VarResolverSession} object that contains {@link VarResolverSession#resolve(String)} and {@link VarResolverSession#resolveTo(String,Writer)} methods
78   * 	that are identical to {@link VarResolver#resolve(String)} and {@link VarResolver#resolveTo(String, Writer)} except that the <code>Var</code> objects
79   * 	have access to the session objects through the {@link VarResolverSession#getSessionObject(Class, String)} method.
80   * <p>
81   * Session objects are specified through either the {@link #createSession(Map)} method or the {@link VarResolverSession#setSessionObject(String, Object)} methods.
82   *
83   * <h6 class='topic'>Cloning</h6>
84   * <p>
85   * Var resolvers can be cloned by using the {@link #clone()} method.
86   * Cloning a resolver will copy it's {@link Var} class names and context objects.
87   * <p>
88   * <h6 class='topic'>Example:</h6>
89   * <p class='bcode'>
90   * 	<jc>// Create a resolver that copies the default resolver and adds $C and $ARG vars.</jc>
91   * 	VarResolver myVarResolver = VarResolver.<jsf>DEFAULT</jsf>.clone().addVars(ConfigVar.<jk>class</jk>, ArgsVar.<jk>class</jk>);
92   * </p>
93   *
94   * @see org.apache.juneau.svl
95   * @author James Bognar (james.bognar@salesforce.com)
96   */
97  public class VarResolver extends CoreApi {
98  
99  	/**
100 	 * Default string variable resolver with support for system properties and environment variables:
101 	 * <p>
102 	 * <ul>
103 	 * 	<li><code>$S{key}</code>,<code>$S{key,default}</code> - System properties.
104 	 * 	<li><code>$E{key}</code>,<code>$E{key,default}</code> - Environment variables.
105 	 * </ul>
106 	 *
107 	 * @see SystemPropertiesVar
108 	 * @see EnvVariablesVar
109 	 */
110 	public static final VarResolver DEFAULT = new VarResolver().addVars(SystemPropertiesVar.class, EnvVariablesVar.class).lock();
111 
112 	/**
113 	 * Construct an empty var resolver with no vars.
114 	 */
115 	public VarResolver() {}
116 
117 	/**
118 	 * Register new variables with this resolver.
119 	 *
120 	 * @param vars The variable resolver classes.
121 	 * 	These classes must subclass from {@link Var} and have no-arg constructors.
122 	 * @return This object (for method chaining).
123 	 */
124 	public VarResolver addVars(Class<?>...vars) {
125 		checkLock();
126 		ContextFactory cf = getContextFactory();
127 		for (Class<?> v : vars) {
128 			try {
129 				v.newInstance();
130 			} catch (InstantiationException e) {
131 				throw new UnsupportedOperationException(format("Cannot instantiate variable class {0}.  Must have a public no-arg constructor.", v.getName()));
132 			} catch (RuntimeException e) {
133 				throw e;
134 			} catch (Exception e) {
135 				throw new RuntimeException(e);
136 			}
137 			cf.addToProperty(SVL_vars, v);
138 		}
139 		return this;
140 	}
141 
142 	/**
143 	 * Associates a context object with this resolver.
144 	 *
145 	 * @param name The name of the context object.
146 	 * @param object The context object.
147 	 * @return This object (for method chaining).
148 	 */
149 	public VarResolver setContextObject(String name, Object object) {
150 		getContextFactory().putToProperty(SVL_context, name, object);
151 		return this;
152 	}
153 
154 	/**
155 	 * Creates a new resolver session with no session objects.
156 	 * <p>
157 	 * Session objects can be associated with the specified session using the {@link VarResolverSession#setSessionObject(String, Object)} method.
158 	 *
159 	 * @return A new resolver session.
160 	 */
161 	public VarResolverSession createSession() {
162 		return new VarResolverSession(getContext(VarResolverContext.class), null);
163 	}
164 
165 	/**
166 	 * Same as {@link #createSession()} except allows you to specify session objects as a map.
167 	 *
168 	 * @param sessionObjects The session objects to associate with the sessino.
169 	 * @return A new resolver session.
170 	 */
171 	public VarResolverSession createSession(Map<String,Object> sessionObjects) {
172 		return new VarResolverSession(getContext(VarResolverContext.class), sessionObjects);
173 	}
174 
175 	/**
176 	 * Resolve variables in the specified string.
177 	 * <p>
178 	 * This is a shortcut for calling <code>createSession(<jk>null</jk>).resolve(s);</code>.
179 	 * This method can only be used if the string doesn't contain variables that rely on the existence of session variables.
180 	 *
181 	 * @param s The input string.
182 	 * @return The string with variables resolved, or the same string if it doesn't contain any variables to resolve.
183 	 */
184 	public String resolve(String s) {
185 		return createSession(null).resolve(s);
186 	}
187 
188 	/**
189 	 * Resolve variables in the specified string and sends the results to the specified writer.
190 	 * <p>
191 	 * This is a shortcut for calling <code>createSession(<jk>null</jk>).resolveTo(s, w);</code>.
192 	 * This method can only be used if the string doesn't contain variables that rely on the existence of session variables.
193 	 *
194 	 * @param s The input string.
195 	 * @param w The writer to send the result to.
196 	 * @throws IOException
197 	 */
198 	public void resolveTo(String s, Writer w) throws IOException {
199 		createSession(null).resolveTo(s, w);
200 	}
201 
202 	//--------------------------------------------------------------------------------
203 	// Overridden methods
204 	//--------------------------------------------------------------------------------
205 
206 	@Override /* Lockable */
207 	public VarResolver lock() {
208 		super.lock();
209 		return this;
210 	}
211 
212 	@Override /* Lockable */
213 	public VarResolver clone() {
214 		try {
215 			return (VarResolver)super.clone();
216 		} catch (CloneNotSupportedException e) {
217 			throw new RuntimeException(e);
218 		}
219 	}
220 }
