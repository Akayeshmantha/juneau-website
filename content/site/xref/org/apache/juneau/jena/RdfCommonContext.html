236 	/**
237 	 * <b>Configuration property:</b>  RDF language.
238 	 * <p>
239 	 * <ul>
240 	 * 	<li><b>Name:</b> <js>"Rdf.language"</js>
241 	 * 	<li><b>Data type:</b> <code>String</code>
242 	 * 	<li><b>Default:</b> <js>"RDF/XML-ABBREV"</js>
243 	 * </ul>
244 	 * <p>
245 	 * 	Can be any of the following:
246 	 * <ul class='spaced-list'>
247 	 * 	<li><js>"RDF/XML"</js>
248 	 * 	<li><js>"RDF/XML-ABBREV"</js>
249 	 * 	<li><js>"N-TRIPLE"</js>
250 	 * 	<li><js>"N3"</js> - General name for the N3 writer.
251 	 * 		Will make a decision on exactly which writer to use (pretty writer, plain writer or simple writer) when created.
252 	 * 		Default is the pretty writer but can be overridden with system property	<code>com.hp.hpl.jena.n3.N3JenaWriter.writer</code>.
253 	 * 	<li><js>"N3-PP"</js> - Name of the N3 pretty writer.
254 	 * 		The pretty writer uses a frame-like layout, with prefixing, clustering like properties and embedding one-referenced bNodes.
255 	 * 	<li><js>"N3-PLAIN"</js> - Name of the N3 plain writer.
256 	 * 		The plain writer writes records by subject.
257 	 * 	<li><js>"N3-TRIPLES"</js> - Name of the N3 triples writer.
258 	 * 		This writer writes one line per statement, like N-Triples, but does N3-style prefixing.
259 	 * 	<li><js>"TURTLE"</js> -  Turtle writer.
260 	 * 		http://www.dajobe.org/2004/01/turtle/
261 	 * </ul>
262 	 */
263 	public static final String RDF_language = "Rdf.language";
