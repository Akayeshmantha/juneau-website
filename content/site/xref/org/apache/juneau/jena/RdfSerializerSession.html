29  /**
30   * Session object that lives for the duration of a single use of {@link RdfSerializer}.
31   * <p>
32   * This class is NOT thread safe.  It is meant to be discarded after one-time use.
33   *
34   * @author James Bognar (james.bognar@salesforce.com)
35   */

38  	private final String rdfLanguage;
39  	private final Namespace juneauNs, juneauBpNs;
40  	private final boolean addLiteralTypes, addRootProperty, useXmlNamespaces, looseCollections, autoDetectNamespaces;
41  	private final Property pRoot, pValue, pClass;
42  	private final Model model;

45  	private final Namespace[] namespaces;

47  	/**
48  	 * Create a new session using properties specified in the context.
49  	 *
50  	 * @param ctx The context creating this session object.
51  	 * 	The context contains all the configuration settings for this object.
52  	 * @param beanContext The bean context being used.
53  	 * @param output The output object.  See {@link JsonSerializerSession#getWriter()} for valid class types.
54  	 * @param op The override properties.
55  	 * 	These override any context properties defined in the context.
56  	 * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
57  	 */
58  	protected RdfSerializerSession(RdfSerializerContext ctx, BeanContext beanContext, Object output, ObjectMap op, Method javaMethod) {
59  		super(ctx, beanContext, output, op, javaMethod);
60  		ObjectMap jenaSettings = new ObjectMap();
61  		jenaSettings.put("rdfXml.tab", isUseIndentation() ? 2 : 0);
62  		jenaSettings.put("rdfXml.attributeQuoteChar", Character.toString(getQuoteChar()));
63  		jenaSettings.putAll(ctx.jenaSettings);
64  		if (op == null || op.isEmpty()) {
65  			this.rdfLanguage = ctx.rdfLanguage;
66  			this.juneauNs = ctx.juneauNs;
67  			this.juneauBpNs = ctx.juneauBpNs;
68  			this.addLiteralTypes = ctx.addLiteralTypes;
69  			this.addRootProperty = ctx.addRootProperty;
70  			this.collectionFormat = ctx.collectionFormat;
71  			this.looseCollections = ctx.looseCollections;
72  			this.useXmlNamespaces = ctx.useXmlNamespaces;
73  			this.autoDetectNamespaces = ctx.autoDetectNamespaces;
74  			this.namespaces = ctx.namespaces;
75  		} else {
76  			this.rdfLanguage = op.getString(RDF_language, ctx.rdfLanguage);
77  			this.juneauNs = (op.containsKey(RDF_juneauNs) ? NamespaceFactory.parseNamespace(op.get(RDF_juneauNs)) : ctx.juneauNs);
78  			this.juneauBpNs = (op.containsKey(RDF_juneauBpNs) ? NamespaceFactory.parseNamespace(op.get(RDF_juneauBpNs)) : ctx.juneauBpNs);
79  			this.addLiteralTypes = op.getBoolean(RDF_addLiteralTypes, ctx.addLiteralTypes);
80  			this.addRootProperty = op.getBoolean(RDF_addRootProperty, ctx.addRootProperty);
81  			for (Map.Entry<String,Object> e : op.entrySet()) {
82  				String key = e.getKey();
83  				if (key.startsWith("Rdf.jena."))
84  					jenaSettings.put(key.substring(9), e.getValue());
85  			}
86  			this.collectionFormat = RdfCollectionFormat.valueOf(op.getString(RDF_collectionFormat, "DEFAULT"));
87  			this.looseCollections = op.getBoolean(RDF_looseCollections, ctx.looseCollections);
88  			this.useXmlNamespaces = op.getBoolean(RDF_useXmlNamespaces, ctx.useXmlNamespaces);
89  			this.autoDetectNamespaces = op.getBoolean(RDF_autoDetectNamespaces, ctx.autoDetectNamespaces);
90  			this.namespaces = op.get(Namespace[].class, RDF_namespaces, ctx.namespaces);
91  		}
92  		this.model = ModelFactory.createDefaultModel();
93  		addModelPrefix(juneauNs);
94  		addModelPrefix(juneauBpNs);
95  		for (Namespace ns : this.namespaces)
96  			addModelPrefix(ns);
97  		this.pRoot = model.createProperty(juneauNs.getUri(), RDF_juneauNs_ROOT);
98  		this.pValue = model.createProperty(juneauNs.getUri(), RDF_juneauNs_VALUE);
99  		this.pClass = model.createProperty(juneauNs.getUri(), RDF_juneauNs_CLASS);
100 		writer = model.getWriter(rdfLanguage);
101 
102 		// Only apply properties with this prefix!
103 		String propPrefix = RdfCommonContext.LANG_PROP_MAP.get(rdfLanguage);
104 		if (propPrefix == null)
105 			throw new RuntimeException("Unknown RDF language encountered: '"+rdfLanguage+"'");
106 
107 		for (Map.Entry<String,Object> e : jenaSettings.entrySet())
108 			if (e.getKey().startsWith(propPrefix))
109 				writer.setProperty(e.getKey().substring(propPrefix.length()), e.getValue());
110 	}

112 	/**
113 	 * Adds the specified namespace as a model prefix.
114 	 *
115 	 * @param ns The XML namespace.
116 	 */
117 	public void addModelPrefix(Namespace ns) {
118 		model.setNsPrefix(ns.getName(), ns.getUri());
119 	}
