30  /**
31   * Parses RDF into POJOs.
32   *
33   *
34   * <h6 class='topic'>Configurable properties</h6>
35   * <p>
36   * 	Refer to <a class='doclink' href='package-summary.html#ParserConfigurableProperties'>Configurable Properties</a>
37   * 		for the entire list of configurable properties.
38   *
39   *
40   * <h6 class='topic'>Behavior-specific subclasses</h6>
41   * <p>
42   * 	The following direct subclasses are provided for language-specific parsers:
43   * <ul class='spaced-list'>
44   * 	<li>{@link RdfParser.Xml} - RDF/XML and RDF/XML-ABBREV.
45   * 	<li>{@link RdfParser.NTriple} - N-TRIPLE.
46   * 	<li>{@link RdfParser.Turtle} - TURTLE.
47   * 	<li>{@link RdfParser.N3} - N3.
48   * </ul>
49   *
50   *
51   * <h6 class='topic'>Additional Information</h6>
52   * <p>
53   * 	See <a class='doclink' href='package-summary.html#TOC'>RDF Overview</a> for an overview of RDF support in Juneau.
54   *
55   *
56   * @author James Bognar (james.bognar@salesforce.com)
57   */

111 	@SuppressWarnings({"unchecked", "rawtypes"})
112 	@Override /* ReaderParser */
113 	protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
114 
115 		RdfParserSession s = (RdfParserSession)session;
116 
117 		type = session.getBeanContext().normalizeClassMeta(type);
118 
119 		Model model = s.getModel();
120 		RDFReader r = s.getRdfReader();
121 		r.read(model, session.getReader(), null);
122 		BeanContext bc = session.getBeanContext();
123 
124 		List<Resource> roots = getRoots(s, model);
125 
126 		// Special case where we're parsing a loose collection of resources.
127 		if (s.isLooseCollections() && (type.isCollection() || type.isArray())) {
128 			Collection c = null;
129 			if (type.isArray())
130 				c = new ArrayList();
131 			else
132 				c = (type.canCreateNewInstance(session.getOuter()) ? (Collection<?>)type.newInstance(session.getOuter()) : new ObjectList(bc));
133 			for (Resource resource : roots)
134 				c.add(parseAnything(s, type.getElementType(), resource, session.getOuter()));
135 
136 			if (type.isArray())
137 				return (T)bc.toArray(type, c);
138 			return (T)c;
139 		}
140 
141 		if (roots.isEmpty())
142 			return null;
143 		if (roots.size() > 1)
144 			throw new ParseException(session, "Too many root nodes found in model:  {0}", roots.size());
145 		Resource resource = roots.get(0);
146 
147 		return parseAnything(s, type, resource, session.getOuter());
148 	}

150 	/*
151 	 * Finds the roots in the model using either the "root" property to identify it,
152 	 * 	or by resorting to scanning the model for all nodes with no incoming predicates.
153 	 */
154 	private List<Resource> getRoots(RdfParserSession session, Model m) {
155 		List<Resource> l = new LinkedList<Resource>();
156 
157 		// First try to find the root using the "http://www.ibm.com/juneau/root" property.
158 		Property root = m.createProperty(session.getJuneauNsUri(), RDF_juneauNs_ROOT);
159 		for (ResIterator i  = m.listResourcesWithProperty(root); i.hasNext();)
160 			l.add(i.next());
161 
162 		if (! l.isEmpty())
163 			return l;
164 
165 		// Otherwise, we need to find all resources that aren't objects.
166 		// We want to explicitly ignore statements where the subject
167 		// and object are the same node.
168 		Set<RDFNode> objects = new HashSet<RDFNode>();
169 		for (StmtIterator i = m.listStatements(); i.hasNext();) {
170 			Statement st = i.next();
171 			RDFNode subject = st.getSubject();
172 			RDFNode object = st.getObject();
173 			if (object.isResource() && ! object.equals(subject))
174 				objects.add(object);
175 		}
176 		for (ResIterator i = m.listSubjects(); i.hasNext();) {
177 			Resource r = i.next();
178 			if (! objects.contains(r))
179 				l.add(r);
180 		}
181 		return l;
182 	}

184 	private <T> BeanMap<T> parseIntoBeanMap(RdfParserSession session, Resource r2, BeanMap<T> m) throws Exception {
185 		BeanMeta<T> bm = m.getMeta();
186 		RdfBeanMeta rbm = bm.getExtendedMeta(RdfBeanMeta.class);
187 		if (rbm.hasBeanUri() && r2.getURI() != null)
188 			rbm.getBeanUriProperty().set(m, r2.getURI());
189 		Property subTypeIdProperty = null;
190 		BeanPropertyMeta stp = bm.getSubTypeProperty();
191 		if (stp != null) {
192 			subTypeIdProperty = session.getProperty(stp.getName());
193 			Statement st = r2.getProperty(subTypeIdProperty);
194 			if (st == null)
195 				throw new ParseException(session, "Could not find subtype ID property for bean of type ''{0}''", m.getClassMeta());
196 			String subTypeId = st.getLiteral().getString();
197 			stp.set(m, subTypeId);
198 		}
199 		for (StmtIterator i = r2.listProperties(); i.hasNext();) {
200 			Statement st = i.next();
201 			Property p = st.getPredicate();
202 			if (p.equals(subTypeIdProperty))
203 				continue;
204 			String key = session.decodeString(p.getLocalName());
205 			BeanPropertyMeta pMeta = m.getPropertyMeta(key);
206 			session.setCurrentProperty(pMeta);
207 			if (pMeta != null) {
208 				RDFNode o = st.getObject();
209 				ClassMeta<?> cm = pMeta.getClassMeta();
210 				if ((cm.isArray() || cm.isCollection()) && isMultiValuedCollections(session, pMeta)) {
211 					ClassMeta<?> et = cm.getElementType();
212 					Object value = parseAnything(session, et, o, m.getBean(false));
213 					setName(et, value, key);
214 					pMeta.add(m, value);
215 				} else {
216 					Object value = parseAnything(session, cm, o, m.getBean(false));
217 					setName(cm, value, key);
218 					pMeta.set(m, value);
219 				}
220 			} else if (! (p.equals(session.getRootProperty()) || p.equals(session.getClassProperty()) || p.equals(subTypeIdProperty))) {
221 				if (bm.isSubTyped()) {
222 					RDFNode o = st.getObject();
223 					Object value = parseAnything(session, object(), o, m.getBean(false));
224 					m.put(key, value);
225 				} else {
226 					onUnknownProperty(session, key, m, -1, -1);
227 				}
228 			}
229 			session.setCurrentProperty(null);
230 		}
231 		return m;
232 	}

234 	private boolean isMultiValuedCollections(RdfParserSession session, BeanPropertyMeta pMeta) {
235 		if (pMeta != null && pMeta.getExtendedMeta(RdfBeanPropertyMeta.class).getCollectionFormat() != RdfCollectionFormat.DEFAULT)
236 			return pMeta.getExtendedMeta(RdfBeanPropertyMeta.class).getCollectionFormat() == RdfCollectionFormat.MULTI_VALUED;
237 		return session.getCollectionFormat() == RdfCollectionFormat.MULTI_VALUED;
238 	}

240 	@SuppressWarnings({ "unchecked", "rawtypes" })
241 	private <T> T parseAnything(RdfParserSession session, ClassMeta<T> eType, RDFNode n, Object outer) throws Exception {
242 
243 		BeanContext bc = session.getBeanContext();
244 
245 		if (eType == null)
246 			eType = (ClassMeta<T>)object();
247 		PojoSwap<T,Object> transform = (PojoSwap<T,Object>)eType.getPojoSwap();
248 		ClassMeta<?> sType = eType.getSerializedClassMeta();
249 		session.setCurrentClass(sType);
250 
251 		if (! sType.canCreateNewInstance(outer)) {
252 			if (n.isResource()) {
253 				Statement st = n.asResource().getProperty(session.getClassProperty());
254 				if (st != null) {
255  					String c = st.getLiteral().getString();
256 					sType = eType = (ClassMeta<T>)bc.getClassMetaFromString(c);
257 				}
258 			}
259 		}
260 
261 		Object o = null;
262 		if (n.isResource() && n.asResource().getURI() != null && n.asResource().getURI().equals(RDF_NIL)) {
263 			// Do nothing.  Leave o == null.
264 		} else if (sType.isObject()) {
265 			if (n.isLiteral()) {
266 				o = n.asLiteral().getValue();
267 				if (o instanceof String) {
268 					o = session.decodeString(o);
269 				}
270 			}
271 			else if (n.isResource()) {
272 				Resource r = n.asResource();
273 				if (session.wasAlreadyProcessed(r))
274 					o = r.getURI();
275 				else if (r.getProperty(session.getValueProperty()) != null) {
276 					o = parseAnything(session, object(), n.asResource().getProperty(session.getValueProperty()).getObject(), outer);
277 				} else if (isSeq(session, r)) {
278 					o = new ObjectList(bc);
279 					parseIntoCollection(session, r.as(Seq.class), (Collection)o, sType.getElementType());
280 				} else if (isBag(session, r)) {
281 					o = new ObjectList(bc);
282 					parseIntoCollection(session, r.as(Bag.class), (Collection)o, sType.getElementType());
283 				} else if (r.canAs(RDFList.class)) {
284 					o = new ObjectList(bc);
285 					parseIntoCollection(session, r.as(RDFList.class), (Collection)o, sType.getElementType());
286 				} else {
287 					// If it has a URI and no child properties, we interpret this as an
288 					// external resource, and convert it to just a URL.
289 					String uri = r.getURI();
290 					if (uri != null && ! r.listProperties().hasNext()) {
291 						o = r.getURI();
292 					} else {
293 						o = new ObjectMap(bc);
294 						parseIntoMap(session, r, (Map)o, null, null);
295 					}
296 				}
297 			} else {
298 				throw new ParseException(session, "Unrecognized node type ''{0}'' for object", n);
299 			}
300 		} else if (sType.isBoolean()) {
301 			o = bc.convertToType(getValue(session, n, outer), boolean.class);
302 		} else if (sType.isCharSequence()) {
303 			o = session.decodeString(getValue(session, n, outer));
304 		} else if (sType.isChar()) {
305 			o = session.decodeString(getValue(session, n, outer)).charAt(0);
306 		} else if (sType.isNumber()) {
307 			o = parseNumber(getValue(session, n, outer).toString(), (Class<? extends Number>)sType.getInnerClass());
308 		} else if (sType.isMap()) {
309 			Resource r = n.asResource();
310 			if (session.wasAlreadyProcessed(r))
311 				return null;
312 			Map m = (sType.canCreateNewInstance(outer) ? (Map)sType.newInstance(outer) : new ObjectMap(bc));
313 			o = parseIntoMap(session, r, m, eType.getKeyType(), eType.getValueType());
314 		} else if (sType.isCollection() || sType.isArray()) {
315 			if (sType.isArray())
316 				o = new ArrayList();
317 			else
318 				o = (sType.canCreateNewInstance(outer) ? (Collection<?>)sType.newInstance(outer) : new ObjectList(bc));
319 			Resource r = n.asResource();
320 			if (session.wasAlreadyProcessed(r))
321 				return null;
322 			if (isSeq(session, r)) {
323 				parseIntoCollection(session, r.as(Seq.class), (Collection)o, sType.getElementType());
324 			} else if (isBag(session, r)) {
325 				parseIntoCollection(session, r.as(Bag.class), (Collection)o, sType.getElementType());
326 			} else if (r.canAs(RDFList.class)) {
327 				parseIntoCollection(session, r.as(RDFList.class), (Collection)o, sType.getElementType());
328 			} else {
329 				throw new ParseException("Unrecognized node type ''{0}'' for collection", n);
330 			}
331 			if (sType.isArray())
332 				o = bc.toArray(sType, (Collection)o);
333 		} else if (sType.canCreateNewInstanceFromObjectMap(outer)) {
334 			Resource r = n.asResource();
335 			if (session.wasAlreadyProcessed(r))
336 				return null;
337 			Map m = new ObjectMap(bc);
338 			parseIntoMap(session, r, m, eType.getKeyType(), eType.getValueType());
339 			o = sType.newInstanceFromObjectMap(outer, (ObjectMap)m);
340 		} else if (sType.canCreateNewBean(outer)) {
341 			Resource r = n.asResource();
342 			if (session.wasAlreadyProcessed(r))
343 				return null;
344 			BeanMap<?> bm = bc.newBeanMap(outer, sType.getInnerClass());
345 			o = parseIntoBeanMap(session, r, bm).getBean();
346 		} else if (sType.isUri() && n.isResource()) {
347 			o = sType.newInstanceFromString(outer, session.decodeString(n.asResource().getURI()));
348 		} else if (sType.canCreateNewInstanceFromString(outer)) {
349 			o = sType.newInstanceFromString(outer, session.decodeString(getValue(session, n, outer)));
350 		} else if (sType.canCreateNewInstanceFromNumber(outer)) {
351 			o = sType.newInstanceFromNumber(outer, parseNumber(getValue(session, n, outer).toString(), sType.getNewInstanceFromNumberClass()));
352 		} else {
353 			throw new ParseException("Class ''{0}'' could not be instantiated.  Reason: ''{1}''", sType.getInnerClass().getName(), sType.getNotABeanReason());
354 		}
355 
356 		if (transform != null && o != null)
357 			o = transform.unswap(o, eType, bc);
358 
359 		if (outer != null)
360 			setParent(eType, o, outer);
361 
362 		return (T)o;
363 	}

365 	private boolean isSeq(RdfParserSession session, RDFNode n) {
366 		if (n.isResource()) {
367 			Statement st = n.asResource().getProperty(session.getTypeProperty());
368 			if (st != null)
369 				return RDF_SEQ.equals(st.getResource().getURI());
370 		}
371 		return false;
372 	}

374 	private boolean isBag(RdfParserSession session, RDFNode n) {
375 		if (n.isResource()) {
376 			Statement st = n.asResource().getProperty(session.getTypeProperty());
377 			if (st != null)
378 				return RDF_BAG.equals(st.getResource().getURI());
379 		}
380 		return false;
381 	}

383 	private Object getValue(RdfParserSession session, RDFNode n, Object outer) throws Exception {
384 		if (n.isLiteral())
385 			return n.asLiteral().getValue();
386 		if (n.isResource()) {
387 			Statement st = n.asResource().getProperty(session.getValueProperty());
388 			if (st != null) {
389 				n = st.getObject();
390 				if (n.isLiteral())
391 					return n.asLiteral().getValue();
392 				return parseAnything(session, object(), st.getObject(), outer);
393 			}
394 		}
395 		throw new ParseException(session, "Unknown value type for node ''{0}''", n);
396 	}

398 	private <K,V> Map<K,V> parseIntoMap(RdfParserSession session, Resource r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType) throws Exception {
399 		// Add URI as "uri" to generic maps.
400 		if (r.getURI() != null) {
401 			K uri = convertAttrToType(session, m, "uri", keyType);
402 			V value = convertAttrToType(session, m, r.getURI(), valueType);
403 			m.put(uri, value);
404 		}
405 		for (StmtIterator i = r.listProperties(); i.hasNext();) {
406 			Statement st = i.next();
407 			Property p = st.getPredicate();
408 			String key = p.getLocalName();
409 			if (! (key.equals("root") && p.getURI().equals(session.getJuneauNsUri()))) {
410 				key = session.decodeString(key);
411 				RDFNode o = st.getObject();
412 				K key2 = convertAttrToType(session, m, key, keyType);
413 				V value = parseAnything(session, valueType, o, m);
414 				setName(valueType, value, key);
415 				m.put(key2, value);
416 			}
417 
418 		}
419 		// TODO Auto-generated method stub
420 		return m;
421 	}

423 	private <E> Collection<E> parseIntoCollection(RdfParserSession session, Container c, Collection<E> l, ClassMeta<E> et) throws Exception {
424 		for (NodeIterator ni = c.iterator(); ni.hasNext();) {
425 			E e = parseAnything(session, et, ni.next(), l);
426 			l.add(e);
427 		}
428 		return l;
429 	}
430 
431 	private <E> Collection<E> parseIntoCollection(RdfParserSession session, RDFList list, Collection<E> l, ClassMeta<E> et) throws Exception {
432 		for (ExtendedIterator<RDFNode> ni = list.iterator(); ni.hasNext();) {
433 			E e = parseAnything(session, et, ni.next(), l);
434 			l.add(e);
435 		}
436 		return l;
437 	}
