13  package org.apache.juneau.jena;
14  
15  import java.util.*;
16  
17  import org.apache.juneau.*;
18  import org.apache.juneau.jena.annotation.*;
19  import org.apache.juneau.xml.*;
20  
21  /**
22   * Utility classes.
23   */
24  public class RdfUtils {
25  
26  	/**
27  	 * Find the namespace given a list of <ja>@Rdf</ja> and <ja>@RdfSchema</ja> annotations.
28  	 * The annotations should be a child-to-parent ordering of annotations found on
29  	 * 	a class or method.
30  	 *
31  	 * @param rdfs The <code>@Rdf</code> annotations to search.
32  	 * @param schemas The list of known RDF schemas.
33  	 * @return The resolved namespace, or <jk>null</jk> if the namespace could not be resolved.
34  	 */
35  	public static Namespace findNamespace(List<Rdf> rdfs, List<RdfSchema> schemas) {
36  
37  		for (Rdf rdf : rdfs) {
38  			Namespace ns = findNamespace(rdf.prefix(), rdf.namespace(), rdfs, schemas);
39  			if (ns != null)
40  				return ns;
41  		}
42  
43  		for (RdfSchema schema : schemas) {
44  			Namespace ns = findNamespace(schema.prefix(), schema.namespace(), null, schemas);
45  			if (ns != null)
46  				return ns;
47  		}
48  
49  		return null;
50  	}
51  
52  	private static Namespace findNamespace(String prefix, String ns, List<Rdf> rdfs, List<RdfSchema> schemas) {
53  
54  		// If both prefix and namespace specified, use that Namespace mapping.
55  		if (! (prefix.isEmpty() || ns.isEmpty()))
56  			return NamespaceFactory.get(prefix, ns);
57  
58  		// If only prefix specified, need to search for namespaceURI.
59  		if (! prefix.isEmpty()) {
60  			if (rdfs != null)
61  				for (Rdf rdf2 : rdfs)
62  					if (rdf2.prefix().equals(prefix) && ! rdf2.namespace().isEmpty())
63  						return NamespaceFactory.get(prefix, rdf2.namespace());
64  			for (RdfSchema schema : schemas) {
65  				if (schema.prefix().equals(prefix) && ! schema.namespace().isEmpty())
66  					return NamespaceFactory.get(prefix, schema.namespace());
67  				for (RdfNs rdfNs : schema.rdfNs())
68  					if (rdfNs.prefix().equals(prefix))
69  						return NamespaceFactory.get(prefix, rdfNs.namespaceURI());
70  			}
71  			throw new BeanRuntimeException("Found @Rdf.prefix annotation with no matching URI.  prefix='"+prefix+"'");
72  		}
73  
74  		// If only namespaceURI specified, need to search for prefix.
75  		if (! ns.isEmpty()) {
76  			if (rdfs != null)
77  				for (Rdf rdf2 : rdfs)
78  					if (rdf2.namespace().equals(ns) && ! rdf2.prefix().isEmpty())
79  						return NamespaceFactory.get(rdf2.prefix(), ns);
80  			for (RdfSchema schema : schemas) {
81  				if (schema.namespace().equals(ns) && ! schema.prefix().isEmpty())
82  					return NamespaceFactory.get(schema.prefix(), ns);
83  				for (RdfNs rdfNs : schema.rdfNs())
84  					if (rdfNs.namespaceURI().equals(ns))
85  						return NamespaceFactory.get(rdfNs.prefix(), ns);
86  			}
87  		}
88  
89  		return null;
90  	}
91  }
