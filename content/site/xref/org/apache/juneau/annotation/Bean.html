13  package org.apache.juneau.annotation;
14  
15  import static java.lang.annotation.ElementType.*;
16  import static java.lang.annotation.RetentionPolicy.*;
17  
18  import java.beans.*;
19  import java.lang.annotation.*;
20  
21  import org.apache.juneau.*;
22  import org.apache.juneau.transform.*;
23  
24  /**
25   * Used to tailor how beans get interpreted by the framework.
26   * <p>
27   * 	Can be used to do the following:
28   * <ul class='spaced-list'>
29   * 	<li>Explicitly specify the set and order of properties on a bean.
30   * 	<li>Associate a {@link PropertyNamer} with a class.
31   * 	<li>Specify subtypes of a bean differentiated by a sub type property.
32   * </ul>
33   * <p>
34   * 	This annotation can be applied to classes and interfaces.
35   *
36   * @author James Bognar (james.bognar@salesforce.com)
37   */
38  @Documented
39  @Target(TYPE)
40  @Retention(RUNTIME)
41  @Inherited
42  public @interface Bean {
43  
44  	/**
45  	 * An identifying name for this class.
46  	 * <p>
47  	 * The name is used to identify the class type during parsing when it cannot be inferred through reflection.
48  	 * For example, if a bean property is of type <code>Object</code>, then the serializer will add the name to the
49  	 * 	output so that the class can be determined during parsing.
50  	 * It is also used to specify element names in XML.
51  	 * <p>
52  	 * The name is used in combination with the bean dictionary defined through {@link BeanProperty#beanDictionary()} or {@link BeanContext#BEAN_beanDictionary}.  Together, they make up
53  	 * 	a simple name/value mapping of names to classes.
54  	 * Names do not need to be universally unique.  However, they must be unique within a dictionary.
55  	 *
56  	 * <dl>
57  	 * 	<dt>Example:</dt>
58  	 * 	<dd>
59  	 * 		<p class='bcode'>
60  	 * 	<ja>@Bean</ja>(typeName=<js>"foo"</js>)
61  	 * 	<jk>public class</jk> Foo {
62  	 * 		<jc>// A bean property where the object types cannot be inferred since it's an Object[].</jc>
63  	 * 	   <ja>@BeanProperty</ja>(typeDictionary={Bar.<jk>class</jk>,Baz.<jk>class</jk>})
64  	 * 	   <jk>public</jk> Object[] x = <jk>new</jk> Object[]{<jk>new</jk> Bar(), <jk>new</jk> Baz()};
65  	 * 	}
66  	 *
67  	 * 	<ja>@Bean</ja>(typeName=<js>"bar"</js>)
68  	 * 	<jk>public class</jk> Bar {}
69  	 *
70  	 * 	<ja>@Bean</ja>(typeName=<js>"baz"</js>)
71  	 * 	<jk>public class</jk> Baz {}
72  	 * 		</p>
73  	 * 		<p>
74  	 * 			When serialized as XML, the bean is rendered as:
75  	 * 		</p>
76  	 * 		<p class='bcode'>
77  	 * 	<xt>&lt;foo&gt;</xt>
78  	 * 	   <xt>&lt;x&gt;</xt>
79  	 * 	      <xt>&lt;bar/&gt;v
80  	 * 	      <xt>&lt;baz/&gt;</xt>
81  	 * 	   <xt>&lt;/x&gt;</xt>
82  	 * 	<xt>&lt;/foo&gt;</xt>
83  	 * 		</p>
84  	 * 		<p>
85  	 * 			When serialized as JSON, <js>'n'</js> attributes would be added when needed to infer the type during parsing:
86  	 * 		</p>
87  	 * 		<p class='bcode'>
88  	 * 	{
89  	 * 	   <jsa>x</jsa>: [
90  	 * 	      {<jsa>_type</jsa>:<jss>'bar'</jss>},
91  	 * 	      {<jsa>_type</jsa>:<jss>'baz'</jss>}
92  	 * 	   ]
93  	 * 	}	 *
94  	 * 	</dd>
95  	 * </dl>
96  	 */
97  	String typeName() default "";
98  
99  	/**
100 	 * The set and order of names of properties associated with a bean class.
101 	 * <p>
102 	 * 	The order specified is the same order that the entries will be returned by the {@link BeanMap#entrySet()} and related methods.
103 	 * <p>
104 	 * 	This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getProperties()} method.
105 	 *
106 	 * <dl>
107 	 * 	<dt>Example:</dt>
108 	 * 	<dd>
109 	 * 		<p class='bcode'>
110 	 * 	<jc>// Address class with only street/city/state properties (in that order).</jc>
111 	 * 	<jc>// All other properties are ignored.</jc>
112 	 * 	<ja>@Bean</ja>(properties={<js>"street"</js>,<js>"city"</js>,<js>"state"</js>})
113 	 * 	<jk>public class</jk> Address {
114 	 * 	...
115 	 * 		</p>
116 	 * 	</dd>
117 	 * </dl>
118 	 */
119 	String[] properties() default {};
120 
121 	/**
122 	 * Sort bean properties in alphabetical order.
123 	 * <p>
124 	 * 	When <jk>true</jk>, all bean properties will be serialized and access in alphabetical order.
125 	 * 	Otherwise, the natural order of the bean properties is used which is dependent on the
126 	 * 		JVM vendor.
127 	 * 	On IBM JVMs, the bean properties are ordered based on their ordering in the Java file.
128 	 * 	On Oracle JVMs, the bean properties are not ordered (which follows the offical JVM specs).
129 	 * <p>
130 	 * 	This property is disabled by default so that IBM JVM users don't have to use {@link Bean @Bean} annotations
131 	 * 	to force bean properties to be in a particular order and can just alter the order of the fields/methods
132 	 * 	in the Java file.
133 	 * <p>
134 	 * 	This annotation is equivalent to using the {@link BeanContext#BEAN_sortProperties} property, but
135 	 * 	applied to individual classes instead of globally at the serializer or parser level.
136 	 */
137 	boolean sort() default false;
138 
139 	/**
140 	 * Specifies a list of properties that should be excluded from {@link BeanMap#entrySet()}.
141 	 * <p>
142 	 * 	This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getExcludeProperties()} method.
143 	 *
144 	 * <dl>
145 	 * 	<dt>Example:</dt>
146 	 * 	<dd>
147 	 * 		<p class='bcode'>
148 	 * 	<jc>// Address class with only street/city/state properties (in that order).</jc>
149 	 * 	<jc>// All other properties are ignored.</jc>
150 	 * 	<ja>@Bean</ja>(excludeProperties={<js>"city"</js>,<js>"state"</js>})
151 	 * 	<jk>public class</jk> Address {
152 	 * 		...
153 	 * 		</p>
154 	 * 	</dd>
155 	 * </dl>
156 	 */
157 	String[] excludeProperties() default {};
158 
159 	/**
160 	 * Associates a {@link PropertyNamer} with this bean to tailor the names of the bean properties.
161 	 * <p>
162 	 * 	Property namers are used to transform bean property names from standard form to some other form.
163 	 * 	For example, the {@link PropertyNamerDashedLC} will convert property names to dashed-lowercase, and
164 	 * 		these will be used as attribute names in JSON, and element names in XML.
165 	 * <p>
166 	 * 	This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getPropertyNamer()} method.
167 	 *
168 	 * <dl>
169 	 * 	<dt>Example:</dt>
170 	 * 	<dd>
171 	 * 		<p class='bcode'>
172 	 * 	<jc>// Define a class with dashed-lowercase property names.</jc>
173 	 * 	<ja>@Bean</ja>(propertyNamer=PropertyNamerDashedLC.<jk>class</jk>)
174 	 * 	<jk>public class</jk> MyClass {
175 	 * 		...
176 	 * 	}
177 	 * 		</p>
178 	 * 	</dd>
179 	 * </dl>
180 	 */
181 	Class<? extends PropertyNamer> propertyNamer() default PropertyNamerDefault.class;
182 
183 	/**
184 	 * Defines a virtual property on a superclass that identifies bean subtype classes.
185 	 * <p>
186 	 * 	In the following example, the abstract class has two subclasses that are differentiated
187 	 * 		by a property called <code>subType</code>
188 	 * <p class='bcode'>
189 	 * 	<jc>// Abstract superclass</jc>
190 	 * 	<ja>@Bean</ja>(
191 	 * 		subTypeProperty=<js>"subType"</js>,
192 	 * 		subTypes={A1.class, A2.class}
193 	 * 	)
194 	 * 	<jk>public class</jk> A {
195 	 * 		<jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
196 	 * 	}
197 	 *
198 	 * 	<jc>// Subclass 1</jc>
199 	 * 	<ja>@Bean</ja>(typeName=<js>"A1"</js>)
200 	 * 	<jk>public class</jk> A1 <jk>extends</jk> A {
201 	 * 		<jk>public</jk> String <jf>f1</jf>;
202 	 * 	}
203 	 *
204 	 * 	<jc>// Subclass 2</jc>
205 	 * 	<ja>@Bean</ja>(typeName=<js>"A2"</js>)
206 	 * 	<jk>public class</jk> A2 <jk>extends</jk> A {
207 	 * 		<jk>public</jk> String <jf>f2</jf>;
208 	 * 	}
209 	 * <p>
210 	 * 	The following shows what happens when serializing a subclassed object to JSON:
211 	 * <p class='bcode'>
212 	 * 	JsonSerializer s = JsonSerializer.<jsf>DEFAULT_LAX</jsf>;
213 	 * 	A1 a1 = <jk>new</jk> A1();
214 	 * 	a1.<jf>f1</jf> = <js>"f1"</js>;
215 	 * 	String r = s.serialize(a1);
216 	 * 	<jsm>assertEquals</jsm>(<js>"{subType:'A1',f1:'f1',f0:'f0'}"</js>, r);
217 	 * </p>
218 	 * <p>
219 	 * 	The following shows what happens when parsing back into the original object.
220 	 * <p>
221 	 * <p class='bcode'>
222 	 * 	JsonParser p = JsonParser.<jsf>DEFAULT</jsf>;
223 	 * 	A a = p.parse(r, A.<jk>class</jk>);
224 	 * 	<jsm>assertTrue</jsm>(a <jk>instanceof</jk> A1);
225 	 * </p>
226 	 * <p>
227 	 * 	This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getSubTypeProperty()} method.
228 	 */
229 	String subTypeProperty() default "_subtype";
230 
231 	/**
232 	 * Used in conjunction with {@link #subTypeProperty()} to set up bean subtypes.
233 	 */
234 	Class<?>[] subTypes() default {};
235 
236 	/**
237 	 * Identifies a class to be used as the interface class for this and all subclasses.
238 	 * <p>
239 	 * 	When specified, only the list of properties defined on the interface class will be used during serialization.
240 	 * 	Additional properties on subclasses will be ignored.
241 	 * <p class='bcode'>
242 	 * 	<jc>// Parent class</jc>
243 	 * 	<ja>@Bean</ja>(interfaceClass=A.<jk>class</jk>)
244 	 * 	<jk>public abstract class</jk> A {
245 	 * 		<jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
246 	 * 	}
247 	 *
248 	 * 	<jc>// Sub class</jc>
249 	 * 	<jk>public class</jk> A1 <jk>extends</jk> A {
250 	 * 		<jk>public</jk> String <jf>f1</jf> = <js>"f1"</js>;
251 	 * 	}
252 	 *
253 	 * 	JsonSerializer s = JsonSerializer.<jsf>DEFAULT_LAX</jsf>;
254 	 * 	A1 a1 = <jk>new</jk> A1();
255 	 * 	String r = s.serialize(a1);
256 	 * 	<jsm>assertEquals</jsm>(<js>"{f0:'f0'}"</js>, r);  // Note f1 is not serialized.
257 	 * </p>
258 	 * <p>
259 	 * 	Note that this annotation can be used on the parent class so that it filters to all child classes,
260 	 * 		or can be set individually on the child classes.
261 	 * <p>
262 	 * 	This annotation is an alternative to using the {@link BeanFilter} class with an implemented {@link BeanFilter#getInterfaceClass()} method.
263 	 */
264 	Class<?> interfaceClass() default Object.class;
265 
266 	/**
267 	 * Identifies a stop class for the annotated class.
268 	 * <p>
269 	 * Identical in purpose to the stop class specified by {@link Introspector#getBeanInfo(Class, Class)}.
270 	 * Any properties in the stop class or in its baseclasses will be ignored during analysis.
271 	 * <p>
272 	 * For example, in the following class hierarchy, instances of <code>C3</code> will include property <code>p3</code>, but
273 	 * 	not <code>p1</code> or <code>p2</code>.
274 	 * <p class='bcode'>
275 	 * 	<jk>public class</jk> C1 {
276 	 * 		<jk>public int</jk> getP1();
277 	 * 	}
278 	 *
279 	 * 	<jk>public class</jk> C2 <jk>extends</jk> C1 {
280 	 * 		<jk>public int</jk> getP2();
281 	 * 	}
282 	 *
283 	 * 	<ja>@Bean</ja>(stopClass=C2.<jk>class</jk>)
284 	 * 	<jk>public class</jk> C3 <jk>extends</jk> C2 {
285 	 * 		<jk>public int</jk> getP3();
286 	 * 	}
287 	 * </p>
288 	 */
289 	Class<?> stopClass() default Object.class;
290 }
