13  package org.apache.juneau.annotation;
14  
15  import static java.lang.annotation.ElementType.*;
16  import static java.lang.annotation.RetentionPolicy.*;
17  
18  import java.lang.annotation.*;
19  
20  import org.apache.juneau.parser.*;
21  
22  /**
23   * Annotation used on subclasses of {@link Parser} to identify the media types that it consumes.
24   *
25   *
26   * <h6 class='topic'>Description</h6>
27   * <p>
28   * 	Provides a way to define the contents of {@link Parser#getMediaTypes()} through an annotation.
29   * <p>
30   * 	The {@link Parser#getMediaTypes()} default implementation gathers the media types by looking
31   * 		for this annotation.
32   * 	It should be noted that this annotation is optional and that the {@link Parser#getMediaTypes()} method can
33   * 		be overridden by subclasses to return the media types programmatically.
34   *
35   *
36   * <h6 class='topic'>Examples</h6>
37   * <p>
38   * 	Standard example:
39   * <p class='bcode'>
40   * 	<ja>@Consumes</ja>({<js>"application/json"</js>,<js>"text/json"</js>})
41   * 	<jk>public class</jk> JsonParser <jk>extends</jk> ReaderParser {...}
42   * </p>
43   * <p>
44   * 	The media types can also be <code>media-range</code> values per
45   * 		<a href='http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1'>RFC2616/14.1</a>.
46   * <p class='bcode'>
47   * 	<jc>// Consumes any text</jc>
48   * 	<ja>@Consumes</ja>({<js>"text\/*"</js>})
49   * 	<jk>public class</jk> AnythingParser <jk>extends</jk> ReaderParser {...}
50   *
51   * 	<jc>// Consumes anything</jc>
52   * 	<ja>@Consumes</ja>({<js>"*\/*"</js>})
53   * 	<jk>public class</jk> AnythingParser <jk>extends</jk> ReaderParser {...}
54   * </p>
55   *
56   *
57   * @author James Bognar (james.bognar@salesforce.com)
58   */
59  @Documented
60  @Target(TYPE)
61  @Retention(RUNTIME)
62  @Inherited
63  public @interface Consumes {
64  
65  	/**
66  	 * The media types that the parser can handle.
67  	 * <p>
68  	 * 	Can contain meta-characters per the <code>media-type</code> specification of
69  	 * 	<a href='http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.1'>RFC2616/14.1</a>
70  	 * @return The media types that the parser can handle.
71  	 */
72  	String[] value() default {};
73  }
