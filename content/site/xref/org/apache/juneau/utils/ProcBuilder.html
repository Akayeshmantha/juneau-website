23  /**
24   * Utility class for running operating system processes.
25   * <p>
26   * Similar to {@link java.lang.ProcessBuilder} but with additional features.
27   *
28   * @author James Bognar (james.bognar@salesforce.com)
29   */
30  @SuppressWarnings("hiding")
31  public class ProcBuilder {
32  
33  	private java.lang.ProcessBuilder pb = new java.lang.ProcessBuilder();
34  	private TeeWriter outWriters = new TeeWriter(), logWriters = new TeeWriter();
35  	private LineProcessor lp;
36  	private Process p;
37  	private int maxExitStatus = 0;
38  	private boolean byLines;
39  	private String divider = "--------------------------------------------------------------------------------";
40  
41  	/**
42  	 * Creates a process builder with the specified arguments.
43  	 * Equivalent to calling <code>ProcessBuilder.create().command(args);</code>
44  	 *
45  	 * @param args The command-line arguments.
46  	 * @return A new process builder.
47  	 */
48  	public static ProcBuilder create(Object...args) {
49  		return new ProcBuilder().command(args);
50  	}
51  
52  	/**
53  	 * Creates an empty process builder.
54  	 *
55  	 * @return A new process builder.
56  	 */
57  	public static ProcBuilder create() {
58  		return new ProcBuilder().command();
59  	}
60  
61  	/**
62  	 * Command arguments.
63  	 * Arguments can be collections or arrays and will be automatically expanded.
64  	 *
65  	 * @param args The command-line arguments.
66  	 * @return This object (for method chaining).
67  	 */
68  	public ProcBuilder command(Object...args) {
69  		return commandIf(ANY, args);
70  	}
71  
72  	/**
73  	 * Command arguments if the specified matcher matches.
74  	 * Can be used for specifying os-specific commands.
75  	 * Example:
76  	 * <p class='bcode'>
77  	 * 	ProcessBuilder pb = ProcessBuilder
78  	 * 		.create()
79  	 * 		.commandIf(<jsf>WINDOWS</jsf>, <js>"cmd /c dir"</js>)
80  	 * 		.commandIf(<jsf>UNIX</jsf>, <js>"bash -c ls"</js>)
81  	 * 		.merge()
82  	 * 		.execute();
83  	 * </p>
84  	 *
85  	 * @param m The matcher.
86  	 * @param args The command line arguments if matcher matches.
87  	 * @return This object (for method chaining).
88  	 */
89  	public ProcBuilder commandIf(Matcher m, Object...args) {
90  		if (m.matches())
91  			pb.command(toList(args));
92  		return this;
93  	}
94  
95  	/**
96  	 * Append to the command arguments.
97  	 * Arguments can be collections or arrays and will be automatically expanded.
98  	 *
99  	 * @param args The command-line arguments.
100 	 * @return This object (for method chaining).
101 	 */
102 	public ProcBuilder append(Object...args) {
103 		return appendIf(ANY, args);
104 	}
105 
106 	/**
107 	 * Append to the command arguments if the specified matcher matches.
108 	 * Arguments can be collections or arrays and will be automatically expanded.
109 	 *
110 	 * @param m The matcher.
111 	 * @param args The command line arguments if matcher matches.
112 	 * @return This object (for method chaining).
113 	 */
114 	public ProcBuilder appendIf(Matcher m, Object...args) {
115 		if (m.matches())
116 			pb.command().addAll(toList(args));
117 		return this;
118 	}

121 	 * Merge STDOUT and STDERR into a single stream.
122 	 *
123 	 * @return This object (for method chaining).
124 	 */
125 	public ProcBuilder merge() {
126 		pb.redirectErrorStream(true);

130 	/**
131 	 * Use by-lines mode.
132 	 * Flushes output after every line of input.
133 	 *
134 	 * @return This object (for method chaining).
135 	 */
136 	public ProcBuilder byLines() {
137 		this.byLines = true;

141 	/**
142 	 * Pipe output to the specified writer.
143 	 * The method can be called multiple times to write to multiple writers.
144 	 *
145 	 * @param w The writer to pipe to.
146 	 * @param close Close the writer afterwards.
147 	 * @return This object (for method chaining).
148 	 */
149 	public ProcBuilder pipeTo(Writer w, boolean close) {
150 		this.outWriters.add(w, close);
151 		return this;
152 	}
153 
154 	/**
155 	 * Pipe output to the specified writer, but don't close the writer.
156 	 *
157 	 * @param w The writer to pipe to.
158 	 * @return This object (for method chaining).
159 	 */
160 	public ProcBuilder pipeTo(Writer w) {
161 		return pipeTo(w, false);
162 	}
163 
164 	/**
165 	 * Pipe output to the specified writer, including the command and return code.
166 	 * The method can be called multiple times to write to multiple writers.
167 	 *
168 	 * @param w The writer to pipe to.
169 	 * @param close Close the writer afterwards.
170 	 * @return This object (for method chaining).
171 	 */
172 	public ProcBuilder logTo(Writer w, boolean close) {
173 		this.logWriters.add(w, close);
174 		this.outWriters.add(w, close);
175 		return this;
176 	}
177 
178 	/**
179 	 * Pipe output to the specified writer, including the command and return code.
180 	 * The method can be called multiple times to write to multiple writers.
181 	 * Don't close the writer afterwards.
182 	 *
183 	 * @param w The writer to pipe to.
184 	 * @return This object (for method chaining).
185 	 */
186 	public ProcBuilder logTo(Writer w) {
187 		return logTo(w, false);
188 	}
189 
190 	/**
191 	 * Pipe output to the specified writer, including the command and return code.
192 	 * The method can be called multiple times to write to multiple writers.
193 	 *
194 	 * @param level The log level.
195 	 * @param logger The logger to log to.
196 	 * @return This object (for method chaining).
197 	 */
198 	public ProcBuilder logTo(final Level level, final Logger logger) {
199 		if (logger.isLoggable(level)) {
200 			logTo(new StringWriter() {
201 				private boolean isClosed;  // Prevents messages from being written twice.
202 				@Override /* Writer */
203 				public void close() {
204 					if (! isClosed)
205 						logger.log(level, this.toString());
206 					isClosed = true;
207 				}
208 			}, true);
209 		}
210 		return this;
211 	}

224 	/**
225 	 * Append the specified environment variables to the process.
226 	 *
227 	 * @param env The new set of environment variables.
228 	 * @return This object (for method chaining).
229 	 */
230 	@SuppressWarnings({"rawtypes", "unchecked"})
231 	public ProcBuilder env(Map env) {
232 		if (env != null)
233 			for (Map.Entry e : (Set<Map.Entry>)env.entrySet())
234 				environment(e.getKey().toString(), e.getValue() == null ? null : e.getValue().toString());
235 		return this;
236 	}

261 	/**
262 	 * Sets the maximum allowed return code on the process call.
263 	 * If the return code exceeds this value, an IOException is returned on the {@link #run()} command.
264 	 * The default value is '0'.
265 	 *
266 	 * @param maxExitStatus The maximum exit status.
267 	 * @return This object (for method chaining).
268 	 */
269 	public ProcBuilder maxExitStatus(int maxExitStatus) {
270 		this.maxExitStatus = maxExitStatus;
271 		return this;
272 	}
273 
274 	/**
275 	 * Run this command and pipes the output to the specified writer or output stream.
276 	 *
277 	 * @return The exit code from the process.
278 	 * @throws IOException
279 	 * @throws InterruptedException
280 	 */
281 	public int run() throws IOException, InterruptedException {
282 		if (pb.command().size() == 0)
283 			throw new IOException("No command specified in ProcBuilder.");
284 		try {
285 			logWriters.append(divider).append('\n').flush();
286 			logWriters.append(StringUtils.join(pb.command(), " ")).append('\n').flush();
287 			p = pb.start();
288 			IOPipe.create(p.getInputStream(), outWriters).lineProcessor(lp).byLines(byLines).run();
289 			int rc = p.waitFor();
290 			logWriters.append("Exit: ").append(String.valueOf(p.exitValue())).append('\n').flush();
291 			if (rc > maxExitStatus)
292 				throw new IOException("Return code "+rc+" from command " + StringUtils.join(pb.command(), " "));
293 			return rc;
294 		} finally {
295 			close();
296 		}
297 	}
298 
299 	/**
300 	 * Run this command and returns the output as a simple string.
301 	 *
302 	 * @return The output from the command.
303 	 * @throws IOException
304 	 * @throws InterruptedException
305 	 */
306 	public String getOutput() throws IOException, InterruptedException {
307 		StringWriter sw = new StringWriter();
308 		pipeTo(sw).run();
309 		return sw.toString();
310 	}
311 
312 	/**
313 	 * Returns the output from this process as a {@link Scanner}.
314 	 *
315 	 * @return The output from the process as a Scanner object.
316 	 * @throws IOException
317 	 * @throws InterruptedException
318 	 */
319 	public Scanner getScanner() throws IOException, InterruptedException {
320 		StringWriter sw = new StringWriter();
321 		pipeTo(sw, true);
322 		run();
323 		return new Scanner(sw.toString());
324 	}
325 
326 	/**
327 	 * Destroys the underlying process.
328 	 * This method is only needed if the {@link #getScanner()} method was used.
329 	 */
330 	private void close() {
331 		IOUtils.closeQuietly(logWriters, outWriters);
332 		if (p != null)
333 			p.destroy();
334 	}

336 	/**
337 	 * Specifies interface for defining OS-specific commands.
338 	 */
339 	public abstract static class Matcher {
340 		abstract boolean matches();
341 	}
342 
343 	private static String OS = System.getProperty("os.name").toLowerCase();
344 
345 	/** Operating system matcher: Any operating system. */
346 	public final static Matcher ANY = new Matcher() {
347 		@Override boolean matches() {
348 			return true;
349 		}
350 	};
351 
352 	/** Operating system matcher: Any Windows system. */
353 	public final static Matcher WINDOWS = new Matcher() {
354 		@Override boolean matches() {
355 			return OS.indexOf("win") >= 0;
356 		}
357 	};
358 
359 	/** Operating system matcher: Any Mac system. */
360 	public final static Matcher MAC = new Matcher() {
361 		@Override boolean matches() {
362 			return OS.indexOf("mac") >= 0;
363 		}
364 	};
365 
366 	/** Operating system matcher: Any Unix or Linux system. */
367 	public final static Matcher UNIX = new Matcher() {
368 		@Override boolean matches() {
369 			return OS.indexOf("nix") >= 0 || OS.indexOf("nux") >= 0 || OS.indexOf("aix") > 0;
370 		}
371 	};
372 
373 	private static List<String> toList(Object...args) {
374 		List<String> l = new LinkedList<String>();
375 		for (Object o : args) {
376 			if (o.getClass().isArray())
377 				for (int i = 0; i < Array.getLength(o); i++)
378 					l.add(Array.get(o, i).toString());
379 			else if (o instanceof Collection)
380 				for (Object o2 : (Collection<?>)o)
381 					l.add(o2.toString());
382 			else
383 				l.add(o.toString());
384 		}
385 		return l;
386 	}
