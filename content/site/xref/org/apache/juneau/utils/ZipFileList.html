13  package org.apache.juneau.utils;
14  
15  import java.io.*;
16  import java.util.*;
17  import java.util.zip.*;
18  
19  /**
20   * Utility class for representing the contents of a zip file as a list of entries
21   * whose contents don't resolve until serialize time.
22   * <p>
23   * Generally associated with <code>RestServlets</code> using the <code>responseHandlers</code>
24   * 	annotation so that REST methods can easily create ZIP file responses by simply returning instances
25   * 	of this class.
26   */
27  @SuppressWarnings("serial")
28  public class ZipFileList extends LinkedList<ZipFileList.ZipFileEntry> {
29  
30  	/**
31  	 * The name of the zip file.
32  	 */
33  	public final String fileName;
34  
35  	/**
36  	 * Constructor.
37  	 *
38  	 * @param fileName The file name of the zip file to create.
39  	 */
40  	public ZipFileList(String fileName) {
41  		this.fileName = fileName;
42  	}
43  
44  	/**
45  	 * Add an entry to this list.
46  	 *
47  	 * @param e The zip file entry.
48  	 * @return This object (for method chaining).
49  	 */
50  	public ZipFileList append(ZipFileEntry e) {
51  		add(e);
52  		return this;
53  	}
54  
55  	/**
56  	 * Interface for ZipFileList entries.
57  	 */
58  	public static interface ZipFileEntry {
59  		/**
60  		 * Write this entry to the specified output stream.
61  		 *
62  		 * @param zos The output stream to write to.
63  		 * @throws IOException
64  		 */
65  		void write(ZipOutputStream zos) throws IOException;
66  	}
67  
68  	/**
69  	 * ZipFileList entry for File entry types.
70  	 */
71  	public static class FileEntry implements ZipFileEntry {
72  
73  		/** The root file to base the entry paths on. */
74  		protected File root;
75  
76  		/** The file being zipped. */
77  		protected File file;
78  
79  		/**
80  		 * Constructor.
81  		 *
82  		 * @param root The root file that represents the base path.
83  		 * @param file The file to add to the zip file.
84  		 */
85  		public FileEntry(File root, File file) {
86  			this.root = root;
87  			this.file = file;
88  		}
89  
90  		/**
91  		 * Constructor.
92  		 *
93  		 * @param file The file to add to the zip file.
94  		 */
95  		public FileEntry(File file) {
96  			this.file = file;
97  			this.root = (file.isDirectory() ? file : file.getParentFile());
98  		}
99  
100 		@Override /* ZipFileEntry */
101 		public void write(ZipOutputStream zos) throws IOException {
102 			addFile(zos, file);
103 		}
104 
105 		/**
106 		 * Subclasses can override this method to customize which files get added to a zip file.
107 		 *
108 		 * @param f The file being added to the zip file.
109 		 * @return Always returns <jk>true</jk>.
110 		 */
111 		public boolean doAdd(File f) {
112 			return true;
113 		}
114 
115 		/**
116 		 * Adds the specified file to the specified output stream.
117 		 *
118 		 * @param zos The output stream.
119 		 * @param f The file to add.
120 		 * @throws IOException
121 		 */
122 		protected void addFile(ZipOutputStream zos, File f) throws IOException {
123 			if (doAdd(f)) {
124 				if (f.isDirectory()) {
125 					File[] fileList = f.listFiles();
126 					if (fileList == null)
127 						throw new IOException(f.toString());
128 					for (File fc : fileList)
129 						addFile(zos, fc);
130 				} else if (f.canRead()) {
131 					String path = f.getAbsolutePath().substring(root.getAbsolutePath().length() + 1).replace('\\', '/');
132 					ZipEntry e = new ZipEntry(path);
133 					e.setSize(f.length());
134 					zos.putNextEntry(e);
135 					IOPipe.create(new FileInputStream(f), zos).run();
136 				}
137 			}
138 		}
139 	}
140 }
