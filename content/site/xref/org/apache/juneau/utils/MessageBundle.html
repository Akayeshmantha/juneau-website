23  /**
24   * Wraps a {@link ResourceBundle} to provide some useful additional functionality.
25   * <ul class='spaced-list'>
26   * 	<li>Instead of throwing {@link MissingResourceException}, the {@link #getString(String)} method
27   * 		will return <js>"{!!key}"</js> if the bundle was not found, and <js>"{!key}"</js> if bundle
28   * 		was found but the key is not in the bundle.
29   * 	<li>A client locale can be set as a {@link ThreadLocal} object using the static {@link #setClientLocale(Locale)} so that client localized
30   * 		messages can be retrieved using the {@link #getClientString(String, Object...)} method on all instances of this class.
31   * 	<li>Resource bundles on parent classes can be added to the search path for this class by using the
32   * 		{@link #addSearchPath(Class, String)} method.  This allows messages to be retrieved from the resource bundles of parent classes.
33   * 	<li>Locale-specific bundles can be retrieved by using the {@link #getBundle(Locale)} method.
34   * 	<li>The {@link #getString(Locale, String, Object...)} method can be used to retrieve locale-specific messages.
35   * 	<li>Messages in the resource bundle can optionally be prefixed with the simple class name.
36   * 		For example, if the class is <code>MyClass</code> and the properties file contains <js>"MyClass.myMessage"</js>,
37   * 		the message can be retrieved using <code>getString(<js>"myMessage"</js>)</code>.
38   * </ul>
39   *
40   * @author James Bognar (james.bognar@salesforce.com)
41   */
42  @ThreadSafe
43  public class MessageBundle extends ResourceBundle {
44  
45  	private static final ThreadLocal<Locale> clientLocale = new ThreadLocal<Locale>();
46  
47  	private final ResourceBundle rb;
48  	private final String bundlePath, className;
49  	private final Class<?> forClass;
50  	private final long creationThreadId;
51  
52  	// A map that contains all keys [shortKeyName->keyName] and [keyName->keyName], where shortKeyName
53  	// refers to keys prefixed and stripped of the class name (e.g. "foobar"->"MyClass.foobar")
54  	private final Map<String,String> keyMap = new ConcurrentHashMap<String,String>();
55  
56  	// Contains all keys present in all bundles in searchBundles.
57  	private final ConcurrentSkipListSet<String> allKeys = new ConcurrentSkipListSet<String>();
58  
59  	// Bundles to search through to find properties.
60  	// Typically this will be a list of resource bundles for each class up the class hierarchy chain.
61  	private final CopyOnWriteArrayList<MessageBundle> searchBundles = new CopyOnWriteArrayList<MessageBundle>();
62  
63  	// Cache of message bundles per locale.
64  	private final ConcurrentHashMap<Locale,MessageBundle> localizedBundles = new ConcurrentHashMap<Locale,MessageBundle>();
65  
66  	/**
67  	 * Sets the locale for this thread so that calls to {@link #getClientString(String, Object...)} return
68  	 * messages in that locale.
69  	 *
70  	 * @param locale The new client locale.
71  	 */
72  	public static void setClientLocale(Locale locale) {
73  		MessageBundle.clientLocale.set(locale);
74  	}
75  
76  	/**
77  	 * Constructor.
78  	 *
79  	 * @param forClass The class using this resource bundle.
80  	 * @param bundlePath The path of the resource bundle to wrap.
81  	 * 	This can be an absolute path (e.g. <js>"com.ibm.MyMessages"</js>) or a path
82  	 * 	relative to the package of the <l>forClass</l> (e.g. <js>"MyMessages"</js> if <l>forClass</l> is <js>"com.ibm.MyClass"</js>).
83  	 */
84  	public MessageBundle(Class<?> forClass, String bundlePath) {
85  		this(forClass, bundlePath, Locale.getDefault());
86  	}
87  
88  	private MessageBundle(Class<?> forClass, String bundlePath, Locale locale) {
89  		this.forClass = forClass;
90  		this.className = forClass.getSimpleName();
91  		this.bundlePath = bundlePath;
92  		this.creationThreadId = Thread.currentThread().getId();
93  		ClassLoader cl = forClass.getClassLoader();
94  		ResourceBundle trb = null;
95  		try {
96  			trb = ResourceBundle.getBundle(bundlePath, locale, cl);
97  		} catch (MissingResourceException e) {
98  			try {
99  				trb = ResourceBundle.getBundle(forClass.getPackage().getName() + '.' + bundlePath, locale, cl);
100 			} catch (MissingResourceException e2) {
101 			}
102 		}
103 		this.rb = trb;
104 		if (rb != null) {
105 
106 			// Populate keyMap with original mappings.
107 			for (Enumeration<String> e = getKeys(); e.hasMoreElements();) {
108 				String key = e.nextElement();
109 				keyMap.put(key, key);
110 			}
111 
112 			// Override/augment with shortname mappings (e.g. "foobar"->"MyClass.foobar")
113 			String c = className + '.';
114 			for (Enumeration<String> e = getKeys(); e.hasMoreElements();) {
115 				String key = e.nextElement();
116 				if (key.startsWith(c)) {
117 					String shortKey = key.substring(className.length() + 1);
118 					keyMap.put(shortKey, key);
119 				}
120 			}
121 
122 			allKeys.addAll(keyMap.keySet());
123 		}
124 		searchBundles.add(this);
125 	}

128 	/**
129 	 * Add another bundle path to this resource bundle.
130 	 * Order of property lookup is first-to-last.
131 	 * <p>
132 	 * This method must be called from the same thread as the call to the constructor.
133 	 * This eliminates the need for synchronization.
134 	 *
135 	 * @param forClass The class using this resource bundle.
136 	 * @param bundlePath The bundle path.
137 	 * @return This object (for method chaining).
138 	 */
139 	@SuppressWarnings("hiding")
140 	public MessageBundle addSearchPath(Class<?> forClass, String bundlePath) {
141 		ThrowableUtils.assertSameThread(creationThreadId, "This method can only be called from the same thread that created the object.");
142 		MessageBundle srb = new MessageBundle(forClass, bundlePath);
143 		if (srb.rb != null) {
144 			allKeys.addAll(srb.keySet());
145 			searchBundles.add(srb);
146 		}
147 		return this;
148 	}
149 
150 	@Override /* ResourceBundle */
151 	public boolean containsKey(String key) {
152 		return allKeys.contains(key);
153 	}
154 
155 	/**
156 	 * Similar to {@link ResourceBundle#getString(String)} except allows you to pass in {@link MessageFormat} objects.
157 	 *
158 	 * @param key The resource bundle key.
159 	 * @param args Optional variable replacement arguments.
160 	 * @return The resolved value.  Never <jk>null</jk>.  <js>"{!!key}"</j> if the bundle is missing.  <js>"{!key}"</j> if the key is missing.
161 	 */
162 	public String getString(String key, Object...args) {
163 		String s = getString(key);
164 		if (s.length() > 0 && s.charAt(0) == '{')
165 			return s;
166 		if (args.length > 0)
167 			return MessageFormat.format(s, args);
168 		return s;
169 	}
170 
171 	/**
172 	 * Same as {@link #getString(String, Object...)} but allows you to specify the locale.
173 	 *
174 	 * @param locale The locale of the resource bundle to retrieve message from.
175 	 * @param key The resource bundle key.
176 	 * @param args Optional variable replacement arguments.
177 	 * @return The resolved value.  Never <jk>null</jk>.  <js>"{!!key}"</j> if the bundle is missing.  <js>"{!key}"</j> if the key is missing.
178 	 */
179 	public String getString(Locale locale, String key, Object...args) {
180 		if (locale == null)
181 			return getString(key, args);
182 		return getBundle(locale).getString(key, args);
183 	}
184 
185 	/**
186 	 * Same as {@link #getString(String, Object...)} but uses the locale specified on the call to {@link #setClientLocale(Locale)}.
187 	 *
188 	 * @param key The resource bundle key.
189 	 * @param args Optional variable replacement arguments.
190 	 * @return The resolved value.  Never <jk>null</jk>.  <js>"{!!key}"</j> if the bundle is missing.  <js>"{!key}"</j> if the key is missing.
191 	 */
192 	public String getClientString(String key, Object...args) {
193 		return getString(clientLocale.get(), key, args);
194 	}
195 
196 	/**
197 	 * Looks for all the specified keys in the resource bundle and returns the first value that exists.
198 	 *
199 	 * @param keys The list of possible keys.
200 	 * @return The resolved value, or <jk>null</jk> if no value is found or the resource bundle is missing.
201 	 */
202 	public String findFirstString(String...keys) {
203 		if (rb == null)
204 			return null;
205 		for (String k : keys) {
206 			if (containsKey(k))
207 				return getString(k);
208 		}
209 		return null;
210 	}
211 
212 	/**
213 	 * Same as {@link #findFirstString(String...)}, but uses the specified locale.
214 	 *
215 	 * @param locale The locale of the resource bundle to retrieve message from.
216 	 * @param keys The list of possible keys.
217 	 * @return The resolved value, or <jk>null</jk> if no value is found or the resource bundle is missing.
218 	 */
219 	public String findFirstString(Locale locale, String...keys) {
220 		MessageBundle srb = getBundle(locale);
221 		return srb.findFirstString(keys);
222 	}

224 	@Override /* ResourceBundle */
225 	public Set<String> keySet() {
226 		return Collections.unmodifiableSet(allKeys);
227 	}
228 
229 	/**
230 	 * Returns all keys in this resource bundle with the specified prefix.
231 	 *
232 	 * @param prefix The prefix.
233 	 * @return The set of all keys in the resource bundle with the prefix.
234 	 */
235 	public Set<String> keySet(String prefix) {
236 		Set<String> set = new HashSet<String>();
237 		for (String s : keySet()) {
238 			if (s.equals(prefix) || (s.startsWith(prefix) && s.charAt(prefix.length()) == '.'))
239 				set.add(s);
240 		}
241 		return set;
242 	}
243 
244 	@Override /* ResourceBundle */
245 	public Enumeration<String> getKeys() {
246 		if (rb == null)
247 			return new Vector<String>(0).elements();
248 		return rb.getKeys();
249 	}
250 
251 	@Override /* ResourceBundle */
252 	protected Object handleGetObject(String key) {
253 		for (MessageBundle srb : searchBundles) {
254 			if (srb.rb != null) {
255 				String key2 = srb.keyMap.get(key);
256 				if (key2 != null) {
257 					try {
258 						return srb.rb.getObject(key2);
259 					} catch (Exception e) {
260 						return "{!"+key+"}";
261 					}
262 				}
263 			}
264 		}
265 		if (rb == null)
266 			return "{!!"+key+"}";
267 		return "{!"+key+"}";
268 	}
269 
270 	/**
271 	 * Returns this resource bundle as an {@link ObjectMap}.
272 	 * <p>
273 	 * Useful for debugging purposes.
274 	 * Note that any class that implements a <code>toObjectMap()</code> method will automatically be serialized by
275 	 * 	calling this method and serializing the result.
276 	 * <p>
277 	 * This method always constructs a new {@link ObjectMap} on each call.
278 	 *
279 	 * @return A new map containing all the keys and values in this bundle.
280 	 */
281 	public ObjectMap toObjectMap() {
282 		ObjectMap om = new ObjectMap();
283 		for (String k : allKeys)
284 			om.put(k, getString(k));
285 		return om;
286 	}
287 
288 	/**
289 	 * Returns the resource bundle for the specified locale.
290 	 *
291 	 * @param locale The client locale.
292 	 * @return The resource bundle for the specified locale.  Never <jk>null</jk>.
293 	 */
294 	public MessageBundle getBundle(Locale locale) {
295 		MessageBundle mb = localizedBundles.get(locale);
296 		if (mb != null)
297 			return mb;
298 		mb = new MessageBundle(forClass, bundlePath, locale);
299 		List<MessageBundle> l = new ArrayList<MessageBundle>(searchBundles.size()-1);
300 		for (int i = 1; i < searchBundles.size(); i++) {
301 			MessageBundle srb = searchBundles.get(i);
302 			srb = new MessageBundle(srb.forClass, srb.bundlePath, locale);
303 			l.add(srb);
304 			mb.allKeys.addAll(srb.keySet());
305 		}
306 		mb.searchBundles.addAll(l);
307 		localizedBundles.putIfAbsent(locale, mb);
308 		return localizedBundles.get(locale);
309 	}
