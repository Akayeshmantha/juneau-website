13  package org.apache.juneau.microservice;
14  
15  import static org.apache.juneau.html.HtmlDocSerializerContext.*;
16  
17  import org.apache.juneau.server.*;
18  import org.apache.juneau.server.annotation.*;
19  import org.apache.juneau.svl.*;
20  import org.apache.juneau.svl.vars.*;
21  
22  /**
23   * Superclass for all REST resources.
24   * <p>
25   * In additional to the functionality of the {@link RestServletDefault} group,
26   * augments the {@link #createVarResolver()} method with the following additional variable types:
27   * <ul class='spaced-list'>
28   * 	<li><code class='snippet'>$ARG{...}</code> - Command line arguments pulled from {@link Microservice#getArgs()}.<br>
29   * 		<h6 class='figure'>Example:</h6>
30   * 		<p class='bcode'>
31   * 			String firstArg = request.getVarResolver().resolve(<js>"$ARG{0}"</js>);  <jc>// First argument.</jc>
32   * 			String namedArg = request.getVarResolver().resolve(<js>"$ARG{myarg}"</js>);  <jc>// Named argument (e.g. "myarg=foo"). </jc>
33   * 		</p>
34   * 	<li><code class='snippet'>$MF{...}</code> - Manifest file entries pulled from {@link Microservice#getManifest()}.<br>
35   * 		<h6 class='figure'>Example:</h6>
36   * 		<p class='bcode'>
37   * 			String mainClass = request.getVarResolver().resolve(<js>"$MF{Main-Class}"</js>);  <jc>// Main class. </jc>
38   * 		</p>
39   * </ul>
40   *
41   * @author James Bognar (james.bognar@salesforce.com)
42   */
43  @SuppressWarnings("serial")
44  @RestResource(
45  	properties={
46  		@Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS'}")
47  	},
48  	config="$S{juneau.configFile}",
49  	stylesheet="$C{REST/stylesheet,styles/juneau.css}"
50  )
51  public abstract class Resource extends RestServletDefault {
52  
53  	/**
54  	 * Adds $ARG and $MF variables to variable resolver defined on {@link RestServlet#createVarResolver()}.
55  	 */
56  	@Override
57  	protected VarResolver createVarResolver() {
58  		return super.createVarResolver()
59  			.addVars(ArgsVar.class, ManifestFileVar.class)
60  			.setContextObject(ArgsVar.SESSION_args, Microservice.getArgs())
61  			.setContextObject(ManifestFileVar.SESSION_manifest, Microservice.getManifest());
62  	}
63  }
