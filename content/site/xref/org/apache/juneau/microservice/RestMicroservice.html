36  /**
37   * Entry point for Juneau microservice that implements a REST interface using Jetty on a single port.
38   *
39   * <h6 class='topic'>Jetty Server Details</h6>
40   * <p>
41   * The Jetty server is created by the {@link #createServer()} method and started with the {@link #startServer()} method.
42   * These methods can be overridden to provided customized behavior.
43   * <p>
44   *
45   * <h6 class='topic'>Defining REST Resources</h6>
46   * <p>
47   * Top-level REST resources are defined by the {@link #getResourceMap()} method.
48   * This method can be overridden to provide a customized list of REST resources.
49   * <p>
50   *
51   * <h6 class='topic'>Logging</h6>
52   * <p>
53   * Logging is initialized by the {@link #initLogging()} method.
54   * This method can be overridden to provide customized logging behavior.
55   *
56   * <h6 class='topic'>Lifecycle Listener Methods</h6>
57   * Subclasses can optionally implement the following event listener methods:
58   * <ul class='spaced-list'>
59   * 	<li>{@link #onStart()} - Gets executed before {@link #start()}.
60   * 	<li>{@link #onStop()} - Gets executed before {@link #stop()}.
61   * 	<li>{@link #onCreateServer()} - Gets executed before {@link #createServer()}.
62   * 	<li>{@link #onStartServer()} - Gets executed before {@link #startServer()}.
63   * 	<li>{@link #onPostStartServer()} - Gets executed after {@link #startServer()}.
64   * 	<li>{@link #onStopServer()} - Gets executed before {@link #stop()}.
65   * 	<li>{@link #onPostStopServer()} - Gets executed after {@link #stop()}.
66   * </ul>
67   *
68   * @author james.bognar@salesforce.com
69   */
70  public class RestMicroservice extends Microservice {

101 	@Override /* Microservice */
102 	public RestMicroservice start() throws Exception {
103 		super.start();
104 		initLogging();
105 		createServer();
106 		startServer();
107 		return this;
108 	}

110 	@Override /* Microservice */
111 	public RestMicroservice join() throws Exception {
112 		server.join();
113 		return this;
114 	}
115 
116 	@Override /* Microservice */
117 	public RestMicroservice stop() {
118 		Thread t = new Thread() {
119 			@Override /* Thread */
120 			public void run() {
121 				try {
122 					onStopServer();
123 					logger.warning("Stopping server.");
124 					System.out.println();
125 					server.stop();
126 					logger.warning("Server stopped.");
127 					onPostStopServer();
128 				} catch (Exception e) {
129 					logger.log(Level.SEVERE, e.getLocalizedMessage(), e);
130 				}
131 			}
132 		};
133 		t.start();
134 		try {
135 			t.join();
136 		} catch (InterruptedException e) {
137 			e.printStackTrace();
138 		}
139 		super.stop();
140 		return this;
141 	}

148 	 * Initialize the logging for this microservice.
149 	 * <p>
150 	 * Subclasses can override this method to provide customized logging.
151 	 * <p>
152 	 * The default implementation uses the <cs>Logging</cs> section in the config file to set up logging:
153 	 * <p class='bcode'>
154 	 * 	<cc>#================================================================================
155 	 * 	# Logger settings
156 	 * 	# See FileHandler Java class for details.
157 	 * 	#================================================================================</cc>
158 	 * 	<cs>[Logging]</cs>
159 	 *
160 	 * 	<cc># The directory where to create the log file.
161 	 * 	# Default is ".".</cc>
162 	 * 	<ck>logDir</ck> = logs
163 	 *
164 	 * 	<cc># The name of the log file to create for the main logger.
165 	 * 	# The logDir and logFile make up the pattern that's passed to the FileHandler
166 	 * 	# constructor.
167 	 * 	# If value is not specified, then logging to a file will not be set up.</cc>
168 	 * 	<ck>logFile</ck> = microservice.%g.log
169 	 *
170 	 * 	<cc># Whether to append to the existing log file or create a new one.
171 	 * 	# Default is false.</cc>
172 	 * 	<ck>append</ck> =
173 	 *
174 	 * 	<cc># The SimpleDateFormat format to use for dates.
175 	 * 	# Default is "yyyy.MM.dd hh:mm:ss".</cc>
176 	 * 	<ck>dateFormat</ck> =
177 	 *
178 	 * 	<cc># The log message format.
179 	 * 	# The value can contain any of the following variables:
180 	 * 	# 	{date} - The date, formatted per dateFormat.
181 	 * 	#	{class} - The class name.
182 	 * 	#	{method} - The method name.
183 	 * 	#	{logger} - The logger name.
184 	 * 	#	{level} - The log level name.
185 	 * 	#	{msg} - The log message.
186 	 * 	#	{threadid} - The thread ID.
187 	 * 	#	{exception} - The localized exception message.
188 	 * 	# Default is "[{date} {level}] {msg}%n".</cc>
189 	 * 	<ck>format</ck> =
190 	 *
191 	 * 	<cc># The maximum log file size.
192 	 * 	# Suffixes available for numbers.
193 	 * 	# See ConfigFile.getInt(String,int) for details.
194 	 * 	# Default is 1M.</cc>
195 	 * 	<ck>limit</ck> = 10M
196 	 *
197 	 * 	<cc># Max number of log files.
198 	 * 	# Default is 1.</cc>
199 	 * 	<ck>count</ck> = 5
200 	 *
201 	 * 	<cc># Default log levels.
202 	 * 	# Keys are logger names.
203 	 * 	# Values are serialized Level POJOs.</cc>
204 	 * 	<ck>levels</ck> = { org.apache.juneau:'INFO' }
205 	 *
206 	 * 	<cc># Only print unique stack traces once and then refer to them by a simple 8 character hash identifier.
207 	 * 	# Useful for preventing log files from filling up with duplicate stack traces.
208 	 * 	# Default is false.</cc>
209 	 * 	<ck>useStackTraceHashes</ck> = true
210 	 *
211 	 * 	<cc># The default level for the console logger.
212 	 * 	# Default is WARNING.</cc>
213 	 * 	<ck>consoleLevel</ck> = WARNING
214 	 * </p>
215 	 *
216 	 * @throws Exception
217 	 */
218 	protected void initLogging() throws Exception {
219 		ConfigFile cf = getConfig();
220 		logger = Logger.getLogger("");
221 		String logFile = cf.getString("Logging/logFile");
222 		if (! StringUtils.isEmpty(logFile)) {
223 			LogManager.getLogManager().reset();
224 			String logDir = cf.getString("Logging/logDir", ".");
225 			FileUtils.mkdirs(new File(logDir), false);
226 			boolean append = cf.getBoolean("Logging/append");
227 			int limit = cf.getInt("Logging/limit", 1024*1024);
228 			int count = cf.getInt("Logging/count", 1);
229 			FileHandler fh = new FileHandler(logDir + '/' + logFile, limit, count, append);
230 
231 			boolean useStackTraceHashes = cf.getBoolean("Logging/useStackTraceHashes");
232 			String format = cf.getString("Logging/format", "[{date} {level}] {msg}%n");
233 			String dateFormat = cf.getString("Logging/dateFormat", "yyyy.MM.dd hh:mm:ss");
234 			fh.setFormatter(new LogEntryFormatter(format, dateFormat, useStackTraceHashes));
235 			logger.addHandler(fh);
236 
237 			ConsoleHandler ch = new ConsoleHandler();
238 			ch.setLevel(Level.parse(cf.getString("Logging/consoleLevel", "WARNING")));
239 			ch.setFormatter(new LogEntryFormatter(format, dateFormat, false));
240 			logger.addHandler(ch);
241 		}
242 		ObjectMap loggerLevels = cf.getObject(ObjectMap.class, "Logging/levels");
243 		if (loggerLevels != null)
244 		for (String l : loggerLevels.keySet())
245 			Logger.getLogger(l).setLevel(loggerLevels.get(Level.class, l));
246 	}

249 	 * Method used to create (but not start) an instance of a Jetty server.
250 	 * <p>
251 	 * Subclasses can override this method to customize the Jetty server before it is started.
252 	 * <p>
253 	 * The default implementation is configured by the following values in the config file:
254 	 * <p>
255 	 * <p class='bcode'>
256 	 * 	<cc>#================================================================================
257 	 * 	# REST settings
258 	 * 	#================================================================================</cc>
259 	 * 	<cs>[REST]</cs>
260 	 *
261 	 * 	<cc># The HTTP port number to use.
262 	 * 	# Default is Rest-Port setting in manifest file, or 8000.</cc>
263 	 * 	<ck>port</ck> = 10000
264 	 *
265 	 * 	<cc># The context root of the Jetty server.
266 	 * 	# Default is Rest-ContextPath in manifest file, or "/".</cc>
267 	 * 	<ck>contextPath</ck> = 10000
268 	 *
269 	 * 	<cc># Authentication:  NONE, BASIC.
270 	 * 	# Default is Rest-AuthType in manifest file, or NONE.</cc>
271 	 * 	<ck>authType</ck> = NONE
272 	 *
273 	 * 	<cc># The BASIC auth username.
274 	 * 	# Default is Rest-LoginUser in manifest file.</cc>
275 	 * 	<ck>loginUser</ck> =
276 	 *
277 	 * 	<cc># The BASIC auth password.
278 	 * 	# Default is Rest-LoginPassword in manifest file.</cc>
279 	 * 	<ck>loginPassword</ck> =
280 	 *
281 	 * 	<cc># The BASIC auth realm.
282 	 * 	# Default is Rest-AuthRealm in manifest file.</cc>
283 	 * 	<ck>authRealm</ck> =
284 	 *
285 	 * 	<cc># Enable SSL support.</cc>
286 	 * 	<ck>useSsl</ck> = false
287 	 *
288 	 * 	<cc>#================================================================================
289 	 * 	# Bean properties on the org.eclipse.jetty.util.ssl.SslSocketFactory class
290 	 * 	#--------------------------------------------------------------------------------
291 	 * 	# Ignored if REST/useSsl is false.
292 	 * 	#================================================================================</cc>
293 	 * 	<cs>[REST-SslContextFactory]</cs>
294 	 * 	<ck>keyStorePath</ck> = client_keystore.jks
295 	 * 	<ck>keyStorePassword*</ck> = {HRAaRQoT}
296 	 * 	<ck>excludeCipherSuites</ck> = TLS_DHE.*, TLS_EDH.*
297 	 * 	<ck>excludeProtocols</ck> = SSLv3
298 	 * 	<ck>allowRenegotiate</ck> = false
299 	 * </p>
300 	 *
301 	 * @return The newly-created server.
302 	 * @throws Exception
303 	 */
304 	protected Server createServer() throws Exception {
305 		onCreateServer();
306 
307 		ConfigFile cf = getConfig();
308 		ObjectMap mf = getManifest();
309 
310 		port = cf.getInt("REST/port", mf.getInt("Rest-Port", 8000));
311 		String contextPath = cf.getString("REST/contextPath", mf.getString("Rest-ContextPath", "/"));
312 
313 		if (cf.getBoolean("REST/useSsl")) {
314 
315 			SslContextFactory sslContextFactory = new SslContextFactory();
316 
317 			// Write the properties in REST-SslContextFactory to the bean setters on sslContextFactory.
318 			// Throws an exception if section contains unknown properties.
319 			// Only look for bean properties of type String/String/boolean/int since class has multiple
320 			// 	setters with the same name (e.g. setKeyStore(KeyStore) and setKeyStore(String)).
321 			ObjectMap m = cf.writeProperties("REST-SslContextFactory", sslContextFactory, false, String.class, String[].class, boolean.class, int.class);
322 
323 			// We're using Jetty 8 that doesn't allow regular expression matching in SslContextFactory.setExcludeCipherSuites(),
324 			// so to prevent having the config file list all old cipher suites, exclude the known bad ones.
325 			String[] excludeCipherSuites = ArrayUtils.combine(
326 				StringUtils.split("SSL_RSA_WITH_DES_CBC_SHA,SSL_DHE_RSA_WITH_DES_CBC_SHA,SSL_DHE_DSS_WITH_DES_CBC_SHA,SSL_RSA_EXPORT_WITH_RC4_40_MD5,SSL_RSA_EXPORT_WITH_DES40_CBC_SHA,SSL_DHE_RSA_EXPORT_WITH_DES40_CBC_SHA,SSL_DHE_RSA_WITH_3DES_EDE_CBC_SHA,SSL_DHE_DSS_WITH_3DES_EDE_CBC_SHA,TLS_DHE_RSA_WITH_AES_256_CBC_SHA256,TLS_DHE_DSS_WITH_AES_256_CBC_SHA256,TLS_DHE_RSA_WITH_AES_256_CBC_SHA,TLS_DHE_DSS_WITH_AES_256_CBC_SHA,TLS_DHE_RSA_WITH_AES_128_CBC_SHA256,TLS_DHE_DSS_WITH_AES_128_CBC_SHA256,TLS_DHE_RSA_WITH_AES_128_CBC_SHA,TLS_DHE_DSS_WITH_AES_128_CBC_SHA", ','),
327 				sslContextFactory.getExcludeCipherSuites()
328 			);
329 			sslContextFactory.setExcludeCipherSuites(excludeCipherSuites);
330 
331 			logger.log(Level.WARNING, "SSL properties set: {0}", JsonSerializer.DEFAULT_LAX.toString(m));
332 
333 			SslSocketConnector connector = new SslSocketConnector(sslContextFactory);
334 			connector.setPort(port);
335 
336 			server = new Server();
337 			server.setConnectors(new Connector[] { connector });
338 
339 		} else {
340 			server = new Server(port);
341 		}
342 
343 		ServletContextHandler context = new ServletContextHandler(ServletContextHandler.SESSIONS);
344 
345 		String authType = cf.getString("REST/authType", mf.getString("Rest-AuthType", "NONE"));
346 		if (authType.equals("BASIC"))
347 			context.setSecurityHandler(basicAuth(cf, mf));
348 
349 		context.setContextPath(contextPath);
350 		server.setHandler(context);
351 
352 		for (Map.Entry<String,Class<? extends Servlet>> e : getResourceMap().entrySet())
353 			context.addServlet(e.getValue(), e.getKey()).setInitOrder(0);
354 
355 		return server;
356 	}

373 	 * Returns the resource map to use for this microservice.
374 	 * <p>
375 	 * <p>
376 	 * Subclasses can override this method to programmatically specify their resources.
377 	 * <p>
378 	 * The default implementation is configured by the following values in the config file:
379 	 * <p>
380 	 * <p class='bcode'>
381 	 *
382 	 * 	<cc>#================================================================================
383 	 * 	# REST settings
384 	 * 	#================================================================================</cc>
385 	 * 	<cs>[REST]</cs>
386 	 *
387 	 * 	<cc># A JSON map of servlet paths to servlet classes.
388 	 * 	# Example:
389 	 * 	# 	resourceMap = {'/*':'com.ibm.MyServlet'}
390 	 * 	# Either resourceMap or resources must be specified if it's not defined in
391 	 * 	# the manifest file.</cc>
392 	 * 	<ck>resourceMap</ck> =
393 	 *
394 	 * 	<cc># A comma-delimited list of names of classes that extend from Servlet.
395 	 * 	# Resource paths are pulled from @RestResource.path() annotation, or
396 	 * 	# 	"/*" if annotation not specified.
397 	 * 	# Example:
398 	 * 	# 	resources = com.ibm.MyServlet
399 	 * 	 * 	# Default is Rest-Resources in manifest file.
400 	 * 	# Either resourceMap or resources must be specified if it's not defined in
401 	 * 	# the manifest file.</cc>
402 	 * 	<ck>resources</ck> =
403 	 * </p>
404 	 * <p>
405 	 * 	In most cases, the rest resources will be specified in the manifest file since
406 	 * 	it's not likely to be a configurable property:
407 	 * <p>
408 	 * <p class='bcode'>
409 	 * 	<mk>Rest-Resources:</mk> org.apache.juneau.microservice.sample.RootResources
410 	 * </p>
411 	 *
412 	 * @return The map of REST resources.
413 	 * @throws ClassNotFoundException
414 	 * @throws ParseException
415 	 */
416 	@SuppressWarnings("unchecked")
417 	protected Map<String,Class<? extends Servlet>> getResourceMap() throws ClassNotFoundException, ParseException {
418 		ConfigFile cf = getConfig();
419 		ObjectMap mf = getManifest();
420 		Map<String,Class<? extends Servlet>> rm = new LinkedHashMap<String,Class<? extends Servlet>>();
421 
422 		ObjectMap resourceMap = cf.getObject(ObjectMap.class, "REST/resourceMap");
423 		String[] resources = cf.getStringArray("REST/resources", mf.getStringArray("Rest-Resources"));
424 
425 		if (resourceMap != null && ! resourceMap.isEmpty()) {
426 			for (Map.Entry<String,Object> e : resourceMap.entrySet()) {
427 				Class<?> c = Class.forName(e.getValue().toString());
428 				if (! ClassUtils.isParentClass(Servlet.class, c))
429 					throw new ClassNotFoundException("Invalid class specified as resource.  Must be a Servlet.  Class='"+c.getName()+"'");
430 				rm.put(e.getKey(), (Class<? extends Servlet>)c);
431 			}
432 		} else if (resources.length > 0) {
433 			for (String resource : resources) {
434 				Class<?> c = Class.forName(resource);
435 				if (! ClassUtils.isParentClass(Servlet.class, c))
436 					throw new ClassNotFoundException("Invalid class specified as resource.  Must be a Servlet.  Class='"+c.getName()+"'");
437 				RestResource rr = c.getAnnotation(RestResource.class);
438 				String path = rr == null ? "/*" : rr.path();
439 				if (! path.endsWith("*"))
440 					path += (path.endsWith("/") ? "*" : "/*");
441 				rm.put(path, (Class<? extends Servlet>)c);
442 			}
443 		}
444 		return rm;
445 	}

447 	/**
448 	 * Called when {@link ConfigFile#save()} is called on the config file.
449 	 * <p>
450 	 * The default behavior is configured by the following value in the config file:
451 	 * <p>
452 	 * <p class='bcode'>
453 	 * 	<cs>[REST]</cs>
454 	 *
455 	 * 	<cc># What to do when the config file is saved.
456 	 * 	# Possible values:
457 	 * 	# 	NOTHING - Don't do anything. (default)
458 	 * 	#	RESTART_SERVER - Restart the Jetty server.
459 	 * 	#	RESTART_SERVICE - Shutdown and exit with code '3'.</cc>
460 	 * 	<ck>saveConfigAction</ck> = RESTART_SERVER
461 	 * </p>
462 	 */
463 	@Override /* Microservice */
464 	protected void onConfigSave(ConfigFile cf) {
465 		try {
466 			String saveConfigAction = cf.getString("REST/saveConfigAction", "NOTHING");
467 			if (saveConfigAction.equals("RESTART_SERVER")) {
468 				new Thread() {
469 					@Override /* Thread */
470 					public void run() {
471 						try {
472 							RestMicroservice.this.stop();
473 							RestMicroservice.this.start();
474 						} catch (Exception e) {
475 							logger.log(Level.SEVERE, e.getLocalizedMessage(), e);
476 						}
477 					}
478 				}.start();
479 			} else if (saveConfigAction.equals("RESTART_SERVICE")) {
480 				stop();
481 				System.exit(3);
482 			}
483 		} catch (Exception e) {
484 			throw new RuntimeException(e);
485 		}
486 	}

531 	private static final SecurityHandler basicAuth(ConfigFile cf, ObjectMap mf) {
532 
533 		HashLoginService l = new HashLoginService();
534 		String user = cf.getString("REST/loginUser", mf.getString("Rest-LoginUser"));
535 		String pw = cf.getString("REST/loginPassword", mf.getString("Rest-LoginPassword"));
536 		String realm = cf.getString("REST/authRealm", mf.getString("Rest-AuthRealm", ""));
537 		String ctx = cf.getString("REST/contextPath", mf.getString("Rest-ContextPath", "/"));
538 
539 		l.putUser(user, Credential.getCredential(pw), new String[] { "user" });
540 		l.setName(realm);
541 
542 		Constraint constraint = new Constraint();
543 		constraint.setName(Constraint.__BASIC_AUTH);
544 		constraint.setRoles(new String[] { "user" });
545 		constraint.setAuthenticate(true);
546 
547 		ConstraintMapping cm = new ConstraintMapping();
548 		cm.setConstraint(constraint);
549 		cm.setPathSpec(ctx);
550 
551 		ConstraintSecurityHandler csh = new ConstraintSecurityHandler();
552 		csh.setAuthenticator(new BasicAuthenticator());
553 		csh.setRealmName("myrealm");
554 		csh.addConstraintMapping(cm);
555 		csh.setLoginService(l);
556 
557 		return csh;
558 	}
