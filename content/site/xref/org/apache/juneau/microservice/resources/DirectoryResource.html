36  /**
37   * REST resource that allows access to a file system directory.
38   * <p>
39   * 	The root directory is specified in one of two ways:
40   * </p>
41   * <ul class='spaced-list'>
42   * 	<li>Specifying the location via a <l>DirectoryResource.rootDir</l> property.
43   * 	<li>Overriding the {@link #getRootDir()} method.
44   * </ul>
45   * <p>
46   * 	Read/write access control is handled through the following properties:
47   * </p>
48   * <ul class='spaced-list'>
49   * 	<li><l>DirectoryResource.allowViews</l> - If <jk>true</jk>, allows view and download access to files.
50   * 	<li><l>DirectoryResource.allowPuts</l> - If <jk>true</jk>, allows files to be created or overwritten.
51   * 	<li><l>DirectoryResource.allowDeletes</l> - If <jk>true</jk>, allows files to be deleted.
52   * </ul>
53   * <p>
54   * 	Access can also be controlled by overriding the {@link #checkAccess(RestRequest)} method.
55   * </p>
56   */
57  @RestResource(
58  	label="File System Explorer",
59  	description="Contents of $A{path}",
60  	messages="nls/DirectoryResource",
61  	properties={
62  		@Property(name=HTML_uriAnchorText, value=PROPERTY_NAME),
63  		@Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.DirectoryResource)'}"),
64  		@Property(name=REST_allowMethodParam, value="*"),
65  		@Property(name="DirectoryResource.rootDir", value=""),
66  		@Property(name="DirectoryResource.allowViews", value="false"),
67  		@Property(name="DirectoryResource.allowDeletes", value="false"),
68  		@Property(name="DirectoryResource.allowPuts", value="false")
69  	}
70  )
71  public class DirectoryResource extends Resource {
72  	private static final long serialVersionUID = 1L;
73  
74  	private File rootDir;     // The root directory
75  
76  	// Settings enabled through servlet init parameters
77  	private boolean allowDeletes, allowPuts, allowViews;
78  
79  	private static Logger logger = Logger.getLogger(DirectoryResource.class.getName());
80  
81  	@Override /* Servlet */
82  	public void init() throws ServletException {
83  		ObjectMap p = getProperties();
84  		rootDir = new File(p.getString("DirectoryResource.rootDir"));
85  		allowViews = p.getBoolean("DirectoryResource.allowViews", false);
86  		allowDeletes = p.getBoolean("DirectoryResource.allowDeletes", false);
87  		allowPuts = p.getBoolean("DirectoryResource.allowPuts", false);
88  	}
89  
90  	/**
91  	 * Returns the root directory defined by the 'rootDir' init parameter.
92  	 * Subclasses can override this method to provide their own root directory.
93  	 * @return The root directory.
94  	 */
95  	protected File getRootDir() {
96  		if (rootDir == null) {
97  			rootDir = new File(getProperties().getString("rootDir"));
98  			if (! rootDir.exists())
99  				if (! rootDir.mkdirs())
100 					throw new RuntimeException("Could not create root dir");
101 		}
102 		return rootDir;
103 	}

105 	/**
106 	 * [GET /*]
107 	 *  On directories, returns a directory listing.
108 	 *  On files, returns information about the file.
109 	 *
110 	 * @param req The HTTP request.
111 	 * @return Either a FileResource or list of FileResources depending on whether it's a
112 	 * 	file or directory.
113 	 * @throws Exception - If file could not be read or access was not granted.
114 	 */
115 	@RestMethod(name="GET", path="/*",
116 		description="On directories, returns a directory listing.\nOn files, returns information about the file.",
117 		converters={Queryable.class}
118 	)
119 	public Object doGet(RestRequest req) throws Exception {
120 		checkAccess(req);
121 
122 		String pathInfo = req.getPathInfo();
123 		File f = pathInfo == null ? rootDir : new File(rootDir.getAbsolutePath() + pathInfo);
124 
125 		if (!f.exists())
126 			throw new RestException(SC_NOT_FOUND, "File not found");
127 
128 		req.setAttribute("path", f.getAbsolutePath());
129 
130 		if (f.isDirectory()) {
131 			List<FileResource> l = new LinkedList<FileResource>();
132 			File[] files = f.listFiles();
133 			if (files != null) {
134 				for (File fc : files) {
135 					URL fUrl = new URL(req.getRequestURL().append("/").append(fc.getName()).toString());
136 					l.add(new FileResource(fc, fUrl));
137 				}
138 			}
139 			return l;
140 		}
141 
142 		return new FileResource(f, new URL(req.getRequestURL().toString()));
143 	}

145 	/**
146 	 * [DELETE /*]
147 	 *  Delete a file on the file system.
148 	 *
149 	 * @param req The HTTP request.
150 	 * @return The message <js>"File deleted"</js> if successful.
151 	 * @throws Exception - If file could not be read or access was not granted.
152 	 */
153 	@RestMethod(name="DELETE", path="/*",
154 		description="Delete a file on the file system."
155 	)
156 	public Object doDelete(RestRequest req) throws Exception {
157 		checkAccess(req);
158 
159 		File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
160 		deleteFile(f);
161 
162 		if (req.getHeader("Accept").contains("text/html"))
163 			return new Redirect();
164 		return "File deleted";
165 	}
166 
167 	/**
168 	 * [PUT /*]
169 	 * Add or overwrite a file on the file system.
170 	 *
171 	 * @param req The HTTP request.
172 	 * @return The message <js>"File added"</js> if successful.
173 	 * @throws Exception - If file could not be read or access was not granted.
174 	 */
175 	@RestMethod(name="PUT", path="/*",
176 		description="Add or overwrite a file on the file system."
177 	)
178 	public Object doPut(RestRequest req) throws Exception {
179 		checkAccess(req);
180 
181 		File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
182 		String parentSubPath = f.getParentFile().getAbsolutePath().substring(rootDir.getAbsolutePath().length());
183 		BufferedOutputStream bos = new BufferedOutputStream(new FileOutputStream(f));
184 		IOPipe.create(req.getInputStream(), bos).closeOut().run();
185 		if (req.getContentType().contains("html"))
186 			return new Redirect(parentSubPath);
187 		return "File added";
188 	}

190 	/**
191 	 * [VIEW /*]
192 	 * 	View the contents of a file.
193 	 * 	Applies to files only.
194 	 *
195 	 * @param req The HTTP request.
196 	 * @param res The HTTP response.
197 	 * @return A Reader containing the contents of the file.
198 	 * @throws Exception - If file could not be read or access was not granted.
199 	 */
200 	@RestMethod(name="VIEW", path="/*",
201 		description="View the contents of a file.\nApplies to files only."
202 	)
203 	public Reader doView(RestRequest req, RestResponse res) throws Exception {
204 		checkAccess(req);
205 
206 		File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
207 
208 		if (!f.exists())
209 			throw new RestException(SC_NOT_FOUND, "File not found");
210 
211 		if (f.isDirectory())
212 			throw new RestException(SC_METHOD_NOT_ALLOWED, "VIEW not available on directories");
213 
214 		res.setContentType("text/plain");
215 		return new FileReader(f);
216 	}
217 
218 	/**
219 	 * [DOWNLOAD /*]
220 	 * 	Download the contents of a file.
221 	 * 	Applies to files only.
222 	 *
223 	 * @param req The HTTP request.
224 	 * @param res The HTTP response.
225 	 * @return A Reader containing the contents of the file.
226 	 * @throws Exception - If file could not be read or access was not granted.
227 	 */
228 	@RestMethod(name="DOWNLOAD", path="/*",
229 		description="Download the contents of a file.\nApplies to files only."
230 	)
231 	public Reader doDownload(RestRequest req, RestResponse res) throws Exception {
232 		checkAccess(req);
233 
234 		File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
235 
236 		if (!f.exists())
237 			throw new RestException(SC_NOT_FOUND, "File not found");
238 
239 		if (f.isDirectory())
240 			throw new RestException(SC_METHOD_NOT_ALLOWED, "DOWNLOAD not available on directories");
241 
242 		res.setContentType("application");
243 		return new FileReader(f);
244 	}

246 	/**
247 	 * Verify that the specified request is allowed.
248 	 * Subclasses can override this method to provide customized behavior.
249 	 * Method should throw a {@link RestException} if the request should be disallowed.
250 	 *
251 	 * @param req The HTTP request.
252 	 */
253 	protected void checkAccess(RestRequest req) {
254 		String method = req.getMethod();
255 		if (method.equals("VIEW") && ! allowViews)
256 			throw new RestException(SC_METHOD_NOT_ALLOWED, "VIEW not enabled");
257 		if (method.equals("PUT") && ! allowPuts)
258 			throw new RestException(SC_METHOD_NOT_ALLOWED, "PUT not enabled");
259 		if (method.equals("DELETE") && ! allowDeletes)
260 			throw new RestException(SC_METHOD_NOT_ALLOWED, "DELETE not enabled");
261 		if (method.equals("DOWNLOAD") && ! allowViews)
262 			throw new RestException(SC_METHOD_NOT_ALLOWED, "DOWNLOAD not enabled");
263 	}

265 	/** File POJO */
266 	public class FileResource {
267 		private File f;
268 		private URL url;
269 
270 		/**
271 		 * Constructor.
272 		 * @param f The file.
273 		 * @param url The URL of the file resource.
274 		 */
275 		public FileResource(File f, URL url) {
276 			this.f = f;
277 			this.url = url;
278 		}
279 
280 		// Bean property getters
281 
282 		/**
283 		 * @return The URL of the file resource.
284 		 */
285 		public URL getUrl() {
286 			return url;
287 		}
288 
289 		/**
290 		 * @return The file type.
291 		 */
292 		public String getType() {
293 			return (f.isDirectory() ? "dir" : "file");
294 		}
295 
296 		/**
297 		 * @return The file name.
298 		 */
299 		public String getName() {
300 			return f.getName();
301 		}
302 
303 		/**
304 		 * @return The file size.
305 		 */
306 		public long getSize() {
307 			return f.length();
308 		}
309 
310 		/**
311 		 * @return The file last modified timestamp.
312 		 */
313 		@BeanProperty(swap=DateSwap.ISO8601DTP.class)
314 		public Date getLastModified() {
315 			return new Date(f.lastModified());
316 		}
317 
318 		/**
319 		 * @return A hyperlink to view the contents of the file.
320 		 * @throws Exception If access is not allowed.
321 		 */
322 		public URL getView() throws Exception {
323 			if (allowViews && f.canRead() && ! f.isDirectory())
324 				return new URL(url + "?method=VIEW");
325 			return null;
326 		}
327 
328 		/**
329 		 * @return A hyperlink to download the contents of the file.
330 		 * @throws Exception If access is not allowed.
331 		 */
332 		public URL getDownload() throws Exception {
333 			if (allowViews && f.canRead() && ! f.isDirectory())
334 				return new URL(url + "?method=DOWNLOAD");
335 			return null;
336 		}
337 
338 		/**
339 		 * @return A hyperlink to delete the file.
340 		 * @throws Exception If access is not allowed.
341 		 */
342 		public URL getDelete() throws Exception {
343 			if (allowDeletes && f.canWrite())
344 				return new URL(url + "?method=DELETE");
345 			return null;
346 		}
347 	}
348 
349 	/** Utility method */
350 	private void deleteFile(File f) {
351 		try {
352 			if (f.isDirectory()) {
353 				File[] files = f.listFiles();
354 				if (files != null) {
355 					for (File fc : files)
356 						deleteFile(fc);
357 				}
358 			}
359 			f.delete();
360 		} catch (Exception e) {
361 			logger.log(WARNING, "Cannot delete file '" + f.getAbsolutePath() + "'", e);
362 		}
363 	}
