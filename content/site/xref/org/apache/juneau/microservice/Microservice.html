27  /**
28   * Parent class for all microservices.
29   * <p>
30   * A microservice defines a simple API for starting and stopping simple Java services
31   * 	contained in executable jars.
32   * <p>
33   * The general command for invoking these services is...
34   * <p class='bcode'>
35   * 	java -jar mymicroservice.jar [mymicroservice.cfg]
36   * </p>
37   * <p>
38   * Your microservice class must be specified as the <jk>Main-Class</jk> entry in
39   * 	the manifest file of your microservice jar file.
40   *
41   * <h6 class='topic'>Microservice Configuration</h6>
42   *
43   * This class defines the following method for accessing configuration for your microservice:
44   * <p>
45   * <ul class='spaced-list'>
46   * 	<li>{@link #getArgs()} - The command-line arguments passed to the jar file.
47   * 	<li>{@link #getConfig()} - An external INI-style configuration file.
48   * 	<li>{@link #getManifest()} - The manifest file for the main jar file.
49   * </ul>
50   *
51   * <h6 class='topic'>Entrypoint Method</h6>
52   *
53   * Subclasses must implement a static void main method as the entry point for the microservice.
54   * Typically, this method will simply consist of the following...
55   * <p>
56   * <p class='bcode'>
57   * 	<jk>public static void</jk> main(String[] args) <jk>throws</jk> Exception {
58   * 		<jk>new</jk> MyMicroservice(args).start();
59   * 	}
60   * </p>
61   *
62   * <h6 class='topic'>Lifecycle Methods</h6>
63   *
64   * Subclasses must implement the following lifecycle methods:
65   * <p>
66   * <ul class='spaced-list'>
67   * 	<li>{@link #start()} - Gets executed during startup.
68   * 	<li>{@link #stop()} - Gets executed when 'exit' is typed in the console or an external shutdown signal is received.
69   * 	<li>{@link #kill()} - Can be used to forcibly shut down the service.  Doesn't get called during normal operation.
70   * </ul>
71   *
72   * <h6 class='topic'>Lifecycle Listener Methods</h6>
73   *
74   * Subclasses can optionally implement the following event listener methods:
75   * <p>
76   * <ul class='spaced-list'>
77   * 	<li>{@link #onStart()} - Gets executed before {@link #start()}.
78   * 	<li>{@link #onStop()} - Gets executed before {@link #stop()}.
79   * 	<li>{@link #onConfigSave(ConfigFile)} - Gets executed after a config file has been saved.
80   * 	<li>{@link #onConfigChange(ConfigFile, Set)} - Gets executed after a config file has been modified.
81   * </ul>
82   *
83   * <h6 class='topic'>Other Methods</h6>
84   *
85   * Subclasses can optionally override the following methods to provide customized behavior:
86   * <p>
87   * <ul class='spaced-list'>
88   * 	<li>{@link #createVarResolver()} - Creates the {@link VarResolver} used to resolve variables in the config file returned by {@link #getConfig()}.
89   * </ul>
90   *
91   * @author james.bognar@salesforce.com
92   */
93  public abstract class Microservice {
94  
95  	private static Args args;
96  	private static ConfigFile cf;
97  	private static ManifestFile mf;

99  	/**
100 	 * Constructor.
101 	 *
102 	 * @param args2 Command line arguments.
103 	 * @throws Exception
104 	 */
105 	protected Microservice(String[] args2) throws Exception {
106 		Microservice.args = new Args(args2);
107 
108 		// --------------------------------------------------------------------------------
109 		// Try to get the manifest file.
110 		// --------------------------------------------------------------------------------
111 		Manifest m = new Manifest();
112 
113 		// If running within an eclipse workspace, need to get it from the file system.
114 		File f = new File("META-INF/MANIFEST.MF");
115 		if (f.exists()) {
116 			try {
117 				m.read(new FileInputStream(f));
118 			} catch (IOException e) {
119 				System.err.println("Problem detected in MANIFEST.MF.  Contents below:\n" + IOUtils.read(f));
120 				throw e;
121 			}
122 		} else {
123 			// Otherwise, read from manifest file in the jar file containing the main class.
124 			URLClassLoader cl = (URLClassLoader)getClass().getClassLoader();
125 			URL url = cl.findResource("META-INF/MANIFEST.MF");
126 			if (url != null) {
127 				try {
128 					m.read(url.openStream());
129 				} catch (IOException e) {
130 					System.err.println("Problem detected in MANIFEST.MF.  Contents below:\n" + IOUtils.read(url.openStream()));
131 					throw e;
132 				}
133 			}
134 		}
135 		mf = new ManifestFile(m);
136 
137 		// --------------------------------------------------------------------------------
138 		// Find config file.
139 		// Can either be passed in as first parameter, or we discover it using
140 		// the 'sun.java.command' system property.
141 		// --------------------------------------------------------------------------------
142 		String cFile = null;
143 		if (args.hasArg(0))
144 			cFile = args.getArg(0);
145 		else if (mf.containsKey("Main-ConfigFile"))
146 			cFile = mf.getString("Main-ConfigFile");
147 		else {
148 			String cmd = System.getProperty("sun.java.command", "not_found").split("\\s+")[0];
149 			if (cmd.endsWith(".jar"))
150 				cFile = cmd.replace(".jar", ".cfg");
151 		}
152 
153 		if (cFile == null) {
154 			System.err.println("Running class ["+getClass().getSimpleName()+"] without a config file.");
155 			cf = ConfigMgr.DEFAULT.create();
156 		} else {
157 			System.out.println("Running class ["+getClass().getSimpleName()+"] using config file ["+cFile+"]");
158 			System.setProperty("juneau.configFile", cFile);
159 			cf = ConfigMgr.DEFAULT.get(cFile).getResolving(createVarResolver());
160 		}
161 
162 		// --------------------------------------------------------------------------------
163 		// Set system properties.
164 		// --------------------------------------------------------------------------------
165 		Set<String> spKeys = cf.getSectionKeys("SystemProperties");
166 		if (spKeys != null)
167 			for (String key : spKeys)
168 				System.setProperty(key, cf.get("SystemProperties", key));
169 
170 		// --------------------------------------------------------------------------------
171 		// Add a config file change listener.
172 		// --------------------------------------------------------------------------------
173 		cf.addListener(new ConfigFileListener() {
174 			@Override /* ConfigFileListener */
175 			public void onSave(ConfigFile cf) {
176 				onConfigSave(cf);
177 			}
178 			@Override /* ConfigFileListener */
179 			public void onChange(ConfigFile cf, Set<String> changes) {
180 				onConfigChange(cf, changes);
181 			}
182 		});
183 	}

185 	/**
186 	 * Creates the {@link VarResolver} used to resolve variables in the
187 	 * config file returned by {@link #getConfig()}.
188 	 * <p>
189 	 * The default implementation resolves the following variables:
190 	 * <ul>
191 	 * 	<li><code>$S{key}</code>, <code>$S{key,default}</code> - System properties.
192 	 * 	<li><code>$E{key}</code>, <code>$E{key,default}</code> - Environment variables.
193 	 * 	<li><code>$C{key}</code>, <code>$C{key,default}</code> - Config file entries.
194 	 * 	<li><code>$MF{key}</code>, <code>$MF{key,default}</code> - Manifest file entries.
195 	 * 	<li><code>$ARG{key}</code>, <code>$ARG{key,default}</code> - Command-line arguments.
196 	 * </ul>
197 	 * <p>
198 	 * Subclasses can override this method to provide their own variables.
199 	 * <dl>
200 	 * 	<dt>Examples:</dt>
201 	 * 	<dd>
202 	 * 		<p class='bcode'>
203 	 * 	<jd>/**
204 	 * 	 * Augment default var resolver with a custom $B{...} variable that simply wraps strings inside square brackets.
205 	 * 	 * /</jd>
206 	 * 	<ja>@Override</ja> <jc>// Microservice</jc>
207 	 * 	<jk>protected</jk> StringVarResolver createVarResolver() {
208 	 * 		<jk>return super</jk>.createVarResolver()
209 	 * 			.addVar(<js>"B"</js>,
210 	 * 				<jk>new</jk> StringVarWithDefault() {
211 	 * 					<ja>@Override</ja> <jc>// StringVar</jc>
212 	 * 					<jk>public</jk> String resolve(String varVal) {
213 	 * 						<jk>return</jk> <js>'['</js> + varVal + <js>']'</js>;
214 	 * 					}
215 	 * 				}
216 	 * 			);
217 	 * 	}
218 	 * 		</p>
219 	 * 		<p class='bcode'>
220 	 * 	<cc># Example config file</cc>
221 	 * 	<cs>[MySection]</cs>
222 	 * 	<ck>myEntry</ck> = $B{foo}
223 	 * 		</p>
224 	 * 		<p class='bcode'>
225 	 * 	<jc>// Example java code</jc>
226 	 * 	String myentry = getConfig().getString(<js>"MySection/myEntry"</js>); <jc>// == "[foo]"</js>
227 	 * 		</p>
228 	 * 	</dd>
229 	 * </dl>
230 	 *
231 	 * @return A new {@link VarResolver}.
232 	 */
233 	protected VarResolver createVarResolver() {
234 		return new VarResolver()
235 			.addVars(SystemPropertiesVar.class, EnvVariablesVar.class, ConfigFileVar.class, ManifestFileVar.class, ArgsVar.class)
236 			.setContextObject(ConfigFileVar.SESSION_config, cf)
237 			.setContextObject(ManifestFileVar.SESSION_manifest, mf)
238 			.setContextObject(ArgsVar.SESSION_args, args);
239 	}

265 	/**
266 	 * Returns the external INI-style configuration file that can be used to configure your microservice.
267 	 * <p>
268 	 * The config file location is determined in the following order:
269 	 * <ol class='spaced-list'>
270 	 * 	<li>The first argument passed to the microservice jar.
271 	 * 	<li>The <code>Main-ConfigFile</code> entry in the microservice jar manifest file.
272 	 * 	<li>The name of the microservice jar with a <js>".cfg"</js> suffix (e.g. <js>"mymicroservice.jar"</js>-&gt;<js>"mymicroservice.cfg"</js>).
273 	 * </ol>
274 	 * <p>
275 	 * If all methods for locating the config file fail, then this method returns <jk>null</jk>.
276 	 * <p>
277 	 * Subclasses can set their own config file by calling the {@link #setConfig(ConfigFile)} method.
278 	 * <p>
279 	 * String variables defined by {@link #createVarResolver()} are automatically resolved when using this method.
280 	 * <p>
281 	 * This method can be called from the class constructor.
282 	 * <dl>
283 	 * 	<dt>Examples:</dt>
284 	 * 	<dd>
285 	 * 		<p class='bcode'>
286 	 * 	<cc>#--------------------------</cc>
287 	 * 	<cc># My section</cc>
288 	 * 	<cc>#--------------------------</cc>
289 	 * 	<cs>[MySection]</cs>
290 	 *
291 	 * 	<cc># An integer</cc>
292 	 * 	<ck>anInt</ck> = 1
293 	 *
294 	 * 	<cc># A boolean</cc>
295 	 * 	<ck>aBoolean</ck> = true
296 	 *
297 	 * 	<cc># An int array</cc>
298 	 * 	<ck>anIntArray</ck> = 1,2,3
299 	 *
300 	 * 	<cc># A POJO that can be converted from a String</cc>
301 	 * 	<ck>aURL</ck> = http://foo
302 	 *
303 	 * 	<cc># A POJO that can be converted from JSON</cc>
304 	 * 	<ck>aBean</ck> = {foo:'bar',baz:123}
305 	 *
306 	 * 	<cc># A system property</cc>
307 	 * 	<ck>locale</ck> = $S{java.locale, en_US}
308 	 *
309 	 * 	<cc># An environment variable</cc>
310 	 * 	<ck>path</ck> = $E{PATH, unknown}
311 	 *
312 	 * 	<cc># A manifest file entry</cc>
313 	 * 	<ck>mainClass</ck> = $MF{Main-Class}
314 	 *
315 	 * 	<cc># Another value in this config file</cc>
316 	 * 	<ck>sameAsAnInt</ck> = $C{MySection/anInt}
317 	 *
318 	 * 	<cc># A command-line argument in the form "myarg=foo"</cc>
319 	 * 	<ck>myArg</ck> = $ARG{myarg}
320 	 *
321 	 * 	<cc># The first command-line argument</cc>
322 	 * 	<ck>firstArg</ck> = $ARG{0}
323 	 *
324 	 * 	<cc># Look for system property, or env var if that doesn't exist, or command-line arg if that doesn't exist.</cc>
325 	 * 	<ck>nested</ck> = $S{mySystemProperty,$E{MY_ENV_VAR,$ARG{0}}}
326 	 *
327 	 * 	<cc># A POJO with embedded variables</cc>
328 	 * 	<ck>aBean2</ck> = {foo:'$ARG{0}',baz:$C{MySection/anInt}}
329 	 *
330 	 * 		</p>
331 	 * 		<p class='bcode'>
332 	 * 	<jc>// Java code for accessing config entries above.</jc>
333 	 * 	ConfigFile cf = getConfig();
334 	 *
335 	 * 	<jk>int</jk> anInt = cf.getInt(<js>"MySection/anInt"</js>);
336 	 * 	<jk>boolean</jk> aBoolean = cf.getBoolean(<js>"MySection/aBoolean"</js>);
337 	 * 	<jk>int</jk>[] anIntArray = cf.getObject(<jk>int</jk>[].<jk>class</jk>, <js>"MySection/anIntArray"</js>);
338 	 * 	URL aURL = cf.getObject(URL.<jk>class</jk>, <js>"MySection/aURL"</js>);
339 	 * 	MyBean aBean = cf.getObject(MyBean.<jk>class</jk>, <js>"MySection/aBean"</js>);
340 	 * 	Locale locale = cf.getObject(Locale.<jk>class</jk>, <js>"MySection/locale"</js>);
341 	 * 	String path = cf.getString(<js>"MySection/path"</js>);
342 	 * 	String mainClass = cf.getString(<js>"MySection/mainClass"</js>);
343 	 * 	<jk>int</jk> sameAsAnInt = cf.getInt(<js>"MySection/sameAsAnInt"</js>);
344 	 * 	String myArg = cf.getString(<js>"MySection/myArg"</js>);
345 	 * 	String firstArg = cf.getString(<js>"MySection/firstArg"</js>);
346 	 * 		</p>
347 	 * 	</dd>
348 	 * </dl>
349 	 *
350 	 * @return The config file for this application, or <jk>null</jk> if no config file is configured.
351 	 */
352 	protected static ConfigFile getConfig() {
353 		return cf;
354 	}

397 	/**
398 	 * Start this application.
399 	 * <p>
400 	 * Default implementation simply calls {@link #onStart()}.
401 	 * <p>
402 	 * Overridden methods MUST call this method FIRST so that the {@link #onStart()} method is called.
403 	 *
404 	 * @return This object (for method chaining).
405 	 * @throws Exception
406 	 */
407 	protected Microservice start() throws Exception {
408 		// --------------------------------------------------------------------------------
409 		// Add exit listeners.
410 		// --------------------------------------------------------------------------------
411 		new Thread() {
412 			@Override /* Thread */
413 			public void run() {
414 				Console c = System.console();
415 				if (c == null)
416 					System.out.println("No available console.");
417 				else {
418 					while (true) {
419 						String l = c.readLine("\nEnter 'exit' to exit.\n");
420 						if (l == null || l.equals("exit")) {
421 							Microservice.this.stop();
422 							break;
423 						}
424 					}
425 				}
426 			}
427 		}.start();
428 		Runtime.getRuntime().addShutdownHook(
429 			new Thread() {
430 				@Override /* Thread */
431 				public void run() {
432 					Microservice.this.stop();
433 				}
434 			}
435 		);
436 		onStart();
437 		return this;
438 	}
