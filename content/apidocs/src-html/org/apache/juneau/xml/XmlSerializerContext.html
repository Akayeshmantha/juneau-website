013package org.apache.juneau.xml;
014
015import org.apache.juneau.*;
016import org.apache.juneau.serializer.*;
017
018/**
019 * Configurable properties on the {@link XmlSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * The following convenience methods are also provided for setting context properties:
025 * <ul>
026 *    <li>{@link XmlSerializer#setProperty(String,Object)}
027 *    <li>{@link XmlSerializer#setProperties(ObjectMap)}
028 *    <li>{@link XmlSerializer#addNotBeanClasses(Class[])}
029 *    <li>{@link XmlSerializer#addBeanFilters(Class[])}
030 *    <li>{@link XmlSerializer#addPojoSwaps(Class[])}
031 *    <li>{@link XmlSerializer#addImplClass(Class,Class)}
032 * </ul>
033 * <p>
034 * See {@link ContextFactory} for more information about context properties.
035 *
036 * @author James Bognar (james.bognar@salesforce.com)
037 */
038public class XmlSerializerContext extends SerializerContext {
039
040   /**
041    * Add JSON type attributes to output ({@link Boolean}, default=<jk>false</jk>).
042    * <p>
043    * If <js>true</jk>, {@code type} attributes will be added to elements in the XML for number/boolean/null nodes.
044    */
045   public static final String XML_addJsonTypeAttrs = "XmlSerializer.addJsonTypeAttrs";
046
047   /**
048    * Add JSON type attributes for strings to output ({@link Boolean}, default=<jk>false</jk>).
049    * <p>
050    * If <jk>true</jk>, {@code type} attributes will be added to elements in the XML for string nodes.
051    * <p>
052    * By default, these attributes are not added, and the parser will assume that the content type
053    * of the node is string by default.
054    * <p>
055    * This feature is disabled if {@link #XML_addJsonTypeAttrs} is disabled.
056    */
057   public static final String XML_addJsonStringTypeAttrs = "XmlSerializer.addJsonStringTypeAttrs";
058
059   /**
060    * Enable support for XML namespaces ({@link Boolean}, default=<jk>true</jk>).
061    * <p>
062    * If not enabled, XML output will not contain any namespaces regardless of any other settings.
063    */
064   public static final String XML_enableNamespaces = "XmlSerializer.enableNamespaces";
065
066   /**
067    * Auto-detect namespace usage ({@link Boolean}, default=<jk>true</jk>).
068    * <p>
069    * Detect namespace usage before serialization.
070    * <p>
071    * Used in conjunction with {@link #XML_addNamespaceUrisToRoot} to reduce
072    * the list of namespace URLs appended to the root element to only those
073    * that will be used in the resulting document.
074    * <p>
075    * If enabled, then the data structure will first be crawled looking for
076    * namespaces that will be encountered before the root element is
077    * serialized.
078    * <p>
079    * This setting is ignored if {@link #XML_enableNamespaces} is not enabled.
080    * <p>
081    * <b>IMPORTANT NOTE:</b>
082    * Auto-detection of namespaces can be costly performance-wise.
083    * In high-performance environments, it's recommended that namespace detection be
084    *    disabled, and that namespaces be manually defined through the {@link #XML_namespaces} property.
085    */
086   public static final String XML_autoDetectNamespaces = "XmlSerializer.autoDetectNamespaces";
087
088   /**
089    * Add namespace URLs to the root element ({@link Boolean}, default=<jk>true</jk>).
090    * <p>
091    * Use this setting to add {@code xmlns:x} attributes to the root
092    * element for the default and all mapped namespaces.
093    * <p>
094    * This setting is ignored if {@link #XML_enableNamespaces} is not enabled.
095    */
096   public static final String XML_addNamespaceUrisToRoot = "XmlSerializer.addNamespaceUrisToRoot";
097
098   /**
099    * Default namespace URI ({@link String}, default=<jk>null</jk>).
100    * <p>
101    * Specifies the default namespace URI for this document.
102    */
103   public static final String XML_defaultNamespaceUri = "XmlSerializer.defaultNamespaceUri";
104
105   /**
106    * XMLSchema namespace ({@link Namespace}, default=<code>{name:<js>'xs'</js>,uri:<js>'http://www.w3.org/2001/XMLSchema'</js>}</code>).
107    * <p>
108    * Specifies the namespace for the <code>XMLSchema</code> namespace, used by the schema generated
109    * by the {@link XmlSchemaSerializer} class.
110    */
111   public static final String XML_xsNamespace = "XmlSerializer.xsNamespace";
112
113   /**
114    * XMLSchema-Instance namespace ({@link Namespace}, default=<code>{name:<js>'xsi'</js>,uri:<js>'http://www.w3.org/2001/XMLSchema-instance'</js>}</code>).
115    * <p>
116    * Specifies the namespace of the <code>XMLSchema-instance</code> namespace used for<code>nil=<jk>true</jk></code> attributes.
117    */
118   public static final String XML_xsiNamespace = "XmlSerializer.xsiNamespace";
119
120   /**
121    * Default namespaces (<code>Set&lt;Namespace&gt;</code>, default=empty set).
122    * <p>
123    * The default list of namespaces associated with this serializer.
124    */
125   public static final String XML_namespaces = "XmlSerializer.namespaces";
126
127
128   final boolean
129      addJsonTypeAttrs,
130      addJsonStringTypeAttrs,
131      autoDetectNamespaces,
132      enableNamespaces,
133      addNamespaceUrlsToRoot;
134
135   final String defaultNamespace;
136
137   final Namespace
138      xsiNamespace,
139      xsNamespace;
140
141   final Namespace[] namespaces;
142
143   /**
144    * Constructor.
145    * <p>
146    * Typically only called from {@link ContextFactory#getContext(Class)}.
147    *
148    * @param cf The factory that created this context.
149    */
150   public XmlSerializerContext(ContextFactory cf) {
151      super(cf);
152      addJsonTypeAttrs = cf.getProperty(XML_addJsonTypeAttrs, boolean.class, false);
153      addJsonStringTypeAttrs = cf.getProperty(XML_addJsonStringTypeAttrs, boolean.class, false);
154      autoDetectNamespaces = cf.getProperty(XML_autoDetectNamespaces, boolean.class, true);
155      enableNamespaces = cf.getProperty(XML_enableNamespaces, boolean.class, true);
156      addNamespaceUrlsToRoot = cf.getProperty(XML_addNamespaceUrisToRoot, boolean.class, true);
157      defaultNamespace = cf.getProperty(XML_defaultNamespaceUri, String.class, "{juneau:'http://www.ibm.com/2013/Juneau'}");
158      xsNamespace = cf.getProperty(XML_xsNamespace, Namespace.class, new Namespace("xs", "http://www.w3.org/2001/XMLSchema"));
159      xsiNamespace = cf.getProperty(XML_xsiNamespace, Namespace.class, new Namespace("xsi", "http://www.w3.org/2001/XMLSchema-instance"));
160      namespaces = cf.getProperty(XML_namespaces, Namespace[].class, new Namespace[0]);
161   }
162}
