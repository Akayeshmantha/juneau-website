029/**
030 * Serializes POJO models to XML.
031 *
032 *
033 * <h6 class='topic'>Media types</h6>
034 * <p>
035 *    Handles <code>Accept</code> types: <code>text/xml</code>
036 * <p>
037 *    Produces <code>Content-Type</code> types: <code>text/xml</code>
038 *
039 *
040 * <h6 class='topic'>Description</h6>
041 * <p>
042 *    See the {@link JsonSerializer} class for details on how Java models map to JSON.
043 * <p>
044 *    For example, the following JSON...
045 * <p class='bcode'>
046 *    {
047 *       name:<js>'John Smith'</js>,
048 *       address: {
049 *          streetAddress: <js>'21 2nd Street'</js>,
050 *          city: <js>'New York'</js>,
051 *          state: <js>'NY'</js>,
052 *          postalCode: <js>10021</js>
053 *       },
054 *       phoneNumbers: [
055 *          <js>'212 555-1111'</js>,
056 *          <js>'212 555-2222'</js>
057 *       ],
058 *       additionalInfo: <jk>null</jk>,
059 *       remote: <jk>false</jk>,
060 *       height: <js>62.4</js>,
061 *       <js>'fico score'</js>:  <js>' &gt; 640'</js>
062 *    }
063 * <p>
064 *    ...maps to the following XML using the default serializer...
065 * <p class='bcode'>
066 *    <xt>&lt;object&gt;</xt>
067 *       <xt>&lt;name&gt;</xt>John Smith<xt>&lt;/name&gt;</xt>
068 *       <xt>&lt;address&gt;</xt>
069 *          <xt>&lt;streetAddress&gt;</xt>21 2nd Street<xt>&lt;/streetAddress&gt;</xt>
070 *          <xt>&lt;city&gt;</xt>New York<xt>&lt;/city&gt;</xt>
071 *          <xt>&lt;state&gt;</xt>NY<xt>&lt;/state&gt;</xt>
072 *          <xt>&lt;postalCode&gt;</xt>10021<xt>&lt;/postalCode&gt;</xt>
073 *       <xt>&lt;/address&gt;</xt>
074 *       <xt>&lt;phoneNumbers&gt;</xt>
075 *          <xt>&lt;string&gt;</xt>212 555-1111<xt>&lt;/string&gt;</xt>
076 *          <xt>&lt;string&gt;</xt>212 555-2222<xt>&lt;/string&gt;</xt>
077 *       <xt>&lt;/phoneNumbers&gt;</xt>
078 *       <xt>&lt;additionalInfo</xt> <xa>_type</xa>=<xs>'null'</xs><xt>&gt;&lt;/additionalInfo&gt;</xt>
079 *       <xt>&lt;remote&gt;</xt>false<xt>&lt;/remote&gt;</xt>
080 *       <xt>&lt;height&gt;</xt>62.4<xt>&lt;/height&gt;</xt>
081 *       <xt>&lt;fico_x0020_score&gt;</xt> &amp;gt; 640<xt>&lt;/fico_x0020_score&gt;</xt>
082 *    <xt>&lt;/object&gt;</xt>
083 * <p>
084 *    An additional "add-json-properties" mode is also provided to prevent loss of JSON data types...
085 * <p class='bcode'>
086 *    <xt>&lt;object&gt;</xt>
087 *       <xt>&lt;name</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt>John Smith<xt>&lt;/name&gt;</xt>
088 *       <xt>&lt;address</xt> <xa>_type</xa>=<xs>'object'</xs><xt>&gt;</xt>
089 *          <xt>&lt;streetAddress</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt>21 2nd Street<xt>&lt;/streetAddress&gt;</xt>
090 *          <xt>&lt;city</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt>New York<xt>&lt;/city&gt;</xt>
091 *          <xt>&lt;state</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt>NY<xt>&lt;/state&gt;</xt>
092 *          <xt>&lt;postalCode</xt> <xa>_type</xa>=<xs>'number'</xs><xt>&gt;</xt>10021<xt>&lt;/postalCode&gt;</xt>
093 *       <xt>&lt;/address&gt;</xt>
094 *       <xt>&lt;phoneNumbers</xt> <xa>_type</xa>=<xs>'array'</xs><xt>&gt;</xt>
095 *          <xt>&lt;string&gt;</xt>212 555-1111<xt>&lt;/string&gt;</xt>
096 *          <xt>&lt;string&gt;</xt>212 555-2222<xt>&lt;/string&gt;</xt>
097 *       <xt>&lt;/phoneNumbers&gt;</xt>
098 *       <xt>&lt;additionalInfo</xt> <xa>_type</xa>=<xs>'null'</xs><xt>&gt;&lt;/additionalInfo&gt;</xt>
099 *       <xt>&lt;remote</xt> <xa>_type</xa>=<xs>'boolean'</xs><xt>&gt;</xt>false<xt>&lt;/remote&gt;</xt>
100 *       <xt>&lt;height</xt> <xa>_type</xa>=<xs>'number'</xs><xt>&gt;</xt>62.4<xt>&lt;/height&gt;</xt>
101 *       <xt>&lt;fico_x0020_score</xt> <xa>_type</xa>=<xs>'string'</xs><xt>&gt;</xt> &amp;gt; 640<xt>&lt;/fico_x0020_score&gt;</xt>
102 *    <xt>&lt;/object&gt;</xt>
103 * <p>
104 *    This serializer provides several serialization options.  Typically, one of the predefined <jsf>DEFAULT</jsf> serializers will be sufficient.
105 *    However, custom serializers can be constructed to fine-tune behavior.
106 * <p>
107 *    If an attribute name contains any non-valid XML element characters, they will be escaped using standard {@code _x####_} notation.
108 *
109 *
110 * <h6 class='topic'>Configurable properties</h6>
111 * <p>
112 *    This class has the following properties associated with it:
113 * <ul>
114 *    <li>{@link XmlSerializerContext}
115 *    <li>{@link BeanContext}
116 * </ul>
117 *
118 *
119 * <h6 class='topic'>Behavior-specific subclasses</h6>
120 * <p>
121 *    The following direct subclasses are provided for convenience:
122 * <ul class='spaced-list'>
123 *    <li>{@link Sq} - Default serializer, single quotes.
124 *    <li>{@link SqReadable} - Default serializer, single quotes, whitespace added.
125 *    <li>{@link XmlJson} - Default serializer with JSON attribute tags.
126 *    <li>{@link XmlJsonSq} - Default serializer with JSON attribute tags, single quotes.
127 * </ul>
128 *
129 *
130 * @author James Bognar (james.bognar@salesforce.com)
131 */

219   /**
220    * Recursively searches for the XML namespaces on the specified POJO and adds them to the serializer context object.
221    *
222    * @param session The context that exists for the duration of a single serialization.
223    * @param o The POJO to check.
224    * @throws SerializeException
225    */
226   protected void findNsfMappings(XmlSerializerSession session, Object o) throws SerializeException {
227      BeanContext bc = session.getBeanContext();
228      ClassMeta<?> aType = null;                // The actual type
229      aType = session.push(null, o, null);
230
231      if (aType != null) {
232         Namespace ns = aType.getExtendedMeta(XmlClassMeta.class).getNamespace();
233         if (ns != null) {
234            if (ns.uri != null)
235               session.addNamespace(ns);
236            else
237               ns = null;
238         }
239      }
240
241      // Handle recursion
242      if (aType != null && ! aType.isPrimitive()) {
243
244         BeanMap<?> bm = null;
245         if (aType.isBeanMap()) {
246            bm = (BeanMap)o;
247         } else if (aType.isBean()) {
248            bm = bc.forBean(o);
249         } else if (aType.isDelegate()) {
250            ClassMeta<?> innerType = ((Delegate)o).getClassMeta();
251            Namespace ns = innerType.getExtendedMeta(XmlClassMeta.class).getNamespace();
252            if (ns != null) {
253               if (ns.uri != null)
254                  session.addNamespace(ns);
255               else
256                  ns = null;
257            }
258
259            if (innerType.isBean()) {
260               for (BeanPropertyMeta bpm : innerType.getBeanMeta().getPropertyMetas()) {
261                  ns = bpm.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace();
262                  if (ns != null && ns.uri != null)
263                     session.addNamespace(ns);
264               }
265
266            } else if (innerType.isMap()) {
267               for (Object o2 : ((Map)o).values())
268                  findNsfMappings(session, o2);
269            } else if (innerType.isCollection()) {
270               for (Object o2 : ((Collection)o))
271                  findNsfMappings(session, o2);
272            }
273
274         } else if (aType.isMap()) {
275            for (Object o2 : ((Map)o).values())
276               findNsfMappings(session, o2);
277         } else if (aType.isCollection()) {
278            for (Object o2 : ((Collection)o))
279               findNsfMappings(session, o2);
280         } else if (aType.isArray() && ! aType.getElementType().isPrimitive()) {
281            for (Object o2 : ((Object[])o))
282               findNsfMappings(session, o2);
283         }
284         if (bm != null) {
285            for (BeanPropertyValue p : bm.getValues(session.isTrimNulls())) {
286
287               Namespace ns = p.getMeta().getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace();
288               if (ns != null && ns.uri != null)
289                  session.addNamespace(ns);
290
291               try {
292                  findNsfMappings(session, p.getValue());
293               } catch (Throwable x) {
294                  // Ignore
295               }
296            }
297         }
298      }
299
300      session.pop();
301   }

303   /**
304    * Workhorse method.
305    *
306    * @param session The serializer context.
307    * @param out The writer to send the output to.
308    * @param o The object to serialize.
309    * @param eType The expected type if this is a bean property value being serialized.
310    * @param elementName The root element name.
311    * @param elementNamespace The namespace of the element.
312    * @param addNamespaceUris Flag indicating that namespace URIs need to be added.
313    * @param format The format to serialize the output to.
314    * @param pMeta The bean property metadata if this is a bean property being serialized.
315    * @return The same writer passed in so that calls to the writer can be chained.
316    * @throws Exception If a problem occurred trying to convert the output.
317    */
318   protected XmlWriter serializeAnything(XmlSerializerSession session, XmlWriter out, Object o,
319         ClassMeta eType, String elementName, Namespace elementNamespace, boolean addNamespaceUris,
320         XmlFormat format, BeanPropertyMeta pMeta) throws Exception {
321
322      BeanContext bc = session.getBeanContext();
323      String ts = null;              // The type string (e.g. <type> or <x x='type'>
324      int indent = session.indent;       // Current indentation
325      ClassMeta<?> aType = null;     // The actual type
326      ClassMeta<?> wType = null;     // The wrapped type
327      ClassMeta<?> sType = object(); // The serialized type
328
329      aType = session.push(elementName, o, eType);
330
331      if (eType == null)
332         eType = object();
333
334      // Handle recursion
335      if (aType == null) {
336         o = null;
337         aType = object();
338      }
339
340      if (o != null) {
341
342         if (aType.isDelegate()) {
343            wType = aType;
344            aType = ((Delegate)o).getClassMeta();
345         }
346
347         sType = aType.getSerializedClassMeta();
348
349         // Swap if necessary
350         PojoSwap swap = aType.getPojoSwap();
351         if (swap != null) {
352            o = swap.swap(o, bc);
353
354            // If the getSwapClass() method returns Object, we need to figure out
355            // the actual type now.
356            if (sType.isObject())
357               sType = bc.getClassMetaForObject(o);
358         }
359      } else {
360         sType = eType.getSerializedClassMeta();
361      }
362
363      String typeName = null;
364      if (session.isAddBeanTypeProperties()) {
365         if (o != null && ! eType.equals(aType))
366            typeName = aType.getDictionaryName();
367      }
368
369      // char '\0' is interpreted as null.
370      if (o != null && sType.isChar() && ((Character)o).charValue() == 0)
371         o = null;
372
373      boolean isCollapsed = false;     // If 'true', this is a collection and we're not rendering the outer element.
374
375      // Get the JSON type string.
376      if (sType.isCharSequence() || sType.isChar())
377         ts = "string";
378      else if (sType.isNumber())
379         ts = "number";
380      else if (sType.isBoolean())
381         ts = "boolean";
382      else if (sType.isMap() || sType.isBean() || sType.hasToObjectMapMethod()) {
383         isCollapsed = sType.getExtendedMeta(XmlClassMeta.class).getFormat() == XmlFormat.COLLAPSED;
384         ts = "object";
385      }
386      else if (sType.isCollection() || sType.isArray()) {
387         isCollapsed = (format == COLLAPSED && ! addNamespaceUris);
388         ts = "array";
389      }
390      else
391         ts = "string";
392
393      // Is there a name associated with this bean?
394      if (elementName == null)
395         elementName = sType.getDictionaryName();
396
397      // If the value is null then it's either going to be <null/> or <XmlSerializer nil='true'/>
398      // depending on whether the element has a name.
399      boolean isNullTag = (elementName == null && o == null);
400
401      if (isNullTag)
402         ts = "null";
403
404      if (session.isEnableNamespaces()) {
405         if (elementNamespace == null)
406            elementNamespace = sType.getExtendedMeta(XmlClassMeta.class).getNamespace();
407         if (elementNamespace == null)
408            elementNamespace = aType.getExtendedMeta(XmlClassMeta.class).getNamespace();
409         if (elementNamespace != null && elementNamespace.uri == null)
410            elementNamespace = null;
411         if (elementNamespace == null)
412            elementNamespace = session.getDefaultNamespace();
413      } else {
414         elementNamespace = null;
415      }
416
417      // Do we need a carriage return after the start tag?
418      boolean cr = o != null && (sType.isMap() || sType.isCollection() || sType.isArray() || sType.isBean() || sType.hasToObjectMapMethod());
419
420      String en = (elementName == null ? ts : elementName);
421      boolean encodeEn = elementName != null;
422      String ns = (elementNamespace == null ? null : elementNamespace.name);
423      String xsi = null, dns = null, elementNs = null;
424      if (session.isEnableNamespaces()) {
425         xsi = session.getXsiNamespace().name;
426         dns = elementName == null && session.getDefaultNamespace() != null ? session.getDefaultNamespace().name : null;
427         elementNs = elementName == null ? dns : ns;
428         if (elementName == null)
429            elementNamespace = null;
430      }
431
432      // Render the start tag.
433      if (! isCollapsed) {
434         out.oTag(indent, elementNs, en, encodeEn);
435         if (addNamespaceUris) {
436            out.attr((String)null, "xmlns", session.getDefaultNamespace().getUri());
437
438            for (Namespace n : session.getNamespaces())
439               out.attr("xmlns", n.getName(), n.getUri());
440
441            Namespace xsiNs = session.getXsiNamespace();
442            if (xsiNs != null)
443               out.attr("xmlns", xsiNs.name, xsiNs.uri);
444         }
445         if (typeName == null && elementName != null && session.isAddJsonTypeAttrs() && (session.isAddJsonStringTypeAttrs() || ! ts.equals("string")))
446            typeName = ts;
447         if (typeName != null && ! typeName.equals(elementName))
448            out.attr(dns, bc.getBeanTypePropertyName(), typeName);
449         if (o == null) {
450            if (! isNullTag)
451               out.attr(xsi, "nil", "true");
452            if ((sType.isBoolean() || sType.isNumber()) && ! sType.isNullable())
453               o = sType.getPrimitiveDefault();
454         }
455
456         if (o != null && !(sType.isMap() || sType.isBean() || sType.hasToObjectMapMethod()))
457            out.append('>');
458
459         if (cr && !(sType.isMap() || sType.isBean() || sType.hasToObjectMapMethod()))
460            out.nl();
461      }
462
463      boolean hasChildren = true;
464
465      // Render the tag contents.
466      if (o != null) {
467         if (sType.isUri() || (pMeta != null && pMeta.isUri()))
468            out.appendUri(o);
469         else if (sType.isCharSequence() || sType.isChar())
470            out.encodeText(session.trim(o));
471         else if (sType.isNumber() || sType.isBoolean())
472            out.append(o);
473         else if (sType.isMap() || (wType != null && wType.isMap())) {
474            if (o instanceof BeanMap)
475               hasChildren = serializeBeanMap(session, out, (BeanMap)o, elementNamespace, isCollapsed);
476            else
477               hasChildren = serializeMap(session, out, (Map)o, sType);
478         }
479         else if (sType.hasToObjectMapMethod())
480            hasChildren = serializeMap(session, out, sType.toObjectMap(o), sType);
481         else if (sType.isBean())
482            hasChildren = serializeBeanMap(session, out, bc.forBean(o), elementNamespace, isCollapsed);
483         else if (sType.isCollection() || (wType != null && wType.isCollection())) {
484            if (isCollapsed)
485               session.indent--;
486            serializeCollection(session, out, (Collection)o, sType, pMeta);
487            if (isCollapsed)
488               session.indent++;
489         }
490         else if (sType.isArray()) {
491            if (isCollapsed)
492               session.indent--;
493            serializeCollection(session, out, toList(sType.getInnerClass(), o), sType, pMeta);
494            if (isCollapsed)
495               session.indent++;
496         }
497         else
498            out.encodeText(session.toString(o));
499      }
500
501      session.pop();
502
503      // Render the end tag.
504      if (! isCollapsed) {
505         if (o == null || ! hasChildren)
506            out.append('/').append('>').nl();
507         else
508            out.i(cr ? indent : 0).eTag(elementNs, en, encodeEn).nl();
509      }
510
511      return out;
512   }

514   private boolean serializeMap(XmlSerializerSession session, XmlWriter out, Map m, ClassMeta<?> type) throws Exception {
515
516      m = session.sort(m);
517
518      ClassMeta<?> keyType = type.getKeyType(), valueType = type.getValueType();
519
520      boolean hasChildren = false;
521      for (Iterator i = m.entrySet().iterator(); i.hasNext();) {
522         Map.Entry e = (Map.Entry)i.next();
523
524         Object k = e.getKey();
525         if (k == null) {
526            k = "\u0000";
527         } else {
528            k = session.generalize(k, keyType);
529            if (session.isTrimStrings() && k instanceof String)
530               k = k.toString().trim();
531         }
532
533         Object value = e.getValue();
534
535         if (! hasChildren) {
536            hasChildren = true;
537            out.append('>').nl();
538         }
539         serializeAnything(session, out, value, valueType, session.toString(k), null, false, NORMAL, null);
540      }
541      return hasChildren;
542   }

544   private boolean serializeBeanMap(XmlSerializerSession session, XmlWriter out, BeanMap<?> m, Namespace elementNs, boolean isCollapsed) throws Exception {
545      boolean hasChildren = false;
546      BeanMeta<?> bm = m.getMeta();
547
548      List<BeanPropertyValue> lp = m.getValues(session.isTrimNulls());
549
550      Map<String,BeanPropertyMeta> xmlAttrs = bm.getExtendedMeta(XmlBeanMeta.class).getXmlAttrProperties();
551      Object content = null;
552      for (BeanPropertyValue p : lp) {
553         if (xmlAttrs.containsKey(p.getName())) {
554            BeanPropertyMeta pMeta = p.getMeta();
555            ClassMeta<?> cMeta = p.getClassMeta();
556
557            String key = p.getName();
558            Object value = p.getValue();
559            Throwable t = p.getThrown();
560            if (t != null)
561               session.addBeanGetterWarning(pMeta, t);
562
563            if (session.canIgnoreValue(cMeta, key, value))
564               continue;
565
566            Namespace ns = (session.isEnableNamespaces() && pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace() != elementNs ? pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace() : null);
567
568            if (pMeta.isUri())
569               out.attrUri(ns, key, value);
570            else
571               out.attr(ns, key, value);
572         }
573      }
574
575      boolean hasContent = false;
576
577      for (BeanPropertyValue p : lp) {
578         BeanPropertyMeta pMeta = p.getMeta();
579         ClassMeta<?> cMeta = p.getClassMeta();
580
581         XmlFormat xf = pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getXmlFormat();
582
583         if (xf == CONTENT) {
584            content = p.getValue();
585            hasContent = true;
586         } else if (xf == ATTR) {
587            // Do nothing
588         } else {
589            String key = p.getName();
590            Object value = p.getValue();
591            Throwable t = p.getThrown();
592            if (t != null)
593               session.addBeanGetterWarning(pMeta, t);
594
595            if (session.canIgnoreValue(cMeta, key, value))
596               continue;
597
598            if (! hasChildren) {
599               hasChildren = true;
600               out.appendIf(! isCollapsed, '>').nl();
601            }
602            serializeAnything(session, out, value, cMeta, key, pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace(), false, pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getXmlFormat(), pMeta);
603         }
604      }
605      if ((! hasContent) || session.canIgnoreValue(string(), null, content))
606         return hasChildren;
607      out.append('>').cr(session.indent);
608
609      // Serialize XML content.
610      XmlContentHandler h = bm.getExtendedMeta(XmlBeanMeta.class).getXmlContentHandler();
611      if (h != null)
612         h.serialize(out, m.getBean());
613      else
614         out.encodeText(content);
615      out.nl();
616      return true;
617   }
618
619   private XmlWriter serializeCollection(XmlSerializerSession session, XmlWriter out, Collection c, ClassMeta<?> type, BeanPropertyMeta ppMeta) throws Exception {
620
621      c = session.sort(c);
622
623      ClassMeta<?> elementType = type.getElementType();
624
625      String eName = null;
626      Namespace eNs = null;
627
628      if (ppMeta != null) {
629         eName = ppMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getChildName();
630         eNs = ppMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getNamespace();
631      }
632
633      if (eName == null) {
634         eName = type.getExtendedMeta(XmlClassMeta.class).getChildName();
635         eNs = type.getExtendedMeta(XmlClassMeta.class).getNamespace();
636      }
637
638      if (eName == null && ! elementType.isObject()) {
639         eName = elementType.getDictionaryName();
640         eNs = elementType.getExtendedMeta(XmlClassMeta.class).getNamespace();
641      }
642
643      for (Iterator i = c.iterator(); i.hasNext();) {
644         Object value = i.next();
645         serializeAnything(session, out, value, elementType, eName, eNs, false, NORMAL, null);
646      }
647      return out;
648   }
