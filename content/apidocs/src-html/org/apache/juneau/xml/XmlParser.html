030/**
031 * Parses text generated by the {@link XmlSerializer} class back into a POJO model.
032 *
033 *
034 * <h6 class='topic'>Media types</h6>
035 * <p>
036 *    Handles <code>Content-Type</code> types: <code>text/xml</code>
037 *
038 *
039 * <h6 class='topic'>Description</h6>
040 * <p>
041 *    See the {@link XmlSerializer} class for a description of Juneau-generated XML.
042 *
043 *
044 * <h6 class='topic'>Configurable properties</h6>
045 * <p>
046 *    This class has the following properties associated with it:
047 * <ul>
048 *    <li>{@link XmlParserContext}
049 *    <li>{@link BeanContext}
050 * </ul>
051 *
052 *
053 * @author James Bognar (james.bognar@salesforce.com)
054 */
055@SuppressWarnings({ "rawtypes", "unchecked" })
056@Consumes({"text/xml","application/xml"})
057public class XmlParser extends ReaderParser {

062   private static final int UNKNOWN=0, OBJECT=1, ARRAY=2, STRING=3, NUMBER=4, BOOLEAN=5, NULL=6;

065   private <T> T parseAnything(XmlParserSession session, ClassMeta<T> nt, String currAttr, XMLStreamReader r, Object outer, boolean isRoot) throws Exception {
066
067      BeanContext bc = session.getBeanContext();
068      if (nt == null)
069         nt = (ClassMeta<T>)object();
070      PojoSwap<T,Object> transform = (PojoSwap<T,Object>)nt.getPojoSwap();
071      ClassMeta<?> ft = nt.getSerializedClassMeta();
072      session.setCurrentClass(ft);
073
074      String wrapperAttr = (isRoot && session.isPreserveRootElement()) ? r.getName().getLocalPart() : null;
075      String typeAttr = r.getAttributeValue(null, "type");
076      int jsonType = getJsonType(typeAttr);
077      String b = r.getAttributeValue(session.getXsiNs(), "nil");
078      if (b == null)
079         b = r.getAttributeValue(null, "nil");
080      boolean isNull = b != null && b.equals("true");
081      if (jsonType == 0) {
082         String elementName = session.decodeString(r.getLocalName());
083         if (elementName == null || elementName.equals(currAttr))
084            jsonType = UNKNOWN;
085         else
086            jsonType = getJsonType(elementName);
087      }
088      if (! ft.canCreateNewInstance(outer)) {
089         String c = r.getAttributeValue(null, "_class");
090         if (c != null) {
091            ft = nt = (ClassMeta<T>)bc.getClassMetaFromString(c);
092         }
093      }
094      Object o = null;
095
096      if (jsonType == NULL) {
097         r.nextTag();   // Discard end tag
098         return null;
099      }
100      if (isNull) {
101         while (true) {
102            int e = r.next();
103            if (e == END_ELEMENT)
104               return null;
105         }
106      }
107
108      if (ft.isObject()) {
109         if (jsonType == OBJECT) {
110            ObjectMap m = new ObjectMap(bc);
111            parseIntoMap(session, r, m, string(), object());
112            if (wrapperAttr != null)
113               m = new ObjectMap(bc).append(wrapperAttr, m);
114            o = m.cast();
115         } else if (jsonType == ARRAY)
116            o = parseIntoCollection(session, r, new ObjectList(bc), object());
117         else if (jsonType == STRING) {
118            o = session.decodeString(r.getElementText());
119            if (ft.isChar())
120               o = o.toString().charAt(0);
121         }
122         else if (jsonType == NUMBER)
123            o = parseNumber(session.decodeLiteral(r.getElementText()), null);
124         else if (jsonType == BOOLEAN)
125            o = Boolean.parseBoolean(session.decodeLiteral(r.getElementText()));
126         else if (jsonType == UNKNOWN)
127            o = getUnknown(session, r);
128      } else if (ft.isBoolean()) {
129         o = Boolean.parseBoolean(session.decodeLiteral(r.getElementText()));
130      } else if (ft.isCharSequence()) {
131         o = session.decodeString(r.getElementText());
132      } else if (ft.isChar()) {
133         o = session.decodeString(r.getElementText()).charAt(0);
134      } else if (ft.isMap()) {
135         Map m = (ft.canCreateNewInstance(outer) ? (Map)ft.newInstance(outer) : new ObjectMap(bc));
136         o = parseIntoMap(session, r, m, ft.getKeyType(), ft.getValueType());
137         if (wrapperAttr != null)
138            o = new ObjectMap(bc).append(wrapperAttr, m);
139      } else if (ft.isCollection()) {
140         Collection l = (ft.canCreateNewInstance(outer) ? (Collection)ft.newInstance(outer) : new ObjectList(bc));
141         o = parseIntoCollection(session, r, l, ft.getElementType());
142      } else if (ft.isNumber()) {
143         o = parseNumber(session.decodeLiteral(r.getElementText()), (Class<? extends Number>)ft.getInnerClass());
144      } else if (ft.canCreateNewInstanceFromObjectMap(outer)) {
145         ObjectMap m = new ObjectMap(bc);
146         parseIntoMap(session, r, m, string(), object());
147         o = ft.newInstanceFromObjectMap(outer, m);
148      } else if (ft.canCreateNewBean(outer)) {
149         if (ft.getExtendedMeta(XmlClassMeta.class).getFormat() == XmlFormat.COLLAPSED) {
150            String fieldName = r.getLocalName();
151            BeanMap<?> m = bc.newBeanMap(outer, ft.getInnerClass());
152            BeanPropertyMeta bpm = m.getMeta().getExtendedMeta(XmlBeanMeta.class).getPropertyMeta(fieldName);
153            ClassMeta<?> cm = m.getMeta().getClassMeta();
154            Object value = parseAnything(session, cm, currAttr, r, m.getBean(false), false);
155            setName(cm, value, currAttr);
156            bpm.set(m, value);
157            o = m.getBean();
158         } else {
159            BeanMap m = bc.newBeanMap(outer, ft.getInnerClass());
160            o = parseIntoBean(session, r, m).getBean();
161         }
162      } else if (ft.isArray()) {
163         ArrayList l = (ArrayList)parseIntoCollection(session, r, new ArrayList(), ft.getElementType());
164         o = bc.toArray(ft, l);
165      } else if (ft.canCreateNewInstanceFromString(outer)) {
166         o = ft.newInstanceFromString(outer, session.decodeString(r.getElementText()));
167      } else if (ft.canCreateNewInstanceFromNumber(outer)) {
168         o = ft.newInstanceFromNumber(outer, parseNumber(session.decodeLiteral(r.getElementText()), ft.getNewInstanceFromNumberClass()));
169      } else {
170         throw new ParseException(session, "Class ''{0}'' could not be instantiated.  Reason: ''{1}''", ft.getInnerClass().getName(), ft.getNotABeanReason());
171      }
172
173      if (transform != null && o != null)
174         o = transform.unswap(o, nt, bc);
175
176      if (outer != null)
177         setParent(nt, o, outer);
178
179      return (T)o;
180   }

182   private <K,V> Map<K,V> parseIntoMap(XmlParserSession session, XMLStreamReader r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType) throws Exception {
183      BeanContext bc = session.getBeanContext();
184      int depth = 0;
185      for (int i = 0; i < r.getAttributeCount(); i++) {
186         String a = r.getAttributeLocalName(i);
187         // TODO - Need better handling of namespaces here.
188         if (! (a.equals("type"))) {
189            K key = session.trim(convertAttrToType(session, m, a, keyType));
190            V value = session.trim(convertAttrToType(session, m, r.getAttributeValue(i), valueType));
191            setName(valueType, value, key);
192            m.put(key, value);
193         }
194      }
195      do {
196         int event = r.nextTag();
197         String currAttr;
198         if (event == START_ELEMENT) {
199            depth++;
200            currAttr = session.decodeString(r.getLocalName());
201            K key = convertAttrToType(session, m, currAttr, keyType);
202            V value = parseAnything(session, valueType, currAttr, r, m, false);
203            setName(valueType, value, currAttr);
204            if (valueType.isObject() && m.containsKey(key)) {
205               Object o = m.get(key);
206               if (o instanceof List)
207                  ((List)o).add(value);
208               else
209                  m.put(key, (V)new ObjectList(o, value).setBeanContext(bc));
210            } else {
211               m.put(key, value);
212            }
213         } else if (event == END_ELEMENT) {
214            depth--;
215            return m;
216         }
217      } while (depth > 0);
218      return m;
219   }

221   private <E> Collection<E> parseIntoCollection(XmlParserSession session, XMLStreamReader r, Collection<E> l, ClassMeta<E> elementType) throws Exception {
222      int depth = 0;
223      do {
224         int event = r.nextTag();
225         if (event == START_ELEMENT) {
226            depth++;
227            E value = parseAnything(session, elementType, null, r, l, false);
228            l.add(value);
229         } else if (event == END_ELEMENT) {
230            depth--;
231            return l;
232         }
233      } while (depth > 0);
234      return l;
235   }

237   private Object[] doParseArgs(XmlParserSession session, XMLStreamReader r, ClassMeta<?>[] argTypes) throws Exception {
238      int depth = 0;
239      Object[] o = new Object[argTypes.length];
240      int i = 0;
241      do {
242         int event = r.nextTag();
243         if (event == START_ELEMENT) {
244            depth++;
245            o[i] = parseAnything(session, argTypes[i], null, r, null, false);
246            i++;
247         } else if (event == END_ELEMENT) {
248            depth--;
249            return o;
250         }
251      } while (depth > 0);
252      return o;
253   }

255   private int getJsonType(String s) {
256      if (s == null)
257         return UNKNOWN;
258      char c = s.charAt(0);
259      switch(c) {
260         case 'o': return (s.equals("object") ? OBJECT : UNKNOWN);
261         case 'a': return (s.equals("array") ? ARRAY : UNKNOWN);
262         case 's': return (s.equals("string") ? STRING : UNKNOWN);
263         case 'b': return (s.equals("boolean") ? BOOLEAN : UNKNOWN);
264         case 'n': {
265            c = s.charAt(2);
266            switch(c) {
267               case 'm': return (s.equals("number") ? NUMBER : UNKNOWN);
268               case 'l': return (s.equals("null") ? NULL : UNKNOWN);
269            }
270            //return NUMBER;
271         }
272      }
273      return UNKNOWN;
274   }

276   private <T> BeanMap<T> parseIntoBean(XmlParserSession session, XMLStreamReader r, BeanMap<T> m) throws Exception {
277      BeanMeta<?> bMeta = m.getMeta();
278      XmlBeanMeta xmlMeta = bMeta.getExtendedMeta(XmlBeanMeta.class);
279
280      for (int i = 0; i < r.getAttributeCount(); i++) {
281         String key = session.decodeString(r.getAttributeLocalName(i));
282         String val = r.getAttributeValue(i);
283         BeanPropertyMeta bpm = xmlMeta.getPropertyMeta(key);
284         if (bpm == null) {
285            if (m.getMeta().isSubTyped()) {
286               m.put(key, val);
287            } else {
288               Location l = r.getLocation();
289               onUnknownProperty(session, key, m, l.getLineNumber(), l.getColumnNumber());
290            }
291         } else {
292            bpm.set(m, val);
293         }
294      }
295
296      BeanPropertyMeta cp = xmlMeta.getXmlContentProperty();
297      if (cp != null) {
298         XmlContentHandler h = xmlMeta.getXmlContentHandler();
299         if (h != null) {
300            h.parse(r, m.getBean());
301         } else {
302            String text = r.getElementText();
303            cp.set(m, session.decodeString(text));
304         }
305         return m;
306      }
307
308      int depth = 0;
309      do {
310         int event = r.nextTag();
311         String currAttr;
312         if (event == START_ELEMENT) {
313            depth++;
314            currAttr = session.decodeString(r.getLocalName());
315            BeanPropertyMeta pMeta = xmlMeta.getPropertyMeta(currAttr);
316            if (pMeta == null) {
317               if (m.getMeta().isSubTyped()) {
318                  Object value = parseAnything(session, string(), currAttr, r, m.getBean(false), false);
319                  m.put(currAttr, value);
320               } else {
321                  Location l = r.getLocation();
322                  onUnknownProperty(session, currAttr, m, l.getLineNumber(), l.getColumnNumber());
323                  skipCurrentTag(r);
324               }
325            } else {
326               session.setCurrentProperty(pMeta);
327               XmlFormat xf = pMeta.getExtendedMeta(XmlBeanPropertyMeta.class).getXmlFormat();
328               if (xf == COLLAPSED) {
329                  ClassMeta<?> et = pMeta.getClassMeta().getElementType();
330                  Object value = parseAnything(session, et, currAttr, r, m.getBean(false), false);
331                  setName(et, value, currAttr);
332                  pMeta.add(m, value);
333               } else if (xf == ATTR)  {
334                  pMeta.set(m, session.decodeString(r.getAttributeValue(0)));
335                  r.nextTag();
336               } else {
337                  ClassMeta<?> cm = pMeta.getClassMeta();
338                  Object value = parseAnything(session, cm, currAttr, r, m.getBean(false), false);
339                  setName(cm, value, currAttr);
340                  pMeta.set(m, value);
341               }
342               session.setCurrentProperty(null);
343            }
344         } else if (event == END_ELEMENT) {
345            depth--;
346            return m;
347         }
348      } while (depth > 0);
349      return m;
350   }

352   private void skipCurrentTag(XMLStreamReader r) throws XMLStreamException {
353      int depth = 1;
354      do {
355         int event = r.next();
356         if (event == START_ELEMENT)
357            depth++;
358         else if (event == END_ELEMENT)
359            depth--;
360      } while (depth > 0);
361   }

363   private Object getUnknown(XmlParserSession session, XMLStreamReader r) throws Exception {
364      BeanContext bc = session.getBeanContext();
365      if (r.getEventType() != XMLStreamConstants.START_ELEMENT) {
366         throw new XMLStreamException("parser must be on START_ELEMENT to read next text", r.getLocation());
367      }
368      ObjectMap m = null;
369
370      // If this element has attributes, then it's always an ObjectMap.
371      if (r.getAttributeCount() > 0) {
372         m = new ObjectMap(bc);
373         for (int i = 0; i < r.getAttributeCount(); i++) {
374            String key = session.decodeString(r.getAttributeLocalName(i));
375            String val = r.getAttributeValue(i);
376            if (! key.equals("type"))
377               m.put(key, val);
378         }
379      }
380      int eventType = r.next();
381      StringBuilder sb = new StringBuilder();
382      while (eventType != XMLStreamConstants.END_ELEMENT) {
383         if (eventType == XMLStreamConstants.CHARACTERS || eventType == XMLStreamConstants.CDATA || eventType == XMLStreamConstants.SPACE || eventType == XMLStreamConstants.ENTITY_REFERENCE) {
384            sb.append(r.getText());
385         } else if (eventType == XMLStreamConstants.PROCESSING_INSTRUCTION || eventType == XMLStreamConstants.COMMENT) {
386            // skipping
387         } else if (eventType == XMLStreamConstants.END_DOCUMENT) {
388            throw new XMLStreamException("Unexpected end of document when reading element text content", r.getLocation());
389         } else if (eventType == XMLStreamConstants.START_ELEMENT) {
390            // Oops...this has an element in it.
391            // Parse it as a map.
392            if (m == null)
393               m = new ObjectMap(bc);
394            int depth = 0;
395            do {
396               int event = (eventType == -1 ? r.nextTag() : eventType);
397               String currAttr;
398               if (event == START_ELEMENT) {
399                  depth++;
400                  currAttr = session.decodeString(r.getLocalName());
401                  String key = convertAttrToType(session, null, currAttr, string());
402                  Object value = parseAnything(session, object(), currAttr, r, null, false);
403                  if (m.containsKey(key)) {
404                     Object o = m.get(key);
405                     if (o instanceof ObjectList)
406                        ((ObjectList)o).add(value);
407                     else
408                        m.put(key, new ObjectList(o, value).setBeanContext(bc));
409                  } else {
410                     m.put(key, value);
411                  }
412
413               } else if (event == END_ELEMENT) {
414                  depth--;
415                  break;
416               }
417               eventType = -1;
418            } while (depth > 0);
419            break;
420         } else {
421            throw new XMLStreamException("Unexpected event type " + eventType, r.getLocation());
422         }
423         eventType = r.next();
424      }
425      String s = sb.toString();
426      s = session.decodeString(s);
427      if (m != null) {
428         if (! s.isEmpty())
429            m.put("contents", s);
430         return m;
431      }
432      return s;
433   }

440   @Override /* Parser */
441   public XmlParserSession createSession(Object input, ObjectMap properties, Method javaMethod, Object outer) {
442      return new XmlParserSession(getContext(XmlParserContext.class), getBeanContext(), input, properties, javaMethod, outer);
443   }

445   @Override /* Parser */
446   protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
447      XmlParserSession s = (XmlParserSession)session;
448      type = s.getBeanContext().normalizeClassMeta(type);
449      return parseAnything(s, type, null, s.getXmlStreamReader(), s.getOuter(), true);
450   }
451
452   @Override /* ReaderParser */
453   protected <K,V> Map<K,V> doParseIntoMap(ParserSession session, Map<K,V> m, Type keyType, Type valueType) throws Exception {
454      XmlParserSession s = (XmlParserSession)session;
455      ClassMeta cm = s.getBeanContext().getMapClassMeta(m.getClass(), keyType, valueType);
456      return parseIntoMap(s, m, cm.getKeyType(), cm.getValueType());
457   }
458
459   @Override /* ReaderParser */
460   protected <E> Collection<E> doParseIntoCollection(ParserSession session, Collection<E> c, Type elementType) throws Exception {
461      XmlParserSession s = (XmlParserSession)session;
462      ClassMeta cm = s.getBeanContext().getCollectionClassMeta(c.getClass(), elementType);
463      return parseIntoCollection(s,c, cm.getElementType());
464   }
465
466   @Override /* ReaderParser */
467   protected Object[] doParseArgs(ParserSession session, ClassMeta<?>[] argTypes) throws Exception {
468      XmlParserSession s = (XmlParserSession)session;
469      return doParseArgs(s, s.getXmlStreamReader(), argTypes);
470   }
