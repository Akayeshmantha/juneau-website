043/**
044 * Represents an HTTP request for a REST resource.
045 * <p>
046 *    Equivalent to {@link HttpServletRequest} except with some additional convenience methods.
047 * </p>
048 * <p>
049 *    For reference, given the URL <js>"http://localhost:9080/contextRoot/servletPath/foo?bar=baz#qux"</js>, the
050 *    following methods return the following values....
051 * </p>
052 * <table class='styled'>
053 *    <tr><th>Method</th><th>Value</th></tr>
054 *    <tr><td>{@code getContextPath()}</td><td>{@code /contextRoot}</td></tr>
055 *    <tr><td>{@code getPathInfo()}</td><td>{@code /foo}</td></tr>
056 *    <tr><td>{@code getPathTranslated()}</td><td>{@code path-to-deployed-war-on-filesystem/foo}</td></tr>
057 *    <tr><td>{@code getQueryString()}</td><td>{@code bar=baz}</td></tr>
058 *    <tr><td>{@code getRequestURI()}</td><td>{@code /contextRoot/servletPath/foo}</td></tr>
059 *    <tr><td>{@code getRequestURL()}</td><td>{@code http://localhost:9080/contextRoot/servletPath/foo}</td></tr>
060 *    <tr><td>{@code getServletPath()}</td><td>{@code /servletPath}</td></tr>
061 * </table>
062 * <p>
063 *    Refer to <a class='doclink' href='package-summary.html#TOC'>REST Servlet API</a> for information about using this class.

091   /**
092    * Constructor.
093    */
094   RestRequest(RestServlet servlet, HttpServletRequest req) throws ServletException {
095      super(req);
096
097      try {
098         this.servlet = servlet;
099         isPost = req.getMethod().equalsIgnoreCase("POST");
100
101         // If this is a POST, we want to parse the query parameters ourselves to prevent
102         // the servlet code from processing the HTTP body as URL-Encoded parameters.
103         if (isPost)
104            queryParams = servlet.getUrlEncodingParser().parseIntoSimpleMap(getQueryString());
105         else {
106            queryParams = req.getParameterMap();
107         }
108
109         // Get the HTTP method.
110         // Can be overridden through a "method" GET attribute.
111         method = super.getMethod();
112
113         String m = getQueryParameter("method");
114         if (! StringUtils.isEmpty(m) && (servlet.context.allowMethodParams.contains(m) || servlet.context.allowMethodParams.contains("*")))
115            method = m;
116
117         if (servlet.context.allowContentParam)
118            content = getQueryParameter("content");
119
120         defaultServletHeaders = servlet.getDefaultRequestHeaders();
121
122         debug = "true".equals(getQueryParameter("debug", "false"));
123
124         if (debug) {
125            servlet.log(Level.INFO, toString());
126         }
127
128      } catch (RestException e) {
129         throw e;
130      } catch (Exception e) {
131         throw new ServletException(e);
132      }
133   }

192   /**
193    * Returns the <code>Content-Type</code> header value on the request, stripped
194    *    of any parameters such as <js>";charset=X"</js>.
195    * <p>
196    * Example: <js>"text/json"</js>.
197    * <p>
198    * If the content type is not specified, and the content is specified via a
199    *    <code>&content</code> query parameter, the content type is assumed to be
200    *    <js>"text/uon"</js>.  Otherwise, the
201    *    content type is assumed to be <js>"text/json"</js>.
202    *
203    * @return The <code>Accept</code> media-type header values on the request.
204    */
205   public String getMediaType() {
206      String cm = getHeader("Content-Type");
207      if (cm == null) {
208         if (content != null)
209            return "text/uon";
210         return "text/json";
211      }
212      int j = cm.indexOf(';');
213      if (j != -1)
214         cm = cm.substring(0, j);
215      return cm;
216   }

227   /**
228    * Returns the charset specified on the <code>Content-Type</code> header, or
229    * <js>"UTF-8"</js> if not specified.
230    */
231   @Override /* ServletRequest */
232   public String getCharacterEncoding() {
233      if (charset == null) {
234         // Determine charset
235         // NOTE:  Don't use super.getCharacterEncoding() because the spec is implemented inconsistently.
236         // Jetty returns the default charset instead of null if the character is not specified on the request.
237         String h = getHeader("Content-Type");
238         if (h != null) {
239            int i = h.indexOf(";charset=");
240            if (i > 0)
241               charset = h.substring(i+9).trim();
242         }
243         if (charset == null)
244            charset = defaultCharset;
245         if (! RestServlet.availableCharsets.containsKey(charset))
246            throw new RestException(SC_UNSUPPORTED_MEDIA_TYPE, "Unsupported charset in header ''Content-Type'': ''{0}''", h);
247      }
248      return charset;
249   }

312   /**
313    * Returns the specified header value, or the specified default value if the
314    *    header value isn't present.
315    * <p>
316    *    If {@code allowHeaderParams} init parameter is <jk>true</jk>, then first looks
317    *    for {@code &HeaderName=x} in the URL query string.
318    *
319    * @param name The HTTP header name.
320    * @param def The default value to return if the header value isn't found.
321    * @return The header value, or the default value if the header isn't present.
322    */
323   public String getHeader(String name, String def) {
324      String h = getOverriddenHeader(name);
325      if (h != null)
326         return h;
327      h = super.getHeader(name);
328      if (h != null && ! h.isEmpty())
329         return h;
330      if (defaultMethodHeaders != null) {
331         h = defaultMethodHeaders.get(name);
332         if (h != null)
333            return h;
334      }
335      h = defaultServletHeaders.get(name);
336      if (h != null)
337         return h;
338      return def;
339   }

349   /*
350    * Returns header value from URL-parameters or set via setHeader() meant
351    * to override actual header values on the request.
352    */
353   private String getOverriddenHeader(String name) {
354      String h = null;
355      if (servlet.context.allowHeaderParams)
356         h = getQueryParameter(name);
357      if (h != null)
358         return h;
359      if (overriddenHeaders != null) {
360         h = overriddenHeaders.get(name);
361         if (h != null)
362            return h;
363      }
364      return h;
365   }

367   @Override /* ServletRequest */
368   public Locale getLocale() {
369      String h = getOverriddenHeader("Accept-Language");
370      if (h != null) {
371         MediaRange[] mr = MediaRange.parse(h);
372         if (mr.length > 0)
373            return toLocale(mr[0].getType());
374      }
375      return super.getLocale();
376   }

378   @Override /* ServletRequest */
379   public Enumeration<Locale> getLocales() {
380      String h = getOverriddenHeader("Accept-Language");
381      if (h != null) {
382         MediaRange[] mr = MediaRange.parse(h);
383         if (mr.length > 0) {
384            List<Locale> l = new ArrayList<Locale>(mr.length);
385            for (MediaRange r : mr)
386               l.add(toLocale(r.getType()));
387            return enumeration(l);
388         }
389      }
390      return super.getLocales();
391   }

393   /**
394    * Converts an Accept-Header value entry to a Locale.
395    */
396   private Locale toLocale(String lang) {
397      String country = "";
398      int i = lang.indexOf('-');
399      if (i > -1) {
400          country = lang.substring(i+1).trim();
401          lang = lang.substring(0,i).trim();
402      }
403      return new Locale(lang, country);
404   }

435   /**
436    * Returns the parameter with the specified name.
437    * <p>
438    * Returns <jk>null</jk> for parameters with no value (e.g. <js>"&foo"</js>).
439    * This is consistent with WAS, but differs from Tomcat behavior.
440    * The presence of parameter <js>"&foo"</js> in this case can be determined using {@link #hasParameter(String)}.
441    * <p>
442    * Parameter lookup is case-insensitive (consistent with WAS, but differs from Tomcat).
443    * <p>
444    * <i>Note:</i> Calling this method on URL-Encoded FORM posts causes the body content to be loaded and parsed by
445    *    the underlying servlet API.
446    * <p>
447    * <i>Note:</i> This method returns the raw unparsed value, and differs from calling <code>getParameter(name, String.<jk>class</js>)</code>
448    *    which will convert the value from UON notation:
449    * <ul>
450    *    <li><js>"\u0000"</js> =&gt; <jk>null</jk>
451    *    <li><js>"$s(foo)"</js> =&gt; <js>"foo"</js>
452    *    <li><js>"(foo)"</js> =&gt; <js>"foo"</js>
453    * </ul>
454    */
455   @Override /* ServletRequest */
456   public String getParameter(String name) {
457      String s = null;
458      if (overriddenParams != null)
459         s = overriddenParams.get(name);
460      if (s != null)
461         return s;
462
463      String val = super.getParameter(name);
464
465      // Fix for behavior difference between Tomcat and WAS.
466      // getParameter("foo") on "&foo" in Tomcat returns "".
467      // getParameter("foo") on "&foo" in WAS returns null.
468      if (val != null && val.isEmpty())
469         if (queryParams.containsKey(name))
470            val = null;
471
472      return val;
473   }

626   /**
627    * Same as {@link #getParameter(String, ClassMeta)} except for use on multi-part parameters
628    *    (e.g. <js>"&key=1&key=2&key=3"</js> instead of <js>"&key=(1,2,3)"</js>)
629    * <p>
630    *    This method must only be called when parsing into classes of type Collection or array.
631    *
632    * @param name The parameter name.
633    * @param cm The class type to convert the parameter value to.
634    * @param <T> The class type to convert the parameter value to.
635    * @return The parameter value converted to the specified class type.
636    * @throws ParseException
637    */
638   @SuppressWarnings("rawtypes")
639   public <T> T getParameters(String name, ClassMeta<T> cm) throws ParseException {
640      String[] p = getParameterValues(name);
641      if (p == null)
642         return null;
