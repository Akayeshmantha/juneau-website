189   @RestResource(path="/testInheritanceProperties", serializers=JsonSerializer.Simple.class)
190   public static class TestProperties extends Sub {
191      private static final long serialVersionUID = 1L;
192
193      // Should show {p1:'v1',p2:'v2a',p3:'v3',p4:'v4'}
194      @RestMethod(name="GET", path="/test1")
195      public ObjectMap test1(@Properties ObjectMap properties) {
196         return transform(properties);
197      }
198
199      // Should show {p1:'v1',p2:'v2a',p3:'v3',p4:'v4a',p5:'v5'} when override is false.
200      // Should show {p1:'x',p2:'x',p3:'x',p4:'x',p5:'x'} when override is true.
201      @RestMethod(name="GET", path="/test2",
202         properties={@Property(name="p4",value="v4a"), @Property(name="p5", value="v5")})
203      public ObjectMap test2(@Properties ObjectMap properties, @HasParam("override") boolean override) {
204         if (override) {
205            properties.put("p1", "x");
206            properties.put("p2", "x");
207            properties.put("p3", "x");
208            properties.put("p4", "x");
209            properties.put("p5", "x");
210         }
211         return transform(properties);
212      }
213
214      private ObjectMap transform(ObjectMap properties) {
215         ObjectMap m = new ObjectMap();
216         for (Map.Entry<String,Object> e : properties.entrySet()) {
217            if (e.getKey().startsWith("p"))
218               m.put(e.getKey(), e.getValue());
219         }
220         return m;
221      }
222   }
