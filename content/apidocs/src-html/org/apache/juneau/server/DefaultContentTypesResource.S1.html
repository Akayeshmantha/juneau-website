013package org.apache.juneau.server;
014
015import static org.apache.juneau.server.annotation.Inherit.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.annotation.*;
019import org.apache.juneau.parser.*;
020import org.apache.juneau.serializer.*;
021import org.apache.juneau.server.annotation.*;
022
023/**
024 * JUnit automated testcase resource.
025 */
026@RestResource(
027   path="/testDefaultContentTypes",
028   defaultRequestHeaders={" Accept : text/s2 "," Content-Type : text/p2 "},
029   parsers={DefaultContentTypesResource.P1.class,DefaultContentTypesResource.P2.class}, serializers={DefaultContentTypesResource.S1.class,DefaultContentTypesResource.S2.class}
030)
031@SuppressWarnings("synthetic-access")
032public class DefaultContentTypesResource extends RestServlet {
033   private static final long serialVersionUID = 1L;
034
035   @Consumes("text/p1")
036   public static class P1 extends DummyParser { public P1() {super("p1");}}
037
038   @Consumes("text/p2")
039   public static class P2 extends DummyParser { public P2() {super("p2");}}
040
041   @Consumes("text/p3")
042   public static class P3 extends DummyParser { public P3() {super("p3");}}
043
044   @Produces("text/s1")
045   public static class S1 extends DummySerializer { public S1() {super("s1");}}
046
047   @Produces("text/s2")
048   public static class S2 extends DummySerializer { public S2() {super("s2");}}
049
050   @Produces("text/s3")
051   public static class S3 extends DummySerializer { public S3() {super("s3");}}
052
053   /**
054    * Test that default Accept and Content-Type headers on servlet annotation are picked up.
055    */
056   @RestMethod(name="PUT", path="/testDefaultHeadersOnServletAnnotation")
057   public String testDefaultHeadersOnServletAnnotation(@Content String in) {
058      return in;
059   }
060
061   //====================================================================================================
062   // Test that default Accept and Content-Type headers on servlet annotation are picked up
063   // when @RestMethod.parsers/serializers annotations are used.
064   //====================================================================================================
065   @RestMethod(name="PUT", path="/testRestMethodParsersSerializers", parsers=P3.class, serializers=S3.class)
066   public String testRestMethodParsersSerializers(@Content String in) {
067      return in;
068   }
069
070   //====================================================================================================
071   // Test that default Accept and Content-Type headers on servlet annotation are picked up
072   // when @RestMethod.addParsers/addSerializers annotations are used.
073   //====================================================================================================
074   @RestMethod(name="PUT", path="/testRestMethodAddParsersSerializers", parsers=P3.class, parsersInherit=PARSERS, serializers=S3.class, serializersInherit=SERIALIZERS)
075   public String testRestMethodAddParsersSerializers(@Content String in) {
076      return in;
077   }
078
079   //====================================================================================================
080   // Various Accept incantations.
081   //====================================================================================================
082   @RestMethod(name="PUT", path="/testAccept")
083   public String testAccept(@Content String in) {
084      return in;
085   }
086
087   //====================================================================================================
088   // Test that default Accept and Content-Type headers on method annotation are picked up
089   // when @RestMethod.parsers/serializers annotations are used.
090   //====================================================================================================
091   @RestMethod(name="PUT", path="/testRestMethodParserSerializerAnnotations", defaultRequestHeaders={"Accept: text/s3","Content-Type: text/p3"}, parsers=P3.class, serializers=S3.class)
092   public String testRestMethodParserSerializerAnnotations(@Content String in) {
093      return in;
094   }
095
096   //====================================================================================================
097   // Test that default Accept and Content-Type headers on method annotation are picked up
098   //    when @RestMethod.addParsers/addSerializers annotations are used.
099   //====================================================================================================
100   @RestMethod(name="PUT", path="/testRestMethodAddParsersSerializersAnnotations", defaultRequestHeaders={"Accept: text/s3","Content-Type: text/p3"}, parsers=P3.class, parsersInherit=PARSERS, serializers=S3.class, serializersInherit=SERIALIZERS)
101   public String testRestMethodAddParsersSerializersAnnotations(@Content String in) {
102      return in;
103   }
104
105   public static class DummyParser extends ReaderParser {
106      private String name;
107      private DummyParser(String name) {
108         this.name = name;
109      }
110      @SuppressWarnings("unchecked")
111      @Override /* Parser */
112      protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
113         return (T)name;
114      }
115   }
116
117   public static class DummySerializer extends WriterSerializer {
118      private String name;
119      private DummySerializer(String name) {
120         this.name = name;
121      }
122      @Override /* Serializer */
123      protected void doSerialize(SerializerSession session, Object output) throws Exception {
124         session.getWriter().write(name + "/" + output);
125      }
126   }
127}
