013package org.apache.juneau.server;
014
015import static org.apache.juneau.server.RestServletContext.*;
016
017import java.io.*;
018
019import org.apache.juneau.*;
020import org.apache.juneau.annotation.*;
021import org.apache.juneau.parser.*;
022import org.apache.juneau.plaintext.*;
023import org.apache.juneau.serializer.*;
024import org.apache.juneau.server.annotation.*;
025
026/**
027 * JUnit automated testcase resource.
028 */
029@RestResource(
030   path="/testAcceptCharset",
031   serializers={PlainTextSerializer.class},
032   properties={
033      // Some versions of Jetty default to ISO8601, so specify UTF-8 for test consistency.
034      @Property(name=REST_defaultCharset,value="utf-8")
035   }
036)
037public class AcceptCharsetResource extends RestServlet {
038   private static final long serialVersionUID = 1L;
039
040   //====================================================================================================
041   // Test that Q-values are being resolved correctly.
042   //====================================================================================================
043   @RestMethod(name="GET", path="/testQValues")
044   public String testQValues() {
045      return "foo";
046   }
047
048   //====================================================================================================
049   // Validate various Accept-Charset variations.
050   //====================================================================================================
051   @RestMethod(name="PUT", path="/testCharsetOnResponse", parsers=TestParser.class, serializers=TestSerializer.class)
052   public String testCharsetOnResponse(@Content String in) {
053      return in;
054   }
055
056   @Consumes("text/plain")
057   public static class TestParser extends InputStreamParser {
058      @SuppressWarnings("unchecked")
059      @Override /* Parser */
060      protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
061         return (T)session.getProperties().getString("characterEncoding");
062      }
063   }
064
065   @Produces("text/plain")
066   public static class TestSerializer extends OutputStreamSerializer {
067      @Override /* Serializer */
068      protected void doSerialize(SerializerSession session, Object o) throws Exception {
069         Writer w = new OutputStreamWriter(session.getOutputStream());
070         w.append(o.toString()).append('/').append(session.getProperties().getString("characterEncoding"));
071         w.flush();
072         w.close();
073      }
074   }
075}
