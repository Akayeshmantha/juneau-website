013package org.apache.juneau.server;
014
015import org.apache.juneau.*;
016import org.apache.juneau.annotation.*;
017import org.apache.juneau.parser.*;
018import org.apache.juneau.plaintext.*;
019import org.apache.juneau.server.annotation.*;
020
021/**
022 * JUnit automated testcase resource.
023 * Validates that headers
024 */
025@RestResource(
026   path="/testOnPreCall",
027   parsers=OnPreCallResource.TestParserA.class,
028   serializers=PlainTextSerializer.class,
029   properties={
030      @Property(name="p1",value="sp1"), // Unchanged servlet-level property.
031      @Property(name="p2",value="sp2"), // Servlet-level property overridden by onPreCall.
032      @Property(name="p3",value="sp3"), // Servlet-level property overridded by method.
033      @Property(name="p4",value="sp4")  // Servlet-level property overridden by method then onPreCall.
034   }
035)
036public class OnPreCallResource extends RestServlet {
037   private static final long serialVersionUID = 1L;
038
039   @Consumes({"text/a1","text/a2","text/a3"})
040   public static class TestParserA extends ReaderParser {
041      @SuppressWarnings("unchecked")
042      @Override /* Parser */
043      protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
044         ObjectMap p = session.getProperties();
045         String matchingContentType = session.getProperties().getString("mediaType");
046         return (T)("p1="+p.get("p1")+",p2="+p.get("p2")+",p3="+p.get("p3")+",p4="+p.get("p4")+",p5="+p.get("p5")+",contentType="+matchingContentType);
047      }
048   }
049
050   @Override /* RestServlet */
051   protected void onPreCall(RestRequest req) {
052      ObjectMap properties = req.getProperties();
053      properties.put("p2", "xp2");
054      properties.put("p4", "xp4");
055      properties.put("p5", "xp5"); // New property
056      String overrideContentType = req.getHeader("Override-Content-Type");
057      if (overrideContentType != null)
058         req.setHeader("Content-Type", overrideContentType);
059   }
060
061
062   //====================================================================================================
063   // Properties overridden via properties annotation.
064   //====================================================================================================
065   @RestMethod(name="PUT", path="/testPropertiesOverriddenByAnnotation",
066      properties={
067         @Property(name="p3",value="mp3"),
068         @Property(name="p4",value="mp4")
069      }
070   )
071   public String testPropertiesOverriddenByAnnotation(@Content String in) {
072      return in;
073   }
074
075   //====================================================================================================
076   // Properties overridden programmatically.
077   //====================================================================================================
078   @RestMethod(name="PUT", path="/testPropertiesOverriddenProgrammatically")
079   public String testPropertiesOverriddenProgrammatically(RestRequest req, @Properties ObjectMap properties) throws Exception {
080      properties.put("p3", "pp3");
081      properties.put("p4", "pp4");
082      return req.getInput(String.class);
083   }
084}
