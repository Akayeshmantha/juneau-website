013package org.apache.juneau.server;
014
015import org.apache.juneau.*;
016import org.apache.juneau.annotation.*;
017import org.apache.juneau.internal.*;
018import org.apache.juneau.parser.*;
019import org.apache.juneau.serializer.*;
020import org.apache.juneau.server.annotation.*;
021
022/**
023 * JUnit automated testcase resource.
024 */
025@RestResource(
026   path="/testGroups"
027)
028public class GroupsResource extends RestServlet {
029   private static final long serialVersionUID = 1L;
030
031   @Produces({"text/s1","text/s2"})
032   public static class SSerializer extends WriterSerializer {
033      @Override /* Serializer */
034      protected void doSerialize(SerializerSession session, Object output) throws Exception {
035         session.getWriter().write("text/s," + output);
036      }
037   }
038
039   @Consumes({"text/p1","text/p2"})
040   public static class PParser extends ReaderParser {
041      @SuppressWarnings("unchecked")
042      @Override /* Parser */
043      protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
044         return (T)IOUtils.read(session.getReader());
045      }
046   }
047
048
049   @Override /* RestServlet */
050   public SerializerGroup createSerializers(ObjectMap properties, Class<?>[] beanFilters, Class<?>[] pojoSwaps) throws Exception {
051      return new SerializerGroup().append(SSerializer.class).setProperties(properties).addBeanFilters(beanFilters).addPojoSwaps(pojoSwaps);
052   }
053
054   @Override /* RestServlet */
055   public ParserGroup createParsers(ObjectMap properties, Class<?>[] beanFilters, Class<?>[] pojoSwaps) throws Exception {
056      return new ParserGroup().append(PParser.class).setProperties(properties).addBeanFilters(beanFilters).addPojoSwaps(pojoSwaps);
057   }
058
059   //====================================================================================================
060   // Serializer defined on class.
061   //====================================================================================================
062   @RestMethod(name="GET", path="/testSerializerDefinedOnClass")
063   public String testSerializerDefinedOnClass_get() {
064      return "GET";
065   }
066
067   @RestMethod(name="PUT", path="/testSerializerDefinedOnClass")
068   public String testSerializerDefinedOnClass_put(@Content String in) {
069      return in;
070   }
071}
