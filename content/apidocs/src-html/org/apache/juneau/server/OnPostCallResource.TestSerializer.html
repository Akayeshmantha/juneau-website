013package org.apache.juneau.server;
014
015import org.apache.juneau.*;
016import org.apache.juneau.annotation.*;
017import org.apache.juneau.serializer.*;
018import org.apache.juneau.server.annotation.*;
019
020/**
021 * JUnit automated testcase resource.
022 * Validates that headers
023 */
024@RestResource(
025   path="/testOnPostCall",
026   serializers=OnPostCallResource.TestSerializer.class,
027   properties={
028      @Property(name="p1",value="sp1"), // Unchanged servlet-level property.
029      @Property(name="p2",value="sp2"), // Servlet-level property overridden by onPostCall.
030      @Property(name="p3",value="sp3"), // Servlet-level property overridded by method.
031      @Property(name="p4",value="sp4")  // Servlet-level property overridden by method then onPostCall.
032   }
033)
034public class OnPostCallResource extends RestServlet {
035   private static final long serialVersionUID = 1L;
036
037   @Produces({"text/s1","text/s2","text/s3"})
038   public static class TestSerializer extends WriterSerializer {
039      @Override /* Serializer */
040      protected void doSerialize(SerializerSession session, Object o) throws Exception {
041         ObjectMap p = session.getProperties();
042         session.getWriter().write("p1="+p.get("p1")+",p2="+p.get("p2")+",p3="+p.get("p3")+",p4="+p.get("p4")+",p5="+p.get("p5")+",contentType="+session.getProperties().getString("mediaType"));
043      }
044      @Override /* Serializer */
045      public ObjectMap getResponseHeaders(ObjectMap properties) {
046         if (properties.containsKey("Override-Content-Type"))
047            return new ObjectMap().append("Content-Type", properties.get("Override-Content-Type"));
048         return null;
049      }
050   }
051
052   @Override /* RestServlet */
053   protected void onPostCall(RestRequest req, RestResponse res) {
054      ObjectMap properties = req.getProperties();
055      properties.put("p2", "xp2");
056      properties.put("p4", "xp4");
057      properties.put("p5", "xp5"); // New property
058      String overrideAccept = req.getHeader("Override-Accept");
059      if (overrideAccept != null)
060         req.setHeader("Accept", overrideAccept);
061      String overrideContentType = req.getHeader("Override-Content-Type");
062      if (overrideContentType != null)
063         properties.put("Override-Content-Type", overrideContentType);
064   }
065
066
067   //====================================================================================================
068   // Test1 - Properties overridden via properties annotation.
069   //====================================================================================================
070   @RestMethod(name="PUT", path="/testPropertiesOverridenByAnnotation",
071      properties={
072         @Property(name="p3",value="mp3"),
073         @Property(name="p4",value="mp4")
074      },
075      defaultRequestHeaders="Accept: text/s2"
076   )
077   public String testPropertiesOverridenByAnnotation() {
078      return "";
079   }
080
081   //====================================================================================================
082   // Test2 - Properties overridden programmatically.
083   //====================================================================================================
084   @RestMethod(name="PUT", path="/testPropertiesOverriddenProgramatically")
085   public String testPropertiesOverriddenProgramatically(RestRequest req, @Properties ObjectMap properties) throws Exception {
086      properties.put("p3", "pp3");
087      properties.put("p4", "pp4");
088      String accept = req.getHeader("Accept");
089      if (accept == null || accept.isEmpty())
090         req.setHeader("Accept", "text/s2");
091      return "";
092   }
093}
