013package org.apache.juneau.server;
014
015import static org.apache.juneau.server.annotation.Inherit.*;
016
017import org.apache.juneau.annotation.*;
018import org.apache.juneau.serializer.*;
019import org.apache.juneau.server.annotation.*;
020
021/**
022 * JUnit automated testcase resource.
023 */
024@RestResource(
025   path="/testSerializers",
026   serializers=SerializersResource.TestSerializerA.class
027)
028public class SerializersResource extends RestServletDefault {
029   private static final long serialVersionUID = 1L;
030
031   @Produces("text/a")
032   public static class TestSerializerA extends WriterSerializer {
033      @Override /* Serializer */
034      protected void doSerialize(SerializerSession session, Object o) throws Exception {
035         session.getWriter().write("text/a - " + o);
036      }
037   }
038
039   @Produces("text/b")
040   public static class TestSerializerB extends WriterSerializer {
041      @Override /* Serializer */
042      protected void doSerialize(SerializerSession session, Object o) throws Exception {
043         session.getWriter().write("text/b - " + o);
044      }
045   }
046
047   //====================================================================================================
048   // Serializer defined on class.
049   //====================================================================================================
050   @RestMethod(name="GET", path="/testSerializerOnClass")
051   public String testSerializerOnClass() {
052      return "test1";
053   }
054
055   //====================================================================================================
056   // Serializer defined on method.
057   //====================================================================================================
058   @RestMethod(name="GET", path="/testSerializerOnMethod", serializers=TestSerializerB.class)
059   public String testSerializerOnMethod() {
060      return "test2";
061   }
062
063   //====================================================================================================
064   // Serializer overridden on method.
065   //====================================================================================================
066   @RestMethod(name="GET", path="/testSerializerOverriddenOnMethod", serializers={TestSerializerB.class,TestSerializerC.class}, serializersInherit=SERIALIZERS)
067   public String testSerializerOverriddenOnMethod() {
068      return "test3";
069   }
070
071   @Produces("text/a")
072   public static class TestSerializerC extends WriterSerializer {
073      @Override /* Serializer */
074      protected void doSerialize(SerializerSession session, Object o) throws Exception {
075         session.getWriter().write("text/c - " + o);
076      }
077   }
078
079   //====================================================================================================
080   // Serializer with different Accept than Content-Type.
081   //====================================================================================================
082   @RestMethod(name="GET", path="/testSerializerWithDifferentMediaTypes", serializers={TestSerializerD.class}, serializersInherit=SERIALIZERS)
083   public String testSerializerWithDifferentMediaTypes() {
084      return "test4";
085   }
086
087   @Produces(value={"text/a","text/d"},contentType="text/d")
088   public static class TestSerializerD extends WriterSerializer {
089      @Override /* Serializer */
090      protected void doSerialize(SerializerSession session, Object o) throws Exception {
091         session.getWriter().write("text/d - " + o);
092      }
093   }
094
095   //====================================================================================================
096   // Check for valid 406 error response.
097   //====================================================================================================
098   @RestMethod(name="GET", path="/test406")
099   public String test406() {
100      return "test406";
101   }
102}
