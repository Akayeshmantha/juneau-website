013package org.apache.juneau.server.labels;
014
015import static javax.servlet.http.HttpServletResponse.*;
016
017import java.util.*;
018
019import org.apache.juneau.annotation.*;
020import org.apache.juneau.server.*;
021import org.apache.juneau.server.annotation.*;
022
023/**
024 * Default POJO bean used for generating OPTIONS page results.
025 * <p>
026 * @author James Bognar (james.bognar@salesforce.com)
027 */
028@Bean(properties={"label","description","className","methods","children","consumes","produces","guards","converters"})
029public class ResourceOptions {
030   private String label, description;
031   private String className;
032   private Collection<MethodDescription> methods;
033   private ChildResourceDescriptions children;
034   private Collection<String> consumes, produces;
035   private Collection<String> guards;
036   private Collection<String> converters;
037
038   /**
039    * Constructor.
040    * @param servlet The servlet that this bean describes.
041    * @param req The HTTP servlet request.
042    */
043   public ResourceOptions(RestServlet servlet, RestRequest req) {
044      try {
045         setClassName(servlet.getClass().getName());
046         setLabel(servlet.getLabel(req));
047         setDescription(servlet.getDescription(req));
048         setMethods(servlet.getMethodDescriptions(req));
049         setConsumes(servlet.getSupportedAcceptTypes());
050         setProduces(servlet.getSupportedContentTypes());
051         setChildren(new ChildResourceDescriptions(servlet, req));
052         setGuards(servlet.getGuards());
053         setConverters(servlet.getConverters());
054      } catch (RestServletException e) {
055         throw new RestException(SC_INTERNAL_SERVER_ERROR, e);
056      }
057   }
058
059   /**
060    * Bean constructor.
061    */
062   public ResourceOptions() {}
063
064   /**
065    * Returns the label of the REST resource.
066    * @return The current bean property value.
067    */
068   public String getLabel() {
069      return label;
070   }
071
072   /**
073    * Sets the label of the REST resource.
074    * @param label The new bean property value.
075    * @return This object (for method chaining).
076    */
077   public ResourceOptions setLabel(String label) {
078      this.label = label;
079      return this;
080   }
081
082   /**
083    * Returns the description of the REST resource.
084    * @return The current bean property value.
085    */
086   public String getDescription() {
087      return description;
088   }
089
090   /**
091    * Sets the description of the REST resource.
092    * @param description The new bean property value.
093    * @return This object (for method chaining).
094    */
095   public ResourceOptions setDescription(String description) {
096      this.description = description;
097      return this;
098   }
099
100   /**
101    * Returns the class name of the REST resource.
102    * @return The current bean property value.
103    */
104   public String getClassName() {
105      return className;
106   }
107
108   /**
109    * Sets the class name of the REST resource.
110    * @param className The new bean property value.
111    * @return This object (for method chaining).
112    */
113   public ResourceOptions setClassName(String className) {
114      this.className = className;
115      return this;
116   }
117
118   /**
119    * Returns the methods provided on this REST resource.
120    * @return The current bean property value.
121    */
122   public Collection<MethodDescription> getMethods() {
123      return methods;
124   }
125
126   /**
127    * Sets the methods provided on this REST resource.
128    * @param methods The new bean property value.
129    * @return This object (for method chaining).
130    */
131   public ResourceOptions setMethods(Collection<MethodDescription> methods) {
132      List<MethodDescription> l = new ArrayList<MethodDescription>(methods);
133      Collections.sort(l);
134      this.methods = l;
135      return this;
136   }
137
138   /**
139    * Returns the list of allowable <code>Accept</code> header values on requests.
140    * @return The current bean property value.
141    */
142   public Collection<String> getConsumes() {
143      return consumes;
144   }
145
146   /**
147    * Sets the list of allowable <code>Accept</code> header values on requests.
148    * @param consumes The new bean property value.
149    * @return This object (for method chaining).
150    */
151   public ResourceOptions setConsumes(Collection<String> consumes) {
152      this.consumes = consumes;
153      return this;
154   }
155
156   /**
157    * Returns the list of allowable <code>Content-Type</code> header values on requests.
158    * @return The current bean property value.
159    */
160   public Collection<String> getProduces() {
161      return produces;
162   }
163
164   /**
165    * Sets the list of allowable <code>Content-Type</code> header values on requests.
166    * @param produces The new bean property value.
167    * @return This object (for method chaining).
168    */
169   public ResourceOptions setProduces(Collection<String> produces) {
170      this.produces = produces;
171      return this;
172   }
173
174   /**
175    * Returns the description of child resources with this resource (typically through {@link RestResource#children()} annotation).
176    * @return The description of child resources of this resource.
177    */
178   public ChildResourceDescriptions getChildren() {
179      return children;
180   }
181
182   /**
183    * Sets the child resource descriptions associated with this resource.
184    * @param children The child resource descriptions.
185    * @return This object (for method chaining).
186    */
187   public ResourceOptions setChildren(ChildResourceDescriptions children) {
188      this.children = children;
189      return this;
190   }
191
192
193   /**
194    * Returns the list of class-wide guards associated with this resource (typically through {@link RestResource#guards()} annotation).
195    * @return The simple class names of the guards.
196    */
197   public Collection<String> getGuards() {
198      return guards;
199   }
200
201   /**
202    * Sets the simple class names of the guards associated with this resource.
203    * @param guards The simple class names of the guards.
204    * @return This object (for method chaining).
205    */
206   public ResourceOptions setGuards(Collection<String> guards) {
207      this.guards = guards;
208      return this;
209   }
210
211   /**
212    * Shortcut for calling {@link #setGuards(Collection)} from {@link RestGuard} instances.
213    * @param guards Instances of guards associated with this resource.
214    * @return This object (for method chaining).
215    */
216   public ResourceOptions setGuards(RestGuard[] guards) {
217      Collection<String> l = new ArrayList<String>(guards.length);
218      for (RestGuard g : guards)
219         l.add(g.getClass().getSimpleName());
220      return setGuards(l);
221   }
222
223   /**
224    * Returns the list of class-wide converters associated with this resource (typically through {@link RestResource#converters()} annotation).
225    * @return The simple class names of the converters.
226    */
227   public Collection<String> getConverters() {
228      return converters;
229   }
230
231   /**
232    * Sets the simple class names of the converters associated with this resource.
233    * @param converters The simple class names of the converters.
234    * @return This object (for method chaining).
235    */
236   public ResourceOptions setConverters(Collection<String> converters) {
237      this.converters = converters;
238      return this;
239   }
240
241   /**
242    * Shortcut for calling {@link #setConverters(Collection)} from {@link RestConverter} instances.
243    * @param converters Converter classes associated with this resource.
244    * @return This object (for method chaining).
245    */
246   public ResourceOptions setConverters(RestConverter[] converters) {
247      Collection<String> l = new ArrayList<String>(converters.length);
248      for (RestConverter c : converters)
249         l.add(c.getClass().getSimpleName());
250      return setConverters(l);
251   }
252}
