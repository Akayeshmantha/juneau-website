013package org.apache.juneau.server;
014
015import static java.lang.String.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.annotation.*;
019import org.apache.juneau.serializer.*;
020import org.apache.juneau.server.annotation.*;
021
022/**
023 * JUnit automated testcase resource.
024 */
025@RestResource(
026   path="/testProperties",
027   properties={
028      @Property(name="A1",value="a1"),
029      @Property(name="A2",value="a2"),
030      @Property(name="foo",value="bar"),
031      @Property(name="bar",value="baz"),
032      @Property(name="R1a",value="$R{requestURI}"),
033      @Property(name="R1b",value="$R{requestParentURI}"),
034      @Property(name="R2",value="$R{foo}"),
035      @Property(name="R3",value="$R{$R{foo}}"),
036      @Property(name="R4",value="$R{A1}"),
037      @Property(name="R5",value="$R{A2}"),
038      @Property(name="R6",value="$R{C}"),
039   }
040)
041public class PropertiesResource extends RestServletDefault {
042   private static final long serialVersionUID = 1L;
043
044   //====================================================================================================
045   // Properties defined on method.
046   //====================================================================================================
047   @RestMethod(name="GET", path="/testPropertiesDefinedOnMethod",
048      properties={
049         @Property(name="B1",value="b1"),
050         @Property(name="B2",value="b2")
051      },
052      serializers=PropertySerializer1.class
053   )
054   public void testPropertiesDefinedOnMethod(RestResponse res) {
055      res.setProperty("A2", "c");
056      res.setProperty("B2", "c");
057      res.setProperty("C", "c");
058      res.setOutput(null);
059   }
060
061   @Produces({"application/json","text/json"})
062   public static class PropertySerializer1 extends WriterSerializer {
063      @Override /* Serializer */
064      protected void doSerialize(SerializerSession session, Object output) throws Exception {
065         ObjectMap p = session.getProperties();
066         session.getWriter().write(format("A1=%s,A2=%s,B1=%s,B2=%s,C=%s,R1a=%s,R1b=%s,R2=%s,R3=%s,R4=%s,R5=%s,R6=%s",
067            p.get("A1"), p.get("A2"), p.get("B1"), p.get("B2"), p.get("C"),
068            p.get("R1a"), p.get("R1b"), p.get("R2"), p.get("R3"), p.get("R4"), p.get("R5"), p.get("R6")));
069      }
070   }
071
072   //====================================================================================================
073   // Make sure attributes/parameters/headers are available through ctx.getProperties().
074   //====================================================================================================
075   @RestMethod(name="GET", path="/testProperties/{A}", serializers=PropertySerializer2.class)
076   public void testProperties(RestResponse res) {
077      res.setOutput(null);
078   }
079
080   @Produces({"application/json","text/json"})
081   public static class PropertySerializer2 extends WriterSerializer {
082      @Override /* Serializer */
083      protected void doSerialize(SerializerSession session, Object output) throws Exception {
084         ObjectMap p = session.getProperties();
085         session.getWriter().write(format("A=%s,P=%s,H=%s", p.get("A"), p.get("P"), p.get("h")));
086      }
087   }
088
089}
