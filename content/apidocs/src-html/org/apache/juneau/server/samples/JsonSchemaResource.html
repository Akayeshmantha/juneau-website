013package org.apache.juneau.server.samples;
014
015import static org.apache.juneau.html.HtmlDocSerializerContext.*;
016
017import org.apache.juneau.dto.jsonschema.*;
018import org.apache.juneau.microservice.*;
019import org.apache.juneau.server.annotation.*;
020
021/**
022 * Sample resource that shows how to serialize JSON-Schema documents.
023 */
024@RestResource(
025   path="/jsonSchema",
026   messages="nls/JsonSchemaResource",
027   properties={
028      @Property(name=HTMLDOC_title, value="Sample JSON-Schema document"),
029      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.JsonSchemaResource)'}")
030   }
031)
032public class JsonSchemaResource extends ResourceJena {
033   private static final long serialVersionUID = 1L;
034
035   private Schema schema;     // The schema document
036
037   @Override /* Servlet */
038   public void init() {
039
040      try {
041         schema = new Schema()
042            .setId("http://example.com/sample-schema#")
043            .setSchemaVersionUri("http://json-schema.org/draft-04/schema#")
044            .setTitle("Example Schema")
045            .setType(JsonType.OBJECT)
046            .addProperties(
047               new SchemaProperty("firstName", JsonType.STRING),
048               new SchemaProperty("lastName", JsonType.STRING),
049               new SchemaProperty("age", JsonType.INTEGER)
050                  .setDescription("Age in years")
051                  .setMinimum(0)
052            )
053            .addRequired("firstName", "lastName");
054      } catch (Exception e) {
055         throw new RuntimeException(e);
056      }
057   }
058
059   /** GET request handler */
060   @RestMethod(name="GET", path="/")
061   public Schema getSchema() throws Exception {
062      return schema;
063   }
064
065   /**
066    * PUT request handler.
067    * Replaces the schema document with the specified content, and then mirrors it as the response.
068    */
069   @RestMethod(name="PUT", path="/")
070   public Schema setSchema(@Content Schema schema) throws Exception {
071      this.schema = schema;
072      return schema;
073   }
074}
