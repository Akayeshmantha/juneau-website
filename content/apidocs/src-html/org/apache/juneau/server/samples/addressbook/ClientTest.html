013package org.apache.juneau.server.samples.addressbook;
014
015import java.text.*;
016import java.util.*;
017
018import org.apache.juneau.client.*;
019import org.apache.juneau.json.*;
020import org.apache.juneau.samples.addressbook.*;
021import org.apache.juneau.xml.*;
022
023/**
024 * Sample client code for interacting with AddressBookResource
025 */
026public class ClientTest {
027
028   public static void main(String[] args) {
029
030      try {
031         System.out.println("Running client test...");
032
033         // Create a client to handle XML requests and responses.
034         RestClient client = new RestClient(JsonSerializer.DEFAULT, JsonParser.DEFAULT);
035         RestClient xmlClient = new RestClient(XmlSerializer.DEFAULT, XmlParser.DEFAULT);
036         try {
037            String root = "http://localhost:10000/addressBook";
038
039            // Get the current contents of the address book
040            AddressBook ab = client.doGet(root + "/people").getResponse(AddressBook.class);
041            System.out.println("Number of entries = " + ab.getPeople().size());
042
043            // Same, but use XML as the protocol both ways
044            ab = xmlClient.doGet(root + "/people").getResponse(AddressBook.class);
045            System.out.println("Number of entries = " + ab.getPeople().size());
046
047
048            // Delete the existing entries
049            for (Person p : ab.getPeople()) {
050               String r = client.doDelete(p.uri).getResponse(String.class);
051               System.out.println("Deleted person " + p.name + ", response = " + r);
052            }
053
054            // Make sure they're gone
055            ab = client.doGet(root + "/people").getResponse(AddressBook.class);
056            System.out.println("Number of entries = " + ab.getPeople().size());
057
058            // Add 1st person again
059            CreatePerson cp = new CreatePerson(
060               "Barack Obama",
061               toCalendar("Aug 4, 1961"),
062               new CreateAddress("1600 Pennsylvania Ave", "Washington", "DC", 20500, true),
063               new CreateAddress("5046 S Greenwood Ave", "Chicago", "IL", 60615, false)
064            );
065            Person p = client.doPost(root + "/people", cp).getResponse(Person.class);
066            System.out.println("Created person " + p.name + ", uri = " + p.uri);
067
068            // Add 2nd person again, but add addresses separately
069            cp = new CreatePerson(
070               "George Walker Bush",
071               toCalendar("Jul 6, 1946")
072            );
073            p = client.doPost(root + "/people", cp).getResponse(Person.class);
074            System.out.println("Created person " + p.name + ", uri = " + p.uri);
075
076            // Add addresses to 2nd person
077            CreateAddress ca = new CreateAddress("43 Prairie Chapel Rd", "Crawford", "TX", 76638, true);
078            Address a = client.doPost(p.uri + "/addresses", ca).getResponse(Address.class);
079            System.out.println("Created address " + a.uri);
080
081            ca = new CreateAddress("1600 Pennsylvania Ave", "Washington", "DC", 20500, false);
082            a = client.doPost(p.uri + "/addresses", ca).getResponse(Address.class);
083            System.out.println("Created address " + a.uri);
084
085            // Find 1st person, and change name
086            Person[] pp = client.doGet(root + "/people?q=(name='Barack+Obama')").getResponse(Person[].class);
087            String r = client.doPut(pp[0].uri + "/name", "Barack Hussein Obama").getResponse(String.class);
088            System.out.println("Changed name, response = " + r);
089            p = client.doGet(pp[0].uri).getResponse(Person.class);
090            System.out.println("New name = " + p.name);
091         } finally {
092            client.closeQuietly();
093            xmlClient.closeQuietly();
094         }
095
096      } catch (Exception e) {
097         e.printStackTrace();
098      }
099   }
100
101   // Utility method
102   public static Calendar toCalendar(String birthDate) throws Exception {
103      Calendar c = new GregorianCalendar();
104      c.setTime(DateFormat.getDateInstance(DateFormat.MEDIUM).parse(birthDate));
105      return c;
106   }
107}
