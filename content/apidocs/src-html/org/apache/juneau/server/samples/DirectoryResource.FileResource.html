015import static java.util.logging.Level.*;
016import static javax.servlet.http.HttpServletResponse.*;
017import static org.apache.juneau.html.HtmlDocSerializerContext.*;
018import static org.apache.juneau.server.RestServletContext.*;
019
020import java.io.*;
021import java.net.*;
022import java.util.*;
023import java.util.logging.*;
024
025import javax.servlet.*;
026
027import org.apache.juneau.*;
028import org.apache.juneau.microservice.*;
029import org.apache.juneau.server.*;
030import org.apache.juneau.server.annotation.*;
031import org.apache.juneau.server.annotation.Properties;
032import org.apache.juneau.server.converters.*;
033import org.apache.juneau.utils.*;
034
035/**
036 * Sample REST resource for exploring local file systems.
037 */
038@RestResource(
039   messages="nls/DirectoryResource",
040   properties={
041      @Property(name=HTML_uriAnchorText, value=PROPERTY_NAME),
042      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.DirectoryResource)'}"),
043      @Property(name=REST_allowMethodParam, value="*"),
044      @Property(name="rootDir", value="$S{java.io.tmpdir}"),
045      @Property(name="allowViews", value="false"),
046      @Property(name="allowDeletes", value="false"),
047      @Property(name="allowPuts", value="false")
048   }
049)
050public class DirectoryResource extends Resource {
051   private static final long serialVersionUID = 1L;
052
053   protected File rootDir;     // The root directory
054
055   // Settings enabled through servlet init parameters
056   protected boolean allowDeletes, allowPuts, allowViews;
057
058   private static Logger logger = Logger.getLogger(DirectoryResource.class.getName());
059
060   @Override /* Servlet */
061   public void init() throws ServletException {
062      ObjectMap p = getProperties();
063      rootDir = new File(p.getString("rootDir"));
064      allowViews = p.getBoolean("allowViews", false);
065      allowDeletes = p.getBoolean("allowDeletes", false);
066      allowPuts = p.getBoolean("allowPuts", false);
067   }
068
069   /** Returns the root directory defined by the 'rootDir' init parameter */
070   protected File getRootDir() {
071      if (rootDir == null) {
072         rootDir = new File(getProperties().getString("rootDir"));
073         if (! rootDir.exists())
074            if (! rootDir.mkdirs())
075               throw new RuntimeException("Could not create root dir");
076      }
077      return rootDir;
078   }
079
080   /** GET request handler */
081   @RestMethod(name="GET", path="/*", converters={Queryable.class})
082   public Object doGet(RestRequest req, @Properties ObjectMap properties) throws Exception {
083
084      String pathInfo = req.getPathInfo();
085      File f = pathInfo == null ? rootDir : new File(rootDir.getAbsolutePath() + pathInfo);
086
087      if (!f.exists())
088         throw new RestException(SC_NOT_FOUND, "File not found");
089
090      properties.put("path", f.getAbsolutePath());
091
092      if (f.isDirectory()) {
093         List<FileResource> l = new LinkedList<FileResource>();
094         for (File fc : f.listFiles()) {
095            URL fUrl = new URL(req.getRequestURL().append("/").append(fc.getName()).toString());
096            l.add(new FileResource(fc, fUrl));
097         }
098         return l;
099      }
100
101      return new FileResource(f, new URL(req.getRequestURL().toString()));
102   }
103
104   /** DELETE request handler */
105   @RestMethod(name="DELETE", path="/*", guards=AdminGuard.class)
106   public Object doDelete(RestRequest req) throws Exception {
107
108      if (! allowDeletes)
109         throw new RestException(SC_METHOD_NOT_ALLOWED, "DELETE not enabled");
110
111      File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
112      deleteFile(f);
113
114      if (req.getHeader("Accept").contains("text/html"))
115         return new Redirect();
116      return "File deleted";
117   }
118
119   /** PUT request handler */
120   @RestMethod(name="PUT", path="/*", guards=AdminGuard.class)
121   public Object doPut(RestRequest req) throws Exception {
122
123      if (! allowPuts)
124         throw new RestException(SC_METHOD_NOT_ALLOWED, "PUT not enabled");
125
126      File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
127      String parentSubPath = f.getParentFile().getAbsolutePath().substring(rootDir.getAbsolutePath().length());
128      BufferedOutputStream bos = new BufferedOutputStream(new FileOutputStream(f));
129      IOPipe.create(req.getInputStream(), bos).closeOut().run();
130      if (req.getContentType().contains("html"))
131         return new Redirect(parentSubPath);
132      return "File added";
133   }
134
135   /** VIEW request handler (overloaded GET for viewing file contents) */
136   @RestMethod(name="VIEW", path="/*")
137   public void doView(RestRequest req, RestResponse res) throws Exception {
138
139      if (! allowViews)
140         throw new RestException(SC_METHOD_NOT_ALLOWED, "VIEW not enabled");
141
142      File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
143
144      if (!f.exists())
145         throw new RestException(SC_NOT_FOUND, "File not found");
146
147      if (f.isDirectory())
148         throw new RestException(SC_METHOD_NOT_ALLOWED, "VIEW not available on directories");
149
150      res.setOutput(new FileReader(f)).setContentType("text/plain");
151   }
152
153   /** DOWNLOAD request handler (overloaded GET for downloading file contents) */
154   @RestMethod(name="DOWNLOAD")
155   public void doDownload(RestRequest req, RestResponse res) throws Exception {
156
157      if (! allowViews)
158         throw new RestException(SC_METHOD_NOT_ALLOWED, "DOWNLOAD not enabled");
159
160      File f = new File(rootDir.getAbsolutePath() + req.getPathInfo());
161
162      if (!f.exists())
163         throw new RestException(SC_NOT_FOUND, "File not found");
164
165      if (f.isDirectory())
166         throw new RestException(SC_METHOD_NOT_ALLOWED, "DOWNLOAD not available on directories");
167
168      res.setOutput(new FileReader(f)).setContentType("application");
169   }
170
171   /** File POJO */
172   public class FileResource {
173      private File f;
174      private URL url;
175
176      /** Constructor */
177      public FileResource(File f, URL url) {
178         this.f = f;
179         this.url = url;
180      }
181
182      // Bean property getters
183
184      public URL getUrl() {
185         return url;
186      }
187
188      public String getType() {
189         return (f.isDirectory() ? "dir" : "file");
190      }
191
192      public String getName() {
193         return f.getName();
194      }
195
196      public long getSize() {
197         return f.length();
198      }
199
200      public Date getLastModified() {
201         return new Date(f.lastModified());
202      }
203
204      public URL getView() throws Exception {
205         if (allowViews && f.canRead() && ! f.isDirectory())
206            return new URL(url + "?method=VIEW");
207         return null;
208      }
209
210      public URL getDownload() throws Exception {
211         if (allowViews && f.canRead() && ! f.isDirectory())
212            return new URL(url + "?method=DOWNLOAD");
213         return null;
214      }
215
216      public URL getDelete() throws Exception {
217         if (allowDeletes && f.canWrite())
218            return new URL(url + "?method=DELETE");
219         return null;
220      }
221   }
222
223   /** Utility method */
224   private void deleteFile(File f) {
225      try {
226         if (f.isDirectory())
227            for (File fc : f.listFiles())
228               deleteFile(fc);
229         f.delete();
230      } catch (Exception e) {
231         logger.log(WARNING, "Cannot delete file '" + f.getAbsolutePath() + "'", e);
232      }
233   }
234}
