013package org.apache.juneau.server.samples;
014
015import static javax.xml.bind.DatatypeConverter.*;
016import static org.apache.juneau.html.HtmlDocSerializerContext.*;
017import static org.apache.juneau.jena.RdfCommonContext.*;
018import static org.apache.juneau.jena.RdfSerializerContext.*;
019
020import java.net.*;
021
022import org.apache.juneau.dto.atom.*;
023import org.apache.juneau.dto.atom.Content;
024import org.apache.juneau.encoders.*;
025import org.apache.juneau.microservice.*;
026import org.apache.juneau.server.annotation.*;
027
028/**
029 * Sample resource that shows how to generate ATOM feeds.
030 */
031@RestResource(
032   path="/atom",
033   messages="nls/AtomFeedResource",
034   properties={
035      @Property(name=SERIALIZER_quoteChar, value="'"),
036      @Property(name=RDF_rdfxml_tab, value="5"),
037      @Property(name=RDF_addRootProperty, value="true"),
038      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.AtomFeedResource)'}")
039   },
040   encoders=GzipEncoder.class
041)
042public class AtomFeedResource extends ResourceJena {
043   private static final long serialVersionUID = 1L;
044
045   private Feed feed;     // The root resource object
046
047   @Override /* Servlet */
048   public void init() {
049
050      try {
051         feed = new Feed()
052            .setTitle(new Text("text", "Juneau ATOM specification"))
053            .setSubTitle(new Text("html", "A <em>lot</em> of effort went into making this effortless"))
054            .setUpdated(parseDateTime("2013-05-08T12:29:29Z"))
055            .setId(new Id("tag:juneau.sample.com,2013:1"))
056            .addLinks(
057               new Link("alternate", "text/html", "http://www.sample.com/").setHreflang("en"),
058               new Link("self", "application/atom+xml", "http://www.sample.com/feed.atom")
059            )
060            .setRights(new Text("Copyright (c) 2013, IBM"))
061            .setGenerator(new Generator("Juneau").setUri(new URI("http://juneau.ibm.com/")).setVersion("1.0"))
062            .addEntries(
063               new Entry()
064                  .setTitle(new Text("Juneau ATOM specification snapshot"))
065                  .addLinks(
066                     new Link("alternate", "text/html", "http://www.sample.com/2012/05/08/juneau.atom"),
067                     new Link("enclosure", "audio/mpeg", "http://www.sample.com/audio/juneau_podcast.mp3").setLength(12345)
068                  )
069                  .setId(new Id("tag:juneau.sample.com,2013:1.2345"))
070                  .setUpdated(parseDateTime("2013-05-08T12:29:29Z"))
071                  .setPublished(parseDateTime("2013-05-08T12:29:29Z"))
072                  .addAuthors(new Person("James Bognar").setUri(new URI("http://www.sample.com/")).setEmail("james.bognar@salesforce.com"))
073                  .addContributors(
074                     new Person("Barry M. Caceres")
075                  )
076                  .setContent(
077                     new Content()
078                        .setLang("en")
079                        .setBase(new URI("http://www.ibm.com/"))
080                        .setType("xhtml")
081                        .setText("<div xmlns=\"http://www.w3.org/1999/xhtml\"><p><i>[Update: Juneau supports ATOM.]</i></p></div>")
082                  )
083            );
084      } catch (Exception e) {
085         throw new RuntimeException(e);
086      }
087   }
088
089   /**
090    * GET request handler
091    */
092   @RestMethod(name="GET", path="/")
093   public Feed getFeed() throws Exception {
094      return feed;
095   }
096
097   /**
098    * PUT request handler.
099    * Replaces the feed with the specified content, and then mirrors it as the response.
100    */
101   @RestMethod(name="PUT", path="/")
102   public Feed setFeed(@org.apache.juneau.server.annotation.Content Feed feed) throws Exception {
103      this.feed = feed;
104      return feed;
105   }
106}
