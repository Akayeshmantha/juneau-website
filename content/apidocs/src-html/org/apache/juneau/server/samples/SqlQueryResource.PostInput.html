013package org.apache.juneau.server.samples;
014
015import static javax.servlet.http.HttpServletResponse.*;
016import static org.apache.juneau.html.HtmlDocSerializerContext.*;
017
018import java.io.*;
019import java.sql.*;
020import java.util.*;
021
022import org.apache.juneau.dto.*;
023import org.apache.juneau.ini.*;
024import org.apache.juneau.internal.*;
025import org.apache.juneau.microservice.*;
026import org.apache.juneau.server.*;
027import org.apache.juneau.server.annotation.*;
028
029/**
030 * Sample resource that shows how Juneau can serialize ResultSets.
031 */
032@RestResource(
033   path="/sqlQuery",
034   messages="nls/SqlQueryResource",
035   properties={
036      @Property(name=HTMLDOC_title, value="SQL query service"),
037      @Property(name=HTMLDOC_description, value="Executes queries against the local derby '$C{SqlQueryResource/connectionUrl}' database"),
038      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.SqlQueryResource)'}"),
039   }
040)
041public class SqlQueryResource extends Resource {
042   private static final long serialVersionUID = 1L;
043
044   private ConfigFile cf = getConfig();
045
046   private String driver = cf.getString("SqlQueryResource/driver");
047   private String connectionUrl = cf.getString("SqlQueryResource/connectionUrl");
048   private boolean
049      allowUpdates = cf.getBoolean("SqlQueryResource/allowUpdates", false),
050      allowTempUpdates = cf.getBoolean("SqlQueryResource/allowTempUpdates", false),
051      includeRowNums = cf.getBoolean("SqlQueryResource/includeRowNums", false);
052
053   @Override /* Servlet */
054   public void init() {
055      try {
056         Class.forName(driver).newInstance();
057      } catch (Exception e) {
058         throw new RuntimeException(e);
059      }
060   }
061
062   /** GET request handler - Display the query entry page. */
063   @RestMethod(name="GET", path="/")
064   public ReaderResource doGet(RestRequest req) throws IOException {
065      return req.getReaderResource("SqlQueryResource.html", true);
066   }
067
068   /** POST request handler - Execute the query. */
069   @RestMethod(name="POST", path="/")
070   public List<Object> doPost(@Content PostInput in) throws Exception {
071
072      List<Object> results = new LinkedList<Object>();
073
074      // Don't try to submit empty input.
075      if (StringUtils.isEmpty(in.sql))
076         return results;
077
078      if (in.pos < 1 || in.pos > 10000)
079         throw new RestException(SC_BAD_REQUEST, "Invalid value for position.  Must be between 1-10000");
080      if (in.limit < 1 || in.limit > 10000)
081         throw new RestException(SC_BAD_REQUEST, "Invalid value for limit.  Must be between 1-10000");
082
083      // Create a connection and statement.
084      // If these fais, let the exception filter up as a 500 error.
085      Connection c = DriverManager.getConnection(connectionUrl);
086      c.setAutoCommit(false);
087      Statement st = c.createStatement();
088      String sql = null;
089
090      try {
091         for (String s : in.sql.split(";")) {
092            sql = s.trim();
093            if (! sql.isEmpty()) {
094               Object o = null;
095               if (allowUpdates || (allowTempUpdates && ! sql.matches("(?:i)commit.*"))) {
096                  if (st.execute(sql)) {
097                     ResultSet rs = st.getResultSet();
098                     o = new ResultSetList(rs, in.pos, in.limit, includeRowNums);
099                  } else {
100                     o = st.getUpdateCount();
101                  }
102               } else {
103                  ResultSet rs = st.executeQuery(sql);
104                  o = new ResultSetList(rs, in.pos, in.limit, includeRowNums);
105               }
106               results.add(o);
107            }
108         }
109         if (allowUpdates)
110            c.commit();
111         else if (allowTempUpdates)
112            c.rollback();
113      } catch (SQLException e) {
114         c.rollback();
115         throw new RestException(SC_BAD_REQUEST, "Invalid query:  {0}", sql).initCause(e);
116      } finally {
117         c.close();
118      }
119
120      return results;
121   }
122
123   /** The parsed form post */
124   public static class PostInput {
125      public String sql;
126      public int pos = 1, limit = 100;
127   }
128}
