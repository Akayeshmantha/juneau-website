013package org.apache.juneau.server.samples;
014
015import static org.apache.juneau.html.HtmlDocSerializerContext.*;
016
017import java.io.*;
018import java.util.*;
019
020import org.apache.juneau.html.annotation.*;
021import org.apache.juneau.internal.*;
022import org.apache.juneau.microservice.*;
023import org.apache.juneau.server.annotation.*;
024
025/**
026 * Servlet for viewing source code on classes whose Java files are present on the classpath.
027 * <p>
028 * This class is by no means perfect but is pretty much the best you can get using only regular expression matching.
029 *
030 * @author James Bognar (james.bognar@salesforce.com)
031 */
032@SuppressWarnings("serial")
033@RestResource(
034   path="/source",
035   messages="nls/SourceResource",
036   properties={
037      @Property(name=HTMLDOC_title, value="Source code viewer"),
038      @Property(name=HTMLDOC_cssImports, value="$R{servletURI}/htdocs/code-highlighting.css"),
039      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.SourceResource)'}"),
040   }
041)
042public class SourceResource extends Resource {
043
044   /** View source on the specified classes. */
045   @RestMethod(name="GET", path="/")
046   public Object getSource(@Param("classes") String[] classes) throws Exception {
047      if (classes == null)
048         return "Specify classes using ?classes=(class1,class2,....) attribute";
049      List<Object> l = new LinkedList<Object>();
050      for (String c : classes) {
051         try {
052            l.add(new Source(Class.forName(c)));
053         } catch (ClassNotFoundException e) {
054            l.add("Class " + c + " not found");
055         } catch (Exception e) {
056            l.add(e.getLocalizedMessage());
057         }
058      }
059      return l;
060   }
061
062   /**
063    * POJO that allows us to serialize HTML directly to the output.
064    */
065   @Html(asPlainText=true)
066   public static class Source {
067      private Class<?> c;
068      private Source(Class<?> c) {
069         this.c = c;
070      }
071      @Override /* Object */
072      public String toString() {
073         String filename = c.getSimpleName() + ".java";
074         InputStream is = c.getResourceAsStream('/' + c.getPackage().getName().replace('.','/') + '/' + filename);
075         if (is == null)
076            return "Source for class " + c.getName() + " not found";
077         StringBuilder sb = new StringBuilder();
078         try {
079               sb.append("<h3>").append(c.getSimpleName()).append(".java").append("</h3>");
080               sb.append("<p class='bcode'>");
081               sb.append(highlight(IOUtils.read(is), "java"));
082               sb.append("</p>");
083         } catch (Exception e) {
084            return e.getLocalizedMessage();
085         }
086         return sb.toString();
087      }
088   }
089
090   public static String highlight(String code, String lang) throws Exception {
091      if (lang.equalsIgnoreCase("xml")) {
092         code = code.replaceAll("&", "&amp;");
093         code = code.replaceAll("<", "&lt;");
094         code = code.replaceAll(">", "&gt;");
095         code = code.replaceAll("(&lt;[^\\s&]+&gt;)", "<xt>$1</xt>");
096         code = code.replaceAll("(&lt;[^\\s&]+)(\\s)", "<xt>$1</xt>$2");
097         code = code.replaceAll("(['\"])(/?&gt;)", "$1<xt>$2</xt>");
098         code = code.replaceAll("([\\S]+)=", "<xa>$1</xa>=");
099         code = code.replaceAll("=(['\"][^'\"]+['\"])", "=<xs>$1</xs>");
100      } else if (lang.equalsIgnoreCase("java")) {
101         code = code.replaceAll("&", "&amp;");
102         code = code.replaceAll("<", "&lt;");
103         code = code.replaceAll(">", "&gt;");
104         code = code.replaceAll("(?s)(\\/\\*\\*.*?\\*\\/)", "<jd>$1</jd>"); // javadoc comments
105         code = code.replaceAll("(@\\w+)", "<ja>$1</ja>"); // annotations
106         code = code.replaceAll("(?s)(?!\\/)(\\/\\*.*?\\*\\/)", "<jc>$1</jc>"); // C style comments
107         code = code.replaceAll("(?m)(\\/\\/.*)", "<jc>$1</jc>"); // C++ style comments
108         code = code.replaceAll("(?m)('[^'\n]*'|\"[^\"\n]*\")", "<js>$1</js>"); // quotes
109         code = code.replaceAll("(?<!@)(import|package|boolean|byte|char|double|float|final|static|transient|synchronized|private|protected|public|int|long|short|abstract|class|interface|extends|implements|null|true|false|void|break|case|catch|continue|default|do|else|finally|for|goto|if|instanceof|native|new|return|super|switch|this|threadsafe|throws|throw|try|while)(?=\\W)", "<jk>$1</jk>"); // quotes
110         code = code.replaceAll("<\\/jk>(\\s+)<jk>", "$1"); // quotes
111      }
112      return code;
113   }
114}
