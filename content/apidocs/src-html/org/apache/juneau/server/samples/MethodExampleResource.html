013package org.apache.juneau.server.samples;
014
015import static org.apache.juneau.html.HtmlDocSerializerContext.*;
016
017import java.util.*;
018
019import org.apache.juneau.microservice.*;
020import org.apache.juneau.server.*;
021import org.apache.juneau.server.annotation.*;
022
023/**
024 * Sample REST resource that shows how to define REST methods and OPTIONS pages
025 */
026@RestResource(
027   path="/methodExample",
028   messages="nls/MethodExampleResource",
029   properties={
030      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.MethodExampleResource)'}")
031   }
032)
033public class MethodExampleResource extends Resource {
034   private static final long serialVersionUID = 1L;
035
036   /** Example GET request that redirects to our example method */
037   @RestMethod(name="GET", path="/")
038   public Redirect doGetExample() throws Exception {
039      return new Redirect("example1/xxx/123/{0}/xRemainder?p1=123&p2=yyy", UUID.randomUUID());
040   }
041
042   /** Example GET request using annotated attributes */
043   @RestMethod(name="GET", path="/example1/{a1}/{a2}/{a3}/*", rc={200})
044   public String doGetExample1(
045         @Method String method,
046         @Attr String a1,
047         @Attr int a2,
048         @Attr UUID a3,
049         @Param("p1") int p1,
050         @Param("p2") String p2,
051         @Param("p3") UUID p3,
052         @PathRemainder String remainder,
053         @Header("Accept-Language") String lang,
054         @Header("Accept") String accept,
055         @Header("DNT") int doNotTrack
056      ) {
057      String output = String.format(
058            "method=%s, a1=%s, a2=%d, a3=%s, remainder=%s, p1=%d, p2=%s, p3=%s, lang=%s, accept=%s, dnt=%d",
059            method, a1, a2, a3, remainder, p1, p2, p3, lang, accept, doNotTrack);
060      return output;
061   }
062
063   /** Example GET request using methods on RestRequest and RestResponse */
064   @RestMethod(name="GET", path="/example2/{a1}/{a2}/{a3}/*", rc={200})
065   public void doGetExample2(RestRequest req, RestResponse res) throws Exception {
066      String method = req.getMethod();
067
068      // Attributes (from URL pattern variables)
069      String a1 = req.getAttribute("a1", String.class);
070      int a2 = req.getAttribute("a2", int.class);
071      UUID a3 = req.getAttribute("a3", UUID.class);
072
073      // Optional GET parameters
074      int p1 = req.getParameter("p1", int.class, 0);
075      String p2 = req.getParameter("p2", String.class);
076      UUID p3 = req.getParameter("p3", UUID.class);
077
078      // URL pattern post-match
079      String remainder = req.getPathRemainder();
080
081      // Headers
082      String lang = req.getHeader("Accept-Language");
083      int doNotTrack = req.getHeader("DNT", int.class);
084
085      // Send back a simple String response
086      String output = String.format(
087            "method=%s, a1=%s, a2=%d, a3=%s, remainder=%s, p1=%d, p2=%s, p3=%s, lang=%s, dnt=%d",
088            method, a1, a2, a3, remainder, p1, p2, p3, lang, doNotTrack);
089      res.setOutput(output);
090   }
091}
