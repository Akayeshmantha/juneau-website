013package org.apache.juneau.server.samples;
014
015import static org.apache.juneau.html.HtmlDocSerializerContext.*;
016
017import java.lang.Object;
018
019import org.apache.juneau.*;
020import org.apache.juneau.client.*;
021import org.apache.juneau.dto.Link;
022import org.apache.juneau.html.dto.*;
023import org.apache.juneau.json.*;
024import org.apache.juneau.microservice.*;
025import org.apache.juneau.server.annotation.*;
026
027@RestResource(
028   path="/tumblrParser",
029   messages="nls/TumblrParserResource",
030   properties={
031      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'?method=OPTIONS',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.TumblrParserResource)'}"),
032      @Property(name=HTMLDOC_title, value="Tumblr parser service"),
033      @Property(name=HTMLDOC_description, value="Specify a URL to a Tumblr blog and parse the results.")
034   }
035)
036public class TumblrParserResource extends Resource {
037   private static final long serialVersionUID = 1L;
038
039   @RestMethod(name="GET", path="/")
040   public String getInstructions() throws Exception {
041      return "Append the Tumblr blog name to the URL above (e.g. /juneau/sample/tumblrParser/mytumblrblog)";
042   }
043
044   @RestMethod(name="GET", path="/{blogName}")
045   public ObjectList parseBlog(@Attr String blogName) throws Exception {
046      ObjectList l = new ObjectList();
047      RestClient rc = new RestClient(JsonSerializer.class, JsonParser.class);
048      try {
049         String site = "http://" + blogName + ".tumblr.com/api/read/json";
050         ObjectMap m = rc.doGet(site).getResponse(ObjectMap.class);
051         int postsTotal = m.getInt("posts-total");
052         for (int i = 0; i < postsTotal; i += 20) {
053            m = rc.doGet(site + "?start=" + i + "&num=20&filter=text").getResponse(ObjectMap.class);
054            ObjectList ol = m.getObjectList("posts");
055            for (int j = 0; j < ol.size(); j++) {
056               ObjectMap om = ol.getObjectMap(j);
057               String type = om.getString("type");
058               Entry e = new Entry();
059               e.date = om.getString("date");
060               if (type.equals("link"))
061                  e.entry = new Link(om.getString("link-text"), om.getString("link-url"));
062               else if (type.equals("audio"))
063                  e.entry = new ObjectMap().append("type","audio").append("audio-caption", om.getString("audio-caption"));
064               else if (type.equals("video"))
065                  e.entry = new ObjectMap().append("type","video").append("video-caption", om.getString("video-caption"));
066               else if (type.equals("quote"))
067                  e.entry = new ObjectMap().append("type","quote").append("quote-source", om.getString("quote-source")).append("quote-text", om.getString("quote-text"));
068               else if (type.equals("regular"))
069                  e.entry = om.getString("regular-body");
070               else if (type.equals("photo"))
071                  e.entry = new Img(om.getString("photo-url-250"));
072               else
073                  e.entry = new ObjectMap().append("type", type);
074               l.add(e);
075            }
076         }
077      } finally {
078         rc.closeQuietly();
079      }
080      return l;
081   }
082
083   public static class Entry {
084      public String date;
085      public Object entry;
086   }
087}
