013package org.apache.juneau.server.samples;
014
015import static org.apache.juneau.html.HtmlDocSerializerContext.*;
016
017import java.io.*;
018
019import org.apache.commons.fileupload.*;
020import org.apache.commons.fileupload.servlet.*;
021import org.apache.juneau.server.*;
022import org.apache.juneau.server.annotation.*;
023import org.apache.juneau.utils.*;
024
025/**
026 * Sample resource that extends {@link DirectoryResource} to open up the temp directory as a REST resource.
027 */
028@RestResource(
029   path="/tempDir",
030   messages="nls/TempDirResource",
031   properties={
032      @Property(name="rootDir", value="$S{java.io.tmpdir}"),
033      @Property(name="allowViews", value="true"),
034      @Property(name="allowDeletes", value="true"),
035      @Property(name="allowPuts", value="false"),
036      @Property(name=HTMLDOC_links, value="{up:'$R{requestParentURI}',options:'$R{servletURI}?method=OPTIONS',upload:'upload',source:'$R{servletParentURI}/source?classes=(org.apache.juneau.server.samples.TempDirResource,org.apache.juneau.server.samples.DirectoryResource)'}"),
037   },
038   stylesheet="styles/devops.css"
039)
040public class TempDirResource extends DirectoryResource {
041   private static final long serialVersionUID = 1L;
042
043   /**
044    * [GET /upload] - Display the form entry page for uploading a file to the temp directory.
045    */
046   @RestMethod(name="GET", path="/upload")
047   public ReaderResource getUploadPage(RestRequest req) throws IOException {
048      return req.getReaderResource("TempDirUploadPage.html", true);
049   }
050
051   /**
052    * [POST /upload] - Upload a file as a multipart form post.
053    * Shows how to use the Apache Commons ServletFileUpload class for handling multi-part form posts.
054    */
055   @RestMethod(name="POST", path="/upload", matchers=TempDirResource.MultipartFormDataMatcher.class)
056   public Redirect uploadFile(RestRequest req) throws Exception {
057      ServletFileUpload upload = new ServletFileUpload();
058      FileItemIterator iter = upload.getItemIterator(req);
059      while (iter.hasNext()) {
060         FileItemStream item = iter.next();
061         if (item.getFieldName().equals("contents")) { //$NON-NLS-1$
062            File f = new File(getRootDir(), item.getName());
063            IOPipe.create(item.openStream(), new FileOutputStream(f)).closeOut().run();
064         }
065      }
066      return new Redirect(); // Redirect to the servlet root.
067   }
068
069   /** Causes a 404 if POST isn't multipart/form-data */
070   public static class MultipartFormDataMatcher extends RestMatcher {
071      @Override /* RestMatcher */
072      public boolean matches(RestRequest req) {
073         String contentType = req.getContentType();
074         return contentType != null && contentType.startsWith("multipart/form-data"); //$NON-NLS-1$
075      }
076   }
077}
