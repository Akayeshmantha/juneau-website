027/**
028 * Context object that lives for the duration of a single use of {@link Serializer}.
029 * <p>
030 * Used by serializers for the following purposes:
031 * <ul class='spaced-list'>
032 *    <li>Keeping track of how deep it is in a model for indentation purposes.
033 *    <li>Ensuring infinite loops don't occur by setting a limit on how deep to traverse a model.
034 *    <li>Ensuring infinite loops don't occur from loops in the model (when detectRecursions is enabled.
035 *    <li>Allowing serializer properties to be overridden on method calls.
036 * </ul>
037 * <p>
038 * This class is NOT thread safe.  It is meant to be discarded after one-time use.
039 *
040 * @author James Bognar (james.bognar@salesforce.com)
041 */
042public class SerializerSession extends Session {

081   /**
082    * Create a new session using properties specified in the context.
083    *
084    * @param ctx The context creating this session object.
085    *    The context contains all the configuration settings for this object.
086    * @param beanContext The bean context being used.
087    * @param output The output object.
088    *    <br>Character-based serializers can handle the following output class types:
089    *    <ul>
090    *       <li>{@link Writer}
091    *       <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
092    *       <li>{@link File} - Output will be written as system-default encoded stream.
093    *    </ul>
094    *    <br>Stream-based serializers can handle the following output class types:
095    *    <ul>
096    *       <li>{@link OutputStream}
097    *       <li>{@link File}
098    *    </ul>
099    * @param op The override properties.
100    *    These override any context properties defined in the context.
101    * @param javaMethod The java method that called this parser, usually the method in a REST servlet.
102    */
103   public SerializerSession(SerializerContext ctx, BeanContext beanContext, Object output, ObjectMap op, Method javaMethod) {
104      super(ctx);
105      this.beanContext = beanContext;
106      this.javaMethod = javaMethod;
107      this.output = output;
108      if (op == null || op.isEmpty()) {
109         overrideProperties = new ObjectMap();
110         maxDepth = ctx.maxDepth;
111         initialDepth = ctx.initialDepth;
112         debug = ctx.debug;
113         detectRecursions = ctx.detectRecursions;
114         ignoreRecursions = ctx.ignoreRecursions;
115         useIndentation = ctx.useIndentation;
116         addBeanTypeProperties = ctx.addBeanTypeProperties;
117         trimNulls = ctx.trimNulls;
118         trimEmptyCollections = ctx.trimEmptyCollections;
119         trimEmptyMaps = ctx.trimEmptyMaps;
120         trimStrings = ctx.trimStrings;
121         quoteChar = ctx.quoteChar;
122         relativeUriBase = ctx.relativeUriBase;
123         absolutePathUriBase = ctx.absolutePathUriBase;
124         sortCollections = ctx.sortCollections;
125         sortMaps = ctx.sortMaps;
126      } else {
127         overrideProperties = op;
128         maxDepth = op.getInt(SERIALIZER_maxDepth, ctx.maxDepth);
129         initialDepth = op.getInt(SERIALIZER_initialDepth, ctx.initialDepth);
130         debug = op.getBoolean(SERIALIZER_debug, ctx.debug);
131         detectRecursions = op.getBoolean(SERIALIZER_detectRecursions, ctx.detectRecursions);
132         ignoreRecursions = op.getBoolean(SERIALIZER_ignoreRecursions, ctx.ignoreRecursions);
133         useIndentation = op.getBoolean(SERIALIZER_useIndentation, ctx.useIndentation);
134         addBeanTypeProperties = op.getBoolean(SERIALIZER_addBeanTypeProperties, ctx.addBeanTypeProperties);
135         trimNulls = op.getBoolean(SERIALIZER_trimNullProperties, ctx.trimNulls);
136         trimEmptyCollections = op.getBoolean(SERIALIZER_trimEmptyCollections, ctx.trimEmptyCollections);
137         trimEmptyMaps = op.getBoolean(SERIALIZER_trimEmptyMaps, ctx.trimEmptyMaps);
138         trimStrings = op.getBoolean(SERIALIZER_trimStrings, ctx.trimStrings);
139         quoteChar = op.getString(SERIALIZER_quoteChar, ""+ctx.quoteChar).charAt(0);
140         relativeUriBase = op.getString(SERIALIZER_relativeUriBase, ctx.relativeUriBase);
141         absolutePathUriBase = op.getString(SERIALIZER_absolutePathUriBase, ctx.absolutePathUriBase);
142         sortCollections = op.getBoolean(SERIALIZER_sortCollections, ctx.sortMaps);
143         sortMaps = op.getBoolean(SERIALIZER_sortMaps, ctx.sortMaps);
144      }
145
146      this.indent = initialDepth;
147      if (detectRecursions || debug) {
148         set = new IdentityHashMap<Object,Object>();
149      } else {
150         set = Collections.emptyMap();
151      }
152   }

154   /**
155    * Wraps the specified output object inside an output stream.
156    * Subclasses can override this method to implement their own specialized output streams.
157    * <p>
158    * This method can be used if the output object is any of the following class types:
159    * <ul>
160    *    <li>{@link OutputStream}
161    *    <li>{@link File}
162    * </ul>
163    *
164    * @return The output object wrapped in an output stream.
165    * @throws Exception If object could not be converted to an output stream.
166    */
167   public OutputStream getOutputStream() throws Exception {
168      if (output == null)
169         throw new SerializeException("Output cannot be null.");
170      if (output instanceof OutputStream)
171         return (OutputStream)output;
172      if (output instanceof File) {
173         if (outputStream == null)
174            outputStream = new BufferedOutputStream(new FileOutputStream((File)output));
175         return outputStream;
176      }
177      throw new SerializeException("Cannot convert object of type {0} to an OutputStream.", output.getClass().getName());
178   }

181   /**
182    * Wraps the specified output object inside a writer.
183    * Subclasses can override this method to implement their own specialized writers.
184    * <p>
185    * This method can be used if the output object is any of the following class types:
186    * <ul>
187    *    <li>{@link Writer}
188    *    <li>{@link OutputStream} - Output will be written as UTF-8 encoded stream.
189    *    <li>{@link File} - Output will be written as system-default encoded stream.
190    * </ul>
191    *
192    * @return The output object wrapped in a Writer.
193    * @throws Exception If object could not be converted to a writer.
194    */
195   public Writer getWriter() throws Exception {
196      if (output == null)
197         throw new SerializeException("Output cannot be null.");
198      if (output instanceof Writer)
199         return (Writer)output;
200      if (output instanceof OutputStream) {
201         if (flushOnlyWriter == null)
202            flushOnlyWriter = new OutputStreamWriter((OutputStream)output, IOUtils.UTF8);
203         return flushOnlyWriter;
204      }
205      if (output instanceof File) {
206         if (writer == null)
207            writer = new OutputStreamWriter(new BufferedOutputStream(new FileOutputStream((File)output)));
208         return writer;
209      }
210      throw new SerializeException("Cannot convert object of type {0} to a Writer.", output.getClass().getName());
211   }

412   /**
413    * Push the specified object onto the stack.
414    *
415    * @param attrName The attribute name.
416    * @param o The current object being serialized.
417    * @param eType The expected class type.
418    * @return The {@link ClassMeta} of the object so that <code>instanceof</code> operations
419    *    only need to be performed once (since they can be expensive).<br>
420    * @throws SerializeException If recursion occurred.
421    */
422   public ClassMeta<?> push(String attrName, Object o, ClassMeta<?> eType) throws SerializeException {
423      indent++;
424      isBottom = true;
425      if (o == null)
426         return null;
427      Class<?> c = o.getClass();
428      ClassMeta<?> cm = (eType != null && c == eType.getInnerClass()) ? eType : beanContext.getClassMeta(c);
429      if (cm.isCharSequence() || cm.isNumber() || cm.isBoolean())
430         return cm;
431      if (detectRecursions || debug) {
432         if (stack.size() > maxDepth)
433            return null;
434         if (willRecurse(attrName, o, cm))
435            return null;
436         isBottom = false;
437         stack.add(new StackElement(stack.size(), attrName, o, cm));
438         if (debug)
439            logger.info(getStack(false));
440         set.put(o, o);
441      }
442      return cm;
443   }

445   /**
446    * Returns <jk>true</jk> if {@link SerializerContext#SERIALIZER_detectRecursions} is enabled, and the specified
447    *    object is already higher up in the serialization chain.
448    *
449    * @param attrName The bean property attribute name, or some other identifier.
450    * @param o The object to check for recursion.
451    * @param cm The metadata on the object class.
452    * @return <jk>true</jk> if recursion detected.
453    * @throws SerializeException If recursion occurred.
454    */
455   public boolean willRecurse(String attrName, Object o, ClassMeta<?> cm) throws SerializeException {
456      if (! (detectRecursions || debug))
457         return false;
458      if (! set.containsKey(o))
459         return false;
460      if (ignoreRecursions && ! debug)
461         return true;
462
463      stack.add(new StackElement(stack.size(), attrName, o, cm));
464      throw new SerializeException("Recursion occurred, stack={0}", getStack(true));
465   }

467   /**
468    * Pop an object off the stack.
469    */
470   public void pop() {
471      indent--;
472      if ((detectRecursions || debug) && ! isBottom)  {
473         Object o = stack.removeLast().o;
474         Object o2 = set.remove(o);
475         if (o2 == null)
476            addWarning("Couldn't remove object of type ''{0}'' on attribute ''{1}'' from object stack.", o.getClass().getName(), stack);
477      }
478      isBottom = false;
479   }

528   /**
529    * Generalize the specified object if a POJO swap is associated with it.
530    *
531    * @param o The object to generalize.
532    * @param type The type of object.
533    * @return The generalized object, or <jk>null</jk> if the object is <jk>null</jk>.
534    * @throws SerializeException If a problem occurred trying to convert the output.
535    */
536   @SuppressWarnings({ "rawtypes", "unchecked" })
537   public final Object generalize(Object o, ClassMeta<?> type) throws SerializeException {
538      if (o == null)
539         return null;
540      PojoSwap f = (type == null || type.isObject() ? getBeanContext().getClassMeta(o.getClass()).getPojoSwap() : type.getPojoSwap());
541      if (f == null)
542         return o;
543      return f.swap(o, getBeanContext());
544   }

546   /**
547    * Returns <jk>true</jk> if the specified value should not be serialized.
548    *
549    * @param cm The class type of the object being serialized.
550    * @param attrName The bean attribute name, or <jk>null</jk> if this isn't a bean attribute.
551    * @param value The object being serialized.
552    * @return <jk>true</jk> if the specified value should not be serialized.
553    * @throws SerializeException If recursion occurred.
554    */
555   public final boolean canIgnoreValue(ClassMeta<?> cm, String attrName, Object value) throws SerializeException {
556
557      if (trimNulls && value == null)
558         return true;
559
560      if (value == null)
561         return false;
562
563      if (cm == null)
564         cm = getBeanContext().object();
565
566      if (trimEmptyCollections) {
567         if (cm.isArray() || (cm.isObject() && value.getClass().isArray())) {
568            if (((Object[])value).length == 0)
569               return true;
570         }
571         if (cm.isCollection() || (cm.isObject() && isParentClass(Collection.class, value.getClass()))) {
572            if (((Collection<?>)value).isEmpty())
573               return true;
574         }
575      }
576
577      if (trimEmptyMaps) {
578         if (cm.isMap() || (cm.isObject() && isParentClass(Map.class, value.getClass()))) {
579            if (((Map<?,?>)value).isEmpty())
580               return true;
581         }
582      }
583
584      if (trimNulls && willRecurse(attrName, value, cm))
585         return true;
586
587      return false;
588   }

614   /**
615    * Converts a String to an absolute URI based on the {@link SerializerContext#SERIALIZER_absolutePathUriBase} and
616    *    {@link SerializerContext#SERIALIZER_relativeUriBase} settings on this context.
617    *
618    * @param uri The input URI.
619    * @return The resolved URI.
620    */
621   public String resolveUri(String uri) {
622      if (uri.indexOf("://") != -1 || (absolutePathUriBase == null && relativeUriBase == null))
623         return uri;
624      StringBuilder sb = new StringBuilder(uri.length() + absolutePathUriBase.length() + 1 + relativeUriBase.length());
625      if (StringUtils.startsWith(uri, '/')) {
626         if (absolutePathUriBase != null)
627            sb.append(absolutePathUriBase);
628      } else {
629         if (relativeUriBase != null) {
630            sb.append(relativeUriBase);
631            if (! uri.equals("/"))
632               sb.append("/");
633         }
634      }
635      sb.append(uri);
636      return sb.toString();
637   }

656   /**
657    * Perform cleanup on this context object if necessary.
658    *
659    * @throws SerializeException If we're in debug mode and one or more warnings occurred.
660    */
661   public void close() throws SerializeException {
662      if (closed)
663         throw new SerializeException("Attempt to close SerializerSession more than once.");
664
665      try {
666         if (outputStream != null)
667            outputStream.close();
668         if (flushOnlyWriter != null)
669            flushOnlyWriter.flush();
670         if (writer != null)
671            writer.close();
672      } catch (IOException e) {
673         throw new SerializeException(e);
674      }
675
676      if (debug && warnings.size() > 0)
677         throw new SerializeException("Warnings occurred during serialization: \n" + StringUtils.join(warnings, "\n"));
678
679      closed = true;
680   }
681
682   @Override /* Object */
683   protected void finalize() throws Throwable {
684      if (! closed)
685         throw new RuntimeException("SerializerSession was not closed.");
686   }

688   private static class StackElement {
689      private int depth;
690      private String name;
691      private Object o;
692      private ClassMeta<?> aType;
693
694      private StackElement(int depth, String name, Object o, ClassMeta<?> aType) {
695         this.depth = depth;
696         this.name = name;
697         this.o = o;
698         this.aType = aType;
699      }
700
701      private String toString(boolean simple) {
702         StringBuilder sb = new StringBuilder().append('[').append(depth).append(']');
703         sb.append(StringUtils.isEmpty(name) ? "<noname>" : name).append(':');
704         sb.append(aType.toString(simple));
705         if (aType != aType.getSerializedClassMeta())
706            sb.append('/').append(aType.getSerializedClassMeta().toString(simple));
707         return sb.toString();
708      }
709   }
710
711   private String getStack(boolean full) {
712      StringBuilder sb = new StringBuilder();
713      for (StackElement e : stack) {
714         if (full) {
715            sb.append("\n\t");
716            for (int i = 1; i < e.depth; i++)
717               sb.append("  ");
718            if (e.depth > 0)
719               sb.append("->");
720            sb.append(e.toString(false));
721         } else {
722            sb.append(" > ").append(e.toString(true));
723         }
724      }
725      return sb.toString();
726   }

728   /**
729    * Returns information used to determine at what location in the parse a failure occurred.
730    *
731    * @return A map, typically containing something like <code>{line:123,column:456,currentProperty:"foobar"}</code>
732    */
733   public Map<String,Object> getLastLocation() {
734      Map<String,Object> m = new LinkedHashMap<String,Object>();
735      if (currentClass != null)
736         m.put("currentClass", currentClass);
737      if (currentProperty != null)
738         m.put("currentProperty", currentProperty);
739      if (stack != null && ! stack.isEmpty())
740         m.put("stack", stack);
741      return m;
742   }
