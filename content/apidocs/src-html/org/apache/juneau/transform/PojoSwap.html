013package org.apache.juneau.transform;
014
015import java.lang.reflect.*;
016import java.util.*;
017
018import org.apache.juneau.*;
019import org.apache.juneau.internal.*;
020import org.apache.juneau.parser.*;
021import org.apache.juneau.serializer.*;
022
023/**
024 * Used to swap out non-serializable objects with serializable replacements during serialization, and vis-versa during parsing.
025 *
026 *
027 * <h6 class='topic'>Description</h6>
028 * <p>
029 *    <code>PojoSwaps</code> are used to extend the functionality of the serializers and parsers to be able to handle POJOs
030 *    that aren't automatically handled by the serializers or parsers.  For example, JSON does not have a standard
031 *    representation for rendering dates.  By defining a special {@code Date} swap and associating it with a serializer and
032 *    parser, you can convert a {@code Date} object to a {@code String} during serialization, and convert that {@code String} object back into
033 *    a {@code Date} object during parsing.
034 * <p>
035 *    Swaps MUST declare a public no-arg constructor so that the bean context can instantiate them.
036 * <p>
037 *    <code>PojoSwaps</code> are associated with instances of {@link BeanContext BeanContexts} by passing the swap class to
038 *    the {@link CoreApi#addPojoSwaps(Class...)} method.<br>
039 *    When associated with a bean context, fields of the specified type will automatically be converted when the
040 *    {@link BeanMap#get(Object)} or {@link BeanMap#put(String, Object)} methods are called.<br>
041 * <p>
042 *    <code>PojoSwaps</code> have two parameters:
043 *    <ol>
044 *       <li>{@code <T>} - The normal representation of an object.
045 *       <li>{@code <S>} - The swapped representation of an object.
046 *    </ol>
047 *    <br>
048 *    {@link Serializer Serializers} use swaps to convert objects of type T into objects of type S, and on calls to {@link BeanMap#get(Object)}.<br>
049 *    {@link Parser Parsers} use swaps to convert objects of type S into objects of type T, and on calls to {@link BeanMap#put(String,Object)}.
050 *
051 *
052 * <h6 class='topic'>Swap Class Type {@code <S>}</h6>
053 * <p>
054 *    The swapped object representation of an object must be an object type that the serializers can
055 *    natively convert to JSON (or language-specific equivalent).  The list of valid transformed types are as follows...
056 *    <ul class='spaced-list'>
057 *       <li>{@link String}
058 *       <li>{@link Number}
059 *       <li>{@link Boolean}
060 *       <li>{@link Collection} containing anything on this list.
061 *       <li>{@link Map} containing anything on this list.
062 *       <li>A java bean with properties of anything on this list.
063 *       <li>An array of anything on this list.
064 *    </ul>
065 *
066 *
067 * <h6 class='topic'>Normal Class Type {@code <T>}</h6>
068 * <p>
069 *    The normal object representation of an object.<br>
070 *
071 *
072 * <h6 class='topic'>One-way vs. Two-way Serialization</h6>
073 * <p>
074 *    Note that while there is a unified interface for handling swaps during both serialization and parsing,
075 *    in many cases only one of the {@link #swap(Object)} or {@link #unswap(Object)} methods will be defined
076 *    because the swap is one-way.  For example, a swap may be defined to convert an {@code Iterator} to a {@code ObjectList}, but
077 *    it's not possible to unswap an {@code Iterator}.  In that case, the {@code swap(Object}} method would
078 *    be implemented, but the {@code unswap(ObjectMap)} object would not, and the swap would be associated on
079 *    the serializer, but not the parser.  Also, you may choose to serialize objects like {@code Dates} to readable {@code Strings},
080 *    in which case it's not possible to reparse it back into a {@code Date}, since there is no way for the {@code Parser} to
081 *    know it's a {@code Date} from just the JSON or XML text.
082 *
083 *
084 * <h6 class='topic'>Additional information</h6>
085 *    See {@link org.apache.juneau.transform} for more information.
086 *
087 *
088 * @author James Bognar (james.bognar@salesforce.com)
089 * @param <T> The normal form of the class.
090 * @param <S> The swapped form of the class.
091 */
092public abstract class PojoSwap<T,S> {
093
094   private final Class<T> normalClass;
095   private final Class<S> swapClass;
096   private ClassMeta<S> swapClassMeta;
097
098   /**
099    * Constructor.
100    */
101   @SuppressWarnings("unchecked")
102   protected PojoSwap() {
103
104      Class<?> t_normalClass = null, t_swapClass = null;
105
106      Class<?> c = this.getClass().getSuperclass();
107      Type t = this.getClass().getGenericSuperclass();
108      while (c != PojoSwap.class) {
109         t = c.getGenericSuperclass();
110         c = c.getSuperclass();
111      }
112
113      // Attempt to determine the T and G classes using reflection.
114      if (t instanceof ParameterizedType) {
115         ParameterizedType pt = (ParameterizedType)t;
116         Type[] pta = pt.getActualTypeArguments();
117         if (pta.length == 2) {
118            Type nType = pta[0];
119            if (nType instanceof Class) {
120               t_normalClass = (Class<T>)nType;
121
122            // <byte[],x> ends up containing a GenericArrayType, so it has to
123            // be handled as a special case.
124            } else if (nType instanceof GenericArrayType) {
125               Class<?> cmpntType = (Class<?>)((GenericArrayType)nType).getGenericComponentType();
126               t_normalClass = Array.newInstance(cmpntType, 0).getClass();
127
128            // <Class<?>,x> ends up containing a ParameterizedType, so just use the raw type.
129            } else if (nType instanceof ParameterizedType) {
130               t_normalClass = (Class<T>)((ParameterizedType)nType).getRawType();
131
132            } else
133               throw new RuntimeException("Unsupported parameter type: " + nType);
134            if (pta[1] instanceof Class)
135               t_swapClass = (Class<S>)pta[1];
136            else if (pta[1] instanceof ParameterizedType)
137               t_swapClass = (Class<S>)((ParameterizedType)pta[1]).getRawType();
138            else
139               throw new RuntimeException("Unexpected transformed class type: " + pta[1].getClass().getName());
140         }
141      }
142
143      this.normalClass = (Class<T>)t_normalClass;
144      this.swapClass = (Class<S>)t_swapClass;
145   }
146
147   /**
148    * Constructor for when the normal and transformed classes are already known.
149    *
150    * @param normalClass The normal class (cannot be serialized).
151    * @param swapClass The transformed class (serializable).
152    */
153   protected PojoSwap(Class<T> normalClass, Class<S> swapClass) {
154      this.normalClass = normalClass;
155      this.swapClass = swapClass;
156   }
157
158   /**
159    * If this transform is to be used to serialize non-serializable POJOs, it must implement this method.
160    * <p>
161    *    The object must be converted into one of the following serializable types:
162    *    <ul class='spaced-list'>
163    *       <li>{@link String}
164    *       <li>{@link Number}
165    *       <li>{@link Boolean}
166    *       <li>{@link Collection} containing anything on this list.
167    *       <li>{@link Map} containing anything on this list.
168    *       <li>A java bean with properties of anything on this list.
169    *       <li>An array of anything on this list.
170    *    </ul>
171    *
172    * @param o The object to be transformed.
173    * @return The transformed object.
174    * @throws SerializeException If a problem occurred trying to convert the output.
175    */
176   public S swap(T o) throws SerializeException {
177      throw new SerializeException("Swap method not implemented on PojoSwap ''{0}''", this.getClass().getName());
178   }
179
180   /**
181    * Same as {@link #swap(Object)}, but override this method instead if you want access to the bean context that created this swap.
182    *
183    * @param o The object to be transformed.
184    * @param beanContext The bean context to use to get the class meta.
185    *    This is always going to be the same bean context that created this swap.
186    * @return The transformed object.
187    * @throws SerializeException If a problem occurred trying to convert the output.
188    */
189   public S swap(T o, BeanContext beanContext) throws SerializeException {
190      return swap(o);
191   }
192
193   /**
194    * If this transform is to be used to reconstitute POJOs that aren't true Java beans, it must implement this method.
195    *
196    * @param f The transformed object.
197    *    This may be <jk>null</jk> if the parser cannot make this determination.
198    * @return The narrowed object.
199    * @throws ParseException If this method is not implemented.
200    */
201   public T unswap(S f) throws ParseException {
202      throw new ParseException("Unswap method not implemented on PojoSwap ''{0}''", this.getClass().getName());
203   }
204
205   /**
206    * Same as {@link #unswap(Object)}, but override this method if you need access to the real class type or the bean context that created this swap.
207    *
208    * @param f The transformed object.
209    * @param hint If possible, the parser will try to tell you the object type being created.  For example,
210    *    on a serialized date, this may tell you that the object being created must be of type {@code GregorianCalendar}.<br>
211    *    This may be <jk>null</jk> if the parser cannot make this determination.
212    * @param beanContext The bean context to use to get the class meta.
213    *    This is always going to be the same bean context that created this swap.
214    * @return The narrowed object.
215    * @throws ParseException If this method is not implemented.
216    */
217   public T unswap(S f, ClassMeta<?> hint, BeanContext beanContext) throws ParseException {
218      return unswap(f);
219   }
220
221   /**
222    * Returns the T class, the normalized form of the class.
223    *
224    * @return The normal form of this class.
225    */
226   public Class<T> getNormalClass() {
227      return normalClass;
228   }
229
230   /**
231    * Returns the G class, the generialized form of the class.
232    * <p>
233    *    Subclasses must override this method if the generialized class is {@code Object},
234    *    meaning it can produce multiple generialized forms.
235    *
236    * @return The transformed form of this class.
237    */
238   public Class<S> getSwapClass() {
239      return swapClass;
240   }
241
242   /**
243    * Returns the {@link ClassMeta} of the transformed class type.
244    * This value is cached for quick lookup.
245    *
246    * @param beanContext The bean context to use to get the class meta.
247    *    This is always going to be the same bean context that created this swap.
248    * @return The {@link ClassMeta} of the transformed class type.
249    */
250   public ClassMeta<S> getSwapClassMeta(BeanContext beanContext) {
251      if (swapClassMeta == null)
252         swapClassMeta = beanContext.getClassMeta(swapClass);
253      return swapClassMeta;
254   }
255
256   /**
257    * Checks if the specified object is an instance of the normal class defined on this swap.
258    *
259    * @param o The object to check.
260    * @return <jk>true</jk> if the specified object is a subclass of the normal class defined on this transform.
261    *    <jk>null</jk> always return <jk>false</jk>.
262    */
263   public boolean isNormalObject(Object o) {
264      if (o == null)
265         return false;
266      return ClassUtils.isParentClass(normalClass, o.getClass());
267   }
268
269   /**
270    * Checks if the specified object is an instance of the swap class defined on this swap.
271    *
272    * @param o The object to check.
273    * @return <jk>true</jk> if the specified object is a subclass of the transformed class defined on this transform.
274    *    <jk>null</jk> always return <jk>false</jk>.
275    */
276   public boolean isSwappedObject(Object o) {
277      if (o == null)
278         return false;
279      return ClassUtils.isParentClass(swapClass, o.getClass());
280   }
281
282   //--------------------------------------------------------------------------------
283   // Overridden methods
284   //--------------------------------------------------------------------------------
285
286   @Override /* Object */
287   public String toString() {
288      return getClass().getSimpleName() + '<' + getNormalClass().getSimpleName() + "," + getSwapClass().getSimpleName() + '>';
289   }
290}
