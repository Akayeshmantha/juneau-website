023/**
024 * Parent class for all bean filters.
025 * <p>
026 *    Bean filters are used to control aspects of how beans are handled during serialization and parsing.
027 * <p>
028 *    This class can be considered a programmatic equivalent to using the {@link Bean @Bean} annotation on bean classes.
029 *    Thus, it can be used to perform the same function as the <code>@Bean</code> annotation when you don't have
030 *       the ability to annotate those classes (e.g. you don't have access to the source code).
031 *
032 * <p>
033 *    When defining bean filters, you can either call the setters in the contructor, or override getters.
034 *
035 * <h6 class='topic'>Example</h6>
036 * <p class='bcode'>
037 *    <jc>// Create our serializer with a bean filter.</jc> 
038 *    WriterSerializer s = <jk>new</jk> JsonSerializer().addBeanFilters(AddressFilter.<jk>class</jk>); 
039 *
040 *    Address a = <jk>new</jk> Address();
041 *    String json = s.serialize(a); <jc>// Serializes only street, city, state.</jc>
042 *
043 *    <jc>// Filter class defined via setters</jc>
044 *    <jk>public class</jk> AddressFilter <jk>extends</jk> BeanFilter&lt;Address&gt; {
045 *       <jk>public</jk> AddressFilter() {
046 *          setProperties(<js>"street"</js>,<js>"city"</js>,<js>"state"</js>);
047 *       }
048 *    }
049 *
050 *    <jc>// Filter class defined by overriding getters</jc>
051 *    <jk>public class</jk> AddressFilter <jk>extends</jk> BeanFilter&lt;Address&gt; {
052 *       <jk>public</jk> String[] getProperties() {
053 *          <jk>return new</jk> String[]{<js>"street"</js>,<js>"city"</js>,<js>"state"</js>};
054 *       }
055 *    }
056 * </p>
057 * <p>
058 *    The examples in this class use the setters approach.
059 *
060 * <h6 class='topic'>Additional information</h6>
061 *    See {@link org.apache.juneau.transform} for more information.
062 *
063 *
064 * @author James Bognar (james.bognar@salesforce.com)
065 * @param <T> The class type that this filter applies to.
066 */
067public abstract class BeanFilter<T> {
068
069   private Class<T> beanClass;
070   private String[] properties, excludeProperties;
071   private LinkedHashMap<Class<?>, String> subTypes;
072   private String subTypeAttr;
073   private Class<? extends PropertyNamer> propertyNamer;
074   private Class<?> interfaceClass, stopClass;
075   private boolean sortProperties;
076
077   /**
078    * Constructor that determines the for-class value using reflection.
079    */
080   @SuppressWarnings("unchecked")
081   public BeanFilter() {
082      super();
083
084      Class<?> c = this.getClass().getSuperclass();
085      Type t = this.getClass().getGenericSuperclass();
086      while (c != BeanFilter.class) {
087         t = c.getGenericSuperclass();
088         c = c.getSuperclass();
089      }
090
091      // Attempt to determine the T and G classes using reflection.
092      if (t instanceof ParameterizedType) {
093         ParameterizedType pt = (ParameterizedType)t;
094         Type[] pta = pt.getActualTypeArguments();
095         if (pta.length > 0) {
096            Type nType = pta[0];
097            if (nType instanceof Class)
098               this.beanClass = (Class<T>)nType;
099
100            else
101               throw new RuntimeException("Unsupported parameter type: " + nType);
102         }
103      }
104   }

110    *

113    *    <dd>
114    *       <p class='bcode'>
115    *    <jk>public class</jk> SomeArbitraryFilter <jk>extends</jk> BeanFilter&lt?&gt; {

312   /**
313    * Defines a virtual property on a superclass that identifies bean subtype classes.
314    * <p>
315    *    In the following example, the abstract class has two subclasses that are differentiated
316    *       by a property called <code>subType</code>
317    *
318    * <p class='bcode'>
319    *    <jc>// Abstract superclass</jc>
320    *    <jk>public abstract class</jk> A {
321    *       <jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
322    *    }
323    *
324    *    <jc>// Subclass 1</jc>
325    *    <jk>public class</jk> A1 <jk>extends</jk> A {
326    *       <jk>public</jk> String <jf>f1</jf>;
327    *    }
328    *
329    *    <jc>// Subclass 2</jc>
330    *    <jk>public class</jk> A2 <jk>extends</jk> A {
331    *       <jk>public</jk> String <jf>f2</jf>;
332    *    }
333    *
334    *    <jc>// Filter for defining subtypes</jc>
335    *    <jk>public class</jk> AFilter <jk>extends</jk> BeanFilter&lt;A&gt; {
336    *       <jk>public</jk> AFilter() {
337    *          setSubTypeProperty(<js>"subType"</js>);
338    *          addSubType(Al.<jk>class</jk>, <js>"A1"</js>);
339    *          addSubType(A2.<jk>class</jk>, <js>"A2"</js>);
340    *       }
341    *    }
342    * </p>
343    * <p>
344    *    The following shows what happens when serializing a subclassed object to JSON:
345    * <p class='bcode'>
346    *    JsonSerializer s = <jk>new</jk> JsonSerializer().addBeanFilters(AFilter.<jk>class</jk>);
347    *    A1 a1 = <jk>new</jk> A1();
348    *    a1.<jf>f1</jf> = <js>"f1"</js>;
349    *    String r = s.serialize(a1);
350    *    <jsm>assertEquals</jsm>(<js>"{subType:'A1',f1:'f1',f0:'f0'}"</js>, r);
351    * </p>
352    * <p>
353    *    The following shows what happens when parsing back into the original object.
354    * <p class='bcode'>
355    *    JsonParser p = <jk>new</jk> JsonParser().addBeanFilters(AFilter.<jk>class</jk>);
356    *    A a = p.parse(r, A.<jk>class</jk>);
357    *    <jsm>assertTrue</jsm>(a <jk>instanceof</jk> A1);
358    * </p>
359    * <p>
360    *    This method is an alternative to using the {@link Bean#subTypeProperty()} annotation on a class.
361    *
362    * @param subTypeAttr The name of the attribute representing the subtype.
363    * @return This object (for method chaining).
364    */
365   public BeanFilter<T> setSubTypeProperty(String subTypeAttr) {
366      this.subTypeAttr = subTypeAttr;
367      return this;
368   }
369
370   /**
371    * Returns the subtypes associated with the bean class.
372    *
373    * @see #setSubTypeProperty(String)
374    * @return The set of sub types associated with this bean class, or <jk>null</jk> if bean has no subtypes defined.
375    */

377      return subTypes;

380   /**
381    * Specifies the set of subclasses of this bean class in addition to a string identifier for that subclass.
382    *
383    * @see #setSubTypeProperty(String)
384    * @param subTypes the map of subtype classes to subtype identifier strings.
385    * @return This object (for method chaining).
386    */
387   public BeanFilter<T> setSubTypes(LinkedHashMap<Class<?>, String> subTypes) {
388      this.subTypes = subTypes;
389      return this;
390   }
391
392   /**
393    * Convenience method for adding a single subtype in leu of using {@link #setSubTypes(LinkedHashMap)} in one call.
394    *
395    * @see #setSubTypeProperty(String)
396    * @param c The subtype class.
397    * @param id The subtype identifier string for the specified subtype class.
398    * @return This object (for method chaining).
399    */
400   public BeanFilter<T> addSubType(Class<?> c, String id) {
401      if (subTypes == null)
402         subTypes = new LinkedHashMap<Class<?>, String>();
403      subTypes.put(c, id);
404      return this;
405   }

417   /**
418    * Identifies a class to be used as the interface class for this and all subclasses.
419    * <p>
420    *    Functionally equivalent to using the {@link Bean#interfaceClass()} annotation.
421    * <p>
422    *    When specified, only the list of properties defined on the interface class will be used during serialization.
423    *    Additional properties on subclasses will be ignored.
424    * <p class='bcode'>
425    *    <jc>// Parent class</jc>
426    *    <jk>public abstract class</jk> A {
427    *       <jk>public</jk> String <jf>f0</jf> = <js>"f0"</js>;
428    *    }
429    *
430    *    <jc>// Sub class</jc>
431    *    <jk>public class</jk> A1 <jk>extends</jk> A {
432    *       <jk>public</jk> String <jf>f1</jf> = <js>"f1"</js>;
433    *    }
434    *
435    *    <jc>// Filter class</jc>
436    *    <jk>public class</jk> AFilter <jk>extends</jk> BeanFilter&lt;A&gt; {
437    *       <jk>public</jk> AFilter() {
438    *          setInterfaceClass(A.<jk>class</jk>);
439    *       }
440    *    }
441    *
442    *    JsonSerializer s = new JsonSerializer().addBeanFilters(AFilter.<jk>class</jk>);
443    *    A1 a1 = <jk>new</jk> A1();
444    *    String r = s.serialize(a1);
445    *    <jsm>assertEquals</jsm>(<js>"{f0:'f0'}"</js>, r);  <jc>// Note f1 is not serialized</jc>
446    * </p>
447    * <p>
448    *    Note that this filter can be used on the parent class so that it filters to all child classes,
449    *       or can be set individually on the child classes.
450    * <p>
451    *    This method is an alternative to using the {@link Bean#interfaceClass()}} annotation.
452    *
453    * @param interfaceClass The interface class.
454    * @return This object (for method chaining).
455    */
456   public BeanFilter<T> setInterfaceClass(Class<?> interfaceClass) {
457      this.interfaceClass = interfaceClass;
458      return this;
459   }

471   /**
472    * Identifies a stop class for this class and all subclasses.
473    * <p>
474    *    Functionally equivalent to using the {@link Bean#stopClass()} annotation.
475    * <p>
476    *    Identical in purpose to the stop class specified by {@link Introspector#getBeanInfo(Class, Class)}.
477    *    Any properties in the stop class or in its baseclasses will be ignored during analysis.
478    * <p>
479    *    For example, in the following class hierarchy, instances of <code>C3</code> will include property <code>p3</code>, but
480    *       not <code>p1</code> or <code>p2</code>.
481    * <p class='bcode'>
482    *    <jk>public class</jk> C1 {
483    *       <jk>public int</jk> getP1();
484    *    }
485    *
486    *    <jk>public class</jk> C2 <jk>extends</jk> C1 {
487    *       <jk>public int</jk> getP2();
488    *    }
489    *
490    *    <ja>@Bean</ja>(stopClass=C2.<jk>class</jk>)
491    *    <jk>public class</jk> C3 <jk>extends</jk> C2 {
492    *       <jk>public int</jk> getP3();
493    *    }
494    * </p>
495    *
496    * @param stopClass The stop class.
497    * @return This object (for method chaining).
498    */
499   public BeanFilter<T> setStopClass(Class<?> stopClass) {
500      this.stopClass = stopClass;
501      return this;
502   }
