013package org.apache.juneau.internal;
014
015import static org.apache.juneau.internal.CollectionUtils.*;
016
017import java.io.*;
018import java.lang.annotation.*;
019import java.util.*;
020
021/**
022 * Reflection utilities.
023 *
024 * @author James Bognar (james.bognar@salesforce.com)
025 */
026public final class ReflectionUtils {
027
028   /**
029    * Similar to {@link Class#getAnnotation(Class)} except also searches annotations on interfaces.
030    *
031    * @param <T> The annotation class type.
032    * @param a The annotation class.
033    * @param c The annotated class.
034    * @return The annotation, or <jk>null</jk> if not found.
035    */
036   public static <T extends Annotation> T getAnnotation(Class<T> a, Class<?> c) {
037      if (c == null)
038         return null;
039
040      T t = getDeclaredAnnotation(a, c);
041      if (t != null)
042         return t;
043
044      t = getAnnotation(a, c.getSuperclass());
045      if (t != null)
046         return t;
047
048      for (Class<?> c2 : c.getInterfaces()) {
049         t = getAnnotation(a, c2);
050         if (t != null)
051            return t;
052      }
053      return null;
054   }
055
056   /**
057    * Returns the specified annotation only if it's been declared on the specified class.
058    * <p>
059    *    More efficient than calling {@link Class#getAnnotation(Class)} since it doesn't
060    *    recursively look for the class up the parent chain.
061    *
062    * @param <T> The annotation class type.
063    * @param a The annotation class.
064    * @param c The annotated class.
065    * @return The annotation, or <jk>null</jk> if not found.
066    */
067   @SuppressWarnings("unchecked")
068   public static <T extends Annotation> T getDeclaredAnnotation(Class<T> a, Class<?> c) {
069      for (Annotation a2 : c.getDeclaredAnnotations())
070         if (a2.annotationType() == a)
071            return (T)a2;
072      return null;
073   }
074
075   /**
076    * Returns all instances of the specified annotation on the specified class.
077    * <p>
078    * Searches all superclasses and superinterfaces.
079    * <p>
080    * Results are ordered child-to-parent.
081    *
082    * @param <T> The annotation class type.
083    * @param a The annotation class type.
084    * @param c The class being searched.
085    * @return The found matches, or an empty array if annotation was not found.
086    */
087   public static <T extends Annotation> List<T> findAnnotations(Class<T> a, Class<?> c) {
088      List<T> l = new LinkedList<T>();
089      appendAnnotations(a, c, l);
090      return l;
091   }
092
093   /**
094    * Sames as {@link #findAnnotations(Class, Class)} except returns the annotations as a map
095    * with the keys being the class on which the annotation was found.
096    * <p>
097    * Results are ordered child-to-parent.
098    *
099    * @param <T> The annotation class type.
100    * @param a The annotation class type.
101    * @param c The class being searched.
102    * @return The found matches, or an empty array if annotation was not found.
103    */
104   public static <T extends Annotation> LinkedHashMap<Class<?>,T> findAnnotationsMap(Class<T> a, Class<?> c) {
105      LinkedHashMap<Class<?>,T> m = new LinkedHashMap<Class<?>,T>();
106      findAnnotationsMap(a, c, m);
107      return m;
108   }
109
110   private static <T extends Annotation> void findAnnotationsMap(Class<T> a, Class<?> c, Map<Class<?>,T> m) {
111      if (c == null)
112         return;
113
114      T t = getDeclaredAnnotation(a, c);
115      if (t != null)
116         m.put(c, t);
117
118      findAnnotationsMap(a, c.getSuperclass(), m);
119
120      for (Class<?> c2 : c.getInterfaces())
121         findAnnotationsMap(a, c2, m);
122   }
123
124   /**
125    * Finds and appends the specified annotation on the specified class and superclasses/interfaces to the specified list.
126    *
127    * @param a The annotation.
128    * @param c The class.
129    * @param l The list of annotations.
130    */
131   public static <T extends Annotation> void appendAnnotations(Class<T> a, Class<?> c, List<T> l) {
132      if (c == null)
133         return;
134
135      addIfNotNull(l, getDeclaredAnnotation(a, c));
136
137      if (c.getPackage() != null)
138         addIfNotNull(l, c.getPackage().getAnnotation(a));
139
140      appendAnnotations(a, c.getSuperclass(), l);
141
142      for (Class<?> c2 : c.getInterfaces())
143         appendAnnotations(a, c2, l);
144   }
145
146   /**
147    * Similar to {@link Class#getResourceAsStream(String)} except looks up the
148    * parent hierarchy for the existence of the specified resource.
149    *
150    * @param c The class to return the resource on.
151    * @param name The resource name.
152    * @return An input stream on the specified resource, or <jk>null</jk> if the resource could not be found.
153    */
154   public static InputStream getResource(Class<?> c, String name) {
155      while (c != null) {
156         InputStream is = c.getResourceAsStream(name);
157         if (is != null)
158            return is;
159         c = c.getSuperclass();
160      }
161      return null;
162   }
163}
