013package org.apache.juneau.internal;
014
015import java.util.*;
016
017/**
018 * Utility methods for collections.
019 *
020 * @author James Bognar (james.bognar@salesforce.com)
021 */
022public class CollectionUtils {
023
024   /**
025    * Reverses the order of a {@link LinkedHashMap}.
026    *
027    * @param in The map to reverse the order on.
028    * @return A new {@link LinkedHashMap} with keys in reverse order.
029    */
030   public static <K,V> LinkedHashMap<K,V> reverse(LinkedHashMap<K,V> in) {
031      if (in == null)
032         return null;
033      LinkedHashMap<K,V> m = new LinkedHashMap<K,V>();
034
035      // Note:  Entry objects are reusable in an entry set, so we simply can't
036      // create a reversed iteration of that set.
037      List<K> keys = new ArrayList<K>(in.keySet());
038      List<V> values = new ArrayList<V>(in.values());
039      for (int i = in.size()-1; i >= 0; i--)
040         m.put(keys.get(i), values.get(i));
041
042      return m;
043   }
044
045   /**
046    * Add a value to a list if the value is not null.
047    *
048    * @param l The list to add to.
049    * @param o The element to add.
050    * @return The same list.
051    */
052   public static <T> List<T> addIfNotNull(List<T> l, T o) {
053      if (o != null)
054         l.add(o);
055      return l;
056   }
057}
