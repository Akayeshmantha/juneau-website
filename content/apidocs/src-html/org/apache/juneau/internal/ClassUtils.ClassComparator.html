031    * @param o The objects.
032    * @return An array of readable class type strings.

034   public static ObjectList getReadableClassNames(Object[] o) {
035      ObjectList l = new ObjectList();
036      for (int i = 0; i < o.length; i++)
037         l.add(o[i] == null ? "null" : getReadableClassName(o[i].getClass()));
038      return l;

065   /**
066    * Converts the specified class name to a readable form when class name is a special construct like <js>"[[Z"</js>.
067    * <p>
068    * Examples:
069    * <p class='bcode'>
070    *    <jsm>getReadableClassName</jsm>(<js>"java.lang.Object"</js>);  <jc>// Returns "java.lang.Object"</jc>
071    *    <jsm>getReadableClassName</jsm>(<js>"boolean"</js>);  <jc>// Returns "boolean"</jc>
072    *    <jsm>getReadableClassName</jsm>(<js>"[Z"</js>);  <jc>// Returns "boolean[]"</jc>
073    *    <jsm>getReadableClassName</jsm>(<js>"[[Z"</js>);  <jc>// Returns "boolean[][]"</jc>
074    *    <jsm>getReadableClassName</jsm>(<js>"[Ljava.lang.Object;"</js>);  <jc>// Returns "java.lang.Object[]"</jc>
075    *    <jsm>getReadableClassName</jsm>(<jk>null</jk>);  <jc>// Returns null</jc>
076    * </p>
077    *
078    * @param className The class name.
079    * @return A readable class type name, or <jk>null</jk> if parameter is <jk>null</jk>.
080    */
081   public static String getReadableClassName(String className) {
082      if (className == null)
083         return null;
084      if (! StringUtils.startsWith(className, '['))
085         return className;
086      int depth = 0;
087      for (int i = 0; i < className.length(); i++) {
088         if (className.charAt(i) == '[')
089            depth++;
090         else
091            break;
092      }
093      char type = className.charAt(depth);
094      String c;
095      switch (type) {
096         case 'Z': c = "boolean"; break;
097         case 'B': c = "byte"; break;
098         case 'C': c = "char"; break;
099         case 'D': c = "double"; break;
100         case 'F': c = "float"; break;
101         case 'I': c = "int"; break;
102         case 'J': c = "long"; break;
103         case 'S': c = "short"; break;
104         default: c = className.substring(depth+1, className.length()-1);
105      }
106      StringBuilder sb = new StringBuilder(c.length() + 2*depth).append(c);
107      for (int i = 0; i < depth; i++)
108         sb.append("[]");
109      return sb.toString();
110   }

112   /**
113    * Converts the string generated by {@link #getReadableClassName(Class)} back into a {@link Class}.
114    * <p>
115    * Generics are stripped from the string since they cannot be converted to a class.
116    *
117    * @param cl The classloader to use to load the class.
118    * @param name The readable class name.
119    * @return The class object.
120    * @throws ClassNotFoundException
121    */
122   public static Class<?> getClassFromReadableName(ClassLoader cl, String name) throws ClassNotFoundException {
123      return cl.loadClass(name);
124   }

126   /**
127    * Returns <jk>true</jk> if <code>parent</code> is a parent class of <code>child</code>.
128    *
129    * @param parent The parent class.
130    * @param child The child class.
131    * @param strict If <jk>true</jk> returns <jk>false</jk> if the classes are the same.
132    * @return <jk>true</jk> if <code>parent</code> is a parent class of <code>child</code>.
133    */
134   public static boolean isParentClass(Class<?> parent, Class<?> child, boolean strict) {
135      return parent.isAssignableFrom(child) && ((!strict) || ! parent.equals(child));
136   }

164   /**
165    * Returns the signature of the specified method.
166    * For no-arg methods, the signature will be a simple string such as <js>"toString"</js>.
167    * For methods with one or more args, the arguments will be fully-qualified class names (e.g. <js>"append(java.util.StringBuilder,boolean)"</js>)
168    *
169    * @param m The methods to get the signature on.
170    * @return The methods signature.
171    */
172   public static String getMethodSignature(Method m) {
173      StringBuilder sb = new StringBuilder(m.getName());
174      Class<?>[] pt = m.getParameterTypes();
175      if (pt.length > 0) {
176         sb.append('(');
177         for (int i = 0; i < pt.length; i++) {
178            if (i > 0)
179               sb.append(',');
180            sb.append(getReadableClassName(pt[i]));
181         }
182         sb.append(')');
183      }
184      return sb.toString();
185   }
