013package org.apache.juneau.internal;
014
015import static org.apache.juneau.internal.ThrowableUtils.*;
016
017/**
018 * Stores a set of language keywords for quick lookup.
019 * <p>
020 * Keywords must be:
021 * <ul class='spaced-list'>
022 *    <li>2 or more characters in length.
023 *    <li>Lowercase ASCII.
024 * </ul>
025 *
026 * @author James Bognar (james.bognar@salesforce.com)
027 */
028public final class KeywordSet {
029   final char[][][][] store;
030
031   /**
032    * Constructor.
033    *
034    * @param keywords The list of keywords.
035    */
036   public KeywordSet(String... keywords) {
037      this.store = new char[26][][][];
038
039      for (String keyword : keywords) {
040         if (keyword.length() < 2)
041            illegalArg("Invalid keyword '{0}' passed to KeywordStore.", keyword);
042         int c0 = keyword.charAt(0) - 'a';
043         int c1 = keyword.charAt(1) - 'a';
044         if (c0 < 0 || c0 > 25 || c1 < 0 || c1 > 25)
045            illegalArg("Invalid keyword '{0}' passed to KeywordStore.", keyword);
046         if (this.store[c0] == null)
047            this.store[c0] = new char[26][][];
048         char[][][] x1 = this.store[c0];
049         char[][] x2;
050         if (x1[c1] == null)
051            x2 = new char[1][];
052         else {
053            x2 = new char[x1[c1].length+1][];
054            System.arraycopy(x1[c1], 0, x2, 0, x1[c1].length);
055         }
056         x2[x2.length-1] = keyword.toCharArray();
057         x1[c1] = x2;
058      }
059   }
060
061   /**
062    * Returns <jk>true</jk> if the specified string exists in this store.
063    *
064    * @param s The string to check.
065    * @return <jk>true</jk> if the specified string exists in this store.
066    */
067   public boolean contains(String s) {
068      if (s == null || s.length() < 2)
069         return false;
070      int c0 = s.charAt(0) - 'a', c1 = s.charAt(1) - 'a';
071      if (c0 < 0 || c0 > 25 || c1 < 0 || c1 > 25)
072         return false;
073      char[][][] x1 = store[c0];
074      if (x1 == null)
075         return false;
076      char[][] x2 = x1[c1];
077      if (x2 == null)
078         return false;
079      for (int i = 0; i < x2.length; i++) {
080         char[] keyword = x2[i];
081         if (keyword.length == s.length()) {
082            for (int j = 0; j < keyword.length; j++)
083               if (keyword[j] != s.charAt(j))
084                  return false;
085            return true;
086         }
087      }
088      return false;
089   }
090}
