013package org.apache.juneau.dto.cognos;
014
015import java.util.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.annotation.*;
019import org.apache.juneau.xml.annotation.*;
020
021/**
022 * Represents a Cognos dataset.
023 * <p>
024 *    When serialized to XML, creates the following construct (example pulled from <code>AddressBookResource</code>):
025 * <p class='bcode'>
026 *    <xt>&lt;?xml</xt> <xa>version</xa>=<xs>'1.0'</xs> <xa>encoding</xa>=<xs>'UTF-8'</xs><xt>?&gt;</xt>
027 *    <xt>&lt;c:dataset <xa>xmlns:c</xa>=<xs>'http://developer.cognos.com/schemas/xmldata/1/'</xs>&gt;</xt>
028 *       <xt>&lt;c:metadata&gt;</xt>
029 *          <xt>&lt;c:item</xt> <xa>name</xa>=<xs>'name'</xs> <xa>type</xa>=<xs>'xs:String'</xs> <xa>length</xa>=<xs>'255'</xs><xt>/&gt;</xt>
030 *          <xt>&lt;c:item</xt> <xa>name</xa>=<xs>'age'</xs> <xa>type</xa>=<xs>'xs:int'</xs><xt>/&gt;</xt>
031 *          <xt>&lt;c:item</xt> <xa>name</xa>=<xs>'numAddresses'</xs> <xa>type</xa>=<xs>'xs:int'</xs><xt>/&gt;</xt>
032 *       <xt>&lt;/c:metadata&gt;</xt>
033 *       <xt>&lt;c:data&gt;</xt>
034 *          <xt>&lt;c:row&gt;</xt>
035 *             <xt>&lt;c:value&gt;</xt>Barack Obama<xt>&lt;/c:value&gt;</xt>
036 *             <xt>&lt;c:value&gt;</xt>52<xt>&lt;/c:value&gt;</xt>
037 *             <xt>&lt;c:value&gt;</xt>2<xt>&lt;/c:value&gt;</xt>
038 *          <xt>&lt;/c:row&gt;</xt>
039 *          <xt>&lt;c:row&gt;</xt>
040 *             <xt>&lt;c:value&gt;</xt>George Walker Bush<xt>&lt;/c:value&gt;</xt>
041 *             <xt>&lt;c:value&gt;</xt>67<xt>&lt;/c:value&gt;</xt>
042 *             <xt>&lt;c:value&gt;</xt>2<xt>&lt;/c:value&gt;</xt>
043 *          <xt>&lt;/c:row&gt;</xt>
044 *       <xt>&lt;/c:data&gt;</xt>
045 *    <xt>&lt;/c:dataset&gt;</xt>
046 * </p>
047 * <p>
048 *    Only 2-dimentional POJOs (arrays or collections of maps or beans) can be serialized to Cognos.
049 *
050 * <h6 class='topic'>Example</h6>
051 * <p>
052 *    The construct shown above is a serialized <code>AddressBook</code> object which is a subclass of <code>LinkedList&lt;Person&gt;</code>.
053 *    The code for generating the XML is as follows...
054 * </p>
055 * <p class='bcode'>
056 *    Column[] items = {
057 *       <jk>new</jk> Column(<js>"name"</js>, <js>"xs:String"</js>, 255),
058 *       <jk>new</jk> Column(<js>"age"</js>, <js>"xs:int"</js>),
059 *       <jk>new</jk> Column(<js>"numAddresses"</js>, <js>"xs:int"</js>)
060 *          .addPojoSwap(
061 *             <jk>new</jk> PojoSwap&ltPerson,Integer&gt;() {
062 *                <ja>@Override</ja>
063 *                <jk>public</jk> Integer swap(Person p) {
064 *                   <jk>return</jk> p.<jf>addresses</jf>.size();
065 *                }
066 *             }
067 *          )
068 *    };
069 *
070 *    DataSet ds = <jk>new</jk> DataSet(items, <jsf>addressBook</jsf>, BeanContext.<jsf>DEFAULT</jsf>);
071 *
072 *    String xml = XmlSerializer.<jsf>DEFAULT_SQ</jsf>.serialize(ds);
073 * </p>
074 *
075 * @author James Bognar (james.bognar@salesforce.com)
076 */
077@Xml(name="dataset")
078@SuppressWarnings("unchecked")
079@Bean(properties={"metadata","data"})
080public class DataSet {
081
082   private Column[] metaData;
083   private List<Row> data;
084
085   /** Bean constructor. */
086   public DataSet() {}
087
088   /**
089    * Constructor.
090    *
091    * @param columns The meta-data that represents the columns in the dataset.
092    * @param o The POJO being serialized to Cognos.
093    *    Must be an array/collection of beans/maps.
094    * @param beanContext The bean context used to convert POJOs to strings.
095    * @throws Exception An error occurred trying to serialize the POJO.
096    */
097   public DataSet(Column[] columns, Object o, BeanContext beanContext) throws Exception {
098      metaData = columns;
099      data = new LinkedList<Row>();
100      if (o != null) {
101         if (o.getClass().isArray())
102            o = Arrays.asList((Object[])o);
103         if (o instanceof Collection) {
104            Collection<?> c = (Collection<?>)o;
105            for (Object o2 : c) {
106               Row r = new Row();
107               Map<?,?> m = null;
108               if (o2 instanceof Map)
109                  m = (Map<?,?>)o2;
110               else
111                  m = beanContext.forBean(o2);
112               for (Column col : columns) {
113                  Object v;
114                  if (col.transform != null)
115                     v = col.transform.swap(o2, beanContext);
116                  else
117                     v = m.get(col.getName());
118                  r.add(v == null ? null : v.toString());
119               }
120               data.add(r);
121            }
122         }
123      }
124   }
125
126   /**
127    * Represents a row of data.
128    * <p>
129    *    When serialized to XML, creates the following construct (example pulled from <code>AddressBookResource</code>):
130    * <p class='bcode'>
131    *    <xt>&lt;row&gt;</xt>
132    *       <xt>&lt;value&gt;</xt>Barack Obama<xt>&lt;/value&gt;</xt>
133    *       <xt>&lt;value&gt;</xt>52<xt>&lt;/value&gt;</xt>
134    *       <xt>&lt;value&gt;</xt>2<xt>&lt;/value&gt;</xt>
135    *    <xt>&lt;/row&gt;</xt>
136    * </p>
137    *
138    * @author James Bognar (james.bognar@salesforce.com)
139    */
140   @Xml(name="row", childName="value")
141   public static class Row extends LinkedList<String> {
142      private static final long serialVersionUID = 1L;
143   }
144
145
146   //--------------------------------------------------------------------------------
147   // Bean properties
148   //--------------------------------------------------------------------------------
149
150   /**
151    * Bean property getter:  <property>metadata</property>.
152    *
153    * @return The value of the <property>metadata</property> property on this bean, or <jk>null</jk> if it is not set.
154    */
155   @BeanProperty(name="metadata")
156   public Column[] getMetaData() {
157      return metaData;
158   }
159
160   /**
161    * Bean property setter:  <property>metadata</property>.
162    *
163    * @param metaData The new value for the <property>metadata</property> property on this bean.
164    * @return This object (for method chaining).
165    */
166   @BeanProperty(name="metadata")
167   public DataSet setMetaData(Column[] metaData) {
168      this.metaData = metaData;
169      return this;
170   }
171
172   /**
173    * Bean property getter:  <property>data</property>.
174    *
175    * @return The value of the <property>data</property> property on this bean, or <jk>null</jk> if it is not set.
176    */
177   @BeanProperty(name="data")
178   public List<Row> getData() {
179      return data;
180   }
181
182   /**
183    * Bean property setter:  <property>data</property>.
184    *
185    * @param data The new value for the <property>data</property> property on this bean.
186    * @return This object (for method chaining).
187    */
188   @BeanProperty(name="data")
189   public DataSet setData(List<Row> data) {
190      this.data = data;
191      return this;
192   }
193}
