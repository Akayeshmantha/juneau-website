026/**
027 * Parses URL-encoded text into POJO models.
028 *
029 *
030 * <h6 class='topic'>Media types</h6>
031 * <p>
032 *    Handles <code>Content-Type</code> types: <code>application/x-www-form-urlencoded</code>
033 *
034 *
035 * <h6 class='topic'>Description</h6>
036 * <p>
037 *    Parses URL-Encoded text (e.g. <js>"foo=bar&baz=bing"</js>) into POJOs.
038 * <p>
039 *    Expects parameter values to be in UON notation.
040 * <p>
041 *    This parser uses a state machine, which makes it very fast and efficient.
042 *
043 *
044 * <h6 class='topic'>Configurable properties</h6>
045 * <p>
046 *    This class has the following properties associated with it:
047 * <ul>
048 *    <li>{@link UonParserContext}
049 *    <li>{@link BeanContext}
050 * </ul>
051 *
052 *
053 * @author James Bognar (james.bognar@salesforce.com)
054 */
055@SuppressWarnings({ "rawtypes", "unchecked", "hiding" })
056@Consumes("application/x-www-form-urlencoded")
057public class UrlEncodingParser extends UonParser {
058

072   private <T> T parseAnything(UrlEncodingParserSession session, ClassMeta<T> eType, ParserReader r, Object outer) throws Exception {
073
074      BeanContext bc = session.getBeanContext();
075      if (eType == null)
076         eType = (ClassMeta<T>)object();
077      PojoSwap<T,Object> transform = (PojoSwap<T,Object>)eType.getPojoSwap();
078      ClassMeta<?> sType = eType.getSerializedClassMeta();
079      BeanDictionary bd = bc.getBeanDictionary();
080
081      int c = r.peek();
082      if (c == '?')
083         r.read();
084
085      Object o;
086
087      if (sType.isObject()) {
088         ObjectMap m = new ObjectMap(bc);
089         parseIntoMap(session, r, m, bc.string(), bc.object());
090         if (m.containsKey("_value"))
091            o = m.get("_value");
092         else
093            o = bd.cast(m);
094      } else if (sType.isMap()) {
095         Map m = (sType.canCreateNewInstance() ? (Map)sType.newInstance() : new ObjectMap(bc));
096         o = parseIntoMap(session, r, m, sType.getKeyType(), sType.getValueType());
097      } else if (sType.canCreateNewInstanceFromObjectMap(outer)) {
098         ObjectMap m = new ObjectMap(bc);
099         parseIntoMap(session, r, m, string(), object());
100         o = sType.newInstanceFromObjectMap(outer, m);
101      } else if (sType.canCreateNewBean(outer)) {
102         BeanMap m = bc.newBeanMap(outer, sType.getInnerClass());
103         m = parseIntoBeanMap(session, r, m);
104         o = m == null ? null : m.getBean();
105      } else {
106         // It could be a non-bean with _type attribute.
107         ObjectMap m = new ObjectMap(bc);
108         ClassMeta<Object> valueType = object();
109         parseIntoMap(session, r, m, string(), valueType);
110         if (m.containsKey(bc.getBeanTypePropertyName()))
111            o = bd.cast(m);
112         else if (m.containsKey("_value"))
113            o = session.getBeanContext().convertToType(m.get("_value"), sType);
114         else if (sType.isCollection()) {
115            // ?1=foo&2=bar...
116            Collection c2 = sType.canCreateNewInstance() ? (Collection)sType.newInstance() : new ObjectList(bc);
117            Map<Integer,Object> t = new TreeMap<Integer,Object>();
118            for (Map.Entry<String,Object> e : m.entrySet()) {
119               String k = e.getKey();
120               if (StringUtils.isNumeric(k))
121                  t.put(Integer.valueOf(k), bc.convertToType(e.getValue(), sType.getElementType()));
122            }
123            c2.addAll(t.values());
124            o = c2;
125         } else {
126            if (sType.getNotABeanReason() != null)
127               throw new ParseException(session, "Class ''{0}'' could not be instantiated as application/x-www-form-urlencoded.  Reason: ''{1}''", sType, sType.getNotABeanReason());
128            throw new ParseException(session, "Malformed application/x-www-form-urlencoded input for class ''{0}''.", sType);
129         }
130      }
131
132      if (transform != null && o != null)
133         o = transform.unswap(o, eType, bc);
134
135      if (outer != null)
136         setParent(eType, o, outer);
137
138      return (T)o;
139   }

141   private <K,V> Map<K,V> parseIntoMap(UonParserSession session, ParserReader r, Map<K,V> m, ClassMeta<K> keyType, ClassMeta<V> valueType) throws Exception {
142
143      if (keyType == null)
144         keyType = (ClassMeta<K>)string();
145
146      int c = r.peek();
147      if (c == -1)
148         return m;
149
150      final int S1=1; // Looking for attrName start.
151      final int S2=2; // Found attrName end, looking for =.
152      final int S3=3; // Found =, looking for valStart.
153      final int S4=4; // Looking for & or end.
154      boolean isInEscape = false;
155
156      int state = S1;
157      K currAttr = null;
158      while (c != -1) {
159         c = r.read();
160         if (! isInEscape) {
161            if (state == S1) {
162               if (c == -1)
163                  return m;
164               r.unread();
165               Object attr = parseAttr(session, r, true);
166               currAttr = session.trim(session.getBeanContext().convertToType(attr, keyType));
167               state = S2;
168               c = 0; // Avoid isInEscape if c was '\'
169            } else if (state == S2) {
170               if (c == '\u0002')
171                  state = S3;
172               else if (c == -1 || c == '\u0001') {
173                  m.put(currAttr, null);
174                  if (c == -1)
175                     return m;
176                  state = S1;
177               }
178            } else if (state == S3) {
179               if (c == -1 || c == '\u0001') {
180                  V value = convertAttrToType(session, m, "", valueType);
181                  m.put(currAttr, value);
182                  if (c == -1)
183                     return m;
184                  state = S1;
185               } else  {
186                  // For performance, we bypass parseAnything for string values.
187                  V value = (V)(valueType.isString() ? super.parseString(session, r.unread(), true) : super.parseAnything(session, valueType, r.unread(), m, true, null));
188
189                  // If we already encountered this parameter, turn it into a list.
190                  if (m.containsKey(currAttr) && valueType.isObject()) {
191                     Object v2 = m.get(currAttr);
192                     if (! (v2 instanceof ObjectList)) {
193                        v2 = new ObjectList(v2);
194                        m.put(currAttr, (V)v2);
195                     }
196                     ((ObjectList)v2).add(value);
197                  } else {
198                     m.put(currAttr, value);
199                  }
200                  state = S4;
201                  c = 0; // Avoid isInEscape if c was '\'
202               }
203            } else if (state == S4) {
204               if (c == '\u0001')
205                  state = S1;
206               else if (c == -1) {
207                  return m;
208               }
209            }
210         }
211         isInEscape = (c == '\\' && ! isInEscape);
212      }
213      if (state == S1)
214         throw new ParseException(session, "Could not find attribute name on object.");
215      if (state == S2)
216         throw new ParseException(session, "Could not find '=' following attribute name on object.");
217      if (state == S3)
218         throw new ParseException(session, "Dangling '=' found in object entry");
219      if (state == S4)
220         throw new ParseException(session, "Could not find end of object.");
221
222      return null; // Unreachable.
223   }

225   private <T> BeanMap<T> parseIntoBeanMap(UrlEncodingParserSession session, ParserReader r, BeanMap<T> m) throws Exception {
226
227      BeanContext bc = session.getBeanContext();
228
229      int c = r.peek();
230      if (c == -1)
231         return m;
232
233      final int S1=1; // Looking for attrName start.
234      final int S2=2; // Found attrName end, looking for =.
235      final int S3=3; // Found =, looking for valStart.
236      final int S4=4; // Looking for , or }
237      boolean isInEscape = false;
238
239      int state = S1;
240      String currAttr = "";
241      int currAttrLine = -1, currAttrCol = -1;
242      while (c != -1) {
243         c = r.read();
244         if (! isInEscape) {
245            if (state == S1) {
246               if (c == -1) {
247                  return m;
248               }
249               r.unread();
250               currAttrLine= r.getLine();
251               currAttrCol = r.getColumn();
252               currAttr = parseAttrName(session, r, true);
253               if (currAttr == null)  // Value was '%00'
254                  return null;
255               state = S2;
256            } else if (state == S2) {
257               if (c == '\u0002')
258                  state = S3;
259               else if (c == -1 || c == '\u0001') {
260                  m.put(currAttr, null);
261                  if (c == -1)
262                     return m;
263                  state = S1;
264               }
265            } else if (state == S3) {
266               if (c == -1 || c == '\u0001') {
267                  if (! currAttr.equals(bc.getBeanTypePropertyName())) {
268                     BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
269                     if (pMeta == null) {
270                        if (m.getMeta().isSubTyped()) {
271                           m.put(currAttr, "");
272                        } else {
273                           onUnknownProperty(session, currAttr, m, currAttrLine, currAttrCol);
274                        }
275                     } else {
276                        session.setCurrentProperty(pMeta);
277                        // In cases of "&foo=", create an empty instance of the value if createable.
278                        // Otherwise, leave it null.
279                        ClassMeta<?> cm = pMeta.getClassMeta();
280                        if (cm.canCreateNewInstance())
281                           pMeta.set(m, cm.newInstance());
282                        session.setCurrentProperty(null);
283                     }
284                  }
285                  if (c == -1)
286                     return m;
287                  state = S1;
288               } else {
289                  if (! currAttr.equals(bc.getBeanTypePropertyName())) {
290                     BeanPropertyMeta pMeta = m.getPropertyMeta(currAttr);
291                     if (pMeta == null) {
292                        if (m.getMeta().isSubTyped()) {
293                           Object value = parseAnything(session, object(), r.unread(), m.getBean(false), true, null);
294                           m.put(currAttr, value);
295                        } else {
296                           onUnknownProperty(session, currAttr, m, currAttrLine, currAttrCol);
297                           parseAnything(session, object(), r.unread(), m.getBean(false), true, null); // Read content anyway to ignore it
298                        }
299                     } else {
300                        session.setCurrentProperty(pMeta);
301                        if (session.shouldUseExpandedParams(pMeta)) {
302                           ClassMeta et = pMeta.getClassMeta().getElementType();
303                           Object value = parseAnything(session, et, r.unread(), m.getBean(false), true, pMeta);
304                           setName(et, value, currAttr);
305                           pMeta.add(m, value);
306                        } else {
307                           ClassMeta<?> cm = pMeta.getClassMeta();
308                           Object value = parseAnything(session, cm, r.unread(), m.getBean(false), true, pMeta);
309                           setName(cm, value, currAttr);
310                           pMeta.set(m, value);
311                        }
312                        session.setCurrentProperty(null);
313                     }
314                  }
315                  state = S4;
316               }
317            } else if (state == S4) {
318               if (c == '\u0001')
319                  state = S1;
320               else if (c == -1) {
321                  return m;
322               }
323            }
324         }
325         isInEscape = (c == '\\' && ! isInEscape);
326      }
327      if (state == S1)
328         throw new ParseException(session, "Could not find attribute name on object.");
329      if (state == S2)
330         throw new ParseException(session, "Could not find '=' following attribute name on object.");
331      if (state == S3)
332         throw new ParseException(session, "Could not find value following '=' on object.");
333      if (state == S4)
334         throw new ParseException(session, "Could not find end of object.");
335
336      return null; // Unreachable.
337   }

339   /**
340    * Parse a URL query string into a simple map of key/value pairs.
341    *
342    * @param qs The query string to parse.
343    * @return A sorted {@link TreeMap} of query string entries.
344    * @throws Exception
345    */
346   public Map<String,String[]> parseIntoSimpleMap(String qs) throws Exception {
347
348      Map<String,String[]> m = new TreeMap<String,String[]>();
349
350      if (StringUtils.isEmpty(qs))
351         return m;
352
353      UonReader r = new UonReader(qs, true);
354
355      final int S1=1; // Looking for attrName start.
356      final int S2=2; // Found attrName start, looking for = or & or end.
357      final int S3=3; // Found =, looking for valStart.
358      final int S4=4; // Found valStart, looking for & or end.
359
360      try {
361         int c = r.peek();
362         if (c == '?')
363            r.read();
364
365         int state = S1;
366         String currAttr = null;
367         while (c != -1) {
368            c = r.read();
369            if (state == S1) {
370               if (c != -1) {
371                  r.unread();
372                  r.mark();
373                  state = S2;
374               }
375            } else if (state == S2) {
376               if (c == -1) {
377                  add(m, r.getMarked(), null);
378               } else if (c == '\u0001') {
379                  m.put(r.getMarked(0,-1), null);
380                  state = S1;
381               } else if (c == '\u0002') {
382                  currAttr = r.getMarked(0,-1);
383                  state = S3;
384               }
385            } else if (state == S3) {
386               if (c == -1 || c == '\u0001') {
387                  add(m, currAttr, "");
388               } else {
389                  if (c == '\u0002')
390                     r.replace('=');
391                  r.unread();
392                  r.mark();
393                  state = S4;
394               }
395            } else if (state == S4) {
396               if (c == -1) {
397                  add(m, currAttr, r.getMarked());
398               } else if (c == '\u0001') {
399                  add(m, currAttr, r.getMarked(0,-1));
400                  state = S1;
401               } else if (c == '\u0002') {
402                  r.replace('=');
403               }
404            }
405         }
406      } finally {
407         r.close();
408      }
409
410      return m;
411   }

413   private static void add(Map<String,String[]> m, String key, String val) {
414      boolean b = m.containsKey(key);
415      if (val == null) {
416         if (! b)
417            m.put(key, null);
418      } else if (b && m.get(key) != null) {
419         m.put(key, ArrayUtils.append(m.get(key), val));
420      } else {
421         m.put(key, new String[]{val});
422      }
423   }

425   private Object[] parseArgs(UrlEncodingParserSession session, ParserReader r, ClassMeta<?>[] argTypes) throws Exception {
426      // TODO - This can be made more efficient.
427      BeanContext bc = session.getBeanContext();
428      ClassMeta<TreeMap<Integer,String>> cm = bc.getMapClassMeta(TreeMap.class, Integer.class, String.class);
429      TreeMap<Integer,String> m = parseAnything(session, cm, r, session.getOuter());
430      Object[] vals = m.values().toArray(new Object[m.size()]);
431      if (vals.length != argTypes.length)
432         throw new ParseException(session, "Argument lengths don't match.  vals={0}, argTypes={1}", vals.length, argTypes.length);
433      for (int i = 0; i < vals.length; i++) {
434         String s = String.valueOf(vals[i]);
435         vals[i] = super.parseAnything(session, argTypes[i], new UonReader(s, false), session.getOuter(), true, null);
436      }
437
438      return vals;
439   }

441   /**
442    * Parses a single query parameter value into the specified class type.
443    *
444    * @param in The input query string value.
445    * @param type The class type of the object to create.
446    * @return A new instance of the specified type.
447    * @throws ParseException
448    */
449   public <T> T parseParameter(CharSequence in, ClassMeta<T> type) throws ParseException {
450      if (in == null)
451         return null;
452      UonParserSession session = createParameterContext(in);
453      try {
454         UonReader r = session.getReader();
455         return super.parseAnything(session, type, r, null, true, null);
456      } catch (ParseException e) {
457         throw e;
458      } catch (Exception e) {
459         throw new ParseException(session, e);
460      } finally {
461         session.close();
462      }
463   }

466    * Parses a single query parameter value into the specified class type.
467    *
468    * @param in The input query string value.
469    * @param type The class type of the object to create.
470    * @return A new instance of the specified type.
471    * @throws ParseException
472    */
473   public <T> T parseParameter(CharSequence in, Class<T> type) throws ParseException {
474      return parseParameter(in, getBeanContext().getClassMeta(type));
475   }

486   @Override /* Parser */
487   protected <T> T doParse(ParserSession session, ClassMeta<T> type) throws Exception {
488      UrlEncodingParserSession s = (UrlEncodingParserSession)session;
489      type = s.getBeanContext().normalizeClassMeta(type);
490      UonReader r = s.getReader();
491      T o = parseAnything(s, type, r, s.getOuter());
492      return o;
493   }
494
495   @Override /* ReaderParser */
496   protected Object[] doParseArgs(ParserSession session, ClassMeta<?>[] argTypes) throws Exception {
497      UrlEncodingParserSession uctx = (UrlEncodingParserSession)session;
498      UonReader r = uctx.getReader();
499      Object[] a = parseArgs(uctx, r, argTypes);
500      return a;
501   }

503   @Override /* ReaderParser */
504   protected <K,V> Map<K,V> doParseIntoMap(ParserSession session, Map<K,V> m, Type keyType, Type valueType) throws Exception {
505      UrlEncodingParserSession s = (UrlEncodingParserSession)session;
506      UonReader r = s.getReader();
507      if (r.peek() == '?')
508         r.read();
509      m = parseIntoMap(s, r, m, s.getBeanContext().getClassMeta(keyType), s.getBeanContext().getClassMeta(valueType));
510      return m;
511   }
