013package org.apache.juneau.urlencoding;
014
015import org.apache.juneau.*;
016import org.apache.juneau.parser.*;
017
018/**
019 * Configurable properties on the {@link UonParser} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * The following convenience methods are also provided for setting context properties:
025 * <ul>
026 *    <li>{@link UonParser#setProperty(String,Object)}
027 *    <li>{@link UonParser#setProperties(ObjectMap)}
028 *    <li>{@link UonParser#addNotBeanClasses(Class[])}
029 *    <li>{@link UonParser#addBeanFilters(Class[])}
030 *    <li>{@link UonParser#addPojoSwaps(Class[])}
031 *    <li>{@link UonParser#addToDictionary(Class[])}
032 *    <li>{@link UonParser#addImplClass(Class,Class)}
033 * </ul>
034 * <p>
035 * See {@link ContextFactory} for more information about context properties.
036 *
037 *
038 * <h6 class='topic' id='ConfigProperties'>Configurable properties on the URL-Encoding and UON parsers</h6>
039 * <table class='styled' style='border-collapse: collapse;'>
040 *    <tr><th>Setting name</th><th>Description</th><th>Data type</th><th>Default value</th></tr>
041 *    <tr>
042 *       <td>{@link #UON_decodeChars}</td>
043 *       <td>Decode <js>"%xx"</js> sequences</td>
044 *       <td><code>Boolean</code></td>
045 *       <td><jk>false</jk> for {@link UonParser}<br><jk>true</jk> for {@link UrlEncodingParser}</td>
046 *    </tr>
047 *    <tr>
048 *       <td>{@link #UON_whitespaceAware}</td>
049 *       <td>Whitespace aware</td>
050 *       <td><code>Boolean</code></td>
051 *       <td><jk>false</jk></td>
052 *    </tr>
053 * </table>
054 *
055 * <h6 class='topic'>Configurable properties inherited from parent classes</h6>
056 * <ul class='javahierarchy'>
057 *    <li class='c'><a class='doclink' href='../BeanContext.html#ConfigProperties'>BeanContext</a> - Properties associated with handling beans on serializers and parsers.
058 *    <ul>
059 *       <li class='c'><a class='doclink' href='../parser/ParserContext.html#ConfigProperties'>ParserContext</a> - Configurable properties common to all parsers.
060 *    </ul>
061 * </ul>
062 *
063 *
064 * @author James Bognar (james.bognar@salesforce.com)
065 */
066public class UonParserContext extends ParserContext {
067
068   /**
069    * <b>Configuration property:</b> Decode <js>"%xx"</js> sequences.
070    * <p>
071    * <ul>
072    *    <li><b>Name:</b> <js>"UonParser.decodeChars"</js>
073    *    <li><b>Data type:</b> <code>Boolean</code>
074    *    <li><b>Default:</b> <jk>false</jk> for {@link UonParser}, <jk>true</jk> for {@link UrlEncodingParser}
075    * </ul>
076    * <p>
077    * Specify <jk>true</jk> if URI encoded characters should be decoded, <jk>false</jk>
078    *    if they've already been decoded before being passed to this parser.
079    */
080   public static final String UON_decodeChars = "UonParser.decodeChars";
081
082   /**
083    * <b>Configuration property:</b> Whitespace aware.
084    * <p>
085    * <ul>
086    *    <li><b>Name:</b> <js>"UonParser.whitespaceAware"</js>
087    *    <li><b>Data type:</b> <code>Boolean</code>
088    *    <li><b>Default:</b> <jk>false</jk>
089    * </ul>
090    * <p>
091    * Expect input to contain readable whitespace characters from using the {@link UonSerializerContext#UON_useWhitespace} setting.
092    */
093   public static final String UON_whitespaceAware = "UonParser.whitespaceAware";
094
095
096   final boolean
097      decodeChars,
098      whitespaceAware;
099
100   /**
101    * Constructor.
102    * <p>
103    * Typically only called from {@link ContextFactory#getContext(Class)}.
104    *
105    * @param cf The factory that created this context.
106    */
107   public UonParserContext(ContextFactory cf) {
108      super(cf);
109      this.decodeChars = cf.getProperty(UON_decodeChars, boolean.class, false);
110      this.whitespaceAware = cf.getProperty(UON_whitespaceAware, boolean.class, false);
111   }
112}
