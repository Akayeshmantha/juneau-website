013package org.apache.juneau.urlencoding;
014
015import org.apache.juneau.*;
016import org.apache.juneau.parser.*;
017
018/**
019 * Configurable properties on the {@link UonParser} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * The following convenience methods are also provided for setting context properties:
025 * <ul>
026 *    <li>{@link UonParser#setProperty(String,Object)}
027 *    <li>{@link UonParser#setProperties(ObjectMap)}
028 *    <li>{@link UonParser#addNotBeanClasses(Class[])}
029 *    <li>{@link UonParser#addBeanFilters(Class[])}
030 *    <li>{@link UonParser#addPojoSwaps(Class[])}
031 *    <li>{@link UonParser#addImplClass(Class,Class)}
032 * </ul>
033 * <p>
034 * See {@link ContextFactory} for more information about context properties.
035 *
036 * @author James Bognar (james.bognar@salesforce.com)
037 */
038public class UonParserContext extends ParserContext {
039
040   /**
041    * Decode <js>"%xx"</js> sequences. ({@link Boolean}, default=<jk>false</jk> for {@link UonParser}, <jk>true</jk> for {@link UrlEncodingParser}).
042    * <p>
043    * Specify <jk>true</jk> if URI encoded characters should be decoded, <jk>false</jk>
044    *    if they've already been decoded before being passed to this parser.
045    */
046   public static final String UON_decodeChars = "UonParser.decodeChars";
047
048   /**
049    * Expect input to contain readable whitespace characters from using the {@link UonSerializerContext#UON_useWhitespace} setting ({@link Boolean}, default=<jk>false</jk>).
050    */
051   public static final String UON_whitespaceAware = "UonParser.whitespaceAware";
052
053
054   final boolean
055      decodeChars,
056      whitespaceAware;
057
058   /**
059    * Constructor.
060    * <p>
061    * Typically only called from {@link ContextFactory#getContext(Class)}.
062    *
063    * @param cf The factory that created this context.
064    */
065   public UonParserContext(ContextFactory cf) {
066      super(cf);
067      this.decodeChars = cf.getProperty(UON_decodeChars, boolean.class, false);
068      this.whitespaceAware = cf.getProperty(UON_whitespaceAware, boolean.class, false);
069   }
070}
