013package org.apache.juneau.urlencoding;
014
015import org.apache.juneau.*;
016import org.apache.juneau.serializer.*;
017
018/**
019 * Configurable properties on the {@link UonSerializer} class.
020 * <p>
021 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
022 * returned {@link CoreApi#getContextFactory()}.
023 * <p>
024 * The following convenience methods are also provided for setting context properties:
025 * <ul>
026 *    <li>{@link UonSerializer#setProperty(String,Object)}
027 *    <li>{@link UonSerializer#setProperties(ObjectMap)}
028 *    <li>{@link UonSerializer#addNotBeanClasses(Class[])}
029 *    <li>{@link UonSerializer#addBeanFilters(Class[])}
030 *    <li>{@link UonSerializer#addPojoSwaps(Class[])}
031 *    <li>{@link UonSerializer#addImplClass(Class,Class)}
032 * </ul>
033 * <p>
034 * See {@link ContextFactory} for more information about context properties.
035 *
036 * @author James Bognar (james.bognar@salesforce.com)
037 */
038public class UonSerializerContext extends SerializerContext {
039
040   /**
041    * Use simplified output ({@link Boolean}, default=<jk>false</jk>).
042    * <p>
043    * If <jk>true</jk>, type flags will not be prepended to values in most cases.
044    * <p>
045    * Use this setting if the data types of the values (e.g. object/array/boolean/number/string)
046    *    is known on the receiving end.
047    * <p>
048    * It should be noted that the default behavior produces a data structure that can
049    *    be losslessly converted into JSON, and any JSON can be losslessly represented
050    *    in a URL-encoded value.  However, this strict equivalency does not exist
051    *    when simple mode is used.
052    * <p>
053    * <table class='styled'>
054    *    <tr>
055    *       <th>Input (in JSON)</th>
056    *       <th>Normal mode output</th>
057    *       <th>Simple mode output</th>
058    *    </tr>
059    *    <tr>
060    *       <td class='code'>{foo:'bar',baz:'bing'}</td>
061    *       <td class='code'>$o(foo=bar,baz=bing)</td>
062    *       <td class='code'>(foo=bar,baz=bing)</td>
063    *    </tr>
064    *    <tr>
065    *       <td class='code'>{foo:{bar:'baz'}}</td>
066    *       <td class='code'>$o(foo=$o(bar=baz))</td>
067    *       <td class='code'>(foo=(bar=baz))</td>
068    *    </tr>
069    *    <tr>
070    *       <td class='code'>['foo','bar']</td>
071    *       <td class='code'>$a(foo,bar)</td>
072    *       <td class='code'>(foo,bar)</td>
073    *    </tr>
074    *    <tr>
075    *       <td class='code'>['foo',['bar','baz']]</td>
076    *       <td class='code'>$a(foo,$a(bar,baz))</td>
077    *       <td class='code'>(foo,(bar,baz))</td>
078    *    </tr>
079    *    <tr>
080    *       <td class='code'>true</td>
081    *       <td class='code'>$b(true)</td>
082    *       <td class='code'>true</td>
083    *    </tr>
084    *    <tr>
085    *       <td class='code'>123</td>
086    *       <td class='code'>$n(123)</td>
087    *       <td class='code'>123</td>
088    *    </tr>
089    * </table>
090    */
091   public static final String UON_simpleMode = "UonSerializer.simpleMode";
092
093   /**
094    * Use whitespace in output ({@link Boolean}, default=<jk>false</jk>).
095    * <p>
096    * If <jk>true</jk>, whitespace is added to the output to improve readability.
097    */
098   public static final String UON_useWhitespace = "UonSerializer.useWhitespace";
099
100   /**
101    * Encode non-valid URI characters to <js>"%xx"</js> constructs. ({@link Boolean}, default=<jk>false</jk> for {@link UonSerializer}, <jk>true</jk> for {@link UrlEncodingSerializer}).
102    * <p>
103    * If <jk>true</jk>, non-valid URI characters will be converted to <js>"%xx"</js> sequences.
104    * Set to <jk>false</jk> if parameter value is being passed to some other code that will already
105    *    perform URL-encoding of non-valid URI characters.
106    */
107   public static final String UON_encodeChars = "UonSerializer.encodeChars";
108
109
110   final boolean
111      simpleMode,
112      useWhitespace,
113      encodeChars;
114
115   /**
116    * Constructor.
117    * <p>
118    * Typically only called from {@link ContextFactory#getContext(Class)}.
119    *
120    * @param cf The factory that created this context.
121    */
122   public UonSerializerContext(ContextFactory cf) {
123      super(cf);
124      simpleMode = cf.getProperty(UON_simpleMode, boolean.class, false);
125      useWhitespace = cf.getProperty(UON_useWhitespace, boolean.class, false);
126      encodeChars = cf.getProperty(UON_encodeChars, boolean.class, false);
127   }
128}
