033/**
034 * Contains metadata about a bean property.
035 * <p>
036 *    Contains information such as type of property (e.g. field/getter/setter), class type of property value,
037 *    and whether any transforms are associated with this property.
038 * <p>
039 *    Developers will typically not need access to this class.  The information provided by it is already
040 *    exposed through several methods on the {@link BeanMap} API.
041 *
042 * @author James Bognar (james.bognar@salesforce.com)
043 */

236   boolean validate() throws Exception {
237
238      BeanContext f = beanMeta.ctx;
239      Map<Class<?>,Class<?>[]> typeVarImpls = beanMeta.typeVarImpls;
240
241      if (field == null && getter == null)
242         return false;
243
244      if (field == null && setter == null && f.beansRequireSettersForGetters && ! isConstructorArg)
245         return false;
246
247      if (field != null) {
248         BeanProperty p = field.getAnnotation(BeanProperty.class);
249         rawTypeMeta = f.getClassMeta(p, field.getGenericType(), typeVarImpls);
250         isUri |= (rawTypeMeta.isUri() || field.isAnnotationPresent(org.apache.juneau.annotation.URI.class));
251         if (p != null) {
252            transform = getPropertyPojoSwap(p);
253            if (p.properties().length != 0)
254               properties = p.properties();
255            isBeanUri |= p.beanUri();
256         }
257      }
258
259      if (getter != null) {
260         BeanProperty p = getter.getAnnotation(BeanProperty.class);
261         if (rawTypeMeta == null)
262            rawTypeMeta = f.getClassMeta(p, getter.getGenericReturnType(), typeVarImpls);
263         isUri |= (rawTypeMeta.isUri() || getter.isAnnotationPresent(org.apache.juneau.annotation.URI.class));
264         if (p != null) {
265            if (transform == null)
266               transform = getPropertyPojoSwap(p);
267            if (properties != null && p.properties().length != 0)
268               properties = p.properties();
269            isBeanUri |= p.beanUri();
270         }
271      }
272
273      if (setter != null) {
274         BeanProperty p = setter.getAnnotation(BeanProperty.class);
275         if (rawTypeMeta == null)
276            rawTypeMeta = f.getClassMeta(p, setter.getGenericParameterTypes()[0], typeVarImpls);
277         isUri |= (rawTypeMeta.isUri() || setter.isAnnotationPresent(org.apache.juneau.annotation.URI.class));
278         if (p != null) {
279         if (transform == null)
280            transform = getPropertyPojoSwap(p);
281            if (properties != null && p.properties().length != 0)
282               properties = p.properties();
283            isBeanUri |= p.beanUri();
284         }
285      }
286
287      if (rawTypeMeta == null)
288         return false;
289
290      // Do some annotation validation.
291      Class<?> c = rawTypeMeta.getInnerClass();
292      if (getter != null && ! isParentClass(getter.getReturnType(), c))
293         return false;
294      if (setter != null && ! isParentClass(setter.getParameterTypes()[0], c))
295         return false;
296      if (field != null && ! isParentClass(field.getType(), c))
297         return false;
298
299      return true;
300   }
301
302   private PojoSwap getPropertyPojoSwap(BeanProperty p) throws Exception {
303      Class<?> c = p.swap();
304      if (c == Null.class)
305         return null;
306      try {
307         if (ClassUtils.isParentClass(PojoSwap.class, c)) {
308            return (PojoSwap)c.newInstance();
309         }
310         throw new RuntimeException("TODO - Surrogate swaps not yet supported.");
311      } catch (Exception e) {
312         throw new BeanRuntimeException(this.beanMeta.c, "Could not instantiate PojoSwap ''{0}'' for bean property ''{1}''", c.getName(), this.name).initCause(e);
313      }
314   }

316   /**
317    * Equivalent to calling {@link BeanMap#get(Object)}, but is faster since it avoids looking up the property meta.
318    *
319    * @param m The bean map to get the transformed value from.
320    * @return The property value.
321    */
322   public Object get(BeanMap<?> m) {
323      try {
324         // Read-only beans have their properties stored in a cache until getBean() is called.
325         Object bean = m.bean;
326         if (bean == null)
327            return m.propertyCache.get(name);
328
329         Object o = null;
330
331         if (getter == null && field == null)
332            throw new BeanRuntimeException(beanMeta.c, "Getter or public field not defined on property ''{0}''", name);
333
334         if (getter != null)
335            o = getter.invoke(bean, (Object[])null);
336
337         else if (field != null)
338            o = field.get(bean);
339
340         o = transform(o);
341         if (o == null)
342            return null;
343         if (properties != null) {
344            if (rawTypeMeta.isArray()) {
345               Object[] a = (Object[])o;
346               List l = new ArrayList(a.length);
347               ClassMeta childType = rawTypeMeta.getElementType();
348               for (Object c : a)
349                  l.add(applyChildPropertiesFilter(childType, c));
350               return l;
351            } else if (rawTypeMeta.isCollection()) {
352               Collection c = (Collection)o;
353               List l = new ArrayList(c.size());
354               ClassMeta childType = rawTypeMeta.getElementType();
355               for (Object cc : c)
356                  l.add(applyChildPropertiesFilter(childType, cc));
357               return l;
358            } else {
359               return applyChildPropertiesFilter(rawTypeMeta, o);
360            }
361         }
362         return o;
363      } catch (SerializeException e) {
364         throw new BeanRuntimeException(e);
365      } catch (Throwable e) {
366         if (beanMeta.ctx.ignoreInvocationExceptionsOnGetters) {
367            if (rawTypeMeta.isPrimitive())
368               return rawTypeMeta.getPrimitiveDefault();
369            return null;
370         }
371         throw new BeanRuntimeException(beanMeta.c, "Exception occurred while getting property ''{0}''", name).initCause(e);
372      }
373   }

375   /**
376    * Equivalent to calling {@link BeanMap#put(String, Object)}, but is faster since it avoids
377    *    looking up the property meta.
378    *
379    * @param m The bean map to set the property value on.
380    * @param value The value to set.
381    * @return The previous property value.
382    * @throws BeanRuntimeException If property could not be set.
383    */
384   public Object set(BeanMap<?> m, Object value) throws BeanRuntimeException {
385      try {
386         // Comvert to raw form.
387         value = unswap(value);
388         BeanContext bc = this.beanMeta.ctx;
389
390      if (m.bean == null) {
391
392         // If this bean has subtypes, and we haven't set the subtype yet,
393         // store the property in a temporary cache until the bean can be instantiated.
394         if (m.meta.subTypeIdProperty != null && m.propertyCache == null)
395            m.propertyCache = new TreeMap<String,Object>();
396
397         // Read-only beans get their properties stored in a cache.
398         if (m.propertyCache != null)
399            return m.propertyCache.put(name, value);
400
401         throw new BeanRuntimeException("Non-existent bean instance on bean.");
402      }
403
404         boolean isMap = rawTypeMeta.isMap();
405         boolean isCollection = rawTypeMeta.isCollection();
406
407      if (field == null && setter == null && ! (isMap || isCollection)) {
408         if ((value == null && bc.ignoreUnknownNullBeanProperties) || bc.ignorePropertiesWithoutSetters)
409            return null;
410         throw new BeanRuntimeException(beanMeta.c, "Setter or public field not defined on property ''{0}''", name);
411      }
412
413      Object bean = m.getBean(true);  // Don't use getBean() because it triggers array creation!
414
415      try {
416
417         Object r = beanMeta.ctx.beanMapPutReturnsOldValue || isMap || isCollection ? get(m) : null;
418            Class<?> propertyClass = rawTypeMeta.getInnerClass();
419
420         if (value == null && (isMap || isCollection)) {
421            if (setter != null) {
422               setter.invoke(bean, new Object[] { null });
423               return r;
424            } else if (field != null) {
425               field.set(bean, null);
426               return r;
427            }
428            throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' to null because no setter or public field is defined", name);
429         }
430
431         if (isMap) {
432
433            if (! (value instanceof Map)) {
434               if (value instanceof CharSequence)
435                  value = new ObjectMap((CharSequence)value).setBeanContext(beanMeta.ctx);
436               else
437                  throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{1}'' to object of type ''{2}''", name, propertyClass.getName(), findClassName(value));
438            }
439
440            Map valueMap = (Map)value;
441            Map propMap = (Map)r;
442               ClassMeta<?> valueType = rawTypeMeta.getValueType();
443
444            // If the property type is abstract, then we either need to reuse the existing
445            // map (if it's not null), or try to assign the value directly.
446               if (! rawTypeMeta.canCreateNewInstance()) {
447               if (propMap == null) {
448                  if (setter == null && field == null)
449                     throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{1}'' to object of type ''{2}'' because no setter or public field is defined, and the current value is null", name, propertyClass.getName(), findClassName(value));
450
451                  if (propertyClass.isInstance(valueMap)) {
452                     if (! valueType.isObject()) {
453                        for (Map.Entry e : (Set<Map.Entry>)valueMap.entrySet()) {
454                           Object v = e.getValue();
455                           if (v != null && ! valueType.getInnerClass().isInstance(v))
456                              throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{1}'' to object of type ''{2}'' because the value types in the assigned map do not match the specified ''elementClass'' attribute on the property, and the property value is currently null", name, propertyClass.getName(), findClassName(value));
457                        }
458                     }
459                     if (setter != null)
460                        setter.invoke(bean, valueMap);
461                     else
462                        field.set(bean, valueMap);
463                     return r;
464                  }
465                  throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{2}'' to object of type ''{2}'' because the assigned map cannot be converted to the specified type because the property type is abstract, and the property value is currently null", name, propertyClass.getName(), findClassName(value));
466               }
467            } else {
468               if (propMap == null) {
469                  propMap = (Map)propertyClass.newInstance();
470                  if (setter != null)
471                     setter.invoke(bean, propMap);
472                  else if (field != null)
473                     field.set(bean, propMap);
474                  else
475                     throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{1}'' to object of type ''{2}'' because no setter or public field is defined on this property, and the existing property value is null", name, propertyClass.getName(), findClassName(value));
476               } else {
477                  propMap.clear();
478               }
479            }
480
481            // Set the values.
482            for (Map.Entry e : (Set<Map.Entry>)valueMap.entrySet()) {
483               Object k = e.getKey();
484               Object v = e.getValue();
485               if (! valueType.isObject())
486                  v = beanMeta.ctx.convertToType(v, valueType);
487               propMap.put(k, v);
488            }
489
490         } else if (isCollection) {
491
492            if (! (value instanceof Collection)) {
493               if (value instanceof CharSequence)
494                  value = new ObjectList((CharSequence)value).setBeanContext(beanMeta.ctx);
495               else
496                  throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{1}'' to object of type ''{2}''", name, propertyClass.getName(), findClassName(value));
497            }
498
499            Collection valueList = (Collection)value;
500            Collection propList = (Collection)r;
501               ClassMeta elementType = rawTypeMeta.getElementType();
502
503            // If the property type is abstract, then we either need to reuse the existing
504            // collection (if it's not null), or try to assign the value directly.
505               if (! rawTypeMeta.canCreateNewInstance()) {
506               if (propList == null) {
507                  if (setter == null && field == null)
508                     throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{1}'' to object of type ''{2}'' because no setter or public field is defined, and the current value is null", name, propertyClass.getName(), findClassName(value));
509
510                  if (propertyClass.isInstance(valueList)) {
511                     if (! elementType.isObject()) {
512                           List l = new ObjectList(valueList);
513                           for (ListIterator<Object> i = l.listIterator(); i.hasNext(); ) {
514                              Object v = i.next();
515                              if (v != null && (! elementType.getInnerClass().isInstance(v))) {
516                                 i.set(bc.convertToType(v, elementType));
517                              }
518                           }
519                           valueList = l;
520                        }
521                     if (setter != null)
522                        setter.invoke(bean, valueList);
523                     else
524                        field.set(bean, valueList);
525                     return r;
526                  }
527                  throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{1}'' to object of type ''{2}'' because the assigned map cannot be converted to the specified type because the property type is abstract, and the property value is currently null", name, propertyClass.getName(), findClassName(value));
528               }
529               propList.clear();
530            } else {
531               if (propList == null) {
532                  propList = (Collection)propertyClass.newInstance();
533                  if (setter != null)
534                     setter.invoke(bean, propList);
535                  else if (field != null)
536                     field.set(bean, propList);
537                  else
538                     throw new BeanRuntimeException(beanMeta.c, "Cannot set property ''{0}'' of type ''{1}'' to object of type ''{2}'' because no setter is defined on this property, and the existing property value is null", name, propertyClass.getName(), findClassName(value));
539               } else {
540                  propList.clear();
541               }
542            }
543
544            // Set the values.
545            for (Object v : valueList) {
546               if (! elementType.isObject())
547                  v = beanMeta.ctx.convertToType(v, elementType);
548               propList.add(v);
549            }
550
551         } else {
552            if (transform != null && value != null && isParentClass(transform.getSwapClass(), value.getClass())) {
553                  value = transform.unswap(value, rawTypeMeta, beanMeta.ctx);
554            } else {
555                  value = beanMeta.ctx.convertToType(value, rawTypeMeta);
556               }
557            if (setter != null)
558               setter.invoke(bean, new Object[] { value });
559            else if (field != null)
560               field.set(bean, value);
561         }
562
563         return r;
564
565      } catch (BeanRuntimeException e) {
566         throw e;
567      } catch (Exception e) {
568         if (beanMeta.ctx.ignoreInvocationExceptionsOnSetters) {
569               if (rawTypeMeta.isPrimitive())
570                  return rawTypeMeta.getPrimitiveDefault();
571            return null;
572         }
573         throw new BeanRuntimeException(beanMeta.c, "Error occurred trying to set property ''{0}''", name).initCause(e);
574      }
575      } catch (ParseException e) {
576         throw new BeanRuntimeException(e);
577      }
578   }

580   /**
581    * Sets an array field on this bean.
582    * Works on both <code>Object</code> and primitive arrays.
583    *
584    * @param bean The bean of the field.
585    * @param l The collection to use to set the array field.
586    * @throws IllegalArgumentException Thrown by method invocation.
587    * @throws IllegalAccessException Thrown by method invocation.
588    * @throws InvocationTargetException Thrown by method invocation.
589    */
590   protected void setArray(Object bean, List l) throws IllegalArgumentException, IllegalAccessException, InvocationTargetException {
591      Object array = ArrayUtils.toArray(l, this.rawTypeMeta.getElementType().getInnerClass());
592      if (setter != null)
593         setter.invoke(bean, array);
594      else if (field != null)
595         field.set(bean, array);
596      else
597         throw new BeanRuntimeException(beanMeta.c, "Attempt to initialize array property ''{0}'', but no setter or field defined.", name);
598   }

600   /**
601    * Adds a value to a {@link Collection} or array property.
602    * Note that adding values to an array property is inefficient for large
603    * arrays since it must copy the array into a larger array on each operation.
604    *
605    * @param m The bean of the field being set.
606    * @param value The value to add to the field.
607    * @throws BeanRuntimeException If field is not a collection or array.
608    */
609   public void add(BeanMap<?> m, Object value) throws BeanRuntimeException {
610
611      BeanContext bc = beanMeta.ctx;
612
613      // Read-only beans get their properties stored in a cache.
614      if (m.bean == null) {
615         if (! m.propertyCache.containsKey(name))
616            m.propertyCache.put(name, new ObjectList(bc));
617         ((ObjectList)m.propertyCache.get(name)).add(value);
618         return;
619      }
620
621      boolean isCollection = rawTypeMeta.isCollection();
622      boolean isArray = rawTypeMeta.isArray();
623
624      if (! (isCollection || isArray))
625         throw new BeanRuntimeException(beanMeta.c, "Attempt to add element to property ''{0}'' which is not a collection or array", name);
626
627      Object bean = m.getBean(true);
628
629      ClassMeta<?> elementType = rawTypeMeta.getElementType();
630
631      try {
632         Object v = bc.convertToType(value, elementType);
633
634         if (isCollection) {
635            Collection c = null;
636            if (getter != null) {
637               c = (Collection)getter.invoke(bean, (Object[])null);
638            } else if (field != null) {
639               c = (Collection)field.get(bean);
640            } else {
641               throw new BeanRuntimeException(beanMeta.c, "Attempt to append to collection property ''{0}'', but no getter or field defined.", name);
642            }
643
644            if (c != null) {
645               c.add(v);
646               return;
647            }
648
649            if (rawTypeMeta.canCreateNewInstance())
650               c = (Collection)rawTypeMeta.newInstance();
651            else
652               c = new ObjectList(bc);
653
654            c.add(v);
655
656            if (setter != null)
657               setter.invoke(bean, c);
658            else if (field != null)
659               field.set(bean, c);
660            else
661               throw new BeanRuntimeException(beanMeta.c, "Attempt to initialize collection property ''{0}'', but no setter or field defined.", name);
662
663         } else /* isArray() */ {
664
665            if (m.arrayPropertyCache == null)
666               m.arrayPropertyCache = new TreeMap<String,List<?>>();
667
668            List l = m.arrayPropertyCache.get(name);
669            if (l == null) {
670               l = new LinkedList();  // ArrayLists and LinkLists appear to perform equally.
671               m.arrayPropertyCache.put(name, l);
672
673               // Copy any existing array values into the temporary list.
674               Object oldArray;
675            if (getter != null)
676                  oldArray = getter.invoke(bean, (Object[])null);
677            else if (field != null)
678                  oldArray = field.get(bean);
679            else
680               throw new BeanRuntimeException(beanMeta.c, "Attempt to append to array property ''{0}'', but no getter or field defined.", name);
681               ArrayUtils.copyToList(oldArray, l);
682            }
683
684            // Add new entry to our array.
685            l.add(v);
686         }
687
688      } catch (BeanRuntimeException e) {
689         throw e;
690      } catch (Exception e) {
691         throw new BeanRuntimeException(e);
692      }
693   }

695   /**
696    * Returns all instances of the specified annotation in the hierarchy of this bean property.
697    * <p>
698    * Searches through the class hierarchy (e.g. superclasses, interfaces, packages) for all
699    * instances of the specified annotation.
700    *
701    * @param a The class to find annotations for.
702    * @return A list of annotations ordered in child-to-parent order.  Never <jk>null</jk>.
703    */
704   public <A extends Annotation> List<A> findAnnotations(Class<A> a) {
705      List<A> l = new LinkedList<A>();
706      if (field != null) {
707         addIfNotNull(l, field.getAnnotation(a));
708         appendAnnotations(a, field.getType(), l);
709      }
710      if (getter != null) {
711         addIfNotNull(l, getter.getAnnotation(a));
712         appendAnnotations(a, getter.getReturnType(), l);
713      }
714      if (setter != null) {
715         addIfNotNull(l, setter.getAnnotation(a));
716         appendAnnotations(a, setter.getReturnType(), l);
717      }
718      appendAnnotations(a, this.getBeanMeta().getClassMeta().getInnerClass(), l);
719      return l;
720   }

722   private Object transform(Object o) throws SerializeException {
723      // First use transform defined via @BeanProperty.
724      if (transform != null)
725         return transform.swap(o, beanMeta.ctx);
726      if (o == null)
727         return null;
728      // Otherwise, look it up via bean context.
729      if (rawTypeMeta.hasChildPojoSwaps()) {
730         Class c = o.getClass();
731         ClassMeta<?> cm = rawTypeMeta.innerClass == c ? rawTypeMeta : beanMeta.ctx.getClassMeta(c);
732         PojoSwap f = cm.getPojoSwap();
733         if (f != null)
734            return f.swap(o, beanMeta.ctx);
735      }
736      return o;
737   }

739   private Object unswap(Object o) throws ParseException {
740      if (transform != null)
741         return transform.unswap(o, rawTypeMeta, beanMeta.ctx);
742      if (o == null)
743         return null;
744      if (rawTypeMeta.hasChildPojoSwaps()) {
745         Class c = o.getClass();
746         ClassMeta<?> cm = rawTypeMeta.innerClass == c ? rawTypeMeta : beanMeta.ctx.getClassMeta(c);
747         PojoSwap f = cm.getPojoSwap();
748         if (f != null)
749            return f.unswap(o, rawTypeMeta, beanMeta.ctx);
750      }
751      return o;
752   }

754   private Object applyChildPropertiesFilter(ClassMeta cm, Object o) {
755      if (o == null)
756         return null;
757      if (cm.isBean())
758         return new BeanMap(o, new BeanMetaFiltered(cm.getBeanMeta(), properties));
759      if (cm.isMap())
760         return new FilteredMap((Map)o, properties);
761      if (cm.isObject()) {
762         if (o instanceof Map)
763            return new FilteredMap((Map)o, properties);
764         BeanMeta bm = this.getBeanMeta().ctx.getBeanMeta(o.getClass());
765         if (bm != null)
766            return new BeanMap(o, new BeanMetaFiltered(cm.getBeanMeta(), properties));
767      }
768      return o;
769   }

772      if (o == null)
773         return null;
774      if (o instanceof Class)
775         return ((Class<?>)o).getName();
776      return o.getClass().getName();
777   }
