013package org.apache.juneau.html;
014
015import java.lang.reflect.*;
016
017import org.apache.juneau.*;
018import org.apache.juneau.parser.*;
019
020/**
021 * Configurable properties on the {@link HtmlParser} class.
022 * <p>
023 * Context properties are set by calling {@link ContextFactory#setProperty(String, Object)} on the context factory
024 * returned {@link CoreApi#getContextFactory()}.
025 * <p>
026 * The following convenience methods are also provided for setting context properties:
027 * <ul>
028 *    <li>{@link HtmlParser#setProperty(String,Object)}
029 *    <li>{@link HtmlParser#setProperties(ObjectMap)}
030 *    <li>{@link HtmlParser#addNotBeanClasses(Class[])}
031 *    <li>{@link HtmlParser#addBeanFilters(Class[])}
032 *    <li>{@link HtmlParser#addPojoSwaps(Class[])}
033 *    <li>{@link HtmlParser#addImplClass(Class,Class)}
034 * </ul>
035 * <p>
036 * See {@link ContextFactory} for more information about context properties.
037 *
038 * @author James Bognar (james.bognar@salesforce.com)
039 */
040public final class HtmlParserContext extends ParserContext {
041
042   /**
043    * Constructor.
044    * <p>
045    * Typically only called from {@link ContextFactory#getContext(Class)}.
046    *
047    * @param cf The factory that created this context.
048    */
049   public HtmlParserContext(ContextFactory cf) {
050      super(cf);
051   }
052
053   /**
054    * Constructor.
055    * <p>
056    * Typically only called from {@link ContextFactory#getContext(Class)}.
057    *
058    * @param cf The factory that created this context.
059    */
060   HtmlParserSession createSession(BeanContext beanContext, Object input, ObjectMap op, Method javaMethod, Object outer) {
061      return new HtmlParserSession(this, beanContext, input, op, javaMethod, outer);
062   }
063}
