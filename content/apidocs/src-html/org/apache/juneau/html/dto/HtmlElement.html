023/**
024 * Superclass for all HTML elements.
025 * <p>
026 * These are beans that when serialized using {@link HtmlSerializer} generate
027 * valid HTML5 elements.
028 *
029 * @author James Bognar (james.bognar@salesforce.com)
030 */

035   /**
036    * The children of this element.
037    */
038   @Xml(format=MIXED)
039   public List<Object> children;
040
041   /**
042    * Adds a child element to this element;
043    *
044    * @param child
045    * @return This object (for method chaining).
046    */
047   public HtmlElement child(Object child) {
048      if (children == null)
049         children = new LinkedList<Object>();
050      children.add(child);
051      return this;
052   }

165    * <code>dropzone</code> - <a href='https://www.w3.org/TR/html-markup/datatypes.html#common.data.dropzonevalue'>Dropzone value</a>.
166    * Specifies what types of content can be dropped on the element, and instructs the UA about which actions to take with content when it is dropped on the element.
167    *
168    * An <a href='https://www.w3.org/TR/html-markup/datatypes.html#data-unordered-tokens'>unordered set of unique space-separated tokens</a>, each of which is a <a href='https://www.w3.org/TR/html-markup/terminology.html#case-insensitive'>case-insensitive match</a> for one of the following:
169    * <ul>
170    *    <li><js>"copy"</js> - Indicates that dropping an accepted item on the element will result in a copy of the dragged data.
171    *    <li><js>"move"</js> - Indicates that dropping an accepted item on the element will result in the dragged data being moved to the new location.
172    *    <li><js>"link"</js> - Indicates that dropping an accepted item on the element will result in a link to the original data.
173    *    <li>Any <a href='https://www.w3.org/TR/html-markup/datatypes.html#data-string'>string</a> with three characters or more, beginning with the literal string <js>"string:"</js>.
174    *       Indicates that Plain Unicode string items, of the type indicated by the part of of the keyword after the <js>"string:"</js> string, can be dropped on this element.
175    *    <li>Any <a href='https://www.w3.org/TR/html-markup/datatypes.html#data-string'>string</a> with three characters or more, beginning with the literal string <js>"file:"</js>.
176    *       Indicates that File items, of the type indicated by the part of of the keyword after the <js>"file:"</js> string, can be dropped on this element.
177    * </ul>
178    *
179    * The value must not have more than one of the three tokens <js>"copy"</js>, <js>"move"</js>, or <js>"link"</js>. If none are specified, the element represents a copy dropzone.
180    */
