028/**
029 * Manager for retrieving shared instances of {@link ConfigFile ConfigFiles}.
030 * <p>
031 * Example:
032 * <p class='bcode'>
033 *    ConfigFile cf = ConfigMgr.<jsf>DEFAULT</jsf>.get(<js>"MyConfig.cfg"</js>);
034 *    String setting = cf.get(<js>"MySection/mysetting"</js>);
035 * </p>
036 */
037public class ConfigMgr {
038
039   /**
040    * Default reusable configuration manager.
041    * <ul class='spaced-list'>
042    *    <li>Read-only: <jk>false</jk>.
043    *    <li>Encoder: {@link XorEncoder}.
044    *    <li>Serializer: {@link JsonSerializer#DEFAULT}.
045    *    <li>Parser: {@link JsonParser#DEFAULT}.
046    *    <li>Charset: {@link Charset#defaultCharset()}.
047    *    <li>Search paths: [<js>"."</js>].
048    * </ul>
049    */
050   public static final ConfigMgr DEFAULT = new ConfigMgr(false, new XorEncoder(), JsonSerializer.DEFAULT, JsonParser.DEFAULT, Charset.defaultCharset(), new String[]{"."});
051
052   private ConcurrentHashMap<String,File> files = new ConcurrentHashMap<String,File>();
053   private ConcurrentHashMap<File,ConfigFile> configs = new ConcurrentHashMap<File,ConfigFile>();
054   private final WriterSerializer serializer;
055   private final ReaderParser parser;
056   private final Encoder encoder;
057   private final boolean readOnly;
058   private final Charset charset;
059   private final List<File> searchPaths = new LinkedList<File>();
060
061   /**
062    * Create a custom configuration manager.
063    *
064    * @param readOnly Make {@link ConfigFile ConfigFiles} read-only.
065    * @param encoder Optional.  Specify the encoder to use for encoded config file entries (e.g. <js>"mySecret*={...}"</js>).
066    * @param serializer Optional.  Specify the serializer to use for serializing POJOs when using {@link ConfigFile#put(String, Object)}.
067    * @param parser Optional.  Specify the parser to use for parsing POJOs when using {@link ConfigFile#getObject(Class,String)}.
068    * @param charset Optional.  Specify the config file character encoding.  If <jk>null</jk>, uses {@link Charset#defaultCharset()}.
069    * @param searchPaths Specify the search paths for config files.  Can contain relative or absolute paths.
070    */
071   public ConfigMgr(boolean readOnly, Encoder encoder, WriterSerializer serializer, ReaderParser parser, Charset charset, String[] searchPaths) {
072      this.readOnly = readOnly;
073      this.encoder = encoder;
074      this.serializer = serializer;
075      this.parser = parser;
076      this.charset = charset;
077      if (searchPaths != null)
078         for (String p : searchPaths)
079            this.searchPaths.add(new File(p));
080   }
081
082   /**
083    * Returns the config file with the specified absolute or relative path.
084    * <p>
085    * Multiple calls to the same path return the same <code>ConfigFile</code> instance.
086    *
087    * @param path The absolute or relative path of the config file.
088    * @return The config file.
089    * @throws IOException If config file could not be parsed.
090    * @throws FileNotFoundException If config file could not be found.
091    */
092   public ConfigFile get(String path) throws IOException {
093      return get(path, false);
094   }
095
096   /**
097    * Returns the config file with the specified absolute or relative path.
098    * <p>
099    * Multiple calls to the same path return the same <code>ConfigFile</code> instance.
100    * <p>
101    * If file doesn't exist and <code>create</code> is <jk>true</jk>, the configuration file will be
102    * create in the location identified by the first entry in the search paths.
103    *
104    * @param path The absolute or relative path of the config file.
105    * @param create Create the config file if it doesn't exist.
106    * @return The config file.
107    * @throws IOException If config file could not be parsed.
108    * @throws FileNotFoundException If config file could not be found or could not be created.
109    */
110   public ConfigFile get(String path, boolean create) throws IOException {
111
112      File f = resolve(path, create);
113
114      ConfigFile cf = configs.get(f);
115      if (cf != null)
116         return cf;
117
118      cf = new ConfigFileImpl(f, readOnly, encoder, serializer, parser, charset);
119      configs.putIfAbsent(f, cf);
120      return configs.get(f);
121   }
122
123   /**
124    * Create a new empty config file not backed by any file.
125    *
126    * @return A new config file.
127    * @throws IOException
128    */
129   public ConfigFile create() throws IOException {
130      return new ConfigFileImpl(null, false, encoder, serializer, parser, charset);
131   }
132
133   /**
134    * Create a new config file backed by the specified file.
135    * Note that {@link #get(String)} is the preferred method for getting access to config files
136    *    since this method will create a new config file each time it is called.
137    * This method is provided primarily for testing purposes.
138    *
139    * @param f The file to create a config file from.
140    * @return A new config file.
141    * @throws IOException
142    */
143   public ConfigFile create(File f) throws IOException {
144      return new ConfigFileImpl(f, false, encoder, serializer, parser, charset);
145   }
146
147   /**
148    * Create a new config file not backed by a file.
149    *
150    * @param r The reader containing an INI-formatted file to initialize the config file from.
151    * @return A new config file.
152    * @throws IOException
153    */
154   public ConfigFile create(Reader r) throws IOException {
155      return new ConfigFileImpl(null, false, encoder, serializer, parser, charset).load(r);
156   }
157
158   /**
159    * Reloads any config files that were modified.
160    * @throws IOException
161    */
162   public void loadIfModified() throws IOException {
163      for (ConfigFile cf : configs.values())
164         cf.loadIfModified();
165   }
166
167   /**
168    * Delete all configuration files registered with this config manager.
169    */
170   public void deleteAll() {
171      for (File f : configs.keySet())
172         FileUtils.delete(f);
173      files.clear();
174      configs.clear();
175   }
176
177   private File resolve(String path, boolean create) throws IOException {
178
179      // See if it's cached.
180      File f = files.get(path);
181      if (f != null)
182         return f;
183
184      // Handle absolute file.
185      f = new File(path);
186      if (f.isAbsolute()) {
187         if (create)
188            FileUtils.create(f);
189         if (f.exists())
190            return addFile(path, f);
191         throw new FileNotFoundException("Could not find config file '"+path+"'");
192      }
193
194      if (searchPaths.isEmpty())
195         throw new FileNotFoundException("No search paths specified on ConfigMgr.");
196
197      // Handle paths relative to search paths.
198      for (File sf : searchPaths) {
199         f = new File(sf.getAbsolutePath() + "/" + path);
200         if (f.exists())
201            return addFile(path, f);
202      }
203
204      if (create) {
205         f = new File(searchPaths.get(0).getAbsolutePath() + "/" + path);
206         FileUtils.create(f);
207            return addFile(path, f);
208      }
209
210      throw new FileNotFoundException("Could not find config file '"+path+"'");
211   }
212
213   private File addFile(String path, File f) {
214      files.putIfAbsent(path, f);
215      return files.get(path);
216   }

218   /**
219    * Implements command-line features for working with INI configuration files.
220    * <p>
221    * Invoke as a normal Java program...
222    * <p>
223    * <p class='bcode'>
224    *    java org.apache.juneau.ini.ConfigMgr [args]
225    * </p>
226    * <p>
227    * Arguments can be any of the following...
228    * <ul class='spaced-list'>
229    *    <li>No arguments<br>
230    *       Prints usage message.<br>
231    *    <li><code>createBatchEnvFile -configfile &lt;configFile&gt; -envfile &lt;batchFile&gt; [-verbose]</code><br>
232    *       Creates a batch file that will set each config file entry as an environment variable.<br>
233    *       Characters in the keys that are not valid as environment variable names (e.g. <js>'/'</js> and <js>'.'</js>)
234    *          will be converted to underscores.<br>
235    *    <li><code>createShellEnvFile -configFile &lt;configFile&gt; -envFile &lt;configFile&gt; [-verbose]</code>
236    *       Creates a shell script that will set each config file entry as an environment variable.<br>
237    *       Characters in the keys that are not valid as environment variable names (e.g. <js>'/'</js> and <js>'.'</js>)
238    *          will be converted to underscores.<br>
239    *    <li><code>setVals -configFile &lt;configFile&gt; -vals [var1=val1 [var2=val2...]] [-verbose]</code>
240    *       Sets values in config files.<br>
241    * </ul>
242    * <p>
243    * For example, the following command will create the file <code>'MyConfig.bat'</code> from the contents of the file <code>'MyConfig.cfg'</code>.
244    * <p class='bcode'>
245    *    java org.apache.juneau.ini.ConfigMgr createBatchEnvFile -configfile C:\foo\MyConfig.cfg -batchfile C:\foo\MyConfig.bat
246    * </p>
247    *
248    * @param args Command-line arguments
249    */
250   public static void main(String[] args) {
251
252      Args a = new Args(args);
253      String command = a.getArg(0);
254      String configFile = a.getArg("configFile");
255      String envFile = a.getArg("envFile");
256      List<String> vals = a.getArgs("vals");
257
258      if (command == null || ! (command.equals("createBatchEnvFile") || command.equals("createShellEnvFile") || command.equals("setVals")))
259         printUsageAndExit();
260      else if (configFile.isEmpty())
261         printUsageAndExit();
262      else if (command.equals("setVals") && vals.isEmpty())
263         printUsageAndExit();
264      else if ((command.equals("createBatchEnvFile") || command.equals("createShellEnvFile")) && envFile.isEmpty())
265         printUsageAndExit();
266      else {
267         try {
268            ConfigFile cf = ConfigMgr.DEFAULT.get(configFile);
269
270            if (command.equalsIgnoreCase("setVals")) {
271               for (String val : vals) {
272                  String[] x = val.split("\\=");
273                  if (x.length != 2)
274                     throw new RuntimeException("Invalid format for value: '"+val+"'.  Must be in the format 'key=value'");
275                  cf.put(x[0], x[1]);
276               }
277               cf.save();
278               return;
279
280            } else if (command.equalsIgnoreCase("createBatchEnvFile")) {
281               Writer fw = new OutputStreamWriter(new FileOutputStream(envFile), Charset.defaultCharset());
282               try {
283                  cf.serializeTo(fw, BATCH);
284               } finally {
285                  fw.close();
286               }
287               return;
288
289            } else if (command.equalsIgnoreCase("createShellEnvFile")) {
290               Writer fw = new OutputStreamWriter(new FileOutputStream(envFile), Charset.defaultCharset());
291               try {
292                  cf.serializeTo(fw, SHELL);
293               } finally {
294                  fw.close();
295               }
296               return;
297            }
298
299         } catch (Exception e) {
300            e.printStackTrace();
301         }
302      }
303   }
304
305   private static void printUsageAndExit() {
306      System.err.println("---Usage---");
307      System.err.println("java -cp juneau.jar org.apache.juneau.ini.ConfigFile createBatchEnvFile -configFile <configFile> -envFile <envFile> [-verbose]");
308      System.err.println("java -cp juneau.jar org.apache.juneau.ini.ConfigFile createShellEnvFile -configFile <configFile> -envFile <envFile> [-verbose]");
309      System.err.println("java -cp juneau.jar org.apache.juneau.ini.ConfigFile setVals -configFile <configFile> -vals [var1 val1 [var2 val2...]] [-verbose]");
310      int rc = Integer.getInteger("exit.2", 2);
311      if (rc != 0)
312         System.exit(rc);
313   }
