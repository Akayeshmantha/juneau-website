013package org.apache.juneau.encoders;
014
015import static org.apache.juneau.internal.ArrayUtils.*;
016
017import java.util.*;
018
019import org.apache.juneau.*;
020
021/**
022 * Represents the group of {@link Encoder encoders} keyed by codings.
023 *
024 *
025 * <h6 class='topic'>Description</h6>
026 * <p>
027 *    Maintains a set of encoders and the codings that they can handle.
028 * <p>
029 *    The {@link #findMatch(String)} and {@link #getEncoder(String)} methods are then
030 *       used to find appropriate encoders for specific <code>Accept-Encoding</code>
031 *       and <code>Content-Encoding</code> header values.
032 *
033 *
034 * <h6 class='topic'>Match ordering</h6>
035 * <p>
036 *    Encoders are matched against <code>Accept-Encoding</code> strings in the order they exist in this group.
037 * <p>
038 *    Adding new entries will cause the entries to be prepended to the group.
039 *    This allows for previous encoders to be overridden through subsequent calls.
040 * <p>
041 *    For example, calling <code>g.append(E1.<jk>class</jk>,E2.<jk>class</jk>).append(E3.<jk>class</jk>,E4.<jk>class</jk>)</code>
042 *    will result in the order <code>E3, E4, E1, E2</code>.
043 *
044 *
045 * <h6 class='topic'>Example</h6>
046 * <p class='bcode'>
047 *    <jc>// Create an encoder group with support for gzip compression.</jc>
048 *    EncoderGroup g = <jk>new</jk> EncoderGroup().append(GzipEncoder.<jk>class</jk>);
049 *
050 *    <jc>// Should return "gzip"</jc>
051 *    String matchedCoding = g.findMatch(<js>"compress;q=1.0, gzip;q=0.8, identity;q=0.5, *;q=0"</js>);
052 *
053 *    <jc>// Get the encoder</jc>
054 *    IEncoder encoder = g.getEncoder(matchedCoding);
055 * </p>
056 *
057 *
058 * @author James Bognar (james.bognar@salesforce.com)
059 */
060public final class EncoderGroup {
061
062   private Map<String,EncoderEntry> entryMap = new TreeMap<String,EncoderEntry>(String.CASE_INSENSITIVE_ORDER);
063   private LinkedList<EncoderEntry> tempEntries = new LinkedList<EncoderEntry>();
064   private EncoderEntry[] entries;
065
066   /**
067    * Returns the coding string for the matching encoder that can handle the specified <code>Accept-Encoding</code>
068    *    or <code>Content-Encoding</code> header value.
069    * <p>
070    *    Returns <jk>null</jk> if no encoders can handle it.
071    * <p>
072    *    This method is fully compliant with the RFC2616/14.3 and 14.11 specifications.
073    *
074    * @param acceptEncoding The <code>Accept-Encoding</code> or <code>Content-Encoding</code> value.
075    * @return The coding value (e.g. <js>"gzip"</js>).
076    */
077   public String findMatch(String acceptEncoding) {
078      if (getEntries().length == 0)
079         return null;
080
081      MediaRange[] ae = MediaRange.parse(acceptEncoding);
082
083      if (ae.length == 0)
084         ae = MediaRange.parse("*");
085
086      for (MediaRange a : ae)
087         for (EncoderEntry e : getEntries())
088            for (MediaRange a2 : e.encodingRanges)
089               if (a.matches(a2))
090                  return a2.getType();
091
092      return null;
093   }
094
095   /**
096    * Adds the specified encoders to this group.
097    *
098    * @param e The encoders to instantiate and add to this group.
099    * @return This object (for method chaining).
100    * @throws Exception If an instantiation error occurred.
101    */
102   public EncoderGroup append(Class<? extends Encoder>...e) throws Exception {
103      for (Class<? extends Encoder> r : reverse(e))
104         append(r.newInstance());
105      return this;
106   }
107
108   /**
109    * Adds the specified encoders to this group.
110    *
111    * @param e The encoder to instantiate and add to this group.
112    * @return This object (for method chaining).
113    * @throws Exception If an instantiation error occurred.
114    */
115   public EncoderGroup append(Class<? extends Encoder> e) throws Exception {
116      append(e.newInstance());
117      return this;
118   }
119
120   /**
121    * Adds the specified encoders to this group.
122    *
123    * @param e The encoders to add to this group.
124    * @return This object (for method chaining).
125    */
126   public EncoderGroup append(Encoder...e) {
127      entries = null;
128      for (Encoder r : reverse(e)) {
129         EncoderEntry ee = new EncoderEntry(r);
130         tempEntries.addFirst(ee);
131         for (String s : ee.encodings)
132            this.entryMap.put(s, ee);
133      }
134      return this;
135   }
136
137   /**
138    * Adds the encoders in the specified group to this group.
139    *
140    * @param g The group containing the encoders to add to this group.
141    * @return This object (for method chaining).
142    */
143   public EncoderGroup append(EncoderGroup g) {
144      for (EncoderEntry e : reverse(g.getEntries()))
145         append(e.encoder);
146      return this;
147   }
148
149   /**
150    * Returns the encoder registered with the specified coding (e.g. <js>"gzip"</js>).
151    *
152    * @param coding The coding string.
153    * @return The encoder, or <jk>null</jk> if encoder isn't registered with that coding.
154    */
155   public Encoder getEncoder(String coding) {
156      EncoderEntry e = entryMap.get(coding);
157      return (e == null ? null : e.encoder);
158   }
159
160   /**
161    * Returns the set of codings supported by all encoders in this group.
162    *
163    * @return The set of codings supported by all encoders in this group.  Never <jk>null</jk>.
164    */
165   public List<String> getSupportedEncodings() {
166      List<String> l = new ArrayList<String>();
167      for (EncoderEntry e : getEntries())
168         for (String enc : e.encodings)
169            if (! l.contains(enc))
170               l.add(enc);
171      return l;
172   }
173
174   private EncoderEntry[] getEntries() {
175      if (entries == null)
176         entries = tempEntries.toArray(new EncoderEntry[tempEntries.size()]);
177      return entries;
178   }
179
180   static class EncoderEntry {
181      Encoder encoder;
182      MediaRange[] encodingRanges;
183      String[] encodings;
184
185      EncoderEntry(Encoder e) {
186         encoder = e;
187
188         encodings = new String[e.getCodings().length];
189         int i = 0;
190         for (String enc : e.getCodings())
191            encodings[i++] = enc;
192
193         List<MediaRange> l = new LinkedList<MediaRange>();
194         for (i = 0; i < encodings.length; i++)
195            l.addAll(Arrays.asList(MediaRange.parse(encodings[i])));
196         encodingRanges = l.toArray(new MediaRange[l.size()]);
197      }
198   }
199}
